 40     @Test
 41     void testMailingList(TestInfo testInfo) throws IOException {
 42         try (var listServer = new TestMailmanServer();
 43              var credentials = new HostCredentials(testInfo);
 44              var tempFolder = new TemporaryDirectory()) {
 45             var repo = credentials.getHostedRepository();
 46             var repoFolder = tempFolder.path().resolve("repo");
 47             var localRepo = CheckableRepository.init(repoFolder, repo.repositoryType());
 48             var masterHash = localRepo.resolve("master").orElseThrow();
 49             credentials.commitLock(localRepo);
 50             localRepo.pushAll(repo.url());
 51 
 52             var listAddress = EmailAddress.parse(listServer.createList("test"));
 53             var mailmanServer = MailingListServerFactory.createMailmanServer(listServer.getArchive(), listServer.getSMTP(), Duration.ZERO);
 54             var mailmanList = mailmanServer.getList(listAddress.address());
 55             var tagStorage = createTagStorage(repo);
 56             var branchStorage = createBranchStorage(repo);
 57             var prStateStorage = createPullRequestStateStorage(repo);
 58             var storageFolder = tempFolder.path().resolve("storage");
 59 
 60             var sender = EmailAddress.from("duke", "duke@duke.duke");
 61             var notifyBot = NotifyBot.newBuilder()
 62                                      .repository(repo)
 63                                      .storagePath(storageFolder)
 64                                      .branches(Pattern.compile("master"))
 65                                      .tagStorageBuilder(tagStorage)
 66                                      .branchStorageBuilder(branchStorage)
 67                                      .prStateStorageBuilder(prStateStorage)
 68                                      .build();
 69             var updater = MailingListNotifier.newBuilder()
 70                                              .list(mailmanList)
 71                                              .recipient(listAddress)
 72                                              .sender(sender)
 73                                              .reportNewTags(false)
 74                                              .reportNewBranches(false)
 75                                              .reportNewBuilds(false)
 76                                              .headers(Map.of("extra1", "value1", "extra2", "value2"))
 77                                              .allowedAuthorDomains(Pattern.compile("none"))
 78                                              .build();
 79             updater.attachTo(notifyBot);
 80 
 81             // No mail should be sent on the first run as there is no history
 82             TestBotRunner.runPeriodicItems(notifyBot);
 83             assertThrows(RuntimeException.class, () -> listServer.processIncoming(Duration.ofMillis(1)));
 84 
 85             var editHash = CheckableRepository.appendAndCommit(localRepo, "Another line", "23456789: More fixes");
 86             localRepo.push(editHash, repo.url(), "master");
 87             TestBotRunner.runPeriodicItems(notifyBot);
 88             listServer.processIncoming();
 89 
 90             var conversations = mailmanList.conversations(Duration.ofDays(1));
 91             var email = conversations.get(0).first();
 92             assertEquals(listAddress, email.sender());
 93             assertEquals(sender, email.author());
 94             assertEquals(email.recipients(), List.of(listAddress));
 95             assertTrue(email.subject().contains(": 23456789: More fixes"));
 96             assertFalse(email.subject().contains("master"));
 97             assertTrue(email.body().contains("Changeset: " + editHash.abbreviate()));
 98             assertTrue(email.body().contains("23456789: More fixes"));
 99             assertFalse(email.body().contains("Committer"));
100             assertFalse(email.body().contains(masterHash.abbreviate()));
101             assertTrue(email.hasHeader("extra1"));
102             assertEquals("value1", email.headerValue("extra1"));
103             assertTrue(email.hasHeader("extra2"));
104             assertEquals("value2", email.headerValue("extra2"));
105             assertTrue(email.hasHeader("X-Git-URL"));
106             assertEquals(repo.webUrl().toString(), email.headerValue("X-Git-URL"));
107             assertTrue(email.hasHeader("X-Git-Changeset"));
108             assertEquals(editHash.hex(), email.headerValue("X-Git-Changeset"));
109         }

386                                              .build();
387             updater.attachTo(notifyBot);

391             localRepo.push(masterHash, repo.url(), "other", true);

394             TestBotRunner.runPeriodicItems(notifyBot);
395             assertThrows(RuntimeException.class, () -> listServer.processIncoming(Duration.ofMillis(1)));

399             localRepo.push(editHash, repo.url(), "edit");
400             var pr = credentials.createPullRequest(repo, "master", "edit", "RFR: My PR");
401 
402             // PR hasn't been integrated yet, so there should be no mail

412 
413             // And an integration (but it hasn't reached master just yet)
414             pr.addComment("Pushed as commit " + editHash.hex() + ".");
415 
416             // Now push the same commit to another branch
417             localRepo.push(editHash, repo.url(), "other");
418             TestBotRunner.runPeriodicItems(notifyBot);
419             listServer.processIncoming();
420 
421             // This one should generate a plain integration mail
422             var conversations = mailmanList.conversations(Duration.ofDays(1));
423             assertEquals(2, conversations.size());
424             var secondEmail = conversations.get(0).first();
425             if (secondEmail.subject().contains("RFR")) {
426                 secondEmail = conversations.get(1).first();
427             }
428             assertEquals("git: test: other: 23456789: More fixes", secondEmail.subject());
429         }
