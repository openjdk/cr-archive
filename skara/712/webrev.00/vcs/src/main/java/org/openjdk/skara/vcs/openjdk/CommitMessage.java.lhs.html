 23 package org.openjdk.skara.vcs.openjdk;
 24 
 25 import org.openjdk.skara.vcs.Author;
 26 
 27 import java.util.List;
 28 
 29 public class CommitMessage {
 30     private final String title;
 31     private final List<Issue> issues;
 32     private final List<String> reviewers;
 33     private final List<Author> contributors;
 34     private final List<String> summaries;
 35     private final List<String> additional;
 36 
 37     public CommitMessage(String title,
 38                          List<Issue> issues,
 39                          List<String> reviewers,
 40                          List<Author> contributors,
 41                          List<String> summaries,
 42                          List<String> additional) {
 43         this.title = title;
 44         this.issues = issues;
 45         this.reviewers = reviewers;
 46         this.contributors = contributors;
 47         this.summaries = summaries;
 48         this.additional = additional;
 49     }
 50 
 51     public String title() {
 52         return title;
 53     }
 54 
 55     public List<Issue> issues() {
 56         return issues;
 57     }
 58 
 59     public List<String> reviewers() {
 60         return reviewers;
 61     }
 62 
 63     public List<Author> contributors() {
 64         return contributors;
 65     }
 66 




 67     public List<String> summaries() {
 68         return summaries;
 69     }
 70 
 71     public List<String> additional() {
 72         return additional;
 73     }
 74 
 75     public static CommitMessageBuilder title(String title) {
 76         if (title == null || title.isEmpty()) {
 77             throw new IllegalArgumentException("Must provide a non-empty title");
 78         }
 79 
 80         var builder = new CommitMessageBuilder();
 81         builder.title(title);
 82         return builder;
 83     }
 84 
 85     public static CommitMessageBuilder title(Issue... issues) {
 86         var builder = new CommitMessageBuilder();
 87         builder.issues(issues);
 88         return builder;
 89     }
 90 
 91     public static CommitMessageBuilder title(List<Issue> issues) {
 92         var builder = new CommitMessageBuilder();
 93         builder.issues(issues);
 94         return builder;
 95     }
 96 
 97     public List<String> format(CommitMessageFormatter formatter) {
 98         return formatter.format(this);
 99     }
100 }
