23 package org.openjdk.skara.vcs;
24 
25 import java.nio.file.Path;
26 import java.util.Objects;
27 import java.util.Optional;
28 
29 public class WebrevStats {
30     private final int added;
31     private final int removed;
32     private final int modified;
33 
34     public WebrevStats(int added, int removed, int modified) {
35         this.added = added;
36         this.removed = removed;
37         this.modified = modified;
38     }
39 
40     public int added() {
41         return added;
42     }
43 
44     public int removed() {
45         return removed;
46     }
47 
48     public int modified() {
49         return modified;
50     }
51 
52     @Override
53     public int hashCode() {
54         return Objects.hash(added, removed, modified);
55     }
56 
57     @Override
58     public boolean equals(Object other) {
59         if (other == this) {
60             return true;
61         }
62 
63         if (!(other instanceof WebrevStats)) {
64             return false;
65         }
66 
67         var o = (WebrevStats) other;
68         return Objects.equals(added, o.added) &&
69                Objects.equals(removed, o.removed) &&
70                Objects.equals(modified, o.modified);
71     }
72 }
