23 package org.openjdk.skara.vcs;
24 
25 import java.nio.file.Path;
26 import java.util.Objects;
27 
28 public class PatchStats {
29     private final Path path;
30     private final int added;
31     private final int removed;
32     private final int modified;
33 
34     public PatchStats(Path path, int added, int removed, int modified) {
35         this.path = path;
36         this.added = added;
37         this.removed = removed;
38         this.modified = modified;
39     }
40 
41     public Path path() {
42         return path;
43     }
44 
45     public int added() {
46         return added;
47     }
48 
49     public int removed() {
50         return removed;
51     }
52 
53     public int modified() {
54         return modified;
55     }
56 
57     @Override
58     public int hashCode() {
59         return Objects.hash(path, added, removed, modified);
60     }
61 
62     @Override
63     public boolean equals(Object other) {
64         if (other == this) {
65             return true;
66         }
67 
68         if (!(other instanceof PatchStats)) {
69             return false;
70         }
71 
72         var o = (PatchStats) other;
73         return Objects.equals(path, o.path) &&
74                Objects.equals(added, o.added) &&
75                Objects.equals(removed, o.removed) &&
76                Objects.equals(modified, o.modified);
77     }
78 }
