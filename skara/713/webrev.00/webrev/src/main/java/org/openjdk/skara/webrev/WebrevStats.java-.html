23 package org.openjdk.skara.webrev;
24 
25 import org.openjdk.skara.vcs.PatchStats;
26 
27 class WebrevStats {
28     private final int added;
29     private final int removed;
30     private final int modified;
31     private final int total;
32 
33     public WebrevStats(PatchStats stats, int total) {
34         this.added = stats.added();
35         this.removed = stats.removed();
36         this.modified = stats.modified();
37         this.total = total;
38     }
39 
40     public WebrevStats(int added, int removed, int modified, int total) {
41         this.added = added;
42         this.removed = removed;
43         this.modified = modified;
44         this.total = total;
45     }
46 
47     public static WebrevStats empty() {
48         return new WebrevStats(0, 0, 0, 0);
49     }
50 
51     public int added() {
52         return added;
53     }
54 
55     public int removed() {
56         return removed;
57     }
58 
59     public int modified() {
60         return modified;
61     }
62 
63     public int changed() {
64         return added() + removed() + modified();
65     }
66 
67     public int unchanged() {
68         return total() - changed();
69     }
70 
71     public int total() {
72         return total;
73     }
74 
75     @Override
76     public String toString() {
77         return String.format("%d lines changed; %d ins; %d del; %d mod; %d unchg",
78                              changed(), added(), removed(), modified(), unchanged());
79     }
80 }
