40 @BenchmarkMode(Mode.AverageTime)
41 @Warmup(iterations = 5, time = 500, timeUnit = TimeUnit.MILLISECONDS)
42 @Measurement(iterations = 10, time = 500, timeUnit = TimeUnit.MILLISECONDS)
43 @State(org.openjdk.jmh.annotations.Scope.Thread)
44 @OutputTimeUnit(TimeUnit.NANOSECONDS)
45 @Fork(3)
46 public class PointsFree {
47 
48     JNIPoint jniPoint;
49     PanamaPoint panamaPoint;
50 
51     @Setup(Level.Invocation)
52     public void setup() {
53         jniPoint = new JNIPoint(0, 0);
54         panamaPoint = new PanamaPoint(0, 0);
55     }
56 
57     @Benchmark
58     public void jni_long_free() throws Throwable {
59         jniPoint.close();
60     }
61 
62     @Benchmark
63     public void panama_free() throws Throwable {
64         panamaPoint.close();
65     }
66 
67 }
