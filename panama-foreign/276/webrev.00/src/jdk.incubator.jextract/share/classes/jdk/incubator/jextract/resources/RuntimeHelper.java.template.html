129         private Object invoke(Object[] args) throws Throwable {
130             // one trailing Object[]
131             int nNamedArgs = function.argumentLayouts().size();
132             assert(args.length == nNamedArgs + 1);
133             // The last argument is the array of vararg collector
134             Object[] unnamedArgs = (Object[]) args[args.length - 1];
135 
136             int argsCount = nNamedArgs + unnamedArgs.length;
137             Class<?>[] argTypes = new Class<?>[argsCount];
138             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
139 
140             int pos = 0;
141             for (pos = 0; pos < nNamedArgs; pos++) {
142                 argTypes[pos] = varargs.parameterType(pos);
143                 argLayouts[pos] = function.argumentLayouts().get(pos);
144             }
145 
146             assert pos == nNamedArgs;
147             for (Object o: unnamedArgs) {
148                 argTypes[pos] = normalize(o.getClass());
149                 argLayouts[pos] = variadicLayout(argTypes[pos]);
150                 pos++;
151             }
152             assert pos == argsCount;
153 
154             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
155             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
156                     FunctionDescriptor.ofVoid(argLayouts) :
157                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
158             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
159             // flatten argument list so that it can be passed to an asSpreader MH
160             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
161             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
162             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
163 
164             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
165         }
