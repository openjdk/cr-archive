 23 package jdk.internal.jextract.impl;
 24 
 25 import java.util.ArrayList;
 26 import java.util.Collections;
 27 import java.util.List;
 28 
 29 public final class Options {
 30     // The args for parsing C
 31     public final List<String> clangArgs;
 32     // The list of library names
 33     public final List<String> libraryNames;
 34     public final List<String> filters;
 35     // target package
 36     public final String targetPackage;
 37     // output directory
 38     public final String outputDir;
 39     public final boolean source;
 40 
 41     private Options(List<String> clangArgs, List<String> libraryNames,
 42             List<String> filters, String targetPackage,
 43             String outputDir, boolean source) {
 44         this.clangArgs = clangArgs;
 45         this.libraryNames = libraryNames;
 46         this.filters = filters;
 47         this.targetPackage = targetPackage;
 48         this.outputDir = outputDir;
 49         this.source = source;
 50     }
 51 
 52     public static Builder builder() {
 53         return new Builder();
 54     }
 55 
 56     public static Options createDefault() {
 57         return builder().build();
 58     }
 59 
 60     public static class Builder {
 61         private final List<String> clangArgs;
 62         private final List<String> libraryNames;
 63         private final List<String> filters;
 64         private String targetPackage;
 65         private String outputDir;
 66         private boolean source;
 67 
 68         public Builder() {
 69             this.clangArgs = new ArrayList<>();
 70             this.libraryNames = new ArrayList<>();
 71             this.filters = new ArrayList<>();
 72             this.targetPackage = "";
 73             this.outputDir = ".";
 74             this.source = false;
 75         }
 76 
 77         public Options build() {
 78             return new Options(
 79                     Collections.unmodifiableList(clangArgs),
 80                     Collections.unmodifiableList(libraryNames),
 81                     Collections.unmodifiableList(filters),
 82                     targetPackage, outputDir, source
 83             );
 84         }
 85 
 86         public void addClangArg(String arg) {
 87             clangArgs.add(arg);
 88         }
 89 
 90         public void addLibraryName(String name) {
 91             libraryNames.add(name);
 92         }
 93 
 94         public void setOutputDir(String outputDir) {
 95             this.outputDir = outputDir;
 96         }
 97 
 98         public void setTargetPackage(String pkg) {
 99             this.targetPackage = pkg;
100         }
101 
102         public void addFilter(String filter) {
103             filters.add(filter);
104         }
105 
106         public void setGenerateSource() {
107             source = true;
108         }
109     }
110 }
