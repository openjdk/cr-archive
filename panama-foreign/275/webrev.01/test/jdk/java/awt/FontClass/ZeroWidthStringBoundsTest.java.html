24 /*
25  * @test
26  * @bug 8245159 8239725
27  * @summary Ensure no exception getting bounds of an empty string when the font has
28  * layout attributes.
29  * @run main/othervm ZeroWidthStringBoundsTest
30  */
31 
32 import java.awt.Font;
33 import java.awt.font.FontRenderContext;
34 import java.awt.font.TextAttribute;
35 import java.text.AttributedCharacterIterator;
36 import java.util.HashMap;
37 
38 public class ZeroWidthStringBoundsTest {
39 
40    public static void main(String[] args) {
41        FontRenderContext frc = new FontRenderContext(null, false, false);
42        Font f1 = new Font(Font.MONOSPACED, Font.PLAIN, 12);
43        f1.getStringBounds("", frc);
44        HashMap<AttributedCharacterIterator.Attribute, Object> attrs = new HashMap<>();
45        attrs.put(TextAttribute.KERNING, TextAttribute.KERNING_ON);
46        Font f2 = f1.deriveFont(attrs);
47        f2.getStringBounds("", frc);
48    }
49 }
