 24 /*
 25  * @test
 26  * @bug 8199307
 27  * @summary NPE in jdk.javadoc.internal.doclets.toolkit.util.Utils.getLineNumber
 28  * @library /tools/lib ../../lib
 29  * @modules
 30  *      jdk.javadoc/jdk.javadoc.internal.tool
 31  *      jdk.compiler/com.sun.tools.javac.api
 32  *      jdk.compiler/com.sun.tools.javac.main
 33  * @build javadoc.tester.*
 34  * @run main TestSerializedFormWithClassFile
 35  */
 36 
 37 import builder.ClassBuilder;
 38 
 39 import java.nio.file.Path;
 40 import java.nio.file.Paths;
 41 
 42 import builder.ClassBuilder.MethodBuilder;
 43 import toolbox.ToolBox;
 44 import toolbox.JavacTask;
 45 
 46 import javadoc.tester.JavadocTester;
 47 
 48 public class TestSerializedFormWithClassFile extends JavadocTester {
 49 
 50     final ToolBox tb;
 51 
 52     public static void main(String... args) throws Exception {
 53         TestSerializedFormWithClassFile tester = new TestSerializedFormWithClassFile();
 54         tester.runTests(m -> new Object[]{Paths.get(m.getName())});
 55     }
 56 
 57     TestSerializedFormWithClassFile() {
 58         tb = new ToolBox();
 59     }
 60 
 61     @Test
 62     public void test(Path base) throws Exception {
 63         Path srcDir = base.resolve("src");
 64         createTestClass(base, srcDir);
 65 
 66         Path outDir = base.resolve("out");
 67         javadoc("-d", outDir.toString(),
 68                 "-linksource",
 69                 "-classpath", base.resolve("classes").toString(),
 70                 "-sourcepath", "",
 71                 srcDir.resolve("B.java").toString());
 72 
 73         checkExit(Exit.OK);
 74 
 75         checkOutput("serialized-form.html", true,
 76                 """
 77                     <div class="member-signature"><span class="modifiers">public</span>&nbsp;<span c\
 78                     lass="return-type">void</span>&nbsp;<span class="member-name">readObject</span>&\
 79                     #8203;(<span class="parameters">java.io.ObjectInputStream&nbsp;arg0)</span>
 80                                     throws <span class="exceptions">java.lang.ClassNotFoundException,
 81                     java.io.IOException</span></div>
 82                     """);
 83     }
 84 
 85     void createTestClass(Path base, Path srcDir) throws Exception {
 86         //create A.java , compile the class in classes dir
 87         Path classes = base.resolve("classes");
 88         tb.createDirectories(classes);
 89 
 90         MethodBuilder method = MethodBuilder
 91                 .parse("public void readObject(ObjectInputStream s)"
 92                         + "throws ClassNotFoundException, IOException {}")
 93                 .setComments(
 94                     "@param s a serialization stream",
 95                     "@throws ClassNotFoundException if class not found",
 96                     "@throws java.io.IOException if an IO error",
 97                     "@serial");
 98 
 99         new ClassBuilder(tb, "A")
100                 .setModifiers("public", "abstract", "class")
101                 .addImplements("Serializable")
102                 .addImports("java.io.*")
103                 .addMembers(method)
104                 .write(srcDir);
105         new JavacTask(tb).files(srcDir.resolve("A.java")).outdir(classes).run();
106 
107         new ClassBuilder(tb, "B")
108                 .setExtends("A")
109                 .setModifiers("public", "class")
110                 .write(srcDir);
111     }
112 }
