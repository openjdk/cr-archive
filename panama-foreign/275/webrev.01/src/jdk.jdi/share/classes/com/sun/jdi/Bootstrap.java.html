26 package com.sun.jdi;
27 
28 import com.sun.jdi.connect.Connector;
29 import com.sun.tools.jdi.VirtualMachineManagerImpl;
30 
31 /**
32  * Initial class that provides access to the default implementation
33  * of JDI interfaces. A debugger application uses this class to access the
34  * single instance of the {@link VirtualMachineManager} interface.
35  *
36  * @author Gordon Hirsch
37  * @since  1.3
38  */
39 
40 public class Bootstrap extends Object {
41     /**
42      * Do not call.
43      */
44     @Deprecated(since="16", forRemoval=true)
45     public Bootstrap() {}
46 
47     /**
48      * Returns the virtual machine manager.
49      *
50      * <p> May throw an unspecified error if initialization of the
51      * {@link VirtualMachineManager} fails or if the virtual machine manager
52      * is unable to locate or create any {@link Connector Connectors}.
53      *
54      * @throws java.lang.SecurityException if a security manager has been
55      * installed and it denies {@link JDIPermission}
56      * ("{@code virtualMachineManager}") or other unspecified
57      * permissions required by the implementation.
58      */
59     static public synchronized VirtualMachineManager virtualMachineManager() {
60         return VirtualMachineManagerImpl.virtualMachineManager();
61     }
62 }
