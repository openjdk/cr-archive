 195 struct pc_nmethod_stats_struct {
 196   int pc_desc_resets;   // number of resets (= number of caches)
 197   int pc_desc_queries;  // queries to nmethod::find_pc_desc
 198   int pc_desc_approx;   // number of those which have approximate true
 199   int pc_desc_repeats;  // number of _pc_descs[0] hits
 200   int pc_desc_hits;     // number of LRU cache hits
 201   int pc_desc_tests;    // total number of PcDesc examinations
 202   int pc_desc_searches; // total number of quasi-binary search steps
 203   int pc_desc_adds;     // number of LUR cache insertions
 204 
 205   void print_pc_stats() {
 206     tty->print_cr("PcDesc Statistics:  %d queries, %.2f comparisons per query",
 207                   pc_desc_queries,
 208                   (double)(pc_desc_tests + pc_desc_searches)
 209                   / pc_desc_queries);
 210     tty->print_cr("  caches=%d queries=%d/%d, hits=%d+%d, tests=%d+%d, adds=%d",
 211                   pc_desc_resets,
 212                   pc_desc_queries, pc_desc_approx,
 213                   pc_desc_repeats, pc_desc_hits,
 214                   pc_desc_tests, pc_desc_searches, pc_desc_adds);
 215   }
 216 };

 271 address ExceptionCache::match(Handle exception, address pc) {
 272   assert(pc != NULL,"Must be non null");
 273   assert(exception.not_null(),"Must be non null");
 274   if (exception->klass() == exception_type()) {
 275     return (test_address(pc));
 276   }
 277 
 278   return NULL;
 279 }

 282 bool ExceptionCache::match_exception_with_space(Handle exception) {
 283   assert(exception.not_null(),"Must be non null");
 284   if (exception->klass() == exception_type() && count() < cache_size) {
 285     return true;
 286   }
 287   return false;

 291 address ExceptionCache::test_address(address addr) {
 292   int limit = count();
 293   for (int i = 0; i < limit; i++) {
 294     if (pc_at(i) == addr) {
 295       return handler_at(i);
 296     }
 297   }
 298   return NULL;
 299 }

 302 bool ExceptionCache::add_address_and_handler(address addr, address handler) {
 303   if (test_address(addr) == handler) return true;
 304 
 305   int index = count();
 306   if (index < cache_size) {
 307     set_pc_at(index, addr);
 308     set_handler_at(index, handler);
 309     increment_count();
 310     return true;
 311   }
 312   return false;
 313 }

 326 // Helper used by both find_pc_desc methods.
 327 static inline bool match_desc(PcDesc* pc, int pc_offset, bool approximate) {
 328   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_tests);
 329   if (!approximate)
 330     return pc->pc_offset() == pc_offset;
 331   else
 332     return (pc-1)->pc_offset() < pc_offset && pc_offset <= pc->pc_offset();
 333 }
 334 
 335 void PcDescCache::reset_to(PcDesc* initial_pc_desc) {
 336   if (initial_pc_desc == NULL) {
 337     _pc_descs[0] = NULL; // native method; no PcDescs at all
 338     return;
 339   }
 340   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_resets);
 341   // reset the cache by filling it with benign (non-null) values
 342   assert(initial_pc_desc->pc_offset() < 0, "must be sentinel");
 343   for (int i = 0; i < cache_size; i++)
 344     _pc_descs[i] = initial_pc_desc;
 345 }
 346 
 347 PcDesc* PcDescCache::find_pc_desc(int pc_offset, bool approximate) {
 348   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_queries);
 349   NOT_PRODUCT(if (approximate) ++pc_nmethod_stats.pc_desc_approx);
 350 
 351   // Note: one might think that caching the most recently
 352   // read value separately would be a win, but one would be
 353   // wrong.  When many threads are updating it, the cache
 354   // line it's in would bounce between caches, negating
 355   // any benefit.
 356 
 357   // In order to prevent race conditions do not load cache elements
 358   // repeatedly, but use a local copy:
 359   PcDesc* res;
 360 
 361   // Step one:  Check the most recently added value.
 362   res = _pc_descs[0];
 363   if (res == NULL) return NULL;  // native method; no PcDescs at all
 364   if (match_desc(res, pc_offset, approximate)) {
 365     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_repeats);
 366     return res;
 367   }
 368 
 369   // Step two:  Check the rest of the LRU cache.
 370   for (int i = 1; i < cache_size; ++i) {
 371     res = _pc_descs[i];
 372     if (res->pc_offset() < 0) break;  // optimization: skip empty cache
 373     if (match_desc(res, pc_offset, approximate)) {
 374       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_hits);
 375       return res;
 376     }
 377   }
 378 
 379   // Report failure.
 380   return NULL;
 381 }
 382 
 383 void PcDescCache::add_pc_desc(PcDesc* pc_desc) {
 384   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_adds);
 385   // Update the LRU cache by shifting pc_desc forward.
 386   for (int i = 0; i < cache_size; i++)  {
 387     PcDesc* next = _pc_descs[i];
 388     _pc_descs[i] = pc_desc;
 389     pc_desc = next;
 390   }
 391 }
 392 
 393 // adjust pcs_size so that it is a multiple of both oopSize and
 394 // sizeof(PcDesc) (assumes that if sizeof(PcDesc) is not a multiple
 395 // of oopSize, then 2*sizeof(PcDesc) is)
 396 static int adjust_pcs_size(int pcs_size) {
 397   int nsize = align_up(pcs_size,   oopSize);
 398   if ((nsize % sizeof(PcDesc)) != 0) {
 399     nsize = pcs_size + sizeof(PcDesc);
 400   }
 401   assert((nsize % oopSize) == 0, "correct alignment");
 402   return nsize;
 403 }

 448 nmethod* nmethod::new_native_nmethod(const methodHandle& method,
 449   int compile_id,
 450   CodeBuffer *code_buffer,
 451   int vep_offset,
 452   int frame_complete,
 453   int frame_size,
 454   ByteSize basic_lock_owner_sp_offset,
 455   ByteSize basic_lock_sp_offset,
 456   OopMapSet* oop_maps) {
 457   code_buffer->finalize_oop_references(method);
 458   // create nmethod
 459   nmethod* nm = NULL;
 460   {
 461     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 462     int native_nmethod_size = CodeBlob::allocation_size(code_buffer, sizeof(nmethod));
 463 
 464     CodeOffsets offsets;
 465     offsets.set_value(CodeOffsets::Verified_Entry, vep_offset);
 466     offsets.set_value(CodeOffsets::Frame_Complete, frame_complete);
 467     nm = new (native_nmethod_size, CompLevel_none)
 468     nmethod(method(), compiler_none, native_nmethod_size,
 469             compile_id, &offsets,
 470             code_buffer, frame_size,
 471             basic_lock_owner_sp_offset,
 472             basic_lock_sp_offset,
 473             oop_maps);
 474     NOT_PRODUCT(if (nm != NULL)  native_nmethod_stats.note_native_nmethod(nm));
 475   }
 476 
 477   if (nm != NULL) {
 478     // verify nmethod
 479     debug_only(nm->verify();) // might block
 480 
 481     nm->log_new_nmethod();
 482   }
 483   return nm;
 484 }

 486 nmethod* nmethod::new_nmethod(const methodHandle& method,
 487   int compile_id,
 488   int entry_bci,
 489   CodeOffsets* offsets,
 490   int orig_pc_offset,
 491   DebugInformationRecorder* debug_info,
 492   Dependencies* dependencies,
 493   CodeBuffer* code_buffer, int frame_size,
 494   OopMapSet* oop_maps,
 495   ExceptionHandlerTable* handler_table,
 496   ImplicitExceptionTable* nul_chk_table,
 497   AbstractCompiler* compiler,
 498   int comp_level,
 499   address* native_stubs,
 500   int num_stubs
 501 #if INCLUDE_JVMCI
 502   , char* speculations,
 503   int speculations_len,
 504   int nmethod_mirror_index,
 505   const char* nmethod_mirror_name,
 506   FailedSpeculation** failed_speculations
 507 #endif
 508 )
 509 {
 510   assert(debug_info->oop_recorder() == code_buffer->oop_recorder(), "shared OR");
 511   code_buffer->finalize_oop_references(method);
 512   // create nmethod
 513   nmethod* nm = NULL;
 514   { MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 515 #if INCLUDE_JVMCI
 516     int jvmci_data_size = !compiler->is_jvmci() ? 0 : JVMCINMethodData::compute_size(nmethod_mirror_name);
 517 #endif
 518     int nmethod_size =
 519       CodeBlob::allocation_size(code_buffer, sizeof(nmethod))
 520       + adjust_pcs_size(debug_info->pcs_size())
 521       + align_up((int)dependencies->size_in_bytes(), oopSize)
 522       + align_up(handler_table->size_in_bytes()    , oopSize)
 523       + align_up(nul_chk_table->size_in_bytes()    , oopSize)
 524 #if INCLUDE_JVMCI
 525       + align_up(speculations_len                  , oopSize)
 526       + align_up(jvmci_data_size                   , oopSize)
 527 #endif
 528       + align_up(debug_info->data_size()           , oopSize);
 529 
 530     nm = new (nmethod_size, comp_level)
 531     nmethod(method(), compiler->type(), nmethod_size, compile_id, entry_bci, offsets,
 532             orig_pc_offset, debug_info, dependencies, code_buffer, frame_size,
 533             oop_maps,
 534             handler_table,
 535             nul_chk_table,
 536             compiler,
 537             comp_level,
 538             native_stubs,
 539             num_stubs
 540 #if INCLUDE_JVMCI
 541             , speculations,
 542             speculations_len,
 543             jvmci_data_size
 544 #endif
 545             );
 546 
 547     if (nm != NULL) {
 548 #if INCLUDE_JVMCI
 549       if (compiler->is_jvmci()) {
 550         // Initialize the JVMCINMethodData object inlined into nm
 551         nm->jvmci_nmethod_data()->initialize(nmethod_mirror_index, nmethod_mirror_name, failed_speculations);
 552       }
 553 #endif
 554       // To make dependency checking during class loading fast, record
 555       // the nmethod dependencies in the classes it is dependent on.
 556       // This allows the dependency checking code to simply walk the
 557       // class hierarchy above the loaded class, checking only nmethods
 558       // which are dependent on those classes.  The slow way is to
 559       // check every nmethod for dependencies which makes it linear in
 560       // the number of methods compiled.  For applications with a lot
 561       // classes the slow way is too slow.
 562       for (Dependencies::DepStream deps(nm); deps.next(); ) {
 563         if (deps.type() == Dependencies::call_site_target_value) {
 564           // CallSite dependencies are managed on per-CallSite instance basis.
 565           oop call_site = deps.argument_oop(0);
 566           MethodHandles::add_dependent_nmethod(call_site, nm);
 567         } else {
 568           Klass* klass = deps.context_type();
 569           if (klass == NULL) {
 570             continue;  // ignore things like evol_method
 571           }
 572           // record this nmethod as dependent on this klass
 573           InstanceKlass::cast(klass)->add_dependent_nmethod(nm);
 574         }
 575       }
 576       NOT_PRODUCT(if (nm != NULL)  note_java_nmethod(nm));
 577     }
 578   }
 579   // Do verification and logging outside CodeCache_lock.
 580   if (nm != NULL) {
 581     // Safepoints in nmethod::verify aren't allowed because nm hasn't been installed yet.
 582     DEBUG_ONLY(nm->verify();)
 583     nm->log_new_nmethod();
 584   }
 585   return nm;
 586 }

1061 void nmethod::fix_oop_relocations(address begin, address end, bool initialize_immediates) {
1062   // re-patch all oop-bearing instructions, just in case some oops moved
1063   RelocIterator iter(this, begin, end);
1064   while (iter.next()) {
1065     if (iter.type() == relocInfo::oop_type) {
1066       oop_Relocation* reloc = iter.oop_reloc();
1067       if (initialize_immediates && reloc->oop_is_immediate()) {
1068         oop* dest = reloc->oop_addr();
1069         initialize_immediate_oop(dest, cast_from_oop<jobject>(*dest));
1070       }
1071       // Refresh the oop-related bits of this instruction.
1072       reloc->fix_oop_relocation();
1073     } else if (iter.type() == relocInfo::metadata_type) {
1074       metadata_Relocation* reloc = iter.metadata_reloc();
1075       reloc->fix_metadata_relocation();
1076     }
1077   }

1081 void nmethod::verify_clean_inline_caches() {
1082   assert(CompiledICLocker::is_safe(this), "mt unsafe call");
1083 
1084   ResourceMark rm;
1085   RelocIterator iter(this, oops_reloc_begin());
1086   while(iter.next()) {
1087     switch(iter.type()) {
1088       case relocInfo::virtual_call_type:
1089       case relocInfo::opt_virtual_call_type: {
1090         CompiledIC *ic = CompiledIC_at(&iter);
1091         // Ok, to lookup references to zombies here
1092         CodeBlob *cb = CodeCache::find_blob_unsafe(ic->ic_destination());
1093         assert(cb != NULL, "destination not in CodeBlob?");
1094         nmethod* nm = cb->as_nmethod_or_null();
1095         if( nm != NULL ) {
1096           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1097           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1098             assert(ic->is_clean(), "IC should be clean");
1099           }
1100         }
1101         break;
1102       }
1103       case relocInfo::static_call_type: {
1104         CompiledStaticCall *csc = compiledStaticCall_at(iter.reloc());
1105         CodeBlob *cb = CodeCache::find_blob_unsafe(csc->destination());
1106         assert(cb != NULL, "destination not in CodeBlob?");
1107         nmethod* nm = cb->as_nmethod_or_null();
1108         if( nm != NULL ) {
1109           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1110           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1111             assert(csc->is_clean(), "IC should be clean");
1112           }
1113         }
1114         break;
1115       }
1116       default:
1117         break;
1118     }
1119   }

1130 // Tell if a non-entrant method can be converted to a zombie (i.e.,
1131 // there are no activations on the stack, not in use by the VM,
1132 // and not in use by the ServiceThread)
1133 bool nmethod::can_convert_to_zombie() {
1134   // Note that this is called when the sweeper has observed the nmethod to be
1135   // not_entrant. However, with concurrent code cache unloading, the state
1136   // might have moved on to unloaded if it is_unloading(), due to racing
1137   // concurrent GC threads.
1138   assert(is_not_entrant() || is_unloading() ||
1139          !Thread::current()->is_Code_cache_sweeper_thread(),
1140          "must be a non-entrant method if called from sweeper");
1141 
1142   // Since the nmethod sweeper only does partial sweep the sweeper's traversal
1143   // count can be greater than the stack traversal count before it hits the
1144   // nmethod for the second time.
1145   // If an is_unloading() nmethod is still not_entrant, then it is not safe to
1146   // convert it to zombie due to GC unloading interactions. However, if it
1147   // has become unloaded, then it is okay to convert such nmethods to zombie.
1148   return stack_traversal_mark() + 1 < NMethodSweeper::traversal_count() &&
1149          !is_locked_by_vm() && (!is_unloading() || is_unloaded());
1150 }

1163 bool nmethod::try_transition(int new_state_int) {
1164   signed char new_state = new_state_int;
1165 #ifdef DEBUG
1166   if (new_state != unloaded) {
1167     assert_lock_strong(CompiledMethod_lock);
1168   }
1169 #endif
1170   for (;;) {
1171     signed char old_state = Atomic::load(&_state);
1172     if (old_state >= new_state) {
1173       // Ensure monotonicity of transitions.
1174       return false;
1175     }
1176     if (Atomic::cmpxchg(&_state, old_state, new_state) == old_state) {
1177       return true;
1178     }
1179   }

1182 void nmethod::make_unloaded() {
1183   post_compiled_method_unload();
1184 
1185   // This nmethod is being unloaded, make sure that dependencies
1186   // recorded in instanceKlasses get flushed.
1187   // Since this work is being done during a GC, defer deleting dependencies from the
1188   // InstanceKlass.
1189   assert(Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread(),
1190          "should only be called during gc");
1191   flush_dependencies(/*delete_immediately*/false);
1192 
1193   // Break cycle between nmethod & method
1194   LogTarget(Trace, class, unload, nmethod) lt;
1195   if (lt.is_enabled()) {
1196     LogStream ls(lt);
1197     ls.print("making nmethod " INTPTR_FORMAT
1198              " unloadable, Method*(" INTPTR_FORMAT
1199              ") ",
1200              p2i(this), p2i(_method));
1201      ls.cr();
1202   }
1203   // Unlink the osr method, so we do not look this up again
1204   if (is_osr_method()) {
1205     // Invalidate the osr nmethod only once. Note that with concurrent
1206     // code cache unloading, OSR nmethods are invalidated before they
1207     // are made unloaded. Therefore, this becomes a no-op then.
1208     if (is_in_use()) {
1209       invalidate_osr_method();
1210     }
1211 #ifdef ASSERT
1212     if (method() != NULL) {
1213       // Make sure osr nmethod is invalidated, i.e. not on the list
1214       bool found = method()->method_holder()->remove_osr_nmethod(this);
1215       assert(!found, "osr nmethod should have been invalidated");
1216     }
1217 #endif
1218   }
1219 
1220   // If _method is already NULL the Method* is about to be unloaded,
1221   // so we don't have to break the cycle. Note that it is possible to
1222   // have the Method* live here, in case we unload the nmethod because
1223   // it is pointing to some oop (other than the Method*) being unloaded.
1224   if (_method != NULL) {
1225     _method->unlink_code(this);
1226   }
1227 
1228   // Make the class unloaded - i.e., change state and notify sweeper
1229   assert(SafepointSynchronize::is_at_safepoint() || Thread::current()->is_ConcurrentGC_thread(),
1230          "must be at safepoint");
1231 
1232   {
1233     // Clear ICStubs and release any CompiledICHolders.
1234     CompiledICLocker ml(this);
1235     clear_ic_callsites();
1236   }
1237 
1238   // Unregister must be done before the state change
1239   {
1240     MutexLocker ml(SafepointSynchronize::is_at_safepoint() ? NULL : CodeCache_lock,
1241                      Mutex::_no_safepoint_check_flag);
1242     Universe::heap()->unregister_nmethod(this);
1243   }
1244 
1245   // Clear the method of this dead nmethod
1246   set_method(NULL);
1247 
1248   // Log the unloading.
1249   log_state_change();
1250 
1251   // The Method* is gone at this point
1252   assert(_method == NULL, "Tautology");
1253 
1254   set_osr_link(NULL);
1255   NMethodSweeper::report_state_change(this);
1256 
1257   bool transition_success = try_transition(unloaded);
1258 
1259   // It is an important invariant that there exists no race between
1260   // the sweeper and GC thread competing for making the same nmethod
1261   // zombie and unloaded respectively. This is ensured by
1262   // can_convert_to_zombie() returning false for any is_unloading()
1263   // nmethod, informing the sweeper not to step on any GC toes.
1264   assert(transition_success, "Invalid nmethod transition to unloaded");
1265 
1266 #if INCLUDE_JVMCI
1267   // Clear the link between this nmethod and a HotSpotNmethod mirror
1268   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1269   if (nmethod_data != NULL) {
1270     nmethod_data->invalidate_nmethod_mirror(this);
1271     nmethod_data->clear_nmethod_mirror(this);
1272   }
1273 #endif
1274 }

1315 /**
1316  * Common functionality for both make_not_entrant and make_zombie
1317  */
1318 bool nmethod::make_not_entrant_or_zombie(int state) {
1319   assert(state == zombie || state == not_entrant, "must be zombie or not_entrant");
1320 
1321   if (Atomic::load(&_state) >= state) {
1322     // Avoid taking the lock if already in required state.
1323     // This is safe from races because the state is an end-state,
1324     // which the nmethod cannot back out of once entered.
1325     // No need for fencing either.
1326     return false;
1327   }
1328 
1329   // Make sure the nmethod is not flushed.
1330   nmethodLocker nml(this);
1331   // This can be called while the system is already at a safepoint which is ok
1332   NoSafepointVerifier nsv;
1333 
1334   // during patching, depending on the nmethod state we must notify the GC that
1335   // code has been unloaded, unregistering it. We cannot do this right while
1336   // holding the CompiledMethod_lock because we need to use the CodeCache_lock. This
1337   // would be prone to deadlocks.
1338   // This flag is used to remember whether we need to later lock and unregister.
1339   bool nmethod_needs_unregister = false;
1340 
1341   {
1342     // Enter critical section.  Does not block for safepoint.
1343     MutexLocker ml(CompiledMethod_lock->owned_by_self() ? NULL : CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1344 
1345     // This logic is equivalent to the logic below for patching the
1346     // verified entry point of regular methods. We check that the
1347     // nmethod is in use to ensure that it is invalidated only once.
1348     if (is_osr_method() && is_in_use()) {
1349       // this effectively makes the osr nmethod not entrant
1350       invalidate_osr_method();
1351     }
1352 
1353     if (Atomic::load(&_state) >= state) {
1354       // another thread already performed this transition so nothing
1355       // to do, but return false to indicate this.
1356       return false;
1357     }
1358 
1359     // The caller can be calling the method statically or through an inline
1360     // cache call.
1361     if (!is_osr_method() && !is_not_entrant()) {
1362       NativeJump::patch_verified_entry(entry_point(), verified_entry_point(),
1363                   SharedRuntime::get_handle_wrong_method_stub());
1364     }
1365 
1366     if (is_in_use() && update_recompile_counts()) {
1367       // It's a true state change, so mark the method as decompiled.
1368       // Do it only for transition from alive.
1369       inc_decompile_count();
1370     }
1371 
1372     // If the state is becoming a zombie, signal to unregister the nmethod with
1373     // the heap.
1374     // This nmethod may have already been unloaded during a full GC.
1375     if ((state == zombie) && !is_unloaded()) {
1376       nmethod_needs_unregister = true;
1377     }
1378 
1379     // Must happen before state change. Otherwise we have a race condition in
1380     // nmethod::can_convert_to_zombie(). I.e., a method can immediately
1381     // transition its state from 'not_entrant' to 'zombie' without having to wait
1382     // for stack scanning.
1383     if (state == not_entrant) {
1384       mark_as_seen_on_stack();
1385       OrderAccess::storestore(); // _stack_traversal_mark and _state
1386     }
1387 
1388     // Change state
1389     if (!try_transition(state)) {
1390       // If the transition fails, it is due to another thread making the nmethod more
1391       // dead. In particular, one thread might be making the nmethod unloaded concurrently.
1392       // If so, having patched in the jump in the verified entry unnecessarily is fine.
1393       // The nmethod is no longer possible to call by Java threads.
1394       // Incrementing the decompile count is also fine as the caller of make_not_entrant()
1395       // had a valid reason to deoptimize the nmethod.
1396       // Marking the nmethod as seen on stack also has no effect, as the nmethod is now
1397       // !is_alive(), and the seen on stack value is only used to convert not_entrant
1398       // nmethods to zombie in can_convert_to_zombie().
1399       return false;
1400     }
1401 
1402     // Log the transition once
1403     log_state_change();
1404 
1405     // Remove nmethod from method.
1406     unlink_from_method();
1407 
1408   } // leave critical region under CompiledMethod_lock
1409 
1410 #if INCLUDE_JVMCI
1411   // Invalidate can't occur while holding the Patching lock
1412   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1413   if (nmethod_data != NULL) {
1414     nmethod_data->invalidate_nmethod_mirror(this);
1415   }
1416 #endif
1417 
1418 #ifdef ASSERT
1419   if (is_osr_method() && method() != NULL) {
1420     // Make sure osr nmethod is invalidated, i.e. not on the list
1421     bool found = method()->method_holder()->remove_osr_nmethod(this);
1422     assert(!found, "osr nmethod should have been invalidated");
1423   }
1424 #endif
1425 
1426   // When the nmethod becomes zombie it is no longer alive so the
1427   // dependencies must be flushed.  nmethods in the not_entrant
1428   // state will be flushed later when the transition to zombie
1429   // happens or they get unloaded.
1430   if (state == zombie) {
1431     {
1432       // Flushing dependencies must be done before any possible
1433       // safepoint can sneak in, otherwise the oops used by the
1434       // dependency logic could have become stale.
1435       MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1436       if (nmethod_needs_unregister) {
1437         Universe::heap()->unregister_nmethod(this);
1438       }
1439       flush_dependencies(/*delete_immediately*/true);
1440     }
1441 
1442 #if INCLUDE_JVMCI
1443     // Now that the nmethod has been unregistered, it's
1444     // safe to clear the HotSpotNmethod mirror oop.
1445     if (nmethod_data != NULL) {
1446       nmethod_data->clear_nmethod_mirror(this);
1447     }
1448 #endif
1449 
1450     // Clear ICStubs to prevent back patching stubs of zombie or flushed
1451     // nmethods during the next safepoint (see ICStub::finalize), as well
1452     // as to free up CompiledICHolder resources.
1453     {
1454       CompiledICLocker ml(this);
1455       clear_ic_callsites();
1456     }
1457 
1458     // zombie only - if a JVMTI agent has enabled the CompiledMethodUnload
1459     // event and it hasn't already been reported for this nmethod then
1460     // report it now. The event may have been reported earlier if the GC
1461     // marked it for unloading). JvmtiDeferredEventQueue support means
1462     // we no longer go to a safepoint here.
1463     post_compiled_method_unload();
1464 
1465 #ifdef ASSERT
1466     // It's no longer safe to access the oops section since zombie
1467     // nmethods aren't scanned for GC.
1468     _oops_are_stale = true;
1469 #endif
1470      // the Method may be reclaimed by class unloading now that the
1471      // nmethod is in zombie state
1472     set_method(NULL);
1473   } else {
1474     assert(state == not_entrant, "other cases may need to be handled differently");
1475   }
1476 
1477   if (TraceCreateZombies && state == zombie) {
1478     ResourceMark m;
1479     tty->print_cr("nmethod <" INTPTR_FORMAT "> %s code made %s", p2i(this), this->method() ? this->method()->name_and_sig_as_C_string() : "null", (state == not_entrant) ? "not entrant" : "zombie");
1480   }
1481 
1482   NMethodSweeper::report_state_change(this);
1483   return true;
1484 }

1486 void nmethod::flush() {
1487   MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1488   // Note that there are no valid oops in the nmethod anymore.
1489   assert(!is_osr_method() || is_unloaded() || is_zombie(),
1490          "osr nmethod must be unloaded or zombie before flushing");
1491   assert(is_zombie() || is_osr_method(), "must be a zombie method");
1492   assert (!is_locked_by_vm(), "locked methods shouldn't be flushed");
1493   assert_locked_or_safepoint(CodeCache_lock);
1494 
1495   // completely deallocate this method
1496   Events::log(JavaThread::current(), "flushing nmethod " INTPTR_FORMAT, p2i(this));
1497   if (PrintMethodFlushing) {
1498     tty->print_cr("*flushing %s nmethod %3d/" INTPTR_FORMAT ". Live blobs:" UINT32_FORMAT
1499                   "/Free CodeCache:" SIZE_FORMAT "Kb",
1500                   is_osr_method() ? "osr" : "",_compile_id, p2i(this), CodeCache::blob_count(),
1501                   CodeCache::unallocated_capacity(CodeCache::get_code_blob_type(this))/1024);
1502   }
1503 
1504   // We need to deallocate any ExceptionCache data.
1505   // Note that we do not need to grab the nmethod lock for this, it
1506   // better be thread safe if we're disposing of it!
1507   ExceptionCache* ec = exception_cache();
1508   set_exception_cache(NULL);
1509   while(ec != NULL) {
1510     ExceptionCache* next = ec->next();
1511     delete ec;
1512     ec = next;
1513   }
1514 
1515   Universe::heap()->flush_nmethod(this);
1516   CodeCache::unregister_old_nmethod(this);
1517 
1518   CodeBlob::flush();
1519   CodeCache::free(this);
1520 }

1536 //
1537 // Notify all classes this nmethod is dependent on that it is no
1538 // longer dependent. This should only be called in two situations.
1539 // First, when a nmethod transitions to a zombie all dependents need
1540 // to be clear.  Since zombification happens at a safepoint there's no
1541 // synchronization issues.  The second place is a little more tricky.
1542 // During phase 1 of mark sweep class unloading may happen and as a
1543 // result some nmethods may get unloaded.  In this case the flushing
1544 // of dependencies must happen during phase 1 since after GC any
1545 // dependencies in the unloaded nmethod won't be updated, so
1546 // traversing the dependency information in unsafe.  In that case this
1547 // function is called with a boolean argument and this function only
1548 // notifies instanceKlasses that are reachable
1549 
1550 void nmethod::flush_dependencies(bool delete_immediately) {
1551   DEBUG_ONLY(bool called_by_gc = Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread();)
1552   assert(called_by_gc != delete_immediately,
1553   "delete_immediately is false if and only if we are called during GC");
1554   if (!has_flushed_dependencies()) {
1555     set_has_flushed_dependencies();
1556     for (Dependencies::DepStream deps(this); deps.next(); ) {
1557       if (deps.type() == Dependencies::call_site_target_value) {
1558         // CallSite dependencies are managed on per-CallSite instance basis.
1559         oop call_site = deps.argument_oop(0);
1560         if (delete_immediately) {
1561           assert_locked_or_safepoint(CodeCache_lock);
1562           MethodHandles::remove_dependent_nmethod(call_site, this);
1563         } else {
1564           MethodHandles::clean_dependency_context(call_site);
1565         }
1566       } else {
1567         Klass* klass = deps.context_type();
1568         if (klass == NULL) {
1569           continue;  // ignore things like evol_method
1570         }
1571         // During GC delete_immediately is false, and liveness
1572         // of dependee determines class that needs to be updated.
1573         if (delete_immediately) {
1574           assert_locked_or_safepoint(CodeCache_lock);
1575           InstanceKlass::cast(klass)->remove_dependent_nmethod(this);
1576         } else if (klass->is_loader_alive()) {
1577           // The GC may clean dependency contexts concurrently and in parallel.
1578           InstanceKlass::cast(klass)->clean_dependency_context();
1579         }
1580       }
1581     }
1582   }
1583 }

1586 // post_compiled_method_load_event
1587 // new method for install_code() path
1588 // Transfer information from compilation to jvmti
1589 void nmethod::post_compiled_method_load_event(JvmtiThreadState* state) {
1590 
1591   // Don't post this nmethod load event if it is already dying
1592   // because the sweeper might already be deleting this nmethod.
1593   if (is_not_entrant() && can_convert_to_zombie()) {
1594     return;
1595   }
1596 
1597   // This is a bad time for a safepoint.  We don't want
1598   // this nmethod to get unloaded while we're queueing the event.
1599   NoSafepointVerifier nsv;
1600 
1601   Method* m = method();
1602   HOTSPOT_COMPILED_METHOD_LOAD(
1603       (char *) m->klass_name()->bytes(),
1604       m->klass_name()->utf8_length(),
1605       (char *) m->name()->bytes(),
1606       m->name()->utf8_length(),
1607       (char *) m->signature()->bytes(),
1608       m->signature()->utf8_length(),
1609       insts_begin(), insts_size());
1610 
1611 
1612   if (JvmtiExport::should_post_compiled_method_load()) {
1613     // Only post unload events if load events are found.
1614     set_load_reported();
1615     // If a JavaThread hasn't been passed in, let the Service thread
1616     // (which is a real Java thread) post the event
1617     JvmtiDeferredEvent event = JvmtiDeferredEvent::compiled_method_load_event(this);
1618     if (state == NULL) {
1619       // Execute any barrier code for this nmethod as if it's called, since
1620       // keeping it alive looks like stack walking.
1621       run_nmethod_entry_barrier();
1622       ServiceThread::enqueue_deferred_event(&event);
1623     } else {
1624       // This enters the nmethod barrier outside in the caller.
1625       state->enqueue_event(&event);
1626     }
1627   }
1628 }

1630 void nmethod::post_compiled_method_unload() {
1631   if (unload_reported()) {
1632     // During unloading we transition to unloaded and then to zombie
1633     // and the unloading is reported during the first transition.
1634     return;
1635   }
1636 
1637   assert(_method != NULL && !is_unloaded(), "just checking");
1638   DTRACE_METHOD_UNLOAD_PROBE(method());
1639 
1640   // If a JVMTI agent has enabled the CompiledMethodUnload event then
1641   // post the event. Sometime later this nmethod will be made a zombie
1642   // by the sweeper but the Method* will not be valid at that point.
1643   // The jmethodID is a weak reference to the Method* so if
1644   // it's being unloaded there's no way to look it up since the weak
1645   // ref will have been cleared.
1646 
1647   // Don't bother posting the unload if the load event wasn't posted.
1648   if (load_reported() && JvmtiExport::should_post_compiled_method_unload()) {
1649     assert(!unload_reported(), "already unloaded");
1650     JvmtiDeferredEvent event =
1651       JvmtiDeferredEvent::compiled_method_unload_event(
1652           method()->jmethod_id(), insts_begin());
1653     ServiceThread::enqueue_deferred_event(&event);
1654   }
1655 
1656   // The JVMTI CompiledMethodUnload event can be enabled or disabled at
1657   // any time. As the nmethod is being unloaded now we mark it has
1658   // having the unload event reported - this will ensure that we don't
1659   // attempt to report the event in the unlikely scenario where the
1660   // event is enabled at the time the nmethod is made a zombie.
1661   set_unload_reported();
1662 }

1664 // Iterate over metadata calling this function.   Used by RedefineClasses
1665 void nmethod::metadata_do(MetadataClosure* f) {
1666   {
1667     // Visit all immediate references that are embedded in the instruction stream.
1668     RelocIterator iter(this, oops_reloc_begin());
1669     while (iter.next()) {
1670       if (iter.type() == relocInfo::metadata_type) {
1671         metadata_Relocation* r = iter.metadata_reloc();
1672         // In this metadata, we must only follow those metadatas directly embedded in
1673         // the code.  Other metadatas (oop_index>0) are seen as part of
1674         // the metadata section below.
1675         assert(1 == (r->metadata_is_immediate()) +
1676                (r->metadata_addr() >= metadata_begin() && r->metadata_addr() < metadata_end()),
1677                "metadata must be found in exactly one place");
1678         if (r->metadata_is_immediate() && r->metadata_value() != NULL) {
1679           Metadata* md = r->metadata_value();
1680           if (md != _method) f->do_metadata(md);
1681         }
1682       } else if (iter.type() == relocInfo::virtual_call_type) {
1683         // Check compiledIC holders associated with this nmethod
1684         ResourceMark rm;
1685         CompiledIC *ic = CompiledIC_at(&iter);
1686         if (ic->is_icholder_call()) {
1687           CompiledICHolder* cichk = ic->cached_icholder();
1688           f->do_metadata(cichk->holder_metadata());
1689           f->do_metadata(cichk->holder_klass());
1690         } else {
1691           Metadata* ic_oop = ic->cached_metadata();
1692           if (ic_oop != NULL) {
1693             f->do_metadata(ic_oop);
1694           }
1695         }
1696       }
1697     }
1698   }
1699 
1700   // Visit the metadata section
1701   for (Metadata** p = metadata_begin(); p < metadata_end(); p++) {
1702     if (*p == Universe::non_oop_word() || *p == NULL)  continue;  // skip non-oops
1703     Metadata* md = *p;
1704     f->do_metadata(md);
1705   }
1706 
1707   // Visit metadata not embedded in the other places.
1708   if (_method != NULL) f->do_metadata(_method);
1709 }

1711 // The _is_unloading_state encodes a tuple comprising the unloading cycle
1712 // and the result of IsUnloadingBehaviour::is_unloading() fpr that cycle.
1713 // This is the bit layout of the _is_unloading_state byte: 00000CCU
1714 // CC refers to the cycle, which has 2 bits, and U refers to the result of
1715 // IsUnloadingBehaviour::is_unloading() for that unloading cycle.
1716 
1717 class IsUnloadingState: public AllStatic {
1718   static const uint8_t _is_unloading_mask = 1;
1719   static const uint8_t _is_unloading_shift = 0;
1720   static const uint8_t _unloading_cycle_mask = 6;
1721   static const uint8_t _unloading_cycle_shift = 1;
1722 
1723   static uint8_t set_is_unloading(uint8_t state, bool value) {
1724     state &= ~_is_unloading_mask;
1725     if (value) {
1726       state |= 1 << _is_unloading_shift;
1727     }
1728     assert(is_unloading(state) == value, "unexpected unloading cycle overflow");
1729     return state;
1730   }
1731 
1732   static uint8_t set_unloading_cycle(uint8_t state, uint8_t value) {
1733     state &= ~_unloading_cycle_mask;
1734     state |= value << _unloading_cycle_shift;
1735     assert(unloading_cycle(state) == value, "unexpected unloading cycle overflow");
1736     return state;
1737   }
1738 
1739 public:
1740   static bool is_unloading(uint8_t state) { return (state & _is_unloading_mask) >> _is_unloading_shift == 1; }
1741   static uint8_t unloading_cycle(uint8_t state) { return (state & _unloading_cycle_mask) >> _unloading_cycle_shift; }
1742 
1743   static uint8_t create(bool is_unloading, uint8_t unloading_cycle) {
1744     uint8_t state = 0;
1745     state = set_is_unloading(state, is_unloading);
1746     state = set_unloading_cycle(state, unloading_cycle);
1747     return state;
1748   }
1749 };
1750 
1751 bool nmethod::is_unloading() {
1752   uint8_t state = RawAccess<MO_RELAXED>::load(&_is_unloading_state);
1753   bool state_is_unloading = IsUnloadingState::is_unloading(state);
1754   uint8_t state_unloading_cycle = IsUnloadingState::unloading_cycle(state);
1755   if (state_is_unloading) {
1756     return true;
1757   }
1758   uint8_t current_cycle = CodeCache::unloading_cycle();
1759   if (state_unloading_cycle == current_cycle) {
1760     return false;
1761   }
1762 
1763   // The IsUnloadingBehaviour is responsible for checking if there are any dead
1764   // oops in the CompiledMethod, by calling oops_do on it.
1765   state_unloading_cycle = current_cycle;
1766 
1767   if (is_zombie()) {
1768     // Zombies without calculated unloading epoch are never unloading due to GC.
1769 
1770     // There are no races where a previously observed is_unloading() nmethod
1771     // suddenly becomes not is_unloading() due to here being observed as zombie.
1772 
1773     // With STW unloading, all is_alive() && is_unloading() nmethods are unlinked
1774     // and unloaded in the safepoint. That makes races where an nmethod is first
1775     // observed as is_alive() && is_unloading() and subsequently observed as
1776     // is_zombie() impossible.
1777 
1778     // With concurrent unloading, all references to is_unloading() nmethods are
1779     // first unlinked (e.g. IC caches and dependency contexts). Then a global
1780     // handshake operation is performed with all JavaThreads before finally
1781     // unloading the nmethods. The sweeper never converts is_alive() && is_unloading()
1782     // nmethods to zombies; it waits for them to become is_unloaded(). So before
1783     // the global handshake, it is impossible for is_unloading() nmethods to
1784     // racingly become is_zombie(). And is_unloading() is calculated for all is_alive()
1785     // nmethods before taking that global handshake, meaning that it will never
1786     // be recalculated after the handshake.
1787 
1788     // After that global handshake, is_unloading() nmethods are only observable
1789     // to the iterators, and they will never trigger recomputation of the cached
1790     // is_unloading_state, and hence may not suffer from such races.
1791 
1792     state_is_unloading = false;
1793   } else {
1794     state_is_unloading = IsUnloadingBehaviour::current()->is_unloading(this);
1795   }
1796 
1797   state = IsUnloadingState::create(state_is_unloading, state_unloading_cycle);
1798 
1799   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1800 
1801   return state_is_unloading;
1802 }

1806   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);

1960 void nmethod::oops_do_process_weak(OopsDoProcessor* p) {
1961   if (!oops_do_try_claim_weak_request()) {
1962     // Failed to claim for weak processing.
1963     oops_do_log_change("oops_do, mark weak request fail");
1964     return;
1965   }
1966 
1967   p->do_regular_processing(this);
1968 
1969   nmethod* old_head = oops_do_try_add_to_list_as_weak_done();
1970   if (old_head == NULL) {
1971     return;
1972   }
1973   oops_do_log_change("oops_do, mark weak done fail");
1974   // Adding to global list failed, another thread added a strong request.
1975   assert(extract_state(_oops_do_mark_link) == claim_strong_request_tag,
1976          "must be but is %u", extract_state(_oops_do_mark_link));
1977 
1978   oops_do_log_change("oops_do, mark weak request -> mark strong done");
1979 
1980   oops_do_set_strong_done(old_head);
1981   // Do missing strong processing.
1982   p->do_remaining_strong_processing(this);
1983 }

1985 void nmethod::oops_do_process_strong(OopsDoProcessor* p) {
1986   oops_do_mark_link* next_raw = oops_do_try_claim_strong_done();
1987   if (next_raw == NULL) {
1988     p->do_regular_processing(this);
1989     oops_do_add_to_list_as_strong_done();
1990     return;
1991   }
1992   // Claim failed. Figure out why and handle it.
1993   if (oops_do_has_weak_request(next_raw)) {
1994     oops_do_mark_link* old = next_raw;
1995     // Claim failed because being weak processed (state == "weak request").
1996     // Try to request deferred strong processing.
1997     next_raw = oops_do_try_add_strong_request(old);
1998     if (next_raw == old) {
1999       // Successfully requested deferred strong processing.
2000       return;
2001     }
2002     // Failed because of a concurrent transition. No longer in "weak request" state.
2003   }
2004   if (oops_do_has_any_strong_state(next_raw)) {
2005     // Already claimed for strong processing or requested for such.
2006     return;
2007   }
2008   if (oops_do_try_claim_weak_done_as_strong_done(next_raw)) {
2009     // Successfully claimed "weak done" as "strong done". Do the missing marking.
2010     p->do_remaining_strong_processing(this);
2011     return;
2012   }
2013   // Claim failed, some other thread got it.
2014 }

2122 // Finds a PcDesc with real-pc equal to "pc"
2123 PcDesc* PcDescContainer::find_pc_desc_internal(address pc, bool approximate, const PcDescSearch& search) {
2124   address base_address = search.code_begin();
2125   if ((pc < base_address) ||
2126       (pc - base_address) >= (ptrdiff_t) PcDesc::upper_offset_limit) {
2127     return NULL;  // PC is wildly out of range
2128   }
2129   int pc_offset = (int) (pc - base_address);
2130 
2131   // Check the PcDesc cache if it contains the desired PcDesc
2132   // (This as an almost 100% hit rate.)
2133   PcDesc* res = _pc_desc_cache.find_pc_desc(pc_offset, approximate);
2134   if (res != NULL) {
2135     assert(res == linear_search(search, pc_offset, approximate), "cache ok");
2136     return res;
2137   }
2138 
2139   // Fallback algorithm: quasi-linear search for the PcDesc
2140   // Find the last pc_offset less than the given offset.
2141   // The successor must be the required match, if there is a match at all.
2142   // (Use a fixed radix to avoid expensive affine pointer arithmetic.)
2143   PcDesc* lower = search.scopes_pcs_begin();
2144   PcDesc* upper = search.scopes_pcs_end();
2145   upper -= 1; // exclude final sentinel
2146   if (lower >= upper)  return NULL;  // native method; no PcDescs at all
2147 
2148 #define assert_LU_OK \
2149   /* invariant on lower..upper during the following search: */ \
2150   assert(lower->pc_offset() <  pc_offset, "sanity"); \
2151   assert(upper->pc_offset() >= pc_offset, "sanity")
2152   assert_LU_OK;
2153 
2154   // Use the last successful return as a split point.
2155   PcDesc* mid = _pc_desc_cache.last_pc_desc();
2156   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2157   if (mid->pc_offset() < pc_offset) {
2158     lower = mid;
2159   } else {
2160     upper = mid;
2161   }
2162 
2163   // Take giant steps at first (4096, then 256, then 16, then 1)
2164   const int LOG2_RADIX = 4 /*smaller steps in debug mode:*/ debug_only(-1);
2165   const int RADIX = (1 << LOG2_RADIX);
2166   for (int step = (1 << (LOG2_RADIX*3)); step > 1; step >>= LOG2_RADIX) {
2167     while ((mid = lower + step) < upper) {
2168       assert_LU_OK;
2169       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2170       if (mid->pc_offset() < pc_offset) {
2171         lower = mid;
2172       } else {
2173         upper = mid;
2174         break;
2175       }
2176     }
2177     assert_LU_OK;
2178   }
2179 
2180   // Sneak up on the value with a linear search of length ~16.
2181   while (true) {
2182     assert_LU_OK;
2183     mid = lower + 1;
2184     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2185     if (mid->pc_offset() < pc_offset) {
2186       lower = mid;
2187     } else {
2188       upper = mid;
2189       break;
2190     }
2191   }
2192 #undef assert_LU_OK
2193 
2194   if (match_desc(upper, pc_offset, approximate)) {
2195     assert(upper == linear_search(search, pc_offset, approximate), "search ok");
2196     _pc_desc_cache.add_pc_desc(upper);
2197     return upper;
2198   } else {
2199     assert(NULL == linear_search(search, pc_offset, approximate), "search ok");
2200     return NULL;
2201   }
2202 }
