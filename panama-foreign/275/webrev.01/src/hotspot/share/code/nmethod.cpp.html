 195 struct pc_nmethod_stats_struct {
 196   int pc_desc_resets;   // number of resets (= number of caches)
 197   int pc_desc_queries;  // queries to nmethod::find_pc_desc
 198   int pc_desc_approx;   // number of those which have approximate true
 199   int pc_desc_repeats;  // number of _pc_descs[0] hits
 200   int pc_desc_hits;     // number of LRU cache hits
 201   int pc_desc_tests;    // total number of PcDesc examinations
 202   int pc_desc_searches; // total number of quasi-binary search steps
 203   int pc_desc_adds;     // number of LUR cache insertions
 204 
 205   void print_pc_stats() {
 206     tty->print_cr("PcDesc Statistics:  %d queries, %.2f comparisons per query",
 207                   pc_desc_queries,
 208                   (double)(pc_desc_tests + pc_desc_searches)
 209                   / pc_desc_queries);
 210     tty->print_cr("  caches=%d queries=%d/%d, hits=%d+%d, tests=%d+%d, adds=%d",
 211                   pc_desc_resets,
 212                   pc_desc_queries, pc_desc_approx,
 213                   pc_desc_repeats, pc_desc_hits,
 214                   pc_desc_tests, pc_desc_searches, pc_desc_adds);
 215   }
 216 };

 271 address ExceptionCache::match(Handle exception, address pc) {
 272   assert(pc != NULL,"Must be non null");
 273   assert(exception.not_null(),"Must be non null");
 274   if (exception->klass() == exception_type()) {
 275     return (test_address(pc));
 276   }
 277 
 278   return NULL;
 279 }

 282 bool ExceptionCache::match_exception_with_space(Handle exception) {
 283   assert(exception.not_null(),"Must be non null");
 284   if (exception->klass() == exception_type() && count() < cache_size) {
 285     return true;
 286   }
 287   return false;

 291 address ExceptionCache::test_address(address addr) {
 292   int limit = count();
 293   for (int i = 0; i < limit; i++) {
 294     if (pc_at(i) == addr) {
 295       return handler_at(i);
 296     }
 297   }
 298   return NULL;
 299 }

 302 bool ExceptionCache::add_address_and_handler(address addr, address handler) {
 303   if (test_address(addr) == handler) return true;
 304 
 305   int index = count();
 306   if (index < cache_size) {
 307     set_pc_at(index, addr);
 308     set_handler_at(index, handler);
 309     increment_count();
 310     return true;
 311   }
 312   return false;
 313 }

 326 // Helper used by both find_pc_desc methods.
 327 static inline bool match_desc(PcDesc* pc, int pc_offset, bool approximate) {
 328   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_tests);
 329   if (!approximate)
 330     return pc->pc_offset() == pc_offset;
 331   else
 332     return (pc-1)->pc_offset() < pc_offset && pc_offset <= pc->pc_offset();
 333 }
 334 
 335 void PcDescCache::reset_to(PcDesc* initial_pc_desc) {
 336   if (initial_pc_desc == NULL) {
 337     _pc_descs[0] = NULL; // native method; no PcDescs at all
 338     return;
 339   }
 340   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_resets);
 341   // reset the cache by filling it with benign (non-null) values
 342   assert(initial_pc_desc->pc_offset() < 0, "must be sentinel");
 343   for (int i = 0; i < cache_size; i++)
 344     _pc_descs[i] = initial_pc_desc;
 345 }
 346 
 347 PcDesc* PcDescCache::find_pc_desc(int pc_offset, bool approximate) {
 348   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_queries);
 349   NOT_PRODUCT(if (approximate) ++pc_nmethod_stats.pc_desc_approx);
 350 
 351   // Note: one might think that caching the most recently
 352   // read value separately would be a win, but one would be
 353   // wrong.  When many threads are updating it, the cache
 354   // line it's in would bounce between caches, negating
 355   // any benefit.
 356 
 357   // In order to prevent race conditions do not load cache elements
 358   // repeatedly, but use a local copy:
 359   PcDesc* res;
 360 
 361   // Step one:  Check the most recently added value.
 362   res = _pc_descs[0];
 363   if (res == NULL) return NULL;  // native method; no PcDescs at all
 364   if (match_desc(res, pc_offset, approximate)) {
 365     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_repeats);
 366     return res;
 367   }
 368 
 369   // Step two:  Check the rest of the LRU cache.
 370   for (int i = 1; i < cache_size; ++i) {
 371     res = _pc_descs[i];
 372     if (res->pc_offset() < 0) break;  // optimization: skip empty cache
 373     if (match_desc(res, pc_offset, approximate)) {
 374       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_hits);
 375       return res;
 376     }
 377   }
 378 
 379   // Report failure.
 380   return NULL;
 381 }
 382 
 383 void PcDescCache::add_pc_desc(PcDesc* pc_desc) {
 384   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_adds);
 385   // Update the LRU cache by shifting pc_desc forward.
 386   for (int i = 0; i < cache_size; i++)  {
 387     PcDesc* next = _pc_descs[i];
 388     _pc_descs[i] = pc_desc;
 389     pc_desc = next;
 390   }
 391 }
 392 
 393 // adjust pcs_size so that it is a multiple of both oopSize and
 394 // sizeof(PcDesc) (assumes that if sizeof(PcDesc) is not a multiple
 395 // of oopSize, then 2*sizeof(PcDesc) is)
 396 static int adjust_pcs_size(int pcs_size) {
 397   int nsize = align_up(pcs_size,   oopSize);
 398   if ((nsize % sizeof(PcDesc)) != 0) {
 399     nsize = pcs_size + sizeof(PcDesc);
 400   }
 401   assert((nsize % oopSize) == 0, "correct alignment");
 402   return nsize;
 403 }

 448 nmethod* nmethod::new_native_nmethod(const methodHandle& method,
 449   int compile_id,
 450   CodeBuffer *code_buffer,
 451   int vep_offset,
 452   int frame_complete,
 453   int frame_size,
 454   ByteSize basic_lock_owner_sp_offset,
 455   ByteSize basic_lock_sp_offset,
 456   OopMapSet* oop_maps) {
 457   code_buffer->finalize_oop_references(method);
 458   // create nmethod
 459   nmethod* nm = NULL;
 460   {
 461     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 462     int native_nmethod_size = CodeBlob::allocation_size(code_buffer, sizeof(nmethod));
 463 
 464     CodeOffsets offsets;
 465     offsets.set_value(CodeOffsets::Verified_Entry, vep_offset);
 466     offsets.set_value(CodeOffsets::Frame_Complete, frame_complete);
 467     nm = new (native_nmethod_size, CompLevel_none)
 468     nmethod(method(), compiler_none, native_nmethod_size,
 469             compile_id, &offsets,
 470             code_buffer, frame_size,
 471             basic_lock_owner_sp_offset,
 472             basic_lock_sp_offset,
 473             oop_maps);
 474     NOT_PRODUCT(if (nm != NULL)  native_nmethod_stats.note_native_nmethod(nm));
 475   }
 476 
 477   if (nm != NULL) {
 478     // verify nmethod
 479     debug_only(nm->verify();) // might block
 480 
 481     nm->log_new_nmethod();
 482   }
 483   return nm;
 484 }

 486 nmethod* nmethod::new_nmethod(const methodHandle& method,
 487   int compile_id,
 488   int entry_bci,
 489   CodeOffsets* offsets,
 490   int orig_pc_offset,
 491   DebugInformationRecorder* debug_info,
 492   Dependencies* dependencies,
 493   CodeBuffer* code_buffer, int frame_size,
 494   OopMapSet* oop_maps,
 495   ExceptionHandlerTable* handler_table,
 496   ImplicitExceptionTable* nul_chk_table,
 497   AbstractCompiler* compiler,
 498   int comp_level,
 499   address* native_stubs,
 500   int num_stubs
 501 #if INCLUDE_JVMCI
 502   , char* speculations,
 503   int speculations_len,
 504   int nmethod_mirror_index,
 505   const char* nmethod_mirror_name,
 506   FailedSpeculation** failed_speculations
 507 #endif
 508 )
 509 {
 510   assert(debug_info->oop_recorder() == code_buffer->oop_recorder(), "shared OR");
 511   code_buffer->finalize_oop_references(method);
 512   // create nmethod
 513   nmethod* nm = NULL;
 514   { MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 515 #if INCLUDE_JVMCI
 516     int jvmci_data_size = !compiler->is_jvmci() ? 0 : JVMCINMethodData::compute_size(nmethod_mirror_name);
 517 #endif
 518     int nmethod_size =
 519       CodeBlob::allocation_size(code_buffer, sizeof(nmethod))
 520       + adjust_pcs_size(debug_info->pcs_size())
 521       + align_up((int)dependencies->size_in_bytes(), oopSize)
 522       + align_up(handler_table->size_in_bytes()    , oopSize)
 523       + align_up(nul_chk_table->size_in_bytes()    , oopSize)
 524 #if INCLUDE_JVMCI
 525       + align_up(speculations_len                  , oopSize)
 526       + align_up(jvmci_data_size                   , oopSize)
 527 #endif
 528       + align_up(debug_info->data_size()           , oopSize);
 529 
 530     nm = new (nmethod_size, comp_level)
 531     nmethod(method(), compiler->type(), nmethod_size, compile_id, entry_bci, offsets,
 532             orig_pc_offset, debug_info, dependencies, code_buffer, frame_size,
 533             oop_maps,
 534             handler_table,
 535             nul_chk_table,
 536             compiler,
 537             comp_level,
 538             native_stubs,
 539             num_stubs
 540 #if INCLUDE_JVMCI
 541             , speculations,
 542             speculations_len,
 543             jvmci_data_size
 544 #endif
 545             );
 546 
 547     if (nm != NULL) {
 548 #if INCLUDE_JVMCI
 549       if (compiler->is_jvmci()) {
 550         // Initialize the JVMCINMethodData object inlined into nm
 551         nm->jvmci_nmethod_data()->initialize(nmethod_mirror_index, nmethod_mirror_name, failed_speculations);
 552       }
 553 #endif
 554       // To make dependency checking during class loading fast, record
 555       // the nmethod dependencies in the classes it is dependent on.
 556       // This allows the dependency checking code to simply walk the
 557       // class hierarchy above the loaded class, checking only nmethods
 558       // which are dependent on those classes.  The slow way is to
 559       // check every nmethod for dependencies which makes it linear in
 560       // the number of methods compiled.  For applications with a lot
 561       // classes the slow way is too slow.
 562       for (Dependencies::DepStream deps(nm); deps.next(); ) {
 563         if (deps.type() == Dependencies::call_site_target_value) {
 564           // CallSite dependencies are managed on per-CallSite instance basis.
 565           oop call_site = deps.argument_oop(0);
 566           MethodHandles::add_dependent_nmethod(call_site, nm);
 567         } else {
 568           Klass* klass = deps.context_type();
 569           if (klass == NULL) {
 570             continue;  // ignore things like evol_method
 571           }
 572           // record this nmethod as dependent on this klass
 573           InstanceKlass::cast(klass)->add_dependent_nmethod(nm);
 574         }
 575       }
 576       NOT_PRODUCT(if (nm != NULL)  note_java_nmethod(nm));
 577     }
 578   }
 579   // Do verification and logging outside CodeCache_lock.
 580   if (nm != NULL) {
 581     // Safepoints in nmethod::verify aren't allowed because nm hasn't been installed yet.
 582     DEBUG_ONLY(nm->verify();)
 583     nm->log_new_nmethod();
 584   }
 585   return nm;
 586 }

1059 void nmethod::fix_oop_relocations(address begin, address end, bool initialize_immediates) {
1060   // re-patch all oop-bearing instructions, just in case some oops moved
1061   RelocIterator iter(this, begin, end);
1062   while (iter.next()) {
1063     if (iter.type() == relocInfo::oop_type) {
1064       oop_Relocation* reloc = iter.oop_reloc();
1065       if (initialize_immediates && reloc->oop_is_immediate()) {
1066         oop* dest = reloc->oop_addr();
1067         initialize_immediate_oop(dest, cast_from_oop<jobject>(*dest));
1068       }
1069       // Refresh the oop-related bits of this instruction.
1070       reloc->fix_oop_relocation();
1071     } else if (iter.type() == relocInfo::metadata_type) {
1072       metadata_Relocation* reloc = iter.metadata_reloc();
1073       reloc->fix_metadata_relocation();
1074     }
1075   }
1076 }

1079 void nmethod::verify_clean_inline_caches() {
1080   assert(CompiledICLocker::is_safe(this), "mt unsafe call");
1081 
1082   ResourceMark rm;
1083   RelocIterator iter(this, oops_reloc_begin());
1084   while(iter.next()) {
1085     switch(iter.type()) {
1086       case relocInfo::virtual_call_type:
1087       case relocInfo::opt_virtual_call_type: {
1088         CompiledIC *ic = CompiledIC_at(&iter);
1089         // Ok, to lookup references to zombies here
1090         CodeBlob *cb = CodeCache::find_blob_unsafe(ic->ic_destination());
1091         assert(cb != NULL, "destination not in CodeBlob?");
1092         nmethod* nm = cb->as_nmethod_or_null();
1093         if( nm != NULL ) {
1094           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1095           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1096             assert(ic->is_clean(), "IC should be clean");
1097           }
1098         }
1099         break;
1100       }
1101       case relocInfo::static_call_type: {
1102         CompiledStaticCall *csc = compiledStaticCall_at(iter.reloc());
1103         CodeBlob *cb = CodeCache::find_blob_unsafe(csc->destination());
1104         assert(cb != NULL, "destination not in CodeBlob?");
1105         nmethod* nm = cb->as_nmethod_or_null();
1106         if( nm != NULL ) {
1107           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1108           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1109             assert(csc->is_clean(), "IC should be clean");
1110           }
1111         }
1112         break;
1113       }
1114       default:
1115         break;
1116     }
1117   }

1128 // Tell if a non-entrant method can be converted to a zombie (i.e.,
1129 // there are no activations on the stack, not in use by the VM,
1130 // and not in use by the ServiceThread)
1131 bool nmethod::can_convert_to_zombie() {
1132   // Note that this is called when the sweeper has observed the nmethod to be
1133   // not_entrant. However, with concurrent code cache unloading, the state
1134   // might have moved on to unloaded if it is_unloading(), due to racing
1135   // concurrent GC threads.
1136   assert(is_not_entrant() || is_unloading() ||
1137          !Thread::current()->is_Code_cache_sweeper_thread(),
1138          "must be a non-entrant method if called from sweeper");
1139 
1140   // Since the nmethod sweeper only does partial sweep the sweeper's traversal
1141   // count can be greater than the stack traversal count before it hits the
1142   // nmethod for the second time.
1143   // If an is_unloading() nmethod is still not_entrant, then it is not safe to
1144   // convert it to zombie due to GC unloading interactions. However, if it
1145   // has become unloaded, then it is okay to convert such nmethods to zombie.
1146   return stack_traversal_mark() + 1 < NMethodSweeper::traversal_count() &&
1147          !is_locked_by_vm() && (!is_unloading() || is_unloaded());
1148 }

1161 bool nmethod::try_transition(int new_state_int) {
1162   signed char new_state = new_state_int;
1163 #ifdef DEBUG
1164   if (new_state != unloaded) {
1165     assert_lock_strong(CompiledMethod_lock);
1166   }
1167 #endif
1168   for (;;) {
1169     signed char old_state = Atomic::load(&_state);
1170     if (old_state >= new_state) {
1171       // Ensure monotonicity of transitions.
1172       return false;
1173     }
1174     if (Atomic::cmpxchg(&_state, old_state, new_state) == old_state) {
1175       return true;
1176     }
1177   }

1180 void nmethod::make_unloaded() {
1181   post_compiled_method_unload();
1182 
1183   // This nmethod is being unloaded, make sure that dependencies
1184   // recorded in instanceKlasses get flushed.
1185   // Since this work is being done during a GC, defer deleting dependencies from the
1186   // InstanceKlass.
1187   assert(Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread(),
1188          "should only be called during gc");
1189   flush_dependencies(/*delete_immediately*/false);
1190 
1191   // Break cycle between nmethod & method
1192   LogTarget(Trace, class, unload, nmethod) lt;
1193   if (lt.is_enabled()) {
1194     LogStream ls(lt);
1195     ls.print("making nmethod " INTPTR_FORMAT
1196              " unloadable, Method*(" INTPTR_FORMAT
1197              ") ",
1198              p2i(this), p2i(_method));
1199      ls.cr();
1200   }
1201   // Unlink the osr method, so we do not look this up again
1202   if (is_osr_method()) {
1203     // Invalidate the osr nmethod only once. Note that with concurrent
1204     // code cache unloading, OSR nmethods are invalidated before they
1205     // are made unloaded. Therefore, this becomes a no-op then.
1206     if (is_in_use()) {
1207       invalidate_osr_method();
1208     }
1209 #ifdef ASSERT
1210     if (method() != NULL) {
1211       // Make sure osr nmethod is invalidated, i.e. not on the list
1212       bool found = method()->method_holder()->remove_osr_nmethod(this);
1213       assert(!found, "osr nmethod should have been invalidated");
1214     }
1215 #endif
1216   }
1217 
1218   // If _method is already NULL the Method* is about to be unloaded,
1219   // so we don't have to break the cycle. Note that it is possible to
1220   // have the Method* live here, in case we unload the nmethod because
1221   // it is pointing to some oop (other than the Method*) being unloaded.
1222   if (_method != NULL) {
1223     _method->unlink_code(this);
1224   }
1225 
1226   // Make the class unloaded - i.e., change state and notify sweeper
1227   assert(SafepointSynchronize::is_at_safepoint() || Thread::current()->is_ConcurrentGC_thread(),
1228          "must be at safepoint");
1229 
1230   {
1231     // Clear ICStubs and release any CompiledICHolders.
1232     CompiledICLocker ml(this);
1233     clear_ic_callsites();
1234   }
1235 
1236   // Unregister must be done before the state change
1237   {
1238     MutexLocker ml(SafepointSynchronize::is_at_safepoint() ? NULL : CodeCache_lock,
1239                      Mutex::_no_safepoint_check_flag);
1240     Universe::heap()->unregister_nmethod(this);
1241   }
1242 
1243   // Clear the method of this dead nmethod
1244   set_method(NULL);
1245 
1246   // Log the unloading.
1247   log_state_change();
1248 
1249   // The Method* is gone at this point
1250   assert(_method == NULL, "Tautology");
1251 
1252   set_osr_link(NULL);
1253   NMethodSweeper::report_state_change(this);
1254 
1255   bool transition_success = try_transition(unloaded);
1256 
1257   // It is an important invariant that there exists no race between
1258   // the sweeper and GC thread competing for making the same nmethod
1259   // zombie and unloaded respectively. This is ensured by
1260   // can_convert_to_zombie() returning false for any is_unloading()
1261   // nmethod, informing the sweeper not to step on any GC toes.
1262   assert(transition_success, "Invalid nmethod transition to unloaded");
1263 
1264 #if INCLUDE_JVMCI
1265   // Clear the link between this nmethod and a HotSpotNmethod mirror
1266   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1267   if (nmethod_data != NULL) {
1268     nmethod_data->invalidate_nmethod_mirror(this);
1269     nmethod_data->clear_nmethod_mirror(this);
1270   }
1271 #endif
1272 }

1313 /**
1314  * Common functionality for both make_not_entrant and make_zombie
1315  */
1316 bool nmethod::make_not_entrant_or_zombie(int state) {
1317   assert(state == zombie || state == not_entrant, "must be zombie or not_entrant");
1318 
1319   if (Atomic::load(&_state) >= state) {
1320     // Avoid taking the lock if already in required state.
1321     // This is safe from races because the state is an end-state,
1322     // which the nmethod cannot back out of once entered.
1323     // No need for fencing either.
1324     return false;
1325   }
1326 
1327   // Make sure the nmethod is not flushed.
1328   nmethodLocker nml(this);
1329   // This can be called while the system is already at a safepoint which is ok
1330   NoSafepointVerifier nsv;
1331 
1332   // during patching, depending on the nmethod state we must notify the GC that
1333   // code has been unloaded, unregistering it. We cannot do this right while
1334   // holding the CompiledMethod_lock because we need to use the CodeCache_lock. This
1335   // would be prone to deadlocks.
1336   // This flag is used to remember whether we need to later lock and unregister.
1337   bool nmethod_needs_unregister = false;
1338 
1339   {
1340     // Enter critical section.  Does not block for safepoint.
1341     MutexLocker ml(CompiledMethod_lock->owned_by_self() ? NULL : CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1342 
1343     // This logic is equivalent to the logic below for patching the
1344     // verified entry point of regular methods. We check that the
1345     // nmethod is in use to ensure that it is invalidated only once.
1346     if (is_osr_method() && is_in_use()) {
1347       // this effectively makes the osr nmethod not entrant
1348       invalidate_osr_method();
1349     }
1350 
1351     if (Atomic::load(&_state) >= state) {
1352       // another thread already performed this transition so nothing
1353       // to do, but return false to indicate this.
1354       return false;
1355     }
1356 
1357     // The caller can be calling the method statically or through an inline
1358     // cache call.
1359     if (!is_osr_method() && !is_not_entrant()) {
1360       NativeJump::patch_verified_entry(entry_point(), verified_entry_point(),
1361                   SharedRuntime::get_handle_wrong_method_stub());
1362     }
1363 
1364     if (is_in_use() && update_recompile_counts()) {
1365       // It's a true state change, so mark the method as decompiled.
1366       // Do it only for transition from alive.
1367       inc_decompile_count();
1368     }
1369 
1370     // If the state is becoming a zombie, signal to unregister the nmethod with
1371     // the heap.
1372     // This nmethod may have already been unloaded during a full GC.
1373     if ((state == zombie) && !is_unloaded()) {
1374       nmethod_needs_unregister = true;
1375     }
1376 
1377     // Must happen before state change. Otherwise we have a race condition in
1378     // nmethod::can_convert_to_zombie(). I.e., a method can immediately
1379     // transition its state from 'not_entrant' to 'zombie' without having to wait
1380     // for stack scanning.
1381     if (state == not_entrant) {
1382       mark_as_seen_on_stack();
1383       OrderAccess::storestore(); // _stack_traversal_mark and _state
1384     }
1385 
1386     // Change state
1387     if (!try_transition(state)) {
1388       // If the transition fails, it is due to another thread making the nmethod more
1389       // dead. In particular, one thread might be making the nmethod unloaded concurrently.
1390       // If so, having patched in the jump in the verified entry unnecessarily is fine.
1391       // The nmethod is no longer possible to call by Java threads.
1392       // Incrementing the decompile count is also fine as the caller of make_not_entrant()
1393       // had a valid reason to deoptimize the nmethod.
1394       // Marking the nmethod as seen on stack also has no effect, as the nmethod is now
1395       // !is_alive(), and the seen on stack value is only used to convert not_entrant
1396       // nmethods to zombie in can_convert_to_zombie().
1397       return false;
1398     }
1399 
1400     // Log the transition once
1401     log_state_change();
1402 
1403     // Remove nmethod from method.
1404     unlink_from_method();
1405 
1406   } // leave critical region under CompiledMethod_lock
1407 
1408 #if INCLUDE_JVMCI
1409   // Invalidate can't occur while holding the Patching lock
1410   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1411   if (nmethod_data != NULL) {
1412     nmethod_data->invalidate_nmethod_mirror(this);
1413   }
1414 #endif
1415 
1416 #ifdef ASSERT
1417   if (is_osr_method() && method() != NULL) {
1418     // Make sure osr nmethod is invalidated, i.e. not on the list
1419     bool found = method()->method_holder()->remove_osr_nmethod(this);
1420     assert(!found, "osr nmethod should have been invalidated");
1421   }
1422 #endif
1423 
1424   // When the nmethod becomes zombie it is no longer alive so the
1425   // dependencies must be flushed.  nmethods in the not_entrant
1426   // state will be flushed later when the transition to zombie
1427   // happens or they get unloaded.
1428   if (state == zombie) {
1429     {
1430       // Flushing dependencies must be done before any possible
1431       // safepoint can sneak in, otherwise the oops used by the
1432       // dependency logic could have become stale.
1433       MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1434       if (nmethod_needs_unregister) {
1435         Universe::heap()->unregister_nmethod(this);
1436       }
1437       flush_dependencies(/*delete_immediately*/true);
1438     }
1439 
1440 #if INCLUDE_JVMCI
1441     // Now that the nmethod has been unregistered, it's
1442     // safe to clear the HotSpotNmethod mirror oop.
1443     if (nmethod_data != NULL) {
1444       nmethod_data->clear_nmethod_mirror(this);
1445     }
1446 #endif
1447 
1448     // Clear ICStubs to prevent back patching stubs of zombie or flushed
1449     // nmethods during the next safepoint (see ICStub::finalize), as well
1450     // as to free up CompiledICHolder resources.
1451     {
1452       CompiledICLocker ml(this);
1453       clear_ic_callsites();
1454     }
1455 
1456     // zombie only - if a JVMTI agent has enabled the CompiledMethodUnload
1457     // event and it hasn't already been reported for this nmethod then
1458     // report it now. The event may have been reported earlier if the GC
1459     // marked it for unloading). JvmtiDeferredEventQueue support means
1460     // we no longer go to a safepoint here.
1461     post_compiled_method_unload();
1462 
1463 #ifdef ASSERT
1464     // It's no longer safe to access the oops section since zombie
1465     // nmethods aren't scanned for GC.
1466     _oops_are_stale = true;
1467 #endif
1468      // the Method may be reclaimed by class unloading now that the
1469      // nmethod is in zombie state
1470     set_method(NULL);
1471   } else {
1472     assert(state == not_entrant, "other cases may need to be handled differently");
1473   }
1474 
1475   if (TraceCreateZombies && state == zombie) {
1476     ResourceMark m;
1477     tty->print_cr("nmethod <" INTPTR_FORMAT "> %s code made %s", p2i(this), this->method() ? this->method()->name_and_sig_as_C_string() : "null", (state == not_entrant) ? "not entrant" : "zombie");
1478   }
1479 
1480   NMethodSweeper::report_state_change(this);
1481   return true;
1482 }

1484 void nmethod::flush() {
1485   MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1486   // Note that there are no valid oops in the nmethod anymore.
1487   assert(!is_osr_method() || is_unloaded() || is_zombie(),
1488          "osr nmethod must be unloaded or zombie before flushing");
1489   assert(is_zombie() || is_osr_method(), "must be a zombie method");
1490   assert (!is_locked_by_vm(), "locked methods shouldn't be flushed");
1491   assert_locked_or_safepoint(CodeCache_lock);
1492 
1493   // completely deallocate this method
1494   Events::log(JavaThread::current(), "flushing nmethod " INTPTR_FORMAT, p2i(this));
1495   if (PrintMethodFlushing) {
1496     tty->print_cr("*flushing %s nmethod %3d/" INTPTR_FORMAT ". Live blobs:" UINT32_FORMAT
1497                   "/Free CodeCache:" SIZE_FORMAT "Kb",
1498                   is_osr_method() ? "osr" : "",_compile_id, p2i(this), CodeCache::blob_count(),
1499                   CodeCache::unallocated_capacity(CodeCache::get_code_blob_type(this))/1024);
1500   }
1501 
1502   // We need to deallocate any ExceptionCache data.
1503   // Note that we do not need to grab the nmethod lock for this, it
1504   // better be thread safe if we're disposing of it!
1505   ExceptionCache* ec = exception_cache();
1506   set_exception_cache(NULL);
1507   while(ec != NULL) {
1508     ExceptionCache* next = ec->next();
1509     delete ec;
1510     ec = next;
1511   }
1512 
1513   Universe::heap()->flush_nmethod(this);
1514   CodeCache::unregister_old_nmethod(this);
1515 
1516   CodeBlob::flush();
1517   CodeCache::free(this);
1518 }

1534 //
1535 // Notify all classes this nmethod is dependent on that it is no
1536 // longer dependent. This should only be called in two situations.
1537 // First, when a nmethod transitions to a zombie all dependents need
1538 // to be clear.  Since zombification happens at a safepoint there's no
1539 // synchronization issues.  The second place is a little more tricky.
1540 // During phase 1 of mark sweep class unloading may happen and as a
1541 // result some nmethods may get unloaded.  In this case the flushing
1542 // of dependencies must happen during phase 1 since after GC any
1543 // dependencies in the unloaded nmethod won't be updated, so
1544 // traversing the dependency information in unsafe.  In that case this
1545 // function is called with a boolean argument and this function only
1546 // notifies instanceKlasses that are reachable
1547 
1548 void nmethod::flush_dependencies(bool delete_immediately) {
1549   DEBUG_ONLY(bool called_by_gc = Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread();)
1550   assert(called_by_gc != delete_immediately,
1551   "delete_immediately is false if and only if we are called during GC");
1552   if (!has_flushed_dependencies()) {
1553     set_has_flushed_dependencies();
1554     for (Dependencies::DepStream deps(this); deps.next(); ) {
1555       if (deps.type() == Dependencies::call_site_target_value) {
1556         // CallSite dependencies are managed on per-CallSite instance basis.
1557         oop call_site = deps.argument_oop(0);
1558         if (delete_immediately) {
1559           assert_locked_or_safepoint(CodeCache_lock);
1560           MethodHandles::remove_dependent_nmethod(call_site, this);
1561         } else {
1562           MethodHandles::clean_dependency_context(call_site);
1563         }
1564       } else {
1565         Klass* klass = deps.context_type();
1566         if (klass == NULL) {
1567           continue;  // ignore things like evol_method
1568         }
1569         // During GC delete_immediately is false, and liveness
1570         // of dependee determines class that needs to be updated.
1571         if (delete_immediately) {
1572           assert_locked_or_safepoint(CodeCache_lock);
1573           InstanceKlass::cast(klass)->remove_dependent_nmethod(this);
1574         } else if (klass->is_loader_alive()) {
1575           // The GC may clean dependency contexts concurrently and in parallel.
1576           InstanceKlass::cast(klass)->clean_dependency_context();
1577         }
1578       }
1579     }
1580   }
1581 }

1584 // post_compiled_method_load_event
1585 // new method for install_code() path
1586 // Transfer information from compilation to jvmti
1587 void nmethod::post_compiled_method_load_event(JvmtiThreadState* state) {
1588 
1589   // Don't post this nmethod load event if it is already dying
1590   // because the sweeper might already be deleting this nmethod.
1591   if (is_not_entrant() && can_convert_to_zombie()) {
1592     return;
1593   }
1594 
1595   // This is a bad time for a safepoint.  We don't want
1596   // this nmethod to get unloaded while we're queueing the event.
1597   NoSafepointVerifier nsv;
1598 
1599   Method* m = method();
1600   HOTSPOT_COMPILED_METHOD_LOAD(
1601       (char *) m->klass_name()->bytes(),
1602       m->klass_name()->utf8_length(),
1603       (char *) m->name()->bytes(),
1604       m->name()->utf8_length(),
1605       (char *) m->signature()->bytes(),
1606       m->signature()->utf8_length(),
1607       insts_begin(), insts_size());
1608 
1609 
1610   if (JvmtiExport::should_post_compiled_method_load()) {
1611     // Only post unload events if load events are found.
1612     set_load_reported();
1613     // If a JavaThread hasn't been passed in, let the Service thread
1614     // (which is a real Java thread) post the event
1615     JvmtiDeferredEvent event = JvmtiDeferredEvent::compiled_method_load_event(this);
1616     if (state == NULL) {
1617       // Execute any barrier code for this nmethod as if it's called, since
1618       // keeping it alive looks like stack walking.
1619       run_nmethod_entry_barrier();
1620       ServiceThread::enqueue_deferred_event(&event);
1621     } else {
1622       // This enters the nmethod barrier outside in the caller.
1623       state->enqueue_event(&event);
1624     }
1625   }
1626 }

1628 void nmethod::post_compiled_method_unload() {
1629   if (unload_reported()) {
1630     // During unloading we transition to unloaded and then to zombie
1631     // and the unloading is reported during the first transition.
1632     return;
1633   }
1634 
1635   assert(_method != NULL && !is_unloaded(), "just checking");
1636   DTRACE_METHOD_UNLOAD_PROBE(method());
1637 
1638   // If a JVMTI agent has enabled the CompiledMethodUnload event then
1639   // post the event. Sometime later this nmethod will be made a zombie
1640   // by the sweeper but the Method* will not be valid at that point.
1641   // The jmethodID is a weak reference to the Method* so if
1642   // it's being unloaded there's no way to look it up since the weak
1643   // ref will have been cleared.
1644 
1645   // Don't bother posting the unload if the load event wasn't posted.
1646   if (load_reported() && JvmtiExport::should_post_compiled_method_unload()) {
1647     assert(!unload_reported(), "already unloaded");
1648     JvmtiDeferredEvent event =
1649       JvmtiDeferredEvent::compiled_method_unload_event(
1650           method()->jmethod_id(), insts_begin());
1651     ServiceThread::enqueue_deferred_event(&event);
1652   }
1653 
1654   // The JVMTI CompiledMethodUnload event can be enabled or disabled at
1655   // any time. As the nmethod is being unloaded now we mark it has
1656   // having the unload event reported - this will ensure that we don't
1657   // attempt to report the event in the unlikely scenario where the
1658   // event is enabled at the time the nmethod is made a zombie.
1659   set_unload_reported();
1660 }

1662 // Iterate over metadata calling this function.   Used by RedefineClasses
1663 void nmethod::metadata_do(MetadataClosure* f) {
1664   {
1665     // Visit all immediate references that are embedded in the instruction stream.
1666     RelocIterator iter(this, oops_reloc_begin());
1667     while (iter.next()) {
1668       if (iter.type() == relocInfo::metadata_type) {
1669         metadata_Relocation* r = iter.metadata_reloc();
1670         // In this metadata, we must only follow those metadatas directly embedded in
1671         // the code.  Other metadatas (oop_index>0) are seen as part of
1672         // the metadata section below.
1673         assert(1 == (r->metadata_is_immediate()) +
1674                (r->metadata_addr() >= metadata_begin() && r->metadata_addr() < metadata_end()),
1675                "metadata must be found in exactly one place");
1676         if (r->metadata_is_immediate() && r->metadata_value() != NULL) {
1677           Metadata* md = r->metadata_value();
1678           if (md != _method) f->do_metadata(md);
1679         }
1680       } else if (iter.type() == relocInfo::virtual_call_type) {
1681         // Check compiledIC holders associated with this nmethod
1682         ResourceMark rm;
1683         CompiledIC *ic = CompiledIC_at(&iter);
1684         if (ic->is_icholder_call()) {
1685           CompiledICHolder* cichk = ic->cached_icholder();
1686           f->do_metadata(cichk->holder_metadata());
1687           f->do_metadata(cichk->holder_klass());
1688         } else {
1689           Metadata* ic_oop = ic->cached_metadata();
1690           if (ic_oop != NULL) {
1691             f->do_metadata(ic_oop);
1692           }
1693         }
1694       }
1695     }
1696   }
1697 
1698   // Visit the metadata section
1699   for (Metadata** p = metadata_begin(); p < metadata_end(); p++) {
1700     if (*p == Universe::non_oop_word() || *p == NULL)  continue;  // skip non-oops
1701     Metadata* md = *p;
1702     f->do_metadata(md);
1703   }
1704 
1705   // Visit metadata not embedded in the other places.
1706   if (_method != NULL) f->do_metadata(_method);
1707 }

1709 // The _is_unloading_state encodes a tuple comprising the unloading cycle
1710 // and the result of IsUnloadingBehaviour::is_unloading() fpr that cycle.
1711 // This is the bit layout of the _is_unloading_state byte: 00000CCU
1712 // CC refers to the cycle, which has 2 bits, and U refers to the result of
1713 // IsUnloadingBehaviour::is_unloading() for that unloading cycle.
1714 
1715 class IsUnloadingState: public AllStatic {
1716   static const uint8_t _is_unloading_mask = 1;
1717   static const uint8_t _is_unloading_shift = 0;
1718   static const uint8_t _unloading_cycle_mask = 6;
1719   static const uint8_t _unloading_cycle_shift = 1;
1720 
1721   static uint8_t set_is_unloading(uint8_t state, bool value) {
1722     state &= ~_is_unloading_mask;
1723     if (value) {
1724       state |= 1 << _is_unloading_shift;
1725     }
1726     assert(is_unloading(state) == value, "unexpected unloading cycle overflow");
1727     return state;
1728   }
1729 
1730   static uint8_t set_unloading_cycle(uint8_t state, uint8_t value) {
1731     state &= ~_unloading_cycle_mask;
1732     state |= value << _unloading_cycle_shift;
1733     assert(unloading_cycle(state) == value, "unexpected unloading cycle overflow");
1734     return state;
1735   }
1736 
1737 public:
1738   static bool is_unloading(uint8_t state) { return (state & _is_unloading_mask) >> _is_unloading_shift == 1; }
1739   static uint8_t unloading_cycle(uint8_t state) { return (state & _unloading_cycle_mask) >> _unloading_cycle_shift; }
1740 
1741   static uint8_t create(bool is_unloading, uint8_t unloading_cycle) {
1742     uint8_t state = 0;
1743     state = set_is_unloading(state, is_unloading);
1744     state = set_unloading_cycle(state, unloading_cycle);
1745     return state;
1746   }
1747 };
1748 
1749 bool nmethod::is_unloading() {
1750   uint8_t state = RawAccess<MO_RELAXED>::load(&_is_unloading_state);
1751   bool state_is_unloading = IsUnloadingState::is_unloading(state);
1752   uint8_t state_unloading_cycle = IsUnloadingState::unloading_cycle(state);
1753   if (state_is_unloading) {
1754     return true;
1755   }
1756   uint8_t current_cycle = CodeCache::unloading_cycle();
1757   if (state_unloading_cycle == current_cycle) {
1758     return false;
1759   }
1760 
1761   // The IsUnloadingBehaviour is responsible for checking if there are any dead
1762   // oops in the CompiledMethod, by calling oops_do on it.
1763   state_unloading_cycle = current_cycle;
1764 
1765   if (is_zombie()) {
1766     // Zombies without calculated unloading epoch are never unloading due to GC.
1767 
1768     // There are no races where a previously observed is_unloading() nmethod
1769     // suddenly becomes not is_unloading() due to here being observed as zombie.
1770 
1771     // With STW unloading, all is_alive() && is_unloading() nmethods are unlinked
1772     // and unloaded in the safepoint. That makes races where an nmethod is first
1773     // observed as is_alive() && is_unloading() and subsequently observed as
1774     // is_zombie() impossible.
1775 
1776     // With concurrent unloading, all references to is_unloading() nmethods are
1777     // first unlinked (e.g. IC caches and dependency contexts). Then a global
1778     // handshake operation is performed with all JavaThreads before finally
1779     // unloading the nmethods. The sweeper never converts is_alive() && is_unloading()
1780     // nmethods to zombies; it waits for them to become is_unloaded(). So before
1781     // the global handshake, it is impossible for is_unloading() nmethods to
1782     // racingly become is_zombie(). And is_unloading() is calculated for all is_alive()
1783     // nmethods before taking that global handshake, meaning that it will never
1784     // be recalculated after the handshake.
1785 
1786     // After that global handshake, is_unloading() nmethods are only observable
1787     // to the iterators, and they will never trigger recomputation of the cached
1788     // is_unloading_state, and hence may not suffer from such races.
1789 
1790     state_is_unloading = false;
1791   } else {
1792     state_is_unloading = IsUnloadingBehaviour::current()->is_unloading(this);
1793   }
1794 
1795   state = IsUnloadingState::create(state_is_unloading, state_unloading_cycle);
1796 
1797   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1798 
1799   return state_is_unloading;
1800 }

1804   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);

1958 void nmethod::oops_do_process_weak(OopsDoProcessor* p) {
1959   if (!oops_do_try_claim_weak_request()) {
1960     // Failed to claim for weak processing.
1961     oops_do_log_change("oops_do, mark weak request fail");
1962     return;
1963   }
1964 
1965   p->do_regular_processing(this);
1966 
1967   nmethod* old_head = oops_do_try_add_to_list_as_weak_done();
1968   if (old_head == NULL) {
1969     return;
1970   }
1971   oops_do_log_change("oops_do, mark weak done fail");
1972   // Adding to global list failed, another thread added a strong request.
1973   assert(extract_state(_oops_do_mark_link) == claim_strong_request_tag,
1974          "must be but is %u", extract_state(_oops_do_mark_link));
1975 
1976   oops_do_log_change("oops_do, mark weak request -> mark strong done");
1977 
1978   oops_do_set_strong_done(old_head);
1979   // Do missing strong processing.
1980   p->do_remaining_strong_processing(this);
1981 }

1983 void nmethod::oops_do_process_strong(OopsDoProcessor* p) {
1984   oops_do_mark_link* next_raw = oops_do_try_claim_strong_done();
1985   if (next_raw == NULL) {
1986     p->do_regular_processing(this);
1987     oops_do_add_to_list_as_strong_done();
1988     return;
1989   }
1990   // Claim failed. Figure out why and handle it.
1991   if (oops_do_has_weak_request(next_raw)) {
1992     oops_do_mark_link* old = next_raw;
1993     // Claim failed because being weak processed (state == "weak request").
1994     // Try to request deferred strong processing.
1995     next_raw = oops_do_try_add_strong_request(old);
1996     if (next_raw == old) {
1997       // Successfully requested deferred strong processing.
1998       return;
1999     }
2000     // Failed because of a concurrent transition. No longer in "weak request" state.
2001   }
2002   if (oops_do_has_any_strong_state(next_raw)) {
2003     // Already claimed for strong processing or requested for such.
2004     return;
2005   }
2006   if (oops_do_try_claim_weak_done_as_strong_done(next_raw)) {
2007     // Successfully claimed "weak done" as "strong done". Do the missing marking.
2008     p->do_remaining_strong_processing(this);
2009     return;
2010   }
2011   // Claim failed, some other thread got it.
2012 }

2120 // Finds a PcDesc with real-pc equal to "pc"
2121 PcDesc* PcDescContainer::find_pc_desc_internal(address pc, bool approximate, const PcDescSearch& search) {
2122   address base_address = search.code_begin();
2123   if ((pc < base_address) ||
2124       (pc - base_address) >= (ptrdiff_t) PcDesc::upper_offset_limit) {
2125     return NULL;  // PC is wildly out of range
2126   }
2127   int pc_offset = (int) (pc - base_address);
2128 
2129   // Check the PcDesc cache if it contains the desired PcDesc
2130   // (This as an almost 100% hit rate.)
2131   PcDesc* res = _pc_desc_cache.find_pc_desc(pc_offset, approximate);
2132   if (res != NULL) {
2133     assert(res == linear_search(search, pc_offset, approximate), "cache ok");
2134     return res;
2135   }
2136 
2137   // Fallback algorithm: quasi-linear search for the PcDesc
2138   // Find the last pc_offset less than the given offset.
2139   // The successor must be the required match, if there is a match at all.
2140   // (Use a fixed radix to avoid expensive affine pointer arithmetic.)
2141   PcDesc* lower = search.scopes_pcs_begin();
2142   PcDesc* upper = search.scopes_pcs_end();
2143   upper -= 1; // exclude final sentinel
2144   if (lower >= upper)  return NULL;  // native method; no PcDescs at all
2145 
2146 #define assert_LU_OK \
2147   /* invariant on lower..upper during the following search: */ \
2148   assert(lower->pc_offset() <  pc_offset, "sanity"); \
2149   assert(upper->pc_offset() >= pc_offset, "sanity")
2150   assert_LU_OK;
2151 
2152   // Use the last successful return as a split point.
2153   PcDesc* mid = _pc_desc_cache.last_pc_desc();
2154   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2155   if (mid->pc_offset() < pc_offset) {
2156     lower = mid;
2157   } else {
2158     upper = mid;
2159   }
2160 
2161   // Take giant steps at first (4096, then 256, then 16, then 1)
2162   const int LOG2_RADIX = 4 /*smaller steps in debug mode:*/ debug_only(-1);
2163   const int RADIX = (1 << LOG2_RADIX);
2164   for (int step = (1 << (LOG2_RADIX*3)); step > 1; step >>= LOG2_RADIX) {
2165     while ((mid = lower + step) < upper) {
2166       assert_LU_OK;
2167       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2168       if (mid->pc_offset() < pc_offset) {
2169         lower = mid;
2170       } else {
2171         upper = mid;
2172         break;
2173       }
2174     }
2175     assert_LU_OK;
2176   }
2177 
2178   // Sneak up on the value with a linear search of length ~16.
2179   while (true) {
2180     assert_LU_OK;
2181     mid = lower + 1;
2182     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2183     if (mid->pc_offset() < pc_offset) {
2184       lower = mid;
2185     } else {
2186       upper = mid;
2187       break;
2188     }
2189   }
2190 #undef assert_LU_OK
2191 
2192   if (match_desc(upper, pc_offset, approximate)) {
2193     assert(upper == linear_search(search, pc_offset, approximate), "search ok");
2194     _pc_desc_cache.add_pc_desc(upper);
2195     return upper;
2196   } else {
2197     assert(NULL == linear_search(search, pc_offset, approximate), "search ok");
2198     return NULL;
2199   }
2200 }
