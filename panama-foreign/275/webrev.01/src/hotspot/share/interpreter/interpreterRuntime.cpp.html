  77 // Helper class to access current interpreter state
  78 class LastFrameAccessor : public StackObj {
  79   frame _last_frame;
  80 public:
  81   LastFrameAccessor(JavaThread* thread) {
  82     assert(thread == Thread::current(), "sanity");
  83     _last_frame = thread->last_frame();
  84   }
  85   bool is_interpreted_frame() const              { return _last_frame.is_interpreted_frame(); }
  86   Method*   method() const                       { return _last_frame.interpreter_frame_method(); }
  87   address   bcp() const                          { return _last_frame.interpreter_frame_bcp(); }
  88   int       bci() const                          { return _last_frame.interpreter_frame_bci(); }
  89   address   mdp() const                          { return _last_frame.interpreter_frame_mdp(); }
  90 
  91   void      set_bcp(address bcp)                 { _last_frame.interpreter_frame_set_bcp(bcp); }
  92   void      set_mdp(address dp)                  { _last_frame.interpreter_frame_set_mdp(dp); }
  93 
  94   // pass method to avoid calling unsafe bcp_to_method (partial fix 4926272)
  95   Bytecodes::Code code() const                   { return Bytecodes::code_at(method(), bcp()); }
  96 
  97   Bytecode  bytecode() const                     { return Bytecode(method(), bcp()); }
  98   int get_index_u1(Bytecodes::Code bc) const     { return bytecode().get_index_u1(bc); }
  99   int get_index_u2(Bytecodes::Code bc) const     { return bytecode().get_index_u2(bc); }
 100   int get_index_u2_cpcache(Bytecodes::Code bc) const
 101                                                  { return bytecode().get_index_u2_cpcache(bc); }
 102   int get_index_u4(Bytecodes::Code bc) const     { return bytecode().get_index_u4(bc); }
 103   int number_of_dimensions() const               { return bcp()[3]; }
 104   ConstantPoolCacheEntry* cache_entry_at(int i) const
 105                                                  { return method()->constants()->cache()->entry_at(i); }
 106   ConstantPoolCacheEntry* cache_entry() const    { return cache_entry_at(Bytes::get_native_u2(bcp() + 1)); }
 107 
 108   oop callee_receiver(Symbol* signature) {
 109     return _last_frame.interpreter_callee_receiver(signature);
 110   }
 111   BasicObjectLock* monitor_begin() const {
 112     return _last_frame.interpreter_frame_monitor_begin();
 113   }
 114   BasicObjectLock* monitor_end() const {
 115     return _last_frame.interpreter_frame_monitor_end();
 116   }
 117   BasicObjectLock* next_monitor(BasicObjectLock* current) const {
 118     return _last_frame.next_monitor_in_interpreter_frame(current);
 119   }
 120 
 121   frame& get_frame()                             { return _last_frame; }
 122 };

 127 void InterpreterRuntime::set_bcp_and_mdp(address bcp, JavaThread *thread) {
 128   LastFrameAccessor last_frame(thread);
 129   last_frame.set_bcp(bcp);
 130   if (ProfileInterpreter) {
 131     // ProfileTraps uses MDOs independently of ProfileInterpreter.
 132     // That is why we must check both ProfileInterpreter and mdo != NULL.
 133     MethodData* mdo = last_frame.method()->method_data();
 134     if (mdo != NULL) {
 135       NEEDS_CLEANUP;
 136       last_frame.set_mdp(mdo->bci_to_dp(last_frame.bci()));
 137     }
 138   }
 139 }

 145 JRT_ENTRY(void, InterpreterRuntime::ldc(JavaThread* thread, bool wide))
 146   // access constant pool
 147   LastFrameAccessor last_frame(thread);
 148   ConstantPool* pool = last_frame.method()->constants();
 149   int index = wide ? last_frame.get_index_u2(Bytecodes::_ldc_w) : last_frame.get_index_u1(Bytecodes::_ldc);
 150   constantTag tag = pool->tag_at(index);
 151 
 152   assert (tag.is_unresolved_klass() || tag.is_klass(), "wrong ldc call");
 153   Klass* klass = pool->klass_at(index, CHECK);
 154     oop java_class = klass->java_mirror();
 155     thread->set_vm_result(java_class);
 156 JRT_END

 158 JRT_ENTRY(void, InterpreterRuntime::resolve_ldc(JavaThread* thread, Bytecodes::Code bytecode)) {
 159   assert(bytecode == Bytecodes::_ldc ||
 160          bytecode == Bytecodes::_ldc_w ||
 161          bytecode == Bytecodes::_ldc2_w ||
 162          bytecode == Bytecodes::_fast_aldc ||
 163          bytecode == Bytecodes::_fast_aldc_w, "wrong bc");
 164   ResourceMark rm(thread);
 165   const bool is_fast_aldc = (bytecode == Bytecodes::_fast_aldc ||
 166                              bytecode == Bytecodes::_fast_aldc_w);
 167   LastFrameAccessor last_frame(thread);
 168   methodHandle m (thread, last_frame.method());
 169   Bytecode_loadconstant ldc(m, last_frame.bci());
 170 
 171   // Double-check the size.  (Condy can have any type.)
 172   BasicType type = ldc.result_type();
 173   switch (type2size[type]) {
 174   case 2: guarantee(bytecode == Bytecodes::_ldc2_w, ""); break;
 175   case 1: guarantee(bytecode != Bytecodes::_ldc2_w, ""); break;
 176   default: ShouldNotReachHere();
 177   }
 178 
 179   // Resolve the constant.  This does not do unboxing.
 180   // But it does replace Universe::the_null_sentinel by null.
 181   oop result = ldc.resolve_constant(CHECK);
 182   assert(result != NULL || is_fast_aldc, "null result only valid for fast_aldc");
 183 
 184 #ifdef ASSERT
 185   {
 186     // The bytecode wrappers aren't GC-safe so construct a new one
 187     Bytecode_loadconstant ldc2(m, last_frame.bci());
 188     int rindex = ldc2.cache_index();
 189     if (rindex < 0)
 190       rindex = m->constants()->cp_to_object_index(ldc2.pool_index());
 191     if (rindex >= 0) {
 192       oop coop = m->constants()->resolved_references()->obj_at(rindex);
 193       oop roop = (result == NULL ? Universe::the_null_sentinel() : result);
 194       assert(roop == coop, "expected result for assembly code");
 195     }
 196   }
 197 #endif
 198   thread->set_vm_result(result);
 199   if (!is_fast_aldc) {
 200     // Tell the interpreter how to unbox the primitive.
 201     guarantee(java_lang_boxing_object::is_instance(result, type), "");
 202     int offset = java_lang_boxing_object::value_offset(type);
 203     intptr_t flags = ((as_TosState(type) << ConstantPoolCacheEntry::tos_state_shift)
 204                       | (offset & ConstantPoolCacheEntry::field_index_mask));
 205     thread->set_vm_result_2((Metadata*)flags);
 206   }
 207 }
 208 JRT_END

 214 JRT_ENTRY(void, InterpreterRuntime::_new(JavaThread* thread, ConstantPool* pool, int index))
 215   Klass* k = pool->klass_at(index, CHECK);
 216   InstanceKlass* klass = InstanceKlass::cast(k);
 217 
 218   // Make sure we are not instantiating an abstract klass
 219   klass->check_valid_for_instantiation(true, CHECK);
 220 
 221   // Make sure klass is initialized
 222   klass->initialize(CHECK);
 223 
 224   // At this point the class may not be fully initialized
 225   // because of recursive initialization. If it is fully
 226   // initialized & has_finalized is not set, we rewrite
 227   // it into its fast version (Note: no locking is needed
 228   // here since this is an atomic byte write and can be
 229   // done more than once).
 230   //
 231   // Note: In case of classes with has_finalized we don't
 232   //       rewrite since that saves us an extra check in
 233   //       the fast version which then would call the
 234   //       slow version anyway (and do a call back into
 235   //       Java).
 236   //       If we have a breakpoint, then we don't rewrite
 237   //       because the _breakpoint bytecode would be lost.
 238   oop obj = klass->allocate_instance(CHECK);
 239   thread->set_vm_result(obj);
 240 JRT_END

 256 JRT_ENTRY(void, InterpreterRuntime::multianewarray(JavaThread* thread, jint* first_size_address))
 257   // We may want to pass in more arguments - could make this slightly faster
 258   LastFrameAccessor last_frame(thread);
 259   ConstantPool* constants = last_frame.method()->constants();
 260   int          i = last_frame.get_index_u2(Bytecodes::_multianewarray);
 261   Klass* klass   = constants->klass_at(i, CHECK);
 262   int   nof_dims = last_frame.number_of_dimensions();
 263   assert(klass->is_klass(), "not a class");
 264   assert(nof_dims >= 1, "multianewarray rank must be nonzero");
 265 
 266   // We must create an array of jints to pass to multi_allocate.
 267   ResourceMark rm(thread);
 268   const int small_dims = 10;
 269   jint dim_array[small_dims];
 270   jint *dims = &dim_array[0];
 271   if (nof_dims > small_dims) {
 272     dims = (jint*) NEW_RESOURCE_ARRAY(jint, nof_dims);
 273   }
 274   for (int index = 0; index < nof_dims; index++) {
 275     // offset from first_size_address is addressed as local[index]
 276     int n = Interpreter::local_offset_in_bytes(index)/jintSize;
 277     dims[index] = first_size_address[n];
 278   }
 279   oop obj = ArrayKlass::cast(klass)->multi_allocate(nof_dims, dims, CHECK);
 280   thread->set_vm_result(obj);
 281 JRT_END

 309 void InterpreterRuntime::note_trap_inner(JavaThread* thread, int reason,
 310                                          const methodHandle& trap_method, int trap_bci, TRAPS) {
 311   if (trap_method.not_null()) {
 312     MethodData* trap_mdo = trap_method->method_data();
 313     if (trap_mdo == NULL) {
 314       Method::build_interpreter_method_data(trap_method, THREAD);
 315       if (HAS_PENDING_EXCEPTION) {
 316         assert((PENDING_EXCEPTION->is_a(SystemDictionary::OutOfMemoryError_klass())),
 317                "we expect only an OOM error here");
 318         CLEAR_PENDING_EXCEPTION;
 319       }
 320       trap_mdo = trap_method->method_data();
 321       // and fall through...
 322     }
 323     if (trap_mdo != NULL) {
 324       // Update per-method count of trap events.  The interpreter
 325       // is updating the MDO to simulate the effect of compiler traps.
 326       Deoptimization::update_method_data_from_interpreter(trap_mdo, trap_bci, reason);
 327     }
 328   }
 329 }

 331 // Assume the compiler is (or will be) interested in this event.
 332 // If necessary, create an MDO to hold the information, and record it.
 333 void InterpreterRuntime::note_trap(JavaThread* thread, int reason, TRAPS) {
 334   assert(ProfileTraps, "call me only if profiling");
 335   LastFrameAccessor last_frame(thread);
 336   methodHandle trap_method(thread, last_frame.method());
 337   int trap_bci = trap_method->bci_from(last_frame.bcp());
 338   note_trap_inner(thread, reason, trap_method, trap_bci, THREAD);
 339 }

 341 static Handle get_preinitialized_exception(Klass* k, TRAPS) {
 342   // get klass
 343   InstanceKlass* klass = InstanceKlass::cast(k);
 344   assert(klass->is_initialized(),
 345          "this klass should have been initialized during VM initialization");
 346   // create instance - do not call constructor since we may have no
 347   // (java) stack space left (should assert constructor is empty)
 348   Handle exception;
 349   oop exception_oop = klass->allocate_instance(CHECK_(exception));
 350   exception = Handle(THREAD, exception_oop);
 351   if (StackTraceInThrowable) {
 352     java_lang_Throwable::fill_in_stack_trace(exception);
 353   }
 354   return exception;
 355 }

 381 JRT_ENTRY(void, InterpreterRuntime::create_exception(JavaThread* thread, char* name, char* message))
 382   // lookup exception klass
 383   TempNewSymbol s = SymbolTable::new_symbol(name);
 384   if (ProfileTraps) {
 385     if (s == vmSymbols::java_lang_ArithmeticException()) {
 386       note_trap(thread, Deoptimization::Reason_div0_check, CHECK);
 387     } else if (s == vmSymbols::java_lang_NullPointerException()) {
 388       note_trap(thread, Deoptimization::Reason_null_check, CHECK);
 389     }
 390   }
 391   // create exception
 392   Handle exception = Exceptions::new_exception(thread, s, message);
 393   thread->set_vm_result(exception());
 394 JRT_END

 440 // exception_handler_for_exception(...) returns the continuation address,
 441 // the exception oop (via TLS) and sets the bci/bcp for the continuation.
 442 // The exception oop is returned to make sure it is preserved over GC (it
 443 // is only on the stack if the exception was thrown explicitly via athrow).
 444 // During this operation, the expression stack contains the values for the
 445 // bci where the exception happened. If the exception was propagated back
 446 // from a call, the expression stack contains the values for the bci at the
 447 // invoke w/o arguments (i.e., as if one were inside the call).
 448 JRT_ENTRY(address, InterpreterRuntime::exception_handler_for_exception(JavaThread* thread, oopDesc* exception))
 449 
 450   LastFrameAccessor last_frame(thread);
 451   Handle             h_exception(thread, exception);
 452   methodHandle       h_method   (thread, last_frame.method());
 453   constantPoolHandle h_constants(thread, h_method->constants());
 454   bool               should_repeat;
 455   int                handler_bci;
 456   int                current_bci = last_frame.bci();
 457 
 458   if (thread->frames_to_pop_failed_realloc() > 0) {
 459     // Allocation of scalar replaced object used in this frame
 460     // failed. Unconditionally pop the frame.
 461     thread->dec_frames_to_pop_failed_realloc();
 462     thread->set_vm_result(h_exception());
 463     // If the method is synchronized we already unlocked the monitor
 464     // during deoptimization so the interpreter needs to skip it when
 465     // the frame is popped.
 466     thread->set_do_not_unlock_if_synchronized(true);
 467     return Interpreter::remove_activation_entry();
 468   }
 469 
 470   // Need to do this check first since when _do_not_unlock_if_synchronized
 471   // is set, we don't want to trigger any classloading which may make calls
 472   // into java, or surprisingly find a matching exception handler for bci 0
 473   // since at this moment the method hasn't been "officially" entered yet.
 474   if (thread->do_not_unlock_if_synchronized()) {
 475     ResourceMark rm;
 476     assert(current_bci == 0,  "bci isn't zero for do_not_unlock_if_synchronized");
 477     thread->set_vm_result(exception);
 478     return Interpreter::remove_activation_entry();
 479   }
 480 
 481   do {
 482     should_repeat = false;
 483 
 484     // assertions
 485 #ifdef ASSERT
 486     assert(h_exception.not_null(), "NULL exceptions should be handled by athrow");
 487     // Check that exception is a subclass of Throwable, otherwise we have a VerifyError
 488     if (!(h_exception->is_a(SystemDictionary::Throwable_klass()))) {
 489       if (ExitVMOnVerifyError) vm_exit(-1);
 490       ShouldNotReachHere();
 491     }
 492 #endif
 493 
 494     // tracing
 495     if (log_is_enabled(Info, exceptions)) {
 496       ResourceMark rm(thread);
 497       stringStream tempst;
 498       tempst.print("interpreter method <%s>\n"
 499                    " at bci %d for thread " INTPTR_FORMAT " (%s)",
 500                    h_method->print_value_string(), current_bci, p2i(thread), thread->name());
 501       Exceptions::log_exception(h_exception, tempst.as_string());
 502     }
 503 // Don't go paging in something which won't be used.
 504 //     else if (extable->length() == 0) {
 505 //       // disabled for now - interpreter is not using shortcut yet
 506 //       // (shortcut is not to call runtime if we have no exception handlers)
 507 //       // warning("performance bug: should not call runtime if method has no exception handlers");
 508 //     }
 509     // for AbortVMOnException flag
 510     Exceptions::debug_check_abort(h_exception);
 511 
 512     // exception handler lookup
 513     Klass* klass = h_exception->klass();
 514     handler_bci = Method::fast_exception_handler_bci_for(h_method, klass, current_bci, THREAD);
 515     if (HAS_PENDING_EXCEPTION) {
 516       // We threw an exception while trying to find the exception handler.
 517       // Transfer the new exception to the exception handle which will
 518       // be set into thread local storage, and do another lookup for an
 519       // exception handler for this exception, this time starting at the
 520       // BCI of the exception handler which caused the exception to be
 521       // thrown (bug 4307310).
 522       h_exception = Handle(THREAD, PENDING_EXCEPTION);
 523       CLEAR_PENDING_EXCEPTION;
 524       if (handler_bci >= 0) {
 525         current_bci = handler_bci;
 526         should_repeat = true;
 527       }
 528     }
 529   } while (should_repeat == true);
 530 
 531 #if INCLUDE_JVMCI
 532   if (EnableJVMCI && h_method->method_data() != NULL) {
 533     ResourceMark rm(thread);
 534     ProfileData* pdata = h_method->method_data()->allocate_bci_to_data(current_bci, NULL);
 535     if (pdata != NULL && pdata->is_BitData()) {
 536       BitData* bit_data = (BitData*) pdata;
 537       bit_data->set_exception_seen();
 538     }
 539   }
 540 #endif
 541 
 542   // notify JVMTI of an exception throw; JVMTI will detect if this is a first
 543   // time throw or a stack unwinding throw and accordingly notify the debugger
 544   if (JvmtiExport::can_post_on_exceptions()) {
 545     JvmtiExport::post_exception_throw(thread, h_method(), last_frame.bcp(), h_exception());
 546   }
 547 
 548   address continuation = NULL;
 549   address handler_pc = NULL;
 550   if (handler_bci < 0 || !thread->reguard_stack((address) &continuation)) {
 551     // Forward exception to callee (leaving bci/bcp untouched) because (a) no
 552     // handler in this method, or (b) after a stack overflow there is not yet
 553     // enough stack space available to reprotect the stack.
 554     continuation = Interpreter::remove_activation_entry();
 555 #if COMPILER2_OR_JVMCI
 556     // Count this for compilation purposes
 557     h_method->interpreter_throwout_increment(THREAD);
 558 #endif
 559   } else {
 560     // handler in this method => change bci/bcp to handler bci/bcp and continue there
 561     handler_pc = h_method->code_base() + handler_bci;
 562 #ifndef ZERO
 563     set_bcp_and_mdp(handler_pc, thread);
 564     continuation = Interpreter::dispatch_table(vtos)[*handler_pc];
 565 #else
 566     continuation = (address)(intptr_t) handler_bci;
 567 #endif
 568   }
 569 
 570   // notify debugger of an exception catch
 571   // (this is good for exceptions caught in native methods as well)
 572   if (JvmtiExport::can_post_on_exceptions()) {
 573     JvmtiExport::notice_unwind_due_to_exception(thread, h_method(), handler_pc, h_exception(), (handler_pc != NULL));
 574   }
 575 
 576   thread->set_vm_result(h_exception());
 577   return continuation;
 578 JRT_END

 635 void InterpreterRuntime::resolve_get_put(JavaThread* thread, Bytecodes::Code bytecode) {
 636   Thread* THREAD = thread;
 637   // resolve field
 638   fieldDescriptor info;
 639   LastFrameAccessor last_frame(thread);
 640   constantPoolHandle pool(thread, last_frame.method()->constants());
 641   methodHandle m(thread, last_frame.method());
 642   bool is_put    = (bytecode == Bytecodes::_putfield  || bytecode == Bytecodes::_nofast_putfield ||
 643                     bytecode == Bytecodes::_putstatic);
 644   bool is_static = (bytecode == Bytecodes::_getstatic || bytecode == Bytecodes::_putstatic);
 645 
 646   {
 647     JvmtiHideSingleStepping jhss(thread);
 648     LinkResolver::resolve_field_access(info, pool, last_frame.get_index_u2_cpcache(bytecode),
 649                                        m, bytecode, CHECK);
 650   } // end JvmtiHideSingleStepping
 651 
 652   // check if link resolution caused cpCache to be updated
 653   ConstantPoolCacheEntry* cp_cache_entry = last_frame.cache_entry();
 654   if (cp_cache_entry->is_resolved(bytecode)) return;
 655 
 656   // compute auxiliary field attributes
 657   TosState state  = as_TosState(info.field_type());
 658 
 659   // Resolution of put instructions on final fields is delayed. That is required so that
 660   // exceptions are thrown at the correct place (when the instruction is actually invoked).
 661   // If we do not resolve an instruction in the current pass, leaving the put_code
 662   // set to zero will cause the next put instruction to the same field to reresolve.
 663 
 664   // Resolution of put instructions to final instance fields with invalid updates (i.e.,
 665   // to final instance fields with updates originating from a method different than <init>)
 666   // is inhibited. A putfield instruction targeting an instance final field must throw
 667   // an IllegalAccessError if the instruction is not in an instance
 668   // initializer method <init>. If resolution were not inhibited, a putfield
 669   // in an initializer method could be resolved in the initializer. Subsequent
 670   // putfield instructions to the same field would then use cached information.
 671   // As a result, those instructions would not pass through the VM. That is,
 672   // checks in resolve_field_access() would not be executed for those instructions
 673   // and the required IllegalAccessError would not be thrown.
 674   //
 675   // Also, we need to delay resolving getstatic and putstatic instructions until the
 676   // class is initialized.  This is required so that access to the static
 677   // field will call the initialization function every time until the class
 678   // is completely initialized ala. in 2.17.5 in JVM Specification.
 679   InstanceKlass* klass = info.field_holder();
 680   bool uninitialized_static = is_static && !klass->is_initialized();
 681   bool has_initialized_final_update = info.field_holder()->major_version() >= 53 &&
 682                                       info.has_initialized_final_update();
 683   assert(!(has_initialized_final_update && !info.access_flags().is_final()), "Fields with initialized final updates must be final");
 684 
 685   Bytecodes::Code get_code = (Bytecodes::Code)0;
 686   Bytecodes::Code put_code = (Bytecodes::Code)0;
 687   if (!uninitialized_static) {
 688     get_code = ((is_static) ? Bytecodes::_getstatic : Bytecodes::_getfield);
 689     if ((is_put && !has_initialized_final_update) || !info.access_flags().is_final()) {
 690       put_code = ((is_static) ? Bytecodes::_putstatic : Bytecodes::_putfield);
 691     }
 692   }
 693 
 694   cp_cache_entry->set_field(
 695     get_code,
 696     put_code,
 697     info.field_holder(),
 698     info.index(),
 699     info.offset(),
 700     state,
 701     info.access_flags().is_final(),
 702     info.access_flags().is_volatile(),
 703     pool->pool_holder()
 704   );
 705 }

 795 void InterpreterRuntime::resolve_invoke(JavaThread* thread, Bytecodes::Code bytecode) {
 796   Thread* THREAD = thread;
 797   LastFrameAccessor last_frame(thread);
 798   // extract receiver from the outgoing argument list if necessary
 799   Handle receiver(thread, NULL);
 800   if (bytecode == Bytecodes::_invokevirtual || bytecode == Bytecodes::_invokeinterface ||
 801       bytecode == Bytecodes::_invokespecial) {
 802     ResourceMark rm(thread);
 803     methodHandle m (thread, last_frame.method());
 804     Bytecode_invoke call(m, last_frame.bci());
 805     Symbol* signature = call.signature();
 806     receiver = Handle(thread, last_frame.callee_receiver(signature));
 807 
 808     assert(Universe::heap()->is_in_or_null(receiver()),
 809            "sanity check");
 810     assert(receiver.is_null() ||
 811            !Universe::heap()->is_in(receiver->klass()),
 812            "sanity check");
 813   }
 814 
 815   // resolve method
 816   CallInfo info;
 817   constantPoolHandle pool(thread, last_frame.method()->constants());
 818 
 819   {
 820     JvmtiHideSingleStepping jhss(thread);
 821     LinkResolver::resolve_invoke(info, receiver, pool,
 822                                  last_frame.get_index_u2_cpcache(bytecode), bytecode,
 823                                  CHECK);
 824     if (JvmtiExport::can_hotswap_or_post_breakpoint()) {
 825       int retry_count = 0;
 826       while (info.resolved_method()->is_old()) {
 827         // It is very unlikely that method is redefined more than 100 times
 828         // in the middle of resolve. If it is looping here more than 100 times
 829         // means then there could be a bug here.
 830         guarantee((retry_count++ < 100),
 831                   "Could not resolve to latest version of redefined method");
 832         // method is redefined in the middle of resolve so re-try.
 833         LinkResolver::resolve_invoke(info, receiver, pool,
 834                                      last_frame.get_index_u2_cpcache(bytecode), bytecode,
 835                                      CHECK);
 836       }
 837     }
 838   } // end JvmtiHideSingleStepping
 839 
 840   // check if link resolution caused cpCache to be updated
 841   ConstantPoolCacheEntry* cp_cache_entry = last_frame.cache_entry();
 842   if (cp_cache_entry->is_resolved(bytecode)) return;
 843 
 844 #ifdef ASSERT
 845   if (bytecode == Bytecodes::_invokeinterface) {
 846     if (info.resolved_method()->method_holder() ==
 847                                             SystemDictionary::Object_klass()) {
 848       // NOTE: THIS IS A FIX FOR A CORNER CASE in the JVM spec
 849       // (see also CallInfo::set_interface for details)
 850       assert(info.call_kind() == CallInfo::vtable_call ||
 851              info.call_kind() == CallInfo::direct_call, "");
 852       Method* rm = info.resolved_method();
 853       assert(rm->is_final() || info.has_vtable_index(),
 854              "should have been set already");
 855     } else if (!info.resolved_method()->has_itable_index()) {
 856       // Resolved something like CharSequence.toString.  Use vtable not itable.
 857       assert(info.call_kind() != CallInfo::itable_call, "");
 858     } else {
 859       // Setup itable entry
 860       assert(info.call_kind() == CallInfo::itable_call, "");
 861       int index = info.resolved_method()->itable_index();
 862       assert(info.itable_index() == index, "");
 863     }
 864   } else if (bytecode == Bytecodes::_invokespecial) {
 865     assert(info.call_kind() == CallInfo::direct_call, "must be direct call");
 866   } else {
 867     assert(info.call_kind() == CallInfo::direct_call ||
 868            info.call_kind() == CallInfo::vtable_call, "");
 869   }
 870 #endif
 871   // Get sender or sender's unsafe_anonymous_host, and only set cpCache entry to resolved if
 872   // it is not an interface.  The receiver for invokespecial calls within interface
 873   // methods must be checked for every call.
 874   InstanceKlass* sender = pool->pool_holder();
 875   sender = sender->is_unsafe_anonymous() ? sender->unsafe_anonymous_host() : sender;
 876   methodHandle resolved_method(THREAD, info.resolved_method());
 877 
 878   switch (info.call_kind()) {
 879   case CallInfo::direct_call:
 880     cp_cache_entry->set_direct_call(
 881       bytecode,
 882       resolved_method,
 883       sender->is_interface());
 884     break;
 885   case CallInfo::vtable_call:
 886     cp_cache_entry->set_vtable_call(
 887       bytecode,
 888       resolved_method,
 889       info.vtable_index());
 890     break;
 891   case CallInfo::itable_call:
 892     cp_cache_entry->set_itable_call(
 893       bytecode,
 894       info.resolved_klass(),
 895       resolved_method,
 896       info.itable_index());
 897     break;
 898   default:  ShouldNotReachHere();
 899   }
 900 }

 977 nmethod* InterpreterRuntime::frequency_counter_overflow(JavaThread* thread, address branch_bcp) {
 978   nmethod* nm = frequency_counter_overflow_inner(thread, branch_bcp);
 979   assert(branch_bcp != NULL || nm == NULL, "always returns null for non OSR requests");
 980   if (branch_bcp != NULL && nm != NULL) {
 981     // This was a successful request for an OSR nmethod.  Because
 982     // frequency_counter_overflow_inner ends with a safepoint check,
 983     // nm could have been unloaded so look it up again.  It's unsafe
 984     // to examine nm directly since it might have been freed and used
 985     // for something else.
 986     LastFrameAccessor last_frame(thread);
 987     Method* method =  last_frame.method();
 988     int bci = method->bci_from(last_frame.bcp());
 989     nm = method->lookup_osr_nmethod_for(bci, CompLevel_none, false);
 990     BarrierSetNMethod* bs_nm = BarrierSet::barrier_set()->barrier_set_nmethod();
 991     if (nm != NULL && bs_nm != NULL) {
 992       // in case the transition passed a safepoint we need to barrier this again
 993       if (!bs_nm->nmethod_osr_entry_barrier(nm)) {
 994         nm = NULL;
 995       }
 996     }
 997   }
 998   if (nm != NULL && thread->is_interp_only_mode()) {
 999     // Normally we never get an nm if is_interp_only_mode() is true, because
1000     // policy()->event has a check for this and won't compile the method when
1001     // true. However, it's possible for is_interp_only_mode() to become true
1002     // during the compilation. We don't want to return the nm in that case
1003     // because we want to continue to execute interpreted.
1004     nm = NULL;
1005   }
1006 #ifndef PRODUCT
1007   if (TraceOnStackReplacement) {
1008     if (nm != NULL) {
1009       tty->print("OSR entry @ pc: " INTPTR_FORMAT ": ", p2i(nm->osr_entry()));
1010       nm->print();
1011     }
1012   }
1013 #endif
1014   return nm;
1015 }

1018           InterpreterRuntime::frequency_counter_overflow_inner(JavaThread* thread, address branch_bcp))
1019   if (HAS_PENDING_EXCEPTION) {
1020     return NULL;
1021   }
1022   // use UnlockFlagSaver to clear and restore the _do_not_unlock_if_synchronized
1023   // flag, in case this method triggers classloading which will call into Java.
1024   UnlockFlagSaver fs(thread);

1027   assert(last_frame.is_interpreted_frame(), "must come from interpreter");
1028   methodHandle method(thread, last_frame.method());
1029   const int branch_bci = branch_bcp != NULL ? method->bci_from(branch_bcp) : InvocationEntryBci;
1030   const int bci = branch_bcp != NULL ? method->bci_from(last_frame.bcp()) : InvocationEntryBci;
1031 
1032   nmethod* osr_nm = CompilationPolicy::policy()->event(method, method, branch_bci, bci, CompLevel_none, NULL, thread);

1038       osr_nm = NULL;
1039     }

1044     // monitors in the activation are biased because their
1045     // BasicObjectLocks will need to migrate during OSR. Force
1046     // unbiasing of all monitors in the activation now (even though
1047     // the OSR nmethod might be invalidated) because we don't have a
1048     // safepoint opportunity later once the migration begins.

1050       ResourceMark rm;
1051       GrowableArray<Handle>* objects_to_revoke = new GrowableArray<Handle>();
1052       for( BasicObjectLock *kptr = last_frame.monitor_end();
1053            kptr < last_frame.monitor_begin();
1054            kptr = last_frame.next_monitor(kptr) ) {
1055         if( kptr->obj() != NULL ) {
1056           objects_to_revoke->append(Handle(THREAD, kptr->obj()));
1057         }
1058       }
1059       BiasedLocking::revoke(objects_to_revoke, thread);
1060     }
1061   }
1062   return osr_nm;

1193 JRT_ENTRY(void, InterpreterRuntime::post_field_modification(JavaThread *thread,
1194   oopDesc* obj, ConstantPoolCacheEntry *cp_entry, jvalue *value))
1195 
1196   Klass* k = cp_entry->f1_as_klass();
1197 
1198   // check the access_flags for the field in the klass
1199   InstanceKlass* ik = InstanceKlass::cast(k);
1200   int index = cp_entry->field_index();
1201   // bail out if field modifications are not watched
1202   if ((ik->field_access_flags(index) & JVM_ACC_FIELD_MODIFICATION_WATCHED) == 0) return;
1203 
1204   char sig_type = '\0';
1205 
1206   switch(cp_entry->flag_state()) {
1207     case btos: sig_type = JVM_SIGNATURE_BYTE;    break;
1208     case ztos: sig_type = JVM_SIGNATURE_BOOLEAN; break;
1209     case ctos: sig_type = JVM_SIGNATURE_CHAR;    break;
1210     case stos: sig_type = JVM_SIGNATURE_SHORT;   break;
1211     case itos: sig_type = JVM_SIGNATURE_INT;     break;
1212     case ftos: sig_type = JVM_SIGNATURE_FLOAT;   break;
1213     case atos: sig_type = JVM_SIGNATURE_CLASS;   break;
1214     case ltos: sig_type = JVM_SIGNATURE_LONG;    break;
1215     case dtos: sig_type = JVM_SIGNATURE_DOUBLE;  break;
1216     default:  ShouldNotReachHere(); return;
1217   }
1218   bool is_static = (obj == NULL);
1219 
1220   HandleMark hm(thread);
1221   jfieldID fid = jfieldIDWorkaround::to_jfieldID(ik, cp_entry->f2_as_index(), is_static);
1222   jvalue fvalue;
1223 #ifdef _LP64
1224   fvalue = *value;
1225 #else
1226   // Long/double values are stored unaligned and also noncontiguously with
1227   // tagged stacks.  We can't just do a simple assignment even in the non-
1228   // J/D cases because a C++ compiler is allowed to assume that a jvalue is
1229   // 8-byte aligned, and interpreter stack slots are only 4-byte aligned.
1230   // We assume that the two halves of longs/doubles are stored in interpreter
1231   // stack slots in platform-endian order.
1232   jlong_accessor u;
1233   jint* newval = (jint*)value;
1234   u.words[0] = newval[0];
1235   u.words[1] = newval[Interpreter::stackElementWords]; // skip if tag
1236   fvalue.j = u.long_value;
1237 #endif // _LP64
1238 
1239   Handle h_obj;
1240   if (!is_static) {
1241     // non-static field accessors have an object, but we need a handle
1242     h_obj = Handle(thread, obj);
1243   }
1244 
1245   LastFrameAccessor last_frame(thread);
1246   JvmtiExport::post_raw_field_modification(thread, last_frame.method(), last_frame.bcp(), ik, h_obj,
1247                                            fid, sig_type, &fvalue);

1321 void SignatureHandlerLibrary::add(const methodHandle& method) {
1322   if (method->signature_handler() == NULL) {
1323     // use slow signature handler if we can't do better
1324     int handler_index = -1;
1325     // check if we can use customized (fast) signature handler
1326     if (UseFastSignatureHandlers && method->size_of_parameters() <= Fingerprinter::fp_max_size_of_parameters) {
1327       // use customized signature handler
1328       MutexLocker mu(SignatureHandlerLibrary_lock);
1329       // make sure data structure is initialized
1330       initialize();
1331       // lookup method signature's fingerprint
1332       uint64_t fingerprint = Fingerprinter(method).fingerprint();
1333       // allow CPU dependant code to optimize the fingerprints for the fast handler
1334       fingerprint = InterpreterRuntime::normalize_fast_native_fingerprint(fingerprint);
1335       handler_index = _fingerprints->find(fingerprint);
1336       // create handler if necessary
1337       if (handler_index < 0) {
1338         ResourceMark rm;
1339         ptrdiff_t align_offset = align_up(_buffer, CodeEntryAlignment) - (address)_buffer;
1340         CodeBuffer buffer((address)(_buffer + align_offset),
1341                           SignatureHandlerLibrary::buffer_size - align_offset);
1342         InterpreterRuntime::SignatureHandlerGenerator(method, &buffer).generate(fingerprint);
1343         // copy into code heap
1344         address handler = set_handler(&buffer);
1345         if (handler == NULL) {
1346           // use slow signature handler (without memorizing it in the fingerprints)
1347         } else {
1348           // debugging suppport
1349           if (PrintSignatureHandlers && (handler != Interpreter::slow_signature_handler())) {
1350             ttyLocker ttyl;
1351             tty->cr();
1352             tty->print_cr("argument handler #%d for: %s %s (fingerprint = " UINT64_FORMAT ", %d bytes generated)",
1353                           _handlers->length(),
1354                           (method->is_static() ? "static" : "receiver"),
1355                           method->name_and_sig_as_C_string(),
1356                           fingerprint,
1357                           buffer.insts_size());
1358             if (buffer.insts_size() > 0) {
1359               Disassembler::decode(handler, handler + buffer.insts_size());
1360             }
1361 #ifndef PRODUCT
1362             address rh_begin = Interpreter::result_handler(method()->result_type());
1363             if (CodeCache::contains(rh_begin)) {
1364               // else it might be special platform dependent values
1365               tty->print_cr(" --- associated result handler ---");
1366               address rh_end = rh_begin;
1367               while (*(int*)rh_end != 0) {
1368                 rh_end += sizeof(int);
1369               }
1370               Disassembler::decode(rh_begin, rh_end);
1371             } else {
1372               tty->print_cr(" associated result handler: " PTR_FORMAT, p2i(rh_begin));
1373             }
1374 #endif
1375           }
1376           // add handler to library
1377           _fingerprints->append(fingerprint);
1378           _handlers->append(handler);
1379           // set handler index
1380           assert(_fingerprints->length() == _handlers->length(), "sanity check");
1381           handler_index = _fingerprints->length() - 1;
1382         }
1383       }
1384       // Set handler under SignatureHandlerLibrary_lock
1385       if (handler_index < 0) {
1386         // use generic signature handler
1387         method->set_signature_handler(Interpreter::slow_signature_handler());
1388       } else {
1389         // set handler
1390         method->set_signature_handler(_handlers->at(handler_index));
1391       }
1392     } else {
1393       DEBUG_ONLY(Thread::current()->check_possible_safepoint());
1394       // use generic signature handler
1395       method->set_signature_handler(Interpreter::slow_signature_handler());
1396     }
1397   }
1398 #ifdef ASSERT
1399   int handler_index = -1;
1400   int fingerprint_index = -2;
1401   {
1402     // '_handlers' and '_fingerprints' are 'GrowableArray's and are NOT synchronized
1403     // in any way if accessed from multiple threads. To avoid races with another
1404     // thread which may change the arrays in the above, mutex protected block, we
1405     // have to protect this read access here with the same mutex as well!
1406     MutexLocker mu(SignatureHandlerLibrary_lock);
1407     if (_handlers != NULL) {
1408       handler_index = _handlers->find(method->signature_handler());
1409       uint64_t fingerprint = Fingerprinter(method).fingerprint();
1410       fingerprint = InterpreterRuntime::normalize_fast_native_fingerprint(fingerprint);
1411       fingerprint_index = _fingerprints->find(fingerprint);
1412     }
1413   }
1414   assert(method->signature_handler() == Interpreter::slow_signature_handler() ||
1415          handler_index == fingerprint_index, "sanity check");
1416 #endif // ASSERT
1417 }
