35 struct hb_aat_feature_mapping_t
36 {
37   hb_tag_t otFeatureTag;
38   hb_aat_layout_feature_type_t aatFeatureType;
39   hb_aat_layout_feature_selector_t selectorToEnable;
40   hb_aat_layout_feature_selector_t selectorToDisable;
41 
42   static int cmp (const void *key_, const void *entry_)
43   {
44     hb_tag_t key = * (unsigned int *) key_;
45     const hb_aat_feature_mapping_t * entry = (const hb_aat_feature_mapping_t *) entry_;
46     return key < entry->otFeatureTag ? -1 :
47            key > entry->otFeatureTag ? 1 :
48            0;
49   }
50 };
