27 #ifndef HB_OT_HHEA_TABLE_HH
28 #define HB_OT_HHEA_TABLE_HH
29 
30 #include "hb-open-type.hh"
31 
32 /*
33  * hhea -- Horizontal Header
34  * https://docs.microsoft.com/en-us/typography/opentype/spec/hhea
35  * vhea -- Vertical Header
36  * https://docs.microsoft.com/en-us/typography/opentype/spec/vhea
37  */
38 #define HB_OT_TAG_hhea HB_TAG('h','h','e','a')
39 #define HB_OT_TAG_vhea HB_TAG('v','h','e','a')
40 
41 
42 namespace OT {
43 
44 
45 template <typename T>
46 struct _hea
47 {
48   bool sanitize (hb_sanitize_context_t *c) const
49   {
50     TRACE_SANITIZE (this);
51     return_trace (c->check_struct (this) && likely (version.major == 1));
52   }
53 
54   public:
55   FixedVersion<>version;                /* 0x00010000u for version 1.0. */
56   FWORD         ascender;               /* Typographic ascent. */
57   FWORD         descender;              /* Typographic descent. */
58   FWORD         lineGap;                /* Typographic line gap. */
59   UFWORD        advanceMax;             /* Maximum advance width/height value in
60                                          * metrics table. */
61   FWORD         minLeadingBearing;      /* Minimum left/top sidebearing value in
62                                          * metrics table. */
63   FWORD         minTrailingBearing;     /* Minimum right/bottom sidebearing value;
64                                          * calculated as Min(aw - lsb -
65                                          * (xMax - xMin)) for horizontal. */
66   FWORD         maxExtent;              /* horizontal: Max(lsb + (xMax - xMin)),
67                                          * vertical: minLeadingBearing+(yMax-yMin). */
68   HBINT16               caretSlopeRise;         /* Used to calculate the slope of the
69                                          * cursor (rise/run); 1 for vertical caret,
70                                          * 0 for horizontal.*/
71   HBINT16               caretSlopeRun;          /* 0 for vertical caret, 1 for horizontal. */
72   HBINT16               caretOffset;            /* The amount by which a slanted
73                                          * highlight on a glyph needs
74                                          * to be shifted to produce the
75                                          * best appearance. Set to 0 for
76                                          * non-slanted fonts. */
77   HBINT16               reserved1;              /* Set to 0. */
78   HBINT16               reserved2;              /* Set to 0. */
79   HBINT16               reserved3;              /* Set to 0. */
80   HBINT16               reserved4;              /* Set to 0. */
81   HBINT16               metricDataFormat;       /* 0 for current format. */
82   HBUINT16      numberOfLongMetrics;    /* Number of LongMetric entries in metric
83                                          * table. */
84   public:
85   DEFINE_SIZE_STATIC (36);
86 };
87 
88 struct hhea : _hea<hhea> {
89   static constexpr hb_tag_t tableTag = HB_OT_TAG_hhea;
90 };
91 struct vhea : _hea<vhea> {
92   static constexpr hb_tag_t tableTag = HB_OT_TAG_vhea;
93 };
94 
95 
96 } /* namespace OT */

99 #endif /* HB_OT_HHEA_TABLE_HH */
