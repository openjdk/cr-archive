27 #ifndef AppLauncher_h
28 #define AppLauncher_h
29 
30 #include "tstrings.h"
31 
32 class Jvm;
33 class LogAppender;
34 
35 class AppLauncher {
36 public:
37     AppLauncher();
38 
39     AppLauncher& setImageRoot(const tstring& v) {
40         imageRoot = v;
41         return *this;
42     }
43 
44     AppLauncher& setDefaultRuntimePath(const tstring& v) {
45         defaultRuntimePath = v;
46         return *this;
47     }
48 
49     AppLauncher& setAppDir(const tstring& v) {
50         appDirPath = v;
51         return *this;
52     }
53 
54     AppLauncher& setInitJvmFromCmdlineOnly(bool v) {
55         initJvmFromCmdlineOnly = v;
56         return *this;
57     }
58 
59     AppLauncher& addJvmLibName(const tstring& v) {
60         jvmLibNames.push_back(v);
61         return *this;
62     }
63 
64     Jvm* createJvmLauncher() const;
65 
66     void launch() const;
67 
68     static LogAppender& defaultLastErrorLogAppender();
69 
70     static bool isWithLogging();
71 
72     typedef void (*LauncherFunc) ();
73 
74     static int launch(const std::nothrow_t&, LauncherFunc func,
75             LogAppender* lastErrorLogAppender = 0);
76 
77     static std::string lastErrorMsg();
78 
79 private:
80     tstring_array args;
81     tstring launcherPath;
82     tstring defaultRuntimePath;
83     tstring appDirPath;
84     tstring imageRoot;
85     tstring_array jvmLibNames;
86     bool initJvmFromCmdlineOnly;
87 };
88 
89 #endif // AppLauncher_h
