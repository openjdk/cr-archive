27 #ifndef AppLauncher_h
28 #define AppLauncher_h
29 
30 #include "tstrings.h"
31 
32 class Jvm;
33 class LogAppender;
34 
35 class AppLauncher {
36 public:
37     AppLauncher();
38 
39     AppLauncher& setImageRoot(const tstring& v) {
40         imageRoot = v;
41         return *this;
42     }
43 
44     AppLauncher& setDefaultRuntimePath(const tstring& v) {
45         defaultRuntimePath = v;
46         return *this;
47     }
48 
49     AppLauncher& setAppDir(const tstring& v) {
50         appDirPath = v;
51         return *this;
52     }
53 
54     AppLauncher& addJvmLibName(const tstring& v) {
55         jvmLibNames.push_back(v);
56         return *this;
57     }
58 
59     Jvm* createJvmLauncher() const;
60 
61     void launch() const;
62 
63     static LogAppender& defaultLastErrorLogAppender();
64 
65     static bool isWithLogging();
66 
67     typedef void (*LauncherFunc) ();
68 
69     static int launch(const std::nothrow_t&, LauncherFunc func,
70             LogAppender* lastErrorLogAppender = 0);
71 
72     static std::string lastErrorMsg();
73 
74 private:
75     tstring_array args;
76     tstring launcherPath;
77     tstring defaultRuntimePath;
78     tstring appDirPath;
79     tstring imageRoot;
80     tstring_array jvmLibNames;
81 };
82 
83 #endif // AppLauncher_h
