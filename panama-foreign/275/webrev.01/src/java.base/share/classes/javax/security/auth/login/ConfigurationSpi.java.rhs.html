27 package javax.security.auth.login;
28 
29 /**
30  * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
31  * for the {@code Configuration} class.
32  * All the abstract methods in this class must be implemented by each
33  * service provider who wishes to supply a Configuration implementation.
34  *
35  * <p> Subclass implementations of this abstract class must provide
36  * a public constructor that takes a {@code Configuration.Parameters}
37  * object as an input parameter.  This constructor also must throw
38  * an IllegalArgumentException if it does not understand the
39  * {@code Configuration.Parameters} input.
40  *
41  *
42  * @since 1.6
43  */
44 
45 public abstract class ConfigurationSpi {
46     /**
47      * Constructor for subclasses to call.
48      */
49     public ConfigurationSpi() {}
50 
51     /**
52      * Retrieve the AppConfigurationEntries for the specified {@code name}.
53      *
54      * @param name the name used to index the Configuration.
55      *
56      * @return an array of AppConfigurationEntries for the specified
57      *          {@code name}, or null if there are no entries.
58      */
59     protected abstract AppConfigurationEntry[] engineGetAppConfigurationEntry
60                                                         (String name);
61 
62     /**
63      * Refresh and reload the Configuration.
64      *
65      * <p> This method causes this Configuration object to refresh/reload its
66      * contents in an implementation-dependent manner.
67      * For example, if this Configuration object stores its entries in a file,
68      * calling {@code refresh} may cause the file to be re-read.
69      *
70      * <p> The default implementation of this method does nothing.
71      * This method should be overridden if a refresh operation is supported
72      * by the implementation.
73      *
74      * @exception SecurityException if the caller does not have permission
75      *          to refresh its Configuration.
76      */
77     protected void engineRefresh() { }
78 }
