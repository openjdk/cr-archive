26 package javax.crypto;
27 
28 import java.security.*;
29 import java.security.spec.*;
30 
31 /**
32  * This class defines the <i>Service Provider Interface</i> (<b>SPI</b>)
33  * for the <code>KeyGenerator</code> class.
34  * All the abstract methods in this class must be implemented by each
35  * cryptographic service provider who wishes to supply the implementation
36  * of a key generator for a particular algorithm.
37  *
38  * <p>In case the client does not explicitly initialize the KeyGenerator
39  * (via a call to an {@code init} method), each provider must
40  * supply (and document) a default initialization.
41  * See the Keysize Restriction sections of the
42  * {@extLink security_guide_jdk_providers JDK Providers}
43  * document for information on the KeyGenerator defaults used by
44  * JDK providers.
45  * However, note that defaults may vary across different providers.
46  * Additionally, the default value for a provider may change in a future
47  * version. Therefore, it is recommended to explicitly initialize the
48  * KeyGenerator instead of relying on provider-specific defaults.
49  *
50  * @author Jan Luehe
51  *
52  * @see SecretKey
53  * @since 1.4
54  */
55 
56 public abstract class KeyGeneratorSpi {
57 
58     /**
59      * Initializes the key generator.
60      *
61      * @param random the source of randomness for this generator
62      */
63     protected abstract void engineInit(SecureRandom random);
64 
65     /**
66      * Initializes the key generator with the specified parameter
67      * set and a user-provided source of randomness.
68      *
69      * @param params the key generation parameters
70      * @param random the source of randomness for this key generator
71      *
72      * @exception InvalidAlgorithmParameterException if <code>params</code> is
73      * inappropriate for this key generator
74      */
75     protected abstract void engineInit(AlgorithmParameterSpec params,
76                                        SecureRandom random)
77         throws InvalidAlgorithmParameterException;
78 
79     /**
80      * Initializes this key generator for a certain keysize, using the given
81      * source of randomness.
82      *
83      * @param keysize the keysize. This is an algorithm-specific metric,
84      * specified in number of bits.
85      * @param random the source of randomness for this key generator
86      *
87      * @exception InvalidParameterException if the keysize is wrong or not
88      * supported.
89      */
90     protected abstract void engineInit(int keysize, SecureRandom random);
91 
92     /**
93      * Generates a secret key.
94      *
95      * @return the new key
96      */
97     protected abstract SecretKey engineGenerateKey();
98 }
