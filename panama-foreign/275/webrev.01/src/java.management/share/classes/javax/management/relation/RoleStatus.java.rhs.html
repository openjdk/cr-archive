26 package javax.management.relation;
27 
28 /**
29  * This class describes the various problems which can be encountered when
30  * accessing a role.
31  *
32  * @since 1.5
33  */
34 public class RoleStatus {
35 
36     /**
37      * Do not call.
38      */
39     @Deprecated(since="16", forRemoval=true)
40     public RoleStatus() {}
41 
42     //
43     // Possible problems
44     //
45 
46     /**
47      * Problem type when trying to access an unknown role.
48      */
49     public static final int NO_ROLE_WITH_NAME = 1;
50     /**
51      * Problem type when trying to read a non-readable attribute.
52      */
53     public static final int ROLE_NOT_READABLE = 2;
54     /**
55      * Problem type when trying to update a non-writable attribute.
56      */
57     public static final int ROLE_NOT_WRITABLE = 3;
58     /**
59      * Problem type when trying to set a role value with less ObjectNames than
60      * the minimum expected cardinality.
61      */
62     public static final int LESS_THAN_MIN_ROLE_DEGREE = 4;
63     /**
64      * Problem type when trying to set a role value with more ObjectNames than
65      * the maximum expected cardinality.
66      */
67     public static final int MORE_THAN_MAX_ROLE_DEGREE = 5;
68     /**
69      * Problem type when trying to set a role value including the ObjectName of
70      * a MBean not of the class expected for that role.
71      */
72     public static final int REF_MBEAN_OF_INCORRECT_CLASS = 6;
73     /**
74      * Problem type when trying to set a role value including the ObjectName of
75      * a MBean not registered in the MBean Server.
76      */
77     public static final int REF_MBEAN_NOT_REGISTERED = 7;
78 
79     /**
80      * Returns true if given value corresponds to a known role status, false
81      * otherwise.
82      *
83      * @param status a status code.
84      *
85      * @return true if this value is a known role status.
86      */
87     public static boolean isRoleStatus(int status) {
88         if (status != NO_ROLE_WITH_NAME &&
89             status != ROLE_NOT_READABLE &&
90             status != ROLE_NOT_WRITABLE &&
91             status != LESS_THAN_MIN_ROLE_DEGREE &&
92             status != MORE_THAN_MAX_ROLE_DEGREE &&
93             status != REF_MBEAN_OF_INCORRECT_CLASS &&
94             status != REF_MBEAN_NOT_REGISTERED) {
95             return false;
96         }
97         return true;
98     }
99 }
