26 package javax.management;
27 
28 // java import
29 import java.io.Serializable;
30 
31 /**
32  * Allows a query to be performed in the context of a specific MBean server.
33  *
34  * @since 1.5
35  */
36 public abstract class QueryEval implements Serializable {
37 
38     /* Serial version */
39     private static final long serialVersionUID = 2675899265640874796L;
40 
41     private static ThreadLocal<MBeanServer> server =
42         new InheritableThreadLocal<MBeanServer>();
43 
44     /**
45      * Constructor for subclasses to call.
46      */
47     public QueryEval() {}
48 
49     /**
50      * <p>Sets the MBean server on which the query is to be performed.
51      * The setting is valid for the thread performing the set.
52      * It is copied to any threads created by that thread at the moment
53      * of their creation.</p>
54      *
55      * <p>For historical reasons, this method is not static, but its
56      * behavior does not depend on the instance on which it is
57      * called.</p>
58      *
59      * @param s The MBean server on which the query is to be performed.
60      *
61      * @see #getMBeanServer
62      */
63     public void setMBeanServer(MBeanServer s) {
64         server.set(s);
65     }
66 
67     /**
68      * <p>Return the MBean server that was most recently given to the
69      * {@link #setMBeanServer setMBeanServer} method by this thread.
70      * If this thread never called that method, the result is the
71      * value its parent thread would have obtained from
72      * <code>getMBeanServer</code> at the moment of the creation of
73      * this thread, or null if there is no parent thread.</p>
74      *
75      * @return the MBean server.
76      *
77      * @see #setMBeanServer
78      *
79      */
80     public static MBeanServer getMBeanServer() {
81         return server.get();
82     }
83 }
