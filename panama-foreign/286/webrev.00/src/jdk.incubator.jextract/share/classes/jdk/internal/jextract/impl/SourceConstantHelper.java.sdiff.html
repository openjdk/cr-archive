247     private String emitMethodHandleField(String javaName, String nativeName, MethodType mtype,
248                                          FunctionDescriptor desc, boolean varargs) {
249         addFunctionDesc(javaName, desc);
250         incrAlign();
251         String fieldName = getMethodHandleFieldName(javaName);
252         indent();
253         append(PRIVATE_MODS + "MethodHandle ");
254         append(fieldName + " = RuntimeHelper.downcallHandle(\n");
255         incrAlign();
256         indent();
257         append("LIBRARIES, \"" + nativeName + "\"");

258         append(",\n");
259         indent();
260         append("\"" + mtype.toMethodDescriptorString() + "\",\n");
261         indent();
262         append(getFunctionDescFieldName(javaName));
263         append(", ");
264         // isVariadic
265         append(varargs);
266         append("\n");
267         decrAlign();
268         indent();
269         append(");\n");
270         decrAlign();

279         addLayout(javaName, layout);

317     private void emitLayoutString(MemoryLayout l) {
318         if (l instanceof ValueLayout) {
319             append(typeToLayoutName((ValueLayout) l));
320         } else if (l instanceof SequenceLayout) {
321             append("MemoryLayout.ofSequence(");
322             if (((SequenceLayout) l).elementCount().isPresent()) {
323                 append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
324             }
325             emitLayoutString(((SequenceLayout) l).elementLayout());
326             append(")");
327         } else if (l instanceof GroupLayout) {
328             if (l == CSupport.SysV.C_COMPLEX_LONGDOUBLE) {
329                 append("C_COMPLEX_LONGDOUBLE");
330             } else {

347                 append(")");
348             }
349         } else {
350             // padding
351             append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
352         }
353         if (l.name().isPresent()) {
354             append(".withName(\"" +  l.name().get() + "\")");
355         }
356     }

362     private String emitFunctionDescField(String javaName, FunctionDescriptor desc) {
363         incrAlign();
364         indent();
365         String fieldName = getFunctionDescFieldName(javaName);
366         final boolean noArgs = desc.argumentLayouts().isEmpty();
367         append(PRIVATE_MODS);
368         append("FunctionDescriptor ");
369         append(fieldName);
370         append(" = ");
371         if (desc.returnLayout().isPresent()) {
372             append("FunctionDescriptor.of(");
373             emitLayoutString(desc.returnLayout().get());
374             if (!noArgs) {
375                 append(",");
376             }
377         } else {
378             append("FunctionDescriptor.ofVoid(");
379         }
380         if (!noArgs) {
381             append("\n");
382             incrAlign();
383             String delim = "";
384             for (MemoryLayout e : desc.argumentLayouts()) {
385                 append(delim);
386                 indent();
387                 emitLayoutString(e);
388                 delim = ",\n";
389             }
390             append("\n");
391             decrAlign();
392             indent();
393         }
394         append(");\n");
395         decrAlign();
396         return fieldName;
397     }

434         StringBuilder buf = new StringBuilder();

438                 buf.append(value);
439                 buf.append("f");
440             } else {
441                 buf.append("Float.valueOf(\"");
442                 buf.append(value.toString());
443                 buf.append("\")");

491     private static boolean matchLayout(ValueLayout a, ValueLayout b) {
492         if (a == b) return true;
493         return (a.bitSize() == b.bitSize() &&
494                 a.order() == b.order() &&
495                 a.bitAlignment() == b.bitAlignment() &&
496                 a.attribute(ABI_CLASS_ATTR).equals(b.attribute(ABI_CLASS_ATTR)));
497     }

 96     private void newConstantClass() {
 97         if (constantCount > CONSTANTS_PER_CLASS) {
 98             classEnd();
 99             constantClassCount++;

103             this.constantCount = 0;

105             classBegin(null, baseClassName, false);

402     private void emitLayoutString(MemoryLayout l) {
403         if (l instanceof ValueLayout) {
404             append(typeToLayoutName((ValueLayout) l));
405         } else if (l instanceof SequenceLayout) {
406             append("MemoryLayout.ofSequence(");
407             if (((SequenceLayout) l).elementCount().isPresent()) {
408                 append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
409             }
410             emitLayoutString(((SequenceLayout) l).elementLayout());
411             append(")");
412         } else if (l instanceof GroupLayout) {
413             if (l == CSupport.SysV.C_COMPLEX_LONGDOUBLE) {
414                 append("C_COMPLEX_LONGDOUBLE");
415             } else {

432                 append(")");
433             }
434         } else {
435             // padding
436             append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
437         }
438         if (l.name().isPresent()) {
439             append(".withName(\"" +  l.name().get() + "\")");
440         }

447     private String emitFunctionDescField(String javaName, FunctionDescriptor desc) {
448         incrAlign();
449         indent();
450         String fieldName = getFunctionDescFieldName(javaName);
451         final boolean noArgs = desc.argumentLayouts().isEmpty();
452         append(PRIVATE_FINAL_MODS);
453         append("FunctionDescriptor ");
454         append(fieldName);
455         append(" = ");
456         if (desc.returnLayout().isPresent()) {
457             append("FunctionDescriptor.of(");
458             emitLayoutString(desc.returnLayout().get());
459             if (!noArgs) {
460                 append(",");
461             }
462         } else {
463             append("FunctionDescriptor.ofVoid(");
464         }
465         if (!noArgs) {
466             append("\n");
467             incrAlign();
468             String delim = "";
469             for (MemoryLayout e : desc.argumentLayouts()) {
470                 append(delim);
471                 indent();
472                 emitLayoutString(e);
473                 delim = ",\n";
474             }
475             append("\n");
476             decrAlign();
477             indent();
478         }
479         append(");\n");
480         decrAlign();
481         return fieldName;
482     }

576     private static boolean matchLayout(ValueLayout a, ValueLayout b) {
577         if (a == b) return true;
578         return (a.bitSize() == b.bitSize() &&
579                 a.order() == b.order() &&
580                 a.bitAlignment() == b.bitAlignment() &&
581                 a.attribute(ABI_CLASS_ATTR).equals(b.attribute(ABI_CLASS_ATTR)));
582     }
