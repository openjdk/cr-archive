134         private Object invoke(Object[] args) throws Throwable {
135             // one trailing Object[]
136             int nNamedArgs = function.argumentLayouts().size();
137             assert(args.length == nNamedArgs + 1);
138             // The last argument is the array of vararg collector
139             Object[] unnamedArgs = (Object[]) args[args.length - 1];
140 
141             int argsCount = nNamedArgs + unnamedArgs.length;
142             Class<?>[] argTypes = new Class<?>[argsCount];
143             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
144 
145             int pos = 0;
146             for (pos = 0; pos < nNamedArgs; pos++) {
147                 argTypes[pos] = varargs.parameterType(pos);
148                 argLayouts[pos] = function.argumentLayouts().get(pos);
149             }
150 
151             assert pos == nNamedArgs;
152             for (Object o: unnamedArgs) {
153                 argTypes[pos] = normalize(o.getClass());
154                 argLayouts[pos] = variadicLayout(argTypes[pos]);
155                 pos++;
156             }
157             assert pos == argsCount;
158 
159             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
160             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
161                     FunctionDescriptor.ofVoid(argLayouts) :
162                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
163             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
164             // flatten argument list so that it can be passed to an asSpreader MH
165             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
166             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
167             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
168 
169             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
170         }
