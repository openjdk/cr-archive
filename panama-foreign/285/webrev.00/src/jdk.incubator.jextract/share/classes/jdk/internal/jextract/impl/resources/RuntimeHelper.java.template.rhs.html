145         private Object invoke(Object[] args) throws Throwable {
146             // one trailing Object[]
147             int nNamedArgs = function.argumentLayouts().size();
148             assert(args.length == nNamedArgs + 1);
149             // The last argument is the array of vararg collector
150             Object[] unnamedArgs = (Object[]) args[args.length - 1];
151 
152             int argsCount = nNamedArgs + unnamedArgs.length;
153             Class<?>[] argTypes = new Class<?>[argsCount];
154             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
155 
156             int pos = 0;
157             for (pos = 0; pos < nNamedArgs; pos++) {
158                 argTypes[pos] = varargs.parameterType(pos);
159                 argLayouts[pos] = function.argumentLayouts().get(pos);
160             }
161 
162             assert pos == nNamedArgs;
163             for (Object o: unnamedArgs) {
164                 argTypes[pos] = normalize(o.getClass());
165                 argLayouts[pos] = variadicLayout(argTypes[pos]);
166                 pos++;
167             }
168             assert pos == argsCount;
169 
170             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
171             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
172                     FunctionDescriptor.ofVoid(argLayouts) :
173                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
174             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
175             // flatten argument list so that it can be passed to an asSpreader MH
176             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
177             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
178             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
179 
180             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
181         }

217         private Class<?> normalize(Class<?> c) {
