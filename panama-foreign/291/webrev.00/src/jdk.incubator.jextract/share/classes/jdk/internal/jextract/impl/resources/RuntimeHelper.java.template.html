135         private Object invoke(Object[] args) throws Throwable {
136             // one trailing Object[]
137             int nNamedArgs = function.argumentLayouts().size();
138             assert(args.length == nNamedArgs + 1);
139             // The last argument is the array of vararg collector
140             Object[] unnamedArgs = (Object[]) args[args.length - 1];
141 
142             int argsCount = nNamedArgs + unnamedArgs.length;
143             Class<?>[] argTypes = new Class<?>[argsCount];
144             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
145 
146             int pos = 0;
147             for (pos = 0; pos < nNamedArgs; pos++) {
148                 argTypes[pos] = varargs.parameterType(pos);
149                 argLayouts[pos] = function.argumentLayouts().get(pos);
150             }
151 
152             assert pos == nNamedArgs;
153             for (Object o: unnamedArgs) {
154                 argTypes[pos] = normalize(o.getClass());
155                 argLayouts[pos] = variadicLayout(argTypes[pos]);
156                 pos++;
157             }
158             assert pos == argsCount;
159 
160             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
161             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
162                     FunctionDescriptor.ofVoid(argLayouts) :
163                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
164             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
165             // flatten argument list so that it can be passed to an asSpreader MH
166             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
167             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
168             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
169 
170             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
171         }
