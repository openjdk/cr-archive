133         private Object invoke(Object[] args) throws Throwable {
134             // one trailing Object[]
135             int nNamedArgs = function.argumentLayouts().size();
136             assert(args.length == nNamedArgs + 1);
137             // The last argument is the array of vararg collector
138             Object[] unnamedArgs = (Object[]) args[args.length - 1];
139 
140             int argsCount = nNamedArgs + unnamedArgs.length;
141             Class<?>[] argTypes = new Class<?>[argsCount];
142             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
143 
144             int pos = 0;
145             for (pos = 0; pos < nNamedArgs; pos++) {
146                 argTypes[pos] = varargs.parameterType(pos);
147                 argLayouts[pos] = function.argumentLayouts().get(pos);
148             }
149 
150             assert pos == nNamedArgs;
151             for (Object o: unnamedArgs) {
152                 argTypes[pos] = normalize(o.getClass());
153                 argLayouts[pos] = variadicLayout(argTypes[pos]);
154                 pos++;
155             }
156             assert pos == argsCount;
157 
158             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
159             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
160                     FunctionDescriptor.ofVoid(argLayouts) :
161                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
162             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
163             // flatten argument list so that it can be passed to an asSpreader MH
164             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
165             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
166             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
167 
168             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
169         }

171         private static Class<?> unboxIfNeeded(Class<?> clazz) {
172             if (clazz == Boolean.class) {
173                 return boolean.class;
174             } else if (clazz == Void.class) {
175                 return void.class;
176             } else if (clazz == Byte.class) {
177                 return byte.class;
178             } else if (clazz == Character.class) {
179                 return char.class;

186             } else if (clazz == Float.class) {
187                 return float.class;

195         private Class<?> promote(Class<?> c) {
196             if (c == byte.class || c == char.class || c == short.class || c == int.class) {
197                 return long.class;

205         private Class<?> normalize(Class<?> c) {
206             c = unboxIfNeeded(c);
207             if (c.isPrimitive()) {
