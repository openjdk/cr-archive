 86 // Returns a string version of the method state.
 87 const char* CompiledMethod::state() const {
 88   int state = get_state();
 89   switch (state) {
 90   case not_installed:
 91     return "not installed";
 92   case in_use:
 93     return "in use";
 94   case not_used:
 95     return "not_used";
 96   case not_entrant:
 97     return "not_entrant";
 98   case zombie:
 99     return "zombie";
100   case unloaded:
101     return "unloaded";
102   default:
103     fatal("unexpected method state: %d", state);
104     return NULL;
105   }
106 }

117 ExceptionCache* CompiledMethod::exception_cache_acquire() const {
118   return Atomic::load_acquire(&_exception_cache);
119 }

121 void CompiledMethod::add_exception_cache_entry(ExceptionCache* new_entry) {
122   assert(ExceptionCache_lock->owned_by_self(),"Must hold the ExceptionCache_lock");
123   assert(new_entry != NULL,"Must be non null");
124   assert(new_entry->next() == NULL, "Must be null");
125 
126   for (;;) {
127     ExceptionCache *ec = exception_cache();
128     if (ec != NULL) {
129       Klass* ex_klass = ec->exception_type();
130       if (!ex_klass->is_loader_alive()) {
131         // We must guarantee that entries are not inserted with new next pointer
132         // edges to ExceptionCache entries with dead klasses, due to bad interactions
133         // with concurrent ExceptionCache cleanup. Therefore, the inserts roll
134         // the head pointer forward to the first live ExceptionCache, so that the new
135         // next pointers always point at live ExceptionCaches, that are not removed due
136         // to concurrent ExceptionCache cleanup.
137         ExceptionCache* next = ec->next();
138         if (Atomic::cmpxchg(&_exception_cache, ec, next) == ec) {
139           CodeCache::release_exception_cache(ec);
140         }
141         continue;
142       }
143       ec = exception_cache();
144       if (ec != NULL) {
145         new_entry->set_next(ec);
146       }
147     }
148     if (Atomic::cmpxchg(&_exception_cache, ec, new_entry) == ec) {
149       return;
150     }
151   }
152 }

154 void CompiledMethod::clean_exception_cache() {
155   // For each nmethod, only a single thread may call this cleanup function
156   // at the same time, whether called in STW cleanup or concurrent cleanup.
157   // Note that if the GC is processing exception cache cleaning in a concurrent phase,
158   // then a single writer may contend with cleaning up the head pointer to the
159   // first ExceptionCache node that has a Klass* that is alive. That is fine,
160   // as long as there is no concurrent cleanup of next pointers from concurrent writers.
161   // And the concurrent writers do not clean up next pointers, only the head.
162   // Also note that concurent readers will walk through Klass* pointers that are not
163   // alive. That does not cause ABA problems, because Klass* is deleted after
164   // a handshake with all threads, after all stale ExceptionCaches have been
165   // unlinked. That is also when the CodeCache::exception_cache_purge_list()
166   // is deleted, with all ExceptionCache entries that were cleaned concurrently.
167   // That similarly implies that CAS operations on ExceptionCache entries do not
168   // suffer from ABA problems as unlinking and deletion is separated by a global
169   // handshake operation.
170   ExceptionCache* prev = NULL;
171   ExceptionCache* curr = exception_cache_acquire();
172 
173   while (curr != NULL) {
174     ExceptionCache* next = curr->next();
175 
176     if (!curr->exception_type()->is_loader_alive()) {
177       if (prev == NULL) {
178         // Try to clean head; this is contended by concurrent inserts, that
179         // both lazily clean the head, and insert entries at the head. If
180         // the CAS fails, the operation is restarted.
181         if (Atomic::cmpxchg(&_exception_cache, curr, next) != curr) {
182           prev = NULL;
183           curr = exception_cache_acquire();
184           continue;
185         }
186       } else {
187         // It is impossible to during cleanup connect the next pointer to
188         // an ExceptionCache that has not been published before a safepoint
189         // prior to the cleanup. Therefore, release is not required.
190         prev->set_next(next);
191       }
192       // prev stays the same.
193 
194       CodeCache::release_exception_cache(curr);
195     } else {
196       prev = curr;
197     }
198 
199     curr = next;
200   }
201 }

203 // public method for accessing the exception cache
204 // These are the public access methods.
205 address CompiledMethod::handler_for_exception_and_pc(Handle exception, address pc) {
206   // We never grab a lock to read the exception cache, so we may
207   // have false negatives. This is okay, as it can only happen during
208   // the first few exception lookups for a given nmethod.
209   ExceptionCache* ec = exception_cache_acquire();
210   while (ec != NULL) {
211     address ret_val;
212     if ((ret_val = ec->match(exception,pc)) != NULL) {
213       return ret_val;
214     }
215     ec = ec->next();
216   }
217   return NULL;
218 }
219 
220 void CompiledMethod::add_handler_for_exception_and_pc(Handle exception, address pc, address handler) {
221   // There are potential race conditions during exception cache updates, so we
222   // must own the ExceptionCache_lock before doing ANY modifications. Because
223   // we don't lock during reads, it is possible to have several threads attempt
224   // to update the cache with the same data. We need to check for already inserted
225   // copies of the current data before adding it.
226 
227   MutexLocker ml(ExceptionCache_lock);
228   ExceptionCache* target_entry = exception_cache_entry_for_exception(exception);
229 
230   if (target_entry == NULL || !target_entry->add_address_and_handler(pc,handler)) {
231     target_entry = new ExceptionCache(exception,pc,handler);
232     add_exception_cache_entry(target_entry);
233   }
234 }

236 // private method for handling exception cache
237 // These methods are private, and used to manipulate the exception cache
238 // directly.
239 ExceptionCache* CompiledMethod::exception_cache_entry_for_exception(Handle exception) {
240   ExceptionCache* ec = exception_cache_acquire();
241   while (ec != NULL) {
242     if (ec->match_exception_with_space(exception)) {
243       return ec;
244     }
245     ec = ec->next();
246   }
247   return NULL;
248 }

302 address CompiledMethod::oops_reloc_begin() const {
303   // If the method is not entrant or zombie then a JMP is plastered over the
304   // first few bytes.  If an oop in the old code was there, that oop
305   // should not get GC'd.  Skip the first few bytes of oops on
306   // not-entrant methods.
307   if (frame_complete_offset() != CodeOffsets::frame_never_safe &&
308       code_begin() + frame_complete_offset() >
309       verified_entry_point() + NativeJump::instruction_size)
310   {
311     // If we have a frame_complete_offset after the native jump, then there
312     // is no point trying to look for oops before that. This is a requirement
313     // for being allowed to scan oops concurrently.
314     return code_begin() + frame_complete_offset();
315   }
316 
317   // It is not safe to read oops concurrently using entry barriers, if their
318   // location depend on whether the nmethod is entrant or not.
319   assert(BarrierSet::barrier_set()->barrier_set_nmethod() == NULL, "Not safe oop scan");
320 
321   address low_boundary = verified_entry_point();
322   if (!is_in_use() && is_nmethod()) {
323     low_boundary += NativeJump::instruction_size;
324     // %%% Note:  On SPARC we patch only a 4-byte trap, not a full NativeJump.
325     // This means that the low_boundary is going to be a little too high.
326     // This shouldn't matter, since oops of non-entrant methods are never used.
327     // In fact, why are we bothering to look at oops in a non-entrant method??
328   }
329   return low_boundary;
330 }

332 int CompiledMethod::verify_icholder_relocations() {
333   ResourceMark rm;
334   int count = 0;
335 
336   RelocIterator iter(this);
337   while(iter.next()) {
338     if (iter.type() == relocInfo::virtual_call_type) {
339       if (CompiledIC::is_icholder_call_site(iter.virtual_call_reloc(), this)) {
340         CompiledIC *ic = CompiledIC_at(&iter);
341         if (TraceCompiledIC) {
342           tty->print("noticed icholder " INTPTR_FORMAT " ", p2i(ic->cached_icholder()));
343           ic->print();
344         }
345         assert(ic->cached_icholder() != NULL, "must be non-NULL");
346         count++;
347       }
348     }
349   }
350 
351   return count;
352 }

354 // Method that knows how to preserve outgoing arguments at call. This method must be
355 // called with a frame corresponding to a Java invoke
356 void CompiledMethod::preserve_callee_argument_oops(frame fr, const RegisterMap *reg_map, OopClosure* f) {
357   if (method() != NULL && !method()->is_native()) {
358     address pc = fr.pc();
359     SimpleScopeDesc ssd(this, pc);
360     if (ssd.is_optimized_linkToNative()) return; // call was replaced
361     Bytecode_invoke call(methodHandle(Thread::current(), ssd.method()), ssd.bci());
362     bool has_receiver = call.has_receiver();
363     bool has_appendix = call.has_appendix();
364     Symbol* signature = call.signature();
365 
366     // The method attached by JIT-compilers should be used, if present.
367     // Bytecode can be inaccurate in such case.
368     Method* callee = attached_method_before_pc(pc);
369     if (callee != NULL) {
370       has_receiver = !(callee->access_flags().is_static());
371       has_appendix = false;
372       signature = callee->signature();
373     }
374 
375     fr.oops_compiled_arguments_do(signature, has_receiver, has_appendix, reg_map, f);
376   }
377 }

379 Method* CompiledMethod::attached_method(address call_instr) {
380   assert(code_contains(call_instr), "not part of the nmethod");
381   RelocIterator iter(this, call_instr, call_instr + 1);
382   while (iter.next()) {
383     if (iter.addr() == call_instr) {
384       switch(iter.type()) {
385         case relocInfo::static_call_type:      return iter.static_call_reloc()->method_value();
386         case relocInfo::opt_virtual_call_type: return iter.opt_virtual_call_reloc()->method_value();
387         case relocInfo::virtual_call_type:     return iter.virtual_call_reloc()->method_value();
388         default:                               break;
389       }
390     }
391   }
392   return NULL; // not found
393 }

395 Method* CompiledMethod::attached_method_before_pc(address pc) {
396   if (NativeCall::is_call_before(pc)) {
397     NativeCall* ncall = nativeCall_before(pc);
398     return attached_method(ncall->instruction_address());
399   }
400   return NULL; // not a call

426   }

450 bool CompiledMethod::clean_ic_if_metadata_is_dead(CompiledIC *ic) {
451   if (ic->is_clean()) {
452     return true;
453   }
454   if (ic->is_icholder_call()) {
455     // The only exception is compiledICHolder metdata which may
456     // yet be marked below. (We check this further below).
457     CompiledICHolder* cichk_metdata = ic->cached_icholder();
458 
459     if (cichk_metdata->is_loader_alive()) {
460       return true;
461     }
462   } else {
463     Metadata* ic_metdata = ic->cached_metadata();
464     if (ic_metdata != NULL) {
465       if (ic_metdata->is_klass()) {
466         if (((Klass*)ic_metdata)->is_loader_alive()) {
467           return true;
468         }
469       } else if (ic_metdata->is_method()) {
470         Method* method = (Method*)ic_metdata;
471         assert(!method->is_old(), "old method should have been cleaned");
472         if (method->method_holder()->is_loader_alive()) {
473           return true;
474         }
475       } else {
476         ShouldNotReachHere();
477       }
478     }
479   }
480 
481   return ic->set_to_clean();
482 }

484 // Clean references to unloaded nmethods at addr from this one, which is not unloaded.
485 template <class CompiledICorStaticCall>
486 static bool clean_if_nmethod_is_unloaded(CompiledICorStaticCall *ic, address addr, CompiledMethod* from,
487                                          bool clean_all) {
488   // Ok, to lookup references to zombies here
489   CodeBlob *cb = CodeCache::find_blob_unsafe(addr);
490   CompiledMethod* nm = (cb != NULL) ? cb->as_compiled_method_or_null() : NULL;
491   if (nm != NULL) {
492     // Clean inline caches pointing to both zombie and not_entrant methods
493     if (clean_all || !nm->is_in_use() || nm->is_unloading() || (nm->method()->code() != nm)) {
494       // Inline cache cleaning should only be initiated on CompiledMethods that have been
495       // observed to be is_alive(). However, with concurrent code cache unloading, it is
496       // possible that by now, the state has become !is_alive. This can happen in two ways:
497       // 1) It can be racingly flipped to unloaded if the nmethod // being cleaned (from the
498       // sweeper) is_unloading(). This is fine, because if that happens, then the inline
499       // caches have already been cleaned under the same CompiledICLocker that we now hold during
500       // inline cache cleaning, and we will simply walk the inline caches again, and likely not
501       // find much of interest to clean. However, this race prevents us from asserting that the
502       // nmethod is_alive(). The is_unloading() function is completely monotonic; once set due
503       // to an oop dying, it remains set forever until freed. Because of that, all unloaded
504       // nmethods are is_unloading(), but notably, an unloaded nmethod may also subsequently
505       // become zombie (when the sweeper converts it to zombie).
506       // 2) It can be racingly flipped to zombie if the nmethod being cleaned (by the concurrent
507       // GC) cleans a zombie nmethod that is concurrently made zombie by the sweeper. In this
508       // scenario, the sweeper will first transition the nmethod to zombie, and then when
509       // unregistering from the GC, it will wait until the GC is done. The GC will then clean
510       // the inline caches *with IC stubs*, even though no IC stubs are needed. This is fine,
511       // as long as the IC stubs are guaranteed to be released until the next safepoint, where
512       // IC finalization requires live IC stubs to not be associated with zombie nmethods.
513       // This is guaranteed, because the sweeper does not have a single safepoint check until
514       // after it completes the whole transition function; it will wake up after the GC is
515       // done with concurrent code cache cleaning (which blocks out safepoints using the
516       // suspendible threads set), and then call clear_ic_callsites, which will release the
517       // associated IC stubs, before a subsequent safepoint poll can be reached. This
518       // guarantees that the spuriously created IC stubs are released appropriately before
519       // IC finalization in a safepoint gets to run. Therefore, this race is fine. This is also
520       // valid in a scenario where an inline cache of a zombie nmethod gets a spurious IC stub,
521       // and then when cleaning another inline cache, fails to request an IC stub because we
522       // exhausted the IC stub buffer. In this scenario, the GC will request a safepoint after
523       // yielding the suspendible therad set, effectively unblocking safepoints. Before such
524       // a safepoint can be reached, the sweeper similarly has to wake up, clear the IC stubs,
525       // and reach the next safepoint poll, after the whole transition function has completed.
526       // Due to the various races that can cause an nmethod to first be is_alive() and then
527       // racingly become !is_alive(), it is unfortunately not possible to assert the nmethod
528       // is_alive(), !is_unloaded() or !is_zombie() here.
529       if (!ic->set_to_clean(!from->is_unloading())) {
530         return false;
531       }
532       assert(ic->is_clean(), "nmethod " PTR_FORMAT "not clean %s", p2i(from), from->method()->name_and_sig_as_C_string());
533     }
534   }
535   return true;
536 }

539                                          bool clean_all) {
540   return clean_if_nmethod_is_unloaded(ic, ic->ic_destination(), from, clean_all);
541 }

548 // Cleans caches in nmethods that point to either classes that are unloaded
549 // or nmethods that are unloaded.
550 //
551 // Can be called either in parallel by G1 currently or after all
552 // nmethods are unloaded.  Return postponed=true in the parallel case for
553 // inline caches found that point to nmethods that are not yet visited during
554 // the do_unloading walk.
555 bool CompiledMethod::unload_nmethod_caches(bool unloading_occurred) {
556   ResourceMark rm;
557 
558   // Exception cache only needs to be called if unloading occurred
559   if (unloading_occurred) {
560     clean_exception_cache();
561   }
562 
563   if (!cleanup_inline_caches_impl(unloading_occurred, false)) {
564     return false;
565   }
566 
567 #ifdef ASSERT
568   // Check that the metadata embedded in the nmethod is alive
569   CheckClass check_class;
570   metadata_do(&check_class);
571 #endif
572   return true;
573 }

575 void CompiledMethod::run_nmethod_entry_barrier() {
576   BarrierSetNMethod* bs_nm = BarrierSet::barrier_set()->barrier_set_nmethod();
577   if (bs_nm != NULL) {
578     // We want to keep an invariant that nmethods found through iterations of a Thread's
579     // nmethods found in safepoints have gone through an entry barrier and are not armed.
580     // By calling this nmethod entry barrier, it plays along and acts
581     // like any other nmethod found on the stack of a thread (fewer surprises).
582     nmethod* nm = as_nmethod_or_null();
583     if (nm != NULL) {
584       bool alive = bs_nm->nmethod_entry_barrier(nm);
585       assert(alive, "should be alive");
586     }
587   }
588 }

590 void CompiledMethod::cleanup_inline_caches(bool clean_all) {
591   for (;;) {
592     ICRefillVerifier ic_refill_verifier;
593     { CompiledICLocker ic_locker(this);
594       if (cleanup_inline_caches_impl(false, clean_all)) {
595         return;
596       }
597     }
598     // Call this nmethod entry barrier from the sweeper.
599     run_nmethod_entry_barrier();
600     InlineCacheBuffer::refill_ic_stubs();
601   }
602 }

604 // Called to clean up after class unloading for live nmethods and from the sweeper
605 // for all methods.
606 bool CompiledMethod::cleanup_inline_caches_impl(bool unloading_occurred, bool clean_all) {
607   assert(CompiledICLocker::is_safe(this), "mt unsafe call");
608   ResourceMark rm;
609 
610   // Find all calls in an nmethod and clear the ones that point to non-entrant,
611   // zombie and unloaded nmethods.
612   RelocIterator iter(this, oops_reloc_begin());
613   bool is_in_static_stub = false;
614   while(iter.next()) {
615 
616     switch (iter.type()) {
617 
618     case relocInfo::virtual_call_type:
619       if (unloading_occurred) {
620         // If class unloading occurred we first clear ICs where the cached metadata
621         // is referring to an unloaded klass or method.
622         if (!clean_ic_if_metadata_is_dead(CompiledIC_at(&iter))) {
623           return false;
624         }
625       }
626 
627       if (!clean_if_nmethod_is_unloaded(CompiledIC_at(&iter), this, clean_all)) {
628         return false;
629       }
630       break;
631 
632     case relocInfo::opt_virtual_call_type:
633       if (!clean_if_nmethod_is_unloaded(CompiledIC_at(&iter), this, clean_all)) {
634         return false;
635       }
636       break;
637 
638     case relocInfo::static_call_type:
639       if (!clean_if_nmethod_is_unloaded(compiledStaticCall_at(iter.reloc()), this, clean_all)) {
640         return false;
641       }
642       break;
643 
644     case relocInfo::static_stub_type: {
645       is_in_static_stub = true;
646       break;
647     }
648 
649     case relocInfo::metadata_type: {
650       // Only the metadata relocations contained in static/opt virtual call stubs
651       // contains the Method* passed to c2i adapters. It is the only metadata
652       // relocation that needs to be walked, as it is the one metadata relocation
653       // that violates the invariant that all metadata relocations have an oop
654       // in the compiled method (due to deferred resolution and code patching).
655 
656       // This causes dead metadata to remain in compiled methods that are not
657       // unloading. Unless these slippery metadata relocations of the static
658       // stubs are at least cleared, subsequent class redefinition operations
659       // will access potentially free memory, and JavaThread execution
660       // concurrent to class unloading may call c2i adapters with dead methods.
661       if (!is_in_static_stub) {
662         // The first metadata relocation after a static stub relocation is the
663         // metadata relocation of the static stub used to pass the Method* to
664         // c2i adapters.
665         continue;
666       }
667       is_in_static_stub = false;
668       if (is_unloading()) {
669         // If the nmethod itself is dying, then it may point at dead metadata.
670         // Nobody should follow that metadata; it is strictly unsafe.
671         continue;
672       }
673       metadata_Relocation* r = iter.metadata_reloc();
674       Metadata* md = r->metadata_value();
675       if (md != NULL && md->is_method()) {
676         Method* method = static_cast<Method*>(md);
677         if (!method->method_holder()->is_loader_alive()) {
678           Atomic::store(r->metadata_addr(), (Method*)NULL);
679 
680           if (!r->metadata_is_immediate()) {
681             r->fix_metadata_relocation();
682           }
683         }
684       }
685       break;
686     }
687 
688     default:
689       break;
690     }
691   }
692 
693   return true;
694 }
695 
696 // Iterating over all nmethods, e.g. with the help of CodeCache::nmethods_do(fun) was found
697 // to not be inherently safe. There is a chance that fields are seen which are not properly
698 // initialized. This happens despite the fact that nmethods_do() asserts the CodeCache_lock
699 // to be held.
700 // To bundle knowledge about necessary checks in one place, this function was introduced.
701 // It is not claimed that these checks are sufficient, but they were found to be necessary.
702 bool CompiledMethod::nmethod_access_is_safe(nmethod* nm) {
703   Method* method = (nm == NULL) ? NULL : nm->method();  // nm->method() may be uninitialized, i.e. != NULL, but invalid
704   return (nm != NULL) && (method != NULL) && (method->signature() != NULL) &&
705          !nm->is_zombie() && !nm->is_not_installed() &&
706          os::is_readable_pointer(method) &&
707          os::is_readable_pointer(method->constants()) &&
708          os::is_readable_pointer(method->signature());
709 }

711 address CompiledMethod::continuation_for_implicit_exception(address pc, bool for_div0_check) {
712   // Exception happened outside inline-cache check code => we are inside
713   // an active nmethod => use cpc to determine a return address
714   int exception_offset = pc - code_begin();
715   int cont_offset = ImplicitExceptionTable(this).continuation_offset( exception_offset );
716 #ifdef ASSERT
717   if (cont_offset == 0) {
718     Thread* thread = Thread::current();
719     ResetNoHandleMark rnm; // Might be called from LEAF/QUICK ENTRY
720     HandleMark hm(thread);
721     ResourceMark rm(thread);
722     CodeBlob* cb = CodeCache::find_blob(pc);
723     assert(cb != NULL && cb == this, "");
724     ttyLocker ttyl;
725     tty->print_cr("implicit exception happened at " INTPTR_FORMAT, p2i(pc));
726     print();
727     method()->print_codes();
728     print_code();
729     print_pcs();
730   }
731 #endif
732   if (cont_offset == 0) {
733     // Let the normal error handling report the exception
734     return NULL;
735   }
736   if (cont_offset == exception_offset) {
737 #if INCLUDE_JVMCI
738     Deoptimization::DeoptReason deopt_reason = for_div0_check ? Deoptimization::Reason_div0_check : Deoptimization::Reason_null_check;
739     JavaThread *thread = JavaThread::current();
740     thread->set_jvmci_implicit_exception_pc(pc);
741     thread->set_pending_deoptimization(Deoptimization::make_trap_request(deopt_reason,
742                                                                          Deoptimization::Action_reinterpret));
743     return (SharedRuntime::deopt_blob()->implicit_exception_uncommon_trap());
744 #else
745     ShouldNotReachHere();
746 #endif
747   }
748   return code_begin() + cont_offset;
749 }

751 class HasEvolDependency : public MetadataClosure {
752   bool _has_evol_dependency;
753  public:
754   HasEvolDependency() : _has_evol_dependency(false) {}
755   void do_metadata(Metadata* md) {
756     if (md->is_method()) {
757       Method* method = (Method*)md;
758       if (method->is_old()) {
759         _has_evol_dependency = true;
760       }
761     }
762   }
763   bool has_evol_dependency() const { return _has_evol_dependency; }
764 };
765 
766 bool CompiledMethod::has_evol_metadata() {
767   // Check the metadata in relocIter and CompiledIC and also deoptimize
768   // any nmethod that has reference to old methods.
769   HasEvolDependency check_evol;
770   metadata_do(&check_evol);
771   if (check_evol.has_evol_dependency() && log_is_enabled(Debug, redefine, class, nmethod)) {
772     ResourceMark rm;
773     log_debug(redefine, class, nmethod)
774             ("Found evol dependency of nmethod %s.%s(%s) compile_id=%d on in nmethod metadata",
775              _method->method_holder()->external_name(),
776              _method->name()->as_C_string(),
777              _method->signature()->as_C_string(),
778              compile_id());
779   }
780   return check_evol.has_evol_dependency();
781 }
