279   static int numa_node_to_cpus(int node, unsigned long *buffer, int bufferlen) {
280     // use the latest version of numa_node_to_cpus if available
281     if (_numa_node_to_cpus_v2 != NULL) {
282 
283       // libnuma bitmask struct
284       struct bitmask {
285         unsigned long size; /* number of bits in the map */
286         unsigned long *maskp;
287       };
288 
289       struct bitmask mask;
290       mask.maskp = (unsigned long *)buffer;
291       mask.size = bufferlen * 8;
292       return _numa_node_to_cpus_v2(node, &mask);
293     } else if (_numa_node_to_cpus != NULL) {
294       return _numa_node_to_cpus(node, buffer, bufferlen);
295     }
296     return -1;
297   }

315   static NumaAllocationPolicy identify_numa_policy() {
316     for (int node = 0; node <= Linux::numa_max_node(); node++) {
317       if (Linux::_numa_bitmask_isbitset(Linux::_numa_interleave_bitmask, node)) {
318         return Interleave;
319       }
320     }
321     return Membind;
322   }

324   static void numa_interleave_memory(void *start, size_t size) {
325     // Prefer v2 API
326     if (_numa_interleave_memory_v2 != NULL) {
327       if (is_running_in_interleave_mode()) {
328         _numa_interleave_memory_v2(start, size, _numa_interleave_bitmask);
329       } else if (_numa_membind_bitmask != NULL) {
330         _numa_interleave_memory_v2(start, size, _numa_membind_bitmask);
331       }
332     } else if (_numa_interleave_memory != NULL) {
333       _numa_interleave_memory(start, size, _numa_all_nodes);
334     }

354   // Check if numa node is configured (non-zero memory node).
355   static bool is_node_in_configured_nodes(unsigned int n) {
356     if (_numa_bitmask_isbitset != NULL && _numa_all_nodes_ptr != NULL) {
357       return _numa_bitmask_isbitset(_numa_all_nodes_ptr, n);
358     } else
359       return false;
360   }
361   // Check if numa node exists in the system (including zero memory nodes).
362   static bool is_node_in_existing_nodes(unsigned int n) {
363     if (_numa_bitmask_isbitset != NULL && _numa_nodes_ptr != NULL) {
364       return _numa_bitmask_isbitset(_numa_nodes_ptr, n);
365     } else if (_numa_bitmask_isbitset != NULL && _numa_all_nodes_ptr != NULL) {
366       // Not all libnuma API v2 implement numa_nodes_ptr, so it's not possible
367       // to trust the API version for checking its absence. On the other hand,
368       // numa_nodes_ptr found in libnuma 2.0.9 and above is the only way to get
369       // a complete view of all numa nodes in the system, hence numa_nodes_ptr
370       // is used to handle CPU and nodes on architectures (like PowerPC) where
371       // there can exist nodes with CPUs but no memory or vice-versa and the
372       // nodes may be non-contiguous. For most of the architectures, like
373       // x86_64, numa_node_ptr presents the same node set as found in
374       // numa_all_nodes_ptr so it's possible to use numa_all_nodes_ptr as a
375       // substitute.
376       return _numa_bitmask_isbitset(_numa_all_nodes_ptr, n);
377     } else
378       return false;
379   }
380   // Check if node is in bound node set.
381   static bool is_node_in_bound_nodes(int node) {
382     if (_numa_bitmask_isbitset != NULL) {
383       if (is_running_in_interleave_mode()) {
384         return _numa_bitmask_isbitset(_numa_interleave_bitmask, node);
385       } else {
386         return _numa_membind_bitmask != NULL ? _numa_bitmask_isbitset(_numa_membind_bitmask, node) : false;
387       }
388     }
389     return false;
390   }
391   // Check if bound to only one numa node.
392   // Returns true if bound to a single numa node, otherwise returns false.
393   static bool is_bound_to_single_node() {
394     int nodes = 0;
395     struct bitmask* bmp = NULL;
396     unsigned int node = 0;
397     unsigned int highest_node_number = 0;
398 
399     if (_numa_get_membind != NULL && _numa_max_node != NULL && _numa_bitmask_isbitset != NULL) {
400       bmp = _numa_get_membind();
401       highest_node_number = _numa_max_node();
402     } else {
403       return false;
404     }
405 
406     for (node = 0; node <= highest_node_number; node++) {
407       if (_numa_bitmask_isbitset(bmp, node)) {
408         nodes++;
409       }
410     }
411 
412     if (nodes == 1) {
413       return true;
414     } else {
415       return false;
416     }
417   }
