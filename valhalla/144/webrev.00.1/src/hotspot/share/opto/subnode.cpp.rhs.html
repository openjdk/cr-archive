  49 //------------------------------Identity---------------------------------------
  50 // If right input is a constant 0, return the left input.
  51 Node* SubNode::Identity(PhaseGVN* phase) {
  52   assert(in(1) != this, "Must already have called Value");
  53   assert(in(2) != this, "Must already have called Value");
  54 
  55   // Remove double negation
  56   const Type *zero = add_id();
  57   if( phase->type( in(1) )->higher_equal( zero ) &&
  58       in(2)->Opcode() == Opcode() &&
  59       phase->type( in(2)->in(1) )->higher_equal( zero ) ) {
  60     return in(2)->in(2);
  61   }
  62 
  63   // Convert "(X+Y) - Y" into X and "(X+Y) - X" into Y
  64   if( in(1)->Opcode() == Op_AddI ) {
  65     if( phase->eqv(in(1)->in(2),in(2)) )
  66       return in(1)->in(1);
  67     if (phase->eqv(in(1)->in(1),in(2)))
  68       return in(1)->in(2);
  69 
  70     // Also catch: "(X + Opaque2(Y)) - Y".  In this case, 'Y' is a loop-varying
  71     // trip counter and X is likely to be loop-invariant (that's how O2 Nodes
  72     // are originally used, although the optimizer sometimes jiggers things).
  73     // This folding through an O2 removes a loop-exit use of a loop-varying
  74     // value and generally lowers register pressure in and around the loop.
  75     if( in(1)->in(2)->Opcode() == Op_Opaque2 &&
  76         phase->eqv(in(1)->in(2)->in(1),in(2)) )
  77       return in(1)->in(1);
  78   }
  79 
  80   return ( phase->type( in(2) )->higher_equal( zero ) ) ? in(1) : this;
  81 }

  83 //------------------------------Value------------------------------------------
  84 // A subtract node differences it's two inputs.
  85 const Type* SubNode::Value_common(PhaseTransform *phase) const {
  86   const Node* in1 = in(1);
  87   const Node* in2 = in(2);
  88   // Either input is TOP ==> the result is TOP
  89   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
  90   if( t1 == Type::TOP ) return Type::TOP;
  91   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
  92   if( t2 == Type::TOP ) return Type::TOP;
  93 
  94   // Not correct for SubFnode and AddFNode (must check for infinity)
  95   // Equal?  Subtract is zero
  96   if (in1->eqv_uncast(in2))  return add_id();
  97 
  98   // Either input is BOTTOM ==> the result is the local BOTTOM
  99   if( t1 == Type::BOTTOM || t2 == Type::BOTTOM )
 100     return bottom_type();
 101 
 102   return NULL;
 103 }

 105 const Type* SubNode::Value(PhaseGVN* phase) const {
 106   const Type* t = Value_common(phase);
 107   if (t != NULL) {
 108     return t;
 109   }
 110   const Type* t1 = phase->type(in(1));
 111   const Type* t2 = phase->type(in(2));
 112   return sub(t1,t2);            // Local flavor of type subtraction
 113 
 114 }

 119 static bool is_cloop_increment(Node* inc) {
 120   precond(inc->Opcode() == Op_AddI || inc->Opcode() == Op_AddL);
 121 
 122   if (!inc->in(1)->is_Phi()) {
 123     return false;
 124   }
 125   const PhiNode* phi = inc->in(1)->as_Phi();
 126 
 127   if (phi->is_copy() || !phi->region()->is_CountedLoop()) {
 128     return false;
 129   }
 130 
 131   return inc == phi->region()->as_CountedLoop()->incr();
 132 }
 133 
 134 // Given the expression '(x + C) - v', or
 135 //                      'v - (x + C)', we examine nodes '+' and 'v':
 136 //
 137 //  1. Do not convert if '+' is a counted-loop increment, because the '-' is
 138 //     loop invariant and converting extends the live-range of 'x' to overlap
 139 //     with the '+', forcing another register to be used in the loop.
 140 //
 141 //  2. Do not convert if 'v' is a counted-loop induction variable, because
 142 //     'x' might be invariant.
 143 //
 144 static bool ok_to_convert(Node* inc, Node* var) {
 145   return !(is_cloop_increment(inc) || var->is_cloop_ind_var());
 146 }

 148 //------------------------------Ideal------------------------------------------
 149 Node *SubINode::Ideal(PhaseGVN *phase, bool can_reshape){
 150   Node *in1 = in(1);
 151   Node *in2 = in(2);
 152   uint op1 = in1->Opcode();
 153   uint op2 = in2->Opcode();
 154 
 155 #ifdef ASSERT
 156   // Check for dead loop
 157   if( phase->eqv( in1, this ) || phase->eqv( in2, this ) ||
 158       ( ( op1 == Op_AddI || op1 == Op_SubI ) &&
 159         ( phase->eqv( in1->in(1), this ) || phase->eqv( in1->in(2), this ) ||
 160           phase->eqv( in1->in(1), in1  ) || phase->eqv( in1->in(2), in1 ) ) ) )
 161     assert(false, "dead loop in SubINode::Ideal");
 162 #endif
 163 
 164   const Type *t2 = phase->type( in2 );
 165   if( t2 == Type::TOP ) return NULL;
 166   // Convert "x-c0" into "x+ -c0".
 167   if( t2->base() == Type::Int ){        // Might be bottom or top...
 168     const TypeInt *i = t2->is_int();
 169     if( i->is_con() )
 170       return new AddINode(in1, phase->intcon(-i->get_con()));
 171   }
 172 
 173   // Convert "(x+c0) - y" into (x-y) + c0"
 174   // Do not collapse (x+c0)-y if "+" is a loop increment or
 175   // if "y" is a loop induction variable.
 176   if( op1 == Op_AddI && ok_to_convert(in1, in2) ) {
 177     const Type *tadd = phase->type( in1->in(2) );
 178     if( tadd->singleton() && tadd != Type::TOP ) {
 179       Node *sub2 = phase->transform( new SubINode( in1->in(1), in2 ));
 180       return new AddINode( sub2, in1->in(2) );
 181     }
 182   }
 183 
 184 
 185   // Convert "x - (y+c0)" into "(x-y) - c0"
 186   // Need the same check as in above optimization but reversed.
 187   if (op2 == Op_AddI && ok_to_convert(in2, in1)) {
 188     Node* in21 = in2->in(1);
 189     Node* in22 = in2->in(2);
 190     const TypeInt* tcon = phase->type(in22)->isa_int();
 191     if (tcon != NULL && tcon->is_con()) {
 192       Node* sub2 = phase->transform( new SubINode(in1, in21) );
 193       Node* neg_c0 = phase->intcon(- tcon->get_con());
 194       return new AddINode(sub2, neg_c0);
 195     }
 196   }
 197 
 198   const Type *t1 = phase->type( in1 );
 199   if( t1 == Type::TOP ) return NULL;
 200 
 201 #ifdef ASSERT
 202   // Check for dead loop
 203   if( ( op2 == Op_AddI || op2 == Op_SubI ) &&
 204       ( phase->eqv( in2->in(1), this ) || phase->eqv( in2->in(2), this ) ||
 205         phase->eqv( in2->in(1), in2  ) || phase->eqv( in2->in(2), in2  ) ) )
 206     assert(false, "dead loop in SubINode::Ideal");
 207 #endif
 208 
 209   // Convert "x - (x+y)" into "-y"
 210   if( op2 == Op_AddI &&
 211       phase->eqv( in1, in2->in(1) ) )
 212     return new SubINode( phase->intcon(0),in2->in(2));
 213   // Convert "(x-y) - x" into "-y"
 214   if( op1 == Op_SubI &&
 215       phase->eqv( in1->in(1), in2 ) )
 216     return new SubINode( phase->intcon(0),in1->in(2));
 217   // Convert "x - (y+x)" into "-y"
 218   if( op2 == Op_AddI &&
 219       phase->eqv( in1, in2->in(2) ) )
 220     return new SubINode( phase->intcon(0),in2->in(1));
 221 
 222   // Convert "0 - (x-y)" into "y-x"
 223   if( t1 == TypeInt::ZERO && op2 == Op_SubI )
 224     return new SubINode( in2->in(2), in2->in(1) );
 225 
 226   // Convert "0 - (x+con)" into "-con-x"
 227   jint con;
 228   if( t1 == TypeInt::ZERO && op2 == Op_AddI &&
 229       (con = in2->in(2)->find_int_con(0)) != 0 )
 230     return new SubINode( phase->intcon(-con), in2->in(1) );
 231 
 232   // Convert "(X+A) - (X+B)" into "A - B"
 233   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(1) == in2->in(1) )
 234     return new SubINode( in1->in(2), in2->in(2) );
 235 
 236   // Convert "(A+X) - (B+X)" into "A - B"
 237   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(2) == in2->in(2) )
 238     return new SubINode( in1->in(1), in2->in(1) );
 239 
 240   // Convert "(A+X) - (X+B)" into "A - B"
 241   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(2) == in2->in(1) )
 242     return new SubINode( in1->in(1), in2->in(2) );
 243 
 244   // Convert "(X+A) - (B+X)" into "A - B"
 245   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(1) == in2->in(2) )
 246     return new SubINode( in1->in(2), in2->in(1) );
 247 
 248   // Convert "A-(B-C)" into (A+C)-B", since add is commutative and generally
 249   // nicer to optimize than subtract.
 250   if( op2 == Op_SubI && in2->outcnt() == 1) {
 251     Node *add1 = phase->transform( new AddINode( in1, in2->in(2) ) );
 252     return new SubINode( add1, in2->in(1) );
 253   }
 254 
 255   // Convert "0-(A>>31)" into "(A>>>31)"
 256   if ( op2 == Op_RShiftI ) {
 257     Node *in21 = in2->in(1);
 258     Node *in22 = in2->in(2);
 259     const TypeInt *zero = phase->type(in1)->isa_int();
 260     const TypeInt *t21 = phase->type(in21)->isa_int();
 261     const TypeInt *t22 = phase->type(in22)->isa_int();
 262     if ( t21 && t22 && zero == TypeInt::ZERO && t22->is_con(31) ) {
 263       return new URShiftINode(in21, in22);
 264     }
 265   }
 266 
 267   return NULL;
 268 }

 270 //------------------------------sub--------------------------------------------
 271 // A subtract node differences it's two inputs.
 272 const Type *SubINode::sub( const Type *t1, const Type *t2 ) const {
 273   const TypeInt *r0 = t1->is_int(); // Handy access
 274   const TypeInt *r1 = t2->is_int();
 275   int32_t lo = java_subtract(r0->_lo, r1->_hi);
 276   int32_t hi = java_subtract(r0->_hi, r1->_lo);
 277 
 278   // We next check for 32-bit overflow.
 279   // If that happens, we just assume all integers are possible.
 280   if( (((r0->_lo ^ r1->_hi) >= 0) ||    // lo ends have same signs OR
 281        ((r0->_lo ^      lo) >= 0)) &&   // lo results have same signs AND
 282       (((r0->_hi ^ r1->_lo) >= 0) ||    // hi ends have same signs OR
 283        ((r0->_hi ^      hi) >= 0)) )    // hi results have same signs
 284     return TypeInt::make(lo,hi,MAX2(r0->_widen,r1->_widen));
 285   else                          // Overflow; assume all integers
 286     return TypeInt::INT;
 287 }

 290 //------------------------------Ideal------------------------------------------
 291 Node *SubLNode::Ideal(PhaseGVN *phase, bool can_reshape) {
 292   Node *in1 = in(1);
 293   Node *in2 = in(2);
 294   uint op1 = in1->Opcode();
 295   uint op2 = in2->Opcode();
 296 
 297 #ifdef ASSERT
 298   // Check for dead loop
 299   if( phase->eqv( in1, this ) || phase->eqv( in2, this ) ||
 300       ( ( op1 == Op_AddL || op1 == Op_SubL ) &&
 301         ( phase->eqv( in1->in(1), this ) || phase->eqv( in1->in(2), this ) ||
 302           phase->eqv( in1->in(1), in1  ) || phase->eqv( in1->in(2), in1  ) ) ) )
 303     assert(false, "dead loop in SubLNode::Ideal");
 304 #endif
 305 
 306   if( phase->type( in2 ) == Type::TOP ) return NULL;
 307   const TypeLong *i = phase->type( in2 )->isa_long();
 308   // Convert "x-c0" into "x+ -c0".
 309   if( i &&                      // Might be bottom or top...
 310       i->is_con() )
 311     return new AddLNode(in1, phase->longcon(-i->get_con()));
 312 
 313   // Convert "(x+c0) - y" into (x-y) + c0"
 314   // Do not collapse (x+c0)-y if "+" is a loop increment or
 315   // if "y" is a loop induction variable.
 316   if( op1 == Op_AddL && ok_to_convert(in1, in2) ) {
 317     Node *in11 = in1->in(1);
 318     const Type *tadd = phase->type( in1->in(2) );
 319     if( tadd->singleton() && tadd != Type::TOP ) {
 320       Node *sub2 = phase->transform( new SubLNode( in11, in2 ));
 321       return new AddLNode( sub2, in1->in(2) );
 322     }
 323   }
 324 
 325   // Convert "x - (y+c0)" into "(x-y) - c0"
 326   // Need the same check as in above optimization but reversed.
 327   if (op2 == Op_AddL && ok_to_convert(in2, in1)) {
 328     Node* in21 = in2->in(1);
 329     Node* in22 = in2->in(2);
 330     const TypeLong* tcon = phase->type(in22)->isa_long();
 331     if (tcon != NULL && tcon->is_con()) {
 332       Node* sub2 = phase->transform( new SubLNode(in1, in21) );
 333       Node* neg_c0 = phase->longcon(- tcon->get_con());
 334       return new AddLNode(sub2, neg_c0);
 335     }
 336   }
 337 
 338   const Type *t1 = phase->type( in1 );
 339   if( t1 == Type::TOP ) return NULL;
 340 
 341 #ifdef ASSERT
 342   // Check for dead loop
 343   if( ( op2 == Op_AddL || op2 == Op_SubL ) &&
 344       ( phase->eqv( in2->in(1), this ) || phase->eqv( in2->in(2), this ) ||
 345         phase->eqv( in2->in(1), in2  ) || phase->eqv( in2->in(2), in2  ) ) )
 346     assert(false, "dead loop in SubLNode::Ideal");
 347 #endif
 348 
 349   // Convert "x - (x+y)" into "-y"
 350   if( op2 == Op_AddL &&
 351       phase->eqv( in1, in2->in(1) ) )
 352     return new SubLNode( phase->makecon(TypeLong::ZERO), in2->in(2));
 353   // Convert "x - (y+x)" into "-y"
 354   if( op2 == Op_AddL &&
 355       phase->eqv( in1, in2->in(2) ) )
 356     return new SubLNode( phase->makecon(TypeLong::ZERO),in2->in(1));
 357 
 358   // Convert "0 - (x-y)" into "y-x"
 359   if( phase->type( in1 ) == TypeLong::ZERO && op2 == Op_SubL )
 360     return new SubLNode( in2->in(2), in2->in(1) );
 361 
 362   // Convert "(X+A) - (X+B)" into "A - B"
 363   if( op1 == Op_AddL && op2 == Op_AddL && in1->in(1) == in2->in(1) )
 364     return new SubLNode( in1->in(2), in2->in(2) );
 365 
 366   // Convert "(A+X) - (B+X)" into "A - B"
 367   if( op1 == Op_AddL && op2 == Op_AddL && in1->in(2) == in2->in(2) )
 368     return new SubLNode( in1->in(1), in2->in(1) );
 369 
 370   // Convert "A-(B-C)" into (A+C)-B"
 371   if( op2 == Op_SubL && in2->outcnt() == 1) {
 372     Node *add1 = phase->transform( new AddLNode( in1, in2->in(2) ) );
 373     return new SubLNode( add1, in2->in(1) );
 374   }
 375 
 376   // Convert "0L-(A>>63)" into "(A>>>63)"
 377   if ( op2 == Op_RShiftL ) {
 378     Node *in21 = in2->in(1);
 379     Node *in22 = in2->in(2);
 380     const TypeLong *zero = phase->type(in1)->isa_long();
 381     const TypeLong *t21 = phase->type(in21)->isa_long();
 382     const TypeInt *t22 = phase->type(in22)->isa_int();
 383     if ( t21 && t22 && zero == TypeLong::ZERO && t22->is_con(63) ) {
 384       return new URShiftLNode(in21, in22);
 385     }
 386   }
 387 
 388   return NULL;
 389 }

 411 //------------------------------Value------------------------------------------
 412 // A subtract node differences its two inputs.
 413 const Type* SubFPNode::Value(PhaseGVN* phase) const {
 414   const Node* in1 = in(1);
 415   const Node* in2 = in(2);
 416   // Either input is TOP ==> the result is TOP
 417   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
 418   if( t1 == Type::TOP ) return Type::TOP;
 419   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
 420   if( t2 == Type::TOP ) return Type::TOP;
 421 
 422   // if both operands are infinity of same sign, the result is NaN; do
 423   // not replace with zero
 424   if( (t1->is_finite() && t2->is_finite()) ) {
 425     if( phase->eqv(in1, in2) ) return add_id();
 426   }
 427 
 428   // Either input is BOTTOM ==> the result is the local BOTTOM
 429   const Type *bot = bottom_type();
 430   if( (t1 == bot) || (t2 == bot) ||
 431       (t1 == Type::BOTTOM) || (t2 == Type::BOTTOM) )
 432     return bot;
 433 
 434   return sub(t1,t2);            // Local flavor of type subtraction
 435 }

 463 //------------------------------sub--------------------------------------------
 464 // A subtract node differences its two inputs.
 465 const Type *SubFNode::sub( const Type *t1, const Type *t2 ) const {
 466   // no folding if one of operands is infinity or NaN, do not do constant folding
 467   if( g_isfinite(t1->getf()) && g_isfinite(t2->getf()) ) {
 468     return TypeF::make( t1->getf() - t2->getf() );
 469   }
 470   else if( g_isnan(t1->getf()) ) {
 471     return t1;
 472   }
 473   else if( g_isnan(t2->getf()) ) {
 474     return t2;
 475   }
 476   else {
 477     return Type::FLOAT;
 478   }
 479 }

 566 // Simplify a CmpI (compare 2 integers) node, based on local information.
 567 // If both inputs are constants, compare them.
 568 const Type *CmpINode::sub( const Type *t1, const Type *t2 ) const {
 569   const TypeInt *r0 = t1->is_int(); // Handy access
 570   const TypeInt *r1 = t2->is_int();
 571 
 572   if( r0->_hi < r1->_lo )       // Range is always low?
 573     return TypeInt::CC_LT;
 574   else if( r0->_lo > r1->_hi )  // Range is always high?
 575     return TypeInt::CC_GT;
 576 
 577   else if( r0->is_con() && r1->is_con() ) { // comparing constants?
 578     assert(r0->get_con() == r1->get_con(), "must be equal");
 579     return TypeInt::CC_EQ;      // Equal results.
 580   } else if( r0->_hi == r1->_lo ) // Range is never high?
 581     return TypeInt::CC_LE;
 582   else if( r0->_lo == r1->_hi ) // Range is never low?
 583     return TypeInt::CC_GE;
 584   return TypeInt::CC;           // else use worst case results
 585 }

 587 // Simplify a CmpU (compare 2 integers) node, based on local information.
 588 // If both inputs are constants, compare them.
 589 const Type *CmpUNode::sub( const Type *t1, const Type *t2 ) const {
 590   assert(!t1->isa_ptr(), "obsolete usage of CmpU");
 591 
 592   // comparing two unsigned ints
 593   const TypeInt *r0 = t1->is_int();   // Handy access
 594   const TypeInt *r1 = t2->is_int();
 595 
 596   // Current installed version
 597   // Compare ranges for non-overlap
 598   juint lo0 = r0->_lo;
 599   juint hi0 = r0->_hi;
 600   juint lo1 = r1->_lo;
 601   juint hi1 = r1->_hi;
 602 
 603   // If either one has both negative and positive values,
 604   // it therefore contains both 0 and -1, and since [0..-1] is the
 605   // full unsigned range, the type must act as an unsigned bottom.
 606   bool bot0 = ((jint)(lo0 ^ hi0) < 0);
 607   bool bot1 = ((jint)(lo1 ^ hi1) < 0);
 608 
 609   if (bot0 || bot1) {
 610     // All unsigned values are LE -1 and GE 0.
 611     if (lo0 == 0 && hi0 == 0) {
 612       return TypeInt::CC_LE;            //   0 <= bot
 613     } else if ((jint)lo0 == -1 && (jint)hi0 == -1) {
 614       return TypeInt::CC_GE;            // -1 >= bot
 615     } else if (lo1 == 0 && hi1 == 0) {
 616       return TypeInt::CC_GE;            // bot >= 0
 617     } else if ((jint)lo1 == -1 && (jint)hi1 == -1) {
 618       return TypeInt::CC_LE;            // bot <= -1
 619     }
 620   } else {
 621     // We can use ranges of the form [lo..hi] if signs are the same.
 622     assert(lo0 <= hi0 && lo1 <= hi1, "unsigned ranges are valid");
 623     // results are reversed, '-' > '+' for unsigned compare
 624     if (hi0 < lo1) {
 625       return TypeInt::CC_LT;            // smaller
 626     } else if (lo0 > hi1) {
 627       return TypeInt::CC_GT;            // greater
 628     } else if (hi0 == lo1 && lo0 == hi1) {
 629       return TypeInt::CC_EQ;            // Equal results
 630     } else if (lo0 >= hi1) {
 631       return TypeInt::CC_GE;
 632     } else if (hi0 <= lo1) {
 633       // Check for special case in Hashtable::get.  (See below.)
 634       if ((jint)lo0 >= 0 && (jint)lo1 >= 0 && is_index_range_check())
 635         return TypeInt::CC_LT;
 636       return TypeInt::CC_LE;
 637     }
 638   }
 639   // Check for special case in Hashtable::get - the hash index is
 640   // mod'ed to the table size so the following range check is useless.
 641   // Check for: (X Mod Y) CmpU Y, where the mod result and Y both have
 642   // to be positive.
 643   // (This is a gross hack, since the sub method never
 644   // looks at the structure of the node in any other case.)
 645   if ((jint)lo0 >= 0 && (jint)lo1 >= 0 && is_index_range_check())
 646     return TypeInt::CC_LT;
 647   return TypeInt::CC;                   // else use worst case results
 648 }

 650 const Type* CmpUNode::Value(PhaseGVN* phase) const {
 651   const Type* t = SubNode::Value_common(phase);
 652   if (t != NULL) {
 653     return t;
 654   }
 655   const Node* in1 = in(1);
 656   const Node* in2 = in(2);
 657   const Type* t1 = phase->type(in1);
 658   const Type* t2 = phase->type(in2);
 659   assert(t1->isa_int(), "CmpU has only Int type inputs");
 660   if (t2 == TypeInt::INT) { // Compare to bottom?
 661     return bottom_type();
 662   }
 663   uint in1_op = in1->Opcode();
 664   if (in1_op == Op_AddI || in1_op == Op_SubI) {
 665     // The problem rise when result of AddI(SubI) may overflow
 666     // signed integer value. Let say the input type is
 667     // [256, maxint] then +128 will create 2 ranges due to
 668     // overflow: [minint, minint+127] and [384, maxint].
 669     // But C2 type system keep only 1 type range and as result
 670     // it use general [minint, maxint] for this case which we
 671     // can't optimize.
 672     //
 673     // Make 2 separate type ranges based on types of AddI(SubI) inputs
 674     // and compare results of their compare. If results are the same
 675     // CmpU node can be optimized.
 676     const Node* in11 = in1->in(1);
 677     const Node* in12 = in1->in(2);
 678     const Type* t11 = (in11 == in1) ? Type::TOP : phase->type(in11);
 679     const Type* t12 = (in12 == in1) ? Type::TOP : phase->type(in12);
 680     // Skip cases when input types are top or bottom.
 681     if ((t11 != Type::TOP) && (t11 != TypeInt::INT) &&
 682         (t12 != Type::TOP) && (t12 != TypeInt::INT)) {
 683       const TypeInt *r0 = t11->is_int();
 684       const TypeInt *r1 = t12->is_int();
 685       jlong lo_r0 = r0->_lo;
 686       jlong hi_r0 = r0->_hi;
 687       jlong lo_r1 = r1->_lo;
 688       jlong hi_r1 = r1->_hi;
 689       if (in1_op == Op_SubI) {
 690         jlong tmp = hi_r1;
 691         hi_r1 = -lo_r1;
 692         lo_r1 = -tmp;
 693         // Note, for substructing [minint,x] type range
 694         // long arithmetic provides correct overflow answer.
 695         // The confusion come from the fact that in 32-bit
 696         // -minint == minint but in 64-bit -minint == maxint+1.
 697       }
 698       jlong lo_long = lo_r0 + lo_r1;
 699       jlong hi_long = hi_r0 + hi_r1;
 700       int lo_tr1 = min_jint;
 701       int hi_tr1 = (int)hi_long;
 702       int lo_tr2 = (int)lo_long;
 703       int hi_tr2 = max_jint;
 704       bool underflow = lo_long != (jlong)lo_tr2;
 705       bool overflow  = hi_long != (jlong)hi_tr1;
 706       // Use sub(t1, t2) when there is no overflow (one type range)
 707       // or when both overflow and underflow (too complex).
 708       if ((underflow != overflow) && (hi_tr1 < lo_tr2)) {
 709         // Overflow only on one boundary, compare 2 separate type ranges.
 710         int w = MAX2(r0->_widen, r1->_widen); // _widen does not matter here
 711         const TypeInt* tr1 = TypeInt::make(lo_tr1, hi_tr1, w);
 712         const TypeInt* tr2 = TypeInt::make(lo_tr2, hi_tr2, w);
 713         const Type* cmp1 = sub(tr1, t2);
 714         const Type* cmp2 = sub(tr2, t2);
 715         if (cmp1 == cmp2) {
 716           return cmp1; // Hit!
 717         }
 718       }
 719     }
 720   }
 721 
 722   return sub(t1, t2);            // Local flavor of type subtraction
 723 }

 732 Node *CmpINode::Ideal( PhaseGVN *phase, bool can_reshape ) {
 733   if (phase->type(in(2))->higher_equal(TypeInt::ZERO)) {
 734     switch (in(1)->Opcode()) {
 735     case Op_CmpL3:              // Collapse a CmpL3/CmpI into a CmpL
 736       return new CmpLNode(in(1)->in(1),in(1)->in(2));
 737     case Op_CmpF3:              // Collapse a CmpF3/CmpI into a CmpF

 740       return new CmpDNode(in(1)->in(1),in(1)->in(2));
 741     //case Op_SubI:
 742       // If (x - y) cannot overflow, then ((x - y) <?> 0)
 743       // can be turned into (x <?> y).
 744       // This is handled (with more general cases) by Ideal_sub_algebra.
 745     }
 746   }
 747   return NULL;                  // No change

 863 //------------------------------sub--------------------------------------------
 864 // Simplify an CmpP (compare 2 pointers) node, based on local information.
 865 // If both inputs are constants, compare them.
 866 const Type *CmpPNode::sub( const Type *t1, const Type *t2 ) const {
 867   const TypePtr *r0 = t1->is_ptr(); // Handy access
 868   const TypePtr *r1 = t2->is_ptr();
 869 
 870   // Undefined inputs makes for an undefined result
 871   if( TypePtr::above_centerline(r0->_ptr) ||
 872       TypePtr::above_centerline(r1->_ptr) )
 873     return Type::TOP;
 874 
 875   if (r0 == r1 && r0->singleton()) {
 876     // Equal pointer constants (klasses, nulls, etc.)
 877     return TypeInt::CC_EQ;
 878   }
 879 
 880   // See if it is 2 unrelated classes.
 881   const TypeOopPtr* oop_p0 = r0->isa_oopptr();
 882   const TypeOopPtr* oop_p1 = r1->isa_oopptr();
 883   bool both_oop_ptr = oop_p0 && oop_p1;
 884 
 885   if (both_oop_ptr) {
 886     Node* in1 = in(1)->uncast();
 887     Node* in2 = in(2)->uncast();
 888     AllocateNode* alloc1 = AllocateNode::Ideal_allocation(in1, NULL);
 889     AllocateNode* alloc2 = AllocateNode::Ideal_allocation(in2, NULL);
 890     if (MemNode::detect_ptr_independence(in1, alloc1, in2, alloc2, NULL)) {
 891       return TypeInt::CC_GT;  // different pointers
 892     }
 893   }
 894 
 895   const TypeKlassPtr* klass_p0 = r0->isa_klassptr();
 896   const TypeKlassPtr* klass_p1 = r1->isa_klassptr();
 897 
 898   if (both_oop_ptr || (klass_p0 && klass_p1)) { // both or neither are klass pointers
 899     ciKlass* klass0 = NULL;
 900     bool    xklass0 = false;
 901     ciKlass* klass1 = NULL;
 902     bool    xklass1 = false;
 903 
 904     if (oop_p0) {
 905       klass0 = oop_p0->klass();
 906       xklass0 = oop_p0->klass_is_exact();
 907     } else {
 908       assert(klass_p0, "must be non-null if oop_p0 is null");
 909       klass0 = klass_p0->klass();
 910       xklass0 = klass_p0->klass_is_exact();
 911     }
 912 
 913     if (oop_p1) {
 914       klass1 = oop_p1->klass();
 915       xklass1 = oop_p1->klass_is_exact();
 916     } else {
 917       assert(klass_p1, "must be non-null if oop_p1 is null");
 918       klass1 = klass_p1->klass();
 919       xklass1 = klass_p1->klass_is_exact();
 920     }
 921 
 922     if (klass0 && klass1 &&
 923         klass0->is_loaded() && !klass0->is_interface() && // do not trust interfaces
 924         klass1->is_loaded() && !klass1->is_interface() &&
 925         (!klass0->is_obj_array_klass() ||
 926          !klass0->as_obj_array_klass()->base_element_klass()->is_interface()) &&
 927         (!klass1->is_obj_array_klass() ||
 928          !klass1->as_obj_array_klass()->base_element_klass()->is_interface())) {
 929       bool unrelated_classes = false;
 930       // See if neither subclasses the other, or if the class on top
 931       // is precise.  In either of these cases, the compare is known
 932       // to fail if at least one of the pointers is provably not null.
 933       if (klass0->equals(klass1)) {  // if types are unequal but klasses are equal
 934         // Do nothing; we know nothing for imprecise types
 935       } else if (klass0->is_subtype_of(klass1)) {
 936         // If klass1's type is PRECISE, then classes are unrelated.
 937         unrelated_classes = xklass1;
 938       } else if (klass1->is_subtype_of(klass0)) {
 939         // If klass0's type is PRECISE, then classes are unrelated.
 940         unrelated_classes = xklass0;
 941       } else {                  // Neither subtypes the other
 942         unrelated_classes = true;
 943       }
 944       if ((r0->flat_array() && (!r1->can_be_inline_type() || (klass1->is_inlinetype() && !klass1->flatten_array()))) ||
 945           (r1->flat_array() && (!r0->can_be_inline_type() || (klass0->is_inlinetype() && !klass0->flatten_array())))) {
 946         // One type is flattened in arrays and the other type is not. Must be unrelated.
 947         unrelated_classes = true;
 948       }
 949       if (unrelated_classes) {
 950         // The oops classes are known to be unrelated. If the joined PTRs of
 951         // two oops is not Null and not Bottom, then we are sure that one
 952         // of the two oops is non-null, and the comparison will always fail.
 953         TypePtr::PTR jp = r0->join_ptr(r1->_ptr);
 954         if (jp != TypePtr::Null && jp != TypePtr::BotPTR) {
 955           return TypeInt::CC_GT;
 956         }
 957       }
 958     }
 959   }
 960 
 961   // Known constants can be compared exactly
 962   // Null can be distinguished from any NotNull pointers
 963   // Unknown inputs makes an unknown result
 964   if( r0->singleton() ) {
 965     intptr_t bits0 = r0->get_con();
 966     if( r1->singleton() )
 967       return bits0 == r1->get_con() ? TypeInt::CC_EQ : TypeInt::CC_GT;
 968     return ( r1->_ptr == TypePtr::NotNull && bits0==0 ) ? TypeInt::CC_GT : TypeInt::CC;
 969   } else if( r1->singleton() ) {
 970     intptr_t bits1 = r1->get_con();
 971     return ( r0->_ptr == TypePtr::NotNull && bits1==0 ) ? TypeInt::CC_GT : TypeInt::CC;
 972   } else
 973     return TypeInt::CC;
 974 }

 976 static inline Node* isa_java_mirror_load(PhaseGVN* phase, Node* n) {
 977   // Return the klass node for (indirect load from OopHandle)
 978   //   LoadBarrier?(LoadP(LoadP(AddP(foo:Klass, #java_mirror))))
 979   //   or NULL if not matching.
 980   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 981     n = bs->step_over_gc_barrier(n);
 982 
 983   if (n->Opcode() != Op_LoadP) return NULL;
 984 
 985   const TypeInstPtr* tp = phase->type(n)->isa_instptr();
 986   if (!tp || tp->klass() != phase->C->env()->Class_klass()) return NULL;
 987 
 988   Node* adr = n->in(MemNode::Address);
 989   // First load from OopHandle: ((OopHandle)mirror)->resolve(); may need barrier.
 990   if (adr->Opcode() != Op_LoadP || !phase->type(adr)->isa_rawptr()) return NULL;
 991   adr = adr->in(MemNode::Address);
 992 
 993   intptr_t off = 0;
 994   Node* k = AddPNode::Ideal_base_and_offset(adr, phase, off);
 995   if (k == NULL)  return NULL;
 996   const TypeKlassPtr* tkp = phase->type(k)->isa_klassptr();
 997   if (!tkp || off != in_bytes(Klass::java_mirror_offset())) return NULL;
 998 
 999   // We've found the klass node of a Java mirror load.
1000   return k;
1001 }

1003 static inline Node* isa_const_java_mirror(PhaseGVN* phase, Node* n) {
1004   // for ConP(Foo.class) return ConP(Foo.klass)
1005   // otherwise return NULL
1006   if (!n->is_Con()) return NULL;
1007 
1008   const TypeInstPtr* tp = phase->type(n)->isa_instptr();
1009   if (!tp) return NULL;
1010 
1011   ciType* mirror_type = tp->java_mirror_type();
1012   // TypeInstPtr::java_mirror_type() returns non-NULL for compile-
1013   // time Class constants only.
1014   if (!mirror_type) return NULL;
1015 
1016   // x.getClass() == int.class can never be true (for all primitive types)
1017   // Return a ConP(NULL) node for this case.
1018   if (mirror_type->is_classless()) {
1019     return phase->makecon(TypePtr::NULL_PTR);
1020   }

1022   // return the ConP(Foo.klass)

1027 //------------------------------Ideal------------------------------------------
1028 // Normalize comparisons between Java mirror loads to compare the klass instead.
1029 //
1030 // Also check for the case of comparing an unknown klass loaded from the primary
1031 // super-type array vs a known klass with no subtypes.  This amounts to
1032 // checking to see an unknown klass subtypes a known klass with no subtypes;
1033 // this only happens on an exact match.  We can shorten this test by 1 load.
1034 Node* CmpPNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1035   // Normalize comparisons between Java mirrors into comparisons of the low-
1036   // level klass, where a dependent load could be shortened.
1037   //
1038   // The new pattern has a nice effect of matching the same pattern used in the
1039   // fast path of instanceof/checkcast/Class.isInstance(), which allows
1040   // redundant exact type check be optimized away by GVN.
1041   // For example, in
1042   //   if (x.getClass() == Foo.class) {
1043   //     Foo foo = (Foo) x;
1044   //     // ... use a ...
1045   //   }
1046   // a CmpPNode could be shared between if_acmpne and checkcast
1047   {
1048     Node* k1 = isa_java_mirror_load(phase, in(1));
1049     Node* k2 = isa_java_mirror_load(phase, in(2));
1050     Node* conk2 = isa_const_java_mirror(phase, in(2));
1051 
1052     if (k1 && (k2 || conk2)) {
1053       Node* lhs = k1;
1054       Node* rhs = (k2 != NULL) ? k2 : conk2;
1055       PhaseIterGVN* igvn = phase->is_IterGVN();
1056       if (igvn != NULL) {
1057         set_req_X(1, lhs, igvn);
1058         set_req_X(2, rhs, igvn);
1059       } else {
1060         set_req(1, lhs);
1061         set_req(2, rhs);
1062       }
1063       return this;
1064     }
1065   }
1066 
1067   // Constant pointer on right?
1068   const TypeKlassPtr* t2 = phase->type(in(2))->isa_klassptr();
1069   if (t2 == NULL || !t2->klass_is_exact())
1070     return NULL;
1071   // Get the constant klass we are comparing to.
1072   ciKlass* superklass = t2->klass();
1073 
1074   // Now check for LoadKlass on left.
1075   Node* ldk1 = in(1);
1076   if (ldk1->is_DecodeNKlass()) {
1077     ldk1 = ldk1->in(1);
1078     if (ldk1->Opcode() != Op_LoadNKlass )
1079       return NULL;
1080   } else if (ldk1->Opcode() != Op_LoadKlass )
1081     return NULL;
1082   // Take apart the address of the LoadKlass:
1083   Node* adr1 = ldk1->in(MemNode::Address);
1084   intptr_t con2 = 0;
1085   Node* ldk2 = AddPNode::Ideal_base_and_offset(adr1, phase, con2);
1086   if (ldk2 == NULL)
1087     return NULL;
1088   if (con2 == oopDesc::klass_offset_in_bytes()) {
1089     // We are inspecting an object's concrete class.
1090     // Short-circuit the check if the query is abstract.
1091     if (superklass->is_interface() ||
1092         superklass->is_abstract()) {
1093       // Make it come out always false:
1094       this->set_req(2, phase->makecon(TypePtr::NULL_PTR));
1095       return this;
1096     }
1097   }
1098 
1099   // Check for a LoadKlass from primary supertype array.
1100   // Any nested loadklass from loadklass+con must be from the p.s. array.
1101   if (ldk2->is_DecodeNKlass()) {
1102     // Keep ldk2 as DecodeN since it could be used in CmpP below.
1103     if (ldk2->in(1)->Opcode() != Op_LoadNKlass )
1104       return NULL;
1105   } else if (ldk2->Opcode() != Op_LoadKlass)
1106     return NULL;
1107 
1108   // Verify that we understand the situation
1109   if (con2 != (intptr_t) superklass->super_check_offset())
1110     return NULL;                // Might be element-klass loading from array klass
1111 
1112   // If 'superklass' has no subklasses and is not an interface, then we are
1113   // assured that the only input which will pass the type check is
1114   // 'superklass' itself.
1115   //
1116   // We could be more liberal here, and allow the optimization on interfaces
1117   // which have a single implementor.  This would require us to increase the
1118   // expressiveness of the add_dependency() mechanism.
1119   // %%% Do this after we fix TypeOopPtr:  Deps are expressive enough now.
1120 
1121   // Object arrays must have their base element have no subtypes
1122   while (superklass->is_obj_array_klass()) {
1123     ciType* elem = superklass->as_obj_array_klass()->element_type();
1124     superklass = elem->as_klass();
1125   }
1126   if (superklass->is_instance_klass()) {
1127     ciInstanceKlass* ik = superklass->as_instance_klass();
1128     if (ik->has_subklass() || ik->is_interface())  return NULL;
1129     // Add a dependency if there is a chance that a subclass will be added later.
1130     if (!ik->is_final()) {
1131       phase->C->dependencies()->assert_leaf_type(ik);
1132     }
1133   }
1134 
1135   // Bypass the dependent load, and compare directly
1136   this->set_req(1,ldk2);
1137 
1138   return this;
1139 }

1157 // Simplify an CmpF (compare 2 floats ) node, based on local information.
1158 // If both inputs are constants, compare them.
1159 const Type* CmpFNode::Value(PhaseGVN* phase) const {
1160   const Node* in1 = in(1);
1161   const Node* in2 = in(2);
1162   // Either input is TOP ==> the result is TOP
1163   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
1164   if( t1 == Type::TOP ) return Type::TOP;
1165   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
1166   if( t2 == Type::TOP ) return Type::TOP;
1167 
1168   // Not constants?  Don't know squat - even if they are the same
1169   // value!  If they are NaN's they compare to LT instead of EQ.
1170   const TypeF *tf1 = t1->isa_float_constant();
1171   const TypeF *tf2 = t2->isa_float_constant();
1172   if( !tf1 || !tf2 ) return TypeInt::CC;
1173 
1174   // This implements the Java bytecode fcmpl, so unordered returns -1.
1175   if( tf1->is_nan() || tf2->is_nan() )
1176     return TypeInt::CC_LT;
1177 
1178   if( tf1->_f < tf2->_f ) return TypeInt::CC_LT;
1179   if( tf1->_f > tf2->_f ) return TypeInt::CC_GT;
1180   assert( tf1->_f == tf2->_f, "do not understand FP behavior" );
1181   return TypeInt::CC_EQ;
1182 }

1214 //------------------------------Ideal------------------------------------------
1215 Node *CmpDNode::Ideal(PhaseGVN *phase, bool can_reshape){
1216   // Check if we can change this to a CmpF and remove a ConvD2F operation.
1217   // Change  (CMPD (F2D (float)) (ConD value))
1218   // To      (CMPF      (float)  (ConF value))
1219   // Valid when 'value' does not lose precision as a float.
1220   // Benefits: eliminates conversion, does not require 24-bit mode
1221 
1222   // NaNs prevent commuting operands.  This transform works regardless of the
1223   // order of ConD and ConvF2D inputs by preserving the original order.
1224   int idx_f2d = 1;              // ConvF2D on left side?
1225   if( in(idx_f2d)->Opcode() != Op_ConvF2D )
1226     idx_f2d = 2;                // No, swap to check for reversed args
1227   int idx_con = 3-idx_f2d;      // Check for the constant on other input
1228 
1229   if( ConvertCmpD2CmpF &&
1230       in(idx_f2d)->Opcode() == Op_ConvF2D &&
1231       in(idx_con)->Opcode() == Op_ConD ) {
1232     const TypeD *t2 = in(idx_con)->bottom_type()->is_double_constant();
1233     double t2_value_as_double = t2->_d;
1234     float  t2_value_as_float  = (float)t2_value_as_double;
1235     if( t2_value_as_double == (double)t2_value_as_float ) {
1236       // Test value can be represented as a float
1237       // Eliminate the conversion to double and create new comparison
1238       Node *new_in1 = in(idx_f2d)->in(1);
1239       Node *new_in2 = phase->makecon( TypeF::make(t2_value_as_float) );
1240       if( idx_f2d != 1 ) {      // Must flip args to match original order
1241         Node *tmp = new_in1;
1242         new_in1 = new_in2;
1243         new_in2 = tmp;
1244       }
1245       CmpFNode *new_cmp = (Opcode() == Op_CmpD3)
1246         ? new CmpF3Node( new_in1, new_in2 )
1247         : new CmpFNode ( new_in1, new_in2 ) ;
1248       return new_cmp;           // Changed to CmpFNode
1249     }
1250     // Testing value required the precision of a double
1251   }
1252   return NULL;                  // No change
1253 }

1257 //------------------------------cc2logical-------------------------------------
1258 // Convert a condition code type to a logical type
1259 const Type *BoolTest::cc2logical( const Type *CC ) const {
1260   if( CC == Type::TOP ) return Type::TOP;
1261   if( CC->base() != Type::Int ) return TypeInt::BOOL; // Bottom or worse
1262   const TypeInt *ti = CC->is_int();
1263   if( ti->is_con() ) {          // Only 1 kind of condition codes set?
1264     // Match low order 2 bits
1265     int tmp = ((ti->get_con()&3) == (_test&3)) ? 1 : 0;
1266     if( _test & 4 ) tmp = 1-tmp;     // Optionally complement result
1267     return TypeInt::make(tmp);       // Boolean result
1268   }
1269 
1270   if( CC == TypeInt::CC_GE ) {
1271     if( _test == ge ) return TypeInt::ONE;
1272     if( _test == lt ) return TypeInt::ZERO;
1273   }
1274   if( CC == TypeInt::CC_LE ) {
1275     if( _test == le ) return TypeInt::ONE;
1276     if( _test == gt ) return TypeInt::ZERO;
1277   }
1278 
1279   return TypeInt::BOOL;
1280 }

1354 // Change "bool eq/ne (cmp (add/sub A B) C)" into false/true if add/sub
1355 // overflows and we can prove that C is not in the two resulting ranges.
1356 // This optimization is similar to the one performed by CmpUNode::Value().
1357 Node* BoolNode::fold_cmpI(PhaseGVN* phase, SubNode* cmp, Node* cmp1, int cmp_op,
1358                           int cmp1_op, const TypeInt* cmp2_type) {
1359   // Only optimize eq/ne integer comparison of add/sub
1360   if((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1361      (cmp_op == Op_CmpI) && (cmp1_op == Op_AddI || cmp1_op == Op_SubI)) {
1362     // Skip cases were inputs of add/sub are not integers or of bottom type
1363     const TypeInt* r0 = phase->type(cmp1->in(1))->isa_int();
1364     const TypeInt* r1 = phase->type(cmp1->in(2))->isa_int();
1365     if ((r0 != NULL) && (r0 != TypeInt::INT) &&
1366         (r1 != NULL) && (r1 != TypeInt::INT) &&
1367         (cmp2_type != TypeInt::INT)) {
1368       // Compute exact (long) type range of add/sub result
1369       jlong lo_long = r0->_lo;
1370       jlong hi_long = r0->_hi;
1371       if (cmp1_op == Op_AddI) {
1372         lo_long += r1->_lo;
1373         hi_long += r1->_hi;
1374       } else {
1375         lo_long -= r1->_hi;
1376         hi_long -= r1->_lo;
1377       }
1378       // Check for over-/underflow by casting to integer
1379       int lo_int = (int)lo_long;
1380       int hi_int = (int)hi_long;
1381       bool underflow = lo_long != (jlong)lo_int;
1382       bool overflow  = hi_long != (jlong)hi_int;
1383       if ((underflow != overflow) && (hi_int < lo_int)) {
1384         // Overflow on one boundary, compute resulting type ranges:
1385         // tr1 [MIN_INT, hi_int] and tr2 [lo_int, MAX_INT]
1386         int w = MAX2(r0->_widen, r1->_widen); // _widen does not matter here
1387         const TypeInt* tr1 = TypeInt::make(min_jint, hi_int, w);
1388         const TypeInt* tr2 = TypeInt::make(lo_int, max_jint, w);
1389         // Compare second input of cmp to both type ranges
1390         const Type* sub_tr1 = cmp->sub(tr1, cmp2_type);
1391         const Type* sub_tr2 = cmp->sub(tr2, cmp2_type);
1392         if (sub_tr1 == TypeInt::CC_LT && sub_tr2 == TypeInt::CC_GT) {
1393           // The result of the add/sub will never equal cmp2. Replace BoolNode
1394           // by false (0) if it tests for equality and by true (1) otherwise.
1395           return ConINode::make((_test._test == BoolTest::eq) ? 0 : 1);
1396         }
1397       }
1398     }
1399   }
1400   return NULL;
1401 }

1412 //------------------------------Ideal------------------------------------------
1413 Node *BoolNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1414   // Change "bool tst (cmp con x)" into "bool ~tst (cmp x con)".
1415   // This moves the constant to the right.  Helps value-numbering.
1416   Node *cmp = in(1);
1417   if( !cmp->is_Sub() ) return NULL;
1418   int cop = cmp->Opcode();
1419   if( cop == Op_FastLock || cop == Op_FastUnlock || cmp->is_SubTypeCheck()) return NULL;
1420   Node *cmp1 = cmp->in(1);
1421   Node *cmp2 = cmp->in(2);
1422   if( !cmp1 ) return NULL;
1423 
1424   if (_test._test == BoolTest::overflow || _test._test == BoolTest::no_overflow) {
1425     return NULL;
1426   }
1427 
1428   // Constant on left?
1429   Node *con = cmp1;
1430   uint op2 = cmp2->Opcode();
1431   // Move constants to the right of compare's to canonicalize.
1432   // Do not muck with Opaque1 nodes, as this indicates a loop
1433   // guard that cannot change shape.
1434   if( con->is_Con() && !cmp2->is_Con() && op2 != Op_Opaque1 &&
1435       // Because of NaN's, CmpD and CmpF are not commutative
1436       cop != Op_CmpD && cop != Op_CmpF &&
1437       // Protect against swapping inputs to a compare when it is used by a
1438       // counted loop exit, which requires maintaining the loop-limit as in(2)
1439       !is_counted_loop_exit_test() ) {
1440     // Ok, commute the constant to the right of the cmp node.
1441     // Clone the Node, getting a new Node of the same class
1442     cmp = cmp->clone();
1443     // Swap inputs to the clone
1444     cmp->swap_edges(1, 2);
1445     cmp = phase->transform( cmp );
1446     return new BoolNode( cmp, _test.commute() );
1447   }
1448 
1449   // Change "bool eq/ne (cmp (and X 16) 16)" into "bool ne/eq (cmp (and X 16) 0)".
1450   if (cop == Op_CmpI &&
1451       (_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1452       cmp1->Opcode() == Op_AndI && cmp2->Opcode() == Op_ConI &&
1453       cmp1->in(2)->Opcode() == Op_ConI) {
1454     const TypeInt *t12 = phase->type(cmp2)->isa_int();
1455     const TypeInt *t112 = phase->type(cmp1->in(2))->isa_int();
1456     if (t12 && t12->is_con() && t112 && t112->is_con() &&
1457         t12->get_con() == t112->get_con() && is_power_of_2(t12->get_con())) {
1458       Node *ncmp = phase->transform(new CmpINode(cmp1, phase->intcon(0)));
1459       return new BoolNode(ncmp, _test.negate());
1460     }
1461   }
1462 
1463   // Same for long type: change "bool eq/ne (cmp (and X 16) 16)" into "bool ne/eq (cmp (and X 16) 0)".
1464   if (cop == Op_CmpL &&
1465       (_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1466       cmp1->Opcode() == Op_AndL && cmp2->Opcode() == Op_ConL &&
1467       cmp1->in(2)->Opcode() == Op_ConL) {
1468     const TypeLong *t12 = phase->type(cmp2)->isa_long();
1469     const TypeLong *t112 = phase->type(cmp1->in(2))->isa_long();
1470     if (t12 && t12->is_con() && t112 && t112->is_con() &&
1471         t12->get_con() == t112->get_con() && is_power_of_2(t12->get_con())) {
1472       Node *ncmp = phase->transform(new CmpLNode(cmp1, phase->longcon(0)));
1473       return new BoolNode(ncmp, _test.negate());
1474     }
1475   }
1476 
1477   // Change "bool eq/ne (cmp (xor X 1) 0)" into "bool ne/eq (cmp X 0)".
1478   // The XOR-1 is an idiom used to flip the sense of a bool.  We flip the
1479   // test instead.
1480   int cmp1_op = cmp1->Opcode();
1481   const TypeInt* cmp2_type = phase->type(cmp2)->isa_int();
1482   if (cmp2_type == NULL)  return NULL;
1483   Node* j_xor = cmp1;
1484   if( cmp2_type == TypeInt::ZERO &&
1485       cmp1_op == Op_XorI &&
1486       j_xor->in(1) != j_xor &&          // An xor of itself is dead
1487       phase->type( j_xor->in(1) ) == TypeInt::BOOL &&
1488       phase->type( j_xor->in(2) ) == TypeInt::ONE &&
1489       (_test._test == BoolTest::eq ||
1490        _test._test == BoolTest::ne) ) {
1491     Node *ncmp = phase->transform(new CmpINode(j_xor->in(1),cmp2));
1492     return new BoolNode( ncmp, _test.negate() );
1493   }
1494 
1495   // Change ((x & m) u<= m) or ((m & x) u<= m) to always true
1496   // Same with ((x & m) u< m+1) and ((m & x) u< m+1)
1497   if (cop == Op_CmpU &&
1498       cmp1_op == Op_AndI) {
1499     Node* bound = NULL;
1500     if (_test._test == BoolTest::le) {
1501       bound = cmp2;
1502     } else if (_test._test == BoolTest::lt &&
1503                cmp2->Opcode() == Op_AddI &&
1504                cmp2->in(2)->find_int_con(0) == 1) {
1505       bound = cmp2->in(1);
1506     }
1507     if (cmp1->in(2) == bound || cmp1->in(1) == bound) {
1508       return ConINode::make(1);
1509     }
1510   }
1511 
1512   // Change ((x & (m - 1)) u< m) into (m > 0)
1513   // This is the off-by-one variant of the above
1514   if (cop == Op_CmpU &&
1515       _test._test == BoolTest::lt &&
1516       cmp1_op == Op_AndI) {
1517     Node* l = cmp1->in(1);
1518     Node* r = cmp1->in(2);
1519     for (int repeat = 0; repeat < 2; repeat++) {
1520       bool match = r->Opcode() == Op_AddI && r->in(2)->find_int_con(0) == -1 &&
1521                    r->in(1) == cmp2;
1522       if (match) {
1523         // arraylength known to be non-negative, so a (arraylength != 0) is sufficient,
1524         // but to be compatible with the array range check pattern, use (arraylength u> 0)
1525         Node* ncmp = cmp2->Opcode() == Op_LoadRange
1526                      ? phase->transform(new CmpUNode(cmp2, phase->intcon(0)))
1527                      : phase->transform(new CmpINode(cmp2, phase->intcon(0)));
1528         return new BoolNode(ncmp, BoolTest::gt);
1529       } else {
1530         // commute and try again
1531         l = cmp1->in(2);
1532         r = cmp1->in(1);
1533       }
1534     }
1535   }
1536 
1537   // Change x u< 1 or x u<= 0 to x == 0
1538   if (cop == Op_CmpU &&
1539       cmp1_op != Op_LoadRange &&
1540       ((_test._test == BoolTest::lt &&
1541         cmp2->find_int_con(-1) == 1) ||
1542        (_test._test == BoolTest::le &&
1543         cmp2->find_int_con(-1) == 0))) {
1544     Node* ncmp = phase->transform(new CmpINode(cmp1, phase->intcon(0)));
1545     return new BoolNode(ncmp, BoolTest::eq);
1546   }
1547 
1548   // Change (arraylength <= 0) or (arraylength == 0)
1549   //   into (arraylength u<= 0)
1550   // Also change (arraylength != 0) into (arraylength u> 0)
1551   // The latter version matches the code pattern generated for
1552   // array range checks, which will more likely be optimized later.
1553   if (cop == Op_CmpI &&
1554       cmp1_op == Op_LoadRange &&
1555       cmp2->find_int_con(-1) == 0) {
1556     if (_test._test == BoolTest::le || _test._test == BoolTest::eq) {
1557       Node* ncmp = phase->transform(new CmpUNode(cmp1, cmp2));
1558       return new BoolNode(ncmp, BoolTest::le);
1559     } else if (_test._test == BoolTest::ne) {
1560       Node* ncmp = phase->transform(new CmpUNode(cmp1, cmp2));
1561       return new BoolNode(ncmp, BoolTest::gt);
1562     }
1563   }
1564 
1565   // Change "bool eq/ne (cmp (Conv2B X) 0)" into "bool eq/ne (cmp X 0)".
1566   // This is a standard idiom for branching on a boolean value.
1567   Node *c2b = cmp1;
1568   if( cmp2_type == TypeInt::ZERO &&
1569       cmp1_op == Op_Conv2B &&
1570       (_test._test == BoolTest::eq ||
1571        _test._test == BoolTest::ne) ) {
1572     Node *ncmp = phase->transform(phase->type(c2b->in(1))->isa_int()
1573        ? (Node*)new CmpINode(c2b->in(1),cmp2)
1574        : (Node*)new CmpPNode(c2b->in(1),phase->makecon(TypePtr::NULL_PTR))
1575     );
1576     return new BoolNode( ncmp, _test._test );
1577   }
1578 
1579   // Comparing a SubI against a zero is equal to comparing the SubI
1580   // arguments directly.  This only works for eq and ne comparisons
1581   // due to possible integer overflow.
1582   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1583         (cop == Op_CmpI) &&
1584         (cmp1_op == Op_SubI) &&
1585         ( cmp2_type == TypeInt::ZERO ) ) {
1586     Node *ncmp = phase->transform( new CmpINode(cmp1->in(1),cmp1->in(2)));
1587     return new BoolNode( ncmp, _test._test );
1588   }
1589 
1590   // Same as above but with and AddI of a constant
1591   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1592       cop == Op_CmpI &&
1593       cmp1_op == Op_AddI &&
1594       cmp1->in(2) != NULL &&
1595       phase->type(cmp1->in(2))->isa_int() &&
1596       phase->type(cmp1->in(2))->is_int()->is_con() &&
1597       cmp2_type == TypeInt::ZERO &&
1598       !is_counted_loop_cmp(cmp) // modifying the exit test of a counted loop messes the counted loop shape
1599       ) {
1600     const TypeInt* cmp1_in2 = phase->type(cmp1->in(2))->is_int();
1601     Node *ncmp = phase->transform( new CmpINode(cmp1->in(1),phase->intcon(-cmp1_in2->_hi)));
1602     return new BoolNode( ncmp, _test._test );
1603   }
1604 
1605   // Change "bool eq/ne (cmp (phi (X -X) 0))" into "bool eq/ne (cmp X 0)"
1606   // since zero check of conditional negation of an integer is equal to
1607   // zero check of the integer directly.
1608   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1609       (cop == Op_CmpI) &&
1610       (cmp2_type == TypeInt::ZERO) &&
1611       (cmp1_op == Op_Phi)) {
1612     // There should be a diamond phi with true path at index 1 or 2
1613     PhiNode *phi = cmp1->as_Phi();
1614     int idx_true = phi->is_diamond_phi();
1615     if (idx_true != 0) {
1616       // True input is in(idx_true) while false input is in(3 - idx_true)
1617       Node *tin = phi->in(idx_true);
1618       Node *fin = phi->in(3 - idx_true);
1619       if ((tin->Opcode() == Op_SubI) &&
1620           (phase->type(tin->in(1)) == TypeInt::ZERO) &&
1621           (tin->in(2) == fin)) {
1622         // Found conditional negation at true path, create a new CmpINode without that
1623         Node *ncmp = phase->transform(new CmpINode(fin, cmp2));
1624         return new BoolNode(ncmp, _test._test);
1625       }
1626       if ((fin->Opcode() == Op_SubI) &&
1627           (phase->type(fin->in(1)) == TypeInt::ZERO) &&
1628           (fin->in(2) == tin)) {
1629         // Found conditional negation at false path, create a new CmpINode without that
1630         Node *ncmp = phase->transform(new CmpINode(tin, cmp2));
1631         return new BoolNode(ncmp, _test._test);
1632       }
1633     }
1634   }
1635 
1636   // Change (-A vs 0) into (A vs 0) by commuting the test.  Disallow in the
1637   // most general case because negating 0x80000000 does nothing.  Needed for
1638   // the CmpF3/SubI/CmpI idiom.
1639   if( cop == Op_CmpI &&
1640       cmp1_op == Op_SubI &&
1641       cmp2_type == TypeInt::ZERO &&
1642       phase->type( cmp1->in(1) ) == TypeInt::ZERO &&
1643       phase->type( cmp1->in(2) )->higher_equal(TypeInt::SYMINT) ) {
1644     Node *ncmp = phase->transform( new CmpINode(cmp1->in(2),cmp2));
1645     return new BoolNode( ncmp, _test.commute() );
1646   }
1647 
1648   // Try to optimize signed integer comparison
1649   return fold_cmpI(phase, cmp->as_Sub(), cmp1, cop, cmp1_op, cmp2_type);
1650 
1651   //  The transformation below is not valid for either signed or unsigned
1652   //  comparisons due to wraparound concerns at MAX_VALUE and MIN_VALUE.
1653   //  This transformation can be resurrected when we are able to
1654   //  make inferences about the range of values being subtracted from
1655   //  (or added to) relative to the wraparound point.
1656   //
1657   //    // Remove +/-1's if possible.
1658   //    // "X <= Y-1" becomes "X <  Y"
1659   //    // "X+1 <= Y" becomes "X <  Y"
1660   //    // "X <  Y+1" becomes "X <= Y"
1661   //    // "X-1 <  Y" becomes "X <= Y"
1662   //    // Do not this to compares off of the counted-loop-end.  These guys are
1663   //    // checking the trip counter and they want to use the post-incremented
1664   //    // counter.  If they use the PRE-incremented counter, then the counter has
1665   //    // to be incremented in a private block on a loop backedge.
1666   //    if( du && du->cnt(this) && du->out(this)[0]->Opcode() == Op_CountedLoopEnd )
1667   //      return NULL;
1668   //  #ifndef PRODUCT
1669   //    // Do not do this in a wash GVN pass during verification.
1670   //    // Gets triggered by too many simple optimizations to be bothered with
1671   //    // re-trying it again and again.
1672   //    if( !phase->allow_progress() ) return NULL;
