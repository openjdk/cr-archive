  74       return true;

  76   }

 106 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 107   st->print(" @");
 108   if (adr_type == NULL) {
 109     st->print("NULL");
 110   } else {
 111     adr_type->dump_on(st);
 112     Compile* C = Compile::current();
 113     Compile::AliasType* atp = NULL;
 114     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 115     if (atp == NULL)
 116       st->print(", idx=?\?;");
 117     else if (atp->index() == Compile::AliasIdxBot)
 118       st->print(", idx=Bot;");
 119     else if (atp->index() == Compile::AliasIdxTop)
 120       st->print(", idx=Top;");
 121     else if (atp->index() == Compile::AliasIdxRaw)
 122       st->print(", idx=Raw;");
 123     else {
 124       ciField* field = atp->field();
 125       if (field) {
 126         st->print(", name=");
 127         field->print_name_on(st);
 128       }
 129       st->print(", idx=%d;", atp->index());
 130     }
 131   }
 132 }

 138 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 139   assert((t_oop != NULL), "sanity");
 140   bool is_instance = t_oop->is_known_instance_field();
 141   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 142                              (load != NULL) && load->is_Load() &&
 143                              (phase->is_IterGVN() != NULL);
 144   if (!(is_instance || is_boxed_value_load))
 145     return mchain;  // don't try to optimize non-instance types
 146   uint instance_id = t_oop->instance_id();
 147   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 148   Node *prev = NULL;
 149   Node *result = mchain;
 150   while (prev != result) {
 151     prev = result;
 152     if (result == start_mem)
 153       break;  // hit one of our sentinels
 154     // skip over a call which does not affect this memory slice
 155     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 156       Node *proj_in = result->in(0);
 157       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 158         break;  // hit one of our sentinels
 159       } else if (proj_in->is_Call()) {
 160         // ArrayCopyNodes processed here as well
 161         CallNode *call = proj_in->as_Call();
 162         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 163           result = call->in(TypeFunc::Memory);
 164         }
 165       } else if (proj_in->is_Initialize()) {
 166         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 167         // Stop if this is the initialization for the object instance which
 168         // contains this memory slice, otherwise skip over it.
 169         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 170           break;
 171         }
 172         if (is_instance) {
 173           result = proj_in->in(TypeFunc::Memory);
 174         } else if (is_boxed_value_load) {
 175           Node* klass = alloc->in(AllocateNode::KlassNode);
 176           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 177           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 178             result = proj_in->in(TypeFunc::Memory); // not related allocation
 179           }
 180         }
 181       } else if (proj_in->is_MemBar()) {
 182         ArrayCopyNode* ac = NULL;
 183         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 184           break;
 185         }
 186         result = proj_in->in(TypeFunc::Memory);
 187       } else {
 188         assert(false, "unexpected projection");
 189       }
 190     } else if (result->is_ClearArray()) {
 191       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 192         // Can not bypass initialization of the instance
 193         // we are looking for.
 194         break;
 195       }
 196       // Otherwise skip it (the call updated 'result' value).
 197     } else if (result->is_MergeMem()) {
 198       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 199     }
 200   }
 201   return result;
 202 }

 204 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 205   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 206   if (t_oop == NULL)
 207     return mchain;  // don't try to optimize non-oop types
 208   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 209   bool is_instance = t_oop->is_known_instance_field();
 210   PhaseIterGVN *igvn = phase->is_IterGVN();
 211   if (is_instance && igvn != NULL && result->is_Phi()) {
 212     PhiNode *mphi = result->as_Phi();
 213     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 214     const TypePtr *t = mphi->adr_type();
 215     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 216         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 217          t->is_oopptr()->cast_to_exactness(true)
 218            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 219             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 220       // clone the Phi with our address type
 221       result = mphi->split_out_instance(t_adr, igvn);
 222     } else {
 223       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 224     }
 225   }
 226   return result;
 227 }

 229 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 230   uint alias_idx = phase->C->get_alias_index(tp);
 231   Node *mem = mmem;
 232 #ifdef ASSERT
 233   {
 234     // Check that current type is consistent with the alias index used during graph construction
 235     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 236     bool consistent =  adr_check == NULL || adr_check->empty() ||
 237                        phase->C->must_alias(adr_check, alias_idx );
 238     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 239     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 240                tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 241         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 242         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 243           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 244           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 245       // don't assert if it is dead code.
 246       consistent = true;
 247     }
 248     if( !consistent ) {
 249       st->print("alias_idx==%d, adr_check==", alias_idx);
 250       if( adr_check == NULL ) {
 251         st->print("NULL");
 252       } else {
 253         adr_check->dump();
 254       }
 255       st->cr();
 256       print_alias_types();
 257       assert(consistent, "adr_check must match alias idx");
 258     }
 259   }
 260 #endif
 261   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 262   // means an array I have not precisely typed yet.  Do not do any
 263   // alias stuff with it any time soon.
 264   const TypeOopPtr *toop = tp->isa_oopptr();
 265   if( tp->base() != Type::AnyPtr &&
 266       !(toop &&
 267         toop->klass() != NULL &&
 268         toop->klass()->is_java_lang_Object() &&
 269         toop->offset() == Type::OffsetBot) ) {
 270     // compress paths and change unreachable cycles to TOP
 271     // If not, we can update the input infinitely along a MergeMem cycle
 272     // Equivalent code in PhiNode::Ideal
 273     Node* m  = phase->transform(mmem);
 274     // If transformed to a MergeMem, get the desired slice
 275     // Otherwise the returned node represents memory for every slice
 276     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 277     // Update input if it is progress over what we have now
 278   }
 279   return mem;
 280 }

 282 //--------------------------Ideal_common---------------------------------------
 283 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 284 // Unhook non-raw memories from complete (macro-expanded) initializations.
 285 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 286   // If our control input is a dead region, kill all below the region
 287   Node *ctl = in(MemNode::Control);
 288   if (ctl && remove_dead_region(phase, can_reshape))
 289     return this;
 290   ctl = in(MemNode::Control);
 291   // Don't bother trying to transform a dead node
 292   if (ctl && ctl->is_top())  return NodeSentinel;
 293 
 294   PhaseIterGVN *igvn = phase->is_IterGVN();
 295   // Wait if control on the worklist.
 296   if (ctl && can_reshape && igvn != NULL) {
 297     Node* bol = NULL;
 298     Node* cmp = NULL;
 299     if (ctl->in(0)->is_If()) {
 300       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 301       bol = ctl->in(0)->in(1);
 302       if (bol->is_Bool())
 303         cmp = ctl->in(0)->in(1)->in(1);
 304     }
 305     if (igvn->_worklist.member(ctl) ||
 306         (bol != NULL && igvn->_worklist.member(bol)) ||
 307         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 308       // This control path may be dead.
 309       // Delay this memory node transformation until the control is processed.
 310       phase->is_IterGVN()->_worklist.push(this);
 311       return NodeSentinel; // caller will return NULL
 312     }
 313   }
 314   // Ignore if memory is dead, or self-loop
 315   Node *mem = in(MemNode::Memory);
 316   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 317   assert(mem != this, "dead loop in MemNode::Ideal");
 318 
 319   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 320     // This memory slice may be dead.
 321     // Delay this mem node transformation until the memory is processed.
 322     phase->is_IterGVN()->_worklist.push(this);
 323     return NodeSentinel; // caller will return NULL
 324   }
 325 
 326   Node *address = in(MemNode::Address);
 327   const Type *t_adr = phase->type(address);
 328   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 329 
 330   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 331     // Unsafe off-heap access with zero address. Remove access and other control users
 332     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 333     assert(ctl != NULL, "unsafe accesses should be control dependent");
 334     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 335       Node* u = ctl->fast_out(i);
 336       if (u != ctl) {
 337         igvn->rehash_node_delayed(u);
 338         int nb = u->replace_edge(ctl, phase->C->top());
 339         --i, imax -= nb;
 340       }
 341     }
 342     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 343     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 344     phase->C->root()->add_req(halt);
 345     return this;
 346   }
 347 
 348   if (can_reshape && igvn != NULL &&
 349       (igvn->_worklist.member(address) ||
 350        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 351     // The address's base and type may change when the address is processed.
 352     // Delay this mem node transformation until the address is processed.
 353     phase->is_IterGVN()->_worklist.push(this);
 354     return NodeSentinel; // caller will return NULL
 355   }
 356 
 357   // Do NOT remove or optimize the next lines: ensure a new alias index
 358   // is allocated for an oop pointer type before Escape Analysis.
 359   // Note: C++ will not remove it since the call has side effect.
 360   if (t_adr->isa_oopptr()) {
 361     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 362   }
 363 
 364   Node* base = NULL;
 365   if (address->is_AddP()) {
 366     base = address->in(AddPNode::Base);
 367   }
 368   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 369       !t_adr->isa_rawptr()) {
 370     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 371     // Skip this node optimization if its address has TOP base.
 372     return NodeSentinel; // caller will return NULL
 373   }
 374 
 375   // Avoid independent memory operations
 376   Node* old_mem = mem;
 377 
 378   // The code which unhooks non-raw memories from complete (macro-expanded)
 379   // initializations was removed. After macro-expansion all stores catched
 380   // by Initialize node became raw stores and there is no information
 381   // which memory slices they modify. So it is unsafe to move any memory
 382   // operation above these stores. Also in most cases hooked non-raw memories
 383   // were already unhooked by using information from detect_ptr_independence()
 384   // and find_previous_store().
 385 
 386   if (mem->is_MergeMem()) {
 387     MergeMemNode* mmem = mem->as_MergeMem();
 388     const TypePtr *tp = t_adr->is_ptr();
 389 
 390     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 391   }
 392 
 393   if (mem != old_mem) {
 394     set_req(MemNode::Memory, mem);
 395     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 396       igvn->_worklist.push(old_mem);
 397     }
 398     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 399     return this;
 400   }
 401 
 402   // let the subclass continue analyzing...
 403   return NULL;
 404 }

 406 // Helper function for proving some simple control dominations.
 407 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 408 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 409 // is not a constant (dominated by the method's StartNode).
 410 // Used by MemNode::find_previous_store to prove that the
 411 // control input of a memory operation predates (dominates)
 412 // an allocation it wants to look past.
 413 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 414   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 415     return false; // Conservative answer for dead code
 416 
 417   // Check 'dom'. Skip Proj and CatchProj nodes.
 418   dom = dom->find_exact_control(dom);
 419   if (dom == NULL || dom->is_top())
 420     return false; // Conservative answer for dead code
 421 
 422   if (dom == sub) {
 423     // For the case when, for example, 'sub' is Initialize and the original
 424     // 'dom' is Proj node of the 'sub'.
 425     return false;
 426   }
 427 
 428   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 429     return true;
 430 
 431   // 'dom' dominates 'sub' if its control edge and control edges
 432   // of all its inputs dominate or equal to sub's control edge.
 433 
 434   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 435   // Or Region for the check in LoadNode::Ideal();
 436   // 'sub' should have sub->in(0) != NULL.
 437   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 438          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 439 
 440   // Get control edge of 'sub'.
 441   Node* orig_sub = sub;
 442   sub = sub->find_exact_control(sub->in(0));
 443   if (sub == NULL || sub->is_top())
 444     return false; // Conservative answer for dead code
 445 
 446   assert(sub->is_CFG(), "expecting control");
 447 
 448   if (sub == dom)
 449     return true;
 450 
 451   if (sub->is_Start() || sub->is_Root())
 452     return false;
 453 
 454   {
 455     // Check all control edges of 'dom'.
 456 
 457     ResourceMark rm;
 458     Node_List nlist;
 459     Unique_Node_List dom_list;
 460 
 461     dom_list.push(dom);
 462     bool only_dominating_controls = false;
 463 
 464     for (uint next = 0; next < dom_list.size(); next++) {
 465       Node* n = dom_list.at(next);
 466       if (n == orig_sub)
 467         return false; // One of dom's inputs dominated by sub.
 468       if (!n->is_CFG() && n->pinned()) {
 469         // Check only own control edge for pinned non-control nodes.
 470         n = n->find_exact_control(n->in(0));
 471         if (n == NULL || n->is_top())
 472           return false; // Conservative answer for dead code
 473         assert(n->is_CFG(), "expecting control");
 474         dom_list.push(n);
 475       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 476         only_dominating_controls = true;
 477       } else if (n->is_CFG()) {
 478         if (n->dominates(sub, nlist))
 479           only_dominating_controls = true;
 480         else
 481           return false;
 482       } else {
 483         // First, own control edge.
 484         Node* m = n->find_exact_control(n->in(0));
 485         if (m != NULL) {
 486           if (m->is_top())
 487             return false; // Conservative answer for dead code
 488           dom_list.push(m);
 489         }
 490         // Now, the rest of edges.
 491         uint cnt = n->req();
 492         for (uint i = 1; i < cnt; i++) {
 493           m = n->find_exact_control(n->in(i));
 494           if (m == NULL || m->is_top())
 495             continue;
 496           dom_list.push(m);
 497         }
 498       }
 499     }
 500     return only_dominating_controls;
 501   }
 502 }

 504 //---------------------detect_ptr_independence---------------------------------
 505 // Used by MemNode::find_previous_store to prove that two base
 506 // pointers are never equal.
 507 // The pointers are accompanied by their associated allocations,
 508 // if any, which have been previously discovered by the caller.
 509 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 510                                       Node* p2, AllocateNode* a2,
 511                                       PhaseTransform* phase) {
 512   // Attempt to prove that these two pointers cannot be aliased.
 513   // They may both manifestly be allocations, and they should differ.
 514   // Or, if they are not both allocations, they can be distinct constants.
 515   // Otherwise, one is an allocation and the other a pre-existing value.
 516   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 517     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 518   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 519     return (a1 != a2);
 520   } else if (a1 != NULL) {                  // one allocation a1
 521     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 522     return all_controls_dominate(p2, a1);
 523   } else { //(a2 != NULL)                   // one allocation a2
 524     return all_controls_dominate(p1, a2);
 525   }
 526   return false;
 527 }

 530 // Find an arraycopy that must have set (can_see_stored_value=true) or
 531 // could have set (can_see_stored_value=false) the value for this load
 532 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 533   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 534                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 535     Node* mb = mem->in(0);
 536     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 537         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 538       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 539       if (ac->is_clonebasic()) {
 540         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 541         if (alloc != NULL && alloc == ld_alloc) {
 542           return ac;
 543         }
 544       }
 545     }
 546   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 547     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 548 
 549     if (ac->is_arraycopy_validated() ||
 550         ac->is_copyof_validated() ||
 551         ac->is_copyofrange_validated()) {
 552       Node* ld_addp = in(MemNode::Address);
 553       if (ld_addp->is_AddP()) {
 554         Node* ld_base = ld_addp->in(AddPNode::Address);
 555         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 556 
 557         Node* dest = ac->in(ArrayCopyNode::Dest);
 558 
 559         if (dest == ld_base) {
 560           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 561           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 562             return ac;
 563           }
 564           if (!can_see_stored_value) {
 565             mem = ac->in(TypeFunc::Memory);
 566           }
 567         }
 568       }
 569     }
 570   }
 571   return NULL;
 572 }

 574 // The logic for reordering loads and stores uses four steps:
 575 // (a) Walk carefully past stores and initializations which we
 576 //     can prove are independent of this load.
 577 // (b) Observe that the next memory state makes an exact match
 578 //     with self (load or store), and locate the relevant store.
 579 // (c) Ensure that, if we were to wire self directly to the store,
 580 //     the optimizer would fold it up somehow.
 581 // (d) Do the rewiring, and return, depending on some other part of
 582 //     the optimizer to fold up the load.
 583 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 584 // specific to loads and stores, so they are handled by the callers.
 585 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 586 //
 587 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 588   Node*         ctrl   = in(MemNode::Control);
 589   Node*         adr    = in(MemNode::Address);
 590   intptr_t      offset = 0;
 591   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 592   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 593 
 594   if (offset == Type::OffsetBot)
 595     return NULL;            // cannot unalias unless there are precise offsets
 596 
 597   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 598   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 599 
 600   intptr_t size_in_bytes = memory_size();
 601 
 602   Node* mem = in(MemNode::Memory);   // start searching here...
 603 
 604   int cnt = 50;             // Cycle limiter
 605   for (;;) {                // While we can dance past unrelated stores...
 606     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 607 
 608     Node* prev = mem;
 609     if (mem->is_Store()) {
 610       Node* st_adr = mem->in(MemNode::Address);
 611       intptr_t st_offset = 0;
 612       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 613       if (st_base == NULL)
 614         break;              // inscrutable pointer
 615 
 616       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 617       // We need the bases to be the equal in order for the offset check to make sense.
 618       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 619         break;
 620       }
 621 
 622       if (st_offset != offset && st_offset != Type::OffsetBot) {
 623         const int MAX_STORE = BytesPerLong;
 624         if (st_offset >= offset + size_in_bytes ||
 625             st_offset <= offset - MAX_STORE ||
 626             st_offset <= offset - mem->as_Store()->memory_size()) {
 627           // Success:  The offsets are provably independent.
 628           // (You may ask, why not just test st_offset != offset and be done?
 629           // The answer is that stores of different sizes can co-exist
 630           // in the same sequence of RawMem effects.  We sometimes initialize
 631           // a whole 'tile' of array elements with a single jint or jlong.)
 632           mem = mem->in(MemNode::Memory);
 633           continue;           // (a) advance through independent store memory
 634         }
 635       }
 636       if (st_base != base &&
 637           detect_ptr_independence(base, alloc,
 638                                   st_base,
 639                                   AllocateNode::Ideal_allocation(st_base, phase),
 640                                   phase)) {
 641         // Success:  The bases are provably independent.
 642         mem = mem->in(MemNode::Memory);
 643         continue;           // (a) advance through independent store memory
 644       }
 645 
 646       // (b) At this point, if the bases or offsets do not agree, we lose,
 647       // since we have not managed to prove 'this' and 'mem' independent.
 648       if (st_base == base && st_offset == offset) {
 649         return mem;         // let caller handle steps (c), (d)
 650       }
 651 
 652     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 653       InitializeNode* st_init = mem->in(0)->as_Initialize();
 654       AllocateNode*  st_alloc = st_init->allocation();
 655       if (st_alloc == NULL)
 656         break;              // something degenerated
 657       bool known_identical = false;
 658       bool known_independent = false;
 659       if (alloc == st_alloc)
 660         known_identical = true;
 661       else if (alloc != NULL)
 662         known_independent = true;
 663       else if (all_controls_dominate(this, st_alloc))
 664         known_independent = true;
 665 
 666       if (known_independent) {
 667         // The bases are provably independent: Either they are
 668         // manifestly distinct allocations, or else the control
 669         // of this load dominates the store's allocation.
 670         int alias_idx = phase->C->get_alias_index(adr_type());
 671         if (alias_idx == Compile::AliasIdxRaw) {
 672           mem = st_alloc->in(TypeFunc::Memory);
 673         } else {
 674           mem = st_init->memory(alias_idx);
 675         }
 676         continue;           // (a) advance through independent store memory
 677       }
 678 
 679       // (b) at this point, if we are not looking at a store initializing
 680       // the same allocation we are loading from, we lose.
 681       if (known_identical) {
 682         // From caller, can_see_stored_value will consult find_captured_store.
 683         return mem;         // let caller handle steps (c), (d)
 684       }
 685 
 686     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 687       if (prev != mem) {
 688         // Found an arraycopy but it doesn't affect that load
 689         continue;
 690       }
 691       // Found an arraycopy that may affect that load
 692       return mem;
 693     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 694       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 695       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 696         // ArrayCopyNodes processed here as well.
 697         CallNode *call = mem->in(0)->as_Call();
 698         if (!call->may_modify(addr_t, phase)) {
 699           mem = call->in(TypeFunc::Memory);
 700           continue;         // (a) advance through independent call memory
 701         }
 702       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 703         ArrayCopyNode* ac = NULL;
 704         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 705           break;
 706         }
 707         mem = mem->in(0)->in(TypeFunc::Memory);
 708         continue;           // (a) advance through independent MemBar memory
 709       } else if (mem->is_ClearArray()) {
 710         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 711           // (the call updated 'mem' value)
 712           continue;         // (a) advance through independent allocation memory
 713         } else {
 714           // Can not bypass initialization of the instance
 715           // we are looking for.
 716           return mem;
 717         }
 718       } else if (mem->is_MergeMem()) {
 719         int alias_idx = phase->C->get_alias_index(adr_type());
 720         mem = mem->as_MergeMem()->memory_at(alias_idx);
 721         continue;           // (a) advance through independent MergeMem memory
 722       }
 723     }
 724 
 725     // Unless there is an explicit 'continue', we must bail out here,
 726     // because 'mem' is an inscrutable memory state (e.g., a call).
 727     break;
 728   }
 729 
 730   return NULL;              // bail out
 731 }

 733 //----------------------calculate_adr_type-------------------------------------
 734 // Helper function.  Notices when the given type of address hits top or bottom.
 735 // Also, asserts a cross-check of the type against the expected address type.
 736 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 737   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 738   #ifdef ASSERT
 739   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 740   #endif
 741   const TypePtr* tp = t->isa_ptr();
 742   if (tp == NULL) {
 743     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 744     return TypePtr::BOTTOM;           // touches lots of memory
 745   } else {
 746     #ifdef ASSERT
 747     // %%%% [phh] We don't check the alias index if cross_check is
 748     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 749     if (cross_check != NULL &&
 750         cross_check != TypePtr::BOTTOM &&
 751         cross_check != TypeRawPtr::BOTTOM) {
 752       // Recheck the alias index, to see if it has changed (due to a bug).
 753       Compile* C = Compile::current();

 757       // disregarding "null"-ness.

 761              "real address must not escape from expected memory type");
 762     }
 763     #endif
 764     return tp;
 765   }

 924 // a load node that reads from the source array so we may be able to
 925 // optimize out the ArrayCopy node later.
 926 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 927   Node* ld_adr = in(MemNode::Address);
 928   intptr_t ld_off = 0;
 929   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 930   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 931   if (ac != NULL) {
 932     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 933 
 934     Node* mem = ac->in(TypeFunc::Memory);
 935     Node* ctl = ac->in(0);
 936     Node* src = ac->in(ArrayCopyNode::Src);

 938     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 939       return NULL;
 940     }

 943     Node* addp = in(MemNode::Address)->clone();
 944     if (ac->as_ArrayCopy()->is_clonebasic()) {

 950       addp->set_req(AddPNode::Base, src);
 951       addp->set_req(AddPNode::Address, src);
 952     } else {
 953       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 954              ac->as_ArrayCopy()->is_copyof_validated() ||
 955              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 956       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 957       addp->set_req(AddPNode::Base, src);
 958       addp->set_req(AddPNode::Address, src);
 959 
 960       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 961       BasicType ary_elem  = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 962       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 963       uint shift  = exact_log2(type2aelembytes(ary_elem));
 964 
 965       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));

 967       diff = phase->transform(new ConvI2LNode(diff));
 968 #endif
 969       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));

 976     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 977     ld->_adr_type = adr_type;
 978 #endif

 981     ld->set_req(MemNode::Memory, mem);
 982     // load depends on the tests that validate the arraycopy
 983     ld->_control_dependency = UnknownControl;
 984     return ld;
 985   }
 986   return NULL;

 990 //---------------------------can_see_stored_value------------------------------
 991 // This routine exists to make sure this set of tests is done the same
 992 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
 993 // will change the graph shape in a way which makes memory alive twice at the
 994 // same time (uses the Oracle model of aliasing), then some
 995 // LoadXNode::Identity will fold things back to the equivalence-class model
 996 // of aliasing.
 997 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
 998   Node* ld_adr = in(MemNode::Address);
 999   intptr_t ld_off = 0;
1000   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1001   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1002   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1003   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1004   // This is more general than load from boxing objects.
1005   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1006     uint alias_idx = atp->index();
1007     bool final = !atp->is_rewritable();
1008     Node* result = NULL;
1009     Node* current = st;
1010     // Skip through chains of MemBarNodes checking the MergeMems for
1011     // new states for the slice of this load.  Stop once any other
1012     // kind of node is encountered.  Loads from final memory can skip
1013     // through any kind of MemBar but normal loads shouldn't skip
1014     // through MemBarAcquire since the could allow them to move out of
1015     // a synchronized region.
1016     while (current->is_Proj()) {
1017       int opc = current->in(0)->Opcode();
1018       if ((final && (opc == Op_MemBarAcquire ||
1019                      opc == Op_MemBarAcquireLock ||
1020                      opc == Op_LoadFence)) ||
1021           opc == Op_MemBarRelease ||
1022           opc == Op_StoreFence ||
1023           opc == Op_MemBarReleaseLock ||
1024           opc == Op_MemBarStoreStore ||
1025           opc == Op_MemBarCPUOrder) {
1026         Node* mem = current->in(0)->in(TypeFunc::Memory);
1027         if (mem->is_MergeMem()) {
1028           MergeMemNode* merge = mem->as_MergeMem();
1029           Node* new_st = merge->memory_at(alias_idx);
1030           if (new_st == merge->base_memory()) {
1031             // Keep searching
1032             current = new_st;
1033             continue;
1034           }
1035           // Save the new memory state for the slice and fall through
1036           // to exit.
1037           result = new_st;
1038         }
1039       }
1040       break;
1041     }
1042     if (result != NULL) {
1043       st = result;
1044     }
1045   }
1046 
1047   // Loop around twice in the case Load -> Initialize -> Store.
1048   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1049   for (int trip = 0; trip <= 1; trip++) {
1050 
1051     if (st->is_Store()) {
1052       Node* st_adr = st->in(MemNode::Address);
1053       if (!phase->eqv(st_adr, ld_adr)) {
1054         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1055         intptr_t st_off = 0;
1056         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1057         if (ld_base == NULL)                                   return NULL;
1058         if (st_base == NULL)                                   return NULL;
1059         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1060         if (ld_off != st_off)                                  return NULL;
1061         if (ld_off == Type::OffsetBot)                         return NULL;
1062         // Same base, same offset.
1063         // Possible improvement for arrays: check index value instead of absolute offset.
1064 
1065         // At this point we have proven something like this setup:
1066         //   B = << base >>
1067         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1068         //   S = StoreQ(AddP(             B , #Off), V)
1069         // (Actually, we haven't yet proven the Q's are the same.)
1070         // In other words, we are loading from a casted version of
1071         // the same pointer-and-offset that we stored to.
1072         // Casted version may carry a dependency and it is respected.
1073         // Thus, we are able to replace L by V.
1074       }
1075       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1076       if (store_Opcode() != st->Opcode())
1077         return NULL;
1078       return st->in(MemNode::ValueIn);
1079     }
1080 
1081     // A load from a freshly-created object always returns zero.
1082     // (This can happen after LoadNode::Ideal resets the load's memory input
1083     // to find_captured_store, which returned InitializeNode::zero_memory.)
1084     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1085         (st->in(0) == ld_alloc) &&
1086         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1087       // return a zero value for the load's basic type
1088       // (This is one of the few places where a generic PhaseTransform
1089       // can create new nodes.  Think of it as lazily manifesting
1090       // virtually pre-existing constants.)
1091       return phase->zerocon(memory_type());
1092     }
1093 
1094     // A load from an initialization barrier can match a captured store.
1095     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1096       InitializeNode* init = st->in(0)->as_Initialize();
1097       AllocateNode* alloc = init->allocation();
1098       if ((alloc != NULL) && (alloc == ld_alloc)) {
1099         // examine a captured store value
1100         st = init->find_captured_store(ld_off, memory_size(), phase);
1101         if (st != NULL) {
1102           continue;             // take one more trip around
1103         }
1104       }
1105     }
1106 
1107     // Load boxed value from result of valueOf() call is input parameter.
1108     if (this->is_Load() && ld_adr->is_AddP() &&
1109         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1110       intptr_t ignore = 0;
1111       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1112       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1113       base = bs->step_over_gc_barrier(base);
1114       if (base != NULL && base->is_Proj() &&
1115           base->as_Proj()->_con == TypeFunc::Parms &&
1116           base->in(0)->is_CallStaticJava() &&
1117           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1118         return base->in(0)->in(TypeFunc::Parms);
1119       }
1120     }
1121 
1122     break;
1123   }
1124 
1125   return NULL;
1126 }

1145 //------------------------------Identity---------------------------------------
1146 // Loads are identity if previous store is to same address
1147 Node* LoadNode::Identity(PhaseGVN* phase) {
1148   // If the previous store-maker is the right kind of Store, and the store is
1149   // to the same address, then we are equal to the value stored.
1150   Node* mem = in(Memory);
1151   Node* value = can_see_stored_value(mem, phase);
1152   if( value ) {
1153     // byte, short & char stores truncate naturally.
1154     // A load has to load the truncated value which requires
1155     // some sort of masking operation and that requires an
1156     // Ideal call instead of an Identity call.
1157     if (memory_size() < BytesPerInt) {
1158       // If the input to the store does not fit with the load's result type,
1159       // it must be truncated via an Ideal call.
1160       if (!phase->type(value)->higher_equal(phase->type(this)))
1161         return this;
1162     }
1163     // (This works even when value is a Con, but LoadNode::Value
1164     // usually runs first, producing the singleton type of the Con.)
1165     return value;
1166   }
1167 
1168   // Search for an existing data phi which was generated before for the same
1169   // instance's field to avoid infinite generation of phis in a loop.
1170   Node *region = mem->in(0);
1171   if (is_instance_field_load_with_local_phi(region)) {
1172     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1173     int this_index  = phase->C->get_alias_index(addr_t);
1174     int this_offset = addr_t->offset();
1175     int this_iid    = addr_t->instance_id();
1176     if (!addr_t->is_known_instance() &&
1177          addr_t->is_ptr_to_boxed_value()) {
1178       // Use _idx of address base (could be Phi node) for boxed values.
1179       intptr_t   ignore = 0;
1180       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1181       if (base == NULL) {
1182         return this;
1183       }
1184       this_iid = base->_idx;
1185     }
1186     const Type* this_type = bottom_type();
1187     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1188       Node* phi = region->fast_out(i);
1189       if (phi->is_Phi() && phi != mem &&
1190           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1191         return phi;
1192       }
1193     }
1194   }
1195 
1196   return this;
1197 }

1237 // We're loading from an object which has autobox behaviour.
1238 // If this object is result of a valueOf call we'll have a phi
1239 // merging a newly allocated object and a load from the cache.
1240 // We want to replace this load with the original incoming
1241 // argument to the valueOf call.
1242 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1243   assert(phase->C->eliminate_boxing(), "sanity");
1244   intptr_t ignore = 0;
1245   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1246   if ((base == NULL) || base->is_Phi()) {
1247     // Push the loads from the phi that comes from valueOf up
1248     // through it to allow elimination of the loads and the recovery
1249     // of the original value. It is done in split_through_phi().
1250     return NULL;
1251   } else if (base->is_Load() ||
1252              (base->is_DecodeN() && base->in(1)->is_Load())) {
1253     // Eliminate the load of boxed value for integer types from the cache
1254     // array by deriving the value from the index into the array.
1255     // Capture the offset of the load and then reverse the computation.
1256 
1257     // Get LoadN node which loads a boxing object from 'cache' array.
1258     if (base->is_DecodeN()) {
1259       base = base->in(1);
1260     }
1261     if (!base->in(Address)->is_AddP()) {
1262       return NULL; // Complex address
1263     }
1264     AddPNode* address = base->in(Address)->as_AddP();
1265     Node* cache_base = address->in(AddPNode::Base);
1266     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1267       // Get ConP node which is static 'cache' field.
1268       cache_base = cache_base->in(1);
1269     }
1270     if ((cache_base != NULL) && cache_base->is_Con()) {
1271       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1272       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1273         Node* elements[4];
1274         int shift = exact_log2(type2aelembytes(T_OBJECT));
1275         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1276         if (count > 0 && elements[0]->is_Con() &&
1277             (count == 1 ||
1278              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1279                             elements[1]->in(2) == phase->intcon(shift)))) {
1280           ciObjArray* array = base_type->const_oop()->as_obj_array();
1281           // Fetch the box object cache[0] at the base of the array and get its value
1282           ciInstance* box = array->obj_at(0)->as_instance();
1283           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1284           assert(ik->is_box_klass(), "sanity");
1285           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1286           if (ik->nof_nonstatic_fields() == 1) {
1287             // This should be true nonstatic_field_at requires calling
1288             // nof_nonstatic_fields so check it anyway
1289             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1290             BasicType bt = c.basic_type();
1291             // Only integer types have boxing cache.
1292             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1293                    bt == T_BYTE    || bt == T_SHORT ||
1294                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1295             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1296             if (cache_low != (int)cache_low) {
1297               return NULL; // should not happen since cache is array indexed by value
1298             }
1299             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1300             if (offset != (int)offset) {
1301               return NULL; // should not happen since cache is array indexed by value
1302             }
1303            // Add up all the offsets making of the address of the load
1304             Node* result = elements[0];
1305             for (int i = 1; i < count; i++) {
1306               result = phase->transform(new AddXNode(result, elements[i]));
1307             }
1308             // Remove the constant offset from the address and then
1309             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1310             // remove the scaling of the offset to recover the original index.
1311             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1312               // Peel the shift off directly but wrap it in a dummy node
1313               // since Ideal can't return existing nodes
1314               result = new RShiftXNode(result->in(1), phase->intcon(0));
1315             } else if (result->is_Add() && result->in(2)->is_Con() &&
1316                        result->in(1)->Opcode() == Op_LShiftX &&
1317                        result->in(1)->in(2) == phase->intcon(shift)) {
1318               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1319               // but for boxing cache access we know that X<<Z will not overflow
1320               // (there is range check) so we do this optimizatrion by hand here.
1321               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1322               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1323             } else {
1324               result = new RShiftXNode(result, phase->intcon(shift));
1325             }
1326 #ifdef _LP64
1327             if (bt != T_LONG) {
1328               result = new ConvL2INode(phase->transform(result));
1329             }
1330 #else
1331             if (bt == T_LONG) {
1332               result = new ConvI2LNode(phase->transform(result));
1333             }
1334 #endif
1335             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1336             // Need to preserve unboxing load type if it is unsigned.
1337             switch(this->Opcode()) {
1338               case Op_LoadUB:
1339                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1340                 break;
1341               case Op_LoadUS:
1342                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1343                 break;
1344             }
1345             return result;
1346           }
1347         }
1348       }
1349     }
1350   }
1351   return NULL;
1352 }

1355   Node* region = phi->in(0);
1356   if (region == NULL) {
1357     return false; // Wait stable graph

1370 //------------------------------split_through_phi------------------------------
1371 // Split instance or boxed field load through Phi.
1372 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1373   Node* mem     = in(Memory);
1374   Node* address = in(Address);
1375   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1376 
1377   assert((t_oop != NULL) &&
1378          (t_oop->is_known_instance_field() ||
1379           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1380 
1381   Compile* C = phase->C;
1382   intptr_t ignore = 0;
1383   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1384   bool base_is_phi = (base != NULL) && base->is_Phi();
1385   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1386                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1387                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1388 
1389   if (!((mem->is_Phi() || base_is_phi) &&
1390         (load_boxed_values || t_oop->is_known_instance_field()))) {
1391     return NULL; // memory is not Phi
1392   }
1393 
1394   if (mem->is_Phi()) {
1395     if (!stable_phi(mem->as_Phi(), phase)) {
1396       return NULL; // Wait stable graph
1397     }
1398     uint cnt = mem->req();
1399     // Check for loop invariant memory.
1400     if (cnt == 3) {
1401       for (uint i = 1; i < cnt; i++) {
1402         Node* in = mem->in(i);
1403         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1404         if (m == mem) {
1405           if (i == 1) {
1406             // if the first edge was a loop, check second edge too.
1407             // If both are replaceable - we are in an infinite loop
1408             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1409             if (n == mem) {
1410               break;
1411             }
1412           }
1413           set_req(Memory, mem->in(cnt - i));
1414           return this; // made change
1415         }
1416       }
1417     }
1418   }
1419   if (base_is_phi) {
1420     if (!stable_phi(base->as_Phi(), phase)) {
1421       return NULL; // Wait stable graph
1422     }
1423     uint cnt = base->req();
1424     // Check for loop invariant memory.
1425     if (cnt == 3) {
1426       for (uint i = 1; i < cnt; i++) {
1427         if (base->in(i) == base) {
1428           return NULL; // Wait stable graph
1429         }
1430       }
1431     }
1432   }
1433 
1434   // Split through Phi (see original code in loopopts.cpp).
1435   assert(C->have_alias_type(t_oop), "instance should have alias type");
1436 
1437   // Do nothing here if Identity will find a value
1438   // (to avoid infinite chain of value phis generation).
1439   if (!phase->eqv(this, this->Identity(phase))) {
1440     return NULL;
1441   }
1442 
1443   // Select Region to split through.
1444   Node* region;
1445   if (!base_is_phi) {
1446     assert(mem->is_Phi(), "sanity");
1447     region = mem->in(0);
1448     // Skip if the region dominates some control edge of the address.
1449     if (!MemNode::all_controls_dominate(address, region))
1450       return NULL;
1451   } else if (!mem->is_Phi()) {
1452     assert(base_is_phi, "sanity");
1453     region = base->in(0);
1454     // Skip if the region dominates some control edge of the memory.
1455     if (!MemNode::all_controls_dominate(mem, region))
1456       return NULL;
1457   } else if (base->in(0) != mem->in(0)) {
1458     assert(base_is_phi && mem->is_Phi(), "sanity");
1459     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1460       region = base->in(0);
1461     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1462       region = mem->in(0);
1463     } else {
1464       return NULL; // complex graph
1465     }
1466   } else {
1467     assert(base->in(0) == mem->in(0), "sanity");
1468     region = mem->in(0);
1469   }
1470 
1471   const Type* this_type = this->bottom_type();
1472   int this_index  = C->get_alias_index(t_oop);
1473   int this_offset = t_oop->offset();
1474   int this_iid    = t_oop->instance_id();
1475   if (!t_oop->is_known_instance() && load_boxed_values) {
1476     // Use _idx of address base for boxed values.
1477     this_iid = base->_idx;
1478   }
1479   PhaseIterGVN* igvn = phase->is_IterGVN();
1480   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1481   for (uint i = 1; i < region->req(); i++) {
1482     Node* x;
1483     Node* the_clone = NULL;
1484     Node* in = region->in(i);
1485     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1486         in != NULL && in->is_OuterStripMinedLoop()) {
1487       // No node should go in the outer strip mined loop
1488       in = in->in(LoopNode::EntryControl);
1489     }
1490     if (in == NULL || in == C->top()) {
1491       x = C->top();      // Dead path?  Use a dead data op
1492     } else {
1493       x = this->clone();        // Else clone up the data op
1494       the_clone = x;            // Remember for possible deletion.
1495       // Alter data node to use pre-phi inputs
1496       if (this->in(0) == region) {
1497         x->set_req(0, in);
1498       } else {
1499         x->set_req(0, NULL);
1500       }
1501       if (mem->is_Phi() && (mem->in(0) == region)) {
1502         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1503       }
1504       if (address->is_Phi() && address->in(0) == region) {
1505         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1506       }
1507       if (base_is_phi && (base->in(0) == region)) {
1508         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1509         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1510         x->set_req(Address, adr_x);
1511       }
1512     }
1513     // Check for a 'win' on some paths
1514     const Type *t = x->Value(igvn);
1515 
1516     bool singleton = t->singleton();
1517 
1518     // See comments in PhaseIdealLoop::split_thru_phi().
1519     if (singleton && t == Type::TOP) {
1520       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1521     }
1522 
1523     if (singleton) {
1524       x = igvn->makecon(t);
1525     } else {
1526       // We now call Identity to try to simplify the cloned node.
1527       // Note that some Identity methods call phase->type(this).
1528       // Make sure that the type array is big enough for
1529       // our new node, even though we may throw the node away.
1530       // (This tweaking with igvn only works because x is a new node.)
1531       igvn->set_type(x, t);
1532       // If x is a TypeNode, capture any more-precise type permanently into Node
1533       // otherwise it will be not updated during igvn->transform since
1534       // igvn->type(x) is set to x->Value() already.
1535       x->raise_bottom_type(t);
1536       Node* y = x->Identity(igvn);
1537       if (y != x) {
1538         x = y;
1539       } else {
1540         y = igvn->hash_find_insert(x);
1541         if (y) {
1542           x = y;
1543         } else {
1544           // Else x is a new node we are keeping
1545           // We do not need register_new_node_with_optimizer
1546           // because set_type has already been called.
1547           igvn->_worklist.push(x);
1548         }
1549       }
1550     }
1551     if (x != the_clone && the_clone != NULL) {
1552       igvn->remove_dead_node(the_clone);
1553     }
1554     phi->set_req(i, x);
1555   }
1556   // Record Phi
1557   igvn->register_new_node_with_optimizer(phi);
1558   return phi;
1559 }

1579 // zero out the control input.
1580 // If the offset is constant and the base is an object allocation,
1581 // try to hook me up to the exact initializing store.
1582 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1583   Node* p = MemNode::Ideal_common(phase, can_reshape);
1584   if (p)  return (p == NodeSentinel) ? NULL : p;
1585 
1586   Node* ctrl    = in(MemNode::Control);
1587   Node* address = in(MemNode::Address);
1588   bool progress = false;
1589 
1590   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1591          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1592 
1593   // Skip up past a SafePoint control.  Cannot do this for Stores because
1594   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1595   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1596       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1597       !addr_mark &&
1598       (depends_only_on_test() || has_unknown_control_dependency())) {
1599     ctrl = ctrl->in(0);
1600     set_req(MemNode::Control,ctrl);
1601     progress = true;
1602   }
1603 
1604   intptr_t ignore = 0;
1605   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1606   if (base != NULL
1607       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1608     // Check for useless control edge in some common special cases
1609     if (in(MemNode::Control) != NULL
1610         && can_remove_control()
1611         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1612         && all_controls_dominate(base, phase->C->start())) {
1613       // A method-invariant, non-null address (constant or 'this' argument).
1614       set_req(MemNode::Control, NULL);
1615       progress = true;
1616     }
1617   }
1618 
1619   Node* mem = in(MemNode::Memory);
1620   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1621 
1622   if (can_reshape && (addr_t != NULL)) {
1623     // try to optimize our memory input
1624     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1625     if (opt_mem != mem) {
1626       set_req(MemNode::Memory, opt_mem);
1627       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1628       return this;
1629     }
1630     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1631     if ((t_oop != NULL) &&
1632         (t_oop->is_known_instance_field() ||
1633          t_oop->is_ptr_to_boxed_value())) {
1634       PhaseIterGVN *igvn = phase->is_IterGVN();
1635       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1636         // Delay this transformation until memory Phi is processed.
1637         phase->is_IterGVN()->_worklist.push(this);
1638         return NULL;
1639       }
1640       // Split instance field load through Phi.
1641       Node* result = split_through_phi(phase);
1642       if (result != NULL) return result;
1643 
1644       if (t_oop->is_ptr_to_boxed_value()) {
1645         Node* result = eliminate_autobox(phase);
1646         if (result != NULL) return result;
1647       }
1648     }
1649   }
1650 
1651   // Is there a dominating load that loads the same value?  Leave
1652   // anything that is not a load of a field/array element (like
1653   // barriers etc.) alone
1654   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1655     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1656       Node *use = mem->fast_out(i);
1657       if (use != this &&
1658           use->Opcode() == Opcode() &&
1659           use->in(0) != NULL &&
1660           use->in(0) != in(0) &&
1661           use->in(Address) == in(Address)) {
1662         Node* ctl = in(0);
1663         for (int i = 0; i < 10 && ctl != NULL; i++) {
1664           ctl = IfNode::up_one_dom(ctl);
1665           if (ctl == use->in(0)) {
1666             set_req(0, use->in(0));
1667             return this;
1668           }
1669         }
1670       }
1671     }
1672   }
1673 
1674   // Check for prior store with a different base or offset; make Load
1675   // independent.  Skip through any number of them.  Bail out if the stores
1676   // are in an endless dead cycle and report no progress.  This is a key
1677   // transform for Reflection.  However, if after skipping through the Stores
1678   // we can't then fold up against a prior store do NOT do the transform as
1679   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1680   // array memory alive twice: once for the hoisted Load and again after the
1681   // bypassed Store.  This situation only works if EVERYBODY who does
1682   // anti-dependence work knows how to bypass.  I.e. we need all
1683   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1684   // the alias index stuff.  So instead, peek through Stores and IFF we can
1685   // fold up, do so.
1686   Node* prev_mem = find_previous_store(phase);
1687   if (prev_mem != NULL) {
1688     Node* value = can_see_arraycopy_value(prev_mem, phase);
1689     if (value != NULL) {
1690       return value;
1691     }
1692   }
1693   // Steps (a), (b):  Walk past independent stores to find an exact match.
1694   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1695     // (c) See if we can fold up on the spot, but don't fold up here.
1696     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1697     // just return a prior value, which is done by Identity calls.
1698     if (can_see_stored_value(prev_mem, phase)) {
1699       // Make ready for step (d):
1700       set_req(MemNode::Memory, prev_mem);
1701       return this;
1702     }
1703   }
1704 
1705   AllocateNode* alloc = is_new_object_mark_load(phase);
1706   if (alloc != NULL && alloc->Opcode() == Op_Allocate && UseBiasedLocking) {
1707     InitializeNode* init = alloc->initialization();
1708     Node* control = init->proj_out(0);
1709     return alloc->make_ideal_mark(phase, address, control, mem);
1710   }
1711 
1712   return progress ? this : NULL;
1713 }

1715 // Helper to recognize certain Klass fields which are invariant across
1716 // some group of array types (e.g., int[] or all T[] where T < Object).
1717 const Type*
1718 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1719                                  ciKlass* klass) const {
1720   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1721     // The field is Klass::_modifier_flags.  Return its (constant) value.
1722     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1723     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1724     return TypeInt::make(klass->modifier_flags());
1725   }
1726   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1727     // The field is Klass::_access_flags.  Return its (constant) value.
1728     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1729     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1730     return TypeInt::make(klass->access_flags());
1731   }
1732   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1733     // The field is Klass::_layout_helper.  Return its constant value if known.
1734     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1735     return TypeInt::make(klass->layout_helper());
1736   }
1737 
1738   // No match.
1739   return NULL;
1740 }

1742 //------------------------------Value-----------------------------------------
1743 const Type* LoadNode::Value(PhaseGVN* phase) const {
1744   // Either input is TOP ==> the result is TOP
1745   Node* mem = in(MemNode::Memory);
1746   const Type *t1 = phase->type(mem);
1747   if (t1 == Type::TOP)  return Type::TOP;
1748   Node* adr = in(MemNode::Address);
1749   const TypePtr* tp = phase->type(adr)->isa_ptr();
1750   if (tp == NULL || tp->empty())  return Type::TOP;
1751   int off = tp->offset();
1752   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1753   Compile* C = phase->C;
1754 
1755   // Try to guess loaded type from pointer type
1756   if (tp->isa_aryptr()) {
1757     const TypeAryPtr* ary = tp->is_aryptr();
1758     const Type* t = ary->elem();
1759 
1760     // Determine whether the reference is beyond the header or not, by comparing
1761     // the offset against the offset of the start of the array's data.
1762     // Different array types begin at slightly different offsets (12 vs. 16).
1763     // We choose T_BYTE as an example base type that is least restrictive
1764     // as to alignment, which will therefore produce the smallest
1765     // possible base offset.
1766     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1767     const bool off_beyond_header = (off >= min_base_off);
1768 
1769     // Try to constant-fold a stable array element.
1770     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1771       // Make sure the reference is not into the header and the offset is constant
1772       ciObject* aobj = ary->const_oop();
1773       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1774         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1775         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1776                                                                       stable_dimension,
1777                                                                       memory_type(), is_unsigned());
1778         if (con_type != NULL) {
1779           return con_type;
1780         }
1781       }
1782     }
1783 
1784     // Don't do this for integer types. There is only potential profit if
1785     // the element type t is lower than _type; that is, for int types, if _type is
1786     // more restrictive than t.  This only happens here if one is short and the other
1787     // char (both 16 bits), and in those cases we've made an intentional decision
1788     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1789     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1790     //
1791     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1792     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1793     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1794     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1795     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1796     // In fact, that could have been the original type of p1, and p1 could have
1797     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1798     // expression (LShiftL quux 3) independently optimized to the constant 8.
1799     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1800         && (_type->isa_vect() == NULL)
1801         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1802       // t might actually be lower than _type, if _type is a unique
1803       // concrete subclass of abstract class t.
1804       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1805         const Type* jt = t->join_speculative(_type);
1806         // In any case, do not allow the join, per se, to empty out the type.
1807         if (jt->empty() && !t->empty()) {
1808           // This can happen if a interface-typed array narrows to a class type.
1809           jt = _type;
1810         }
1811 #ifdef ASSERT
1812         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1813           // The pointers in the autobox arrays are always non-null
1814           Node* base = adr->in(AddPNode::Base);
1815           if ((base != NULL) && base->is_DecodeN()) {
1816             // Get LoadN node which loads IntegerCache.cache field
1817             base = base->in(1);
1818           }
1819           if ((base != NULL) && base->is_Con()) {
1820             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1821             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1822               // It could be narrow oop
1823               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1824             }
1825           }
1826         }
1827 #endif
1828         return jt;
1829       }
1830     }
1831   } else if (tp->base() == Type::InstPtr) {
1832     assert( off != Type::OffsetBot ||
1833             // arrays can be cast to Objects
1834             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1835             // unsafe field access may not have a constant offset
1836             C->has_unsafe_access(),
1837             "Field accesses must be precise" );
1838     // For oop loads, we expect the _type to be precise.
1839 
1840     // Optimize loads from constant fields.
1841     const TypeInstPtr* tinst = tp->is_instptr();
1842     ciObject* const_oop = tinst->const_oop();
1843     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1844       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), memory_type());
1845       if (con_type != NULL) {
1846         return con_type;
1847       }
1848     }
1849   } else if (tp->base() == Type::KlassPtr) {
1850     assert( off != Type::OffsetBot ||
1851             // arrays can be cast to Objects
1852             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1853             // also allow array-loading from the primary supertype
1854             // array during subtype checks
1855             Opcode() == Op_LoadKlass,
1856             "Field accesses must be precise" );
1857     // For klass/static loads, we expect the _type to be precise
1858   } else if (tp->base() == Type::RawPtr && adr->is_Load() && off == 0) {
1859     /* With mirrors being an indirect in the Klass*
1860      * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1861      * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1862      *
1863      * So check the type and klass of the node before the LoadP.
1864      */
1865     Node* adr2 = adr->in(MemNode::Address);
1866     const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1867     if (tkls != NULL && !StressReflectiveCode) {
1868       ciKlass* klass = tkls->klass();
1869       if (klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1870         assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1871         assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1872         return TypeInstPtr::make(klass->java_mirror());
1873       }
1874     }
1875   }
1876 
1877   const TypeKlassPtr *tkls = tp->isa_klassptr();
1878   if (tkls != NULL && !StressReflectiveCode) {
1879     ciKlass* klass = tkls->klass();
1880     if (klass->is_loaded() && tkls->klass_is_exact()) {
1881       // We are loading a field from a Klass metaobject whose identity
1882       // is known at compile time (the type is "exact" or "precise").
1883       // Check for fields we know are maintained as constants by the VM.
1884       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1885         // The field is Klass::_super_check_offset.  Return its (constant) value.
1886         // (Folds up type checking code.)
1887         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1888         return TypeInt::make(klass->super_check_offset());
1889       }
1890       // Compute index into primary_supers array
1891       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1892       // Check for overflowing; use unsigned compare to handle the negative case.
1893       if( depth < ciKlass::primary_super_limit() ) {
1894         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1895         // (Folds up type checking code.)
1896         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1897         ciKlass *ss = klass->super_of_depth(depth);
1898         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1899       }
1900       const Type* aift = load_array_final_field(tkls, klass);
1901       if (aift != NULL)  return aift;
1902     }
1903 
1904     // We can still check if we are loading from the primary_supers array at a
1905     // shallow enough depth.  Even though the klass is not exact, entries less
1906     // than or equal to its super depth are correct.
1907     if (klass->is_loaded() ) {
1908       ciType *inner = klass;
1909       while( inner->is_obj_array_klass() )
1910         inner = inner->as_obj_array_klass()->base_element_type();
1911       if( inner->is_instance_klass() &&
1912           !inner->as_instance_klass()->flags().is_interface() ) {
1913         // Compute index into primary_supers array
1914         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1915         // Check for overflowing; use unsigned compare to handle the negative case.
1916         if( depth < ciKlass::primary_super_limit() &&
1917             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1918           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1919           // (Folds up type checking code.)
1920           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1921           ciKlass *ss = klass->super_of_depth(depth);
1922           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1923         }
1924       }
1925     }
1926 
1927     // If the type is enough to determine that the thing is not an array,
1928     // we can give the layout_helper a positive interval type.
1929     // This will help short-circuit some reflective code.
1930     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
1931         && !klass->is_array_klass() // not directly typed as an array
1932         && !klass->is_interface()  // specifically not Serializable & Cloneable
1933         && !klass->is_java_lang_Object()   // not the supertype of all T[]
1934         ) {
1935       // Note:  When interfaces are reliable, we can narrow the interface
1936       // test to (klass != Serializable && klass != Cloneable).
1937       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
1938       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
1939       // The key property of this type is that it folds up tests
1940       // for array-ness, since it proves that the layout_helper is positive.
1941       // Thus, a generic value like the basic object layout helper works fine.
1942       return TypeInt::make(min_size, max_jint, Type::WidenMin);
1943     }
1944   }
1945 
1946   // If we are loading from a freshly-allocated object, produce a zero,
1947   // if the load is provably beyond the header of the object.
1948   // (Also allow a variable load from a fresh array to produce zero.)
1949   const TypeOopPtr *tinst = tp->isa_oopptr();
1950   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
1951   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
1952   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
1953     Node* value = can_see_stored_value(mem,phase);
1954     if (value != NULL && value->is_Con()) {
1955       assert(value->bottom_type()->higher_equal(_type),"sanity");
1956       return value->bottom_type();
1957     }
1958   }
1959 
1960   if (is_instance) {
1961     // If we have an instance type and our memory input is the
1962     // programs's initial memory state, there is no matching store,
1963     // so just return a zero of the appropriate type
1964     Node *mem = in(MemNode::Memory);
1965     if (mem->is_Parm() && mem->in(0)->is_Start()) {
1966       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
1967       return Type::get_zero_type(_type->basic_type());
1968     }
1969   }
1970 
1971   Node* alloc = is_new_object_mark_load(phase);
1972   if (alloc != NULL && !(alloc->Opcode() == Op_Allocate && UseBiasedLocking)) {
1973     return TypeX::make(markWord::prototype().value());
1974   }
1975 
1976   return _type;
1977 }

2507 //------------------------------Ideal------------------------------------------
2508 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2509 // When a store immediately follows a relevant allocation/initialization,
2510 // try to capture it into the initialization, or hoist it above.
2511 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {

2513   if (p)  return (p == NodeSentinel) ? NULL : p;
2514 
2515   Node* mem     = in(MemNode::Memory);
2516   Node* address = in(MemNode::Address);
2517   // Back-to-back stores to same address?  Fold em up.  Generally
2518   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2519   // since they must follow each StoreP operation.  Redundant StoreCMs
2520   // are eliminated just before matching in final_graph_reshape.
2521   {
2522     Node* st = mem;
2523     // If Store 'st' has more than one use, we cannot fold 'st' away.
2524     // For example, 'st' might be the final state at a conditional
2525     // return.  Or, 'st' might be used by some node which is live at
2526     // the same time 'st' is live, which might be unschedulable.  So,
2527     // require exactly ONE user until such time as we clone 'mem' for
2528     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2529     // true).
2530     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2531       // Looking at a dead closed cycle of memory?
2532       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2533       assert(Opcode() == st->Opcode() ||
2534              st->Opcode() == Op_StoreVector ||
2535              Opcode() == Op_StoreVector ||
2536              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2537              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2538              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2539              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2540              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2541 
2542       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2543           st->as_Store()->memory_size() <= this->memory_size()) {
2544         Node* use = st->raw_out(0);
2545         phase->igvn_rehash_node_delayed(use);
2546         if (can_reshape) {
2547           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2548         } else {
2549           // It's OK to do this in the parser, since DU info is always accurate,
2550           // and the parser always refers to nodes via SafePointNode maps.
2551           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2552         }

2554       }
2555       st = st->in(MemNode::Memory);
2556     }
2557   }
2558 
2559 
2560   // Capture an unaliased, unconditional, simple store into an initializer.

2593 //------------------------------Identity---------------------------------------
2594 // Remove redundant stores:
2595 //   Store(m, p, Load(m, p)) changes to m.
2596 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2597 Node* StoreNode::Identity(PhaseGVN* phase) {
2598   Node* mem = in(MemNode::Memory);
2599   Node* adr = in(MemNode::Address);
2600   Node* val = in(MemNode::ValueIn);
2601 
2602   Node* result = this;
2603 
2604   // Load then Store?  Then the Store is useless
2605   if (val->is_Load() &&
2606       val->in(MemNode::Address)->eqv_uncast(adr) &&
2607       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2608       val->as_Load()->store_Opcode() == Opcode()) {
2609     result = mem;
2610   }
2611 
2612   // Two stores in a row of the same value?
2613   if (result == this &&
2614       mem->is_Store() &&
2615       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2616       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2617       mem->Opcode() == Opcode()) {
2618     result = mem;
2619   }
2620 
2621   // Store of zero anywhere into a freshly-allocated object?
2622   // Then the store is useless.
2623   // (It must already have been captured by the InitializeNode.)
2624   if (result == this &&
2625       ReduceFieldZeroing && phase->type(val)->is_zero_type()) {
2626     // a newly allocated object is already all-zeroes everywhere
2627     if (mem->is_Proj() && mem->in(0)->is_Allocate()) {
2628       result = mem;
2629     }
2630 
2631     if (result == this) {
2632       // the store may also apply to zero-bits in an earlier object
2633       Node* prev_mem = find_previous_store(phase);
2634       // Steps (a), (b):  Walk past independent stores to find an exact match.
2635       if (prev_mem != NULL) {
2636         Node* prev_val = can_see_stored_value(prev_mem, phase);
2637         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2638           // prev_val and val might differ by a cast; it would be good
2639           // to keep the more informative of the two.
2640           result = mem;
2641         }
2642       }
2643     }
2644   }
2645 
2646   if (result != this && phase->is_IterGVN() != NULL) {
2647     MemBarNode* trailing = trailing_membar();
2648     if (trailing != NULL) {
2649 #ifdef ASSERT
2650       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2651       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2652 #endif
2653       PhaseIterGVN* igvn = phase->is_IterGVN();
2654       trailing->remove(igvn);
2655     }
2656   }
2657 
2658   return result;
2659 }

2928 //------------------------------Idealize---------------------------------------
2929 // Clearing a short array is faster with stores
2930 Node *ClearArrayNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2931   // Already know this is a large node, do not try to ideal it
2932   if (!IdealizeClearArrayNode || _is_large) return NULL;
2933 
2934   const int unit = BytesPerLong;
2935   const TypeX* t = phase->type(in(2))->isa_intptr_t();
2936   if (!t)  return NULL;
2937   if (!t->is_con())  return NULL;
2938   intptr_t raw_count = t->get_con();
2939   intptr_t size = raw_count;
2940   if (!Matcher::init_array_count_is_in_bytes) size *= unit;
2941   // Clearing nothing uses the Identity call.
2942   // Negative clears are possible on dead ClearArrays
2943   // (see jck test stmt114.stmt11402.val).
2944   if (size <= 0 || size % unit != 0)  return NULL;
2945   intptr_t count = size / unit;
2946   // Length too long; communicate this to matchers and assemblers.
2947   // Assemblers are responsible to produce fast hardware clears for it.
2948   if (size > InitArrayShortSize) {
2949     return new ClearArrayNode(in(0), in(1), in(2), in(3), true);
2950   }
2951   Node *mem = in(1);
2952   if( phase->type(mem)==Type::TOP ) return NULL;
2953   Node *adr = in(3);
2954   const Type* at = phase->type(adr);
2955   if( at==Type::TOP ) return NULL;
2956   const TypePtr* atp = at->isa_ptr();
2957   // adjust atp to be the correct array element address type
2958   if (atp == NULL)  atp = TypePtr::BOTTOM;
2959   else              atp = atp->add_offset(Type::OffsetBot);
2960   // Get base for derived pointer purposes
2961   if( adr->Opcode() != Op_AddP ) Unimplemented();
2962   Node *base = adr->in(1);
2963 
2964   Node *zero = phase->makecon(TypeLong::ZERO);
2965   Node *off  = phase->MakeConX(BytesPerLong);
2966   mem = new StoreLNode(in(0),mem,adr,atp,zero,MemNode::unordered,false);
2967   count--;
2968   while( count-- ) {
2969     mem = phase->transform(mem);
2970     adr = phase->transform(new AddPNode(base,adr,off));
2971     mem = new StoreLNode(in(0),mem,adr,atp,zero,MemNode::unordered,false);
2972   }
2973   return mem;
2974 }

3025 Node* ClearArrayNode::clear_memory(Node* ctl, Node* mem, Node* dest,
3026                                    Node* start_offset,
3027                                    Node* end_offset,
3028                                    PhaseGVN* phase) {
3029   if (start_offset == end_offset) {
3030     // nothing to do
3031     return mem;
3032   }
3033 
3034   int unit = BytesPerLong;
3035   Node* zbase = start_offset;
3036   Node* zend  = end_offset;
3037 
3038   // Scale to the unit required by the CPU:
3039   if (!Matcher::init_array_count_is_in_bytes) {
3040     Node* shift = phase->intcon(exact_log2(unit));
3041     zbase = phase->transform(new URShiftXNode(zbase, shift) );
3042     zend  = phase->transform(new URShiftXNode(zend,  shift) );
3043   }
3044 
3045   // Bulk clear double-words
3046   Node* zsize = phase->transform(new SubXNode(zend, zbase) );
3047   Node* adr = phase->transform(new AddPNode(dest, dest, start_offset) );
3048   mem = new ClearArrayNode(ctl, mem, zsize, adr, false);
3049   return phase->transform(mem);
3050 }

3053                                    intptr_t start_offset,

3353 //===========================InitializeNode====================================
3354 // SUMMARY:
3355 // This node acts as a memory barrier on raw memory, after some raw stores.
3356 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3357 // The Initialize can 'capture' suitably constrained stores as raw inits.
3358 // It can coalesce related raw stores into larger units (called 'tiles').
3359 // It can avoid zeroing new storage for memory units which have raw inits.
3360 // At macro-expansion, it is marked 'complete', and does not optimize further.
3361 //
3362 // EXAMPLE:
3363 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3364 //   ctl = incoming control; mem* = incoming memory
3365 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3366 // First allocate uninitialized memory and fill in the header:
3367 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3368 //   ctl := alloc.Control; mem* := alloc.Memory*
3369 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3370 // Then initialize to zero the non-header parts of the raw memory block:
3371 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3372 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3373 // After the initialize node executes, the object is ready for service:
3374 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3375 // Suppose its body is immediately initialized as {1,2}:
3376 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3377 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3378 //   mem.SLICE(#short[*]) := store2
3379 //
3380 // DETAILS:
3381 // An InitializeNode collects and isolates object initialization after
3382 // an AllocateNode and before the next possible safepoint.  As a
3383 // memory barrier (MemBarNode), it keeps critical stores from drifting
3384 // down past any safepoint or any publication of the allocation.
3385 // Before this barrier, a newly-allocated object may have uninitialized bits.
3386 // After this barrier, it may be treated as a real oop, and GC is allowed.
3387 //
3388 // The semantics of the InitializeNode include an implicit zeroing of
3389 // the new object from object header to the end of the object.
3390 // (The object header and end are determined by the AllocateNode.)
3391 //
3392 // Certain stores may be added as direct inputs to the InitializeNode.
3393 // These stores must update raw memory, and they must be to addresses
3394 // derived from the raw address produced by AllocateNode, and with
3395 // a constant offset.  They must be ordered by increasing offset.
3396 // The first one is at in(RawStores), the last at in(req()-1).
3397 // Unlike most memory operations, they are not linked in a chain,
3398 // but are displayed in parallel as users of the rawmem output of
3399 // the allocation.
3400 //
3401 // (See comments in InitializeNode::capture_store, which continue
3402 // the example given above.)
3403 //
3404 // When the associated Allocate is macro-expanded, the InitializeNode
3405 // may be rewritten to optimize collected stores.  A ClearArrayNode
3406 // may also be created at that point to represent any required zeroing.
3407 // The InitializeNode is then marked 'complete', prohibiting further
3408 // capturing of nearby memory operations.
3409 //
3410 // During macro-expansion, all captured initializations which store
3411 // constant values of 32 bits or smaller are coalesced (if advantageous)
3412 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3413 // initialized in fewer memory operations.  Memory words which are
3414 // covered by neither tiles nor non-constant stores are pre-zeroed
3415 // by explicit stores of zero.  (The code shape happens to do all
3416 // zeroing first, then all other stores, with both sequences occurring
3417 // in order of ascending offsets.)
3418 //
3419 // Alternatively, code may be inserted between an AllocateNode and its
3420 // InitializeNode, to perform arbitrary initialization of the new object.
3421 // E.g., the object copying intrinsics insert complex data transfers here.
3422 // The initialization must then be marked as 'complete' disable the
3423 // built-in zeroing semantics and the collection of initializing stores.
3424 //
3425 // While an InitializeNode is incomplete, reads from the memory state
3426 // produced by it are optimizable if they match the control edge and
3427 // new oop address associated with the allocation/initialization.
3428 // They return a stored value (if the offset matches) or else zero.
3429 // A write to the memory state, if it matches control and address,
3430 // and if it is to a constant offset, may be 'captured' by the
3431 // InitializeNode.  It is cloned as a raw memory operation and rewired
3432 // inside the initialization, to the raw oop produced by the allocation.
3433 // Operations on addresses which are provably distinct (e.g., to
3434 // other AllocateNodes) are allowed to bypass the initialization.
3435 //
3436 // The effect of all this is to consolidate object initialization
3437 // (both arrays and non-arrays, both piecewise and bulk) into a
3438 // single location, where it can be optimized as a unit.
3439 //
3440 // Only stores with an offset less than TrackedInitializationLimit words
3441 // will be considered for capture by an InitializeNode.  This puts a
3442 // reasonable limit on the complexity of optimized initializations.
3443

3536 // Attempts to prove that a store's initial value 'n' can be captured

3548       return false;  // Bail out if processed too many nodes
3549     }

3552     if (n == NULL)      continue;   // (can this really happen?)
3553     if (n->is_Proj())   n = n->in(0);

3557     if (n->is_Root())   continue;   // even better

3575         return false;                  // failed to prove a good control

3579     for (uint i = 1; i < n->req(); i++) {

3591 // Here are all the checks a Store must pass before it can be moved into
3592 // an initialization.  Returns zero if a check fails.
3593 // On success, returns the (constant) offset to which the store applies,
3594 // within the initialized memory.
3595 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3596   const int FAIL = 0;
3597   if (st->req() != MemNode::ValueIn + 1)
3598     return FAIL;                // an inscrutable StoreNode (card mark?)
3599   Node* ctl = st->in(MemNode::Control);
3600   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3601     return FAIL;                // must be unconditional after the initialization
3602   Node* mem = st->in(MemNode::Memory);
3603   if (!(mem->is_Proj() && mem->in(0) == this))
3604     return FAIL;                // must not be preceded by other stores
3605   Node* adr = st->in(MemNode::Address);
3606   intptr_t offset;
3607   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3608   if (alloc == NULL)
3609     return FAIL;                // inscrutable address
3610   if (alloc != allocation())
3611     return FAIL;                // wrong allocation!  (store needs to float up)
3612   int size_in_bytes = st->memory_size();
3613   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3614     return FAIL;                // mismatched access
3615   }
3616   Node* val = st->in(MemNode::ValueIn);
3617 
3618   if (!detect_init_independence(val, phase))
3619     return FAIL;                // stored value must be 'simple enough'
3620 
3621   // The Store can be captured only if nothing after the allocation
3622   // and before the Store is using the memory location that the store
3623   // overwrites.
3624   bool failed = false;
3625   // If is_complete_with_arraycopy() is true the shape of the graph is
3626   // well defined and is safe so no need for extra checks.
3627   if (!is_complete_with_arraycopy()) {
3628     // We are going to look at each use of the memory state following
3629     // the allocation to make sure nothing reads the memory that the
3630     // Store writes.
3631     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3632     int alias_idx = phase->C->get_alias_index(t_adr);
3633     ResourceMark rm;
3634     Unique_Node_List mems;
3635     mems.push(mem);
3636     Node* unique_merge = NULL;
3637     for (uint next = 0; next < mems.size(); ++next) {
3638       Node *m  = mems.at(next);
3639       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3640         Node *n = m->fast_out(j);
3641         if (n->outcnt() == 0) {
3642           continue;
3643         }
3644         if (n == st) {
3645           continue;
3646         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3647           // If the control of this use is different from the control
3648           // of the Store which is right after the InitializeNode then
3649           // this node cannot be between the InitializeNode and the
3650           // Store.
3651           continue;
3652         } else if (n->is_MergeMem()) {
3653           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3654             // We can hit a MergeMemNode (that will likely go away
3655             // later) that is a direct use of the memory state
3656             // following the InitializeNode on the same slice as the
3657             // store node that we'd like to capture. We need to check
3658             // the uses of the MergeMemNode.
3659             mems.push(n);
3660           }
3661         } else if (n->is_Mem()) {
3662           Node* other_adr = n->in(MemNode::Address);
3663           if (other_adr == adr) {
3664             failed = true;
3665             break;
3666           } else {
3667             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3668             if (other_t_adr != NULL) {
3669               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3670               if (other_alias_idx == alias_idx) {
3671                 // A load from the same memory slice as the store right
3672                 // after the InitializeNode. We check the control of the
3673                 // object/array that is loaded from. If it's the same as
3674                 // the store control then we cannot capture the store.

3709 // Find the captured store in(i) which corresponds to the range
3710 // [start..start+size) in the initialized object.
3711 // If there is one, return its index i.  If there isn't, return the
3712 // negative of the index where it should be inserted.
3713 // Return 0 if the queried range overlaps an initialization boundary
3714 // or if dead code is encountered.
3715 // If size_in_bytes is zero, do not bother with overlap checks.
3716 int InitializeNode::captured_store_insertion_point(intptr_t start,
3717                                                    int size_in_bytes,
3718                                                    PhaseTransform* phase) {
3719   const int FAIL = 0, MAX_STORE = BytesPerLong;
3720 
3721   if (is_complete())
3722     return FAIL;                // arraycopy got here first; punt
3723 
3724   assert(allocation() != NULL, "must be present");
3725 
3726   // no negatives, no header fields:
3727   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3728 
3729   // after a certain size, we bail out on tracking all the stores:
3730   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3731   if (start >= ti_limit)  return FAIL;
3732 
3733   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3734     if (i >= limit)  return -(int)i; // not found; here is where to put it
3735 
3736     Node*    st     = in(i);
3737     intptr_t st_off = get_store_offset(st, phase);
3738     if (st_off < 0) {
3739       if (st != zero_memory()) {
3740         return FAIL;            // bail out if there is dead garbage
3741       }
3742     } else if (st_off > start) {
3743       // ...we are done, since stores are ordered
3744       if (st_off < start + size_in_bytes) {
3745         return FAIL;            // the next store overlaps
3746       }
3747       return -(int)i;           // not found; here is where to put it
3748     } else if (st_off < start) {
3749       if (size_in_bytes != 0 &&
3750           start < st_off + MAX_STORE &&
3751           start < st_off + st->as_Store()->memory_size()) {
3752         return FAIL;            // the previous store overlaps
3753       }
3754     } else {
3755       if (size_in_bytes != 0 &&
3756           st->as_Store()->memory_size() != size_in_bytes) {
3757         return FAIL;            // mismatched store size
3758       }
3759       return i;
3760     }
3761 
3762     ++i;
3763   }
3764 }

3797 // Clone the given store, converting it into a raw store
3798 // initializing a field or element of my new object.
3799 // Caller is responsible for retiring the original store,
3800 // with subsume_node or the like.
3801 //
3802 // From the example above InitializeNode::InitializeNode,
3803 // here are the old stores to be captured:
3804 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3805 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3806 //
3807 // Here is the changed code; note the extra edges on init:
3808 //   alloc = (Allocate ...)
3809 //   rawoop = alloc.RawAddress
3810 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3811 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3812 //   init = (Initialize alloc.Control alloc.Memory rawoop
3813 //                      rawstore1 rawstore2)
3814 //
3815 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3816                                     PhaseGVN* phase, bool can_reshape) {
3817   assert(stores_are_sane(phase), "");
3818 
3819   if (start < 0)  return NULL;
3820   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3821 
3822   Compile* C = phase->C;
3823   int size_in_bytes = st->memory_size();
3824   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3825   if (i == 0)  return NULL;     // bail out
3826   Node* prev_mem = NULL;        // raw memory for the captured store
3827   if (i > 0) {
3828     prev_mem = in(i);           // there is a pre-existing store under this one
3829     set_req(i, C->top());       // temporarily disconnect it
3830     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3831   } else {
3832     i = -i;                     // no pre-existing store
3833     prev_mem = zero_memory();   // a slice of the newly allocated object
3834     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3835       set_req(--i, C->top());   // reuse this edge; it has been folded away
3836     else
3837       ins_req(i, C->top());     // build a new edge
3838   }
3839   Node* new_st = st->clone();
3840   new_st->set_req(MemNode::Control, in(Control));
3841   new_st->set_req(MemNode::Memory,  prev_mem);
3842   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3843   new_st = phase->transform(new_st);
3844 
3845   // At this point, new_st might have swallowed a pre-existing store
3846   // at the same offset, or perhaps new_st might have disappeared,
3847   // if it redundantly stored the same value (or zero to fresh memory).
3848 
3849   // In any case, wire it in:
3850   phase->igvn_rehash_node_delayed(this);
3851   set_req(i, new_st);
3852 
3853   // The caller may now kill the old guy.
3854   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3855   assert(check_st == new_st || check_st == NULL, "must be findable");
3856   assert(!is_complete(), "");
3857   return new_st;
3858 }

3860 static bool store_constant(jlong* tiles, int num_tiles,
3861                            intptr_t st_off, int st_size,
3862                            jlong con) {
3863   if ((st_off & (st_size-1)) != 0)
3864     return false;               // strange store offset (assume size==2**N)
3865   address addr = (address)tiles + st_off;
3866   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
3867   switch (st_size) {
3868   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
3869   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
3870   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
3871   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3872   default: return false;        // strange store size (detect size!=2**N here)
3873   }
3874   return true;                  // return success to caller
3875 }

3877 // Coalesce subword constants into int constants and possibly
3878 // into long constants.  The goal, if the CPU permits,
3879 // is to initialize the object with a small number of 64-bit tiles.
3880 // Also, convert floating-point constants to bit patterns.
3881 // Non-constants are not relevant to this pass.
3882 //
3883 // In terms of the running example on InitializeNode::InitializeNode
3884 // and InitializeNode::capture_store, here is the transformation
3885 // of rawstore1 and rawstore2 into rawstore12:
3886 //   alloc = (Allocate ...)
3887 //   rawoop = alloc.RawAddress
3888 //   tile12 = 0x00010002
3889 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
3890 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
3891 //
3892 void
3893 InitializeNode::coalesce_subword_stores(intptr_t header_size,
3894                                         Node* size_in_bytes,
3895                                         PhaseGVN* phase) {
3896   Compile* C = phase->C;
3897 
3898   assert(stores_are_sane(phase), "");
3899   // Note:  After this pass, they are not completely sane,
3900   // since there may be some overlaps.
3901 
3902   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
3903 
3904   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3905   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
3906   size_limit = MIN2(size_limit, ti_limit);
3907   size_limit = align_up(size_limit, BytesPerLong);
3908   int num_tiles = size_limit / BytesPerLong;
3909 
3910   // allocate space for the tile map:
3911   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
3912   jlong  tiles_buf[small_len];
3913   Node*  nodes_buf[small_len];
3914   jlong  inits_buf[small_len];

3916                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
3917   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
3918                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
3919   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
3920                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
3921   // tiles: exact bitwise model of all primitive constants
3922   // nodes: last constant-storing node subsumed into the tiles model
3923   // inits: which bytes (in each tile) are touched by any initializations
3924 
3925   //// Pass A: Fill in the tile model with any relevant stores.
3926 
3927   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
3928   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
3929   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
3930   Node* zmem = zero_memory(); // initially zero memory state
3931   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
3932     Node* st = in(i);
3933     intptr_t st_off = get_store_offset(st, phase);
3934 
3935     // Figure out the store's offset and constant value:
3936     if (st_off < header_size)             continue; //skip (ignore header)
3937     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
3938     int st_size = st->as_Store()->memory_size();
3939     if (st_off + st_size > size_limit)    break;
3940 
3941     // Record which bytes are touched, whether by constant or not.
3942     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
3943       continue;                 // skip (strange store size)
3944 
3945     const Type* val = phase->type(st->in(MemNode::ValueIn));
3946     if (!val->singleton())                continue; //skip (non-con store)
3947     BasicType type = val->basic_type();
3948 
3949     jlong con = 0;
3950     switch (type) {
3951     case T_INT:    con = val->is_int()->get_con();  break;
3952     case T_LONG:   con = val->is_long()->get_con(); break;
3953     case T_FLOAT:  con = jint_cast(val->getf());    break;
3954     case T_DOUBLE: con = jlong_cast(val->getd());   break;
3955     default:                              continue; //skip (odd store type)
3956     }
3957 
3958     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
3959         st->Opcode() == Op_StoreL) {
3960       continue;                 // This StoreL is already optimal.
3961     }
3962 
3963     // Store down the constant.
3964     store_constant(tiles, num_tiles, st_off, st_size, con);
3965 
3966     intptr_t j = st_off >> LogBytesPerLong;
3967 
3968     if (type == T_INT && st_size == BytesPerInt
3969         && (st_off & BytesPerInt) == BytesPerInt) {
3970       jlong lcon = tiles[j];
3971       if (!Matcher::isSimpleConstant64(lcon) &&
3972           st->Opcode() == Op_StoreI) {
3973         // This StoreI is already optimal by itself.
3974         jint* intcon = (jint*) &tiles[j];
3975         intcon[1] = 0;  // undo the store_constant()
3976 
3977         // If the previous store is also optimal by itself, back up and
3978         // undo the action of the previous loop iteration... if we can.
3979         // But if we can't, just let the previous half take care of itself.
3980         st = nodes[j];
3981         st_off -= BytesPerInt;
3982         con = intcon[0];
3983         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
3984           assert(st_off >= header_size, "still ignoring header");
3985           assert(get_store_offset(st, phase) == st_off, "must be");
3986           assert(in(i-1) == zmem, "must be");
3987           DEBUG_ONLY(const Type* tcon = phase->type(st->in(MemNode::ValueIn)));
3988           assert(con == tcon->is_int()->get_con(), "must be");
3989           // Undo the effects of the previous loop trip, which swallowed st:
3990           intcon[0] = 0;        // undo store_constant()
3991           set_req(i-1, st);     // undo set_req(i, zmem)
3992           nodes[j] = NULL;      // undo nodes[j] = st
3993           --old_subword;        // undo ++old_subword
3994         }
3995         continue;               // This StoreI is already optimal.
3996       }
3997     }
3998 
3999     // This store is not needed.
4000     set_req(i, zmem);
4001     nodes[j] = st;              // record for the moment
4002     if (st_size < BytesPerLong) // something has changed
4003           ++old_subword;        // includes int/float, but who's counting...
4004     else  ++old_long;
4005   }
4006 
4007   if ((old_subword + old_long) == 0)
4008     return;                     // nothing more to do
4009 
4010   //// Pass B: Convert any non-zero tiles into optimal constant stores.
4011   // Be sure to insert them before overlapping non-constant stores.
4012   // (E.g., byte[] x = { 1,2,y,4 }  =>  x[int 0] = 0x01020004, x[2]=y.)
4013   for (int j = 0; j < num_tiles; j++) {
4014     jlong con  = tiles[j];
4015     jlong init = inits[j];
4016     if (con == 0)  continue;
4017     jint con0,  con1;           // split the constant, address-wise
4018     jint init0, init1;          // split the init map, address-wise
4019     { union { jlong con; jint intcon[2]; } u;
4020       u.con = con;
4021       con0  = u.intcon[0];
4022       con1  = u.intcon[1];
4023       u.con = init;
4024       init0 = u.intcon[0];
4025       init1 = u.intcon[1];
4026     }
4027 
4028     Node* old = nodes[j];
4029     assert(old != NULL, "need the prior store");
4030     intptr_t offset = (j * BytesPerLong);
4031 
4032     bool split = !Matcher::isSimpleConstant64(con);
4033 
4034     if (offset < header_size) {
4035       assert(offset + BytesPerInt >= header_size, "second int counts");
4036       assert(*(jint*)&tiles[j] == 0, "junk in header");
4037       split = true;             // only the second word counts
4038       // Example:  int a[] = { 42 ... }
4039     } else if (con0 == 0 && init0 == -1) {
4040       split = true;             // first word is covered by full inits
4041       // Example:  int a[] = { ... foo(), 42 ... }
4042     } else if (con1 == 0 && init1 == -1) {
4043       split = true;             // second word is covered by full inits
4044       // Example:  int a[] = { ... 42, foo() ... }
4045     }
4046

4048     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4049     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4050     // In this case the tile is not split; it is (jlong)42.
4051     // The big tile is stored down, and then the foo() value is inserted.
4052     // (If there were foo(),foo() instead of foo(),0, init0 would be -1.)
4053 
4054     Node* ctl = old->in(MemNode::Control);

4056     const TypePtr* atp = TypeRawPtr::BOTTOM;

4073                                     phase->intcon(con0), T_INT, MemNode::unordered);
4074       }

4076         ++new_int;
4077         offset += BytesPerInt;

4079         off[nst]  = offset;
4080         st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,

4082       }
4083     }

4086     // Insert each one just before any overlapping non-constant stores.

4094       guarantee(ins_idx != 0, "must re-insert constant store");

4096       if (ins_idx > InitializeNode::RawStores && in(ins_idx-1) == zmem)
4097         set_req(--ins_idx, st1);
4098       else
4099         ins_req(ins_idx, st1);
4100     }
4101   }

4103   if (PrintCompilation && WizardMode)
4104     tty->print_cr("Changed %d/%d subword/long constants into %d/%d int/long",
4105                   old_subword, old_long, new_int, new_long);

4108                    old_subword, old_long, new_int, new_long);

4111   remove_extra_zeroes();

4114 // Explore forward from in(start) to find the first fully initialized
4115 // word, and return its offset.  Skip groups of subword stores which
4116 // together initialize full words.  If in(start) is itself part of a
4117 // fully initialized word, return the offset of in(start).  If there
4118 // are no following full-word stores, or if something is fishy, return
4119 // a negative value.
4120 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4121   int       int_map = 0;
4122   intptr_t  int_map_off = 0;
4123   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4124 
4125   for (uint i = start, limit = req(); i < limit; i++) {
4126     Node* st = in(i);
4127 
4128     intptr_t st_off = get_store_offset(st, phase);
4129     if (st_off < 0)  break;  // return conservative answer
4130 
4131     int st_size = st->as_Store()->memory_size();
4132     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4133       return st_off;            // we found a complete word init
4134     }
4135 
4136     // update the map:
4137 
4138     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4139     if (this_int_off != int_map_off) {
4140       // reset the map:
4141       int_map = 0;
4142       int_map_off = this_int_off;
4143     }
4144 
4145     int subword_off = st_off - this_int_off;
4146     int_map |= right_n_bits(st_size) << subword_off;
4147     if ((int_map & FULL_MAP) == FULL_MAP) {
4148       return this_int_off;      // we found a complete word init
4149     }
4150 
4151     // Did this store hit or cross the word boundary?
4152     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4153     if (next_int_off == this_int_off + BytesPerInt) {
4154       // We passed the current int, without fully initializing it.
4155       int_map_off = next_int_off;
4156       int_map >>= BytesPerInt;
4157     } else if (next_int_off > this_int_off + BytesPerInt) {
4158       // We passed the current and next int.
4159       return this_int_off + BytesPerInt;
4160     }
4161   }
4162 
4163   return -1;
4164 }

4193   intptr_t zeroes_done = header_size;
4194 
4195   bool do_zeroing = true;       // we might give up if inits are very sparse
4196   int  big_init_gaps = 0;       // how many large gaps have we seen?
4197 
4198   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4199   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4200 
4201   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4202     Node* st = in(i);
4203     intptr_t st_off = get_store_offset(st, phase);
4204     if (st_off < 0)
4205       break;                    // unknown junk in the inits
4206     if (st->in(MemNode::Memory) != zmem)
4207       break;                    // complicated store chains somehow in list
4208 
4209     int st_size = st->as_Store()->memory_size();
4210     intptr_t next_init_off = st_off + st_size;
4211 
4212     if (do_zeroing && zeroes_done < next_init_off) {
4213       // See if this store needs a zero before it or under it.
4214       intptr_t zeroes_needed = st_off;
4215 
4216       if (st_size < BytesPerInt) {
4217         // Look for subword stores which only partially initialize words.
4218         // If we find some, we must lay down some word-level zeroes first,
4219         // underneath the subword stores.
4220         //
4221         // Examples:
4222         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4223         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4224         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4225         //
4226         // Note:  coalesce_subword_stores may have already done this,
4227         // if it was prompted by constant non-zero subword initializers.
4228         // But this case can still arise with non-constant stores.
4229 
4230         intptr_t next_full_store = find_next_fullword_store(i, phase);
4231 
4232         // In the examples above:
4233         //   in(i)          p   q   r   s     x   y     z
4234         //   st_off        12  13  14  15    12  13    14
4235         //   st_size        1   1   1   1     1   1     1
4236         //   next_full_s.  12  16  16  16    16  16    16
4237         //   z's_done      12  16  16  16    12  16    12
4238         //   z's_needed    12  16  16  16    16  16    16
4239         //   zsize          0   0   0   0     4   0     4
4240         if (next_full_store < 0) {
4241           // Conservative tack:  Zero to end of current word.
4242           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4243         } else {
4244           // Zero to beginning of next fully initialized word.
4245           // Or, don't zero at all, if we are already in that word.
4246           assert(next_full_store >= zeroes_needed, "must go forward");
4247           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4248           zeroes_needed = next_full_store;
4249         }
4250       }
4251 
4252       if (zeroes_needed > zeroes_done) {
4253         intptr_t zsize = zeroes_needed - zeroes_done;
4254         // Do some incremental zeroing on rawmem, in parallel with inits.
4255         zeroes_done = align_down(zeroes_done, BytesPerInt);
4256         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4257                                               zeroes_done, zeroes_needed,
4258                                               phase);
4259         zeroes_done = zeroes_needed;
4260         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4261           do_zeroing = false;   // leave the hole, next time
4262       }
4263     }
4264 
4265     // Collect the store and move on:
4266     phase->replace_input_of(st, MemNode::Memory, inits);
4267     inits = st;                 // put it on the linearized chain
4268     set_req(i, zmem);           // unhook from previous position
4269 
4270     if (zeroes_done == st_off)
4271       zeroes_done = next_init_off;
4272 
4273     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4274 
4275     #ifdef ASSERT
4276     // Various order invariants.  Weaker than stores_are_sane because
4277     // a large constant tile can be filled in by smaller non-constant stores.
4278     assert(st_off >= last_init_off, "inits do not reverse");
4279     last_init_off = st_off;
4280     const Type* val = NULL;
4281     if (st_size >= BytesPerInt &&
4282         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4283         (int)val->basic_type() < (int)T_OBJECT) {
4284       assert(st_off >= last_tile_end, "tiles do not overlap");
4285       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4286       last_tile_end = MAX2(last_tile_end, next_init_off);
4287     } else {
4288       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4289       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4290       assert(st_off      >= last_init_end, "inits do not overlap");
4291       last_init_end = next_init_off;  // it's a non-tile
4292     }
4293     #endif //ASSERT
4294   }
4295 
4296   remove_extra_zeroes();        // clear out all the zmems left over
4297   add_req(inits);
4298 
4299   if (!(UseTLAB && ZeroTLAB)) {
4300     // If anything remains to be zeroed, zero it all now.
4301     zeroes_done = align_down(zeroes_done, BytesPerInt);
4302     // if it is the last unused 4 bytes of an instance, forget about it
4303     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4304     if (zeroes_done + BytesPerLong >= size_limit) {
4305       AllocateNode* alloc = allocation();
4306       assert(alloc != NULL, "must be present");
4307       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4308         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4309         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4310         if (zeroes_done == k->layout_helper())
4311           zeroes_done = size_limit;
4312       }
4313     }
4314     if (zeroes_done < size_limit) {
4315       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4316                                             zeroes_done, size_in_bytes, phase);
4317     }
4318   }
4319 
4320   set_complete(phase);
4321   return rawmem;
4322 }

4350 //============================MergeMemNode=====================================
4351 //
4352 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4353 // contributing store or call operations.  Each contributor provides the memory
4354 // state for a particular "alias type" (see Compile::alias_type).  For example,
4355 // if a MergeMem has an input X for alias category #6, then any memory reference
4356 // to alias category #6 may use X as its memory state input, as an exact equivalent
4357 // to using the MergeMem as a whole.
4358 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4359 //
4360 // (Here, the <N> notation gives the index of the relevant adr_type.)
4361 //
4362 // In one special case (and more cases in the future), alias categories overlap.
4363 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4364 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4365 // it is exactly equivalent to that state W:
4366 //   MergeMem(<Bot>: W) <==> W
4367 //
4368 // Usually, the merge has more than one input.  In that case, where inputs
4369 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4370 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4371 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4372 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4373 //
4374 // A merge can take a "wide" memory state as one of its narrow inputs.
4375 // This simply means that the merge observes out only the relevant parts of
4376 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4377 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4378 //
4379 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4380 // and that memory slices "leak through":
4381 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4382 //
4383 // But, in such a cascade, repeated memory slices can "block the leak":
4384 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4385 //
4386 // In the last example, Y is not part of the combined memory state of the
4387 // outermost MergeMem.  The system must, of course, prevent unschedulable
4388 // memory states from arising, so you can be sure that the state Y is somehow
4389 // a precursor to state Y'.
4390 //
4391 //
4392 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4393 // of each MergeMemNode array are exactly the numerical alias indexes, including
4394 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4395 // Compile::alias_type (and kin) produce and manage these indexes.
4396 //
4397 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4398 // (Note that this provides quick access to the top node inside MergeMem methods,
4399 // without the need to reach out via TLS to Compile::current.)
4400 //
4401 // As a consequence of what was just described, a MergeMem that represents a full
4402 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4403 // containing all alias categories.
4404 //
4405 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4406 //
4407 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4408 // a memory state for the alias type <N>, or else the top node, meaning that
4409 // there is no particular input for that alias type.  Note that the length of
4410 // a MergeMem is variable, and may be extended at any time to accommodate new
4411 // memory states at larger alias indexes.  When merges grow, they are of course
4412 // filled with "top" in the unused in() positions.
4413 //
4414 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4415 // (Top was chosen because it works smoothly with passes like GCM.)
4416 //
4417 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4418 // the type of random VM bits like TLS references.)  Since it is always the
4419 // first non-Bot memory slice, some low-level loops use it to initialize an
4420 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4421 //
4422 //
4423 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4424 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4425 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4426 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4427 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4428 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4429 //
4430 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4431 // really that different from the other memory inputs.  An abbreviation called
4432 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4433 //
4434 //
4435 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4436 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4437 // that "emerges though" the base memory will be marked as excluding the alias types
4438 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4439 //
4440 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4441 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4442 //
4443 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4444 // (It is currently unimplemented.)  As you can see, the resulting merge is
4445 // actually a disjoint union of memory states, rather than an overlay.
4446 //

4508 //------------------------------Ideal------------------------------------------
4509 // This method is invoked recursively on chains of MergeMem nodes
4510 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4511   // Remove chain'd MergeMems
4512   //
4513   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4514   // relative to the "in(Bot)".  Since we are patching both at the same time,
4515   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4516   // but rewrite each "in(i)" relative to the new "in(Bot)".
4517   Node *progress = NULL;
4518 
4519 
4520   Node* old_base = base_memory();
4521   Node* empty_mem = empty_memory();
4522   if (old_base == empty_mem)
4523     return NULL; // Dead memory path.
4524 
4525   MergeMemNode* old_mbase;
4526   if (old_base != NULL && old_base->is_MergeMem())
4527     old_mbase = old_base->as_MergeMem();
4528   else
4529     old_mbase = NULL;
4530   Node* new_base = old_base;
4531 
4532   // simplify stacked MergeMems in base memory
4533   if (old_mbase)  new_base = old_mbase->base_memory();
4534 
4535   // the base memory might contribute new slices beyond my req()
4536   if (old_mbase)  grow_to_match(old_mbase);
4537 
4538   // Look carefully at the base node if it is a phi.
4539   PhiNode* phi_base;
4540   if (new_base != NULL && new_base->is_Phi())
4541     phi_base = new_base->as_Phi();
4542   else
4543     phi_base = NULL;
4544 
4545   Node*    phi_reg = NULL;
4546   uint     phi_len = (uint)-1;
4547   if (phi_base != NULL && !phi_base->is_copy()) {
4548     // do not examine phi if degraded to a copy
4549     phi_reg = phi_base->region();
4550     phi_len = phi_base->req();
4551     // see if the phi is unfinished
4552     for (uint i = 1; i < phi_len; i++) {
4553       if (phi_base->in(i) == NULL) {
4554         // incomplete phi; do not look at it yet!
4555         phi_reg = NULL;
4556         phi_len = (uint)-1;
4557         break;
4558       }
4559     }
4560   }
4561 
4562   // Note:  We do not call verify_sparse on entry, because inputs
4563   // can normalize to the base_memory via subsume_node or similar
4564   // mechanisms.  This method repairs that damage.
4565 
4566   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4567 
4568   // Look at each slice.
4569   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4570     Node* old_in = in(i);
4571     // calculate the old memory value
4572     Node* old_mem = old_in;
4573     if (old_mem == empty_mem)  old_mem = old_base;
4574     assert(old_mem == memory_at(i), "");
4575 
4576     // maybe update (reslice) the old memory value
4577 
4578     // simplify stacked MergeMems
4579     Node* new_mem = old_mem;
4580     MergeMemNode* old_mmem;
4581     if (old_mem != NULL && old_mem->is_MergeMem())
4582       old_mmem = old_mem->as_MergeMem();
4583     else
4584       old_mmem = NULL;
4585     if (old_mmem == this) {
4586       // This can happen if loops break up and safepoints disappear.
4587       // A merge of BotPtr (default) with a RawPtr memory derived from a
4588       // safepoint can be rewritten to a merge of the same BotPtr with
4589       // the BotPtr phi coming into the loop.  If that phi disappears
4590       // also, we can end up with a self-loop of the mergemem.
4591       // In general, if loops degenerate and memory effects disappear,
4592       // a mergemem can be left looking at itself.  This simply means
4593       // that the mergemem's default should be used, since there is
4594       // no longer any apparent effect on this slice.
4595       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4596       //       from start.  Update the input to TOP.
4597       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4598     }
4599     else if (old_mmem != NULL) {
4600       new_mem = old_mmem->memory_at(i);
4601     }
4602     // else preceding memory was not a MergeMem
4603 
4604     // replace equivalent phis (unfortunately, they do not GVN together)
4605     if (new_mem != NULL && new_mem != new_base &&
4606         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4607       if (new_mem->is_Phi()) {
4608         PhiNode* phi_mem = new_mem->as_Phi();
4609         for (uint i = 1; i < phi_len; i++) {
4610           if (phi_base->in(i) != phi_mem->in(i)) {
4611             phi_mem = NULL;
4612             break;
4613           }
4614         }
4615         if (phi_mem != NULL) {
4616           // equivalent phi nodes; revert to the def
4617           new_mem = new_base;
4618         }
4619       }
4620     }
4621 
4622     // maybe store down a new value
4623     Node* new_in = new_mem;
4624     if (new_in == new_base)  new_in = empty_mem;
4625 
4626     if (new_in != old_in) {
4627       // Warning:  Do not combine this "if" with the previous "if"
4628       // A memory slice might have be be rewritten even if it is semantically
4629       // unchanged, if the base_memory value has changed.
4630       set_req(i, new_in);
4631       progress = this;          // Report progress
4632     }
4633   }
4634 
4635   if (new_base != old_base) {
4636     set_req(Compile::AliasIdxBot, new_base);
4637     // Don't use set_base_memory(new_base), because we need to update du.
4638     assert(base_memory() == new_base, "");
4639     progress = this;
4640   }
4641 
4642   if( base_memory() == this ) {
4643     // a self cycle indicates this memory path is dead
4644     set_req(Compile::AliasIdxBot, empty_mem);
4645   }
4646 
4647   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4648   // Recursion must occur after the self cycle check above
4649   if( base_memory()->is_MergeMem() ) {
4650     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4651     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4652     if( m != NULL &&
4653         (m->is_top() ||
4654          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4655       // propagate rollup of dead cycle to self
4656       set_req(Compile::AliasIdxBot, empty_mem);
4657     }
4658   }
4659 
4660   if( base_memory() == empty_mem ) {
4661     progress = this;
4662     // Cut inputs during Parse phase only.
4663     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4664     if( !can_reshape ) {
4665       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4666         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4667       }
4668     }
4669   }
4670 
4671   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4672     // Check if PhiNode::Ideal's "Split phis through memory merges"
4673     // transform should be attempted. Look for this->phi->this cycle.
4674     uint merge_width = req();
4675     if (merge_width > Compile::AliasIdxRaw) {
4676       PhiNode* phi = base_memory()->as_Phi();
4677       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4678         if (phi->in(i) == this) {
4679           phase->is_IterGVN()->_worklist.push(phi);
4680           break;
4681         }
4682       }
4683     }
4684   }
4685 
4686   assert(progress || verify_sparse(), "please, no dups of base");
4687   return progress;
4688 }

4732 // verify a narrow slice (either incoming or outgoing)
4733 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4734   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4735   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4736   if (Node::in_dump())               return;  // muzzle asserts when printing
4737   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4738   assert(n != NULL, "");
4739   // Elide intervening MergeMem's
4740   while (n->is_MergeMem()) {
4741     n = n->as_MergeMem()->memory_at(alias_idx);
4742   }
4743   Compile* C = Compile::current();
4744   const TypePtr* n_adr_type = n->adr_type();
4745   if (n == m->empty_memory()) {
4746     // Implicit copy of base_memory()
4747   } else if (n_adr_type != TypePtr::BOTTOM) {
4748     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4749     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4750   } else {
4751     // A few places like make_runtime_call "know" that VM calls are narrow,
4752     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4753     bool expected_wide_mem = false;
4754     if (n == m->base_memory()) {
4755       expected_wide_mem = true;
4756     } else if (alias_idx == Compile::AliasIdxRaw ||
4757                n == m->memory_at(Compile::AliasIdxRaw)) {
4758       expected_wide_mem = true;
4759     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4760       // memory can "leak through" calls on channels that
4761       // are write-once.  Allow this also.
4762       expected_wide_mem = true;
4763     }
4764     assert(expected_wide_mem, "expected narrow slice replacement");
4765   }
4766 }

4772 //-----------------------------memory_at---------------------------------------
4773 Node* MergeMemNode::memory_at(uint alias_idx) const {
4774   assert(alias_idx >= Compile::AliasIdxRaw ||
4775          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4776          "must avoid base_memory and AliasIdxTop");
4777 
4778   // Otherwise, it is a narrow slice.
4779   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4780   Compile *C = Compile::current();
4781   if (is_empty_memory(n)) {
4782     // the array is sparse; empty slots are the "top" node
4783     n = base_memory();
4784     assert(Node::in_dump()
4785            || n == NULL || n->bottom_type() == Type::TOP
4786            || n->adr_type() == NULL // address is TOP
4787            || n->adr_type() == TypePtr::BOTTOM
4788            || n->adr_type() == TypeRawPtr::BOTTOM
4789            || Compile::current()->AliasLevel() == 0,
4790            "must be a wide memory");
4791     // AliasLevel == 0 if we are organizing the memory states manually.
4792     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.
4793   } else {
4794     // make sure the stored slice is sane
4795     #ifdef ASSERT
4796     if (VMError::is_error_reported() || Node::in_dump()) {
4797     } else if (might_be_same(n, base_memory())) {
4798       // Give it a pass:  It is a mostly harmless repetition of the base.
4799       // This can arise normally from node subsumption during optimization.
4800     } else {
4801       verify_memory_slice(this, alias_idx, n);
4802     }
4803     #endif
4804   }
4805   return n;
4806 }

4808 //---------------------------set_memory_at-------------------------------------
4809 void MergeMemNode::set_memory_at(uint alias_idx, Node *n) {
4810   verify_memory_slice(this, alias_idx, n);
4811   Node* empty_mem = empty_memory();
4812   if (n == base_memory())  n = empty_mem;  // collapse default
4813   uint need_req = alias_idx+1;
4814   if (req() < need_req) {
4815     if (n == empty_mem)  return;  // already the default, so do not grow me
4816     // grow the sparse array
4817     do {
4818       add_req(empty_mem);
4819     } while (req() < need_req);
4820   }
4821   set_req( alias_idx, n );
4822 }

4826 //--------------------------iteration_setup------------------------------------
4827 void MergeMemNode::iteration_setup(const MergeMemNode* other) {
4828   if (other != NULL) {
4829     grow_to_match(other);
4830     // invariant:  the finite support of mm2 is within mm->req()
4831     #ifdef ASSERT
4832     for (uint i = req(); i < other->req(); i++) {
4833       assert(other->is_empty_memory(other->in(i)), "slice left uncovered");
4834     }
4835     #endif
4836   }
4837   // Replace spurious copies of base_memory by top.
4838   Node* base_mem = base_memory();
4839   if (base_mem != NULL && !base_mem->is_top()) {
4840     for (uint i = Compile::AliasIdxBot+1, imax = req(); i < imax; i++) {
4841       if (in(i) == base_mem)
4842         set_req(i, empty_memory());
4843     }
4844   }
4845 }

4847 //---------------------------grow_to_match-------------------------------------
4848 void MergeMemNode::grow_to_match(const MergeMemNode* other) {
4849   Node* empty_mem = empty_memory();
4850   assert(other->is_empty_memory(empty_mem), "consistent sentinels");
4851   // look for the finite support of the other memory
4852   for (uint i = other->req(); --i >= req(); ) {
4853     if (other->in(i) != empty_mem) {
4854       uint new_len = i+1;
4855       while (req() < new_len)  add_req(empty_mem);
4856       break;
4857     }
4858   }
4859 }

4876   Node* n;
