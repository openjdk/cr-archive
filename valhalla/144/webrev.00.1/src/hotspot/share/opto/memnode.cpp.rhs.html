 108 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 109   st->print(" @");
 110   if (adr_type == NULL) {
 111     st->print("NULL");
 112   } else {
 113     adr_type->dump_on(st);
 114     Compile* C = Compile::current();
 115     Compile::AliasType* atp = NULL;
 116     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 117     if (atp == NULL)
 118       st->print(", idx=?\?;");
 119     else if (atp->index() == Compile::AliasIdxBot)
 120       st->print(", idx=Bot;");
 121     else if (atp->index() == Compile::AliasIdxTop)
 122       st->print(", idx=Top;");
 123     else if (atp->index() == Compile::AliasIdxRaw)
 124       st->print(", idx=Raw;");
 125     else {
 126       ciField* field = atp->field();
 127       if (field) {
 128         st->print(", name=");
 129         field->print_name_on(st);
 130       }
 131       st->print(", idx=%d;", atp->index());
 132     }
 133   }
 134 }

 140 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 141   assert((t_oop != NULL), "sanity");
 142   bool is_instance = t_oop->is_known_instance_field();
 143   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 144                              (load != NULL) && load->is_Load() &&
 145                              (phase->is_IterGVN() != NULL);
 146   if (!(is_instance || is_boxed_value_load))
 147     return mchain;  // don't try to optimize non-instance types
 148   uint instance_id = t_oop->instance_id();
 149   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 150   Node *prev = NULL;
 151   Node *result = mchain;
 152   while (prev != result) {
 153     prev = result;
 154     if (result == start_mem)
 155       break;  // hit one of our sentinels
 156     // skip over a call which does not affect this memory slice
 157     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 158       Node *proj_in = result->in(0);
 159       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 160         break;  // hit one of our sentinels
 161       } else if (proj_in->is_Call()) {
 162         // ArrayCopyNodes processed here as well
 163         CallNode *call = proj_in->as_Call();
 164         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 165           result = call->in(TypeFunc::Memory);
 166         }
 167       } else if (proj_in->is_Initialize()) {
 168         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 169         // Stop if this is the initialization for the object instance which
 170         // contains this memory slice, otherwise skip over it.
 171         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 172           break;
 173         }
 174         if (is_instance) {
 175           result = proj_in->in(TypeFunc::Memory);
 176         } else if (is_boxed_value_load) {
 177           Node* klass = alloc->in(AllocateNode::KlassNode);
 178           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 179           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 180             result = proj_in->in(TypeFunc::Memory); // not related allocation
 181           }
 182         }
 183       } else if (proj_in->is_MemBar()) {
 184         ArrayCopyNode* ac = NULL;
 185         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 186           break;
 187         }
 188         result = proj_in->in(TypeFunc::Memory);
 189       } else {
 190         assert(false, "unexpected projection");
 191       }
 192     } else if (result->is_ClearArray()) {
 193       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 194         // Can not bypass initialization of the instance
 195         // we are looking for.
 196         break;
 197       }
 198       // Otherwise skip it (the call updated 'result' value).
 199     } else if (result->is_MergeMem()) {
 200       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 201     }
 202   }
 203   return result;
 204 }

 206 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 207   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 208   if (t_oop == NULL)
 209     return mchain;  // don't try to optimize non-oop types
 210   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 211   bool is_instance = t_oop->is_known_instance_field();
 212   PhaseIterGVN *igvn = phase->is_IterGVN();
 213   if (is_instance && igvn != NULL && result->is_Phi()) {
 214     PhiNode *mphi = result->as_Phi();
 215     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 216     const TypePtr *t = mphi->adr_type();
 217     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 218         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 219          t->is_oopptr()->cast_to_exactness(true)
 220            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 221             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 222       // clone the Phi with our address type
 223       result = mphi->split_out_instance(t_adr, igvn);
 224     } else {
 225       if (t->isa_aryptr()) {
 226         // In the case of a flattened inline type array, each field has its own slice.
 227         // TODO This should be re-evaluated with JDK-8251039
 228         t = t->is_aryptr()->with_field_offset(t_adr->is_aryptr()->field_offset().get());
 229       }
 230       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 231     }
 232   }
 233   return result;
 234 }

 236 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 237   uint alias_idx = phase->C->get_alias_index(tp);
 238   Node *mem = mmem;
 239 #ifdef ASSERT
 240   {
 241     // Check that current type is consistent with the alias index used during graph construction
 242     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 243     bool consistent =  adr_check == NULL || adr_check->empty() ||
 244                        phase->C->must_alias(adr_check, alias_idx );
 245     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 246     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 247         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 248         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 249         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 250           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 251           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 252       // don't assert if it is dead code.
 253       consistent = true;
 254     }
 255     if( !consistent ) {
 256       st->print("alias_idx==%d, adr_check==", alias_idx);
 257       if( adr_check == NULL ) {
 258         st->print("NULL");
 259       } else {
 260         adr_check->dump();
 261       }
 262       st->cr();
 263       print_alias_types();
 264       assert(consistent, "adr_check must match alias idx");
 265     }
 266   }
 267 #endif
 268   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 269   // means an array I have not precisely typed yet.  Do not do any
 270   // alias stuff with it any time soon.
 271   const TypeOopPtr *toop = tp->isa_oopptr();
 272   if( tp->base() != Type::AnyPtr &&
 273       !(toop &&
 274         toop->klass() != NULL &&
 275         toop->klass()->is_java_lang_Object() &&
 276         toop->offset() == Type::OffsetBot) ) {
 277     // compress paths and change unreachable cycles to TOP
 278     // If not, we can update the input infinitely along a MergeMem cycle
 279     // Equivalent code in PhiNode::Ideal
 280     Node* m  = phase->transform(mmem);
 281     // If transformed to a MergeMem, get the desired slice
 282     // Otherwise the returned node represents memory for every slice
 283     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 284     // Update input if it is progress over what we have now
 285   }
 286   return mem;
 287 }

 289 //--------------------------Ideal_common---------------------------------------
 290 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 291 // Unhook non-raw memories from complete (macro-expanded) initializations.
 292 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 293   // If our control input is a dead region, kill all below the region
 294   Node *ctl = in(MemNode::Control);
 295   if (ctl && remove_dead_region(phase, can_reshape))
 296     return this;
 297   ctl = in(MemNode::Control);
 298   // Don't bother trying to transform a dead node
 299   if (ctl && ctl->is_top())  return NodeSentinel;
 300 
 301   PhaseIterGVN *igvn = phase->is_IterGVN();
 302   // Wait if control on the worklist.
 303   if (ctl && can_reshape && igvn != NULL) {
 304     Node* bol = NULL;
 305     Node* cmp = NULL;
 306     if (ctl->in(0)->is_If()) {
 307       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 308       bol = ctl->in(0)->in(1);
 309       if (bol->is_Bool())
 310         cmp = ctl->in(0)->in(1)->in(1);
 311     }
 312     if (igvn->_worklist.member(ctl) ||
 313         (bol != NULL && igvn->_worklist.member(bol)) ||
 314         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 315       // This control path may be dead.
 316       // Delay this memory node transformation until the control is processed.
 317       phase->is_IterGVN()->_worklist.push(this);
 318       return NodeSentinel; // caller will return NULL
 319     }
 320   }
 321   // Ignore if memory is dead, or self-loop
 322   Node *mem = in(MemNode::Memory);
 323   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 324   assert(mem != this, "dead loop in MemNode::Ideal");
 325 
 326   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 327     // This memory slice may be dead.
 328     // Delay this mem node transformation until the memory is processed.
 329     phase->is_IterGVN()->_worklist.push(this);
 330     return NodeSentinel; // caller will return NULL
 331   }
 332 
 333   Node *address = in(MemNode::Address);
 334   const Type *t_adr = phase->type(address);
 335   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 336 
 337   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 338     // Unsafe off-heap access with zero address. Remove access and other control users
 339     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 340     assert(ctl != NULL, "unsafe accesses should be control dependent");
 341     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 342       Node* u = ctl->fast_out(i);
 343       if (u != ctl) {
 344         igvn->rehash_node_delayed(u);
 345         int nb = u->replace_edge(ctl, phase->C->top());
 346         --i, imax -= nb;
 347       }
 348     }
 349     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 350     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 351     phase->C->root()->add_req(halt);
 352     return this;
 353   }
 354 
 355   if (can_reshape && igvn != NULL &&
 356       (igvn->_worklist.member(address) ||
 357        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 358     // The address's base and type may change when the address is processed.
 359     // Delay this mem node transformation until the address is processed.
 360     phase->is_IterGVN()->_worklist.push(this);
 361     return NodeSentinel; // caller will return NULL
 362   }
 363 
 364   // Do NOT remove or optimize the next lines: ensure a new alias index
 365   // is allocated for an oop pointer type before Escape Analysis.
 366   // Note: C++ will not remove it since the call has side effect.
 367   if (t_adr->isa_oopptr()) {
 368     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 369   }
 370 
 371   Node* base = NULL;
 372   if (address->is_AddP()) {
 373     base = address->in(AddPNode::Base);
 374   }
 375   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 376       !t_adr->isa_rawptr()) {
 377     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 378     // Skip this node optimization if its address has TOP base.
 379     return NodeSentinel; // caller will return NULL
 380   }
 381 
 382   // Avoid independent memory operations
 383   Node* old_mem = mem;
 384 
 385   // The code which unhooks non-raw memories from complete (macro-expanded)
 386   // initializations was removed. After macro-expansion all stores catched
 387   // by Initialize node became raw stores and there is no information
 388   // which memory slices they modify. So it is unsafe to move any memory
 389   // operation above these stores. Also in most cases hooked non-raw memories
 390   // were already unhooked by using information from detect_ptr_independence()
 391   // and find_previous_store().
 392 
 393   if (mem->is_MergeMem()) {
 394     MergeMemNode* mmem = mem->as_MergeMem();
 395     const TypePtr *tp = t_adr->is_ptr();
 396 
 397     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 398   }
 399 
 400   if (mem != old_mem) {
 401     set_req(MemNode::Memory, mem);
 402     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 403       igvn->_worklist.push(old_mem);
 404     }
 405     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 406     return this;
 407   }
 408 
 409   // let the subclass continue analyzing...
 410   return NULL;
 411 }

 413 // Helper function for proving some simple control dominations.
 414 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 415 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 416 // is not a constant (dominated by the method's StartNode).
 417 // Used by MemNode::find_previous_store to prove that the
 418 // control input of a memory operation predates (dominates)
 419 // an allocation it wants to look past.
 420 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 421   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 422     return false; // Conservative answer for dead code
 423 
 424   // Check 'dom'. Skip Proj and CatchProj nodes.
 425   dom = dom->find_exact_control(dom);
 426   if (dom == NULL || dom->is_top())
 427     return false; // Conservative answer for dead code
 428 
 429   if (dom == sub) {
 430     // For the case when, for example, 'sub' is Initialize and the original
 431     // 'dom' is Proj node of the 'sub'.
 432     return false;
 433   }
 434 
 435   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 436     return true;
 437 
 438   // 'dom' dominates 'sub' if its control edge and control edges
 439   // of all its inputs dominate or equal to sub's control edge.
 440 
 441   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 442   // Or Region for the check in LoadNode::Ideal();
 443   // 'sub' should have sub->in(0) != NULL.
 444   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 445          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 446 
 447   // Get control edge of 'sub'.
 448   Node* orig_sub = sub;
 449   sub = sub->find_exact_control(sub->in(0));
 450   if (sub == NULL || sub->is_top())
 451     return false; // Conservative answer for dead code
 452 
 453   assert(sub->is_CFG(), "expecting control");
 454 
 455   if (sub == dom)
 456     return true;
 457 
 458   if (sub->is_Start() || sub->is_Root())
 459     return false;
 460 
 461   {
 462     // Check all control edges of 'dom'.
 463 
 464     ResourceMark rm;
 465     Node_List nlist;
 466     Unique_Node_List dom_list;
 467 
 468     dom_list.push(dom);
 469     bool only_dominating_controls = false;
 470 
 471     for (uint next = 0; next < dom_list.size(); next++) {
 472       Node* n = dom_list.at(next);
 473       if (n == orig_sub)
 474         return false; // One of dom's inputs dominated by sub.
 475       if (!n->is_CFG() && n->pinned()) {
 476         // Check only own control edge for pinned non-control nodes.
 477         n = n->find_exact_control(n->in(0));
 478         if (n == NULL || n->is_top())
 479           return false; // Conservative answer for dead code
 480         assert(n->is_CFG(), "expecting control");
 481         dom_list.push(n);
 482       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 483         only_dominating_controls = true;
 484       } else if (n->is_CFG()) {
 485         if (n->dominates(sub, nlist))
 486           only_dominating_controls = true;
 487         else
 488           return false;
 489       } else {
 490         // First, own control edge.
 491         Node* m = n->find_exact_control(n->in(0));
 492         if (m != NULL) {
 493           if (m->is_top())
 494             return false; // Conservative answer for dead code
 495           dom_list.push(m);
 496         }
 497         // Now, the rest of edges.
 498         uint cnt = n->req();
 499         for (uint i = 1; i < cnt; i++) {
 500           m = n->find_exact_control(n->in(i));
 501           if (m == NULL || m->is_top())
 502             continue;
 503           dom_list.push(m);
 504         }
 505       }
 506     }
 507     return only_dominating_controls;
 508   }
 509 }

 511 //---------------------detect_ptr_independence---------------------------------
 512 // Used by MemNode::find_previous_store to prove that two base
 513 // pointers are never equal.
 514 // The pointers are accompanied by their associated allocations,
 515 // if any, which have been previously discovered by the caller.
 516 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 517                                       Node* p2, AllocateNode* a2,
 518                                       PhaseTransform* phase) {
 519   // Attempt to prove that these two pointers cannot be aliased.
 520   // They may both manifestly be allocations, and they should differ.
 521   // Or, if they are not both allocations, they can be distinct constants.
 522   // Otherwise, one is an allocation and the other a pre-existing value.
 523   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 524     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 525   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 526     return (a1 != a2);
 527   } else if (a1 != NULL) {                  // one allocation a1
 528     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 529     return all_controls_dominate(p2, a1);
 530   } else { //(a2 != NULL)                   // one allocation a2
 531     return all_controls_dominate(p1, a2);
 532   }
 533   return false;
 534 }

 537 // Find an arraycopy that must have set (can_see_stored_value=true) or
 538 // could have set (can_see_stored_value=false) the value for this load
 539 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 540   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 541                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 542     Node* mb = mem->in(0);
 543     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 544         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 545       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 546       if (ac->is_clonebasic()) {
 547         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 548         if (alloc != NULL && alloc == ld_alloc) {
 549           return ac;
 550         }
 551       }
 552     }
 553   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 554     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 555 
 556     if (ac->is_arraycopy_validated() ||
 557         ac->is_copyof_validated() ||
 558         ac->is_copyofrange_validated()) {
 559       Node* ld_addp = in(MemNode::Address);
 560       if (ld_addp->is_AddP()) {
 561         Node* ld_base = ld_addp->in(AddPNode::Address);
 562         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 563 
 564         Node* dest = ac->in(ArrayCopyNode::Dest);
 565 
 566         if (dest == ld_base) {
 567           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 568           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 569             return ac;
 570           }
 571           if (!can_see_stored_value) {
 572             mem = ac->in(TypeFunc::Memory);
 573           }
 574         }
 575       }
 576     }
 577   }
 578   return NULL;
 579 }

 581 // The logic for reordering loads and stores uses four steps:
 582 // (a) Walk carefully past stores and initializations which we
 583 //     can prove are independent of this load.
 584 // (b) Observe that the next memory state makes an exact match
 585 //     with self (load or store), and locate the relevant store.
 586 // (c) Ensure that, if we were to wire self directly to the store,
 587 //     the optimizer would fold it up somehow.
 588 // (d) Do the rewiring, and return, depending on some other part of
 589 //     the optimizer to fold up the load.
 590 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 591 // specific to loads and stores, so they are handled by the callers.
 592 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 593 //
 594 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 595   Node*         ctrl   = in(MemNode::Control);
 596   Node*         adr    = in(MemNode::Address);
 597   intptr_t      offset = 0;
 598   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 599   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 600 
 601   if (offset == Type::OffsetBot)
 602     return NULL;            // cannot unalias unless there are precise offsets
 603 
 604   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 605   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 606 
 607   intptr_t size_in_bytes = memory_size();
 608 
 609   Node* mem = in(MemNode::Memory);   // start searching here...
 610 
 611   int cnt = 50;             // Cycle limiter
 612   for (;;) {                // While we can dance past unrelated stores...
 613     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 614 
 615     Node* prev = mem;
 616     if (mem->is_Store()) {
 617       Node* st_adr = mem->in(MemNode::Address);
 618       intptr_t st_offset = 0;
 619       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 620       if (st_base == NULL)
 621         break;              // inscrutable pointer
 622 
 623       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 624       // We need the bases to be the equal in order for the offset check to make sense.
 625       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 626         break;
 627       }
 628 
 629       if (st_offset != offset && st_offset != Type::OffsetBot) {
 630         const int MAX_STORE = BytesPerLong;
 631         if (st_offset >= offset + size_in_bytes ||
 632             st_offset <= offset - MAX_STORE ||
 633             st_offset <= offset - mem->as_Store()->memory_size()) {
 634           // Success:  The offsets are provably independent.
 635           // (You may ask, why not just test st_offset != offset and be done?
 636           // The answer is that stores of different sizes can co-exist
 637           // in the same sequence of RawMem effects.  We sometimes initialize
 638           // a whole 'tile' of array elements with a single jint or jlong.)
 639           mem = mem->in(MemNode::Memory);
 640           continue;           // (a) advance through independent store memory
 641         }
 642       }
 643       if (st_base != base &&
 644           detect_ptr_independence(base, alloc,
 645                                   st_base,
 646                                   AllocateNode::Ideal_allocation(st_base, phase),
 647                                   phase)) {
 648         // Success:  The bases are provably independent.
 649         mem = mem->in(MemNode::Memory);
 650         continue;           // (a) advance through independent store memory
 651       }
 652 
 653       // (b) At this point, if the bases or offsets do not agree, we lose,
 654       // since we have not managed to prove 'this' and 'mem' independent.
 655       if (st_base == base && st_offset == offset) {
 656         return mem;         // let caller handle steps (c), (d)
 657       }
 658 
 659     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 660       InitializeNode* st_init = mem->in(0)->as_Initialize();
 661       AllocateNode*  st_alloc = st_init->allocation();
 662       if (st_alloc == NULL)
 663         break;              // something degenerated
 664       bool known_identical = false;
 665       bool known_independent = false;
 666       if (alloc == st_alloc)
 667         known_identical = true;
 668       else if (alloc != NULL)
 669         known_independent = true;
 670       else if (all_controls_dominate(this, st_alloc))
 671         known_independent = true;
 672 
 673       if (known_independent) {
 674         // The bases are provably independent: Either they are
 675         // manifestly distinct allocations, or else the control
 676         // of this load dominates the store's allocation.
 677         int alias_idx = phase->C->get_alias_index(adr_type());
 678         if (alias_idx == Compile::AliasIdxRaw) {
 679           mem = st_alloc->in(TypeFunc::Memory);
 680         } else {
 681           mem = st_init->memory(alias_idx);
 682         }
 683         continue;           // (a) advance through independent store memory
 684       }
 685 
 686       // (b) at this point, if we are not looking at a store initializing
 687       // the same allocation we are loading from, we lose.
 688       if (known_identical) {
 689         // From caller, can_see_stored_value will consult find_captured_store.
 690         return mem;         // let caller handle steps (c), (d)
 691       }
 692 
 693     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 694       if (prev != mem) {
 695         // Found an arraycopy but it doesn't affect that load
 696         continue;
 697       }
 698       // Found an arraycopy that may affect that load
 699       return mem;
 700     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 701       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 702       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 703         // ArrayCopyNodes processed here as well.
 704         CallNode *call = mem->in(0)->as_Call();
 705         if (!call->may_modify(addr_t, phase)) {
 706           mem = call->in(TypeFunc::Memory);
 707           continue;         // (a) advance through independent call memory
 708         }
 709       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 710         ArrayCopyNode* ac = NULL;
 711         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 712           break;
 713         }
 714         mem = mem->in(0)->in(TypeFunc::Memory);
 715         continue;           // (a) advance through independent MemBar memory
 716       } else if (mem->is_ClearArray()) {
 717         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 718           // (the call updated 'mem' value)
 719           continue;         // (a) advance through independent allocation memory
 720         } else {
 721           // Can not bypass initialization of the instance
 722           // we are looking for.
 723           return mem;
 724         }
 725       } else if (mem->is_MergeMem()) {
 726         int alias_idx = phase->C->get_alias_index(adr_type());
 727         mem = mem->as_MergeMem()->memory_at(alias_idx);
 728         continue;           // (a) advance through independent MergeMem memory
 729       }
 730     }
 731 
 732     // Unless there is an explicit 'continue', we must bail out here,
 733     // because 'mem' is an inscrutable memory state (e.g., a call).
 734     break;
 735   }
 736 
 737   return NULL;              // bail out
 738 }

 741 // Helper function.  Notices when the given type of address hits top or bottom.
 742 // Also, asserts a cross-check of the type against the expected address type.
 743 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 744   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 745   #ifdef ASSERT

 750     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 751     return TypePtr::BOTTOM;           // touches lots of memory
 752   } else {
 753     #ifdef ASSERT

 755     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 756     if (cross_check != NULL &&
 757         cross_check != TypePtr::BOTTOM &&
 758         cross_check != TypeRawPtr::BOTTOM) {

 768              "real address must not escape from expected memory type");
 769     }
 770     #endif
 771     return tp;
 772   }

 931 // Is the value loaded previously stored by an arraycopy? If so return
 932 // a load node that reads from the source array so we may be able to

 935   Node* ld_adr = in(MemNode::Address);
 936   intptr_t ld_off = 0;
 937   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 938   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 939   if (ac != NULL) {
 940     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 941 
 942     Node* mem = ac->in(TypeFunc::Memory);
 943     Node* ctl = ac->in(0);
 944     Node* src = ac->in(ArrayCopyNode::Src);
 945 
 946     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 947       return NULL;

 958       addp->set_req(AddPNode::Base, src);
 959       addp->set_req(AddPNode::Address, src);
 960     } else {
 961       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 962              ac->as_ArrayCopy()->is_copyof_validated() ||
 963              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 964       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 965       addp->set_req(AddPNode::Base, src);
 966       addp->set_req(AddPNode::Address, src);
 967 
 968       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 969       BasicType ary_elem = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 970       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 971       uint shift  = exact_log2(type2aelembytes(ary_elem));
 972       if (ary_t->klass()->is_flat_array_klass()) {
 973         ciFlatArrayKlass* vak = ary_t->klass()->as_flat_array_klass();
 974         shift = vak->log2_element_size();
 975       }
 976 
 977       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 978 #ifdef _LP64
 979       diff = phase->transform(new ConvI2LNode(diff));
 980 #endif
 981       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 982 
 983       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 984       addp->set_req(AddPNode::Offset, offset);
 985     }
 986     addp = phase->transform(addp);
 987 #ifdef ASSERT
 988     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 989     ld->_adr_type = adr_type;
 990 #endif
 991     ld->set_req(MemNode::Address, addp);
 992     ld->set_req(0, ctl);
 993     ld->set_req(MemNode::Memory, mem);
 994     // load depends on the tests that validate the arraycopy
 995     ld->_control_dependency = UnknownControl;
 996     return ld;
 997   }
 998   return NULL;

1002 //---------------------------can_see_stored_value------------------------------
1003 // This routine exists to make sure this set of tests is done the same
1004 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1005 // will change the graph shape in a way which makes memory alive twice at the
1006 // same time (uses the Oracle model of aliasing), then some
1007 // LoadXNode::Identity will fold things back to the equivalence-class model
1008 // of aliasing.
1009 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1010   Node* ld_adr = in(MemNode::Address);
1011   intptr_t ld_off = 0;
1012   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1013   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1014   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1015   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1016   // This is more general than load from boxing objects.
1017   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1018     uint alias_idx = atp->index();
1019     bool final = !atp->is_rewritable();
1020     Node* result = NULL;
1021     Node* current = st;
1022     // Skip through chains of MemBarNodes checking the MergeMems for
1023     // new states for the slice of this load.  Stop once any other
1024     // kind of node is encountered.  Loads from final memory can skip
1025     // through any kind of MemBar but normal loads shouldn't skip
1026     // through MemBarAcquire since the could allow them to move out of
1027     // a synchronized region.
1028     while (current->is_Proj()) {
1029       int opc = current->in(0)->Opcode();
1030       if ((final && (opc == Op_MemBarAcquire ||
1031                      opc == Op_MemBarAcquireLock ||
1032                      opc == Op_LoadFence)) ||
1033           opc == Op_MemBarRelease ||
1034           opc == Op_StoreFence ||
1035           opc == Op_MemBarReleaseLock ||
1036           opc == Op_MemBarStoreStore ||
1037           opc == Op_MemBarCPUOrder) {
1038         Node* mem = current->in(0)->in(TypeFunc::Memory);
1039         if (mem->is_MergeMem()) {
1040           MergeMemNode* merge = mem->as_MergeMem();
1041           Node* new_st = merge->memory_at(alias_idx);
1042           if (new_st == merge->base_memory()) {
1043             // Keep searching
1044             current = new_st;
1045             continue;
1046           }
1047           // Save the new memory state for the slice and fall through
1048           // to exit.
1049           result = new_st;
1050         }
1051       }
1052       break;
1053     }
1054     if (result != NULL) {
1055       st = result;
1056     }
1057   }
1058 
1059   // Loop around twice in the case Load -> Initialize -> Store.
1060   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1061   for (int trip = 0; trip <= 1; trip++) {
1062 
1063     if (st->is_Store()) {
1064       Node* st_adr = st->in(MemNode::Address);
1065       if (!phase->eqv(st_adr, ld_adr)) {
1066         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1067         intptr_t st_off = 0;
1068         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1069         if (ld_base == NULL)                                   return NULL;
1070         if (st_base == NULL)                                   return NULL;
1071         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1072         if (ld_off != st_off)                                  return NULL;
1073         if (ld_off == Type::OffsetBot)                         return NULL;
1074         // Same base, same offset.
1075         // Possible improvement for arrays: check index value instead of absolute offset.
1076 
1077         // At this point we have proven something like this setup:
1078         //   B = << base >>
1079         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1080         //   S = StoreQ(AddP(             B , #Off), V)
1081         // (Actually, we haven't yet proven the Q's are the same.)
1082         // In other words, we are loading from a casted version of
1083         // the same pointer-and-offset that we stored to.
1084         // Casted version may carry a dependency and it is respected.
1085         // Thus, we are able to replace L by V.
1086       }
1087       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1088       if (store_Opcode() != st->Opcode())
1089         return NULL;
1090       return st->in(MemNode::ValueIn);
1091     }
1092 
1093     // A load from a freshly-created object always returns zero.
1094     // (This can happen after LoadNode::Ideal resets the load's memory input
1095     // to find_captured_store, which returned InitializeNode::zero_memory.)
1096     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1097         (st->in(0) == ld_alloc) &&
1098         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1099       // return a zero value for the load's basic type
1100       // (This is one of the few places where a generic PhaseTransform
1101       // can create new nodes.  Think of it as lazily manifesting
1102       // virtually pre-existing constants.)
1103       assert(memory_type() != T_INLINE_TYPE, "should not be used for inline types");
1104       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1105       if (default_value != NULL) {
1106         return default_value;
1107       }
1108       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1109       return phase->zerocon(memory_type());
1110     }
1111 
1112     // A load from an initialization barrier can match a captured store.
1113     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1114       InitializeNode* init = st->in(0)->as_Initialize();
1115       AllocateNode* alloc = init->allocation();
1116       if ((alloc != NULL) && (alloc == ld_alloc)) {
1117         // examine a captured store value
1118         st = init->find_captured_store(ld_off, memory_size(), phase);
1119         if (st != NULL) {
1120           continue;             // take one more trip around
1121         }
1122       }
1123     }
1124 
1125     // Load boxed value from result of valueOf() call is input parameter.
1126     if (this->is_Load() && ld_adr->is_AddP() &&
1127         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1128       intptr_t ignore = 0;
1129       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1130       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1131       base = bs->step_over_gc_barrier(base);
1132       if (base != NULL && base->is_Proj() &&
1133           base->as_Proj()->_con == TypeFunc::Parms &&
1134           base->in(0)->is_CallStaticJava() &&
1135           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1136         return base->in(0)->in(TypeFunc::Parms);
1137       }
1138     }
1139 
1140     break;
1141   }
1142 
1143   return NULL;
1144 }

1163 //------------------------------Identity---------------------------------------
1164 // Loads are identity if previous store is to same address
1165 Node* LoadNode::Identity(PhaseGVN* phase) {
1166   // Loading from an InlineTypePtr? The InlineTypePtr has the values of
1167   // all fields as input. Look for the field with matching offset.
1168   Node* addr = in(Address);
1169   intptr_t offset;
1170   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1171   if (base != NULL && base->is_InlineTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1172     Node* value = base->as_InlineTypePtr()->field_value_by_offset((int)offset, true);
1173     if (value->is_InlineType()) {
1174       // Non-flattened inline type field
1175       InlineTypeNode* vt = value->as_InlineType();
1176       if (vt->is_allocated(phase)) {
1177         value = vt->get_oop();
1178       } else {
1179         // Not yet allocated, bail out
1180         value = NULL;
1181       }
1182     }
1183     if (value != NULL) {
1184       if (Opcode() == Op_LoadN) {
1185         // Encode oop value if we are loading a narrow oop
1186         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1187         value = phase->transform(new EncodePNode(value, bottom_type()));
1188       }
1189       return value;
1190     }
1191   }
1192 
1193   // If the previous store-maker is the right kind of Store, and the store is
1194   // to the same address, then we are equal to the value stored.
1195   Node* mem = in(Memory);
1196   Node* value = can_see_stored_value(mem, phase);
1197   if( value ) {
1198     // byte, short & char stores truncate naturally.
1199     // A load has to load the truncated value which requires
1200     // some sort of masking operation and that requires an
1201     // Ideal call instead of an Identity call.
1202     if (memory_size() < BytesPerInt) {
1203       // If the input to the store does not fit with the load's result type,
1204       // it must be truncated via an Ideal call.
1205       if (!phase->type(value)->higher_equal(phase->type(this)))
1206         return this;
1207     }
1208     // (This works even when value is a Con, but LoadNode::Value
1209     // usually runs first, producing the singleton type of the Con.)
1210     return value;
1211   }
1212 
1213   // Search for an existing data phi which was generated before for the same
1214   // instance's field to avoid infinite generation of phis in a loop.
1215   Node *region = mem->in(0);
1216   if (is_instance_field_load_with_local_phi(region)) {
1217     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1218     int this_index  = phase->C->get_alias_index(addr_t);
1219     int this_offset = addr_t->offset();
1220     int this_iid    = addr_t->instance_id();
1221     if (!addr_t->is_known_instance() &&
1222          addr_t->is_ptr_to_boxed_value()) {
1223       // Use _idx of address base (could be Phi node) for boxed values.
1224       intptr_t   ignore = 0;
1225       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1226       if (base == NULL) {
1227         return this;
1228       }
1229       this_iid = base->_idx;
1230     }
1231     const Type* this_type = bottom_type();
1232     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1233       Node* phi = region->fast_out(i);
1234       if (phi->is_Phi() && phi != mem &&
1235           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1236         return phi;
1237       }
1238     }
1239   }
1240 
1241   return this;
1242 }

1282 // We're loading from an object which has autobox behaviour.
1283 // If this object is result of a valueOf call we'll have a phi
1284 // merging a newly allocated object and a load from the cache.
1285 // We want to replace this load with the original incoming
1286 // argument to the valueOf call.
1287 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1288   assert(phase->C->eliminate_boxing(), "sanity");
1289   intptr_t ignore = 0;
1290   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1291   if ((base == NULL) || base->is_Phi()) {
1292     // Push the loads from the phi that comes from valueOf up
1293     // through it to allow elimination of the loads and the recovery
1294     // of the original value. It is done in split_through_phi().
1295     return NULL;
1296   } else if (base->is_Load() ||
1297              (base->is_DecodeN() && base->in(1)->is_Load())) {
1298     // Eliminate the load of boxed value for integer types from the cache
1299     // array by deriving the value from the index into the array.
1300     // Capture the offset of the load and then reverse the computation.
1301 
1302     // Get LoadN node which loads a boxing object from 'cache' array.
1303     if (base->is_DecodeN()) {
1304       base = base->in(1);
1305     }
1306     if (!base->in(Address)->is_AddP()) {
1307       return NULL; // Complex address
1308     }
1309     AddPNode* address = base->in(Address)->as_AddP();
1310     Node* cache_base = address->in(AddPNode::Base);
1311     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1312       // Get ConP node which is static 'cache' field.
1313       cache_base = cache_base->in(1);
1314     }
1315     if ((cache_base != NULL) && cache_base->is_Con()) {
1316       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1317       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1318         Node* elements[4];
1319         int shift = exact_log2(type2aelembytes(T_OBJECT));
1320         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1321         if (count > 0 && elements[0]->is_Con() &&
1322             (count == 1 ||
1323              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1324                             elements[1]->in(2) == phase->intcon(shift)))) {
1325           ciObjArray* array = base_type->const_oop()->as_obj_array();
1326           // Fetch the box object cache[0] at the base of the array and get its value
1327           ciInstance* box = array->obj_at(0)->as_instance();
1328           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1329           assert(ik->is_box_klass(), "sanity");
1330           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1331           if (ik->nof_nonstatic_fields() == 1) {
1332             // This should be true nonstatic_field_at requires calling
1333             // nof_nonstatic_fields so check it anyway
1334             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1335             BasicType bt = c.basic_type();
1336             // Only integer types have boxing cache.
1337             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1338                    bt == T_BYTE    || bt == T_SHORT ||
1339                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1340             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1341             if (cache_low != (int)cache_low) {
1342               return NULL; // should not happen since cache is array indexed by value
1343             }
1344             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1345             if (offset != (int)offset) {
1346               return NULL; // should not happen since cache is array indexed by value
1347             }
1348            // Add up all the offsets making of the address of the load
1349             Node* result = elements[0];
1350             for (int i = 1; i < count; i++) {
1351               result = phase->transform(new AddXNode(result, elements[i]));
1352             }
1353             // Remove the constant offset from the address and then
1354             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1355             // remove the scaling of the offset to recover the original index.
1356             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1357               // Peel the shift off directly but wrap it in a dummy node
1358               // since Ideal can't return existing nodes
1359               result = new RShiftXNode(result->in(1), phase->intcon(0));
1360             } else if (result->is_Add() && result->in(2)->is_Con() &&
1361                        result->in(1)->Opcode() == Op_LShiftX &&
1362                        result->in(1)->in(2) == phase->intcon(shift)) {
1363               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1364               // but for boxing cache access we know that X<<Z will not overflow
1365               // (there is range check) so we do this optimizatrion by hand here.
1366               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1367               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1368             } else {
1369               result = new RShiftXNode(result, phase->intcon(shift));
1370             }
1371 #ifdef _LP64
1372             if (bt != T_LONG) {
1373               result = new ConvL2INode(phase->transform(result));
1374             }
1375 #else
1376             if (bt == T_LONG) {
1377               result = new ConvI2LNode(phase->transform(result));
1378             }
1379 #endif
1380             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1381             // Need to preserve unboxing load type if it is unsigned.
1382             switch(this->Opcode()) {
1383               case Op_LoadUB:
1384                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1385                 break;
1386               case Op_LoadUS:
1387                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1388                 break;
1389             }
1390             return result;
1391           }
1392         }
1393       }
1394     }
1395   }
1396   return NULL;
1397 }

1399 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1400   Node* region = phi->in(0);
1401   if (region == NULL) {
1402     return false; // Wait stable graph
1403   }
1404   uint cnt = phi->req();
1405   for (uint i = 1; i < cnt; i++) {
1406     Node* rc = region->in(i);
1407     if (rc == NULL || phase->type(rc) == Type::TOP)
1408       return false; // Wait stable graph
1409     Node* in = phi->in(i);
1410     if (in == NULL || phase->type(in) == Type::TOP)
1411       return false; // Wait stable graph
1412   }
1413   return true;

1415 //------------------------------split_through_phi------------------------------
1416 // Split instance or boxed field load through Phi.
1417 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1418   Node* mem     = in(Memory);
1419   Node* address = in(Address);
1420   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1421 
1422   assert((t_oop != NULL) &&
1423          (t_oop->is_known_instance_field() ||
1424           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1425 
1426   Compile* C = phase->C;
1427   intptr_t ignore = 0;
1428   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1429   bool base_is_phi = (base != NULL) && base->is_Phi();
1430   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1431                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1432                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1433 
1434   if (!((mem->is_Phi() || base_is_phi) &&
1435         (load_boxed_values || t_oop->is_known_instance_field()))) {
1436     return NULL; // memory is not Phi
1437   }
1438 
1439   if (mem->is_Phi()) {
1440     if (!stable_phi(mem->as_Phi(), phase)) {
1441       return NULL; // Wait stable graph
1442     }
1443     uint cnt = mem->req();
1444     // Check for loop invariant memory.
1445     if (cnt == 3) {
1446       for (uint i = 1; i < cnt; i++) {
1447         Node* in = mem->in(i);
1448         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1449         if (m == mem) {
1450           if (i == 1) {
1451             // if the first edge was a loop, check second edge too.
1452             // If both are replaceable - we are in an infinite loop
1453             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1454             if (n == mem) {
1455               break;
1456             }
1457           }
1458           set_req(Memory, mem->in(cnt - i));
1459           return this; // made change
1460         }
1461       }
1462     }
1463   }
1464   if (base_is_phi) {
1465     if (!stable_phi(base->as_Phi(), phase)) {
1466       return NULL; // Wait stable graph
1467     }
1468     uint cnt = base->req();
1469     // Check for loop invariant memory.
1470     if (cnt == 3) {
1471       for (uint i = 1; i < cnt; i++) {
1472         if (base->in(i) == base) {
1473           return NULL; // Wait stable graph
1474         }
1475       }
1476     }
1477   }
1478 
1479   // Split through Phi (see original code in loopopts.cpp).
1480   assert(C->have_alias_type(t_oop), "instance should have alias type");
1481 
1482   // Do nothing here if Identity will find a value
1483   // (to avoid infinite chain of value phis generation).
1484   if (!phase->eqv(this, this->Identity(phase))) {
1485     return NULL;
1486   }
1487 
1488   // Select Region to split through.
1489   Node* region;
1490   if (!base_is_phi) {
1491     assert(mem->is_Phi(), "sanity");
1492     region = mem->in(0);
1493     // Skip if the region dominates some control edge of the address.
1494     if (!MemNode::all_controls_dominate(address, region))
1495       return NULL;
1496   } else if (!mem->is_Phi()) {
1497     assert(base_is_phi, "sanity");
1498     region = base->in(0);
1499     // Skip if the region dominates some control edge of the memory.
1500     if (!MemNode::all_controls_dominate(mem, region))
1501       return NULL;
1502   } else if (base->in(0) != mem->in(0)) {
1503     assert(base_is_phi && mem->is_Phi(), "sanity");
1504     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1505       region = base->in(0);
1506     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1507       region = mem->in(0);
1508     } else {
1509       return NULL; // complex graph
1510     }
1511   } else {
1512     assert(base->in(0) == mem->in(0), "sanity");
1513     region = mem->in(0);
1514   }
1515 
1516   const Type* this_type = this->bottom_type();
1517   int this_index  = C->get_alias_index(t_oop);
1518   int this_offset = t_oop->offset();
1519   int this_iid    = t_oop->instance_id();
1520   if (!t_oop->is_known_instance() && load_boxed_values) {
1521     // Use _idx of address base for boxed values.
1522     this_iid = base->_idx;
1523   }
1524   PhaseIterGVN* igvn = phase->is_IterGVN();
1525   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1526   for (uint i = 1; i < region->req(); i++) {
1527     Node* x;
1528     Node* the_clone = NULL;
1529     Node* in = region->in(i);
1530     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1531         in != NULL && in->is_OuterStripMinedLoop()) {
1532       // No node should go in the outer strip mined loop
1533       in = in->in(LoopNode::EntryControl);
1534     }
1535     if (in == NULL || in == C->top()) {
1536       x = C->top();      // Dead path?  Use a dead data op
1537     } else {
1538       x = this->clone();        // Else clone up the data op
1539       the_clone = x;            // Remember for possible deletion.
1540       // Alter data node to use pre-phi inputs
1541       if (this->in(0) == region) {
1542         x->set_req(0, in);
1543       } else {
1544         x->set_req(0, NULL);
1545       }
1546       if (mem->is_Phi() && (mem->in(0) == region)) {
1547         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1548       }
1549       if (address->is_Phi() && address->in(0) == region) {
1550         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1551       }
1552       if (base_is_phi && (base->in(0) == region)) {
1553         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1554         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1555         x->set_req(Address, adr_x);
1556       }
1557     }
1558     // Check for a 'win' on some paths
1559     const Type *t = x->Value(igvn);
1560 
1561     bool singleton = t->singleton();
1562 
1563     // See comments in PhaseIdealLoop::split_thru_phi().
1564     if (singleton && t == Type::TOP) {
1565       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1566     }
1567 
1568     if (singleton) {
1569       x = igvn->makecon(t);
1570     } else {
1571       // We now call Identity to try to simplify the cloned node.
1572       // Note that some Identity methods call phase->type(this).
1573       // Make sure that the type array is big enough for
1574       // our new node, even though we may throw the node away.
1575       // (This tweaking with igvn only works because x is a new node.)
1576       igvn->set_type(x, t);
1577       // If x is a TypeNode, capture any more-precise type permanently into Node
1578       // otherwise it will be not updated during igvn->transform since
1579       // igvn->type(x) is set to x->Value() already.
1580       x->raise_bottom_type(t);
1581       Node* y = x->Identity(igvn);
1582       if (y != x) {
1583         x = y;
1584       } else {
1585         y = igvn->hash_find_insert(x);
1586         if (y) {
1587           x = y;
1588         } else {
1589           // Else x is a new node we are keeping
1590           // We do not need register_new_node_with_optimizer
1591           // because set_type has already been called.
1592           igvn->_worklist.push(x);
1593         }
1594       }
1595     }
1596     if (x != the_clone && the_clone != NULL) {
1597       igvn->remove_dead_node(the_clone);
1598     }
1599     phi->set_req(i, x);
1600   }
1601   // Record Phi
1602   igvn->register_new_node_with_optimizer(phi);
1603   return phi;
1604 }

1622 //------------------------------Ideal------------------------------------------
1623 // If the load is from Field memory and the pointer is non-null, it might be possible to
1624 // zero out the control input.
1625 // If the offset is constant and the base is an object allocation,
1626 // try to hook me up to the exact initializing store.
1627 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1628   Node* p = MemNode::Ideal_common(phase, can_reshape);
1629   if (p)  return (p == NodeSentinel) ? NULL : p;
1630 
1631   Node* ctrl    = in(MemNode::Control);
1632   Node* address = in(MemNode::Address);
1633   bool progress = false;
1634 
1635   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1636          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1637 
1638   // Skip up past a SafePoint control.  Cannot do this for Stores because
1639   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1640   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1641       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1642       !addr_mark &&
1643       (depends_only_on_test() || has_unknown_control_dependency())) {
1644     ctrl = ctrl->in(0);
1645     set_req(MemNode::Control,ctrl);
1646     progress = true;
1647   }
1648 
1649   intptr_t ignore = 0;
1650   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1651   if (base != NULL
1652       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1653     // Check for useless control edge in some common special cases
1654     if (in(MemNode::Control) != NULL
1655         && can_remove_control()
1656         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1657         && all_controls_dominate(base, phase->C->start())) {
1658       // A method-invariant, non-null address (constant or 'this' argument).
1659       set_req(MemNode::Control, NULL);
1660       progress = true;
1661     }
1662   }
1663 
1664   Node* mem = in(MemNode::Memory);
1665   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1666 
1667   if (can_reshape && (addr_t != NULL)) {
1668     // try to optimize our memory input
1669     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1670     if (opt_mem != mem) {
1671       set_req(MemNode::Memory, opt_mem);
1672       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1673       return this;
1674     }
1675     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1676     if ((t_oop != NULL) &&
1677         (t_oop->is_known_instance_field() ||
1678          t_oop->is_ptr_to_boxed_value())) {
1679       PhaseIterGVN *igvn = phase->is_IterGVN();
1680       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1681         // Delay this transformation until memory Phi is processed.
1682         phase->is_IterGVN()->_worklist.push(this);
1683         return NULL;
1684       }
1685       // Split instance field load through Phi.
1686       Node* result = split_through_phi(phase);
1687       if (result != NULL) return result;
1688 
1689       if (t_oop->is_ptr_to_boxed_value()) {
1690         Node* result = eliminate_autobox(phase);
1691         if (result != NULL) return result;
1692       }
1693     }
1694   }
1695 
1696   // Is there a dominating load that loads the same value?  Leave
1697   // anything that is not a load of a field/array element (like
1698   // barriers etc.) alone
1699   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1700     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1701       Node *use = mem->fast_out(i);
1702       if (use != this &&
1703           use->Opcode() == Opcode() &&
1704           use->in(0) != NULL &&
1705           use->in(0) != in(0) &&
1706           use->in(Address) == in(Address)) {
1707         Node* ctl = in(0);
1708         for (int i = 0; i < 10 && ctl != NULL; i++) {
1709           ctl = IfNode::up_one_dom(ctl);
1710           if (ctl == use->in(0)) {
1711             set_req(0, use->in(0));
1712             return this;
1713           }
1714         }
1715       }
1716     }
1717   }
1718 
1719   // Check for prior store with a different base or offset; make Load
1720   // independent.  Skip through any number of them.  Bail out if the stores
1721   // are in an endless dead cycle and report no progress.  This is a key
1722   // transform for Reflection.  However, if after skipping through the Stores
1723   // we can't then fold up against a prior store do NOT do the transform as
1724   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1725   // array memory alive twice: once for the hoisted Load and again after the
1726   // bypassed Store.  This situation only works if EVERYBODY who does
1727   // anti-dependence work knows how to bypass.  I.e. we need all
1728   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1729   // the alias index stuff.  So instead, peek through Stores and IFF we can
1730   // fold up, do so.
1731   Node* prev_mem = find_previous_store(phase);
1732   if (prev_mem != NULL) {
1733     Node* value = can_see_arraycopy_value(prev_mem, phase);
1734     if (value != NULL) {
1735       return value;
1736     }
1737   }
1738   // Steps (a), (b):  Walk past independent stores to find an exact match.
1739   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1740     // (c) See if we can fold up on the spot, but don't fold up here.
1741     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1742     // just return a prior value, which is done by Identity calls.
1743     if (can_see_stored_value(prev_mem, phase)) {
1744       // Make ready for step (d):
1745       set_req(MemNode::Memory, prev_mem);
1746       return this;
1747     }
1748   }
1749 
1750   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1751   if (alloc != NULL && mem->is_Proj() &&
1752       mem->in(0) != NULL &&
1753       mem->in(0) == alloc->initialization() &&
1754       Opcode() == Op_LoadX &&
1755       alloc->initialization()->proj_out_or_null(0) != NULL) {
1756     InitializeNode* init = alloc->initialization();
1757     Node* control = init->proj_out(0);
1758     return alloc->make_ideal_mark(phase, control, mem);
1759   }
1760 
1761   return progress ? this : NULL;
1762 }

1764 // Helper to recognize certain Klass fields which are invariant across
1765 // some group of array types (e.g., int[] or all T[] where T < Object).
1766 const Type*
1767 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1768                                  ciKlass* klass) const {
1769   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1770     // The field is Klass::_modifier_flags.  Return its (constant) value.
1771     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1772     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1773     return TypeInt::make(klass->modifier_flags());
1774   }
1775   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1776     // The field is Klass::_access_flags.  Return its (constant) value.
1777     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1778     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1779     return TypeInt::make(klass->access_flags());
1780   }
1781   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1782     // The field is Klass::_layout_helper.  Return its constant value if known.
1783     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1784     return TypeInt::make(klass->layout_helper());
1785   }
1786 
1787   // No match.
1788   return NULL;
1789 }

1791 //------------------------------Value-----------------------------------------
1792 const Type* LoadNode::Value(PhaseGVN* phase) const {
1793   // Either input is TOP ==> the result is TOP
1794   Node* mem = in(MemNode::Memory);
1795   const Type *t1 = phase->type(mem);
1796   if (t1 == Type::TOP)  return Type::TOP;
1797   Node* adr = in(MemNode::Address);
1798   const TypePtr* tp = phase->type(adr)->isa_ptr();
1799   if (tp == NULL || tp->empty())  return Type::TOP;
1800   int off = tp->offset();
1801   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1802   Compile* C = phase->C;
1803 
1804   // Try to guess loaded type from pointer type
1805   if (tp->isa_aryptr()) {
1806     const TypeAryPtr* ary = tp->is_aryptr();
1807     const Type* t = ary->elem();
1808 
1809     // Determine whether the reference is beyond the header or not, by comparing
1810     // the offset against the offset of the start of the array's data.
1811     // Different array types begin at slightly different offsets (12 vs. 16).
1812     // We choose T_BYTE as an example base type that is least restrictive
1813     // as to alignment, which will therefore produce the smallest
1814     // possible base offset.
1815     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1816     const bool off_beyond_header = (off >= min_base_off);
1817 
1818     // Try to constant-fold a stable array element.
1819     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1820       // Make sure the reference is not into the header and the offset is constant
1821       ciObject* aobj = ary->const_oop();
1822       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1823         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1824         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1825                                                                       stable_dimension,
1826                                                                       memory_type(), is_unsigned());
1827         if (con_type != NULL) {
1828           return con_type;
1829         }
1830       }
1831     }
1832 
1833     // Don't do this for integer types. There is only potential profit if
1834     // the element type t is lower than _type; that is, for int types, if _type is
1835     // more restrictive than t.  This only happens here if one is short and the other
1836     // char (both 16 bits), and in those cases we've made an intentional decision
1837     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1838     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1839     //
1840     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1841     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1842     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1843     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1844     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1845     // In fact, that could have been the original type of p1, and p1 could have
1846     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1847     // expression (LShiftL quux 3) independently optimized to the constant 8.
1848     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1849         && (_type->isa_vect() == NULL)
1850         && t->isa_inlinetype() == NULL
1851         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1852       // t might actually be lower than _type, if _type is a unique
1853       // concrete subclass of abstract class t.
1854       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1855         const Type* jt = t->join_speculative(_type);
1856         // In any case, do not allow the join, per se, to empty out the type.
1857         if (jt->empty() && !t->empty()) {
1858           // This can happen if a interface-typed array narrows to a class type.
1859           jt = _type;
1860         }
1861 #ifdef ASSERT
1862         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1863           // The pointers in the autobox arrays are always non-null
1864           Node* base = adr->in(AddPNode::Base);
1865           if ((base != NULL) && base->is_DecodeN()) {
1866             // Get LoadN node which loads IntegerCache.cache field
1867             base = base->in(1);
1868           }
1869           if ((base != NULL) && base->is_Con()) {
1870             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1871             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1872               // It could be narrow oop
1873               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1874             }
1875           }
1876         }
1877 #endif
1878         return jt;
1879       }
1880     }
1881   } else if (tp->base() == Type::InstPtr) {
1882     assert( off != Type::OffsetBot ||
1883             // arrays can be cast to Objects
1884             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1885             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1886             // unsafe field access may not have a constant offset
1887             C->has_unsafe_access(),
1888             "Field accesses must be precise" );
1889     // For oop loads, we expect the _type to be precise.
1890 
1891     const TypeInstPtr* tinst = tp->is_instptr();
1892     BasicType bt = memory_type();
1893 
1894     // Optimize loads from constant fields.
1895     ciObject* const_oop = tinst->const_oop();
1896     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1897       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1898       if (mirror_type != NULL && mirror_type->is_inlinetype()) {
1899         ciInlineKlass* vk = mirror_type->as_inline_klass();
1900         if (off == vk->default_value_offset()) {
1901           // Loading a special hidden field that contains the oop of the default inline type
1902           const Type* const_oop = TypeInstPtr::make(vk->default_instance());
1903           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1904         }
1905       }
1906       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1907       if (con_type != NULL) {
1908         return con_type;
1909       }
1910     }
1911   } else if (tp->base() == Type::KlassPtr) {
1912     assert( off != Type::OffsetBot ||
1913             // arrays can be cast to Objects
1914             tp->is_klassptr()->klass() == NULL ||
1915             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1916             // also allow array-loading from the primary supertype
1917             // array during subtype checks
1918             Opcode() == Op_LoadKlass,
1919             "Field accesses must be precise" );
1920     // For klass/static loads, we expect the _type to be precise
1921   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1922     if (adr->is_Load() && off == 0) {
1923       /* With mirrors being an indirect in the Klass*
1924        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1925        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1926        *
1927        * So check the type and klass of the node before the LoadP.
1928        */
1929       Node* adr2 = adr->in(MemNode::Address);
1930       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1931       if (tkls != NULL) {
1932         ciKlass* klass = tkls->klass();
1933         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1934           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1935           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1936           return TypeInstPtr::make(klass->java_mirror());
1937         }
1938       }
1939     } else {
1940       // Check for a load of the default value offset from the InlineKlassFixedBlock:
1941       // LoadI(LoadP(inline_klass, adr_inlineklass_fixed_block_offset), default_value_offset_offset)
1942       intptr_t offset = 0;
1943       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1944       if (base != NULL && base->is_Load() && offset == in_bytes(InlineKlass::default_value_offset_offset())) {
1945         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1946         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_inlinetype() &&
1947             tkls->offset() == in_bytes(InstanceKlass::adr_inlineklass_fixed_block_offset())) {
1948           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1949           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1950           return TypeInt::make(tkls->klass()->as_inline_klass()->default_value_offset());
1951         }
1952       }
1953     }
1954   }
1955 
1956   const TypeKlassPtr *tkls = tp->isa_klassptr();
1957   if (tkls != NULL && !StressReflectiveCode) {
1958     ciKlass* klass = tkls->klass();
1959     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1960       // We are loading a field from a Klass metaobject whose identity
1961       // is known at compile time (the type is "exact" or "precise").
1962       // Check for fields we know are maintained as constants by the VM.
1963       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1964         // The field is Klass::_super_check_offset.  Return its (constant) value.
1965         // (Folds up type checking code.)
1966         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1967         return TypeInt::make(klass->super_check_offset());
1968       }
1969       // Compute index into primary_supers array
1970       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1971       // Check for overflowing; use unsigned compare to handle the negative case.
1972       if( depth < ciKlass::primary_super_limit() ) {
1973         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1974         // (Folds up type checking code.)
1975         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1976         ciKlass *ss = klass->super_of_depth(depth);
1977         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1978       }
1979       const Type* aift = load_array_final_field(tkls, klass);
1980       if (aift != NULL)  return aift;
1981     }
1982 
1983     // We can still check if we are loading from the primary_supers array at a
1984     // shallow enough depth.  Even though the klass is not exact, entries less
1985     // than or equal to its super depth are correct.
1986     if (tkls->is_loaded()) {
1987       ciType *inner = klass;
1988       while( inner->is_obj_array_klass() )
1989         inner = inner->as_obj_array_klass()->base_element_type();
1990       if( inner->is_instance_klass() &&
1991           !inner->as_instance_klass()->flags().is_interface() ) {
1992         // Compute index into primary_supers array
1993         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1994         // Check for overflowing; use unsigned compare to handle the negative case.
1995         if( depth < ciKlass::primary_super_limit() &&
1996             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1997           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1998           // (Folds up type checking code.)
1999           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
2000           ciKlass *ss = klass->super_of_depth(depth);
2001           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
2002         }
2003       }
2004     }
2005 
2006     // If the type is enough to determine that the thing is not an array,
2007     // we can give the layout_helper a positive interval type.
2008     // This will help short-circuit some reflective code.
2009     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2010         && !klass->is_array_klass() // not directly typed as an array
2011         && !klass->is_interface()  // specifically not Serializable & Cloneable
2012         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2013         ) {
2014       // Note:  When interfaces are reliable, we can narrow the interface
2015       // test to (klass != Serializable && klass != Cloneable).
2016       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2017       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2018       // The key property of this type is that it folds up tests
2019       // for array-ness, since it proves that the layout_helper is positive.
2020       // Thus, a generic value like the basic object layout helper works fine.
2021       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2022     }
2023   }
2024 
2025   // If we are loading from a freshly-allocated object, produce a zero,
2026   // if the load is provably beyond the header of the object.
2027   // (Also allow a variable load from a fresh array to produce zero.)
2028   const TypeOopPtr *tinst = tp->isa_oopptr();
2029   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2030   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
2031   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
2032     Node* value = can_see_stored_value(mem,phase);
2033     if (value != NULL && value->is_Con()) {
2034       assert(value->bottom_type()->higher_equal(_type),"sanity");
2035       return value->bottom_type();
2036     }
2037   }
2038 
2039   if (is_instance) {
2040     // If we have an instance type and our memory input is the
2041     // programs's initial memory state, there is no matching store,
2042     // so just return a zero of the appropriate type
2043     Node *mem = in(MemNode::Memory);
2044     if (mem->is_Parm() && mem->in(0)->is_Start()) {
2045       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
2046       return Type::get_zero_type(_type->basic_type());
2047     }
2048   }
2049 
2050   Node* alloc = is_new_object_mark_load(phase);

2218 const Type* LoadNode::klass_value_common(PhaseGVN* phase) const {
2219   // Either input is TOP ==> the result is TOP
2220   const Type *t1 = phase->type( in(MemNode::Memory) );
2221   if (t1 == Type::TOP)  return Type::TOP;
2222   Node *adr = in(MemNode::Address);
2223   const Type *t2 = phase->type( adr );
2224   if (t2 == Type::TOP)  return Type::TOP;
2225   const TypePtr *tp = t2->is_ptr();
2226   if (TypePtr::above_centerline(tp->ptr()) ||
2227       tp->ptr() == TypePtr::Null)  return Type::TOP;
2228 
2229   // Return a more precise klass, if possible
2230   const TypeInstPtr *tinst = tp->isa_instptr();
2231   if (tinst != NULL) {
2232     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
2233     int offset = tinst->offset();
2234     if (ik == phase->C->env()->Class_klass()
2235         && (offset == java_lang_Class::klass_offset() ||
2236             offset == java_lang_Class::array_klass_offset())) {
2237       // We are loading a special hidden field from a Class mirror object,
2238       // the field which points to the VM's Klass metaobject.
2239       ciType* t = tinst->java_mirror_type();
2240       // java_mirror_type returns non-null for compile-time Class constants.
2241       if (t != NULL) {
2242         // constant oop => constant klass
2243         if (offset == java_lang_Class::array_klass_offset()) {
2244           if (t->is_void()) {
2245             // We cannot create a void array.  Since void is a primitive type return null
2246             // klass.  Users of this result need to do a null check on the returned klass.
2247             return TypePtr::NULL_PTR;
2248           }
2249           return TypeKlassPtr::make(ciArrayKlass::make(t));
2250         }
2251         if (!t->is_klass()) {
2252           // a primitive Class (e.g., int.class) has NULL for a klass field
2253           return TypePtr::NULL_PTR;
2254         }
2255         // (Folds up the 1st indirection in aClassConstant.getModifiers().)
2256         return TypeKlassPtr::make(t->as_klass());
2257       }
2258       // non-constant mirror, so we can't tell what's going on
2259     }
2260     if( !ik->is_loaded() )
2261       return _type;             // Bail out if not loaded
2262     if (offset == oopDesc::klass_offset_in_bytes()) {
2263       if (tinst->klass_is_exact()) {
2264         return TypeKlassPtr::make(ik);
2265       }
2266       // See if we can become precise: no subklasses and no interface
2267       // (Note:  We need to support verified interfaces.)
2268       if (!ik->is_interface() && !ik->has_subklass()) {
2269         // Add a dependence; if any subclass added we need to recompile
2270         if (!ik->is_final()) {
2271           // %%% should use stronger assert_unique_concrete_subtype instead
2272           phase->C->dependencies()->assert_leaf_type(ik);
2273         }
2274         // Return precise klass
2275         return TypeKlassPtr::make(ik);
2276       }
2277 
2278       // Return root of possible klass
2279       return TypeKlassPtr::make(TypePtr::NotNull, ik, Type::Offset(0), tinst->flat_array());
2280     }
2281   }
2282 
2283   // Check for loading klass from an array
2284   const TypeAryPtr *tary = tp->isa_aryptr();
2285   if (tary != NULL) {
2286     ciKlass *tary_klass = tary->klass();
2287     if (tary_klass != NULL   // can be NULL when at BOTTOM or TOP
2288         && tary->offset() == oopDesc::klass_offset_in_bytes()) {
2289       if (tary->klass_is_exact()) {
2290         return TypeKlassPtr::make(tary_klass);
2291       }
2292       ciArrayKlass* ak = tary_klass->as_array_klass();
2293       // If the klass is an object array, we defer the question to the
2294       // array component klass.
2295       if (ak->is_obj_array_klass()) {
2296         assert(ak->is_loaded(), "");
2297         ciKlass *base_k = ak->as_obj_array_klass()->base_element_klass();
2298         if (base_k->is_loaded() && base_k->is_instance_klass()) {
2299           ciInstanceKlass *ik = base_k->as_instance_klass();
2300           // See if we can become precise: no subklasses and no interface
2301           if (!ik->is_interface() && !ik->has_subklass()) {
2302             // Add a dependence; if any subclass added we need to recompile
2303             if (!ik->is_final()) {
2304               phase->C->dependencies()->assert_leaf_type(ik);
2305             }
2306             // Return precise array klass
2307             return TypeKlassPtr::make(ak);
2308           }
2309         }
2310         return TypeKlassPtr::make(TypePtr::NotNull, ak, Type::Offset(0), false);
2311       } else if (ak->is_type_array_klass()) {
2312         //assert(!UseExactTypes, "this code should be useless with exact types");
2313         return TypeKlassPtr::make(ak); // These are always precise
2314       }
2315     }
2316   }
2317 
2318   // Check for loading klass from an array klass
2319   const TypeKlassPtr *tkls = tp->isa_klassptr();
2320   if (tkls != NULL && !StressReflectiveCode) {
2321     if (!tkls->is_loaded()) {

2322       return _type;             // Bail out if not loaded
2323     }
2324     ciKlass* klass = tkls->klass();
2325     if( klass->is_obj_array_klass() &&
2326         tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2327       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
2328       // // Always returning precise element type is incorrect,
2329       // // e.g., element type could be object and array may contain strings
2330       // return TypeKlassPtr::make(TypePtr::Constant, elem, 0);
2331 
2332       // The array's TypeKlassPtr was declared 'precise' or 'not precise'
2333       // according to the element type's subclassing.
2334       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), elem->flatten_array());
2335     } else if (klass->is_flat_array_klass() &&
2336                tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2337       ciKlass* elem = klass->as_flat_array_klass()->element_klass();
2338       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), /* flat_array= */ true);
2339     }
2340     if( klass->is_instance_klass() && tkls->klass_is_exact() &&
2341         tkls->offset() == in_bytes(Klass::super_offset())) {
2342       ciKlass* sup = klass->as_instance_klass()->super();
2343       // The field is Klass::_super.  Return its (constant) value.
2344       // (Folds up the 2nd indirection in aClassConstant.getSuperClass().)
2345       return sup ? TypeKlassPtr::make(sup) : TypePtr::NULL_PTR;
2346     }
2347   }
2348

2360 Node* LoadNode::klass_identity_common(PhaseGVN* phase) {
2361   Node* x = LoadNode::Identity(phase);
2362   if (x != this)  return x;
2363 
2364   // Take apart the address into an oop and and offset.
2365   // Return 'this' if we cannot.
2366   Node*    adr    = in(MemNode::Address);
2367   intptr_t offset = 0;
2368   Node*    base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
2369   if (base == NULL)     return this;
2370   const TypeOopPtr* toop = phase->type(adr)->isa_oopptr();
2371   if (toop == NULL)     return this;
2372 
2373   // Step over potential GC barrier for OopHandle resolve
2374   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2375   if (bs->is_gc_barrier_node(base)) {
2376     base = bs->step_over_gc_barrier(base);
2377   }
2378 
2379   // We can fetch the klass directly through an AllocateNode.
2380   // This works even if the klass is not constant (clone or newArray).
2381   if (offset == oopDesc::klass_offset_in_bytes()) {
2382     Node* allocated_klass = AllocateNode::Ideal_klass(base, phase);
2383     if (allocated_klass != NULL) {
2384       return allocated_klass;
2385     }
2386   }
2387 
2388   // Simplify k.java_mirror.as_klass to plain k, where k is a Klass*.
2389   // See inline_native_Class_query for occurrences of these patterns.
2390   // Java Example:  x.getClass().isAssignableFrom(y)
2391   //
2392   // This improves reflective code, often making the Class
2393   // mirror go completely dead.  (Current exception:  Class
2394   // mirrors may appear in debug info, but we could clean them out by
2395   // introducing a new debug info operator for Klass.java_mirror).
2396 
2397   if (toop->isa_instptr() && toop->klass() == phase->C->env()->Class_klass()
2398       && offset == java_lang_Class::klass_offset()) {
2399     if (base->is_Load()) {
2400       Node* base2 = base->in(MemNode::Address);
2401       if (base2->is_Load()) { /* direct load of a load which is the OopHandle */
2402         Node* adr2 = base2->in(MemNode::Address);
2403         const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
2404         if (tkls != NULL && !tkls->empty()
2405             && (tkls->klass()->is_instance_klass() ||
2406               tkls->klass()->is_array_klass())
2407             && adr2->is_AddP()
2408            ) {
2409           int mirror_field = in_bytes(Klass::java_mirror_offset());
2410           if (tkls->offset() == mirror_field) {
2411             return adr2->in(AddPNode::Base);
2412           }
2413         }
2414       }
2415     }
2416   }
2417 
2418   return this;
2419 }

2547   case T_OBJECT:

2593 //------------------------------Ideal------------------------------------------
2594 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2595 // When a store immediately follows a relevant allocation/initialization,
2596 // try to capture it into the initialization, or hoist it above.
2597 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2598   Node* p = MemNode::Ideal_common(phase, can_reshape);
2599   if (p)  return (p == NodeSentinel) ? NULL : p;
2600 
2601   Node* mem     = in(MemNode::Memory);
2602   Node* address = in(MemNode::Address);
2603   // Back-to-back stores to same address?  Fold em up.  Generally
2604   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2605   // since they must follow each StoreP operation.  Redundant StoreCMs
2606   // are eliminated just before matching in final_graph_reshape.
2607   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::INLINES) {
2608     Node* st = mem;
2609     // If Store 'st' has more than one use, we cannot fold 'st' away.
2610     // For example, 'st' might be the final state at a conditional
2611     // return.  Or, 'st' might be used by some node which is live at
2612     // the same time 'st' is live, which might be unschedulable.  So,
2613     // require exactly ONE user until such time as we clone 'mem' for
2614     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2615     // true).
2616     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2617       // Looking at a dead closed cycle of memory?
2618       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2619       assert(Opcode() == st->Opcode() ||
2620              st->Opcode() == Op_StoreVector ||
2621              Opcode() == Op_StoreVector ||
2622              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2623              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2624              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2625              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2626              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2627              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2628 
2629       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2630           st->as_Store()->memory_size() <= this->memory_size()) {
2631         Node* use = st->raw_out(0);
2632         phase->igvn_rehash_node_delayed(use);
2633         if (can_reshape) {
2634           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2635         } else {
2636           // It's OK to do this in the parser, since DU info is always accurate,
2637           // and the parser always refers to nodes via SafePointNode maps.
2638           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2639         }
2640         return this;
2641       }
2642       st = st->in(MemNode::Memory);
2643     }
2644   }
2645 
2646 
2647   // Capture an unaliased, unconditional, simple store into an initializer.
2648   // Or, if it is independent of the allocation, hoist it above the allocation.
2649   if (ReduceFieldZeroing && /*can_reshape &&*/
2650       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2651     InitializeNode* init = mem->in(0)->as_Initialize();
2652     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2653     if (offset > 0) {
2654       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2655       // If the InitializeNode captured me, it made a raw copy of me,
2656       // and I need to disappear.
2657       if (moved != NULL) {
2658         // %%% hack to ensure that Ideal returns a new node:
2659         mem = MergeMemNode::make(mem);
2660         return mem;             // fold me away
2661       }
2662     }
2663   }
2664 
2665   return NULL;                  // No further progress
2666 }

2680 //------------------------------Identity---------------------------------------
2681 // Remove redundant stores:
2682 //   Store(m, p, Load(m, p)) changes to m.
2683 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2684 Node* StoreNode::Identity(PhaseGVN* phase) {
2685   Node* mem = in(MemNode::Memory);
2686   Node* adr = in(MemNode::Address);
2687   Node* val = in(MemNode::ValueIn);
2688 
2689   Node* result = this;
2690 
2691   // Load then Store?  Then the Store is useless
2692   if (val->is_Load() &&
2693       val->in(MemNode::Address)->eqv_uncast(adr) &&
2694       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2695       val->as_Load()->store_Opcode() == Opcode()) {
2696     result = mem;
2697   }
2698 
2699   // Two stores in a row of the same value?
2700   if (result == this &&
2701       mem->is_Store() &&
2702       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2703       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2704       mem->Opcode() == Opcode()) {
2705     result = mem;
2706   }
2707 
2708   // Store of zero anywhere into a freshly-allocated object?
2709   // Then the store is useless.
2710   // (It must already have been captured by the InitializeNode.)
2711   if (result == this && ReduceFieldZeroing) {

2712     // a newly allocated object is already all-zeroes everywhere
2713     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2714         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2715       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to inline type array is forbidden");
2716       result = mem;
2717     }
2718 
2719     if (result == this) {
2720       // the store may also apply to zero-bits in an earlier object
2721       Node* prev_mem = find_previous_store(phase);
2722       // Steps (a), (b):  Walk past independent stores to find an exact match.
2723       if (prev_mem != NULL) {
2724         Node* prev_val = can_see_stored_value(prev_mem, phase);
2725         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2726           // prev_val and val might differ by a cast; it would be good
2727           // to keep the more informative of the two.
2728           if (phase->type(val)->is_zero_type()) {
2729             result = mem;
2730           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2731             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2732             AllocateNode* alloc = init->allocation();
2733             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2734               result = mem;
2735             }
2736           }
2737         }
2738       }
2739     }
2740   }
2741 
2742   if (result != this && phase->is_IterGVN() != NULL) {
2743     MemBarNode* trailing = trailing_membar();
2744     if (trailing != NULL) {
2745 #ifdef ASSERT
2746       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2747       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2748 #endif
2749       PhaseIterGVN* igvn = phase->is_IterGVN();
2750       trailing->remove(igvn);
2751     }
2752   }
2753 
2754   return result;
2755 }

3024 //------------------------------Idealize---------------------------------------
3025 // Clearing a short array is faster with stores
3026 Node *ClearArrayNode::Ideal(PhaseGVN *phase, bool can_reshape) {
3027   // Already know this is a large node, do not try to ideal it
3028   if (!IdealizeClearArrayNode || _is_large) return NULL;
3029 
3030   const int unit = BytesPerLong;
3031   const TypeX* t = phase->type(in(2))->isa_intptr_t();
3032   if (!t)  return NULL;
3033   if (!t->is_con())  return NULL;
3034   intptr_t raw_count = t->get_con();
3035   intptr_t size = raw_count;
3036   if (!Matcher::init_array_count_is_in_bytes) size *= unit;
3037   // Clearing nothing uses the Identity call.
3038   // Negative clears are possible on dead ClearArrays
3039   // (see jck test stmt114.stmt11402.val).
3040   if (size <= 0 || size % unit != 0)  return NULL;
3041   intptr_t count = size / unit;
3042   // Length too long; communicate this to matchers and assemblers.
3043   // Assemblers are responsible to produce fast hardware clears for it.
3044   if (size > InitArrayShortSize) {
3045     return new ClearArrayNode(in(0), in(1), in(2), in(3), in(4), true);
3046   }
3047   Node *mem = in(1);
3048   if( phase->type(mem)==Type::TOP ) return NULL;
3049   Node *adr = in(3);
3050   const Type* at = phase->type(adr);
3051   if( at==Type::TOP ) return NULL;
3052   const TypePtr* atp = at->isa_ptr();
3053   // adjust atp to be the correct array element address type
3054   if (atp == NULL)  atp = TypePtr::BOTTOM;
3055   else              atp = atp->add_offset(Type::OffsetBot);
3056   // Get base for derived pointer purposes
3057   if( adr->Opcode() != Op_AddP ) Unimplemented();
3058   Node *base = adr->in(1);
3059 
3060   Node *val = in(4);
3061   Node *off  = phase->MakeConX(BytesPerLong);
3062   mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3063   count--;
3064   while( count-- ) {
3065     mem = phase->transform(mem);
3066     adr = phase->transform(new AddPNode(base,adr,off));
3067     mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3068   }

3469 //===========================InitializeNode====================================
3470 // SUMMARY:
3471 // This node acts as a memory barrier on raw memory, after some raw stores.
3472 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3473 // The Initialize can 'capture' suitably constrained stores as raw inits.
3474 // It can coalesce related raw stores into larger units (called 'tiles').
3475 // It can avoid zeroing new storage for memory units which have raw inits.
3476 // At macro-expansion, it is marked 'complete', and does not optimize further.
3477 //
3478 // EXAMPLE:
3479 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3480 //   ctl = incoming control; mem* = incoming memory
3481 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3482 // First allocate uninitialized memory and fill in the header:
3483 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3484 //   ctl := alloc.Control; mem* := alloc.Memory*
3485 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3486 // Then initialize to zero the non-header parts of the raw memory block:
3487 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3488 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3489 // After the initialize node executes, the object is ready for service:
3490 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3491 // Suppose its body is immediately initialized as {1,2}:
3492 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3493 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3494 //   mem.SLICE(#short[*]) := store2
3495 //
3496 // DETAILS:
3497 // An InitializeNode collects and isolates object initialization after
3498 // an AllocateNode and before the next possible safepoint.  As a
3499 // memory barrier (MemBarNode), it keeps critical stores from drifting
3500 // down past any safepoint or any publication of the allocation.
3501 // Before this barrier, a newly-allocated object may have uninitialized bits.
3502 // After this barrier, it may be treated as a real oop, and GC is allowed.
3503 //
3504 // The semantics of the InitializeNode include an implicit zeroing of
3505 // the new object from object header to the end of the object.
3506 // (The object header and end are determined by the AllocateNode.)
3507 //
3508 // Certain stores may be added as direct inputs to the InitializeNode.
3509 // These stores must update raw memory, and they must be to addresses
3510 // derived from the raw address produced by AllocateNode, and with
3511 // a constant offset.  They must be ordered by increasing offset.
3512 // The first one is at in(RawStores), the last at in(req()-1).
3513 // Unlike most memory operations, they are not linked in a chain,
3514 // but are displayed in parallel as users of the rawmem output of
3515 // the allocation.
3516 //
3517 // (See comments in InitializeNode::capture_store, which continue
3518 // the example given above.)
3519 //
3520 // When the associated Allocate is macro-expanded, the InitializeNode
3521 // may be rewritten to optimize collected stores.  A ClearArrayNode
3522 // may also be created at that point to represent any required zeroing.
3523 // The InitializeNode is then marked 'complete', prohibiting further
3524 // capturing of nearby memory operations.
3525 //
3526 // During macro-expansion, all captured initializations which store
3527 // constant values of 32 bits or smaller are coalesced (if advantageous)
3528 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3529 // initialized in fewer memory operations.  Memory words which are
3530 // covered by neither tiles nor non-constant stores are pre-zeroed
3531 // by explicit stores of zero.  (The code shape happens to do all
3532 // zeroing first, then all other stores, with both sequences occurring
3533 // in order of ascending offsets.)
3534 //
3535 // Alternatively, code may be inserted between an AllocateNode and its
3536 // InitializeNode, to perform arbitrary initialization of the new object.
3537 // E.g., the object copying intrinsics insert complex data transfers here.
3538 // The initialization must then be marked as 'complete' disable the
3539 // built-in zeroing semantics and the collection of initializing stores.
3540 //
3541 // While an InitializeNode is incomplete, reads from the memory state
3542 // produced by it are optimizable if they match the control edge and
3543 // new oop address associated with the allocation/initialization.
3544 // They return a stored value (if the offset matches) or else zero.
3545 // A write to the memory state, if it matches control and address,
3546 // and if it is to a constant offset, may be 'captured' by the
3547 // InitializeNode.  It is cloned as a raw memory operation and rewired
3548 // inside the initialization, to the raw oop produced by the allocation.
3549 // Operations on addresses which are provably distinct (e.g., to
3550 // other AllocateNodes) are allowed to bypass the initialization.
3551 //
3552 // The effect of all this is to consolidate object initialization
3553 // (both arrays and non-arrays, both piecewise and bulk) into a
3554 // single location, where it can be optimized as a unit.
3555 //
3556 // Only stores with an offset less than TrackedInitializationLimit words
3557 // will be considered for capture by an InitializeNode.  This puts a
3558 // reasonable limit on the complexity of optimized initializations.
3559

3656 //     { Foo p = new Foo(); p.next = p; }
3657 // True for constants and parameters and small combinations thereof.
3658 bool InitializeNode::detect_init_independence(Node* value, PhaseGVN* phase) {
3659   ResourceMark rm;
3660   Unique_Node_List worklist;
3661   worklist.push(value);
3662 
3663   uint complexity_limit = 20;
3664   for (uint j = 0; j < worklist.size(); j++) {
3665     if (j >= complexity_limit) {
3666       return false;  // Bail out if processed too many nodes
3667     }
3668 
3669     Node* n = worklist.at(j);
3670     if (n == NULL)      continue;   // (can this really happen?)
3671     if (n->is_Proj())   n = n->in(0);
3672     if (n == this)      return false;  // found a cycle
3673     if (n->is_Con())    continue;
3674     if (n->is_Start())  continue;   // params, etc., are OK
3675     if (n->is_Root())   continue;   // even better
3676 
3677     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3678     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3679       continue;
3680     }
3681 
3682     Node* ctl = n->in(0);
3683     if (ctl != NULL && !ctl->is_top()) {
3684       if (ctl->is_Proj())  ctl = ctl->in(0);
3685       if (ctl == this)  return false;
3686 
3687       // If we already know that the enclosing memory op is pinned right after
3688       // the init, then any control flow that the store has picked up
3689       // must have preceded the init, or else be equal to the init.
3690       // Even after loop optimizations (which might change control edges)
3691       // a store is never pinned *before* the availability of its inputs.
3692       if (!MemNode::all_controls_dominate(n, this))
3693         return false;                  // failed to prove a good control
3694     }
3695 
3696     // Check data edges for possible dependencies on 'this'.
3697     for (uint i = 1; i < n->req(); i++) {
3698       Node* m = n->in(i);
3699       if (m == NULL || m == n || m->is_top())  continue;
3700 
3701       // Only process data inputs once
3702       worklist.push(m);
3703     }
3704   }
3705 
3706   return true;
3707 }

3709 // Here are all the checks a Store must pass before it can be moved into
3710 // an initialization.  Returns zero if a check fails.
3711 // On success, returns the (constant) offset to which the store applies,
3712 // within the initialized memory.
3713 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3714   const int FAIL = 0;
3715   if (st->req() != MemNode::ValueIn + 1)
3716     return FAIL;                // an inscrutable StoreNode (card mark?)
3717   Node* ctl = st->in(MemNode::Control);
3718   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3719     return FAIL;                // must be unconditional after the initialization
3720   Node* mem = st->in(MemNode::Memory);
3721   if (!(mem->is_Proj() && mem->in(0) == this))
3722     return FAIL;                // must not be preceded by other stores
3723   Node* adr = st->in(MemNode::Address);
3724   intptr_t offset;
3725   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3726   if (alloc == NULL)
3727     return FAIL;                // inscrutable address
3728   if (alloc != allocation())
3729     return FAIL;                // wrong allocation!  (store needs to float up)
3730   int size_in_bytes = st->memory_size();
3731   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3732     return FAIL;                // mismatched access
3733   }
3734   Node* val = st->in(MemNode::ValueIn);
3735 
3736   if (!detect_init_independence(val, phase))
3737     return FAIL;                // stored value must be 'simple enough'
3738 
3739   // The Store can be captured only if nothing after the allocation
3740   // and before the Store is using the memory location that the store
3741   // overwrites.
3742   bool failed = false;
3743   // If is_complete_with_arraycopy() is true the shape of the graph is
3744   // well defined and is safe so no need for extra checks.
3745   if (!is_complete_with_arraycopy()) {
3746     // We are going to look at each use of the memory state following
3747     // the allocation to make sure nothing reads the memory that the
3748     // Store writes.
3749     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3750     int alias_idx = phase->C->get_alias_index(t_adr);
3751     ResourceMark rm;
3752     Unique_Node_List mems;
3753     mems.push(mem);
3754     Node* unique_merge = NULL;
3755     for (uint next = 0; next < mems.size(); ++next) {
3756       Node *m  = mems.at(next);
3757       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3758         Node *n = m->fast_out(j);
3759         if (n->outcnt() == 0) {
3760           continue;
3761         }
3762         if (n == st) {
3763           continue;
3764         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3765           // If the control of this use is different from the control
3766           // of the Store which is right after the InitializeNode then
3767           // this node cannot be between the InitializeNode and the
3768           // Store.
3769           continue;
3770         } else if (n->is_MergeMem()) {
3771           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3772             // We can hit a MergeMemNode (that will likely go away
3773             // later) that is a direct use of the memory state
3774             // following the InitializeNode on the same slice as the
3775             // store node that we'd like to capture. We need to check
3776             // the uses of the MergeMemNode.
3777             mems.push(n);
3778           }
3779         } else if (n->is_Mem()) {
3780           Node* other_adr = n->in(MemNode::Address);
3781           if (other_adr == adr) {
3782             failed = true;
3783             break;
3784           } else {
3785             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3786             if (other_t_adr != NULL) {
3787               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3788               if (other_alias_idx == alias_idx) {
3789                 // A load from the same memory slice as the store right
3790                 // after the InitializeNode. We check the control of the
3791                 // object/array that is loaded from. If it's the same as
3792                 // the store control then we cannot capture the store.
3793                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3794                 Node* base = other_adr;
3795                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3796                 base = base->in(AddPNode::Base);
3797                 if (base != NULL) {
3798                   base = base->uncast();
3799                   if (base->is_Proj() && base->in(0) == alloc) {
3800                     failed = true;
3801                     break;
3802                   }
3803                 }
3804               }
3805             }
3806           }
3807         } else {
3808           failed = true;
3809           break;
3810         }
3811       }
3812     }
3813   }
3814   if (failed) {
3815     if (!can_reshape) {
3816       // We decided we couldn't capture the store during parsing. We
3817       // should try again during the next IGVN once the graph is
3818       // cleaner.
3819       phase->C->record_for_igvn(st);
3820     }
3821     return FAIL;
3822   }
3823 
3824   return offset;                // success
3825 }

3827 // Find the captured store in(i) which corresponds to the range
3828 // [start..start+size) in the initialized object.
3829 // If there is one, return its index i.  If there isn't, return the
3830 // negative of the index where it should be inserted.
3831 // Return 0 if the queried range overlaps an initialization boundary
3832 // or if dead code is encountered.
3833 // If size_in_bytes is zero, do not bother with overlap checks.
3834 int InitializeNode::captured_store_insertion_point(intptr_t start,
3835                                                    int size_in_bytes,
3836                                                    PhaseTransform* phase) {
3837   const int FAIL = 0, MAX_STORE = BytesPerLong;
3838 
3839   if (is_complete())
3840     return FAIL;                // arraycopy got here first; punt
3841 
3842   assert(allocation() != NULL, "must be present");
3843 
3844   // no negatives, no header fields:
3845   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3846 
3847   // after a certain size, we bail out on tracking all the stores:
3848   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3849   if (start >= ti_limit)  return FAIL;
3850 
3851   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3852     if (i >= limit)  return -(int)i; // not found; here is where to put it
3853 
3854     Node*    st     = in(i);
3855     intptr_t st_off = get_store_offset(st, phase);
3856     if (st_off < 0) {
3857       if (st != zero_memory()) {
3858         return FAIL;            // bail out if there is dead garbage
3859       }
3860     } else if (st_off > start) {
3861       // ...we are done, since stores are ordered
3862       if (st_off < start + size_in_bytes) {
3863         return FAIL;            // the next store overlaps
3864       }
3865       return -(int)i;           // not found; here is where to put it
3866     } else if (st_off < start) {
3867       if (size_in_bytes != 0 &&
3868           start < st_off + MAX_STORE &&
3869           start < st_off + st->as_Store()->memory_size()) {
3870         return FAIL;            // the previous store overlaps
3871       }
3872     } else {
3873       if (size_in_bytes != 0 &&
3874           st->as_Store()->memory_size() != size_in_bytes) {
3875         return FAIL;            // mismatched store size
3876       }
3877       return i;
3878     }
3879 
3880     ++i;
3881   }
3882 }

3915 // Clone the given store, converting it into a raw store
3916 // initializing a field or element of my new object.
3917 // Caller is responsible for retiring the original store,
3918 // with subsume_node or the like.
3919 //
3920 // From the example above InitializeNode::InitializeNode,
3921 // here are the old stores to be captured:
3922 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3923 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3924 //
3925 // Here is the changed code; note the extra edges on init:
3926 //   alloc = (Allocate ...)
3927 //   rawoop = alloc.RawAddress
3928 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3929 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3930 //   init = (Initialize alloc.Control alloc.Memory rawoop
3931 //                      rawstore1 rawstore2)
3932 //
3933 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3934                                     PhaseGVN* phase, bool can_reshape) {
3935   assert(stores_are_sane(phase), "");
3936 
3937   if (start < 0)  return NULL;
3938   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3939 
3940   Compile* C = phase->C;
3941   int size_in_bytes = st->memory_size();
3942   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3943   if (i == 0)  return NULL;     // bail out
3944   Node* prev_mem = NULL;        // raw memory for the captured store
3945   if (i > 0) {
3946     prev_mem = in(i);           // there is a pre-existing store under this one
3947     set_req(i, C->top());       // temporarily disconnect it
3948     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3949   } else {
3950     i = -i;                     // no pre-existing store
3951     prev_mem = zero_memory();   // a slice of the newly allocated object
3952     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3953       set_req(--i, C->top());   // reuse this edge; it has been folded away
3954     else
3955       ins_req(i, C->top());     // build a new edge
3956   }
3957   Node* new_st = st->clone();
3958   new_st->set_req(MemNode::Control, in(Control));
3959   new_st->set_req(MemNode::Memory,  prev_mem);
3960   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3961   new_st = phase->transform(new_st);
3962 
3963   // At this point, new_st might have swallowed a pre-existing store
3964   // at the same offset, or perhaps new_st might have disappeared,
3965   // if it redundantly stored the same value (or zero to fresh memory).
3966 
3967   // In any case, wire it in:
3968   phase->igvn_rehash_node_delayed(this);
3969   set_req(i, new_st);
3970 
3971   // The caller may now kill the old guy.
3972   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3973   assert(check_st == new_st || check_st == NULL, "must be findable");
3974   assert(!is_complete(), "");
3975   return new_st;
3976 }

3978 static bool store_constant(jlong* tiles, int num_tiles,
3979                            intptr_t st_off, int st_size,
3980                            jlong con) {
3981   if ((st_off & (st_size-1)) != 0)
3982     return false;               // strange store offset (assume size==2**N)
3983   address addr = (address)tiles + st_off;
3984   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
3985   switch (st_size) {
3986   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
3987   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
3988   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
3989   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3990   default: return false;        // strange store size (detect size!=2**N here)
3991   }
3992   return true;                  // return success to caller
3993 }

3995 // Coalesce subword constants into int constants and possibly
3996 // into long constants.  The goal, if the CPU permits,
3997 // is to initialize the object with a small number of 64-bit tiles.
3998 // Also, convert floating-point constants to bit patterns.
3999 // Non-constants are not relevant to this pass.
4000 //
4001 // In terms of the running example on InitializeNode::InitializeNode
4002 // and InitializeNode::capture_store, here is the transformation
4003 // of rawstore1 and rawstore2 into rawstore12:
4004 //   alloc = (Allocate ...)
4005 //   rawoop = alloc.RawAddress
4006 //   tile12 = 0x00010002
4007 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4008 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4009 //
4010 void
4011 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4012                                         Node* size_in_bytes,
4013                                         PhaseGVN* phase) {
4014   Compile* C = phase->C;
4015 
4016   assert(stores_are_sane(phase), "");
4017   // Note:  After this pass, they are not completely sane,
4018   // since there may be some overlaps.
4019 
4020   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4021 
4022   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4023   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4024   size_limit = MIN2(size_limit, ti_limit);
4025   size_limit = align_up(size_limit, BytesPerLong);
4026   int num_tiles = size_limit / BytesPerLong;
4027 
4028   // allocate space for the tile map:
4029   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4030   jlong  tiles_buf[small_len];
4031   Node*  nodes_buf[small_len];
4032   jlong  inits_buf[small_len];

4034                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));

4036                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4037   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4038                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));

4040   // nodes: last constant-storing node subsumed into the tiles model
4041   // inits: which bytes (in each tile) are touched by any initializations
4042 
4043   //// Pass A: Fill in the tile model with any relevant stores.
4044

4069     case T_INT:    con = val->is_int()->get_con();  break;
4070     case T_LONG:   con = val->is_long()->get_con(); break;
4071     case T_FLOAT:  con = jint_cast(val->getf());    break;
4072     case T_DOUBLE: con = jlong_cast(val->getd());   break;

4082     store_constant(tiles, num_tiles, st_off, st_size, con);

4085 
4086     if (type == T_INT && st_size == BytesPerInt

4090           st->Opcode() == Op_StoreI) {

4092         jint* intcon = (jint*) &tiles[j];
4093         intcon[1] = 0;  // undo the store_constant()

4096         // undo the action of the previous loop iteration... if we can.
4097         // But if we can't, just let the previous half take care of itself.
4098         st = nodes[j];
4099         st_off -= BytesPerInt;
4100         con = intcon[0];

4102           assert(st_off >= header_size, "still ignoring header");
4103           assert(get_store_offset(st, phase) == st_off, "must be");
4104           assert(in(i-1) == zmem, "must be");

4106           assert(con == tcon->is_int()->get_con(), "must be");

4108           intcon[0] = 0;        // undo store_constant()

4110           nodes[j] = NULL;      // undo nodes[j] = st

4142       init0 = u.intcon[0];

4153       assert(offset + BytesPerInt >= header_size, "second int counts");
4154       assert(*(jint*)&tiles[j] == 0, "junk in header");

4156       // Example:  int a[] = { 42 ... }
4157     } else if (con0 == 0 && init0 == -1) {
4158       split = true;             // first word is covered by full inits
4159       // Example:  int a[] = { ... foo(), 42 ... }
4160     } else if (con1 == 0 && init1 == -1) {
4161       split = true;             // second word is covered by full inits
4162       // Example:  int a[] = { ... 42, foo() ... }

4166     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4167     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4168     // In this case the tile is not split; it is (jlong)42.
4169     // The big tile is stored down, and then the foo() value is inserted.
4170     // (If there were foo(),foo() instead of foo(),0, init0 would be -1.)

4232 // Explore forward from in(start) to find the first fully initialized
4233 // word, and return its offset.  Skip groups of subword stores which
4234 // together initialize full words.  If in(start) is itself part of a
4235 // fully initialized word, return the offset of in(start).  If there
4236 // are no following full-word stores, or if something is fishy, return
4237 // a negative value.
4238 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4239   int       int_map = 0;
4240   intptr_t  int_map_off = 0;
4241   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4242 
4243   for (uint i = start, limit = req(); i < limit; i++) {
4244     Node* st = in(i);
4245 
4246     intptr_t st_off = get_store_offset(st, phase);
4247     if (st_off < 0)  break;  // return conservative answer
4248 
4249     int st_size = st->as_Store()->memory_size();
4250     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4251       return st_off;            // we found a complete word init
4252     }
4253 
4254     // update the map:
4255 
4256     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4257     if (this_int_off != int_map_off) {
4258       // reset the map:
4259       int_map = 0;
4260       int_map_off = this_int_off;
4261     }
4262 
4263     int subword_off = st_off - this_int_off;
4264     int_map |= right_n_bits(st_size) << subword_off;
4265     if ((int_map & FULL_MAP) == FULL_MAP) {
4266       return this_int_off;      // we found a complete word init
4267     }
4268 
4269     // Did this store hit or cross the word boundary?
4270     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4271     if (next_int_off == this_int_off + BytesPerInt) {
4272       // We passed the current int, without fully initializing it.
4273       int_map_off = next_int_off;
4274       int_map >>= BytesPerInt;
4275     } else if (next_int_off > this_int_off + BytesPerInt) {
4276       // We passed the current and next int.
4277       return this_int_off + BytesPerInt;
4278     }
4279   }
4280 
4281   return -1;
4282 }

4286 // At this point, we may perform additional optimizations.
4287 // Linearize the stores by ascending offset, to make memory
4288 // activity as coherent as possible.
4289 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4290                                       intptr_t header_size,
4291                                       Node* size_in_bytes,
4292                                       PhaseIterGVN* phase) {
4293   assert(!is_complete(), "not already complete");
4294   assert(stores_are_sane(phase), "");
4295   assert(allocation() != NULL, "must be present");
4296 
4297   remove_extra_zeroes();
4298 
4299   if (ReduceFieldZeroing || ReduceBulkZeroing)
4300     // reduce instruction count for common initialization patterns
4301     coalesce_subword_stores(header_size, size_in_bytes, phase);
4302 
4303   Node* zmem = zero_memory();   // initially zero memory state
4304   Node* inits = zmem;           // accumulating a linearized chain of inits
4305   #ifdef ASSERT
4306   intptr_t first_offset = allocation()->minimum_header_size();
4307   intptr_t last_init_off = first_offset;  // previous init offset
4308   intptr_t last_init_end = first_offset;  // previous init offset+size
4309   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4310   #endif
4311   intptr_t zeroes_done = header_size;
4312 
4313   bool do_zeroing = true;       // we might give up if inits are very sparse
4314   int  big_init_gaps = 0;       // how many large gaps have we seen?
4315 
4316   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4317   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4318 
4319   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4320     Node* st = in(i);
4321     intptr_t st_off = get_store_offset(st, phase);
4322     if (st_off < 0)
4323       break;                    // unknown junk in the inits
4324     if (st->in(MemNode::Memory) != zmem)
4325       break;                    // complicated store chains somehow in list
4326 
4327     int st_size = st->as_Store()->memory_size();
4328     intptr_t next_init_off = st_off + st_size;
4329 
4330     if (do_zeroing && zeroes_done < next_init_off) {
4331       // See if this store needs a zero before it or under it.
4332       intptr_t zeroes_needed = st_off;
4333 
4334       if (st_size < BytesPerInt) {
4335         // Look for subword stores which only partially initialize words.
4336         // If we find some, we must lay down some word-level zeroes first,
4337         // underneath the subword stores.
4338         //
4339         // Examples:
4340         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4341         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4342         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4343         //
4344         // Note:  coalesce_subword_stores may have already done this,
4345         // if it was prompted by constant non-zero subword initializers.
4346         // But this case can still arise with non-constant stores.
4347 
4348         intptr_t next_full_store = find_next_fullword_store(i, phase);
4349 
4350         // In the examples above:
4351         //   in(i)          p   q   r   s     x   y     z
4352         //   st_off        12  13  14  15    12  13    14
4353         //   st_size        1   1   1   1     1   1     1
4354         //   next_full_s.  12  16  16  16    16  16    16
4355         //   z's_done      12  16  16  16    12  16    12
4356         //   z's_needed    12  16  16  16    16  16    16
4357         //   zsize          0   0   0   0     4   0     4
4358         if (next_full_store < 0) {
4359           // Conservative tack:  Zero to end of current word.
4360           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4361         } else {
4362           // Zero to beginning of next fully initialized word.
4363           // Or, don't zero at all, if we are already in that word.
4364           assert(next_full_store >= zeroes_needed, "must go forward");
4365           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4366           zeroes_needed = next_full_store;
4367         }
4368       }
4369 
4370       if (zeroes_needed > zeroes_done) {
4371         intptr_t zsize = zeroes_needed - zeroes_done;
4372         // Do some incremental zeroing on rawmem, in parallel with inits.
4373         zeroes_done = align_down(zeroes_done, BytesPerInt);
4374         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4375                                               allocation()->in(AllocateNode::DefaultValue),
4376                                               allocation()->in(AllocateNode::RawDefaultValue),
4377                                               zeroes_done, zeroes_needed,
4378                                               phase);
4379         zeroes_done = zeroes_needed;
4380         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4381           do_zeroing = false;   // leave the hole, next time
4382       }
4383     }
4384 
4385     // Collect the store and move on:
4386     phase->replace_input_of(st, MemNode::Memory, inits);
4387     inits = st;                 // put it on the linearized chain
4388     set_req(i, zmem);           // unhook from previous position
4389 
4390     if (zeroes_done == st_off)
4391       zeroes_done = next_init_off;
4392 
4393     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4394 
4395     #ifdef ASSERT
4396     // Various order invariants.  Weaker than stores_are_sane because
4397     // a large constant tile can be filled in by smaller non-constant stores.
4398     assert(st_off >= last_init_off, "inits do not reverse");
4399     last_init_off = st_off;
4400     const Type* val = NULL;
4401     if (st_size >= BytesPerInt &&
4402         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4403         (int)val->basic_type() < (int)T_OBJECT) {
4404       assert(st_off >= last_tile_end, "tiles do not overlap");
4405       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4406       last_tile_end = MAX2(last_tile_end, next_init_off);
4407     } else {
4408       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4409       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4410       assert(st_off      >= last_init_end, "inits do not overlap");
4411       last_init_end = next_init_off;  // it's a non-tile
4412     }
4413     #endif //ASSERT
4414   }
4415 
4416   remove_extra_zeroes();        // clear out all the zmems left over
4417   add_req(inits);
4418 
4419   if (!(UseTLAB && ZeroTLAB)) {
4420     // If anything remains to be zeroed, zero it all now.
4421     zeroes_done = align_down(zeroes_done, BytesPerInt);
4422     // if it is the last unused 4 bytes of an instance, forget about it
4423     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4424     if (zeroes_done + BytesPerLong >= size_limit) {
4425       AllocateNode* alloc = allocation();
4426       assert(alloc != NULL, "must be present");
4427       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4428         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4429         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4430         if (zeroes_done == k->layout_helper())
4431           zeroes_done = size_limit;
4432       }
4433     }
4434     if (zeroes_done < size_limit) {
4435       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4436                                             allocation()->in(AllocateNode::DefaultValue),
4437                                             allocation()->in(AllocateNode::RawDefaultValue),
4438                                             zeroes_done, size_in_bytes, phase);
4439     }
4440   }
4441 
4442   set_complete(phase);
4443   return rawmem;
4444 }

4472 //============================MergeMemNode=====================================
4473 //
4474 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4475 // contributing store or call operations.  Each contributor provides the memory
4476 // state for a particular "alias type" (see Compile::alias_type).  For example,
4477 // if a MergeMem has an input X for alias category #6, then any memory reference
4478 // to alias category #6 may use X as its memory state input, as an exact equivalent
4479 // to using the MergeMem as a whole.
4480 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4481 //
4482 // (Here, the <N> notation gives the index of the relevant adr_type.)
4483 //
4484 // In one special case (and more cases in the future), alias categories overlap.
4485 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4486 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4487 // it is exactly equivalent to that state W:
4488 //   MergeMem(<Bot>: W) <==> W
4489 //
4490 // Usually, the merge has more than one input.  In that case, where inputs
4491 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4492 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4493 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4494 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4495 //
4496 // A merge can take a "wide" memory state as one of its narrow inputs.
4497 // This simply means that the merge observes out only the relevant parts of
4498 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4499 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4500 //
4501 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4502 // and that memory slices "leak through":
4503 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4504 //
4505 // But, in such a cascade, repeated memory slices can "block the leak":
4506 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4507 //
4508 // In the last example, Y is not part of the combined memory state of the
4509 // outermost MergeMem.  The system must, of course, prevent unschedulable
4510 // memory states from arising, so you can be sure that the state Y is somehow
4511 // a precursor to state Y'.
4512 //
4513 //
4514 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4515 // of each MergeMemNode array are exactly the numerical alias indexes, including
4516 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4517 // Compile::alias_type (and kin) produce and manage these indexes.
4518 //
4519 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4520 // (Note that this provides quick access to the top node inside MergeMem methods,
4521 // without the need to reach out via TLS to Compile::current.)
4522 //
4523 // As a consequence of what was just described, a MergeMem that represents a full
4524 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4525 // containing all alias categories.
4526 //
4527 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4528 //
4529 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4530 // a memory state for the alias type <N>, or else the top node, meaning that
4531 // there is no particular input for that alias type.  Note that the length of
4532 // a MergeMem is variable, and may be extended at any time to accommodate new
4533 // memory states at larger alias indexes.  When merges grow, they are of course
4534 // filled with "top" in the unused in() positions.
4535 //
4536 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4537 // (Top was chosen because it works smoothly with passes like GCM.)
4538 //
4539 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4540 // the type of random VM bits like TLS references.)  Since it is always the
4541 // first non-Bot memory slice, some low-level loops use it to initialize an
4542 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4543 //
4544 //
4545 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4546 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4547 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4548 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4549 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4550 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4551 //
4552 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4553 // really that different from the other memory inputs.  An abbreviation called
4554 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4555 //
4556 //
4557 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4558 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4559 // that "emerges though" the base memory will be marked as excluding the alias types
4560 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4561 //
4562 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4563 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4564 //
4565 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4566 // (It is currently unimplemented.)  As you can see, the resulting merge is
4567 // actually a disjoint union of memory states, rather than an overlay.
4568 //

4631 // This method is invoked recursively on chains of MergeMem nodes
4632 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4633   // Remove chain'd MergeMems
4634   //
4635   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4636   // relative to the "in(Bot)".  Since we are patching both at the same time,
4637   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4638   // but rewrite each "in(i)" relative to the new "in(Bot)".
4639   Node *progress = NULL;
4640 
4641 
4642   Node* old_base = base_memory();
4643   Node* empty_mem = empty_memory();
4644   if (old_base == empty_mem)
4645     return NULL; // Dead memory path.
4646 
4647   MergeMemNode* old_mbase;
4648   if (old_base != NULL && old_base->is_MergeMem())
4649     old_mbase = old_base->as_MergeMem();
4650   else
4651     old_mbase = NULL;
4652   Node* new_base = old_base;
4653 
4654   // simplify stacked MergeMems in base memory
4655   if (old_mbase)  new_base = old_mbase->base_memory();
4656 
4657   // the base memory might contribute new slices beyond my req()
4658   if (old_mbase)  grow_to_match(old_mbase);
4659 
4660   // Look carefully at the base node if it is a phi.
4661   PhiNode* phi_base;
4662   if (new_base != NULL && new_base->is_Phi())
4663     phi_base = new_base->as_Phi();
4664   else
4665     phi_base = NULL;
4666 
4667   Node*    phi_reg = NULL;
4668   uint     phi_len = (uint)-1;
4669   if (phi_base != NULL && !phi_base->is_copy()) {
4670     // do not examine phi if degraded to a copy
4671     phi_reg = phi_base->region();
4672     phi_len = phi_base->req();
4673     // see if the phi is unfinished
4674     for (uint i = 1; i < phi_len; i++) {
4675       if (phi_base->in(i) == NULL) {
4676         // incomplete phi; do not look at it yet!
4677         phi_reg = NULL;
4678         phi_len = (uint)-1;
4679         break;
4680       }
4681     }
4682   }
4683 
4684   // Note:  We do not call verify_sparse on entry, because inputs
4685   // can normalize to the base_memory via subsume_node or similar
4686   // mechanisms.  This method repairs that damage.
4687 
4688   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4689 
4690   // Look at each slice.
4691   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4692     Node* old_in = in(i);
4693     // calculate the old memory value
4694     Node* old_mem = old_in;
4695     if (old_mem == empty_mem)  old_mem = old_base;
4696     assert(old_mem == memory_at(i), "");
4697 
4698     // maybe update (reslice) the old memory value
4699 
4700     // simplify stacked MergeMems
4701     Node* new_mem = old_mem;
4702     MergeMemNode* old_mmem;
4703     if (old_mem != NULL && old_mem->is_MergeMem())
4704       old_mmem = old_mem->as_MergeMem();
4705     else
4706       old_mmem = NULL;
4707     if (old_mmem == this) {
4708       // This can happen if loops break up and safepoints disappear.
4709       // A merge of BotPtr (default) with a RawPtr memory derived from a
4710       // safepoint can be rewritten to a merge of the same BotPtr with
4711       // the BotPtr phi coming into the loop.  If that phi disappears
4712       // also, we can end up with a self-loop of the mergemem.
4713       // In general, if loops degenerate and memory effects disappear,
4714       // a mergemem can be left looking at itself.  This simply means
4715       // that the mergemem's default should be used, since there is
4716       // no longer any apparent effect on this slice.
4717       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4718       //       from start.  Update the input to TOP.
4719       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4720     }
4721     else if (old_mmem != NULL) {
4722       new_mem = old_mmem->memory_at(i);
4723     }
4724     // else preceding memory was not a MergeMem
4725 
4726     // replace equivalent phis (unfortunately, they do not GVN together)
4727     if (new_mem != NULL && new_mem != new_base &&
4728         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4729       if (new_mem->is_Phi()) {
4730         PhiNode* phi_mem = new_mem->as_Phi();
4731         for (uint i = 1; i < phi_len; i++) {
4732           if (phi_base->in(i) != phi_mem->in(i)) {
4733             phi_mem = NULL;
4734             break;
4735           }
4736         }
4737         if (phi_mem != NULL) {
4738           // equivalent phi nodes; revert to the def
4739           new_mem = new_base;
4740         }
4741       }
4742     }
4743 
4744     // maybe store down a new value
4745     Node* new_in = new_mem;
4746     if (new_in == new_base)  new_in = empty_mem;
4747 
4748     if (new_in != old_in) {
4749       // Warning:  Do not combine this "if" with the previous "if"
4750       // A memory slice might have be be rewritten even if it is semantically
4751       // unchanged, if the base_memory value has changed.
4752       set_req(i, new_in);
4753       progress = this;          // Report progress
4754     }
4755   }
4756 
4757   if (new_base != old_base) {
4758     set_req(Compile::AliasIdxBot, new_base);
4759     // Don't use set_base_memory(new_base), because we need to update du.
4760     assert(base_memory() == new_base, "");
4761     progress = this;
4762   }
4763 
4764   if( base_memory() == this ) {
4765     // a self cycle indicates this memory path is dead
4766     set_req(Compile::AliasIdxBot, empty_mem);
4767   }
4768 
4769   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4770   // Recursion must occur after the self cycle check above
4771   if( base_memory()->is_MergeMem() ) {
4772     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4773     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4774     if( m != NULL &&
4775         (m->is_top() ||
4776          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4777       // propagate rollup of dead cycle to self
4778       set_req(Compile::AliasIdxBot, empty_mem);
4779     }
4780   }
4781 
4782   if( base_memory() == empty_mem ) {
4783     progress = this;
4784     // Cut inputs during Parse phase only.
4785     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4786     if( !can_reshape ) {
4787       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4788         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4789       }
4790     }
4791   }
4792 
4793   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4794     // Check if PhiNode::Ideal's "Split phis through memory merges"
4795     // transform should be attempted. Look for this->phi->this cycle.
4796     uint merge_width = req();
4797     if (merge_width > Compile::AliasIdxRaw) {
4798       PhiNode* phi = base_memory()->as_Phi();
4799       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4800         if (phi->in(i) == this) {
4801           phase->is_IterGVN()->_worklist.push(phi);
4802           break;
4803         }
4804       }
4805     }
4806   }
4807 
4808   assert(progress || verify_sparse(), "please, no dups of base");
4809   return progress;
4810 }

4854 // verify a narrow slice (either incoming or outgoing)
4855 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4856   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4857   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4858   if (Node::in_dump())               return;  // muzzle asserts when printing
4859   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4860   assert(n != NULL, "");
4861   // Elide intervening MergeMem's
4862   while (n->is_MergeMem()) {
4863     n = n->as_MergeMem()->memory_at(alias_idx);
4864   }
4865   Compile* C = Compile::current();
4866   const TypePtr* n_adr_type = n->adr_type();
4867   if (n == m->empty_memory()) {
4868     // Implicit copy of base_memory()
4869   } else if (n_adr_type != TypePtr::BOTTOM) {
4870     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4871     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4872   } else {
4873     // A few places like make_runtime_call "know" that VM calls are narrow,
4874     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4875     bool expected_wide_mem = false;
4876     if (n == m->base_memory()) {
4877       expected_wide_mem = true;
4878     } else if (alias_idx == Compile::AliasIdxRaw ||
4879                n == m->memory_at(Compile::AliasIdxRaw)) {
4880       expected_wide_mem = true;
4881     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4882       // memory can "leak through" calls on channels that
4883       // are write-once.  Allow this also.
4884       expected_wide_mem = true;
4885     }
4886     assert(expected_wide_mem, "expected narrow slice replacement");
4887   }
4888 }

4894 //-----------------------------memory_at---------------------------------------
4895 Node* MergeMemNode::memory_at(uint alias_idx) const {
4896   assert(alias_idx >= Compile::AliasIdxRaw ||
4897          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4898          "must avoid base_memory and AliasIdxTop");
4899 
4900   // Otherwise, it is a narrow slice.
4901   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4902   Compile *C = Compile::current();
4903   if (is_empty_memory(n)) {
4904     // the array is sparse; empty slots are the "top" node
4905     n = base_memory();
4906     assert(Node::in_dump()
4907            || n == NULL || n->bottom_type() == Type::TOP
4908            || n->adr_type() == NULL // address is TOP
4909            || n->adr_type() == TypePtr::BOTTOM
4910            || n->adr_type() == TypeRawPtr::BOTTOM
4911            || Compile::current()->AliasLevel() == 0,
4912            "must be a wide memory");
4913     // AliasLevel == 0 if we are organizing the memory states manually.
4914     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.
4915   } else {
4916     // make sure the stored slice is sane
4917     #ifdef ASSERT
4918     if (VMError::is_error_reported() || Node::in_dump()) {
4919     } else if (might_be_same(n, base_memory())) {
4920       // Give it a pass:  It is a mostly harmless repetition of the base.
4921       // This can arise normally from node subsumption during optimization.
4922     } else {
4923       verify_memory_slice(this, alias_idx, n);
4924     }
4925     #endif
4926   }
4927   return n;

4931 void MergeMemNode::set_memory_at(uint alias_idx, Node *n) {
4932   verify_memory_slice(this, alias_idx, n);
4933   Node* empty_mem = empty_memory();
4934   if (n == base_memory())  n = empty_mem;  // collapse default

4970 void MergeMemNode::grow_to_match(const MergeMemNode* other) {
4971   Node* empty_mem = empty_memory();

4973   // look for the finite support of the other memory
4974   for (uint i = other->req(); --i >= req(); ) {
4975     if (other->in(i) != empty_mem) {
4976       uint new_len = i+1;
4977       while (req() < new_len)  add_req(empty_mem);
4978       break;
