  83 static inline void* addr_from_java(jlong addr) {
  84   // This assert fails in a variety of ways on 32-bit systems.
  85   // It is impossible to predict whether native code that converts
  86   // pointers to longs will sign-extend or zero-extend the addresses.
  87   //assert(addr == (uintptr_t)addr, "must not be odd high bits");
  88   return (void*)(uintptr_t)addr;
  89 }
  90 
  91 static inline jlong addr_to_java(void* p) {
  92   assert(p == (void*)(uintptr_t)p, "must not be odd high bits");
  93   return (uintptr_t)p;
  94 }

  97 // Note: The VM's obj_field and related accessors use byte-scaled
  98 // ("unscaled") offsets, just as the unsafe methods do.
  99 
 100 // However, the method Unsafe.fieldOffset explicitly declines to
 101 // guarantee this.  The field offset values manipulated by the Java user
 102 // through the Unsafe API are opaque cookies that just happen to be byte
 103 // offsets.  We represent this state of affairs by passing the cookies
 104 // through conversion functions when going between the VM and the Unsafe API.
 105 // The conversion functions just happen to be no-ops at present.
 106 
 107 static inline jlong field_offset_to_byte_offset(jlong field_offset) {
 108   return field_offset;
 109 }
 110 
 111 static inline jlong field_offset_from_byte_offset(jlong byte_offset) {
 112   return byte_offset;
 113 }
 114 
 115 static inline void assert_field_offset_sane(oop p, jlong field_offset) {
 116 #ifdef ASSERT
 117   jlong byte_offset = field_offset_to_byte_offset(field_offset);
 118 
 119   if (p != NULL) {
 120     assert(byte_offset >= 0 && byte_offset <= (jlong)MAX_OBJECT_SIZE, "sane offset");
 121     if (byte_offset == (jint)byte_offset) {
 122       void* ptr_plus_disp = cast_from_oop<address>(p) + byte_offset;
 123       assert(p->field_addr_raw((jint)byte_offset) == ptr_plus_disp,
 124              "raw [ptr+disp] must be consistent with oop::field_addr_raw");
 125     }
 126     jlong p_size = HeapWordSize * (jlong)(p->size());
 127     assert(byte_offset < p_size, "Unsafe access: offset " INT64_FORMAT " > object's size " INT64_FORMAT, (int64_t)byte_offset, (int64_t)p_size);
 128   }
 129 #endif
 130 }
 131 
 132 static inline void* index_oop_from_field_offset_long(oop p, jlong field_offset) {
 133   assert_field_offset_sane(p, field_offset);
 134   jlong byte_offset = field_offset_to_byte_offset(field_offset);
 135 
 136   if (p != NULL) {
 137     p = Access<>::resolve(p);
 138   }
 139 
 140   if (sizeof(char*) == sizeof(jint)) {   // (this constant folds!)
 141     return cast_from_oop<address>(p) + (jint) byte_offset;
 142   } else {
 143     return cast_from_oop<address>(p) +        byte_offset;
 144   }
 145 }

 158 /**
 159  * Helper class to wrap memory accesses in JavaThread::doing_unsafe_access()
 160  */
 161 class GuardUnsafeAccess {
 162   JavaThread* _thread;
 163 
 164 public:
 165   GuardUnsafeAccess(JavaThread* thread) : _thread(thread) {
 166     // native/off-heap access which may raise SIGBUS if accessing
 167     // memory mapped file data in a region of the file which has
 168     // been truncated and is now invalid.
 169     _thread->set_doing_unsafe_access(true);
 170   }
 171 
 172   ~GuardUnsafeAccess() {
 173     _thread->set_doing_unsafe_access(false);
 174   }
 175 };
 176 
 177 /**
 178  * Helper class for accessing memory.
 179  *
 180  * Normalizes values and wraps accesses in
 181  * JavaThread::doing_unsafe_access() if needed.
 182  */
 183 template <typename T>
 184 class MemoryAccess : StackObj {
 185   JavaThread* _thread;
 186   oop _obj;
 187   ptrdiff_t _offset;
 188 
 189   // Resolves and returns the address of the memory access.
 190   // This raw memory access may fault, so we make sure it happens within the
 191   // guarded scope by making the access volatile at least. Since the store
 192   // of Thread::set_doing_unsafe_access() is also volatile, these accesses
 193   // can not be reordered by the compiler. Therefore, if the access triggers
 194   // a fault, we will know that Thread::doing_unsafe_access() returns true.
 195   volatile T* addr() {
 196     void* addr = index_oop_from_field_offset_long(_obj, _offset);
 197     return static_cast<volatile T*>(addr);
 198   }
 199 
 200   template <typename U>
 201   U normalize_for_write(U x) {
 202     return x;
 203   }
 204 
 205   jboolean normalize_for_write(jboolean x) {
 206     return x & 1;
 207   }
 208 
 209   template <typename U>
 210   U normalize_for_read(U x) {
 211     return x;
 212   }
 213 
 214   jboolean normalize_for_read(jboolean x) {
 215     return x != 0;
 216   }
 217 
 218 public:
 219   MemoryAccess(JavaThread* thread, jobject obj, jlong offset)
 220     : _thread(thread), _obj(JNIHandles::resolve(obj)), _offset((ptrdiff_t)offset) {
 221     assert_field_offset_sane(_obj, offset);
 222   }
 223 
 224   T get() {
 225     if (_obj == NULL) {
 226       GuardUnsafeAccess guard(_thread);
 227       T ret = RawAccess<>::load(addr());
 228       return normalize_for_read(ret);
 229     } else {
 230       T ret = HeapAccess<>::load_at(_obj, _offset);
 231       return normalize_for_read(ret);
 232     }
 233   }
 234 
 235   void put(T x) {
 236     if (_obj == NULL) {
 237       GuardUnsafeAccess guard(_thread);
 238       RawAccess<>::store(addr(), normalize_for_write(x));
 239     } else {
 240       assert(!_obj->is_inline_type() || _obj->mark().is_larval_state(), "must be an object instance or a larval inline type");
 241       HeapAccess<>::store_at(_obj, _offset, normalize_for_write(x));
 242     }
 243   }
 244 

 245   T get_volatile() {
 246     if (_obj == NULL) {
 247       GuardUnsafeAccess guard(_thread);
 248       volatile T ret = RawAccess<MO_SEQ_CST>::load(addr());
 249       return normalize_for_read(ret);
 250     } else {
 251       T ret = HeapAccess<MO_SEQ_CST>::load_at(_obj, _offset);
 252       return normalize_for_read(ret);
 253     }
 254   }
 255 
 256   void put_volatile(T x) {
 257     if (_obj == NULL) {
 258       GuardUnsafeAccess guard(_thread);
 259       RawAccess<MO_SEQ_CST>::store(addr(), normalize_for_write(x));
 260     } else {
 261       HeapAccess<MO_SEQ_CST>::store_at(_obj, _offset, normalize_for_write(x));
 262     }
 263   }
 264 };

 267 /*
 268  * Get the field descriptor of the field of the given object at the given offset.
 269  */
 270 static bool get_field_descriptor(oop p, jlong offset, fieldDescriptor* fd) {
 271   bool found = false;
 272   Klass* k = p->klass();
 273   if (k->is_instance_klass()) {
 274     InstanceKlass* ik = InstanceKlass::cast(k);
 275     found = ik->find_field_from_offset((int)offset, false, fd);
 276     if (!found && ik->is_mirror_instance_klass()) {
 277       Klass* k2 = java_lang_Class::as_Klass(p);
 278       if (k2->is_instance_klass()) {
 279         ik = InstanceKlass::cast(k2);
 280         found = ik->find_field_from_offset((int)offset, true, fd);
 281       }
 282     }
 283   }
 284   return found;
 285 }
 286 #endif // ASSERT

 288 static void assert_and_log_unsafe_value_access(oop p, jlong offset, InlineKlass* vk) {
 289   Klass* k = p->klass();
 290 #ifdef ASSERT
 291   if (k->is_instance_klass()) {
 292     assert_field_offset_sane(p, offset);
 293     fieldDescriptor fd;
 294     bool found = get_field_descriptor(p, offset, &fd);
 295     if (found) {
 296       assert(found, "value field not found");
 297       assert(fd.is_inlined(), "field not flat");
 298     } else {
 299       if (log_is_enabled(Trace, valuetypes)) {
 300         log_trace(valuetypes)("not a field in %s at offset " SIZE_FORMAT_HEX,
 301                               p->klass()->external_name(), offset);
 302       }
 303     }
 304   } else if (k->is_flatArray_klass()) {
 305     FlatArrayKlass* vak = FlatArrayKlass::cast(k);
 306     int index = (offset - vak->array_header_in_bytes()) / vak->element_byte_size();
 307     address dest = (address)((flatArrayOop)p)->value_at_addr(index, vak->layout_helper());
 308     assert(dest == (cast_from_oop<address>(p) + offset), "invalid offset");
 309   } else {
 310     ShouldNotReachHere();
 311   }
 312 #endif // ASSERT
 313   if (log_is_enabled(Trace, valuetypes)) {
 314     if (k->is_flatArray_klass()) {
 315       FlatArrayKlass* vak = FlatArrayKlass::cast(k);
 316       int index = (offset - vak->array_header_in_bytes()) / vak->element_byte_size();
 317       address dest = (address)((flatArrayOop)p)->value_at_addr(index, vak->layout_helper());
 318       log_trace(valuetypes)("%s array type %s index %d element size %d offset " SIZE_FORMAT_HEX " at " INTPTR_FORMAT,
 319                             p->klass()->external_name(), vak->external_name(),
 320                             index, vak->element_byte_size(), offset, p2i(dest));
 321     } else {
 322       log_trace(valuetypes)("%s field type %s at offset " SIZE_FORMAT_HEX,
 323                             p->klass()->external_name(), vk->external_name(), offset);
 324     }
 325   }
 326 }

 863 // define a class but do not make it known to the class loader or system dictionary
 864 // - host_class:  supplies context for linkage, access control, protection domain, and class loader
 865 //                if host_class is itself anonymous then it is replaced with its host class.
 866 // - data:  bytes of a class file, a raw memory address (length gives the number of bytes)
 867 // - cp_patches:  where non-null entries exist, they replace corresponding CP entries in data
 868 
 869 // When you load an anonymous class U, it works as if you changed its name just before loading,
 870 // to a name that you will never use again.  Since the name is lost, no other class can directly
 871 // link to any member of U.  Just after U is loaded, the only way to use it is reflectively,
 872 // through java.lang.Class methods like Class.newInstance.
 873 
 874 // The package of an anonymous class must either match its host's class's package or be in the
 875 // unnamed package.  If it is in the unnamed package then it will be put in its host class's
 876 // package.
 877 //
 878 
 879 // Access checks for linkage sites within U continue to follow the same rules as for named classes.
 880 // An anonymous class also has special privileges to access any member of its host class.
 881 // This is the main reason why this loading operation is unsafe.  The purpose of this is to
 882 // allow language implementations to simulate "open classes"; a host class in effect gets
 883 // new code when an anonymous class is loaded alongside it.  A less convenient but more
 884 // standard way to do this is with reflection, which can also be set to ignore access
 885 // restrictions.
 886 
 887 // Access into an anonymous class is possible only through reflection.  Therefore, there
 888 // are no special access rules for calling into an anonymous class.  The relaxed access
 889 // rule for the host class is applied in the opposite direction:  A host class reflectively
 890 // access one of its anonymous classes.
 891 
 892 // If you load the same bytecodes twice, you get two different classes.  You can reload
 893 // the same bytecodes with or without varying CP patches.
 894 
 895 // By using the CP patching array, you can have a new anonymous class U2 refer to an older one U1.
 896 // The bytecodes for U2 should refer to U1 by a symbolic name (doesn't matter what the name is).
 897 // The CONSTANT_Class entry for that name can be patched to refer directly to U1.
 898 
 899 // This allows, for example, U2 to use U1 as a superclass or super-interface, or as
 900 // an outer class (so that U2 is an anonymous inner class of anonymous U1).
 901 // It is not possible for a named class, or an older anonymous class, to refer by
 902 // name (via its CP) to a newer anonymous class.
 903 
 904 // CP patching may also be used to modify (i.e., hack) the names of methods, classes,
 905 // or type descriptors used in the loaded anonymous class.
 906 
 907 // Finally, CP patching may be used to introduce "live" objects into the constant pool,
 908 // instead of "dead" strings.  A compiled statement like println((Object)"hello") can
 909 // be changed to println(greeting), where greeting is an arbitrary object created before
 910 // the anonymous class is loaded.  This is useful in dynamic languages, in which
 911 // various kinds of metaobjects must be introduced as constants into bytecode.
 912 // Note the cast (Object), which tells the verifier to expect an arbitrary object,
 913 // not just a literal string.  For such ldc instructions, the verifier uses the
 914 // type Object instead of String, if the loaded constant is not in fact a String.
 915 //
 916 // An anonymous class cannot be an inline type.
 917
