  93     private static Class<? extends Annotation> findPolymorphicSignatureClass() {
  94         Class<? extends Annotation> signaturePolyAnnotation = null;
  95         try {
  96             for (Class<?> clazz : TestResolvedJavaType.class.getClassLoader().loadClass("java.lang.invoke.MethodHandle").getDeclaredClasses()) {
  97                 if (clazz.getName().endsWith("PolymorphicSignature") && Annotation.class.isAssignableFrom(clazz)) {
  98                     signaturePolyAnnotation = (Class<? extends Annotation>) clazz;
  99                     break;
 100                 }
 101             }
 102         } catch (Throwable e) {
 103             throw new AssertionError("Could not find annotation PolymorphicSignature in java.lang.invoke.MethodHandle", e);
 104         }
 105         assertNotNull(signaturePolyAnnotation);
 106         return signaturePolyAnnotation;
 107     }

 287     public Class<?> getSupertype(Class<?> c) {
 288         assert !c.isPrimitive();
 289         if (c.isArray()) {
 290             Class<?> componentType = c.getComponentType();
 291             if (componentType.isPrimitive() || componentType == Object.class) {
 292                 return Object.class;
 293             }
 294             return getArrayClass(getSupertype(componentType));
 295         }
 296         if (c.isInterface()) {
 297             return Object.class;
 298         }
 299         return c.getSuperclass();
 300     }

 302     public Class<?> findLeastCommonAncestor(Class<?> c1Initial, Class<?> c2Initial) {
 303         if (c1Initial.isPrimitive() || c2Initial.isPrimitive()) {
 304             return null;
 305         } else {
 306             Class<?> c1 = c1Initial;
 307             Class<?> c2 = c2Initial;
 308             while (true) {
 309                 if (c1.isAssignableFrom(c2)) {
 310                     return c1;
 311                 }
 312                 if (c2.isAssignableFrom(c1)) {
 313                     return c2;
 314                 }
 315                 c1 = getSupertype(c1);
 316                 c2 = getSupertype(c2);
 317             }
 318         }
 319     }

 350     private class HidingClassLoader extends ClassLoader {
 351         @Override
 352         protected Class<?> findClass(final String name) throws ClassNotFoundException {
 353             if (name.endsWith("MissingInterface")) {
 354                 throw new ClassNotFoundException("missing");
 355             }
 356             byte[] classData = null;
 357             try {
 358                 InputStream is = HidingClassLoader.class.getResourceAsStream("/" + name.replace('.', '/') + ".class");
 359                 classData = new byte[is.available()];
 360                 new DataInputStream(is).readFully(classData);
 361             } catch (IOException e) {
 362                 Assert.fail("can't access class: " + name);
 363             }
 364 
 365             return defineClass(null, classData, 0, classData.length);
 366         }
 367 
 368         ResolvedJavaType lookupJavaType(String name) throws ClassNotFoundException {
 369             return metaAccess.lookupJavaType(loadClass(name));
 370         }

 373             super(null);
 374         }

 421     private static boolean hasDefaultMethods(ResolvedJavaType type) {
 422         if (!type.isInterface() && type.getSuperclass() != null && hasDefaultMethods(type.getSuperclass())) {
 423             return true;
 424         }
 425         for (ResolvedJavaType iface : type.getInterfaces()) {
 426             if (hasDefaultMethods(iface)) {
 427                 return true;
 428             }
 429         }
 430         return declaresDefaultMethods(type);
 431     }
 432 
 433     static boolean declaresDefaultMethods(ResolvedJavaType type) {
 434         if (!type.isInterface()) {
 435             /* Only interfaces can declare default methods. */
 436             return false;
 437         }
 438         for (ResolvedJavaMethod method : type.getDeclaredMethods()) {
 439             if (method.isDefault()) {
 440                 assert !Modifier.isStatic(method.getModifiers()) : "Default method that is static?";
 441                 return true;
 442             }
 443         }
 444         return false;
 445     }

 471     void checkConcreteSubtype(ResolvedJavaType type, ResolvedJavaType expected) {
 472         AssumptionResult<ResolvedJavaType> leafConcreteSubtype = type.findLeafConcreteSubtype();
 473         if (leafConcreteSubtype == null) {
 474             // findLeafConcreteSubtype() is conservative
 475         } else {
 476             if (expected == null) {
 477                 assertNull(leafConcreteSubtype);
 478             } else {
 479                 assertTrue(leafConcreteSubtype.getResult().equals(expected));
 480             }
 481             assertTrue(!type.isLeaf() || leafConcreteSubtype.isAssumptionFree());
 482         }
 483 
 484         if (!type.isArray()) {
 485             ResolvedJavaType arrayType = type.getArrayClass();
 486             AssumptionResult<ResolvedJavaType> arraySubtype = arrayType.findLeafConcreteSubtype();
 487             if (arraySubtype != null) {
 488                 assertEquals(arraySubtype.getResult(), arrayType);
 489             } else {
 490                 // findLeafConcreteSubtype() method is conservative
 491             }
 492         }
 493     }

 681     /**
 682      * See <a href="http://docs.oracle.com/javase/specs/jvms/se7/html/jvms-5.html#jvms-5.4.5">Method
 683      * overriding</a>.
 684      */
 685     static boolean isOverriderOf(Method impl, Method m) {
 686         if (!isPrivate(m.getModifiers()) && !isFinal(m.getModifiers())) {
 687             if (m.getName().equals(impl.getName())) {
 688                 if (m.getReturnType() == impl.getReturnType()) {
 689                     if (Arrays.equals(m.getParameterTypes(), impl.getParameterTypes())) {
 690                         if (isPublic(m.getModifiers()) || isProtected(m.getModifiers())) {
 691                             // m is public or protected
 692                             return isPublic(impl.getModifiers()) || isProtected(impl.getModifiers());
 693                         } else {
 694                             // m is package-private
 695                             return impl.getDeclaringClass().getPackage() == m.getDeclaringClass().getPackage();
 696                         }
 697                     }
 698                 }
 699             }
 700         }
 701         return false;
 702     }

 706     static class VTable {
 707 
 708         final Map<NameAndSignature, Method> methods = new HashMap<>();
 709     }

 711     static synchronized VTable getVTable(Class<?> c) {
 712         VTable vtable = vtables.get(c);
 713         if (vtable == null) {
 714             vtable = new VTable();
 715             if (c != Object.class) {
 716                 VTable superVtable = getVTable(c.getSuperclass());
 717                 vtable.methods.putAll(superVtable.methods);
 718             }
 719             for (Method m : c.getDeclaredMethods()) {
 720                 if (!isStatic(m.getModifiers()) && !isPrivate(m.getModifiers())) {
 721                     if (isAbstract(m.getModifiers())) {
 722                         // A subclass makes a concrete method in a superclass abstract
 723                         vtable.methods.remove(new NameAndSignature(m));
 724                     } else {
 725                         vtable.methods.put(new NameAndSignature(m), m);
 726                     }
 727                 }
 728             }
 729             vtables.put(c, vtable);
 730         }
 731         return vtable;
 732     }

 734     static Set<Method> findDeclarations(Method impl, Class<?> c) {
 735         Set<Method> declarations = new HashSet<>();
 736         NameAndSignature implSig = new NameAndSignature(impl);
 737         if (c != null) {
 738             for (Method m : c.getDeclaredMethods()) {
 739                 if (new NameAndSignature(m).equals(implSig)) {
 740                     declarations.add(m);
 741                     break;
 742                 }
 743             }
 744             if (!c.isInterface()) {
 745                 declarations.addAll(findDeclarations(impl, c.getSuperclass()));
 746             }
 747             for (Class<?> i : c.getInterfaces()) {
 748                 declarations.addAll(findDeclarations(impl, i));
 749             }
 750         }
 751         return declarations;
 752     }

 885     private static boolean isHiddenFromReflection(ResolvedJavaField f) {
 886         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Throwable.class)) && f.getName().equals("backtrace")) {
 887             return true;
 888         }
 889         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ConstantPool.class)) && f.getName().equals("constantPoolOop")) {
 890             return true;
 891         }
 892         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Class.class))) {
 893             return f.getName().equals("classLoader") || f.getName().equals("classData");
 894         }
 895         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Lookup.class))) {
 896             return f.getName().equals("allowedModes") || f.getName().equals("lookupClass");
 897         }
 898         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ClassLoader.class)) ||
 899             f.getDeclaringClass().equals(metaAccess.lookupJavaType(AccessibleObject.class)) ||
 900             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Constructor.class)) ||
 901             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Field.class)) ||
 902             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Method.class)) ||
 903             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Module.class))) {
 904             return true;
 905         }
 906         return false;
 907     }
