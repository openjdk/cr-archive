  65 //
  66 // Replace any references to "oldref" in inputs to "use" with "newref".
  67 // Returns the number of replacements made.
  68 //
  69 int PhaseMacroExpand::replace_input(Node *use, Node *oldref, Node *newref) {
  70   int nreplacements = 0;
  71   uint req = use->req();
  72   for (uint j = 0; j < use->len(); j++) {
  73     Node *uin = use->in(j);
  74     if (uin == oldref) {
  75       if (j < req)
  76         use->set_req(j, newref);
  77       else
  78         use->set_prec(j, newref);
  79       nreplacements++;
  80     } else if (j >= req && uin == NULL) {
  81       break;
  82     }
  83   }
  84   return nreplacements;
  85 }

  87 Node* PhaseMacroExpand::opt_bits_test(Node* ctrl, Node* region, int edge, Node* word, int mask, int bits, bool return_fast_path) {
  88   Node* cmp;
  89   if (mask != 0) {
  90     Node* and_node = transform_later(new AndXNode(word, MakeConX(mask)));
  91     cmp = transform_later(new CmpXNode(and_node, MakeConX(bits)));
  92   } else {
  93     cmp = word;
  94   }
  95   Node* bol = transform_later(new BoolNode(cmp, BoolTest::ne));
  96   IfNode* iff = new IfNode( ctrl, bol, PROB_MIN, COUNT_UNKNOWN );
  97   transform_later(iff);
  98 
  99   // Fast path taken.
 100   Node *fast_taken = transform_later(new IfFalseNode(iff));
 101 
 102   // Fast path not-taken, i.e. slow path
 103   Node *slow_taken = transform_later(new IfTrueNode(iff));
 104 
 105   if (return_fast_path) {
 106     region->init_req(edge, slow_taken); // Capture slow-control
 107     return fast_taken;
 108   } else {
 109     region->init_req(edge, fast_taken); // Capture fast-control
 110     return slow_taken;
 111   }
 112 }

 124 //------------------------------make_slow_call---------------------------------
 125 CallNode* PhaseMacroExpand::make_slow_call(CallNode *oldcall, const TypeFunc* slow_call_type,
 126                                            address slow_call, const char* leaf_name, Node* slow_path,
 127                                            Node* parm0, Node* parm1, Node* parm2) {
 128 
 129   // Slow-path call
 130  CallNode *call = leaf_name
 131    ? (CallNode*)new CallLeafNode      ( slow_call_type, slow_call, leaf_name, TypeRawPtr::BOTTOM )
 132    : (CallNode*)new CallStaticJavaNode( slow_call_type, slow_call, OptoRuntime::stub_name(slow_call), oldcall->jvms()->bci(), TypeRawPtr::BOTTOM );
 133 
 134   // Slow path call has no side-effects, uses few values
 135   copy_predefined_input_for_runtime_call(slow_path, oldcall, call );
 136   if (parm0 != NULL)  call->init_req(TypeFunc::Parms+0, parm0);
 137   if (parm1 != NULL)  call->init_req(TypeFunc::Parms+1, parm1);
 138   if (parm2 != NULL)  call->init_req(TypeFunc::Parms+2, parm2);
 139   call->copy_call_debug_info(&_igvn, oldcall);
 140   call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
 141   _igvn.replace_node(oldcall, call);
 142   transform_later(call);
 143 
 144   return call;
 145 }

 147 void PhaseMacroExpand::extract_call_projections(CallNode *call) {
 148   _fallthroughproj = NULL;
 149   _fallthroughcatchproj = NULL;
 150   _ioproj_fallthrough = NULL;
 151   _ioproj_catchall = NULL;
 152   _catchallcatchproj = NULL;
 153   _memproj_fallthrough = NULL;
 154   _memproj_catchall = NULL;
 155   _resproj = NULL;
 156   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
 157     ProjNode *pn = call->fast_out(i)->as_Proj();
 158     switch (pn->_con) {
 159       case TypeFunc::Control:
 160       {
 161         // For Control (fallthrough) and I_O (catch_all_index) we have CatchProj -> Catch -> Proj
 162         _fallthroughproj = pn;
 163         DUIterator_Fast jmax, j = pn->fast_outs(jmax);
 164         const Node *cn = pn->fast_out(j);
 165         if (cn->is_Catch()) {
 166           ProjNode *cpn = NULL;
 167           for (DUIterator_Fast kmax, k = cn->fast_outs(kmax); k < kmax; k++) {
 168             cpn = cn->fast_out(k)->as_Proj();
 169             assert(cpn->is_CatchProj(), "must be a CatchProjNode");
 170             if (cpn->_con == CatchProjNode::fall_through_index)
 171               _fallthroughcatchproj = cpn;
 172             else {
 173               assert(cpn->_con == CatchProjNode::catch_all_index, "must be correct index.");
 174               _catchallcatchproj = cpn;
 175             }
 176           }
 177         }
 178         break;
 179       }
 180       case TypeFunc::I_O:
 181         if (pn->_is_io_use)
 182           _ioproj_catchall = pn;
 183         else
 184           _ioproj_fallthrough = pn;
 185         break;
 186       case TypeFunc::Memory:
 187         if (pn->_is_io_use)
 188           _memproj_catchall = pn;
 189         else
 190           _memproj_fallthrough = pn;
 191         break;
 192       case TypeFunc::Parms:
 193         _resproj = pn;
 194         break;
 195       default:
 196         assert(false, "unexpected projection from allocation node.");
 197     }
 198   }
 199 
 200 }

 207 // Search for a memory operation for the specified memory slice.
 208 static Node *scan_mem_chain(Node *mem, int alias_idx, int offset, Node *start_mem, Node *alloc, PhaseGVN *phase) {
 209   Node *orig_mem = mem;
 210   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 211   const TypeOopPtr *tinst = phase->C->get_adr_type(alias_idx)->isa_oopptr();
 212   while (true) {
 213     if (mem == alloc_mem || mem == start_mem ) {
 214       return mem;  // hit one of our sentinels
 215     } else if (mem->is_MergeMem()) {
 216       mem = mem->as_MergeMem()->memory_at(alias_idx);
 217     } else if (mem->is_Proj() && mem->as_Proj()->_con == TypeFunc::Memory) {
 218       Node *in = mem->in(0);
 219       // we can safely skip over safepoints, calls, locks and membars because we
 220       // already know that the object is safe to eliminate.
 221       if (in->is_Initialize() && in->as_Initialize()->allocation() == alloc) {
 222         return in;
 223       } else if (in->is_Call()) {
 224         CallNode *call = in->as_Call();
 225         if (call->may_modify(tinst, phase)) {
 226           assert(call->is_ArrayCopy(), "ArrayCopy is the only call node that doesn't make allocation escape");
 227           if (call->as_ArrayCopy()->modifies(offset, offset, phase, false)) {
 228             return in;
 229           }
 230         }
 231         mem = in->in(TypeFunc::Memory);
 232       } else if (in->is_MemBar()) {
 233         ArrayCopyNode* ac = NULL;
 234         if (ArrayCopyNode::may_modify(tinst, in->as_MemBar(), phase, ac)) {
 235           assert(ac != NULL && ac->is_clonebasic(), "Only basic clone is a non escaping clone");
 236           return ac;
 237         }
 238         mem = in->in(TypeFunc::Memory);
 239       } else {
 240         assert(false, "unexpected projection");
 241       }
 242     } else if (mem->is_Store()) {
 243       const TypePtr* atype = mem->as_Store()->adr_type();
 244       int adr_idx = phase->C->get_alias_index(atype);
 245       if (adr_idx == alias_idx) {
 246         assert(atype->isa_oopptr(), "address type must be oopptr");
 247         int adr_offset = atype->flattened_offset();
 248         uint adr_iid = atype->is_oopptr()->instance_id();
 249         // Array elements references have the same alias_idx
 250         // but different offset and different instance_id.
 251         if (adr_offset == offset && adr_iid == alloc->_idx)
 252           return mem;
 253       } else {
 254         assert(adr_idx == Compile::AliasIdxRaw, "address must match or be raw");
 255       }
 256       mem = mem->in(MemNode::Memory);
 257     } else if (mem->is_ClearArray()) {
 258       if (!ClearArrayNode::step_through(&mem, alloc->_idx, phase)) {
 259         // Can not bypass initialization of the instance
 260         // we are looking.
 261         debug_only(intptr_t offset;)
 262         assert(alloc == AllocateNode::Ideal_allocation(mem->in(3), phase, offset), "sanity");
 263         InitializeNode* init = alloc->as_Allocate()->initialization();
 264         // We are looking for stored value, return Initialize node
 265         // or memory edge from Allocate node.
 266         if (init != NULL)
 267           return init;
 268         else
 269           return alloc->in(TypeFunc::Memory); // It will produce zero value (see callers).
 270       }
 271       // Otherwise skip it (the call updated 'mem' value).
 272     } else if (mem->Opcode() == Op_SCMemProj) {
 273       mem = mem->in(0);
 274       Node* adr = NULL;
 275       if (mem->is_LoadStore()) {
 276         adr = mem->in(MemNode::Address);
 277       } else {
 278         assert(mem->Opcode() == Op_EncodeISOArray ||
 279                mem->Opcode() == Op_StrCompressedCopy, "sanity");
 280         adr = mem->in(3); // Destination array
 281       }
 282       const TypePtr* atype = adr->bottom_type()->is_ptr();
 283       int adr_idx = phase->C->get_alias_index(atype);
 284       if (adr_idx == alias_idx) {
 285         DEBUG_ONLY(mem->dump();)
 286         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 287         return NULL;
 288       }
 289       mem = mem->in(MemNode::Memory);
 290     } else if (mem->Opcode() == Op_StrInflatedCopy) {
 291       Node* adr = mem->in(3); // Destination array
 292       const TypePtr* atype = adr->bottom_type()->is_ptr();
 293       int adr_idx = phase->C->get_alias_index(atype);
 294       if (adr_idx == alias_idx) {
 295         DEBUG_ONLY(mem->dump();)
 296         assert(false, "Object is not scalar replaceable if a StrInflatedCopy node accesses its field");
 297         return NULL;
 298       }
 299       mem = mem->in(MemNode::Memory);
 300     } else {
 301       return mem;
 302     }
 303     assert(mem != orig_mem, "dead memory loop");
 304   }
 305 }

 307 // Generate loads from source of the arraycopy for fields of
 308 // destination needed at a deoptimization point
 309 Node* PhaseMacroExpand::make_arraycopy_load(ArrayCopyNode* ac, intptr_t offset, Node* ctl, Node* mem, BasicType ft, const Type *ftype, AllocateNode *alloc) {
 310   BasicType bt = ft;
 311   const Type *type = ftype;
 312   if (ft == T_NARROWOOP) {
 313     bt = T_OBJECT;
 314     type = ftype->make_oopptr();
 315   }
 316   Node* res = NULL;
 317   if (ac->is_clonebasic()) {
 318     assert(ac->in(ArrayCopyNode::Src) != ac->in(ArrayCopyNode::Dest), "clone source equals destination");
 319     Node* base = ac->in(ArrayCopyNode::Src);
 320     Node* adr = _igvn.transform(new AddPNode(base, base, MakeConX(offset)));
 321     const TypePtr* adr_type = _igvn.type(base)->is_ptr()->add_offset(offset);
 322     res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 323   } else {
 324     if (ac->modifies(offset, offset, &_igvn, true)) {
 325       assert(ac->in(ArrayCopyNode::Dest) == alloc->result_cast(), "arraycopy destination should be allocation's result");
 326       uint shift = exact_log2(type2aelembytes(bt));
 327       Node* src_pos = ac->in(ArrayCopyNode::SrcPos);
 328       Node* dest_pos = ac->in(ArrayCopyNode::DestPos);
 329       const TypeInt* src_pos_t = _igvn.type(src_pos)->is_int();
 330       const TypeInt* dest_pos_t = _igvn.type(dest_pos)->is_int();
 331 
 332       Node* adr = NULL;
 333       Node* base = ac->in(ArrayCopyNode::Src);
 334       const TypePtr* adr_type = _igvn.type(base)->is_ptr();
 335       assert(adr_type->isa_aryptr(), "only arrays here");
 336       if (src_pos_t->is_con() && dest_pos_t->is_con()) {
 337         intptr_t off = ((src_pos_t->get_con() - dest_pos_t->get_con()) << shift) + offset;
 338         adr = _igvn.transform(new AddPNode(base, base, MakeConX(off)));
 339         adr_type = _igvn.type(adr)->is_ptr();
 340         assert(adr_type == _igvn.type(base)->is_aryptr()->add_field_offset_and_offset(off), "incorrect address type");
 341         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 342           // Don't emit a new load from src if src == dst but try to get the value from memory instead
 343           return value_from_mem(ac->in(TypeFunc::Memory), ctl, ft, ftype, adr_type->isa_oopptr(), alloc);
 344         }
 345       } else {
 346         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 347           // Non constant offset in the array: we can't statically
 348           // determine the value
 349           return NULL;
 350         }
 351         Node* diff = _igvn.transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 352 #ifdef _LP64
 353         diff = _igvn.transform(new ConvI2LNode(diff));
 354 #endif
 355         diff = _igvn.transform(new LShiftXNode(diff, intcon(shift)));
 356 
 357         Node* off = _igvn.transform(new AddXNode(MakeConX(offset), diff));
 358         adr = _igvn.transform(new AddPNode(base, base, off));
 359         // In the case of a flattened inline type array, each field has its
 360         // own slice so we need to extract the field being accessed from
 361         // the address computation
 362         adr_type = adr_type->is_aryptr()->add_field_offset_and_offset(offset)->add_offset(Type::OffsetBot);
 363         adr = _igvn.transform(new CastPPNode(adr, adr_type));
 364       }
 365       res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 366     }
 367   }
 368   if (res != NULL) {
 369     res = _igvn.transform(res);
 370     if (ftype->isa_narrowoop()) {
 371       // PhaseMacroExpand::scalar_replacement adds DecodeN nodes
 372       assert(res->isa_DecodeN(), "should be narrow oop");
 373       res = _igvn.transform(new EncodePNode(res, ftype));
 374     }
 375     return res;
 376   }
 377   return NULL;
 378 }

 380 //
 381 // Given a Memory Phi, compute a value Phi containing the values from stores
 382 // on the input paths.
 383 // Note: this function is recursive, its depth is limited by the "level" argument
 384 // Returns the computed Phi, or NULL if it cannot compute it.
 385 Node *PhaseMacroExpand::value_from_mem_phi(Node *mem, BasicType ft, const Type *phi_type, const TypeOopPtr *adr_t, AllocateNode *alloc, Node_Stack *value_phis, int level) {
 386   assert(mem->is_Phi(), "sanity");
 387   int alias_idx = C->get_alias_index(adr_t);
 388   int offset = adr_t->flattened_offset();
 389   int instance_id = adr_t->instance_id();
 390 
 391   // Check if an appropriate value phi already exists.
 392   Node* region = mem->in(0);
 393   for (DUIterator_Fast kmax, k = region->fast_outs(kmax); k < kmax; k++) {
 394     Node* phi = region->fast_out(k);
 395     if (phi->is_Phi() && phi != mem &&
 396         phi->as_Phi()->is_same_inst_field(phi_type, (int)mem->_idx, instance_id, alias_idx, offset)) {
 397       return phi;
 398     }
 399   }
 400   // Check if an appropriate new value phi already exists.
 401   Node* new_phi = value_phis->find(mem->_idx);
 402   if (new_phi != NULL)
 403     return new_phi;
 404 
 405   if (level <= 0) {
 406     return NULL; // Give up: phi tree too deep
 407   }
 408   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 409   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 410 
 411   uint length = mem->req();
 412   GrowableArray <Node *> values(length, length, NULL);
 413 
 414   // create a new Phi for the value
 415   PhiNode *phi = new PhiNode(mem->in(0), phi_type, NULL, mem->_idx, instance_id, alias_idx, offset);
 416   transform_later(phi);
 417   value_phis->push(phi, mem->_idx);
 418 
 419   for (uint j = 1; j < length; j++) {
 420     Node *in = mem->in(j);
 421     if (in == NULL || in->is_top()) {
 422       values.at_put(j, in);
 423     } else  {
 424       Node *val = scan_mem_chain(in, alias_idx, offset, start_mem, alloc, &_igvn);
 425       if (val == start_mem || val == alloc_mem) {
 426         // hit a sentinel, return appropriate 0 value
 427         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 428         if (default_value != NULL) {
 429           values.at_put(j, default_value);
 430         } else {
 431           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 432           values.at_put(j, _igvn.zerocon(ft));
 433         }
 434         continue;
 435       }
 436       if (val->is_Initialize()) {
 437         val = val->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 438       }
 439       if (val == NULL) {
 440         return NULL;  // can't find a value on this path
 441       }
 442       if (val == mem) {
 443         values.at_put(j, mem);
 444       } else if (val->is_Store()) {
 445         Node* n = val->in(MemNode::ValueIn);
 446         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 447         n = bs->step_over_gc_barrier(n);
 448         values.at_put(j, n);
 449       } else if(val->is_Proj() && val->in(0) == alloc) {
 450         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 451         if (default_value != NULL) {
 452           values.at_put(j, default_value);
 453         } else {
 454           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 455           values.at_put(j, _igvn.zerocon(ft));
 456         }
 457       } else if (val->is_Phi()) {
 458         val = value_from_mem_phi(val, ft, phi_type, adr_t, alloc, value_phis, level-1);
 459         if (val == NULL) {
 460           return NULL;
 461         }
 462         values.at_put(j, val);
 463       } else if (val->Opcode() == Op_SCMemProj) {
 464         assert(val->in(0)->is_LoadStore() ||
 465                val->in(0)->Opcode() == Op_EncodeISOArray ||
 466                val->in(0)->Opcode() == Op_StrCompressedCopy, "sanity");
 467         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 468         return NULL;
 469       } else if (val->is_ArrayCopy()) {
 470         Node* res = make_arraycopy_load(val->as_ArrayCopy(), offset, val->in(0), val->in(TypeFunc::Memory), ft, phi_type, alloc);
 471         if (res == NULL) {
 472           return NULL;
 473         }
 474         values.at_put(j, res);
 475       } else {
 476 #ifdef ASSERT
 477         val->dump();
 478         assert(false, "unknown node on this path");
 479 #endif
 480         return NULL;  // unknown node on this path
 481       }
 482     }
 483   }
 484   // Set Phi's inputs
 485   for (uint j = 1; j < length; j++) {
 486     if (values.at(j) == mem) {
 487       phi->init_req(j, phi);
 488     } else {
 489       phi->init_req(j, values.at(j));
 490     }
 491   }
 492   return phi;
 493 }

 495 // Search the last value stored into the object's field.
 496 Node *PhaseMacroExpand::value_from_mem(Node *sfpt_mem, Node *sfpt_ctl, BasicType ft, const Type *ftype, const TypeOopPtr *adr_t, AllocateNode *alloc) {
 497   assert(adr_t->is_known_instance_field(), "instance required");
 498   int instance_id = adr_t->instance_id();
 499   assert((uint)instance_id == alloc->_idx, "wrong allocation");
 500 
 501   int alias_idx = C->get_alias_index(adr_t);
 502   int offset = adr_t->flattened_offset();
 503   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 504   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 505   Arena *a = Thread::current()->resource_area();
 506   VectorSet visited(a);
 507 
 508   bool done = sfpt_mem == alloc_mem;
 509   Node *mem = sfpt_mem;
 510   while (!done) {
 511     if (visited.test_set(mem->_idx)) {
 512       return NULL;  // found a loop, give up
 513     }
 514     mem = scan_mem_chain(mem, alias_idx, offset, start_mem, alloc, &_igvn);
 515     if (mem == start_mem || mem == alloc_mem) {
 516       done = true;  // hit a sentinel, return appropriate 0 value
 517     } else if (mem->is_Initialize()) {
 518       mem = mem->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 519       if (mem == NULL) {
 520         done = true; // Something went wrong.
 521       } else if (mem->is_Store()) {
 522         const TypePtr* atype = mem->as_Store()->adr_type();
 523         assert(C->get_alias_index(atype) == Compile::AliasIdxRaw, "store is correct memory slice");
 524         done = true;
 525       }
 526     } else if (mem->is_Store()) {
 527       const TypeOopPtr* atype = mem->as_Store()->adr_type()->isa_oopptr();
 528       assert(atype != NULL, "address type must be oopptr");
 529       assert(C->get_alias_index(atype) == alias_idx &&
 530              atype->is_known_instance_field() && atype->flattened_offset() == offset &&
 531              atype->instance_id() == instance_id, "store is correct memory slice");
 532       done = true;
 533     } else if (mem->is_Phi()) {
 534       // try to find a phi's unique input
 535       Node *unique_input = NULL;
 536       Node *top = C->top();
 537       for (uint i = 1; i < mem->req(); i++) {
 538         Node *n = scan_mem_chain(mem->in(i), alias_idx, offset, start_mem, alloc, &_igvn);
 539         if (n == NULL || n == top || n == mem) {
 540           continue;
 541         } else if (unique_input == NULL) {
 542           unique_input = n;
 543         } else if (unique_input != n) {
 544           unique_input = top;
 545           break;
 546         }
 547       }
 548       if (unique_input != NULL && unique_input != top) {
 549         mem = unique_input;
 550       } else {
 551         done = true;
 552       }
 553     } else if (mem->is_ArrayCopy()) {
 554       done = true;
 555     } else {
 556       assert(false, "unexpected node");
 557     }
 558   }
 559   if (mem != NULL) {
 560     if (mem == start_mem || mem == alloc_mem) {
 561       // hit a sentinel, return appropriate 0 value
 562       Node* default_value = alloc->in(AllocateNode::DefaultValue);
 563       if (default_value != NULL) {
 564         return default_value;
 565       }
 566       assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 567       return _igvn.zerocon(ft);
 568     } else if (mem->is_Store()) {
 569       Node* n = mem->in(MemNode::ValueIn);
 570       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 571       n = bs->step_over_gc_barrier(n);
 572       return n;
 573     } else if (mem->is_Phi()) {
 574       // attempt to produce a Phi reflecting the values on the input paths of the Phi
 575       Node_Stack value_phis(a, 8);
 576       Node * phi = value_from_mem_phi(mem, ft, ftype, adr_t, alloc, &value_phis, ValueSearchLimit);
 577       if (phi != NULL) {
 578         return phi;
 579       } else {
 580         // Kill all new Phis
 581         while(value_phis.is_nonempty()) {
 582           Node* n = value_phis.node();
 583           _igvn.replace_node(n, C->top());
 584           value_phis.pop();
 585         }
 586       }
 587     } else if (mem->is_ArrayCopy()) {
 588       Node* ctl = mem->in(0);
 589       Node* m = mem->in(TypeFunc::Memory);
 590       if (sfpt_ctl->is_Proj() && sfpt_ctl->as_Proj()->is_uncommon_trap_proj(Deoptimization::Reason_none)) {
 591         // pin the loads in the uncommon trap path
 592         ctl = sfpt_ctl;
 593         m = sfpt_mem;
 594       }
 595       return make_arraycopy_load(mem->as_ArrayCopy(), offset, ctl, m, ft, ftype, alloc);
 596     }
 597   }
 598   // Something went wrong.
 599   return NULL;
 600 }

 602 // Search the last value stored into the inline type's fields.
 603 Node* PhaseMacroExpand::inline_type_from_mem(Node* mem, Node* ctl, ciInlineKlass* vk, const TypeAryPtr* adr_type, int offset, AllocateNode* alloc) {
 604   // Subtract the offset of the first field to account for the missing oop header
 605   offset -= vk->first_field_offset();
 606   // Create a new InlineTypeNode and retrieve the field values from memory
 607   InlineTypeNode* vt = InlineTypeNode::make_uninitialized(_igvn, vk)->as_InlineType();
 608   for (int i = 0; i < vk->nof_declared_nonstatic_fields(); ++i) {
 609     ciType* field_type = vt->field_type(i);
 610     int field_offset = offset + vt->field_offset(i);
 611     // Each inline type field has its own memory slice
 612     adr_type = adr_type->with_field_offset(field_offset);
 613     Node* value = NULL;
 614     if (vt->field_is_flattened(i)) {
 615       value = inline_type_from_mem(mem, ctl, field_type->as_inline_klass(), adr_type, field_offset, alloc);
 616     } else {
 617       const Type* ft = Type::get_const_type(field_type);
 618       BasicType bt = field_type->basic_type();
 619       if (UseCompressedOops && !is_java_primitive(bt)) {
 620         ft = ft->make_narrowoop();
 621         bt = T_NARROWOOP;
 622       }
 623       value = value_from_mem(mem, ctl, bt, ft, adr_type, alloc);
 624       if (value != NULL && ft->isa_narrowoop()) {
 625         assert(UseCompressedOops, "unexpected narrow oop");
 626         value = transform_later(new DecodeNNode(value, value->get_ptr_type()));
 627       }
 628     }
 629     if (value != NULL) {
 630       vt->set_field_value(i, value);
 631     } else {
 632       // We might have reached the TrackedInitializationLimit
 633       return NULL;
 634     }
 635   }
 636   return transform_later(vt);
 637 }

 639 // Check the possibility of scalar replacement.
 640 bool PhaseMacroExpand::can_eliminate_allocation(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 641   //  Scan the uses of the allocation to check for anything that would
 642   //  prevent us from eliminating it.
 643   NOT_PRODUCT( const char* fail_eliminate = NULL; )
 644   DEBUG_ONLY( Node* disq_node = NULL; )
 645   bool  can_eliminate = true;
 646 
 647   Node* res = alloc->result_cast();
 648   const TypeOopPtr* res_type = NULL;
 649   if (res == NULL) {
 650     // All users were eliminated.
 651   } else if (!res->is_CheckCastPP()) {
 652     NOT_PRODUCT(fail_eliminate = "Allocation does not have unique CheckCastPP";)
 653     can_eliminate = false;
 654   } else {
 655     res_type = _igvn.type(res)->isa_oopptr();
 656     if (res_type == NULL) {
 657       NOT_PRODUCT(fail_eliminate = "Neither instance or array allocation";)
 658       can_eliminate = false;
 659     } else if (res_type->isa_aryptr()) {
 660       int length = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 661       if (length < 0) {
 662         NOT_PRODUCT(fail_eliminate = "Array's size is not constant";)
 663         can_eliminate = false;
 664       }
 665     }
 666   }
 667 
 668   if (can_eliminate && res != NULL) {
 669     for (DUIterator_Fast jmax, j = res->fast_outs(jmax);
 670                                j < jmax && can_eliminate; j++) {
 671       Node* use = res->fast_out(j);
 672 
 673       if (use->is_AddP()) {
 674         const TypePtr* addp_type = _igvn.type(use)->is_ptr();
 675         int offset = addp_type->offset();
 676 
 677         if (offset == Type::OffsetTop || offset == Type::OffsetBot) {
 678           NOT_PRODUCT(fail_eliminate = "Undefined field referrence";)
 679           can_eliminate = false;
 680           break;
 681         }
 682         for (DUIterator_Fast kmax, k = use->fast_outs(kmax);
 683                                    k < kmax && can_eliminate; k++) {
 684           Node* n = use->fast_out(k);
 685           if (!n->is_Store() && n->Opcode() != Op_CastP2X
 686               SHENANDOAHGC_ONLY(&& (!UseShenandoahGC || !ShenandoahBarrierSetC2::is_shenandoah_wb_pre_call(n))) ) {
 687             DEBUG_ONLY(disq_node = n;)
 688             if (n->is_Load() || n->is_LoadStore()) {
 689               NOT_PRODUCT(fail_eliminate = "Field load";)
 690             } else {
 691               NOT_PRODUCT(fail_eliminate = "Not store field reference";)
 692             }
 693             can_eliminate = false;
 694           }
 695         }
 696       } else if (use->is_ArrayCopy() &&
 697                  (use->as_ArrayCopy()->is_clonebasic() ||
 698                   use->as_ArrayCopy()->is_arraycopy_validated() ||
 699                   use->as_ArrayCopy()->is_copyof_validated() ||
 700                   use->as_ArrayCopy()->is_copyofrange_validated()) &&
 701                  use->in(ArrayCopyNode::Dest) == res) {
 702         // ok to eliminate
 703       } else if (use->is_SafePoint()) {
 704         SafePointNode* sfpt = use->as_SafePoint();
 705         if (sfpt->is_Call() && sfpt->as_Call()->has_non_debug_use(res)) {
 706           // Object is passed as argument.
 707           DEBUG_ONLY(disq_node = use;)
 708           NOT_PRODUCT(fail_eliminate = "Object is passed as argument";)
 709           can_eliminate = false;
 710         }
 711         Node* sfptMem = sfpt->memory();
 712         if (sfptMem == NULL || sfptMem->is_top()) {
 713           DEBUG_ONLY(disq_node = use;)
 714           NOT_PRODUCT(fail_eliminate = "NULL or TOP memory";)
 715           can_eliminate = false;
 716         } else {
 717           safepoints.append_if_missing(sfpt);
 718         }
 719       } else if (use->is_InlineType() && use->isa_InlineType()->get_oop() == res) {
 720         // ok to eliminate
 721       } else if (use->Opcode() == Op_StoreX && use->in(MemNode::Address) == res) {
 722         // store to mark work
 723       } else if (use->Opcode() != Op_CastP2X) { // CastP2X is used by card mark
 724         if (use->is_Phi()) {
 725           if (use->outcnt() == 1 && use->unique_out()->Opcode() == Op_Return) {
 726             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 727           } else {
 728             NOT_PRODUCT(fail_eliminate = "Object is referenced by Phi";)
 729           }
 730           DEBUG_ONLY(disq_node = use;)
 731         } else {
 732           if (use->Opcode() == Op_Return) {
 733             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 734           } else {
 735             NOT_PRODUCT(fail_eliminate = "Object is referenced by node";)
 736           }
 737           DEBUG_ONLY(disq_node = use;)
 738         }
 739         can_eliminate = false;
 740       } else {
 741         assert(use->Opcode() == Op_CastP2X, "should be");
 742         assert(!use->has_out_with(Op_OrL), "should have been removed because oop is never null");
 743       }
 744     }
 745   }
 746 
 747 #ifndef PRODUCT
 748   if (PrintEliminateAllocations) {
 749     if (can_eliminate) {
 750       tty->print("Scalar ");
 751       if (res == NULL)
 752         alloc->dump();
 753       else
 754         res->dump();
 755     } else if (alloc->_is_scalar_replaceable) {
 756       tty->print("NotScalar (%s)", fail_eliminate);
 757       if (res == NULL)
 758         alloc->dump();
 759       else
 760         res->dump();
 761 #ifdef ASSERT
 762       if (disq_node != NULL) {
 763           tty->print("  >>>> ");
 764           disq_node->dump();
 765       }
 766 #endif /*ASSERT*/
 767     }
 768   }
 769 #endif
 770   return can_eliminate;
 771 }

 773 // Do scalar replacement.
 774 bool PhaseMacroExpand::scalar_replacement(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 775   GrowableArray <SafePointNode *> safepoints_done;
 776 
 777   ciKlass* klass = NULL;
 778   ciInstanceKlass* iklass = NULL;
 779   int nfields = 0;
 780   int array_base = 0;
 781   int element_size = 0;
 782   BasicType basic_elem_type = T_ILLEGAL;
 783   ciType* elem_type = NULL;
 784 
 785   Node* res = alloc->result_cast();
 786   assert(res == NULL || res->is_CheckCastPP(), "unexpected AllocateNode result");
 787   const TypeOopPtr* res_type = NULL;
 788   if (res != NULL) { // Could be NULL when there are no users
 789     res_type = _igvn.type(res)->isa_oopptr();
 790   }
 791 
 792   if (res != NULL) {
 793     klass = res_type->klass();
 794     if (res_type->isa_instptr()) {
 795       // find the fields of the class which will be needed for safepoint debug information
 796       assert(klass->is_instance_klass(), "must be an instance klass.");
 797       iklass = klass->as_instance_klass();
 798       nfields = iklass->nof_nonstatic_fields();
 799     } else {
 800       // find the array's elements which will be needed for safepoint debug information
 801       nfields = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 802       assert(klass->is_array_klass() && nfields >= 0, "must be an array klass.");
 803       elem_type = klass->as_array_klass()->element_type();
 804       basic_elem_type = elem_type->basic_type();
 805       if (elem_type->is_inlinetype() && !klass->is_flat_array_klass()) {
 806         assert(basic_elem_type == T_INLINE_TYPE, "unexpected element basic type");
 807         basic_elem_type = T_OBJECT;
 808       }
 809       array_base = arrayOopDesc::base_offset_in_bytes(basic_elem_type);
 810       element_size = type2aelembytes(basic_elem_type);
 811       if (klass->is_flat_array_klass()) {
 812         // Flattened inline type array
 813         element_size = klass->as_flat_array_klass()->element_byte_size();
 814       }
 815     }
 816   }
 817   //
 818   // Process the safepoint uses
 819   //
 820   Unique_Node_List value_worklist;
 821   while (safepoints.length() > 0) {
 822     SafePointNode* sfpt = safepoints.pop();
 823     Node* mem = sfpt->memory();
 824     Node* ctl = sfpt->control();
 825     assert(sfpt->jvms() != NULL, "missed JVMS");
 826     // Fields of scalar objs are referenced only at the end
 827     // of regular debuginfo at the last (youngest) JVMS.
 828     // Record relative start index.
 829     uint first_ind = (sfpt->req() - sfpt->jvms()->scloff());
 830     SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(res_type,
 831 #ifdef ASSERT
 832                                                  alloc,
 833 #endif
 834                                                  first_ind, nfields);
 835     sobj->init_req(0, C->root());
 836     transform_later(sobj);
 837 
 838     // Scan object's fields adding an input to the safepoint for each field.
 839     for (int j = 0; j < nfields; j++) {
 840       intptr_t offset;
 841       ciField* field = NULL;
 842       if (iklass != NULL) {
 843         field = iklass->nonstatic_field_at(j);
 844         offset = field->offset();
 845         elem_type = field->type();
 846         basic_elem_type = field->layout_type();
 847         assert(!field->is_flattened(), "flattened inline type fields should not have safepoint uses");
 848       } else {
 849         offset = array_base + j * (intptr_t)element_size;
 850       }
 851 
 852       const Type *field_type;
 853       // The next code is taken from Parse::do_get_xxx().
 854       if (is_reference_type(basic_elem_type)) {
 855         if (!elem_type->is_loaded()) {
 856           field_type = TypeInstPtr::BOTTOM;
 857         } else if (field != NULL && field->is_static_constant()) {
 858           // This can happen if the constant oop is non-perm.
 859           ciObject* con = field->constant_value().as_object();
 860           // Do not "join" in the previous type; it doesn't add value,
 861           // and may yield a vacuous result if the field is of interface type.
 862           field_type = TypeOopPtr::make_from_constant(con)->isa_oopptr();
 863           assert(field_type != NULL, "field singleton type must be consistent");
 864         } else {
 865           field_type = TypeOopPtr::make_from_klass(elem_type->as_klass());
 866         }
 867         if (UseCompressedOops) {
 868           field_type = field_type->make_narrowoop();
 869           basic_elem_type = T_NARROWOOP;
 870         }
 871       } else {
 872         field_type = Type::get_const_basic_type(basic_elem_type);
 873       }
 874 
 875       Node* field_val = NULL;
 876       const TypeOopPtr* field_addr_type = res_type->add_offset(offset)->isa_oopptr();
 877       if (klass->is_flat_array_klass()) {
 878         ciInlineKlass* vk = elem_type->as_inline_klass();
 879         assert(vk->flatten_array(), "must be flattened");
 880         field_val = inline_type_from_mem(mem, ctl, vk, field_addr_type->isa_aryptr(), 0, alloc);
 881       } else {
 882         field_val = value_from_mem(mem, ctl, basic_elem_type, field_type, field_addr_type, alloc);
 883       }
 884       if (field_val == NULL) {
 885         // We weren't able to find a value for this field,
 886         // give up on eliminating this allocation.
 887 
 888         // Remove any extra entries we added to the safepoint.
 889         uint last = sfpt->req() - 1;
 890         for (int k = 0;  k < j; k++) {
 891           sfpt->del_req(last--);
 892         }
 893         _igvn._worklist.push(sfpt);
 894         // rollback processed safepoints
 895         while (safepoints_done.length() > 0) {
 896           SafePointNode* sfpt_done = safepoints_done.pop();
 897           // remove any extra entries we added to the safepoint
 898           last = sfpt_done->req() - 1;
 899           for (int k = 0;  k < nfields; k++) {
 900             sfpt_done->del_req(last--);
 901           }
 902           JVMState *jvms = sfpt_done->jvms();
 903           jvms->set_endoff(sfpt_done->req());
 904           // Now make a pass over the debug information replacing any references
 905           // to SafePointScalarObjectNode with the allocated object.
 906           int start = jvms->debug_start();
 907           int end   = jvms->debug_end();
 908           for (int i = start; i < end; i++) {
 909             if (sfpt_done->in(i)->is_SafePointScalarObject()) {
 910               SafePointScalarObjectNode* scobj = sfpt_done->in(i)->as_SafePointScalarObject();
 911               if (scobj->first_index(jvms) == sfpt_done->req() &&
 912                   scobj->n_fields() == (uint)nfields) {
 913                 assert(scobj->alloc() == alloc, "sanity");
 914                 sfpt_done->set_req(i, res);
 915               }
 916             }
 917           }
 918           _igvn._worklist.push(sfpt_done);
 919         }
 920 #ifndef PRODUCT
 921         if (PrintEliminateAllocations) {
 922           if (field != NULL) {
 923             tty->print("=== At SafePoint node %d can't find value of Field: ",
 924                        sfpt->_idx);
 925             field->print();
 926             int field_idx = C->get_alias_index(field_addr_type);
 927             tty->print(" (alias_idx=%d)", field_idx);
 928           } else { // Array's element
 929             tty->print("=== At SafePoint node %d can't find value of array element [%d]",
 930                        sfpt->_idx, j);
 931           }
 932           tty->print(", which prevents elimination of: ");
 933           if (res == NULL)
 934             alloc->dump();
 935           else
 936             res->dump();
 937         }
 938 #endif
 939         return false;
 940       }
 941       if (field_val->is_InlineType()) {
 942         // Keep track of inline types to scalarize them later
 943         value_worklist.push(field_val);
 944       } else if (UseCompressedOops && field_type->isa_narrowoop()) {
 945         // Enable "DecodeN(EncodeP(Allocate)) --> Allocate" transformation
 946         // to be able scalar replace the allocation.
 947         if (field_val->is_EncodeP()) {
 948           field_val = field_val->in(1);
 949         } else {
 950           field_val = transform_later(new DecodeNNode(field_val, field_val->get_ptr_type()));
 951         }
 952       }
 953       sfpt->add_req(field_val);
 954     }
 955     JVMState *jvms = sfpt->jvms();
 956     jvms->set_endoff(sfpt->req());
 957     // Now make a pass over the debug information replacing any references
 958     // to the allocated object with "sobj"
 959     int start = jvms->debug_start();
 960     int end   = jvms->debug_end();
 961     sfpt->replace_edges_in_range(res, sobj, start, end);
 962     _igvn._worklist.push(sfpt);
 963     safepoints_done.append_if_missing(sfpt); // keep it for rollback
 964   }
 965   // Scalarize inline types that were added to the safepoint
 966   for (uint i = 0; i < value_worklist.size(); ++i) {
 967     Node* vt = value_worklist.at(i);
 968     vt->as_InlineType()->make_scalar_in_safepoints(&_igvn);
 969   }
 970   return true;
 971 }

 973 static void disconnect_projections(MultiNode* n, PhaseIterGVN& igvn) {
 974   Node* ctl_proj = n->proj_out_or_null(TypeFunc::Control);
 975   Node* mem_proj = n->proj_out_or_null(TypeFunc::Memory);
 976   if (ctl_proj != NULL) {
 977     igvn.replace_node(ctl_proj, n->in(0));
 978   }
 979   if (mem_proj != NULL) {
 980     igvn.replace_node(mem_proj, n->in(TypeFunc::Memory));
 981   }

 984 // Process users of eliminated allocation.
 985 void PhaseMacroExpand::process_users_of_allocation(CallNode *alloc, bool inline_alloc) {
 986   Node* res = alloc->result_cast();
 987   if (res != NULL) {
 988     for (DUIterator_Last jmin, j = res->last_outs(jmin); j >= jmin; ) {
 989       Node *use = res->last_out(j);
 990       uint oc1 = res->outcnt();
 991 
 992       if (use->is_AddP()) {
 993         for (DUIterator_Last kmin, k = use->last_outs(kmin); k >= kmin; ) {
 994           Node *n = use->last_out(k);
 995           uint oc2 = use->outcnt();
 996           if (n->is_Store()) {
 997             for (DUIterator_Fast pmax, p = n->fast_outs(pmax); p < pmax; p++) {
 998               MemBarNode* mb = n->fast_out(p)->isa_MemBar();
 999               if (mb != NULL && mb->req() <= MemBarNode::Precedent && mb->in(MemBarNode::Precedent) == n) {
1000                 // MemBarVolatiles should have been removed by MemBarNode::Ideal() for non-inline allocations
1001                 assert(inline_alloc, "MemBarVolatile should be eliminated for non-escaping object");
1002                 mb->remove(&_igvn);
1003               }
1004             }
1005             _igvn.replace_node(n, n->in(MemNode::Memory));
1006           } else {
1007             eliminate_gc_barrier(n);
1008           }
1009           k -= (oc2 - use->outcnt());
1010         }
1011         _igvn.remove_dead_node(use);
1012       } else if (use->is_ArrayCopy()) {
1013         // Disconnect ArrayCopy node
1014         ArrayCopyNode* ac = use->as_ArrayCopy();
1015         if (ac->is_clonebasic()) {
1016           Node* membar_after = ac->proj_out(TypeFunc::Control)->unique_ctrl_out();
1017           disconnect_projections(ac, _igvn);
1018           assert(alloc->in(TypeFunc::Memory)->is_Proj() && alloc->in(TypeFunc::Memory)->in(0)->Opcode() == Op_MemBarCPUOrder, "mem barrier expected before allocation");
1019           Node* membar_before = alloc->in(TypeFunc::Memory)->in(0);
1020           disconnect_projections(membar_before->as_MemBar(), _igvn);
1021           if (membar_after->is_MemBar()) {
1022             disconnect_projections(membar_after->as_MemBar(), _igvn);
1023           }
1024         } else {
1025           assert(ac->is_arraycopy_validated() ||
1026                  ac->is_copyof_validated() ||
1027                  ac->is_copyofrange_validated(), "unsupported");
1028           CallProjections* callprojs = ac->extract_projections(true);
1029 
1030           _igvn.replace_node(callprojs->fallthrough_ioproj, ac->in(TypeFunc::I_O));
1031           _igvn.replace_node(callprojs->fallthrough_memproj, ac->in(TypeFunc::Memory));
1032           _igvn.replace_node(callprojs->fallthrough_catchproj, ac->in(TypeFunc::Control));
1033 
1034           // Set control to top. IGVN will remove the remaining projections
1035           ac->set_req(0, top());
1036           ac->replace_edge(res, top());
1037 
1038           // Disconnect src right away: it can help find new
1039           // opportunities for allocation elimination
1040           Node* src = ac->in(ArrayCopyNode::Src);
1041           ac->replace_edge(src, top());
1042           // src can be top at this point if src and dest of the
1043           // arraycopy were the same
1044           if (src->outcnt() == 0 && !src->is_top()) {
1045             _igvn.remove_dead_node(src);
1046           }
1047         }
1048         _igvn._worklist.push(ac);
1049       } else if (use->is_InlineType()) {
1050         assert(use->isa_InlineType()->get_oop() == res, "unexpected inline type use");
1051         _igvn.rehash_node_delayed(use);
1052         use->isa_InlineType()->set_oop(_igvn.zerocon(T_INLINE_TYPE));
1053       } else if (use->is_Store()) {
1054         _igvn.replace_node(use, use->in(MemNode::Memory));
1055       } else {
1056         eliminate_gc_barrier(use);
1057       }
1058       j -= (oc1 - res->outcnt());
1059     }
1060     assert(res->outcnt() == 0, "all uses of allocated objects must be deleted");
1061     _igvn.remove_dead_node(res);
1062   }
1063 
1064   //
1065   // Process other users of allocation's projections
1066   //
1067   if (_resproj != NULL && _resproj->outcnt() != 0) {
1068     // First disconnect stores captured by Initialize node.
1069     // If Initialize node is eliminated first in the following code,
1070     // it will kill such stores and DUIterator_Last will assert.
1071     for (DUIterator_Fast jmax, j = _resproj->fast_outs(jmax);  j < jmax; j++) {
1072       Node *use = _resproj->fast_out(j);
1073       if (use->is_AddP()) {
1074         // raw memory addresses used only by the initialization
1075         _igvn.replace_node(use, C->top());
1076         --j; --jmax;
1077       }
1078     }
1079     for (DUIterator_Last jmin, j = _resproj->last_outs(jmin); j >= jmin; ) {
1080       Node *use = _resproj->last_out(j);
1081       uint oc1 = _resproj->outcnt();
1082       if (use->is_Initialize()) {
1083         // Eliminate Initialize node.
1084         InitializeNode *init = use->as_Initialize();
1085         assert(init->outcnt() <= 2, "only a control and memory projection expected");
1086         Node *ctrl_proj = init->proj_out_or_null(TypeFunc::Control);
1087         if (ctrl_proj != NULL) {
1088           // Inline type buffer allocations are followed by a membar
1089           Node* membar_after = ctrl_proj->unique_ctrl_out();
1090           if (inline_alloc && membar_after->Opcode() == Op_MemBarCPUOrder) {
1091             membar_after->as_MemBar()->remove(&_igvn);
1092           }
1093           _igvn.replace_node(ctrl_proj, init->in(TypeFunc::Control));
1094 #ifdef ASSERT
1095           Node* tmp = init->in(TypeFunc::Control);
1096           assert(tmp == _fallthroughcatchproj, "allocation control projection");
1097 #endif
1098         }
1099         Node *mem_proj = init->proj_out_or_null(TypeFunc::Memory);
1100         if (mem_proj != NULL) {
1101           Node *mem = init->in(TypeFunc::Memory);
1102 #ifdef ASSERT
1103           if (mem->is_MergeMem()) {
1104             assert(mem->in(TypeFunc::Memory) == _memproj_fallthrough, "allocation memory projection");
1105           } else {
1106             assert(mem == _memproj_fallthrough, "allocation memory projection");
1107           }
1108 #endif
1109           _igvn.replace_node(mem_proj, mem);
1110         }
1111       } else if (use->Opcode() == Op_MemBarStoreStore) {
1112         // Inline type buffer allocations are followed by a membar
1113         assert(inline_alloc, "Unexpected MemBarStoreStore");
1114         use->as_MemBar()->remove(&_igvn);
1115       } else  {
1116         assert(false, "only Initialize or AddP expected");
1117       }
1118       j -= (oc1 - _resproj->outcnt());
1119     }
1120   }
1121   if (_fallthroughcatchproj != NULL) {
1122     _igvn.replace_node(_fallthroughcatchproj, alloc->in(TypeFunc::Control));
1123   }
1124   if (_memproj_fallthrough != NULL) {
1125     _igvn.replace_node(_memproj_fallthrough, alloc->in(TypeFunc::Memory));
1126   }
1127   if (_memproj_catchall != NULL) {
1128     _igvn.replace_node(_memproj_catchall, C->top());
1129   }
1130   if (_ioproj_fallthrough != NULL) {
1131     _igvn.replace_node(_ioproj_fallthrough, alloc->in(TypeFunc::I_O));
1132   }
1133   if (_ioproj_catchall != NULL) {
1134     _igvn.replace_node(_ioproj_catchall, C->top());
1135   }
1136   if (_catchallcatchproj != NULL) {
1137     _igvn.replace_node(_catchallcatchproj, C->top());
1138   }
1139 }

1141 bool PhaseMacroExpand::eliminate_allocate_node(AllocateNode *alloc) {
1142   // Don't do scalar replacement if the frame can be popped by JVMTI:
1143   // if reallocation fails during deoptimization we'll pop all
1144   // interpreter frames for this compiled frame and that won't play
1145   // nice with JVMTI popframe.
1146   if (!EliminateAllocations || JvmtiExport::can_pop_frame()) {
1147     return false;
1148   }
1149   Node* klass = alloc->in(AllocateNode::KlassNode);
1150   const TypeKlassPtr* tklass = _igvn.type(klass)->is_klassptr();
1151 
1152   // Attempt to eliminate inline type buffer allocations
1153   // regardless of usage and escape/replaceable status.
1154   bool inline_alloc = tklass->klass()->is_inlinetype();
1155   if (!alloc->_is_non_escaping && !inline_alloc) {
1156     return false;
1157   }
1158   // Eliminate boxing allocations which are not used
1159   // regardless of scalar replaceable status.
1160   Node* res = alloc->result_cast();
1161   bool boxing_alloc = (res == NULL) && C->eliminate_boxing() &&
1162                       tklass->klass()->is_instance_klass() &&
1163                       tklass->klass()->as_instance_klass()->is_box_klass();
1164   if (!alloc->_is_scalar_replaceable && !boxing_alloc && !inline_alloc) {
1165     return false;
1166   }
1167 
1168   extract_call_projections(alloc);
1169 
1170   GrowableArray <SafePointNode *> safepoints;
1171   if (!can_eliminate_allocation(alloc, safepoints)) {
1172     return false;
1173   }
1174 
1175   if (!alloc->_is_scalar_replaceable) {
1176     assert(res == NULL || inline_alloc, "sanity");
1177     // We can only eliminate allocation if all debug info references
1178     // are already replaced with SafePointScalarObject because
1179     // we can't search for a fields value without instance_id.
1180     if (safepoints.length() > 0) {
1181       assert(!inline_alloc, "Inline type allocations should not have safepoint uses");
1182       return false;
1183     }
1184   }
1185 
1186   if (!scalar_replacement(alloc, safepoints)) {
1187     return false;
1188   }
1189 
1190   CompileLog* log = C->log();
1191   if (log != NULL) {
1192     log->head("eliminate_allocation type='%d'",
1193               log->identify(tklass->klass()));
1194     JVMState* p = alloc->jvms();
1195     while (p != NULL) {
1196       log->elem("jvms bci='%d' method='%d'", p->bci(), log->identify(p->method()));
1197       p = p->caller();
1198     }
1199     log->tail("eliminate_allocation");
1200   }
1201 
1202   process_users_of_allocation(alloc, inline_alloc);
1203 
1204 #ifndef PRODUCT
1205   if (PrintEliminateAllocations) {
1206     if (alloc->is_AllocateArray())
1207       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1208     else
1209       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1210   }
1211 #endif
1212 
1213   return true;
1214 }

1290 //=============================================================================
1291 //
1292 //                              A L L O C A T I O N
1293 //
1294 // Allocation attempts to be fast in the case of frequent small objects.
1295 // It breaks down like this:
1296 //
1297 // 1) Size in doublewords is computed.  This is a constant for objects and
1298 // variable for most arrays.  Doubleword units are used to avoid size
1299 // overflow of huge doubleword arrays.  We need doublewords in the end for
1300 // rounding.
1301 //
1302 // 2) Size is checked for being 'too large'.  Too-large allocations will go
1303 // the slow path into the VM.  The slow path can throw any required
1304 // exceptions, and does all the special checks for very large arrays.  The
1305 // size test can constant-fold away for objects.  For objects with
1306 // finalizers it constant-folds the otherway: you always go slow with
1307 // finalizers.
1308 //
1309 // 3) If NOT using TLABs, this is the contended loop-back point.
1310 // Load-Locked the heap top.  If using TLABs normal-load the heap top.
1311 //
1312 // 4) Check that heap top + size*8 < max.  If we fail go the slow ` route.
1313 // NOTE: "top+size*8" cannot wrap the 4Gig line!  Here's why: for largish
1314 // "size*8" we always enter the VM, where "largish" is a constant picked small
1315 // enough that there's always space between the eden max and 4Gig (old space is
1316 // there so it's quite large) and large enough that the cost of entering the VM
1317 // is dwarfed by the cost to initialize the space.
1318 //
1319 // 5) If NOT using TLABs, Store-Conditional the adjusted heap top back
1320 // down.  If contended, repeat at step 3.  If using TLABs normal-store
1321 // adjusted heap top back down; there is no contention.
1322 //
1323 // 6) If !ZeroTLAB then Bulk-clear the object/array.  Fill in klass & mark
1324 // fields.
1325 //
1326 // 7) Merge with the slow-path; cast the raw memory pointer to the correct
1327 // oop flavor.
1328 //
1329 //=============================================================================
1330 // FastAllocateSizeLimit value is in DOUBLEWORDS.
1331 // Allocations bigger than this always go the slow route.
1332 // This value must be small enough that allocation attempts that need to
1333 // trigger exceptions go the slow route.  Also, it must be small enough so
1334 // that heap_top + size_in_bytes does not wrap around the 4Gig limit.
1335 //=============================================================================j//
1336 // %%% Here is an old comment from parseHelper.cpp; is it outdated?
1337 // The allocator will coalesce int->oop copies away.  See comment in
1338 // coalesce.cpp about how this works.  It depends critically on the exact
1339 // code shape produced here, so if you are changing this code shape
1340 // make sure the GC info for the heap-top is correct in and around the
1341 // slow-path call.
1342 //
1343 
1344 void PhaseMacroExpand::expand_allocate_common(
1345             AllocateNode* alloc, // allocation node to be expanded
1346             Node* length,  // array length for an array allocation
1347             const TypeFunc* slow_call_type, // Type of slow call
1348             address slow_call_address  // Address of slow call
1349     )
1350 {
1351   Node* ctrl = alloc->in(TypeFunc::Control);
1352   Node* mem  = alloc->in(TypeFunc::Memory);
1353   Node* i_o  = alloc->in(TypeFunc::I_O);
1354   Node* size_in_bytes     = alloc->in(AllocateNode::AllocSize);
1355   Node* klass_node        = alloc->in(AllocateNode::KlassNode);
1356   Node* initial_slow_test = alloc->in(AllocateNode::InitialTest);
1357   assert(ctrl != NULL, "must have control");
1358 
1359   // We need a Region and corresponding Phi's to merge the slow-path and fast-path results.
1360   // they will not be used if "always_slow" is set
1361   enum { slow_result_path = 1, fast_result_path = 2 };
1362   Node *result_region = NULL;
1363   Node *result_phi_rawmem = NULL;
1364   Node *result_phi_rawoop = NULL;
1365   Node *result_phi_i_o = NULL;
1366 
1367   // The initial slow comparison is a size check, the comparison
1368   // we want to do is a BoolTest::gt
1369   bool expand_fast_path = true;
1370   int tv = _igvn.find_int_con(initial_slow_test, -1);
1371   if (tv >= 0) {
1372     // InitialTest has constant result
1373     //   0 - can fit in TLAB
1374     //   1 - always too big or negative
1375     assert(tv <= 1, "0 or 1 if a constant");
1376     expand_fast_path = (tv == 0);
1377     initial_slow_test = NULL;
1378   } else {
1379     initial_slow_test = BoolNode::make_predicate(initial_slow_test, &_igvn);
1380   }
1381 
1382   if (C->env()->dtrace_alloc_probes() ||
1383       (!UseTLAB && !Universe::heap()->supports_inline_contig_alloc())) {
1384     // Force slow-path allocation
1385     expand_fast_path = false;
1386     initial_slow_test = NULL;
1387   }
1388 
1389   bool allocation_has_use = (alloc->result_cast() != NULL);
1390   if (!allocation_has_use) {
1391     InitializeNode* init = alloc->initialization();
1392     if (init != NULL) {
1393       init->remove(&_igvn);
1394     }
1395     if (expand_fast_path && (initial_slow_test == NULL)) {
1396       // Remove allocation node and return.
1397       // Size is a non-negative constant -> no initial check needed -> directly to fast path.
1398       // Also, no usages -> empty fast path -> no fall out to slow path -> nothing left.
1399 #ifndef PRODUCT
1400       if (PrintEliminateAllocations) {
1401         tty->print("NotUsed ");
1402         Node* res = alloc->proj_out_or_null(TypeFunc::Parms);
1403         if (res != NULL) {
1404           res->dump();
1405         } else {
1406           alloc->dump();
1407         }
1408       }
1409 #endif
1410       yank_alloc_node(alloc);
1411       return;
1412     }
1413   }
1414 
1415   enum { too_big_or_final_path = 1, need_gc_path = 2 };
1416   Node *slow_region = NULL;
1417   Node *toobig_false = ctrl;
1418 
1419   // generate the initial test if necessary
1420   if (initial_slow_test != NULL ) {
1421     assert (expand_fast_path, "Only need test if there is a fast path");
1422     slow_region = new RegionNode(3);
1423 
1424     // Now make the initial failure test.  Usually a too-big test but
1425     // might be a TRUE for finalizers or a fancy class check for
1426     // newInstance0.
1427     IfNode* toobig_iff = new IfNode(ctrl, initial_slow_test, PROB_MIN, COUNT_UNKNOWN);
1428     transform_later(toobig_iff);
1429     // Plug the failing-too-big test into the slow-path region
1430     Node* toobig_true = new IfTrueNode(toobig_iff);
1431     transform_later(toobig_true);
1432     slow_region    ->init_req( too_big_or_final_path, toobig_true );
1433     toobig_false = new IfFalseNode(toobig_iff);
1434     transform_later(toobig_false);
1435   } else {
1436     // No initial test, just fall into next case
1437     assert(allocation_has_use || !expand_fast_path, "Should already have been handled");
1438     toobig_false = ctrl;
1439     debug_only(slow_region = NodeSentinel);
1440   }
1441 
1442   // If we are here there are several possibilities
1443   // - expand_fast_path is false - then only a slow path is expanded. That's it.
1444   // no_initial_check means a constant allocation.
1445   // - If check always evaluates to false -> expand_fast_path is false (see above)
1446   // - If check always evaluates to true -> directly into fast path (but may bailout to slowpath)
1447   // if !allocation_has_use the fast path is empty
1448   // if !allocation_has_use && no_initial_check
1449   // - Then there are no fastpath that can fall out to slowpath -> no allocation code at all.
1450   //   removed by yank_alloc_node above.
1451 
1452   Node *slow_mem = mem;  // save the current memory state for slow path
1453   // generate the fast allocation code unless we know that the initial test will always go slow
1454   if (expand_fast_path) {
1455     // Fast path modifies only raw memory.
1456     if (mem->is_MergeMem()) {
1457       mem = mem->as_MergeMem()->memory_at(Compile::AliasIdxRaw);
1458     }
1459 
1460     // allocate the Region and Phi nodes for the result
1461     result_region = new RegionNode(3);
1462     result_phi_rawmem = new PhiNode(result_region, Type::MEMORY, TypeRawPtr::BOTTOM);
1463     result_phi_i_o    = new PhiNode(result_region, Type::ABIO); // I/O is used for Prefetch
1464 
1465     // Grab regular I/O before optional prefetch may change it.
1466     // Slow-path does no I/O so just set it to the original I/O.
1467     result_phi_i_o->init_req(slow_result_path, i_o);
1468 
1469     // Name successful fast-path variables
1470     Node* fast_oop_ctrl;
1471     Node* fast_oop_rawmem;
1472 
1473     if (allocation_has_use) {
1474       Node* needgc_ctrl = NULL;
1475       result_phi_rawoop = new PhiNode(result_region, TypeRawPtr::BOTTOM);
1476 
1477       intx prefetch_lines = length != NULL ? AllocatePrefetchLines : AllocateInstancePrefetchLines;
1478       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1479       Node* fast_oop = bs->obj_allocate(this, ctrl, mem, toobig_false, size_in_bytes, i_o, needgc_ctrl,
1480                                         fast_oop_ctrl, fast_oop_rawmem,
1481                                         prefetch_lines);
1482 
1483       if (initial_slow_test != NULL) {
1484         // This completes all paths into the slow merge point
1485         slow_region->init_req(need_gc_path, needgc_ctrl);
1486         transform_later(slow_region);
1487       } else {
1488         // No initial slow path needed!
1489         // Just fall from the need-GC path straight into the VM call.
1490         slow_region = needgc_ctrl;
1491       }
1492 
1493       InitializeNode* init = alloc->initialization();
1494       fast_oop_rawmem = initialize_object(alloc,
1495                                           fast_oop_ctrl, fast_oop_rawmem, fast_oop,
1496                                           klass_node, length, size_in_bytes);
1497       expand_initialize_membar(alloc, init, fast_oop_ctrl, fast_oop_rawmem);
1498       expand_dtrace_alloc_probe(alloc, fast_oop, fast_oop_ctrl, fast_oop_rawmem);
1499 
1500       result_phi_rawoop->init_req(fast_result_path, fast_oop);
1501     } else {
1502       assert (initial_slow_test != NULL, "sanity");
1503       fast_oop_ctrl   = toobig_false;
1504       fast_oop_rawmem = mem;
1505       transform_later(slow_region);
1506     }
1507 
1508     // Plug in the successful fast-path into the result merge point
1509     result_region    ->init_req(fast_result_path, fast_oop_ctrl);
1510     result_phi_i_o   ->init_req(fast_result_path, i_o);
1511     result_phi_rawmem->init_req(fast_result_path, fast_oop_rawmem);
1512   } else {
1513     slow_region = ctrl;
1514     result_phi_i_o = i_o; // Rename it to use in the following code.
1515   }
1516 
1517   // Generate slow-path call
1518   CallNode *call = new CallStaticJavaNode(slow_call_type, slow_call_address,
1519                                OptoRuntime::stub_name(slow_call_address),
1520                                alloc->jvms()->bci(),
1521                                TypePtr::BOTTOM);
1522   call->init_req(TypeFunc::Control,   slow_region);
1523   call->init_req(TypeFunc::I_O,       top());    // does no i/o
1524   call->init_req(TypeFunc::Memory,    slow_mem); // may gc ptrs
1525   call->init_req(TypeFunc::ReturnAdr, alloc->in(TypeFunc::ReturnAdr));
1526   call->init_req(TypeFunc::FramePtr,  alloc->in(TypeFunc::FramePtr));
1527 
1528   call->init_req(TypeFunc::Parms+0, klass_node);
1529   if (length != NULL) {
1530     call->init_req(TypeFunc::Parms+1, length);
1531   } else {
1532     // Let the runtime know if this is a larval allocation
1533     call->init_req(TypeFunc::Parms+1, _igvn.intcon(alloc->_larval));
1534   }
1535 
1536   // Copy debug information and adjust JVMState information, then replace
1537   // allocate node with the call
1538   call->copy_call_debug_info(&_igvn, alloc);
1539   if (expand_fast_path) {
1540     call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
1541   } else {
1542     // Hook i_o projection to avoid its elimination during allocation
1543     // replacement (when only a slow call is generated).
1544     call->set_req(TypeFunc::I_O, result_phi_i_o);
1545   }
1546   _igvn.replace_node(alloc, call);
1547   transform_later(call);
1548 
1549   // Identify the output projections from the allocate node and
1550   // adjust any references to them.
1551   // The control and io projections look like:
1552   //
1553   //        v---Proj(ctrl) <-----+   v---CatchProj(ctrl)
1554   //  Allocate                   Catch
1555   //        ^---Proj(io) <-------+   ^---CatchProj(io)
1556   //
1557   //  We are interested in the CatchProj nodes.
1558   //
1559   extract_call_projections(call);
1560 
1561   // An allocate node has separate memory projections for the uses on
1562   // the control and i_o paths. Replace the control memory projection with
1563   // result_phi_rawmem (unless we are only generating a slow call when
1564   // both memory projections are combined)
1565   if (expand_fast_path && _memproj_fallthrough != NULL) {
1566     _igvn.replace_in_uses(_memproj_fallthrough, result_phi_rawmem);
1567   }
1568   // Now change uses of _memproj_catchall to use _memproj_fallthrough and delete
1569   // _memproj_catchall so we end up with a call that has only 1 memory projection.
1570   if (_memproj_catchall != NULL) {
1571     if (_memproj_fallthrough == NULL) {
1572       _memproj_fallthrough = new ProjNode(call, TypeFunc::Memory);
1573       transform_later(_memproj_fallthrough);
1574     }
1575     _igvn.replace_in_uses(_memproj_catchall, _memproj_fallthrough);
1576     _igvn.remove_dead_node(_memproj_catchall);
1577   }
1578 
1579   // An allocate node has separate i_o projections for the uses on the control
1580   // and i_o paths. Always replace the control i_o projection with result i_o
1581   // otherwise incoming i_o become dead when only a slow call is generated
1582   // (it is different from memory projections where both projections are
1583   // combined in such case).
1584   if (_ioproj_fallthrough != NULL) {
1585     _igvn.replace_in_uses(_ioproj_fallthrough, result_phi_i_o);
1586   }
1587   // Now change uses of _ioproj_catchall to use _ioproj_fallthrough and delete
1588   // _ioproj_catchall so we end up with a call that has only 1 i_o projection.
1589   if (_ioproj_catchall != NULL) {
1590     if (_ioproj_fallthrough == NULL) {
1591       _ioproj_fallthrough = new ProjNode(call, TypeFunc::I_O);
1592       transform_later(_ioproj_fallthrough);
1593     }
1594     _igvn.replace_in_uses(_ioproj_catchall, _ioproj_fallthrough);
1595     _igvn.remove_dead_node(_ioproj_catchall);
1596   }
1597 
1598   // if we generated only a slow call, we are done
1599   if (!expand_fast_path) {
1600     // Now we can unhook i_o.
1601     if (result_phi_i_o->outcnt() > 1) {
1602       call->set_req(TypeFunc::I_O, top());
1603     } else {
1604       assert(result_phi_i_o->unique_ctrl_out() == call, "sanity");
1605       // Case of new array with negative size known during compilation.
1606       // AllocateArrayNode::Ideal() optimization disconnect unreachable
1607       // following code since call to runtime will throw exception.
1608       // As result there will be no users of i_o after the call.
1609       // Leave i_o attached to this call to avoid problems in preceding graph.
1610     }
1611     return;
1612   }
1613 
1614   if (_fallthroughcatchproj != NULL) {
1615     ctrl = _fallthroughcatchproj->clone();
1616     transform_later(ctrl);
1617     _igvn.replace_node(_fallthroughcatchproj, result_region);
1618   } else {
1619     ctrl = top();
1620   }
1621   Node *slow_result;
1622   if (_resproj == NULL) {
1623     // no uses of the allocation result
1624     slow_result = top();
1625   } else {
1626     slow_result = _resproj->clone();
1627     transform_later(slow_result);
1628     _igvn.replace_node(_resproj, result_phi_rawoop);
1629   }
1630 
1631   // Plug slow-path into result merge point
1632   result_region->init_req( slow_result_path, ctrl);
1633   transform_later(result_region);
1634   if (allocation_has_use) {
1635     result_phi_rawoop->init_req(slow_result_path, slow_result);
1636     transform_later(result_phi_rawoop);
1637   }
1638   result_phi_rawmem->init_req(slow_result_path, _memproj_fallthrough);
1639   transform_later(result_phi_rawmem);
1640   transform_later(result_phi_i_o);
1641   // This completes all paths into the result merge point
1642 }

1645 void PhaseMacroExpand::yank_alloc_node(AllocateNode* alloc) {
1646   Node* ctrl = alloc->in(TypeFunc::Control);
1647   Node* mem  = alloc->in(TypeFunc::Memory);
1648   Node* i_o  = alloc->in(TypeFunc::I_O);
1649 
1650   extract_call_projections(alloc);
1651   if (_resproj != NULL) {
1652     for (DUIterator_Fast imax, i = _resproj->fast_outs(imax); i < imax; i++) {
1653       Node* use = _resproj->fast_out(i);
1654       use->isa_MemBar()->remove(&_igvn);
1655       --imax;
1656       --i; // back up iterator
1657     }
1658     assert(_resproj->outcnt() == 0, "all uses must be deleted");
1659     _igvn.remove_dead_node(_resproj);
1660   }
1661   if (_fallthroughcatchproj != NULL) {
1662     _igvn.replace_in_uses(_fallthroughcatchproj, ctrl);
1663     _igvn.remove_dead_node(_fallthroughcatchproj);
1664   }
1665   if (_catchallcatchproj != NULL) {
1666     _igvn.rehash_node_delayed(_catchallcatchproj);
1667     _catchallcatchproj->set_req(0, top());
1668   }
1669   if (_fallthroughproj != NULL) {
1670     Node* catchnode = _fallthroughproj->unique_ctrl_out();
1671     _igvn.remove_dead_node(catchnode);
1672     _igvn.remove_dead_node(_fallthroughproj);
1673   }
1674   if (_memproj_fallthrough != NULL) {
1675     _igvn.replace_in_uses(_memproj_fallthrough, mem);
1676     _igvn.remove_dead_node(_memproj_fallthrough);
1677   }
1678   if (_ioproj_fallthrough != NULL) {
1679     _igvn.replace_in_uses(_ioproj_fallthrough, i_o);
1680     _igvn.remove_dead_node(_ioproj_fallthrough);
1681   }
1682   if (_memproj_catchall != NULL) {
1683     _igvn.rehash_node_delayed(_memproj_catchall);
1684     _memproj_catchall->set_req(0, top());
1685   }
1686   if (_ioproj_catchall != NULL) {
1687     _igvn.rehash_node_delayed(_ioproj_catchall);
1688     _ioproj_catchall->set_req(0, top());
1689   }
1690 #ifndef PRODUCT
1691   if (PrintEliminateAllocations) {
1692     if (alloc->is_AllocateArray()) {
1693       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1694     } else {
1695       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1696     }
1697   }
1698 #endif
1699   _igvn.remove_dead_node(alloc);
1700 }

1702 void PhaseMacroExpand::expand_initialize_membar(AllocateNode* alloc, InitializeNode* init,
1703                                                 Node*& fast_oop_ctrl, Node*& fast_oop_rawmem) {
1704   // If initialization is performed by an array copy, any required
1705   // MemBarStoreStore was already added. If the object does not
1706   // escape no need for a MemBarStoreStore. If the object does not
1707   // escape in its initializer and memory barrier (MemBarStoreStore or
1708   // stronger) is already added at exit of initializer, also no need
1709   // for a MemBarStoreStore. Otherwise we need a MemBarStoreStore
1710   // so that stores that initialize this object can't be reordered
1711   // with a subsequent store that makes this object accessible by
1712   // other threads.
1713   // Other threads include java threads and JVM internal threads
1714   // (for example concurrent GC threads). Current concurrent GC
1715   // implementation: G1 will not scan newly created object,
1716   // so it's safe to skip storestore barrier when allocation does
1717   // not escape.
1718   if (!alloc->does_not_escape_thread() &&
1719     !alloc->is_allocation_MemBar_redundant() &&
1720     (init == NULL || !init->is_complete_with_arraycopy())) {
1721     if (init == NULL || init->req() < InitializeNode::RawStores) {
1722       // No InitializeNode or no stores captured by zeroing
1723       // elimination. Simply add the MemBarStoreStore after object
1724       // initialization.
1725       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1726       transform_later(mb);
1727 
1728       mb->init_req(TypeFunc::Memory, fast_oop_rawmem);
1729       mb->init_req(TypeFunc::Control, fast_oop_ctrl);
1730       fast_oop_ctrl = new ProjNode(mb, TypeFunc::Control);
1731       transform_later(fast_oop_ctrl);
1732       fast_oop_rawmem = new ProjNode(mb, TypeFunc::Memory);
1733       transform_later(fast_oop_rawmem);
1734     } else {
1735       // Add the MemBarStoreStore after the InitializeNode so that
1736       // all stores performing the initialization that were moved
1737       // before the InitializeNode happen before the storestore
1738       // barrier.
1739 
1740       Node* init_ctrl = init->proj_out_or_null(TypeFunc::Control);
1741       Node* init_mem = init->proj_out_or_null(TypeFunc::Memory);
1742 
1743       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1744       transform_later(mb);
1745 
1746       Node* ctrl = new ProjNode(init, TypeFunc::Control);
1747       transform_later(ctrl);
1748       Node* mem = new ProjNode(init, TypeFunc::Memory);
1749       transform_later(mem);
1750 
1751       // The MemBarStoreStore depends on control and memory coming
1752       // from the InitializeNode
1753       mb->init_req(TypeFunc::Memory, mem);
1754       mb->init_req(TypeFunc::Control, ctrl);
1755 
1756       ctrl = new ProjNode(mb, TypeFunc::Control);
1757       transform_later(ctrl);
1758       mem = new ProjNode(mb, TypeFunc::Memory);
1759       transform_later(mem);
1760 
1761       // All nodes that depended on the InitializeNode for control
1762       // and memory must now depend on the MemBarNode that itself
1763       // depends on the InitializeNode
1764       if (init_ctrl != NULL) {
1765         _igvn.replace_node(init_ctrl, ctrl);
1766       }
1767       if (init_mem != NULL) {
1768         _igvn.replace_node(init_mem, mem);
1769       }
1770     }
1771   }
1772 }

1803 // Helper for PhaseMacroExpand::expand_allocate_common.
1804 // Initializes the newly-allocated storage.
1805 Node* PhaseMacroExpand::initialize_object(AllocateNode* alloc,
1806                                           Node* control, Node* rawmem, Node* object,
1807                                           Node* klass_node, Node* length,
1808                                           Node* size_in_bytes) {
1809   InitializeNode* init = alloc->initialization();
1810   // Store the klass & mark bits
1811   Node* mark_node = alloc->make_ideal_mark(&_igvn, control, rawmem);
1812   if (!mark_node->is_Con()) {
1813     transform_later(mark_node);
1814   }
1815   rawmem = make_store(control, rawmem, object, oopDesc::mark_offset_in_bytes(), mark_node, TypeX_X->basic_type());
1816 
1817   rawmem = make_store(control, rawmem, object, oopDesc::klass_offset_in_bytes(), klass_node, T_METADATA);
1818   int header_size = alloc->minimum_header_size();  // conservatively small
1819 
1820   // Array length
1821   if (length != NULL) {         // Arrays need length field
1822     rawmem = make_store(control, rawmem, object, arrayOopDesc::length_offset_in_bytes(), length, T_INT);
1823     // conservatively small header size:
1824     header_size = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1825     ciKlass* k = _igvn.type(klass_node)->is_klassptr()->klass();
1826     if (k->is_array_klass())    // we know the exact header size in most cases:
1827       header_size = Klass::layout_helper_header_size(k->layout_helper());
1828   }
1829 
1830   // Clear the object body, if necessary.
1831   if (init == NULL) {
1832     // The init has somehow disappeared; be cautious and clear everything.
1833     //
1834     // This can happen if a node is allocated but an uncommon trap occurs
1835     // immediately.  In this case, the Initialize gets associated with the
1836     // trap, and may be placed in a different (outer) loop, if the Allocate
1837     // is in a loop.  If (this is rare) the inner loop gets unrolled, then
1838     // there can be two Allocates to one Initialize.  The answer in all these
1839     // edge cases is safety first.  It is always safe to clear immediately
1840     // within an Allocate, and then (maybe or maybe not) clear some more later.
1841     if (!(UseTLAB && ZeroTLAB)) {
1842       rawmem = ClearArrayNode::clear_memory(control, rawmem, object,
1843                                             alloc->in(AllocateNode::DefaultValue),
1844                                             alloc->in(AllocateNode::RawDefaultValue),
1845                                             header_size, size_in_bytes,
1846                                             &_igvn);
1847     }
1848   } else {
1849     if (!init->is_complete()) {
1850       // Try to win by zeroing only what the init does not store.
1851       // We can also try to do some peephole optimizations,
1852       // such as combining some adjacent subword stores.
1853       rawmem = init->complete_stores(control, rawmem, object,
1854                                      header_size, size_in_bytes, &_igvn);
1855     }
1856     // We have no more use for this link, since the AllocateNode goes away:
1857     init->set_req(InitializeNode::RawAddress, top());
1858     // (If we keep the link, it just confuses the register allocator,
1859     // who thinks he sees a real use of the address by the membar.)
1860   }
1861 
1862   return rawmem;
1863 }

1865 // Generate prefetch instructions for next allocations.
1866 Node* PhaseMacroExpand::prefetch_allocation(Node* i_o, Node*& needgc_false,
1867                                         Node*& contended_phi_rawmem,
1868                                         Node* old_eden_top, Node* new_eden_top,
1869                                         intx lines) {
1870    enum { fall_in_path = 1, pf_path = 2 };
1871    if( UseTLAB && AllocatePrefetchStyle == 2 ) {
1872       // Generate prefetch allocation with watermark check.
1873       // As an allocation hits the watermark, we will prefetch starting
1874       // at a "distance" away from watermark.
1875 
1876       Node *pf_region = new RegionNode(3);
1877       Node *pf_phi_rawmem = new PhiNode( pf_region, Type::MEMORY,
1878                                                 TypeRawPtr::BOTTOM );
1879       // I/O is used for Prefetch
1880       Node *pf_phi_abio = new PhiNode( pf_region, Type::ABIO );
1881 
1882       Node *thread = new ThreadLocalNode();
1883       transform_later(thread);
1884 
1885       Node *eden_pf_adr = new AddPNode( top()/*not oop*/, thread,
1886                    _igvn.MakeConX(in_bytes(JavaThread::tlab_pf_top_offset())) );
1887       transform_later(eden_pf_adr);
1888 
1889       Node *old_pf_wm = new LoadPNode(needgc_false,
1890                                    contended_phi_rawmem, eden_pf_adr,
1891                                    TypeRawPtr::BOTTOM, TypeRawPtr::BOTTOM,
1892                                    MemNode::unordered);
1893       transform_later(old_pf_wm);
1894 
1895       // check against new_eden_top
1896       Node *need_pf_cmp = new CmpPNode( new_eden_top, old_pf_wm );
1897       transform_later(need_pf_cmp);
1898       Node *need_pf_bol = new BoolNode( need_pf_cmp, BoolTest::ge );
1899       transform_later(need_pf_bol);
1900       IfNode *need_pf_iff = new IfNode( needgc_false, need_pf_bol,
1901                                        PROB_UNLIKELY_MAG(4), COUNT_UNKNOWN );
1902       transform_later(need_pf_iff);
1903 
1904       // true node, add prefetchdistance
1905       Node *need_pf_true = new IfTrueNode( need_pf_iff );
1906       transform_later(need_pf_true);
1907 
1908       Node *need_pf_false = new IfFalseNode( need_pf_iff );
1909       transform_later(need_pf_false);
1910 
1911       Node *new_pf_wmt = new AddPNode( top(), old_pf_wm,
1912                                     _igvn.MakeConX(AllocatePrefetchDistance) );
1913       transform_later(new_pf_wmt );
1914       new_pf_wmt->set_req(0, need_pf_true);
1915 
1916       Node *store_new_wmt = new StorePNode(need_pf_true,
1917                                        contended_phi_rawmem, eden_pf_adr,
1918                                        TypeRawPtr::BOTTOM, new_pf_wmt,
1919                                        MemNode::unordered);
1920       transform_later(store_new_wmt);
1921 
1922       // adding prefetches
1923       pf_phi_abio->init_req( fall_in_path, i_o );
1924 
1925       Node *prefetch_adr;
1926       Node *prefetch;
1927       uint step_size = AllocatePrefetchStepSize;
1928       uint distance = 0;
1929 
1930       for ( intx i = 0; i < lines; i++ ) {
1931         prefetch_adr = new AddPNode( old_pf_wm, new_pf_wmt,
1932                                             _igvn.MakeConX(distance) );
1933         transform_later(prefetch_adr);
1934         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
1935         transform_later(prefetch);
1936         distance += step_size;
1937         i_o = prefetch;
1938       }
1939       pf_phi_abio->set_req( pf_path, i_o );
1940 
1941       pf_region->init_req( fall_in_path, need_pf_false );
1942       pf_region->init_req( pf_path, need_pf_true );
1943 
1944       pf_phi_rawmem->init_req( fall_in_path, contended_phi_rawmem );
1945       pf_phi_rawmem->init_req( pf_path, store_new_wmt );
1946 
1947       transform_later(pf_region);
1948       transform_later(pf_phi_rawmem);
1949       transform_later(pf_phi_abio);
1950 
1951       needgc_false = pf_region;
1952       contended_phi_rawmem = pf_phi_rawmem;
1953       i_o = pf_phi_abio;
1954    } else if( UseTLAB && AllocatePrefetchStyle == 3 ) {
1955       // Insert a prefetch instruction for each allocation.
1956       // This code is used to generate 1 prefetch instruction per cache line.
1957 
1958       // Generate several prefetch instructions.
1959       uint step_size = AllocatePrefetchStepSize;
1960       uint distance = AllocatePrefetchDistance;
1961 
1962       // Next cache address.
1963       Node *cache_adr = new AddPNode(old_eden_top, old_eden_top,
1964                                      _igvn.MakeConX(step_size + distance));
1965       transform_later(cache_adr);
1966       cache_adr = new CastP2XNode(needgc_false, cache_adr);
1967       transform_later(cache_adr);
1968       // Address is aligned to execute prefetch to the beginning of cache line size
1969       // (it is important when BIS instruction is used on SPARC as prefetch).
1970       Node* mask = _igvn.MakeConX(~(intptr_t)(step_size-1));
1971       cache_adr = new AndXNode(cache_adr, mask);
1972       transform_later(cache_adr);
1973       cache_adr = new CastX2PNode(cache_adr);
1974       transform_later(cache_adr);
1975 
1976       // Prefetch
1977       Node *prefetch = new PrefetchAllocationNode( contended_phi_rawmem, cache_adr );
1978       prefetch->set_req(0, needgc_false);
1979       transform_later(prefetch);
1980       contended_phi_rawmem = prefetch;
1981       Node *prefetch_adr;
1982       distance = step_size;
1983       for ( intx i = 1; i < lines; i++ ) {
1984         prefetch_adr = new AddPNode( cache_adr, cache_adr,
1985                                             _igvn.MakeConX(distance) );
1986         transform_later(prefetch_adr);
1987         prefetch = new PrefetchAllocationNode( contended_phi_rawmem, prefetch_adr );
1988         transform_later(prefetch);
1989         distance += step_size;
1990         contended_phi_rawmem = prefetch;
1991       }
1992    } else if( AllocatePrefetchStyle > 0 ) {
1993       // Insert a prefetch for each allocation only on the fast-path
1994       Node *prefetch_adr;
1995       Node *prefetch;
1996       // Generate several prefetch instructions.
1997       uint step_size = AllocatePrefetchStepSize;
1998       uint distance = AllocatePrefetchDistance;
1999       for ( intx i = 0; i < lines; i++ ) {
2000         prefetch_adr = new AddPNode( old_eden_top, new_eden_top,
2001                                             _igvn.MakeConX(distance) );
2002         transform_later(prefetch_adr);
2003         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
2004         // Do not let it float too high, since if eden_top == eden_end,
2005         // both might be null.
2006         if( i == 0 ) { // Set control for first prefetch, next follows it
2007           prefetch->init_req(0, needgc_false);
2008         }
2009         transform_later(prefetch);
2010         distance += step_size;
2011         i_o = prefetch;
2012       }
2013    }
2014    return i_o;
2015 }

2043 //-------------------mark_eliminated_box----------------------------------
2044 //
2045 // During EA obj may point to several objects but after few ideal graph
2046 // transformations (CCP) it may point to only one non escaping object
2047 // (but still using phi), corresponding locks and unlocks will be marked
2048 // for elimination. Later obj could be replaced with a new node (new phi)
2049 // and which does not have escape information. And later after some graph
2050 // reshape other locks and unlocks (which were not marked for elimination
2051 // before) are connected to this new obj (phi) but they still will not be
2052 // marked for elimination since new obj has no escape information.
2053 // Mark all associated (same box and obj) lock and unlock nodes for
2054 // elimination if some of them marked already.
2055 void PhaseMacroExpand::mark_eliminated_box(Node* oldbox, Node* obj) {
2056   if (oldbox->as_BoxLock()->is_eliminated())
2057     return; // This BoxLock node was processed already.
2058 
2059   // New implementation (EliminateNestedLocks) has separate BoxLock
2060   // node for each locked region so mark all associated locks/unlocks as
2061   // eliminated even if different objects are referenced in one locked region
2062   // (for example, OSR compilation of nested loop inside locked scope).
2063   if (EliminateNestedLocks ||
2064       oldbox->as_BoxLock()->is_simple_lock_region(NULL, obj)) {
2065     // Box is used only in one lock region. Mark this box as eliminated.
2066     _igvn.hash_delete(oldbox);
2067     oldbox->as_BoxLock()->set_eliminated(); // This changes box's hash value
2068      _igvn.hash_insert(oldbox);
2069 
2070     for (uint i = 0; i < oldbox->outcnt(); i++) {
2071       Node* u = oldbox->raw_out(i);
2072       if (u->is_AbstractLock() && !u->as_AbstractLock()->is_non_esc_obj()) {
2073         AbstractLockNode* alock = u->as_AbstractLock();
2074         // Check lock's box since box could be referenced by Lock's debug info.
2075         if (alock->box_node() == oldbox) {
2076           // Mark eliminated all related locks and unlocks.
2077 #ifdef ASSERT

2088   // instead of oldbox for the same object.
2089   BoxLockNode* newbox = oldbox->clone()->as_BoxLock();

2092   // to indicate that all associated lock and unlock nodes are marked
2093   // for elimination.
2094   newbox->set_eliminated();

2096

2098   for (uint i = 0; i < oldbox->outcnt();) {

2102     if (u->is_AbstractLock()) {
2103       AbstractLockNode* alock = u->as_AbstractLock();

2106 #ifdef ASSERT
2107         alock->log_lock_optimization(C, "eliminate_lock_set_non_esc5");
2108 #endif
2109         alock->set_non_esc_obj();

2111         alock->set_box_node(newbox);
2112         next_edge = false;
2113       }
2114     }
2115     if (u->is_FastLock() && u->as_FastLock()->obj_node()->eqv_uncast(obj)) {
2116       FastLockNode* flock = u->as_FastLock();
2117       assert(flock->box_node() == oldbox, "sanity");
2118       _igvn.rehash_node_delayed(flock);
2119       flock->set_box_node(newbox);
2120       next_edge = false;
2121     }
2122 
2123     // Replace old box in monitor debug info.
2124     if (u->is_SafePoint() && u->as_SafePoint()->jvms()) {
2125       SafePointNode* sfn = u->as_SafePoint();
2126       JVMState* youngest_jvms = sfn->jvms();
2127       int max_depth = youngest_jvms->depth();
2128       for (int depth = 1; depth <= max_depth; depth++) {
2129         JVMState* jvms = youngest_jvms->of_depth(depth);
2130         int num_mon  = jvms->nof_monitors();
2131         // Loop over monitors
2132         for (int idx = 0; idx < num_mon; idx++) {
2133           Node* obj_node = sfn->monitor_obj(jvms, idx);
2134           Node* box_node = sfn->monitor_box(jvms, idx);
2135           if (box_node == oldbox && obj_node->eqv_uncast(obj)) {
2136             int j = jvms->monitor_box_offset(idx);
2137             _igvn.replace_input_of(u, j, newbox);
2138             next_edge = false;
2139           }
2140         }
2141       }
2142     }
2143     if (next_edge) i++;
2144   }

2301 //------------------------------expand_lock_node----------------------
2302 void PhaseMacroExpand::expand_lock_node(LockNode *lock) {
2303 
2304   Node* ctrl = lock->in(TypeFunc::Control);
2305   Node* mem = lock->in(TypeFunc::Memory);
2306   Node* obj = lock->obj_node();
2307   Node* box = lock->box_node();
2308   Node* flock = lock->fastlock_node();
2309 
2310   assert(!box->as_BoxLock()->is_eliminated(), "sanity");
2311 
2312   // Make the merge point
2313   Node *region;
2314   Node *mem_phi;
2315   Node *slow_path;
2316 
2317   if (UseOptoBiasInlining) {
2318     /*
2319      *  See the full description in MacroAssembler::biased_locking_enter().
2320      *
2321      *  if( (mark_word & biased_lock_mask) == biased_lock_pattern ) {
2322      *    // The object is biased.
2323      *    proto_node = klass->prototype_header;
2324      *    o_node = thread | proto_node;
2325      *    x_node = o_node ^ mark_word;
2326      *    if( (x_node & ~age_mask) == 0 ) { // Biased to the current thread ?
2327      *      // Done.
2328      *    } else {
2329      *      if( (x_node & biased_lock_mask) != 0 ) {
2330      *        // The klass's prototype header is no longer biased.
2331      *        cas(&mark_word, mark_word, proto_node)
2332      *        goto cas_lock;
2333      *      } else {
2334      *        // The klass's prototype header is still biased.
2335      *        if( (x_node & epoch_mask) != 0 ) { // Expired epoch?
2336      *          old = mark_word;
2337      *          new = o_node;
2338      *        } else {
2339      *          // Different thread or anonymous biased.
2340      *          old = mark_word & (epoch_mask | age_mask | biased_lock_mask);
2341      *          new = thread | old;
2342      *        }
2343      *        // Try to rebias.
2344      *        if( cas(&mark_word, old, new) == 0 ) {
2345      *          // Done.
2346      *        } else {
2347      *          goto slow_path; // Failed.
2348      *        }
2349      *      }
2350      *    }
2351      *  } else {
2352      *    // The object is not biased.
2353      *    cas_lock:
2354      *    if( FastLock(obj) == 0 ) {
2355      *      // Done.
2356      *    } else {
2357      *      slow_path:
2358      *      OptoRuntime::complete_monitor_locking_Java(obj);
2359      *    }
2360      *  }
2361      */
2362 
2363     region  = new RegionNode(5);
2364     // create a Phi for the memory state
2365     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2366 
2367     Node* fast_lock_region  = new RegionNode(3);
2368     Node* fast_lock_mem_phi = new PhiNode( fast_lock_region, Type::MEMORY, TypeRawPtr::BOTTOM);
2369 
2370     // First, check mark word for the biased lock pattern.
2371     Node* mark_node = make_load(ctrl, mem, obj, oopDesc::mark_offset_in_bytes(), TypeX_X, TypeX_X->basic_type());
2372 
2373     // Get fast path - mark word has the biased lock pattern.
2374     ctrl = opt_bits_test(ctrl, fast_lock_region, 1, mark_node,
2375                          markWord::biased_lock_mask_in_place,
2376                          markWord::biased_lock_pattern, true);
2377     // fast_lock_region->in(1) is set to slow path.
2378     fast_lock_mem_phi->init_req(1, mem);
2379 
2380     // Now check that the lock is biased to the current thread and has
2381     // the same epoch and bias as Klass::_prototype_header.
2382 
2383     // Special-case a fresh allocation to avoid building nodes:
2384     Node* klass_node = AllocateNode::Ideal_klass(obj, &_igvn);
2385     if (klass_node == NULL) {
2386       Node* k_adr = basic_plus_adr(obj, oopDesc::klass_offset_in_bytes());
2387       klass_node = transform_later(LoadKlassNode::make(_igvn, NULL, mem, k_adr, _igvn.type(k_adr)->is_ptr()));
2388 #ifdef _LP64
2389       if (UseCompressedClassPointers && klass_node->is_DecodeNKlass()) {
2390         assert(klass_node->in(1)->Opcode() == Op_LoadNKlass, "sanity");
2391         klass_node->in(1)->init_req(0, ctrl);
2392       } else
2393 #endif
2394       klass_node->init_req(0, ctrl);
2395     }
2396     Node *proto_node = make_load(ctrl, mem, klass_node, in_bytes(Klass::prototype_header_offset()), TypeX_X, TypeX_X->basic_type());
2397 
2398     Node* thread = transform_later(new ThreadLocalNode());
2399     Node* cast_thread = transform_later(new CastP2XNode(ctrl, thread));
2400     Node* o_node = transform_later(new OrXNode(cast_thread, proto_node));
2401     Node* x_node = transform_later(new XorXNode(o_node, mark_node));
2402 
2403     // Get slow path - mark word does NOT match the value.
2404     STATIC_ASSERT(markWord::age_mask_in_place <= INT_MAX);
2405     Node* not_biased_ctrl =  opt_bits_test(ctrl, region, 3, x_node,
2406                                       (~(int)markWord::age_mask_in_place), 0);
2407     // region->in(3) is set to fast path - the object is biased to the current thread.
2408     mem_phi->init_req(3, mem);
2409 
2410 
2411     // Mark word does NOT match the value (thread | Klass::_prototype_header).
2412 
2413 
2414     // First, check biased pattern.
2415     // Get fast path - _prototype_header has the same biased lock pattern.
2416     ctrl =  opt_bits_test(not_biased_ctrl, fast_lock_region, 2, x_node,
2417                           markWord::biased_lock_mask_in_place, 0, true);
2418 
2419     not_biased_ctrl = fast_lock_region->in(2); // Slow path
2420     // fast_lock_region->in(2) - the prototype header is no longer biased
2421     // and we have to revoke the bias on this object.
2422     // We are going to try to reset the mark of this object to the prototype
2423     // value and fall through to the CAS-based locking scheme.
2424     Node* adr = basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
2425     Node* cas = new StoreXConditionalNode(not_biased_ctrl, mem, adr,
2426                                           proto_node, mark_node);
2427     transform_later(cas);
2428     Node* proj = transform_later(new SCMemProjNode(cas));
2429     fast_lock_mem_phi->init_req(2, proj);
2430 
2431 
2432     // Second, check epoch bits.
2433     Node* rebiased_region  = new RegionNode(3);
2434     Node* old_phi = new PhiNode( rebiased_region, TypeX_X);
2435     Node* new_phi = new PhiNode( rebiased_region, TypeX_X);
2436 
2437     // Get slow path - mark word does NOT match epoch bits.
2438     Node* epoch_ctrl =  opt_bits_test(ctrl, rebiased_region, 1, x_node,
2439                                       markWord::epoch_mask_in_place, 0);
2440     // The epoch of the current bias is not valid, attempt to rebias the object
2441     // toward the current thread.
2442     rebiased_region->init_req(2, epoch_ctrl);
2443     old_phi->init_req(2, mark_node);
2444     new_phi->init_req(2, o_node);
2445 
2446     // rebiased_region->in(1) is set to fast path.
2447     // The epoch of the current bias is still valid but we know
2448     // nothing about the owner; it might be set or it might be clear.
2449     Node* cmask   = MakeConX(markWord::biased_lock_mask_in_place |
2450                              markWord::age_mask_in_place |
2451                              markWord::epoch_mask_in_place);
2452     Node* old = transform_later(new AndXNode(mark_node, cmask));
2453     cast_thread = transform_later(new CastP2XNode(ctrl, thread));
2454     Node* new_mark = transform_later(new OrXNode(cast_thread, old));
2455     old_phi->init_req(1, old);
2456     new_phi->init_req(1, new_mark);
2457 
2458     transform_later(rebiased_region);
2459     transform_later(old_phi);
2460     transform_later(new_phi);
2461 
2462     // Try to acquire the bias of the object using an atomic operation.
2463     // If this fails we will go in to the runtime to revoke the object's bias.
2464     cas = new StoreXConditionalNode(rebiased_region, mem, adr, new_phi, old_phi);
2465     transform_later(cas);
2466     proj = transform_later(new SCMemProjNode(cas));
2467 
2468     // Get slow path - Failed to CAS.
2469     not_biased_ctrl = opt_bits_test(rebiased_region, region, 4, cas, 0, 0);
2470     mem_phi->init_req(4, proj);
2471     // region->in(4) is set to fast path - the object is rebiased to the current thread.
2472 
2473     // Failed to CAS.
2474     slow_path  = new RegionNode(3);
2475     Node *slow_mem = new PhiNode( slow_path, Type::MEMORY, TypeRawPtr::BOTTOM);
2476 
2477     slow_path->init_req(1, not_biased_ctrl); // Capture slow-control
2478     slow_mem->init_req(1, proj);
2479 
2480     // Call CAS-based locking scheme (FastLock node).
2481 
2482     transform_later(fast_lock_region);
2483     transform_later(fast_lock_mem_phi);
2484 
2485     // Get slow path - FastLock failed to lock the object.
2486     ctrl = opt_bits_test(fast_lock_region, region, 2, flock, 0, 0);
2487     mem_phi->init_req(2, fast_lock_mem_phi);
2488     // region->in(2) is set to fast path - the object is locked to the current thread.
2489 
2490     slow_path->init_req(2, ctrl); // Capture slow-control
2491     slow_mem->init_req(2, fast_lock_mem_phi);
2492 
2493     transform_later(slow_path);
2494     transform_later(slow_mem);
2495     // Reset lock's memory edge.
2496     lock->set_req(TypeFunc::Memory, slow_mem);
2497 
2498   } else {
2499     region  = new RegionNode(3);
2500     // create a Phi for the memory state
2501     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2502 
2503     // Optimize test; set region slot 2
2504     slow_path = opt_bits_test(ctrl, region, 2, flock, 0, 0);
2505     mem_phi->init_req(2, mem);
2506   }
2507 
2508   const TypeOopPtr* objptr = _igvn.type(obj)->make_oopptr();
2509   if (objptr->can_be_inline_type()) {
2510     // Deoptimize and re-execute if a value
2511     assert(EnableValhalla, "should only be used if inline types are enabled");
2512     Node* mark = make_load(slow_path, mem, obj, oopDesc::mark_offset_in_bytes(), TypeX_X, TypeX_X->basic_type());
2513     Node* value_mask = _igvn.MakeConX(markWord::always_locked_pattern);
2514     Node* is_value = _igvn.transform(new AndXNode(mark, value_mask));
2515     Node* cmp = _igvn.transform(new CmpXNode(is_value, value_mask));
2516     Node* bol = _igvn.transform(new BoolNode(cmp, BoolTest::eq));
2517     Node* unc_ctrl = generate_slow_guard(&slow_path, bol, NULL);
2518 
2519     int trap_request = Deoptimization::make_trap_request(Deoptimization::Reason_class_check, Deoptimization::Action_none);
2520     address call_addr = SharedRuntime::uncommon_trap_blob()->entry_point();
2521     const TypePtr* no_memory_effects = NULL;
2522     JVMState* jvms = lock->jvms();
2523     CallNode* unc = new CallStaticJavaNode(OptoRuntime::uncommon_trap_Type(), call_addr, "uncommon_trap",
2524                                            jvms->bci(), no_memory_effects);
2525 
2526     unc->init_req(TypeFunc::Control, unc_ctrl);
2527     unc->init_req(TypeFunc::I_O, lock->i_o());
2528     unc->init_req(TypeFunc::Memory, mem); // may gc ptrs
2529     unc->init_req(TypeFunc::FramePtr,  lock->in(TypeFunc::FramePtr));
2530     unc->init_req(TypeFunc::ReturnAdr, lock->in(TypeFunc::ReturnAdr));
2531     unc->init_req(TypeFunc::Parms+0, _igvn.intcon(trap_request));
2532     unc->set_cnt(PROB_UNLIKELY_MAG(4));
2533     unc->copy_call_debug_info(&_igvn, lock);
2534

2536     assert(unc->peek_monitor_obj() == obj, "wrong monitor");
2537 
2538     // pop monitor and push obj back on stack: we trap before the monitorenter
2539     unc->pop_monitor();
2540     unc->grow_stack(unc->jvms(), 1);
2541     unc->set_stack(unc->jvms(), unc->jvms()->stk_size()-1, obj);
2542

2544

2551   CallNode *call = make_slow_call((CallNode *) lock, OptoRuntime::complete_monitor_enter_Type(),
2552                                   OptoRuntime::complete_monitor_locking_Java(), NULL, slow_path,
2553                                   obj, box, NULL);
2554

2558   // de-opted.  So the compiler thinks the slow-call can never throw an
2559   // exception.  If it DOES throw an exception we would need the debug
2560   // info removed first (since if it throws there is no monitor).

2562            _memproj_catchall == NULL && _catchallcatchproj == NULL, "Unexpected projection from Lock");
2563

2565   // disconnect fall-through projection from call and create a new one
2566   // hook up users of fall-through projection to region
2567   Node *slow_ctrl = _fallthroughproj->clone();
2568   transform_later(slow_ctrl);
2569   _igvn.hash_delete(_fallthroughproj);
2570   _fallthroughproj->disconnect_inputs(NULL, C);
2571   region->init_req(1, slow_ctrl);
2572   // region inputs are now complete
2573   transform_later(region);
2574   _igvn.replace_node(_fallthroughproj, region);
2575 
2576   Node *memproj = transform_later(new ProjNode(call, TypeFunc::Memory));

2893 //---------------------------eliminate_macro_nodes----------------------
2894 // Eliminate scalar replaced allocations and associated locks.
2895 void PhaseMacroExpand::eliminate_macro_nodes() {
2896   if (C->macro_count() == 0)
2897     return;
2898 
2899   // First, attempt to eliminate locks
2900   int cnt = C->macro_count();
2901   for (int i=0; i < cnt; i++) {
2902     Node *n = C->macro_node(i);
2903     if (n->is_AbstractLock()) { // Lock and Unlock nodes
2904       // Before elimination mark all associated (same box and obj)
2905       // lock and unlock nodes.
2906       mark_eliminated_locking_nodes(n->as_AbstractLock());
2907     }
2908   }
2909   bool progress = true;
2910   while (progress) {
2911     progress = false;
2912     for (int i = C->macro_count(); i > 0; i--) {
2913       Node * n = C->macro_node(i-1);
2914       bool success = false;
2915       debug_only(int old_macro_count = C->macro_count(););
2916       if (n->is_AbstractLock()) {
2917         success = eliminate_locking_node(n->as_AbstractLock());
2918       }
2919       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2920       progress = progress || success;
2921     }
2922   }
2923   // Next, attempt to eliminate allocations
2924   _has_locks = false;
2925   progress = true;
2926   while (progress) {
2927     progress = false;
2928     for (int i = C->macro_count(); i > 0; i--) {
2929       Node * n = C->macro_node(i-1);
2930       bool success = false;
2931       debug_only(int old_macro_count = C->macro_count(););
2932       switch (n->class_id()) {
2933       case Node::Class_Allocate:
2934       case Node::Class_AllocateArray:
2935         success = eliminate_allocate_node(n->as_Allocate());
2936         break;
2937       case Node::Class_CallStaticJava: {
2938         CallStaticJavaNode* call = n->as_CallStaticJava();
2939         if (!call->method()->is_method_handle_intrinsic()) {
2940           success = eliminate_boxing_node(n->as_CallStaticJava());
2941         }
2942         break;
2943       }
2944       case Node::Class_Lock:
2945       case Node::Class_Unlock:
2946         assert(!n->as_AbstractLock()->is_eliminated(), "sanity");
2947         _has_locks = true;
2948         break;
2949       case Node::Class_ArrayCopy:
2950         break;
2951       case Node::Class_OuterStripMinedLoop:
2952         break;
2953       case Node::Class_SubTypeCheck:
2954         break;
2955       default:
2956         assert(n->Opcode() == Op_LoopLimit ||
2957                n->Opcode() == Op_Opaque1   ||
2958                n->Opcode() == Op_Opaque2   ||
2959                n->Opcode() == Op_Opaque3   ||
2960                BarrierSet::barrier_set()->barrier_set_c2()->is_gc_barrier_node(n),
2961                "unknown node type in macro list");
2962       }
2963       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2964       progress = progress || success;
2965     }
2966   }
2967 }
