  78 Method* Method::allocate(ClassLoaderData* loader_data,
  79                          int byte_code_size,
  80                          AccessFlags access_flags,
  81                          InlineTableSizes* sizes,
  82                          ConstMethod::MethodType method_type,
  83                          TRAPS) {
  84   assert(!access_flags.is_native() || byte_code_size == 0,
  85          "native methods should not contain byte codes");
  86   ConstMethod* cm = ConstMethod::allocate(loader_data,
  87                                           byte_code_size,
  88                                           sizes,
  89                                           method_type,
  90                                           CHECK_NULL);
  91   int size = Method::size(access_flags.is_native());
  92   return new (loader_data, size, MetaspaceObj::MethodType, THREAD) Method(cm, access_flags);
  93 }

 182 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature) {
 183   const char* klass_name = klass->external_name();
 184   int klass_name_len  = (int)strlen(klass_name);
 185   int method_name_len = method_name->utf8_length();
 186   int len             = klass_name_len + 1 + method_name_len + signature->utf8_length();
 187   char* dest          = NEW_RESOURCE_ARRAY(char, len + 1);
 188   strcpy(dest, klass_name);
 189   dest[klass_name_len] = '.';
 190   strcpy(&dest[klass_name_len + 1], method_name->as_C_string());
 191   strcpy(&dest[klass_name_len + 1 + method_name_len], signature->as_C_string());
 192   dest[len] = 0;
 193   return dest;
 194 }

 196 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature, char* buf, int size) {

 205     len = (int)strlen(buf);
 206 
 207     signature->as_C_string(&(buf[len]), size - len);
 208   }
 209 
 210   return buf;

 234 int Method::fast_exception_handler_bci_for(const methodHandle& mh, Klass* ex_klass, int throw_bci, TRAPS) {
 235   // exception table holds quadruple entries of the form (beg_bci, end_bci, handler_bci, klass_index)
 236   // access exception table
 237   ExceptionTable table(mh());
 238   int length = table.length();
 239   // iterate through all entries sequentially
 240   constantPoolHandle pool(THREAD, mh->constants());
 241   for (int i = 0; i < length; i ++) {
 242     //reacquire the table in case a GC happened
 243     ExceptionTable table(mh());
 244     int beg_bci = table.start_pc(i);
 245     int end_bci = table.end_pc(i);
 246     assert(beg_bci <= end_bci, "inconsistent exception table");
 247     if (beg_bci <= throw_bci && throw_bci < end_bci) {
 248       // exception handler bci range covers throw_bci => investigate further
 249       int handler_bci = table.handler_pc(i);
 250       int klass_index = table.catch_type_index(i);
 251       if (klass_index == 0) {
 252         return handler_bci;
 253       } else if (ex_klass == NULL) {
 254         return handler_bci;
 255       } else {
 256         // we know the exception class => get the constraint class
 257         // this may require loading of the constraint class; if verification
 258         // fails or some other exception occurs, return handler_bci
 259         Klass* k = pool->klass_at(klass_index, CHECK_(handler_bci));
 260         assert(k != NULL, "klass not loaded");
 261         if (ex_klass->is_subtype_of(k)) {
 262           return handler_bci;
 263         }
 264       }
 265     }
 266   }
 267 
 268   return -1;
 269 }

 271 void Method::mask_for(int bci, InterpreterOopMap* mask) {
 272   methodHandle h_this(Thread::current(), this);
 273   // Only GC uses the OopMapCache during thread stack root scanning
 274   // any other uses generate an oopmap but do not save it in the cache.
 275   if (Universe::heap()->is_gc_active()) {
 276     method_holder()->mask_for(h_this, bci, mask);
 277   } else {
 278     OopMapCache::compute_one_oop_map(h_this, bci, mask);
 279   }
 280   return;
 281 }

 304 // Return bci if it appears to be a valid bcp
 305 // Return -1 otherwise.
 306 // Used by profiling code, when invalid data is a possibility.
 307 // The caller is responsible for validating the Method* itself.
 308 int Method::validate_bci_from_bcp(address bcp) const {
 309   // keep bci as -1 if not a valid bci
 310   int bci = -1;
 311   if (bcp == 0 || bcp == code_base()) {
 312     // code_size() may return 0 and we allow 0 here
 313     // the method may be native
 314     bci = 0;
 315   } else if (contains(bcp)) {
 316     bci = bcp - code_base();
 317   }
 318   // Assert that if we have dodged any asserts, bci is negative.
 319   assert(bci == -1 || bci == bci_from(bcp_from(bci)), "sane bci if >=0");
 320   return bci;
 321 }

 323 address Method::bcp_from(int bci) const {
 324   assert((is_native() && bci == 0) || (!is_native() && 0 <= bci && bci < code_size()),
 325          "illegal bci: %d for %s method", bci, is_native() ? "native" : "non-native");
 326   address bcp = code_base() + bci;
 327   assert(is_native() && bcp == code_base() || contains(bcp), "bcp doesn't belong to this method");
 328   return bcp;

 400 // The RegisterNatives call being attempted tried to register with a method that
 401 // is not native.  Ask JVM TI what prefixes have been specified.  Then check
 402 // to see if the native method is now wrapped with the prefixes.  See the
 403 // SetNativeMethodPrefix(es) functions in the JVM TI Spec for details.
 404 static Method* find_prefixed_native(Klass* k, Symbol* name, Symbol* signature, TRAPS) {
 405 #if INCLUDE_JVMTI
 406   ResourceMark rm(THREAD);
 407   Method* method;
 408   int name_len = name->utf8_length();
 409   char* name_str = name->as_utf8();
 410   int prefix_count;
 411   char** prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
 412   for (int i = 0; i < prefix_count; i++) {
 413     char* prefix = prefixes[i];
 414     int prefix_len = (int)strlen(prefix);
 415 
 416     // try adding this prefix to the method name and see if it matches another method name
 417     int trial_len = name_len + prefix_len;
 418     char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
 419     strcpy(trial_name_str, prefix);
 420     strcat(trial_name_str, name_str);
 421     TempNewSymbol trial_name = SymbolTable::probe(trial_name_str, trial_len);
 422     if (trial_name == NULL) {
 423       continue; // no such symbol, so this prefix wasn't used, try the next prefix
 424     }
 425     method = k->lookup_method(trial_name, signature);
 426     if (method == NULL) {
 427       continue; // signature doesn't match, try the next prefix
 428     }
 429     if (method->is_native()) {
 430       method->set_is_prefixed_native();
 431       return method; // wahoo, we found a prefixed version of the method, return it
 432     }
 433     // found as non-native, so prefix is good, add it, probably just need more prefixes
 434     name_len = trial_len;
 435     name_str = trial_name_str;
 436   }
 437 #endif // INCLUDE_JVMTI
 438   return NULL; // not found
 439 }

 441 bool Method::register_native(Klass* k, Symbol* name, Symbol* signature, address entry, TRAPS) {
 442   Method* method = k->lookup_method(name, signature);
 443   if (method == NULL) {
 444     ResourceMark rm(THREAD);
 445     stringStream st;
 446     st.print("Method '");
 447     print_external_name(&st, k, name, signature);
 448     st.print("' name or signature does not match");
 449     THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 450   }
 451   if (!method->is_native()) {
 452     // trying to register to a non-native method, see if a JVM TI agent has added prefix(es)
 453     method = find_prefixed_native(k, name, signature, THREAD);
 454     if (method == NULL) {
 455       ResourceMark rm(THREAD);
 456       stringStream st;
 457       st.print("Method '");
 458       print_external_name(&st, k, name, signature);
 459       st.print("' is not declared as native");
 460       THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 461     }
 462   }
 463 
 464   if (entry != NULL) {
 465     method->set_native_function(entry, native_bind_event_is_interesting);
 466   } else {
 467     method->clear_native_function();
 468   }
 469   if (log_is_enabled(Debug, jni, resolve)) {
 470     ResourceMark rm(THREAD);
 471     log_debug(jni, resolve)("[Registering JNI native method %s.%s]",
 472                             method->method_holder()->external_name(),
 473                             method->name()->as_C_string());
 474   }
 475   return true;
 476 }

 478 bool Method::was_executed_more_than(int n) {
 479   // Invocation counter is reset when the Method* is compiled.
 480   // If the method has compiled code we therefore assume it has
 481   // be excuted more than n times.
 482   if (is_accessor() || is_empty_method() || (code() != NULL)) {
 483     // interpreter doesn't bump invocation counter of trivial methods
 484     // compiler does not bump invocation counter of compiled methods
 485     return true;
 486   }
 487   else if ((method_counters() != NULL &&
 488             method_counters()->invocation_counter()->carry()) ||
 489            (method_data() != NULL &&
 490             method_data()->invocation_counter()->carry())) {
 491     // The carry bit is set when the counter overflows and causes
 492     // a compilation to occur.  We don't know how many times
 493     // the counter has been reset, so we simply assume it has
 494     // been executed more than n times.
 495     return true;
 496   } else {
 497     return invocation_count() > n;
 498   }
 499 }

 501 void Method::print_invocation_count() {
 502   if (is_static()) tty->print("static ");
 503   if (is_final()) tty->print("final ");
 504   if (is_synchronized()) tty->print("synchronized ");
 505   if (is_native()) tty->print("native ");
 506   tty->print("%s::", method_holder()->external_name());
 507   name()->print_symbol_on(tty);
 508   signature()->print_symbol_on(tty);
 509 
 510   if (WizardMode) {
 511     // dump the size of the byte codes
 512     tty->print(" {%d}", code_size());
 513   }
 514   tty->cr();
 515 
 516   tty->print_cr ("  interpreter_invocation_count: %8d ", interpreter_invocation_count());
 517   tty->print_cr ("  invocation_counter:           %8d ", invocation_count());
 518   tty->print_cr ("  backedge_counter:             %8d ", backedge_count());
 519 #ifndef PRODUCT
 520   if (CountCompiledCalls) {
 521     tty->print_cr ("  compiled_invocation_count: %8d ", compiled_invocation_count());
 522   }
 523 #endif
 524 }

 526 // Build a MethodData* object to hold information about this method
 527 // collected in the interpreter.
 528 void Method::build_interpreter_method_data(const methodHandle& method, TRAPS) {
 529   // Do not profile the method if metaspace has hit an OOM previously
 530   // allocating profiling data. Callers clear pending exception so don't
 531   // add one here.
 532   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 533     return;
 534   }
 535 
 536   // Grab a lock here to prevent multiple
 537   // MethodData*s from being created.
 538   MutexLocker ml(THREAD, MethodData_lock);
 539   if (method->method_data() == NULL) {
 540     ClassLoaderData* loader_data = method->method_holder()->class_loader_data();
 541     MethodData* method_data = MethodData::allocate(loader_data, method, THREAD);
 542     if (HAS_PENDING_EXCEPTION) {
 543       CompileBroker::log_metaspace_failure();
 544       ClassLoaderDataGraph::set_metaspace_oom(true);
 545       return;   // return the exception (which is cleared)
 546     }
 547 
 548     method->set_method_data(method_data);
 549     if (PrintMethodData && (Verbose || WizardMode)) {
 550       ResourceMark rm(THREAD);
 551       tty->print("build_interpreter_method_data for ");
 552       method->print_name(tty);
 553       tty->cr();
 554       // At the end of the run, the MDO, full of data, will be dumped.
 555     }
 556   }
 557 }

 559 MethodCounters* Method::build_method_counters(Method* m, TRAPS) {
 560   // Do not profile the method if metaspace has hit an OOM previously
 561   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 562     return NULL;
 563   }
 564 
 565   methodHandle mh(THREAD, m);
 566   MethodCounters* counters = MethodCounters::allocate(mh, THREAD);
 567   if (HAS_PENDING_EXCEPTION) {
 568     CompileBroker::log_metaspace_failure();
 569     ClassLoaderDataGraph::set_metaspace_oom(true);
 570     return NULL;   // return the exception (which is cleared)
 571   }
 572   if (!mh->init_method_counters(counters)) {
 573     MetadataFactory::free_metadata(mh->method_holder()->class_loader_data(), counters);
 574   }
 575 
 576   if (LogTouchedMethods) {
 577     mh->log_touched(CHECK_NULL);
 578   }
 579 
 580   return mh->method_counters();
 581 }

 628 bool Method::is_vanilla_constructor() const {
 629   // Returns true if this method is a vanilla constructor, i.e. an "<init>" "()V" method
 630   // which only calls the superclass vanilla constructor and possibly does stores of
 631   // zero constants to local fields:
 632   //
 633   //   aload_0, _fast_aload_0, or _nofast_aload_0
 634   //   invokespecial
 635   //   indexbyte1
 636   //   indexbyte2
 637   //
 638   // followed by an (optional) sequence of:
 639   //
 640   //   aload_0
 641   //   aconst_null / iconst_0 / fconst_0 / dconst_0
 642   //   putfield
 643   //   indexbyte1
 644   //   indexbyte2
 645   //
 646   // followed by:
 647   //
 648   //   return
 649 
 650   assert(name() == vmSymbols::object_initializer_name(),    "Should only be called for default constructors");
 651   assert(signature() == vmSymbols::void_method_signature(), "Should only be called for default constructors");
 652   int size = code_size();
 653   // Check if size match
 654   if (size == 0 || size % 5 != 0) return false;
 655   address cb = code_base();
 656   int last = size - 1;
 657   if ((cb[0] != Bytecodes::_aload_0 && cb[0] != Bytecodes::_fast_aload_0 && cb[0] != Bytecodes::_nofast_aload_0) ||
 658        cb[1] != Bytecodes::_invokespecial || cb[last] != Bytecodes::_return) {
 659     // Does not call superclass default constructor
 660     return false;
 661   }
 662   // Check optional sequence
 663   for (int i = 4; i < last; i += 5) {
 664     if (cb[i] != Bytecodes::_aload_0) return false;
 665     if (!Bytecodes::is_zero_const(Bytecodes::cast(cb[i+1]))) return false;
 666     if (cb[i+2] != Bytecodes::_putfield) return false;
 667   }
 668   return true;
 669 }

 673   BytecodeStream bcs(methodHandle(Thread::current(), this));
 674   Bytecodes::Code bc;

 676   while ((bc = bcs.next()) >= 0) {

 679       case Bytecodes::_ifnull:
 680       case Bytecodes::_iflt:
 681       case Bytecodes::_ifle:
 682       case Bytecodes::_ifne:
 683       case Bytecodes::_ifnonnull:
 684       case Bytecodes::_ifgt:
 685       case Bytecodes::_ifge:
 686       case Bytecodes::_if_icmpeq:
 687       case Bytecodes::_if_icmpne:
 688       case Bytecodes::_if_icmplt:
 689       case Bytecodes::_if_icmpgt:
 690       case Bytecodes::_if_icmple:
 691       case Bytecodes::_if_icmpge:
 692       case Bytecodes::_if_acmpeq:
 693       case Bytecodes::_if_acmpne:
 694       case Bytecodes::_goto:

 862 int Method::line_number_from_bci(int bci) const {
 863   int best_bci  =  0;
 864   int best_line = -1;
 865   if (bci == SynchronizationEntryBCI) bci = 0;
 866   if (0 <= bci && bci < code_size() && has_linenumber_table()) {
 867     // The line numbers are a short array of 2-tuples [start_pc, line_number].
 868     // Not necessarily sorted and not necessarily one-to-one.
 869     CompressedLineNumberReadStream stream(compressed_linenumber_table());
 870     while (stream.read_pair()) {
 871       if (stream.bci() == bci) {
 872         // perfect match
 873         return stream.line();
 874       } else {
 875         // update best_bci/line
 876         if (stream.bci() < bci && stream.bci() >= best_bci) {
 877           best_bci  = stream.bci();
 878           best_line = stream.line();
 879         }
 880       }
 881     }
 882   }
 883   return best_line;
 884 }

 912 void Method::set_native_function(address function, bool post_event_flag) {
 913   assert(function != NULL, "use clear_native_function to unregister natives");
 914   assert(!is_method_handle_intrinsic() || function == SharedRuntime::native_method_throw_unsatisfied_link_error_entry(), "");
 915   address* native_function = native_function_addr();
 916 
 917   // We can see racers trying to place the same native function into place. Once
 918   // is plenty.
 919   address current = *native_function;
 920   if (current == function) return;
 921   if (post_event_flag && JvmtiExport::should_post_native_method_bind() &&
 922       function != NULL) {
 923     // native_method_throw_unsatisfied_link_error_entry() should only
 924     // be passed when post_event_flag is false.
 925     assert(function !=
 926       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
 927       "post_event_flag mis-match");
 928 
 929     // post the bind event, and possible change the bind function
 930     JvmtiExport::post_native_method_bind(this, &function);
 931   }
 932   *native_function = function;
 933   // This function can be called more than once. We must make sure that we always
 934   // use the latest registered method -> check if a stub already has been generated.
 935   // If so, we have to make it not_entrant.
 936   CompiledMethod* nm = code(); // Put it into local variable to guard against concurrent updates
 937   if (nm != NULL) {
 938     nm->make_not_entrant();
 939   }
 940 }

1156 //
1157 // In this example, both Methods A and B have the _i2i_entry of "zero_locals".
1158 // They also have similar signatures so that they will share the same
1159 // AdapterHandlerEntry.
1160 //
1161 // _adapter_trampoline points to a fixed location in the RW section of
1162 // the CDS archive. This location initially contains a NULL pointer. When the
1163 // first of method A or B is linked, an AdapterHandlerEntry is allocated
1164 // dynamically, and its c2i/i2c entries are generated.
1165 //
1166 // _i2i_entry and _from_interpreted_entry initially points to the same
1167 // (fixed) location in the CODE section of the CDS archive. This contains
1168 // an unconditional branch to the actual entry for "zero_locals", which is
1169 // generated at run time and may be on an arbitrary address. Thus, the
1170 // unconditional branch is also generated at run time to jump to the correct
1171 // address.
1172 //
1173 // Similarly, _from_compiled_entry points to a fixed address in the CODE
1174 // section. This address has enough space for an unconditional branch
1175 // instruction, and is initially zero-filled. After the AdapterHandlerEntry is
1176 // initialized, and the address for the actual c2i_entry is known, we emit a
1177 // branch instruction here to branch to the actual c2i_entry.
1178 //
1179 // The effect of the extra branch on the i2i and c2i entries is negligible.
1180 //
1181 // The reason for putting _adapter_trampoline in RO is many shared Methods
1182 // share the same AdapterHandlerEntry, so we can save space in the RW section
1183 // by having the extra indirection.
1184 
1185 
1186 [Method A: RW]
1187   _constMethod ----> [ConstMethod: RO]
1188                        _adapter_trampoline -----------+
1189                                                       |
1190   _i2i_entry              (same value as method B)    |
1191   _from_interpreted_entry (same value as method B)    |
1192   _from_compiled_entry    (same value as method B)    |
1193                                                       |
1194                                                       |
1195 [Method B: RW]                               +--------+
1196   _constMethod ----> [ConstMethod: RO]       |
1197                        _adapter_trampoline --+--->(AdapterHandlerEntry* ptr: RW)-+
1198                                                                                  |
1199                                                  +-------------------------------+
1200                                                  |
1201                                                  +----> [AdapterHandlerEntry] (allocated at run time)
1202                                                               _fingerprint
1203                                                               _c2i_entry ---------------------------------+->[c2i entry..]
1204  _i2i_entry  -------------+                                   _i2c_entry ---------------+-> [i2c entry..] |
1205  _from_interpreted_entry  |                                   _c2i_unverified_entry     |                 |
1206          |                |                                   _c2i_no_clinit_check_entry|                 |
1207          |                |  (_cds_entry_table: CODE)                                   |                 |
1208          |                +->[0]: jmp _entry_table[0] --> (i2i_entry_for "zero_locals") |                 |
1209          |                |                               (allocated at run time)       |                 |
1210          |                |  ...                           [asm code ...]               |                 |
1211          +-[not compiled]-+  [n]: jmp _entry_table[n]                                   |                 |
1212          |                                                                              |                 |
1213          |                                                                              |                 |
1214          +-[compiled]-------------------------------------------------------------------+                 |
1215                                                                                                           |
1216  _from_compiled_entry------------>  (_c2i_entry_trampoline: CODE)                                         |
1217                                     [jmp c2i_entry] ------------------------------------------------------+
1218 
1219 ***/

1221 // Called when the method_holder is getting linked. Setup entrypoints so the method
1222 // is ready to be called from interpreter, compiler, and vtables.
1223 void Method::link_method(const methodHandle& h_method, TRAPS) {
1224   // If the code cache is full, we may reenter this function for the
1225   // leftover methods that weren't linked.
1226   if (is_shared()) {
1227     // Can't assert that the adapters are sane, because methods get linked before
1228     // the interpreter is generated, and hence before its adapters are generated.
1229     // If you messed them up you will notice soon enough though, don't you worry.
1230     if (adapter() != NULL) {
1231       return;
1232     }
1233   } else if (_i2i_entry != NULL) {
1234     return;
1235   }
1236   assert( _code == NULL, "nothing compiled yet" );
1237 
1238   // Setup interpreter entrypoint
1239   assert(this == h_method(), "wrong h_method()" );
1240 
1241   if (!is_shared()) {
1242     assert(adapter() == NULL, "init'd to NULL");
1243     address entry = Interpreter::entry_for_method(h_method);
1244     assert(entry != NULL, "interpreter entry must be non-null");
1245     // Sets both _i2i_entry and _from_interpreted_entry
1246     set_interpreter_entry(entry);
1247   }
1248 
1249   // Don't overwrite already registered native entries.
1250   if (is_native() && !has_native_function()) {
1251     set_native_function(
1252       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
1253       !native_bind_event_is_interesting);
1254   }
1255 
1256   // Setup compiler entrypoint.  This is made eagerly, so we do not need
1257   // special handling of vtables.  An alternative is to make adapters more
1258   // lazily by calling make_adapter() from from_compiled_entry() for the
1259   // normal calls.  For vtable calls life gets more complicated.  When a
1260   // call-site goes mega-morphic we need adapters in all methods which can be
1261   // called from the vtable.  We need adapters on such methods that get loaded
1262   // later.  Ditto for mega-morphic itable calls.  If this proves to be a
1263   // problem we'll make these lazily later.
1264   (void) make_adapters(h_method, CHECK);
1265 
1266   // ONLY USE the h_method now as make_adapter may have blocked
1267 
1268 }

1270 address Method::make_adapters(const methodHandle& mh, TRAPS) {
1271   // Adapters for compiled code are made eagerly here.  They are fairly
1272   // small (generally < 100 bytes) and quick to make (and cached and shared)
1273   // so making them eagerly shouldn't be too expensive.
1274   AdapterHandlerEntry* adapter = AdapterHandlerLibrary::get_adapter(mh);
1275   if (adapter == NULL ) {
1276     if (!is_init_completed()) {
1277       // Don't throw exceptions during VM initialization because java.lang.* classes
1278       // might not have been initialized, causing problems when constructing the
1279       // Java exception object.
1280       vm_exit_during_initialization("Out of space in CodeCache for adapters");
1281     } else {
1282       THROW_MSG_NULL(vmSymbols::java_lang_VirtualMachineError(), "Out of space in CodeCache for adapters");
1283     }
1284   }
1285 
1286   if (mh->is_shared()) {
1287     assert(mh->adapter() == adapter, "must be");
1288     assert(mh->_from_compiled_entry != NULL, "must be");
1289     assert(mh->_from_compiled_inline_entry != NULL, "must be");
1290     assert(mh->_from_compiled_inline_ro_entry != NULL, "must be");
1291   } else {
1292     mh->set_adapter_entry(adapter);
1293     mh->_from_compiled_entry = adapter->get_c2i_entry();
1294     mh->_from_compiled_inline_entry = adapter->get_c2i_inline_entry();
1295     mh->_from_compiled_inline_ro_entry = adapter->get_c2i_inline_ro_entry();
1296   }
1297   return adapter->get_c2i_entry();
1298 }

1377 // Install compiled code.  Instantly it can execute.
1378 void Method::set_code(const methodHandle& mh, CompiledMethod *code) {
1379   assert_lock_strong(CompiledMethod_lock);
1380   assert( code, "use clear_code to remove code" );
1381   assert( mh->check_code(), "" );
1382 
1383   guarantee(mh->adapter() != NULL, "Adapter blob must already exist!");
1384 
1385   // These writes must happen in this order, because the interpreter will
1386   // directly jump to from_interpreted_entry which jumps to an i2c adapter
1387   // which jumps to _from_compiled_entry.
1388   mh->_code = code;             // Assign before allowing compiled code to exec
1389 
1390   int comp_level = code->comp_level();
1391   // In theory there could be a race here. In practice it is unlikely
1392   // and not worth worrying about.
1393   if (comp_level > mh->highest_comp_level()) {
1394     mh->set_highest_comp_level(comp_level);
1395   }
1396 
1397   OrderAccess::storestore();
1398   mh->_from_compiled_entry = code->verified_entry_point();
1399   mh->_from_compiled_inline_entry = code->verified_inline_entry_point();
1400   mh->_from_compiled_inline_ro_entry = code->verified_inline_ro_entry_point();
1401   OrderAccess::storestore();
1402   // Instantly compiled code can execute.
1403   if (!mh->is_method_handle_intrinsic())
1404     mh->_from_interpreted_entry = mh->get_i2c_entry();
1405 }

1496 // Make an instance of a signature-polymorphic internal MH primitive.
1497 methodHandle Method::make_method_handle_intrinsic(vmIntrinsics::ID iid,
1498                                                          Symbol* signature,
1499                                                          TRAPS) {
1500   ResourceMark rm(THREAD);
1501   methodHandle empty;
1502 
1503   InstanceKlass* holder = SystemDictionary::MethodHandle_klass();
1504   Symbol* name = MethodHandles::signature_polymorphic_intrinsic_name(iid);
1505   assert(iid == MethodHandles::signature_polymorphic_name_id(name), "");
1506 
1507   log_info(methodhandles)("make_method_handle_intrinsic MH.%s%s", name->as_C_string(), signature->as_C_string());
1508 
1509   // invariant:   cp->symbol_at_put is preceded by a refcount increment (more usually a lookup)
1510   name->increment_refcount();
1511   signature->increment_refcount();
1512 
1513   int cp_length = _imcp_limit;
1514   ClassLoaderData* loader_data = holder->class_loader_data();
1515   constantPoolHandle cp;
1516   {
1517     ConstantPool* cp_oop = ConstantPool::allocate(loader_data, cp_length, CHECK_(empty));
1518     cp = constantPoolHandle(THREAD, cp_oop);
1519   }
1520   cp->copy_fields(holder->constants());
1521   cp->set_pool_holder(holder);
1522   cp->symbol_at_put(_imcp_invoke_name,       name);
1523   cp->symbol_at_put(_imcp_invoke_signature,  signature);
1524   cp->set_has_preresolution();
1525 
1526   // decide on access bits:  public or not?
1527   int flags_bits = (JVM_ACC_NATIVE | JVM_ACC_SYNTHETIC | JVM_ACC_FINAL);
1528   bool must_be_static = MethodHandles::is_signature_polymorphic_static(iid);
1529   if (must_be_static)  flags_bits |= JVM_ACC_STATIC;
1530   assert((flags_bits & JVM_ACC_PUBLIC) == 0, "do not expose these methods");
1531 
1532   methodHandle m;
1533   {
1534     InlineTableSizes sizes;
1535     Method* m_oop = Method::allocate(loader_data, 0,
1536                                      accessFlags_from(flags_bits), &sizes,
1537                                      ConstMethod::NORMAL, CHECK_(empty));
1538     m = methodHandle(THREAD, m_oop);
1539   }
1540   m->set_constants(cp());
1541   m->set_name_index(_imcp_invoke_name);
1542   m->set_signature_index(_imcp_invoke_signature);
1543   assert(MethodHandles::is_signature_polymorphic_name(m->name()), "");
1544   assert(m->signature() == signature, "");
1545   m->compute_from_signature(signature);
1546   m->init_intrinsic_id();
1547   assert(m->is_method_handle_intrinsic(), "");
1548 #ifdef ASSERT
1549   if (!MethodHandles::is_signature_polymorphic(m->intrinsic_id()))  m->print();
1550   assert(MethodHandles::is_signature_polymorphic(m->intrinsic_id()), "must be an invoker");
1551   assert(m->intrinsic_id() == iid, "correctly predicted iid");
1552 #endif //ASSERT
1553 
1554   // Finally, set up its entry points.
1555   assert(m->can_be_statically_bound(), "");
1556   m->set_vtable_index(Method::nonvirtual_vtable_index);
1557   m->link_method(m, CHECK_(empty));
1558 
1559   if (log_is_enabled(Info, methodhandles) && (Verbose || WizardMode)) {
1560     LogTarget(Info, methodhandles) lt;
1561     LogStream ls(lt);
1562     m->print_on(&ls);
1563   }
1564 
1565   return m;
1566 }

1579                                                 u_char* new_compressed_linenumber_table, int new_compressed_linenumber_size, TRAPS) {

1596       localvariable_len,
1597       new_compressed_linenumber_size,
1598       exception_table_len,
1599       checked_exceptions_len,
1600       method_parameters_len,
1601       cm->generic_signature_index(),
1602       method_annotations_len,
1603       parameter_annotations_len,
1604       type_annotations_len,
1605       default_annotations_len,
1606       0);
1607 
1608   ClassLoaderData* loader_data = m->method_holder()->class_loader_data();
1609   Method* newm_oop = Method::allocate(loader_data,
1610                                       new_code_length,
1611                                       flags,
1612                                       &sizes,
1613                                       m->method_type(),
1614                                       CHECK_(methodHandle()));
1615   methodHandle newm (THREAD, newm_oop);
1616 
1617   // Create a shallow copy of Method part, but be careful to preserve the new ConstMethod*

1630   newm->set_constMethod(newcm);

1641   if (new_compressed_linenumber_size > 0) {
1642     memcpy(newm->compressed_linenumber_table(),
1643            new_compressed_linenumber_table,
1644            new_compressed_linenumber_size);
1645   }
1646   // Copy method_parameters
1647   if (method_parameters_len > 0) {
1648     memcpy(newm->method_parameters_start(),
1649            m->method_parameters_start(),
1650            method_parameters_len * sizeof(MethodParametersElement));
1651   }
1652   // Copy checked_exceptions
1653   if (checked_exceptions_len > 0) {
1654     memcpy(newm->checked_exceptions_start(),
1655            m->checked_exceptions_start(),
1656            checked_exceptions_len * sizeof(CheckedExceptionElement));
1657   }
1658   // Copy exception table
1659   if (exception_table_len > 0) {
1660     memcpy(newm->exception_table_start(),
1661            m->exception_table_start(),
1662            exception_table_len * sizeof(ExceptionTableElement));
1663   }
1664   // Copy local variable number table
1665   if (localvariable_len > 0) {
1666     memcpy(newm->localvariable_table_start(),
1667            m->localvariable_table_start(),
1668            localvariable_len * sizeof(LocalVariableTableElement));
1669   }
1670   // Copy stackmap table
1671   if (m->has_stackmap_table()) {
1672     int code_attribute_length = m->stackmap_data()->length();
1673     Array<u1>* stackmap_data =
1674       MetadataFactory::new_array<u1>(loader_data, code_attribute_length, 0, CHECK_(methodHandle()));
1675     memcpy((void*)stackmap_data->adr_at(0),
1676            (void*)m->stackmap_data()->adr_at(0), code_attribute_length);
1677     newm->set_stackmap_data(stackmap_data);
1678   }
1679 
1680   // copy annotations over to new method
1681   newcm->copy_annotations_from(loader_data, cm, CHECK_(methodHandle()));
1682   return newm;
1683 }

1700 void Method::init_intrinsic_id() {

1702   const uintptr_t max_id_uint = right_n_bits((int)(sizeof(_intrinsic_id) * BitsPerByte));
1703   assert((uintptr_t)vmIntrinsics::ID_LIMIT <= max_id_uint, "else fix size");
1704   assert(intrinsic_id_size_in_bytes() == sizeof(_intrinsic_id), "");
1705 
1706   // the klass name is well-known:
1707   vmSymbols::SID klass_id = klass_id_for_intrinsics(method_holder());
1708   assert(klass_id != vmSymbols::NO_SID, "caller responsibility");
1709 
1710   // ditto for method and signature:
1711   vmSymbols::SID  name_id = vmSymbols::find_sid(name());
1712   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1713       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1714       && name_id == vmSymbols::NO_SID) {
1715     return;
1716   }
1717   vmSymbols::SID   sig_id = vmSymbols::find_sid(signature());
1718   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1719       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1720       && sig_id == vmSymbols::NO_SID) {
1721     return;
1722   }
1723   jshort flags = access_flags().as_short();
1724 
1725   vmIntrinsics::ID id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1726   if (id != vmIntrinsics::_none) {
1727     set_intrinsic_id(id);
1728     if (id == vmIntrinsics::_Class_cast) {
1729       // Even if the intrinsic is rejected, we want to inline this simple method.
1730       set_force_inline(true);
1731     }
1732     return;
1733   }
1734 
1735   // A few slightly irregular cases:
1736   switch (klass_id) {
1737   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_StrictMath):
1738     // Second chance: check in regular Math.
1739     switch (name_id) {
1740     case vmSymbols::VM_SYMBOL_ENUM_NAME(min_name):
1741     case vmSymbols::VM_SYMBOL_ENUM_NAME(max_name):
1742     case vmSymbols::VM_SYMBOL_ENUM_NAME(sqrt_name):
1743       // pretend it is the corresponding method in the non-strict class:
1744       klass_id = vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_Math);
1745       id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1746       break;
1747     default:
1748       break;
1749     }
1750     break;
1751 
1752   // Signature-polymorphic methods: MethodHandle.invoke*, InvokeDynamic.*., VarHandle
1753   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle):
1754   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle):
1755     if (!is_native())  break;
1756     id = MethodHandles::signature_polymorphic_name_id(method_holder(), name());
1757     if (is_static() != MethodHandles::is_signature_polymorphic_static(id))
1758       id = vmIntrinsics::_none;
1759     break;
1760 
1761   default:
1762     break;
1763   }
1764 
1765   if (id != vmIntrinsics::_none) {
1766     // Set up its iid.  It is an alias method.
1767     set_intrinsic_id(id);
1768     return;
1769   }
1770 }

1976 static void clear_matches(Method* m, int bci) {
1977   InstanceKlass* ik = m->method_holder();
1978   BreakpointInfo* prev_bp = NULL;
1979   BreakpointInfo* next_bp;
1980   for (BreakpointInfo* bp = ik->breakpoints(); bp != NULL; bp = next_bp) {
1981     next_bp = bp->next();
1982     // bci value of -1 is used to delete all breakpoints in method m (ex: clear_all_breakpoint).
1983     if (bci >= 0 ? bp->match(m, bci) : bp->match(m)) {
1984       // do this first:
1985       bp->clear(m);
1986       // unhook it
1987       if (prev_bp != NULL)
1988         prev_bp->set_next(next_bp);
1989       else
1990         ik->set_breakpoints(next_bp);
1991       delete bp;
1992       // When class is redefined JVMTI sets breakpoint in all versions of EMCP methods
1993       // at same location. So we have multiple matching (method_index and bci)
1994       // BreakpointInfo nodes in BreakpointInfo list. We should just delete one
1995       // breakpoint for clear_breakpoint request and keep all other method versions
1996       // BreakpointInfo for future clear_breakpoint request.
1997       // bcivalue of -1 is used to clear all breakpoints (see clear_all_breakpoints)
1998       // which is being called when class is unloaded. We delete all the Breakpoint
1999       // information for all versions of method. We may not correctly restore the original
2000       // bytecode in all method versions, but that is ok. Because the class is being unloaded
2001       // so these methods won't be used anymore.
2002       if (bci >= 0) {
2003         break;
2004       }
2005     } else {
2006       // This one is a keeper.
2007       prev_bp = bp;
2008     }
2009   }
2010 }

2136 class JNIMethodBlockNode : public CHeapObj<mtClass> {
2137   friend class JNIMethodBlock;

2150     if (_top < _number_of_methods) {
2151       num_addl_methods -= _number_of_methods - _top;
2152       if (num_addl_methods <= 0) {
2153         return;
2154       }
2155     }
2156     if (_next == NULL) {
2157       _next = new JNIMethodBlockNode(MAX2(num_addl_methods, min_block_size));

2164 class JNIMethodBlock : public CHeapObj<mtClass> {
2165   JNIMethodBlockNode _head;
2166   JNIMethodBlockNode *_last_free;
2167  public:
2168   static Method* const _free_method;
2169 
2170   JNIMethodBlock(int initial_capacity = min_block_size)
2171       : _head(initial_capacity), _last_free(&_head) {}
2172 
2173   void ensure_methods(int num_addl_methods) {
2174     _last_free->ensure_methods(num_addl_methods);
2175   }
2176 
2177   Method** add_method(Method* m) {
2178     for (JNIMethodBlockNode* b = _last_free; b != NULL; b = b->_next) {
2179       if (b->_top < b->_number_of_methods) {
2180         // top points to the next free entry.
2181         int i = b->_top;
2182         b->_methods[i] = m;
2183         b->_top++;
2184         _last_free = b;
2185         return &(b->_methods[i]);
2186       } else if (b->_top == b->_number_of_methods) {
2187         // if the next free entry ran off the block see if there's a free entry
2188         for (int i = 0; i < b->_number_of_methods; i++) {
2189           if (b->_methods[i] == _free_method) {
2190             b->_methods[i] = m;
2191             _last_free = b;
2192             return &(b->_methods[i]);
2193           }
2194         }
2195         // Only check each block once for frees.  They're very unlikely.
2196         // Increment top past the end of the block.
2197         b->_top++;
2198       }
2199       // need to allocate a next block.
2200       if (b->_next == NULL) {
2201         b->_next = _last_free = new JNIMethodBlockNode();
2202       }
2203     }
2204     guarantee(false, "Should always allocate a free block");
2205     return NULL;
2206   }
2207 
2208   bool contains(Method** m) {
2209     if (m == NULL) return false;
2210     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2211       if (b->_methods <= m && m < b->_methods + b->_number_of_methods) {
2212         // This is a bit of extra checking, for two reasons.  One is
2213         // that contains() deals with pointers that are passed in by
2214         // JNI code, so making sure that the pointer is aligned
2215         // correctly is valuable.  The other is that <= and > are
2216         // technically not defined on pointers, so the if guard can
2217         // pass spuriously; no modern compiler is likely to make that
2218         // a problem, though (and if one did, the guard could also
2219         // fail spuriously, which would be bad).
2220         ptrdiff_t idx = m - b->_methods;
2221         if (b->_methods + idx == m) {
2222           return true;
2223         }
2224       }
2225     }
2226     return false;  // not found
2227   }

2229   // Doesn't really destroy it, just marks it as free so it can be reused.
2230   void destroy_method(Method** m) {

2240     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2241       for (int i = 0; i< b->_number_of_methods; i++) {
2242         b->_methods[i] = NULL;
2243       }
2244     }
2245   }

2524 // TouchedMethodRecord -- we can't use a HashtableEntry<Method*> because
2525 // the Method may be garbage collected. Let's roll our own hash table.
2526 class TouchedMethodRecord : CHeapObj<mtTracing> {
2527 public:
2528   // It's OK to store Symbols here because they will NOT be GC'ed if
2529   // LogTouchedMethods is enabled.
2530   TouchedMethodRecord* _next;
2531   Symbol* _class_name;
2532   Symbol* _method_name;
2533   Symbol* _method_signature;
2534 };
2535 
2536 static const int TOUCHED_METHOD_TABLE_SIZE = 20011;
2537 static TouchedMethodRecord** _touched_method_table = NULL;
2538 
2539 void Method::log_touched(TRAPS) {
2540 
2541   const int table_size = TOUCHED_METHOD_TABLE_SIZE;
2542   Symbol* my_class = klass_name();
2543   Symbol* my_name  = name();
2544   Symbol* my_sig   = signature();
2545 
2546   unsigned int hash = my_class->identity_hash() +
2547                       my_name->identity_hash() +
2548                       my_sig->identity_hash();
2549   juint index = juint(hash) % table_size;
2550 
2551   MutexLocker ml(THREAD, TouchedMethodLog_lock);
2552   if (_touched_method_table == NULL) {
2553     _touched_method_table = NEW_C_HEAP_ARRAY2(TouchedMethodRecord*, table_size,
2554                                               mtTracing, CURRENT_PC);
2555     memset(_touched_method_table, 0, sizeof(TouchedMethodRecord*)*table_size);
2556   }
2557 
2558   TouchedMethodRecord* ptr = _touched_method_table[index];
2559   while (ptr) {
2560     if (ptr->_class_name       == my_class &&
2561         ptr->_method_name      == my_name &&
2562         ptr->_method_signature == my_sig) {
2563       return;
2564     }
2565     if (ptr->_next == NULL) break;
2566     ptr = ptr->_next;
2567   }
2568   TouchedMethodRecord* nptr = NEW_C_HEAP_OBJ(TouchedMethodRecord, mtTracing);
2569   my_class->increment_refcount();
2570   my_name->increment_refcount();
2571   my_sig->increment_refcount();
2572   nptr->_class_name         = my_class;
2573   nptr->_method_name        = my_name;
2574   nptr->_method_signature   = my_sig;
2575   nptr->_next               = NULL;
2576 
2577   if (ptr == NULL) {
2578     // first
2579     _touched_method_table[index] = nptr;
2580   } else {
2581     ptr->_next = nptr;
2582   }
2583 }

2586   MutexLocker ml(Thread::current()->is_VM_thread() ? NULL : TouchedMethodLog_lock);
2587   out->print_cr("# Method::print_touched_methods version 1");
2588   if (_touched_method_table) {
