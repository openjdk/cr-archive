  89     private static Class<? extends Annotation> findPolymorphicSignatureClass() {
  90         Class<? extends Annotation> signaturePolyAnnotation = null;
  91         try {
  92             for (Class<?> clazz : TestResolvedJavaType.class.getClassLoader().loadClass("java.lang.invoke.MethodHandle").getDeclaredClasses()) {
  93                 if (clazz.getName().endsWith("PolymorphicSignature") && Annotation.class.isAssignableFrom(clazz)) {
  94                     signaturePolyAnnotation = (Class<? extends Annotation>) clazz;
  95                     break;
  96                 }
  97             }
  98         } catch (Throwable e) {
  99             throw new AssertionError("Could not find annotation PolymorphicSignature in java.lang.invoke.MethodHandle", e);
 100         }
 101         assertNotNull(signaturePolyAnnotation);
 102         return signaturePolyAnnotation;
 103     }

 285     public Class<?> getSupertype(Class<?> c) {
 286         assert !c.isPrimitive();
 287         if (c.isArray()) {
 288             Class<?> componentType = c.getComponentType();
 289             if (componentType.isPrimitive() || componentType == Object.class) {
 290                 return Object.class;
 291             }
 292             return getArrayClass(getSupertype(componentType));
 293         }
 294         if (c.isInterface()) {
 295             return Object.class;
 296         }
 297         return c.getSuperclass();
 298     }

 300     public Class<?> findLeastCommonAncestor(Class<?> c1Initial, Class<?> c2Initial) {
 301         if (c1Initial.isPrimitive() || c2Initial.isPrimitive()) {
 302             return null;
 303         } else {
 304             Class<?> c1 = c1Initial;
 305             Class<?> c2 = c2Initial;
 306             while (true) {
 307                 if (c1.isAssignableFrom(c2)) {
 308                     return c1;
 309                 }
 310                 if (c2.isAssignableFrom(c1)) {
 311                     return c2;
 312                 }
 313                 c1 = getSupertype(c1);
 314                 c2 = getSupertype(c2);
 315             }
 316         }
 317     }

 383                 // findLeafConcreteSubtype() method is conservative

 574     /**
 575      * See <a href="http://docs.oracle.com/javase/specs/jvms/se7/html/jvms-5.html#jvms-5.4.5">Method
 576      * overriding</a>.
 577      */
 578     static boolean isOverriderOf(Method impl, Method m) {
 579         if (!isPrivate(m.getModifiers()) && !isFinal(m.getModifiers())) {
 580             if (m.getName().equals(impl.getName())) {
 581                 if (m.getReturnType() == impl.getReturnType()) {
 582                     if (Arrays.equals(m.getParameterTypes(), impl.getParameterTypes())) {
 583                         if (isPublic(m.getModifiers()) || isProtected(m.getModifiers())) {
 584                             // m is public or protected
 585                             return isPublic(impl.getModifiers()) || isProtected(impl.getModifiers());
 586                         } else {
 587                             // m is package-private
 588                             return impl.getDeclaringClass().getPackage() == m.getDeclaringClass().getPackage();
 589                         }
 590                     }
 591                 }
 592             }
 593         }
 594         return false;
 595     }

 597     static final Map<Class<?>, VTable> vtables = new HashMap<>();
 598 
 599     static class VTable {
 600 
 601         final Map<NameAndSignature, Method> methods = new HashMap<>();
 602     }
 603 
 604     static synchronized VTable getVTable(Class<?> c) {
 605         VTable vtable = vtables.get(c);
 606         if (vtable == null) {
 607             vtable = new VTable();
 608             if (c != Object.class) {
 609                 VTable superVtable = getVTable(c.getSuperclass());
 610                 vtable.methods.putAll(superVtable.methods);
 611             }
 612             for (Method m : c.getDeclaredMethods()) {
 613                 if (!isStatic(m.getModifiers()) && !isPrivate(m.getModifiers())) {
 614                     if (isAbstract(m.getModifiers())) {
 615                         // A subclass makes a concrete method in a superclass abstract
 616                         vtable.methods.remove(new NameAndSignature(m));
 617                     } else {
 618                         vtable.methods.put(new NameAndSignature(m), m);
 619                     }
 620                 }
 621             }
 622             vtables.put(c, vtable);
 623         }
 624         return vtable;
 625     }

 627     static Set<Method> findDeclarations(Method impl, Class<?> c) {
 628         Set<Method> declarations = new HashSet<>();
 629         NameAndSignature implSig = new NameAndSignature(impl);
 630         if (c != null) {
 631             for (Method m : c.getDeclaredMethods()) {
 632                 if (new NameAndSignature(m).equals(implSig)) {
 633                     declarations.add(m);
 634                     break;
 635                 }
 636             }
 637             if (!c.isInterface()) {
 638                 declarations.addAll(findDeclarations(impl, c.getSuperclass()));
 639             }
 640             for (Class<?> i : c.getInterfaces()) {
 641                 declarations.addAll(findDeclarations(impl, i));
 642             }
 643         }
 644         return declarations;
 645     }

 729     public static Set<Field> getInstanceFields(Class<?> c, boolean includeSuperclasses) {
 730         if (c.isArray() || c.isPrimitive() || c.isInterface()) {
 731             return Collections.emptySet();
 732         }
 733         Set<Field> result = new HashSet<>();
 734         for (Field f : c.getDeclaredFields()) {
 735             if (!Modifier.isStatic(f.getModifiers())) {
 736                 result.add(f);
 737             }
 738         }
 739         if (includeSuperclasses && c != Object.class) {
 740             result.addAll(getInstanceFields(c.getSuperclass(), true));
 741         }
 742         return result;
 743     }

 778     private static boolean isHiddenFromReflection(ResolvedJavaField f) {
 779         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Throwable.class)) && f.getName().equals("backtrace")) {
 780             return true;
 781         }
 782         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ConstantPool.class)) && f.getName().equals("constantPoolOop")) {
 783             return true;
 784         }
 785         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Class.class))) {
 786             return f.getName().equals("classLoader") || f.getName().equals("classData");
 787         }
 788         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(Lookup.class))) {
 789             return f.getName().equals("allowedModes") || f.getName().equals("lookupClass");
 790         }
 791         if (f.getDeclaringClass().equals(metaAccess.lookupJavaType(ClassLoader.class)) ||
 792             f.getDeclaringClass().equals(metaAccess.lookupJavaType(AccessibleObject.class)) ||
 793             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Constructor.class)) ||
 794             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Field.class)) ||
 795             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Method.class)) ||
 796             f.getDeclaringClass().equals(metaAccess.lookupJavaType(Module.class))) {
 797             return true;
 798         }
 799         return false;
 800     }
