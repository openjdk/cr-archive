26 package jdk.experimental.bytecode;
27 
28 import java.util.EnumSet;
29 
30 public enum Flag {
31     ACC_PUBLIC(0x0001),
32     ACC_PROTECTED(0x0004),
33     ACC_PRIVATE(0x0002),
34     ACC_INTERFACE(0x0200),
35     ACC_ENUM(0x4000),
36     ACC_ANNOTATION(0x2000),
37     ACC_SUPER(0x0020),
38     ACC_ABSTRACT(0x0400),
39     ACC_VOLATILE(0x0040),
40     ACC_TRANSIENT(0x0080),
41     ACC_SYNTHETIC(0x1000),
42     ACC_STATIC(0x0008),
43     ACC_FINAL(0x0010),
44     ACC_SYNCHRONIZED(0x0020),
45     ACC_BRIDGE(0x0040),
46     ACC_VARARGS(0x0080),
47     ACC_VALUE(0x0100),
48     ACC_NATIVE(0x0100),
49     ACC_STRICT(0x0800);
50 
51     public int flag;
52 
53     Flag(int flag) {
54         this.flag = flag;
55     }
56 
57     static Flag[] parse(int flagsMask) {
58         EnumSet<Flag> flags = EnumSet.noneOf(Flag.class);
59         for (Flag f : Flag.values()) {
60             if ((f.flag & flagsMask) != 0) {
61                 flags.add(f);
62             }
63         }
64         return flags.stream().toArray(Flag[]::new);
65     }
66 }
