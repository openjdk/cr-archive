26 /*
27  * @test
28  * @summary Check value flag in class file
29  * @modules jdk.jdeps/com.sun.tools.classfile
30  * @compile -XDallowWithFieldOperator Point.java
31  * @run main CheckFlags
32  */
33 
34 import com.sun.tools.classfile.*;
35 
36 public class CheckFlags {
37     public static void main(String[] args) throws Exception {
38         ClassFile cls = ClassFile.read(CheckFlags.class.getResourceAsStream("Point.class"));
39 
40         if (!cls.access_flags.is(AccessFlags.ACC_INLINE))
41             throw new Exception("Value flag not set");
42 
43         if (!cls.access_flags.is(AccessFlags.ACC_FINAL))
44             throw new Exception("Final flag not set");
45 
46         Field [] flds = cls.fields;
47 
48         for (Field fld : flds) {
49             if (!fld.access_flags.is(AccessFlags.ACC_FINAL))
50                 throw new Exception("Final flag not set");
51         }
52     }
53 }
