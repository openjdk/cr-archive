  63 //-----------------------------ParseGenerator---------------------------------
  64 // Internal class which handles all direct bytecode traversal.
  65 class ParseGenerator : public InlineCallGenerator {
  66 private:
  67   bool  _is_osr;
  68   float _expected_uses;
  69 
  70 public:
  71   ParseGenerator(ciMethod* method, float expected_uses, bool is_osr = false)
  72     : InlineCallGenerator(method)
  73   {
  74     _is_osr        = is_osr;
  75     _expected_uses = expected_uses;
  76     assert(InlineTree::check_can_parse(method) == NULL, "parse must be possible");
  77   }
  78 
  79   virtual bool      is_parse() const           { return true; }
  80   virtual JVMState* generate(JVMState* jvms);
  81   int is_osr() { return _is_osr; }
  82 
  83 };

  85 JVMState* ParseGenerator::generate(JVMState* jvms) {
  86   Compile* C = Compile::current();
  87   C->print_inlining_update(this);
  88 
  89   if (is_osr()) {
  90     // The JVMS for a OSR has a single argument (see its TypeFunc).
  91     assert(jvms->depth() == 1, "no inline OSR");
  92   }
  93 
  94   if (C->failing()) {
  95     return NULL;  // bailing out of the compile; do not try to parse
  96   }
  97 
  98   Parse parser(jvms, method(), _expected_uses);
  99   // Grab signature for matching/allocation
 100   GraphKit& exits = parser.exits();
 101 
 102   if (C->failing()) {
 103     while (exits.pop_exception_state() != NULL) ;
 104     return NULL;
 105   }
 106 
 107   assert(exits.jvms()->same_calls_as(jvms), "sanity");
 108 
 109   // Simply return the exit state of the parser,
 110   // augmented by any exceptional states.
 111   return exits.transfer_exceptions_into_jvms();
 112 }

 114 //---------------------------DirectCallGenerator------------------------------
 115 // Internal class which handles all out-of-line calls w/o receiver type checks.
 116 class DirectCallGenerator : public CallGenerator {
 117  private:
 118   CallStaticJavaNode* _call_node;
 119   // Force separate memory and I/O projections for the exceptional
 120   // paths to facilitate late inlining.
 121   bool                _separate_io_proj;
 122 
 123  public:
 124   DirectCallGenerator(ciMethod* method, bool separate_io_proj)
 125     : CallGenerator(method),
 126       _call_node(NULL),
 127       _separate_io_proj(separate_io_proj)
 128   {
 129     if (InlineTypeReturnedAsFields && method->is_method_handle_intrinsic()) {
 130       // If that call has not been optimized by the time optimizations are over,
 131       // we'll need to add a call to create an inline type instance from the klass
 132       // returned by the call (see PhaseMacroExpand::expand_mh_intrinsic_return).
 133       // Separating memory and I/O projections for exceptions is required to
 134       // perform that graph transformation.
 135       _separate_io_proj = true;
 136     }
 137   }
 138   virtual JVMState* generate(JVMState* jvms);
 139 
 140   CallStaticJavaNode* call_node() const { return _call_node; }
 141 };
 142 
 143 JVMState* DirectCallGenerator::generate(JVMState* jvms) {
 144   GraphKit kit(jvms);
 145   kit.C->print_inlining_update(this);
 146   PhaseGVN& gvn = kit.gvn();
 147   bool is_static = method()->is_static();
 148   address target = is_static ? SharedRuntime::get_resolve_static_call_stub()
 149                              : SharedRuntime::get_resolve_opt_virtual_call_stub();
 150 
 151   if (kit.C->log() != NULL) {
 152     kit.C->log()->elem("direct_call bci='%d'", jvms->bci());
 153   }
 154 
 155   CallStaticJavaNode *call = new CallStaticJavaNode(kit.C, tf(), target, method(), kit.bci());
 156   if (is_inlined_method_handle_intrinsic(jvms, method())) {
 157     // To be able to issue a direct call and skip a call to MH.linkTo*/invokeBasic adapter,
 158     // additional information about the method being invoked should be attached
 159     // to the call site to make resolution logic work
 160     // (see SharedRuntime::resolve_static_call_C).
 161     call->set_override_symbolic_info(true);
 162   }
 163   _call_node = call;  // Save the call node in case we need it later
 164   if (!is_static) {
 165     // Make an explicit receiver null_check as part of this call.
 166     // Since we share a map with the caller, his JVMS gets adjusted.
 167     kit.null_check_receiver_before_call(method());
 168     if (kit.stopped()) {
 169       // And dump it back to the caller, decorated with any exceptions:
 170       return kit.transfer_exceptions_into_jvms();
 171     }
 172     // Mark the call node as virtual, sort of:
 173     call->set_optimized_virtual(true);
 174     if (method()->is_method_handle_intrinsic() ||
 175         method()->is_compiled_lambda_form()) {
 176       call->set_method_handle_invoke(true);
 177     }
 178   }
 179   kit.set_arguments_for_java_call(call, is_late_inline());
 180   if (kit.stopped()) {
 181     return kit.transfer_exceptions_into_jvms();
 182   }
 183   kit.set_edges_for_java_call(call, false, _separate_io_proj);
 184   Node* ret = kit.set_results_for_java_call(call, _separate_io_proj);
 185   kit.push_node(method()->return_type()->basic_type(), ret);
 186   return kit.transfer_exceptions_into_jvms();
 187 }

 189 //--------------------------VirtualCallGenerator------------------------------
 190 // Internal class which handles all out-of-line calls checking receiver type.
 191 class VirtualCallGenerator : public CallGenerator {
 192 private:
 193   int _vtable_index;
 194 public:
 195   VirtualCallGenerator(ciMethod* method, int vtable_index)
 196     : CallGenerator(method), _vtable_index(vtable_index)
 197   {
 198     assert(vtable_index == Method::invalid_vtable_index ||
 199            vtable_index >= 0, "either invalid or usable");
 200   }
 201   virtual bool      is_virtual() const          { return true; }
 202   virtual JVMState* generate(JVMState* jvms);
 203 };

 205 JVMState* VirtualCallGenerator::generate(JVMState* jvms) {
 206   GraphKit kit(jvms);
 207   Node* receiver = kit.argument(0);
 208   kit.C->print_inlining_update(this);
 209 
 210   if (kit.C->log() != NULL) {
 211     kit.C->log()->elem("virtual_call bci='%d'", jvms->bci());
 212   }
 213 
 214   // If the receiver is a constant null, do not torture the system
 215   // by attempting to call through it.  The compile will proceed
 216   // correctly, but may bail out in final_graph_reshaping, because
 217   // the call instruction will have a seemingly deficient out-count.
 218   // (The bailout says something misleading about an "infinite loop".)
 219   if (!receiver->is_InlineType() && kit.gvn().type(receiver)->higher_equal(TypePtr::NULL_PTR)) {
 220     assert(Bytecodes::is_invoke(kit.java_bc()), "%d: %s", kit.java_bc(), Bytecodes::name(kit.java_bc()));
 221     ciMethod* declared_method = kit.method()->get_method_at_bci(kit.bci());
 222     int arg_size = declared_method->signature()->arg_size_for_bc(kit.java_bc());
 223     kit.inc_sp(arg_size);  // restore arguments
 224     kit.uncommon_trap(Deoptimization::Reason_null_check,
 225                       Deoptimization::Action_none,
 226                       NULL, "null receiver");
 227     return kit.transfer_exceptions_into_jvms();
 228   }
 229 
 230   // Ideally we would unconditionally do a null check here and let it
 231   // be converted to an implicit check based on profile information.
 232   // However currently the conversion to implicit null checks in
 233   // Block::implicit_null_check() only looks for loads and stores, not calls.
 234   ciMethod *caller = kit.method();
 235   ciMethodData *caller_md = (caller == NULL) ? NULL : caller->method_data();
 236   if (!UseInlineCaches || !ImplicitNullChecks || !os::zero_page_read_protected() ||
 237        ((ImplicitNullCheckThreshold > 0) && caller_md &&
 238        (caller_md->trap_count(Deoptimization::Reason_null_check)
 239        >= (uint)ImplicitNullCheckThreshold))) {
 240     // Make an explicit receiver null_check as part of this call.
 241     // Since we share a map with the caller, his JVMS gets adjusted.
 242     receiver = kit.null_check_receiver_before_call(method());
 243     if (kit.stopped()) {
 244       // And dump it back to the caller, decorated with any exceptions:
 245       return kit.transfer_exceptions_into_jvms();
 246     }
 247   }
 248 
 249   assert(!method()->is_static(), "virtual call must not be to static");
 250   assert(!method()->is_final(), "virtual call should not be to final");
 251   assert(!method()->is_private(), "virtual call should not be to private");
 252   assert(_vtable_index == Method::invalid_vtable_index || !UseInlineCaches,
 253          "no vtable calls if +UseInlineCaches ");
 254   address target = SharedRuntime::get_resolve_virtual_call_stub();
 255   // Normal inline cache used for call
 256   CallDynamicJavaNode *call = new CallDynamicJavaNode(tf(), target, method(), _vtable_index, kit.bci());
 257   if (is_inlined_method_handle_intrinsic(jvms, method())) {
 258     // To be able to issue a direct call (optimized virtual or virtual)
 259     // and skip a call to MH.linkTo*/invokeBasic adapter, additional information
 260     // about the method being invoked should be attached to the call site to
 261     // make resolution logic work (see SharedRuntime::resolve_{virtual,opt_virtual}_call_C).
 262     call->set_override_symbolic_info(true);
 263   }
 264   kit.set_arguments_for_java_call(call);
 265   if (kit.stopped()) {
 266     return kit.transfer_exceptions_into_jvms();
 267   }
 268   kit.set_edges_for_java_call(call);
 269   Node* ret = kit.set_results_for_java_call(call);
 270   kit.push_node(method()->return_type()->basic_type(), ret);
 271 
 272   // Represent the effect of an implicit receiver null_check
 273   // as part of this call.  Since we share a map with the caller,
 274   // his JVMS gets adjusted.
 275   kit.cast_not_null(receiver);
 276   return kit.transfer_exceptions_into_jvms();
 277 }

 279 CallGenerator* CallGenerator::for_inline(ciMethod* m, float expected_uses) {
 280   if (InlineTree::check_can_parse(m) != NULL)  return NULL;
 281   return new ParseGenerator(m, expected_uses);
 282 }

 284 // As a special case, the JVMS passed to this CallGenerator is
 285 // for the method execution already in progress, not just the JVMS
 286 // of the caller.  Thus, this CallGenerator cannot be mixed with others!
 287 CallGenerator* CallGenerator::for_osr(ciMethod* m, int osr_bci) {
 288   if (InlineTree::check_can_parse(m) != NULL)  return NULL;
 289   float past_uses = m->interpreter_invocation_count();
 290   float expected_uses = past_uses;
 291   return new ParseGenerator(m, expected_uses, true);
 292 }

 305 // Allow inlining decisions to be delayed
 306 class LateInlineCallGenerator : public DirectCallGenerator {
 307  private:
 308   jlong _unique_id;   // unique id for log compilation
 309   bool _is_pure_call; // a hint that the call doesn't have important side effects to care about
 310 
 311  protected:
 312   CallGenerator* _inline_cg;
 313   virtual bool do_late_inline_check(JVMState* jvms) { return true; }
 314 
 315  public:
 316   LateInlineCallGenerator(ciMethod* method, CallGenerator* inline_cg, bool is_pure_call = false) :
 317     DirectCallGenerator(method, true), _unique_id(0), _is_pure_call(is_pure_call), _inline_cg(inline_cg) {}
 318 
 319   virtual bool is_late_inline() const { return true; }
 320 
 321   // Convert the CallStaticJava into an inline
 322   virtual void do_late_inline();
 323 
 324   virtual JVMState* generate(JVMState* jvms) {
 325     Compile *C = Compile::current();
 326 
 327     C->log_inline_id(this);
 328 
 329     // Record that this call site should be revisited once the main
 330     // parse is finished.
 331     if (!is_mh_late_inline()) {
 332       C->add_late_inline(this);
 333     }
 334 
 335     // Emit the CallStaticJava and request separate projections so
 336     // that the late inlining logic can distinguish between fall
 337     // through and exceptional uses of the memory and io projections
 338     // as is done for allocations and macro expansion.
 339     return DirectCallGenerator::generate(jvms);
 340   }
 341 
 342   virtual void print_inlining_late(const char* msg) {
 343     CallNode* call = call_node();
 344     Compile* C = Compile::current();
 345     C->print_inlining_assert_ready();
 346     C->print_inlining(method(), call->jvms()->depth()-1, call->jvms()->bci(), msg);
 347     C->print_inlining_move_to(this);
 348     C->print_inlining_update_delayed(this);
 349   }
 350 
 351   virtual void set_unique_id(jlong id) {
 352     _unique_id = id;
 353   }
 354 
 355   virtual jlong unique_id() const {
 356     return _unique_id;
 357   }
 358 
 359   virtual CallGenerator* inline_cg() {
 360     return _inline_cg;
 361   }
 362 };

 364 void LateInlineCallGenerator::do_late_inline() {
 365   // Can't inline it
 366   CallStaticJavaNode* call = call_node();
 367   if (call == NULL || call->outcnt() == 0 ||
 368       call->in(0) == NULL || call->in(0)->is_top()) {
 369     return;
 370   }
 371 
 372   const GrowableArray<SigEntry>* sig_cc = method()->get_sig_cc();
 373   const TypeTuple* r = call->tf()->domain_cc();
 374   for (uint i1 = TypeFunc::Parms, i2 = 0; i1 < r->cnt(); i1++) {
 375     if (sig_cc != NULL) {
 376       // Skip reserved entries
 377       while (!SigEntry::skip_value_delimiters(sig_cc, i2)) {
 378         i2++;
 379       }
 380       if (SigEntry::is_reserved_entry(sig_cc, i2++)) {
 381         assert(call->in(i1)->is_top(), "should be top");
 382         continue;
 383       }
 384     }
 385     if (call->in(i1)->is_top() && r->field_at(i1) != Type::HALF) {
 386       assert(Compile::current()->inlining_incrementally(), "shouldn't happen during parsing");
 387       return;
 388     }
 389   }
 390 
 391   if (call->in(TypeFunc::Memory)->is_top()) {
 392     assert(Compile::current()->inlining_incrementally(), "shouldn't happen during parsing");
 393     return;
 394   }
 395 
 396   // check for unreachable loop
 397   CallProjections* callprojs = call->extract_projections(true);
 398   if (callprojs->fallthrough_catchproj == call->in(0) ||
 399       callprojs->catchall_catchproj == call->in(0) ||
 400       callprojs->fallthrough_memproj == call->in(TypeFunc::Memory) ||
 401       callprojs->catchall_memproj == call->in(TypeFunc::Memory) ||
 402       callprojs->fallthrough_ioproj == call->in(TypeFunc::I_O) ||
 403       callprojs->catchall_ioproj == call->in(TypeFunc::I_O) ||
 404       (callprojs->exobj != NULL && call->find_edge(callprojs->exobj) != -1)) {
 405     return;
 406   }
 407   bool result_not_used = true;
 408   for (uint i = 0; i < callprojs->nb_resproj; i++) {
 409     if (callprojs->resproj[i] != NULL) {
 410       if (callprojs->resproj[i]->outcnt() != 0) {
 411         result_not_used = false;
 412       }
 413       if (call->find_edge(callprojs->resproj[i]) != -1) {
 414         return;
 415       }
 416     }
 417   }
 418 
 419   Compile* C = Compile::current();
 420   // Remove inlined methods from Compiler's lists.
 421   if (call->is_macro()) {
 422     C->remove_macro_node(call);
 423   }
 424 
 425   if (_is_pure_call && result_not_used) {
 426     // The call is marked as pure (no important side effects), but result isn't used.
 427     // It's safe to remove the call.
 428     GraphKit kit(call->jvms());
 429     kit.replace_call(call, C->top(), true);
 430   } else {
 431     // Make a clone of the JVMState that appropriate to use for driving a parse
 432     JVMState* old_jvms = call->jvms();
 433     JVMState* jvms = old_jvms->clone_shallow(C);
 434     uint size = call->req();
 435     SafePointNode* map = new SafePointNode(size, jvms);
 436     for (uint i1 = 0; i1 < size; i1++) {
 437       map->init_req(i1, call->in(i1));
 438     }
 439 
 440     PhaseGVN& gvn = *C->initial_gvn();
 441     // Make sure the state is a MergeMem for parsing.
 442     if (!map->in(TypeFunc::Memory)->is_MergeMem()) {
 443       Node* mem = MergeMemNode::make(map->in(TypeFunc::Memory));
 444       gvn.set_type_bottom(mem);
 445       map->set_req(TypeFunc::Memory, mem);
 446     }
 447 
 448     // blow away old call arguments
 449     for (uint i1 = TypeFunc::Parms; i1 < r->cnt(); i1++) {
 450       map->set_req(i1, C->top());
 451     }
 452     jvms->set_map(map);
 453 
 454     // Make enough space in the expression stack to transfer
 455     // the incoming arguments and return value.
 456     map->ensure_stack(jvms, jvms->method()->max_stack());
 457     const TypeTuple *domain_sig = call->_tf->domain_sig();
 458     ExtendedSignature sig_cc = ExtendedSignature(method()->get_sig_cc(), SigEntryFilter());
 459     uint nargs = method()->arg_size();
 460     assert(domain_sig->cnt() - TypeFunc::Parms == nargs, "inconsistent signature");
 461 
 462     uint j = TypeFunc::Parms;
 463     for (uint i1 = 0; i1 < nargs; i1++) {
 464       const Type* t = domain_sig->field_at(TypeFunc::Parms + i1);
 465       if (method()->has_scalarized_args() && t->is_inlinetypeptr() && !t->maybe_null()) {
 466         // Inline type arguments are not passed by reference: we get an argument per
 467         // field of the inline type. Build InlineTypeNodes from the inline type arguments.
 468         GraphKit arg_kit(jvms, &gvn);
 469         InlineTypeNode* vt = InlineTypeNode::make_from_multi(&arg_kit, call, sig_cc, t->inline_klass(), j, true);
 470         map->set_control(arg_kit.control());
 471         map->set_argument(jvms, i1, vt);
 472       } else {
 473         map->set_argument(jvms, i1, call->in(j++));
 474         BasicType bt = t->basic_type();
 475         while (SigEntry::next_is_reserved(sig_cc, bt, true)) {
 476           j += type2size[bt]; // Skip reserved arguments
 477         }
 478       }
 479     }
 480 
 481     C->print_inlining_assert_ready();
 482 
 483     C->print_inlining_move_to(this);
 484 
 485     C->log_late_inline(this);
 486 
 487     // This check is done here because for_method_handle_inline() method
 488     // needs jvms for inlined state.
 489     if (!do_late_inline_check(jvms)) {
 490       map->disconnect_inputs(NULL, C);
 491       return;
 492     }
 493 
 494     // Allocate a buffer for the returned InlineTypeNode because the caller expects an oop return.
 495     // Do this before the method handle call in case the buffer allocation triggers deoptimization.
 496     Node* buffer_oop = NULL;
 497     if (is_mh_late_inline() && _inline_cg->method()->return_type()->is_inlinetype()) {
 498       GraphKit arg_kit(jvms, &gvn);
 499       {
 500         PreserveReexecuteState preexecs(&arg_kit);
 501         arg_kit.jvms()->set_should_reexecute(true);
 502         arg_kit.inc_sp(nargs);
 503         Node* klass_node = arg_kit.makecon(TypeKlassPtr::make(_inline_cg->method()->return_type()->as_inline_klass()));
 504         buffer_oop = arg_kit.new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true);
 505       }
 506       jvms = arg_kit.transfer_exceptions_into_jvms();
 507     }
 508 
 509     // Setup default node notes to be picked up by the inlining
 510     Node_Notes* old_nn = C->node_notes_at(call->_idx);
 511     if (old_nn != NULL) {
 512       Node_Notes* entry_nn = old_nn->clone(C);
 513       entry_nn->set_jvms(jvms);
 514       C->set_default_node_notes(entry_nn);
 515     }
 516 
 517     // Now perform the inlining using the synthesized JVMState
 518     JVMState* new_jvms = _inline_cg->generate(jvms);
 519     if (new_jvms == NULL)  return;  // no change
 520     if (C->failing())      return;
 521 
 522     // Capture any exceptional control flow
 523     GraphKit kit(new_jvms);
 524 
 525     // Find the result object
 526     Node* result = C->top();
 527     int   result_size = method()->return_type()->size();
 528     if (result_size != 0 && !kit.stopped()) {
 529       result = (result_size == 1) ? kit.pop() : kit.pop_pair();
 530     }
 531 
 532     C->set_has_loops(C->has_loops() || _inline_cg->method()->has_loops());
 533     C->env()->notice_inlined_method(_inline_cg->method());
 534     C->set_inlining_progress(true);
 535     C->set_do_cleanup(kit.stopped()); // path is dead; needs cleanup
 536 
 537     // Handle inline type returns
 538     bool returned_as_fields = call->tf()->returns_inline_type_as_fields();
 539     if (result->is_InlineType()) {
 540       // Only possible if is_mh_late_inline() when the callee does not "know" that the caller expects an oop
 541       assert(is_mh_late_inline() && !returned_as_fields, "sanity");
 542       assert(buffer_oop != NULL, "should have allocated a buffer");
 543       InlineTypeNode* vt = result->as_InlineType();
 544       vt->store(&kit, buffer_oop, buffer_oop, vt->type()->inline_klass(), 0);
 545       // Do not let stores that initialize this buffer be reordered with a subsequent
 546       // store that would make this buffer accessible by other threads.
 547       AllocateNode* alloc = AllocateNode::Ideal_allocation(buffer_oop, &kit.gvn());
 548       assert(alloc != NULL, "must have an allocation node");
 549       kit.insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
 550       result = buffer_oop;
 551     } else if (result->is_InlineTypePtr() && returned_as_fields) {

 796 JVMState* PredictedCallGenerator::generate(JVMState* jvms) {
 797   GraphKit kit(jvms);
 798   kit.C->print_inlining_update(this);
 799   PhaseGVN& gvn = kit.gvn();
 800   // We need an explicit receiver null_check before checking its type.
 801   // We share a map with the caller, so his JVMS gets adjusted.
 802   Node* receiver = kit.argument(0);
 803   CompileLog* log = kit.C->log();
 804   if (log != NULL) {
 805     log->elem("predicted_call bci='%d' exact='%d' klass='%d'",
 806               jvms->bci(), (_exact_check ? 1 : 0), log->identify(_predicted_receiver));
 807   }
 808 
 809   receiver = kit.null_check_receiver_before_call(method());
 810   if (kit.stopped()) {
 811     return kit.transfer_exceptions_into_jvms();
 812   }
 813 
 814   // Make a copy of the replaced nodes in case we need to restore them
 815   ReplacedNodes replaced_nodes = kit.map()->replaced_nodes();
 816   replaced_nodes.clone();
 817 
 818   Node* casted_receiver = receiver;  // will get updated in place...
 819   Node* slow_ctl = NULL;
 820   if (_exact_check) {
 821     slow_ctl = kit.type_check_receiver(receiver, _predicted_receiver, _hit_prob,
 822                                        &casted_receiver);
 823   } else {
 824     slow_ctl = kit.subtype_check_receiver(receiver, _predicted_receiver,
 825                                           &casted_receiver);
 826   }
 827 
 828   SafePointNode* slow_map = NULL;
 829   JVMState* slow_jvms = NULL;
 830   { PreserveJVMState pjvms(&kit);
 831     kit.set_control(slow_ctl);
 832     if (!kit.stopped()) {
 833       slow_jvms = _if_missed->generate(kit.sync_jvms());
 834       if (kit.failing())
 835         return NULL;  // might happen because of NodeCountInliningCutoff
 836       assert(slow_jvms != NULL, "must be");
 837       kit.add_exception_states_from(slow_jvms);
 838       kit.set_map(slow_jvms->map());
 839       if (!kit.stopped())
 840         slow_map = kit.stop();
 841     }
 842   }
 843 
 844   if (kit.stopped()) {
 845     // Instance exactly does not matches the desired type.
 846     kit.set_jvms(slow_jvms);
 847     return kit.transfer_exceptions_into_jvms();
 848   }
 849 
 850   // fall through if the instance exactly matches the desired type
 851   kit.replace_in_map(receiver, casted_receiver);
 852 
 853   // Make the hot call:
 854   JVMState* new_jvms = _if_hit->generate(kit.sync_jvms());
 855   if (new_jvms == NULL) {
 856     // Inline failed, so make a direct call.
 857     assert(_if_hit->is_inline(), "must have been a failed inline");
 858     CallGenerator* cg = CallGenerator::for_direct_call(_if_hit->method());
 859     new_jvms = cg->generate(kit.sync_jvms());
 860   }
 861   kit.add_exception_states_from(new_jvms);
 862   kit.set_jvms(new_jvms);
 863 
 864   // Need to merge slow and fast?
 865   if (slow_map == NULL) {
 866     // The fast path is the only path remaining.
 867     return kit.transfer_exceptions_into_jvms();
 868   }
 869 
 870   if (kit.stopped()) {
 871     // Inlined method threw an exception, so it's just the slow path after all.
 872     kit.set_jvms(slow_jvms);
 873     return kit.transfer_exceptions_into_jvms();
 874   }
 875 
 876   // Allocate inline types if they are merged with objects (similar to Parse::merge_common())
 877   uint tos = kit.jvms()->stkoff() + kit.sp();
 878   uint limit = slow_map->req();
 879   for (uint i = TypeFunc::Parms; i < limit; i++) {
 880     Node* m = kit.map()->in(i);
 881     Node* n = slow_map->in(i);
 882     const Type* t = gvn.type(m)->meet_speculative(gvn.type(n));
 883     if (m->is_InlineType() && !t->isa_inlinetype()) {
 884       // Allocate inline type in fast path
 885       m = m->as_InlineType()->buffer(&kit);
 886       kit.map()->set_req(i, m);
 887     }
 888     if (n->is_InlineType() && !t->isa_inlinetype()) {
 889       // Allocate inline type in slow path
 890       PreserveJVMState pjvms(&kit);
 891       kit.set_map(slow_map);
 892       n = n->as_InlineType()->buffer(&kit);
 893       kit.map()->set_req(i, n);
 894       slow_map = kit.stop();
 895     }
 896   }
 897 
 898   // There are 2 branches and the replaced nodes are only valid on
 899   // one: restore the replaced nodes to what they were before the
 900   // branch.
 901   kit.map()->set_replaced_nodes(replaced_nodes);
 902 
 903   // Finish the diamond.
 904   kit.C->set_has_split_ifs(true); // Has chance for split-if optimization
 905   RegionNode* region = new RegionNode(3);
 906   region->init_req(1, kit.control());
 907   region->init_req(2, slow_map->control());
 908   kit.set_control(gvn.transform(region));
 909   Node* iophi = PhiNode::make(region, kit.i_o(), Type::ABIO);
 910   iophi->set_req(2, slow_map->i_o());
 911   kit.set_i_o(gvn.transform(iophi));
 912   // Merge memory
 913   kit.merge_memory(slow_map->merged_memory(), region, 2);
 914   // Transform new memory Phis.
 915   for (MergeMemStream mms(kit.merged_memory()); mms.next_non_empty();) {
 916     Node* phi = mms.memory();
 917     if (phi->is_Phi() && phi->in(0) == region) {
 918       mms.set_memory(gvn.transform(phi));
 919     }
 920   }
 921   for (uint i = TypeFunc::Parms; i < limit; i++) {
 922     // Skip unused stack slots; fast forward to monoff();
 923     if (i == tos) {
 924       i = kit.jvms()->monoff();
 925       if( i >= limit ) break;
 926     }
 927     Node* m = kit.map()->in(i);
 928     Node* n = slow_map->in(i);
 929     if (m != n) {
 930       const Type* t = gvn.type(m)->meet_speculative(gvn.type(n));
 931       Node* phi = PhiNode::make(region, m, t);
 932       phi->set_req(2, n);
 933       kit.map()->set_req(i, gvn.transform(phi));
 934     }
 935   }
 936   return kit.transfer_exceptions_into_jvms();
 937 }

 940 CallGenerator* CallGenerator::for_method_handle_call(JVMState* jvms, ciMethod* caller, ciMethod* callee) {
 941   assert(callee->is_method_handle_intrinsic(), "for_method_handle_call mismatch");
 942   bool input_not_const;
 943   CallGenerator* cg = CallGenerator::for_method_handle_inline(jvms, caller, callee, input_not_const);
 944   Compile* C = Compile::current();

 946     if (AlwaysIncrementalInline) {
 947       return CallGenerator::for_late_inline(callee, cg);
 948     } else {
 949       return cg;
 950     }
 951   }
 952   int bci = jvms->bci();
 953   ciCallProfile profile = caller->call_profile_at_bci(bci);
 954   int call_site_count = caller->scale_count(profile.count());
 955 
 956   if (IncrementalInline && (AlwaysIncrementalInline ||
 957                             (call_site_count > 0 && (input_not_const || !C->inlining_incrementally() || C->over_inlining_cutoff())))) {
 958     return CallGenerator::for_mh_late_inline(caller, callee, input_not_const);
 959   } else {
 960     // Out-of-line call.
 961     return CallGenerator::for_direct_call(callee);
 962   }

 965 static void cast_argument(int nargs, int arg_nb, ciType* t, GraphKit& kit) {
 966   PhaseGVN& gvn = kit.gvn();
 967   Node* arg = kit.argument(arg_nb);
 968   const Type* arg_type = arg->bottom_type();
 969   const Type* sig_type = TypeOopPtr::make_from_klass(t->as_klass());
 970   if (arg_type->isa_oopptr() && !arg_type->higher_equal(sig_type)) {
 971     const Type* narrowed_arg_type = arg_type->join_speculative(sig_type); // keep speculative part
 972     arg = gvn.transform(new CheckCastPPNode(kit.control(), arg, narrowed_arg_type));
 973     kit.set_argument(arg_nb, arg);
 974   }
 975   if (sig_type->is_inlinetypeptr() && !arg->is_InlineType() &&
 976       !kit.gvn().type(arg)->maybe_null() && t->as_inline_klass()->is_scalarizable()) {
 977     arg = InlineTypeNode::make_from_oop(&kit, arg, t->as_inline_klass());
 978     kit.set_argument(arg_nb, arg);
 979   }
 980 }

 982 CallGenerator* CallGenerator::for_method_handle_inline(JVMState* jvms, ciMethod* caller, ciMethod* callee, bool& input_not_const) {
 983   GraphKit kit(jvms);
 984   PhaseGVN& gvn = kit.gvn();
 985   Compile* C = kit.C;
 986   vmIntrinsics::ID iid = callee->intrinsic_id();
 987   input_not_const = true;
 988   switch (iid) {
 989   case vmIntrinsics::_invokeBasic:
 990     {
 991       // Get MethodHandle receiver:
 992       Node* receiver = kit.argument(0);
 993       if (receiver->Opcode() == Op_ConP) {
 994         input_not_const = false;
 995         const TypeOopPtr* oop_ptr = receiver->bottom_type()->is_oopptr();
 996         ciMethod* target = oop_ptr->const_oop()->as_method_handle()->get_vmtarget();
 997         const int vtable_index = Method::invalid_vtable_index;
 998 
 999         if (!ciMethod::is_consistent_info(callee, target)) {
1000           print_inlining_failure(C, callee, jvms->depth() - 1, jvms->bci(),
1001                                  "signatures mismatch");
1002           return NULL;
1003         }
1004 
1005         CallGenerator* cg = C->call_generator(target, vtable_index,
1006                                               false /* call_does_dispatch */,
1007                                               jvms,
1008                                               true /* allow_inline */,
1009                                               PROB_ALWAYS,
1010                                               NULL,
1011                                               true);
1012         return cg;
1013       } else {
1014         print_inlining_failure(C, callee, jvms->depth() - 1, jvms->bci(),
1015                                "receiver not constant");
1016       }
1017     }
1018     break;
1019 
1020   case vmIntrinsics::_linkToVirtual:
1021   case vmIntrinsics::_linkToStatic:
1022   case vmIntrinsics::_linkToSpecial:
1023   case vmIntrinsics::_linkToInterface:
1024     {
1025       int nargs = callee->arg_size();
1026       // Get MemberName argument:
1027       Node* member_name = kit.argument(nargs - 1);
1028       if (member_name->Opcode() == Op_ConP) {
1029         input_not_const = false;
1030         const TypeOopPtr* oop_ptr = member_name->bottom_type()->is_oopptr();
1031         ciMethod* target = oop_ptr->const_oop()->as_member_name()->get_vmtarget();
1032 
1033         if (!ciMethod::is_consistent_info(callee, target)) {
1034           print_inlining_failure(C, callee, jvms->depth() - 1, jvms->bci(),
1035                                  "signatures mismatch");
1036           return NULL;
1037         }
1038 
1039         // In lambda forms we erase signature types to avoid resolving issues
1040         // involving class loaders.  When we optimize a method handle invoke
1041         // to a direct call we must cast the receiver and arguments to its
1042         // actual types.
1043         ciSignature* signature = target->signature();
1044         const int receiver_skip = target->is_static() ? 0 : 1;
1045         // Cast receiver to its type.
1046         if (!target->is_static()) {
1047           cast_argument(nargs, 0, signature->accessing_klass(), kit);
1048         }
1049         // Cast reference arguments to its type.
1050         for (int i = 0, j = 0; i < signature->count(); i++) {
1051           ciType* t = signature->type_at(i);
1052           if (t->is_klass()) {
1053             cast_argument(nargs, receiver_skip + j, t, kit);
1054           }
1055           j += t->size();  // long and double take two slots
1056         }
1057 
1058         // Try to get the most accurate receiver type
1059         const bool is_virtual              = (iid == vmIntrinsics::_linkToVirtual);
1060         const bool is_virtual_or_interface = (is_virtual || iid == vmIntrinsics::_linkToInterface);
1061         int  vtable_index       = Method::invalid_vtable_index;
1062         bool call_does_dispatch = false;
1063 
1064         ciKlass* speculative_receiver_type = NULL;
1065         if (is_virtual_or_interface) {
1066           ciInstanceKlass* klass = target->holder();
1067           Node*             receiver_node = kit.argument(0);
1068           const TypeOopPtr* receiver_type = gvn.type(receiver_node)->isa_oopptr();
1069           // call_does_dispatch and vtable_index are out-parameters.  They might be changed.
1070           // optimize_virtual_call() takes 2 different holder
1071           // arguments for a corner case that doesn't apply here (see
1072           // Parse::do_call())
1073           target = C->optimize_virtual_call(caller, jvms->bci(), klass, klass,
1074                                             target, receiver_type, is_virtual,
1075                                             call_does_dispatch, vtable_index, // out-parameters
1076                                             false /* check_access */);
1077           // We lack profiling at this call but type speculation may
1078           // provide us with a type
1079           speculative_receiver_type = (receiver_type != NULL) ? receiver_type->speculative_type() : NULL;
1080         }
1081         CallGenerator* cg = C->call_generator(target, vtable_index, call_does_dispatch, jvms,
1082                                               !StressMethodHandleLinkerInlining /* allow_inline */,
1083                                               PROB_ALWAYS,
1084                                               speculative_receiver_type,
1085                                               true);
1086         return cg;
1087       } else {
1088         print_inlining_failure(C, callee, jvms->depth() - 1, jvms->bci(),
1089                                "member_name not constant");
1090       }
1091     }
1092     break;
1093 
1094   default:
1095     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
1096     break;
1097   }
1098   return NULL;
1099 }

1131 JVMState* PredicatedIntrinsicGenerator::generate(JVMState* jvms) {
1132   // The code we want to generate here is:
1133   //    if (receiver == NULL)
1134   //        uncommon_Trap
1135   //    if (predicate(0))
1136   //        do_intrinsic(0)
1137   //    else
1138   //    if (predicate(1))
1139   //        do_intrinsic(1)
1140   //    ...
1141   //    else
1142   //        do_java_comp
1143 
1144   GraphKit kit(jvms);
1145   PhaseGVN& gvn = kit.gvn();
1146 
1147   CompileLog* log = kit.C->log();
1148   if (log != NULL) {
1149     log->elem("predicated_intrinsic bci='%d' method='%d'",
1150               jvms->bci(), log->identify(method()));
1151   }
1152 
1153   if (!method()->is_static()) {
1154     // We need an explicit receiver null_check before checking its type in predicate.
1155     // We share a map with the caller, so his JVMS gets adjusted.
1156     kit.null_check_receiver_before_call(method());
1157     if (kit.stopped()) {
1158       return kit.transfer_exceptions_into_jvms();
1159     }
1160   }
1161 
1162   int n_predicates = _intrinsic->predicates_count();
1163   assert(n_predicates > 0, "sanity");
1164 
1165   JVMState** result_jvms = NEW_RESOURCE_ARRAY(JVMState*, (n_predicates+1));
1166 
1167   // Region for normal compilation code if intrinsic failed.
1168   Node* slow_region = new RegionNode(1);
1169 
1170   int results = 0;
1171   for (int predicate = 0; (predicate < n_predicates) && !kit.stopped(); predicate++) {
1172 #ifdef ASSERT
1173     JVMState* old_jvms = kit.jvms();
1174     SafePointNode* old_map = kit.map();
1175     Node* old_io  = old_map->i_o();
1176     Node* old_mem = old_map->memory();
1177     Node* old_exc = old_map->next_exception();
1178 #endif
1179     Node* else_ctrl = _intrinsic->generate_predicate(kit.sync_jvms(), predicate);
1180 #ifdef ASSERT
1181     // Assert(no_new_memory && no_new_io && no_new_exceptions) after generate_predicate.
1182     assert(old_jvms == kit.jvms(), "generate_predicate should not change jvm state");
1183     SafePointNode* new_map = kit.map();
1184     assert(old_io  == new_map->i_o(), "generate_predicate should not change i_o");
1185     assert(old_mem == new_map->memory(), "generate_predicate should not change memory");
1186     assert(old_exc == new_map->next_exception(), "generate_predicate should not add exceptions");
1187 #endif
1188     if (!kit.stopped()) {
1189       PreserveJVMState pjvms(&kit);
1190       // Generate intrinsic code:
1191       JVMState* new_jvms = _intrinsic->generate(kit.sync_jvms());
1192       if (new_jvms == NULL) {
1193         // Intrinsic failed, use normal compilation path for this predicate.
1194         slow_region->add_req(kit.control());
1195       } else {
1196         kit.add_exception_states_from(new_jvms);
1197         kit.set_jvms(new_jvms);
1198         if (!kit.stopped()) {
1199           result_jvms[results++] = kit.jvms();
1200         }
1201       }
1202     }
1203     if (else_ctrl == NULL) {
1204       else_ctrl = kit.C->top();
1205     }
1206     kit.set_control(else_ctrl);
1207   }
1208   if (!kit.stopped()) {
1209     // Final 'else' after predicates.
1210     slow_region->add_req(kit.control());
1211   }
1212   if (slow_region->req() > 1) {
1213     PreserveJVMState pjvms(&kit);
1214     // Generate normal compilation code:
1215     kit.set_control(gvn.transform(slow_region));
1216     JVMState* new_jvms = _cg->generate(kit.sync_jvms());
1217     if (kit.failing())
1218       return NULL;  // might happen because of NodeCountInliningCutoff
1219     assert(new_jvms != NULL, "must be");
1220     kit.add_exception_states_from(new_jvms);
1221     kit.set_jvms(new_jvms);
1222     if (!kit.stopped()) {
1223       result_jvms[results++] = kit.jvms();
1224     }
1225   }
1226 
1227   if (results == 0) {
1228     // All paths ended in uncommon traps.
1229     (void) kit.stop();
1230     return kit.transfer_exceptions_into_jvms();
1231   }
1232 
1233   if (results == 1) { // Only one path
1234     kit.set_jvms(result_jvms[0]);
1235     return kit.transfer_exceptions_into_jvms();
1236   }
1237 
1238   // Merge all paths.
1239   kit.C->set_has_split_ifs(true); // Has chance for split-if optimization
1240   RegionNode* region = new RegionNode(results + 1);
1241   Node* iophi = PhiNode::make(region, kit.i_o(), Type::ABIO);
1242   for (int i = 0; i < results; i++) {
1243     JVMState* jvms = result_jvms[i];
1244     int path = i + 1;
1245     SafePointNode* map = jvms->map();
1246     region->init_req(path, map->control());
1247     iophi->set_req(path, map->i_o());
1248     if (i == 0) {
1249       kit.set_jvms(jvms);
1250     } else {
1251       kit.merge_memory(map->merged_memory(), region, path);
1252     }
1253   }
1254   kit.set_control(gvn.transform(region));
1255   kit.set_i_o(gvn.transform(iophi));
1256   // Transform new memory Phis.
1257   for (MergeMemStream mms(kit.merged_memory()); mms.next_non_empty();) {
1258     Node* phi = mms.memory();
1259     if (phi->is_Phi() && phi->in(0) == region) {
1260       mms.set_memory(gvn.transform(phi));
1261     }
1262   }
1263 
1264   // Merge debug info.
1265   Node** ins = NEW_RESOURCE_ARRAY(Node*, results);
1266   uint tos = kit.jvms()->stkoff() + kit.sp();
1267   Node* map = kit.map();
1268   uint limit = map->req();
1269   for (uint i = TypeFunc::Parms; i < limit; i++) {
1270     // Skip unused stack slots; fast forward to monoff();
1271     if (i == tos) {
1272       i = kit.jvms()->monoff();
1273       if( i >= limit ) break;
1274     }
1275     Node* n = map->in(i);
1276     ins[0] = n;
1277     const Type* t = gvn.type(n);
1278     bool needs_phi = false;
1279     for (int j = 1; j < results; j++) {
1280       JVMState* jvms = result_jvms[j];
1281       Node* jmap = jvms->map();
1282       Node* m = NULL;
1283       if (jmap->req() > i) {
1284         m = jmap->in(i);
1285         if (m != n) {
1286           needs_phi = true;
1287           t = t->meet_speculative(gvn.type(m));
1288         }
1289       }
1290       ins[j] = m;
1291     }
1292     if (needs_phi) {
1293       Node* phi = PhiNode::make(region, n, t);
1294       for (int j = 1; j < results; j++) {
1295         phi->set_req(j + 1, ins[j]);
1296       }
1297       map->set_req(i, gvn.transform(phi));
1298     }
1299   }
1300 
1301   return kit.transfer_exceptions_into_jvms();
1302 }

1366 void WarmCallInfo::init(JVMState* call_site, ciMethod* call_method, ciCallProfile& profile, float prof_factor) {
1367   int call_count = profile.count();
1368   int code_size = call_method->code_size();
1369 
1370   // Expected execution count is based on the historical count:
1371   _count = call_count < 0 ? 1 : call_site->method()->scale_count(call_count, prof_factor);
1372 
1373   // Expected profit from inlining, in units of simple call-overheads.
1374   _profit = 1.0;
1375 
1376   // Expected work performed by the call in units of call-overheads.
1377   // %%% need an empirical curve fit for "work" (time in call)
1378   float bytecodes_per_call = 3;
1379   _work = 1.0 + code_size / bytecodes_per_call;
1380 
1381   // Expected size of compilation graph:
1382   // -XX:+PrintParseStatistics once reported:
1383   //  Methods seen: 9184  Methods parsed: 9184  Nodes created: 1582391
1384   //  Histogram of 144298 parsed bytecodes:
1385   // %%% Need an better predictor for graph size.
1386   _size = NODES_OVERHEAD_PER_METHOD + (NODES_PER_BYTECODE * code_size);
1387 }

1389 // is_cold:  Return true if the node should never be inlined.
1390 // This is true if any of the key metrics are extreme.
1391 bool WarmCallInfo::is_cold() const {
1392   if (count()  <  WarmCallMinCount)        return true;
1393   if (profit() <  WarmCallMinProfit)       return true;
1394   if (work()   >  WarmCallMaxWork)         return true;
1395   if (size()   >  WarmCallMaxSize)         return true;
1396   return false;
1397 }

1399 // is_hot:  Return true if the node should be inlined immediately.
1400 // This is true if any of the key metrics are extreme.
1401 bool WarmCallInfo::is_hot() const {
1402   assert(!is_cold(), "eliminate is_cold cases before testing is_hot");
1403   if (count()  >= HotCallCountThreshold)   return true;
1404   if (profit() >= HotCallProfitThreshold)  return true;
1405   if (work()   <= HotCallTrivialWork)      return true;
1406   if (size()   <= HotCallTrivialSize)      return true;
1407   return false;
1408 }

1410 // compute_heat:
1411 float WarmCallInfo::compute_heat() const {
1412   assert(!is_cold(), "compute heat only on warm nodes");
1413   assert(!is_hot(),  "compute heat only on warm nodes");
1414   int min_size = MAX2(0,   (int)HotCallTrivialSize);
1415   int max_size = MIN2(500, (int)WarmCallMaxSize);
1416   float method_size = (size() - min_size) / MAX2(1, max_size - min_size);
1417   float size_factor;
1418   if      (method_size < 0.05)  size_factor = 4;   // 2 sigmas better than avg.
1419   else if (method_size < 0.15)  size_factor = 2;   // 1 sigma better than avg.
1420   else if (method_size < 0.5)   size_factor = 1;   // better than avg.
1421   else                          size_factor = 0.5; // worse than avg.
1422   return (count() * profit() * size_factor);
1423 }

1425 bool WarmCallInfo::warmer_than(WarmCallInfo* that) {
1426   assert(this != that, "compare only different WCIs");
1427   assert(this->heat() != 0 && that->heat() != 0, "call compute_heat 1st");
1428   if (this->heat() > that->heat())   return true;
1429   if (this->heat() < that->heat())   return false;
1430   assert(this->heat() == that->heat(), "no NaN heat allowed");
1431   // Equal heat.  Break the tie some other way.
1432   if (!this->call() || !that->call())  return (address)this > (address)that;
1433   return this->call()->_idx > that->call()->_idx;
1434 }

1439 WarmCallInfo* WarmCallInfo::insert_into(WarmCallInfo* head) {
1440   assert(next() == UNINIT_NEXT, "not yet on any list");
1441   WarmCallInfo* prev_p = NULL;
1442   WarmCallInfo* next_p = head;
1443   while (next_p != NULL && next_p->warmer_than(this)) {
1444     prev_p = next_p;
1445     next_p = prev_p->next();
1446   }
1447   // Install this between prev_p and next_p.
1448   this->set_next(next_p);
1449   if (prev_p == NULL)
1450     head = this;
1451   else
1452     prev_p->set_next(this);
1453   return head;
1454 }

1456 WarmCallInfo* WarmCallInfo::remove_from(WarmCallInfo* head) {
1457   WarmCallInfo* prev_p = NULL;
1458   WarmCallInfo* next_p = head;
1459   while (next_p != this) {
1460     assert(next_p != NULL, "this must be in the list somewhere");
1461     prev_p = next_p;
1462     next_p = prev_p->next();
1463   }
1464   next_p = this->next();
1465   debug_only(this->set_next(UNINIT_NEXT));
1466   // Remove this from between prev_p and next_p.
1467   if (prev_p == NULL)
1468     head = next_p;
1469   else
1470     prev_p->set_next(next_p);
1471   return head;
1472 }
