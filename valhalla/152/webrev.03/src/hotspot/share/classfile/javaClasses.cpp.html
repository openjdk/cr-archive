 113 InjectedField* JavaClasses::get_injected(Symbol* class_name, int* field_count) {
 114   *field_count = 0;
 115 
 116   vmSymbols::SID sid = vmSymbols::find_sid(class_name);
 117   if (sid == vmSymbols::NO_SID) {
 118     // Only well known classes can inject fields
 119     return NULL;
 120   }
 121 
 122   int count = 0;
 123   int start = -1;
 124 
 125 #define LOOKUP_INJECTED_FIELD(klass, name, signature, may_be_java) \
 126   if (sid == vmSymbols::VM_SYMBOL_ENUM_NAME(klass)) {              \
 127     count++;                                                       \
 128     if (start == -1) start = klass##_##name##_enum;                \
 129   }
 130   ALL_INJECTED_FIELDS(LOOKUP_INJECTED_FIELD);
 131 #undef LOOKUP_INJECTED_FIELD
 132 
 133   if (start != -1) {
 134     *field_count = count;
 135     return _injected_fields + start;
 136   }
 137   return NULL;
 138 }

 141 // Helpful routine for computing field offsets at run time rather than hardcoding them
 142 // Finds local fields only, including static fields.  Static field offsets are from the
 143 // beginning of the mirror.
 144 static void compute_offset(int &dest_offset,
 145                            InstanceKlass* ik, Symbol* name_symbol, Symbol* signature_symbol,
 146                            bool is_static = false) {
 147   fieldDescriptor fd;
 148   if (ik == NULL) {
 149     ResourceMark rm;
 150     log_error(class)("Mismatch JDK version for field: %s type: %s", name_symbol->as_C_string(), signature_symbol->as_C_string());
 151     vm_exit_during_initialization("Invalid layout of well-known class");
 152   }
 153 
 154   if (!ik->find_local_field(name_symbol, signature_symbol, &fd) || fd.is_static() != is_static) {
 155     ResourceMark rm;
 156     log_error(class)("Invalid layout of %s field: %s type: %s", ik->external_name(),
 157                      name_symbol->as_C_string(), signature_symbol->as_C_string());
 158 #ifndef PRODUCT
 159     // Prints all fields and offsets
 160     Log(class) lt;
 161     LogStream ls(lt.error());
 162     ik->print_on(&ls);
 163 #endif //PRODUCT
 164     vm_exit_during_initialization("Invalid layout of well-known class: use -Xlog:class+load=info to see the origin of the problem class");
 165   }
 166   dest_offset = fd.offset();
 167 }

 169 // Overloading to pass name as a string.
 170 static void compute_offset(int& dest_offset, InstanceKlass* ik,
 171                            const char* name_string, Symbol* signature_symbol,
 172                            bool is_static = false) {
 173   TempNewSymbol name = SymbolTable::probe(name_string, (int)strlen(name_string));
 174   if (name == NULL) {
 175     ResourceMark rm;
 176     log_error(class)("Name %s should be in the SymbolTable since its class is loaded", name_string);
 177     vm_exit_during_initialization("Invalid layout of well-known class", ik->external_name());
 178   }
 179   compute_offset(dest_offset, ik, name, signature_symbol, is_static);
 180 }

 229 class CompactStringsFixup : public FieldClosure {
 230 private:
 231   bool _value;
 232 
 233 public:
 234   CompactStringsFixup(bool value) : _value(value) {}
 235 
 236   void do_field(fieldDescriptor* fd) {
 237     if (fd->name() == vmSymbols::compact_strings_name()) {
 238       oop mirror = fd->field_holder()->java_mirror();
 239       assert(fd->field_holder() == SystemDictionary::String_klass(), "Should be String");
 240       assert(mirror != NULL, "String must have mirror already");
 241       mirror->bool_field_put(fd->offset(), _value);
 242     }
 243   }
 244 };

 251 Handle java_lang_String::basic_create(int length, bool is_latin1, TRAPS) {
 252   assert(_initialized, "Must be initialized");
 253   assert(CompactStrings || !is_latin1, "Must be UTF16 without CompactStrings");
 254 
 255   // Create the String object first, so there's a chance that the String
 256   // and the char array it points to end up in the same cache line.
 257   oop obj;
 258   obj = SystemDictionary::String_klass()->allocate_instance(CHECK_NH);
 259 
 260   // Create the char array.  The String object must be handlized here
 261   // because GC can happen as a result of the allocation attempt.
 262   Handle h_obj(THREAD, obj);
 263   int arr_length = is_latin1 ? length : length << 1; // 2 bytes per UTF16.
 264   typeArrayOop buffer = oopFactory::new_byteArray(arr_length, CHECK_NH);;
 265 
 266   // Point the String at the char array
 267   obj = h_obj();
 268   set_value(obj, buffer);
 269   // No need to zero the offset, allocation zero'ed the entire String object
 270   set_coder(obj, is_latin1 ? CODER_LATIN1 : CODER_UTF16);
 271   return h_obj;
 272 }

 274 Handle java_lang_String::create_from_unicode(const jchar* unicode, int length, TRAPS) {
 275   bool is_latin1 = CompactStrings && UNICODE::is_latin1(unicode, length);
 276   Handle h_obj = basic_create(length, is_latin1, CHECK_NH);
 277   typeArrayOop buffer = value(h_obj());
 278   assert(TypeArrayKlass::cast(buffer->klass())->element_type() == T_BYTE, "only byte[]");
 279   if (is_latin1) {
 280     for (int index = 0; index < length; index++) {
 281       buffer->byte_at_put(index, (jbyte)unicode[index]);
 282     }
 283   } else {
 284     for (int index = 0; index < length; index++) {
 285       buffer->char_at_put(index, unicode[index]);
 286     }
 287   }
 288 
 289 #ifdef ASSERT
 290   {
 291     ResourceMark rm;
 292     char* expected = UNICODE::as_utf8(unicode, length);
 293     char* actual = as_utf8_string(h_obj());
 294     if (strcmp(expected, actual) != 0) {
 295       fatal("Unicode conversion failure: %s --> %s", expected, actual);
 296     }
 297   }
 298 #endif
 299 
 300   return h_obj;
 301 }

 308 Handle java_lang_String::create_from_str(const char* utf8_str, TRAPS) {
 309   if (utf8_str == NULL) {
 310     return Handle();
 311   }
 312   bool has_multibyte, is_latin1;
 313   int length = UTF8::unicode_length(utf8_str, is_latin1, has_multibyte);
 314   if (!CompactStrings) {
 315     has_multibyte = true;
 316     is_latin1 = false;
 317   }
 318 
 319   Handle h_obj = basic_create(length, is_latin1, CHECK_NH);
 320   if (length > 0) {
 321     if (!has_multibyte) {
 322       const jbyte* src = reinterpret_cast<const jbyte*>(utf8_str);
 323       ArrayAccess<>::arraycopy_from_native(src, value(h_obj()), typeArrayOopDesc::element_offset<jbyte>(0), length);
 324     } else if (is_latin1) {
 325       UTF8::convert_to_unicode(utf8_str, value(h_obj())->byte_at_addr(0), length);
 326     } else {
 327       UTF8::convert_to_unicode(utf8_str, value(h_obj())->char_at_addr(0), length);
 328     }
 329   }
 330 
 331 #ifdef ASSERT
 332   // This check is too strict when the input string is not a valid UTF8.
 333   // For example, it may be created with arbitrary content via jni_NewStringUTF.
 334   if (UTF8::is_legal_utf8((const unsigned char*)utf8_str, (int)strlen(utf8_str), false)) {
 335     ResourceMark rm;
 336     const char* expected = utf8_str;
 337     char* actual = as_utf8_string(h_obj());
 338     if (strcmp(expected, actual) != 0) {
 339       fatal("String conversion failure: %s --> %s", expected, actual);
 340     }
 341   }
 342 #endif
 343 
 344   return h_obj;
 345 }

 389 // Converts a C string to a Java String based on current encoding
 390 Handle java_lang_String::create_from_platform_dependent_str(const char* str, TRAPS) {
 391   assert(str != NULL, "bad arguments");
 392 
 393   typedef jstring (JNICALL *to_java_string_fn_t)(JNIEnv*, const char *);
 394   static to_java_string_fn_t _to_java_string_fn = NULL;
 395 
 396   if (_to_java_string_fn == NULL) {
 397     void *lib_handle = os::native_java_library();
 398     _to_java_string_fn = CAST_TO_FN_PTR(to_java_string_fn_t, os::dll_lookup(lib_handle, "JNU_NewStringPlatform"));
 399 #if defined(_WIN32) && !defined(_WIN64)
 400     if (_to_java_string_fn == NULL) {
 401       // On 32 bit Windows, also try __stdcall decorated name
 402       _to_java_string_fn = CAST_TO_FN_PTR(to_java_string_fn_t, os::dll_lookup(lib_handle, "_JNU_NewStringPlatform@8"));
 403     }
 404 #endif
 405     if (_to_java_string_fn == NULL) {
 406       fatal("JNU_NewStringPlatform missing");
 407     }
 408   }
 409 
 410   jstring js = NULL;
 411   {
 412     assert(THREAD->is_Java_thread(), "must be java thread");
 413     JavaThread* thread = (JavaThread*)THREAD;
 414     HandleMark hm(thread);
 415     ThreadToNativeFromVM ttn(thread);
 416     js = (_to_java_string_fn)(thread->jni_environment(), str);
 417   }
 418 
 419   Handle native_platform_string(THREAD, JNIHandles::resolve(js));
 420   JNIHandles::destroy_local(js);  // destroy local JNIHandle.
 421   return native_platform_string;
 422 }

 424 // Converts a Java String to a native C string that can be used for
 425 // native OS calls.
 426 char* java_lang_String::as_platform_dependent_str(Handle java_string, TRAPS) {
 427   typedef char* (*to_platform_string_fn_t)(JNIEnv*, jstring, bool*);
 428   static to_platform_string_fn_t _to_platform_string_fn = NULL;
 429 
 430   if (_to_platform_string_fn == NULL) {
 431     void *lib_handle = os::native_java_library();
 432     _to_platform_string_fn = CAST_TO_FN_PTR(to_platform_string_fn_t, os::dll_lookup(lib_handle, "GetStringPlatformChars"));
 433     if (_to_platform_string_fn == NULL) {
 434       fatal("GetStringPlatformChars missing");
 435     }
 436   }
 437 
 438   char *native_platform_string;
 439   { JavaThread* thread = (JavaThread*)THREAD;
 440     assert(thread->is_Java_thread(), "must be java thread");
 441     JNIEnv *env = thread->jni_environment();
 442     jstring js = (jstring) JNIHandles::make_local(env, java_string());
 443     bool is_copy;
 444     HandleMark hm(thread);
 445     ThreadToNativeFromVM ttn(thread);
 446     native_platform_string = (_to_platform_string_fn)(env, js, &is_copy);
 447     assert(is_copy == JNI_TRUE, "is_copy value changed");
 448     JNIHandles::destroy_local(js);
 449   }
 450   return native_platform_string;
 451 }

 453 Handle java_lang_String::char_converter(Handle java_string, jchar from_char, jchar to_char, TRAPS) {
 454   oop          obj    = java_string();
 455   // Typical usage is to convert all '/' to '.' in string.
 456   typeArrayOop value  = java_lang_String::value(obj);
 457   int          length = java_lang_String::length(obj, value);
 458   bool      is_latin1 = java_lang_String::is_latin1(obj);
 459 
 460   // First check if any from_char exist
 461   int index; // Declared outside, used later
 462   for (index = 0; index < length; index++) {
 463     jchar c = !is_latin1 ? value->char_at(index) :
 464                   ((jchar) value->byte_at(index)) & 0xff;
 465     if (c == from_char) {
 466       break;
 467     }
 468   }
 469   if (index == length) {
 470     // No from_char, so do not copy.
 471     return java_string;
 472   }
 473 
 474   // Check if result string will be latin1
 475   bool to_is_latin1 = false;
 476 
 477   // Replacement char must be latin1
 478   if (CompactStrings && UNICODE::is_latin1(to_char)) {
 479     if (is_latin1) {
 480       // Source string is latin1 as well
 481       to_is_latin1 = true;
 482     } else if (!UNICODE::is_latin1(from_char)) {
 483       // We are replacing an UTF16 char. Scan string to
 484       // check if result can be latin1 encoded.
 485       to_is_latin1 = true;
 486       for (index = 0; index < length; index++) {
 487         jchar c = value->char_at(index);
 488         if (c != from_char && !UNICODE::is_latin1(c)) {
 489           to_is_latin1 = false;
 490           break;
 491         }
 492       }
 493     }
 494   }
 495 
 496   // Create new UNICODE (or byte) buffer. Must handlize value because GC
 497   // may happen during String and char array creation.
 498   typeArrayHandle h_value(THREAD, value);
 499   Handle string = basic_create(length, to_is_latin1, CHECK_NH);
 500   typeArrayOop from_buffer = h_value();
 501   typeArrayOop to_buffer = java_lang_String::value(string());
 502 
 503   // Copy contents
 504   for (index = 0; index < length; index++) {
 505     jchar c = (!is_latin1) ? from_buffer->char_at(index) :
 506                     ((jchar) from_buffer->byte_at(index)) & 0xff;
 507     if (c == from_char) {
 508       c = to_char;
 509     }
 510     if (!to_is_latin1) {
 511       to_buffer->char_at_put(index, c);
 512     } else {
 513       to_buffer->byte_at_put(index, (jbyte) c);
 514     }
 515   }
 516   return string;
 517 }

 519 jchar* java_lang_String::as_unicode_string(oop java_string, int& length, TRAPS) {
 520   typeArrayOop value  = java_lang_String::value(java_string);
 521                length = java_lang_String::length(java_string, value);
 522   bool      is_latin1 = java_lang_String::is_latin1(java_string);
 523 
 524   jchar* result = NEW_RESOURCE_ARRAY_RETURN_NULL(jchar, length);
 525   if (result != NULL) {
 526     if (!is_latin1) {
 527       for (int index = 0; index < length; index++) {
 528         result[index] = value->char_at(index);
 529       }
 530     } else {
 531       for (int index = 0; index < length; index++) {
 532         result[index] = ((jchar) value->byte_at(index)) & 0xff;
 533       }
 534     }
 535   } else {
 536     THROW_MSG_0(vmSymbols::java_lang_OutOfMemoryError(), "could not allocate Unicode string");
 537   }
 538   return result;
 539 }

 541 unsigned int java_lang_String::hash_code(oop java_string) {
 542   // The hash and hashIsZero fields are subject to a benign data race,
 543   // making it crucial to ensure that any observable result of the
 544   // calculation in this method stays correct under any possible read of
 545   // these fields. Necessary restrictions to allow this to be correct
 546   // without explicit memory fences or similar concurrency primitives is
 547   // that we can ever only write to one of these two fields for a given
 548   // String instance, and that the computation is idempotent and derived
 549   // from immutable state
 550   assert(_initialized && (_hash_offset > 0) && (_hashIsZero_offset > 0), "Must be initialized");
 551   if (java_lang_String::hash_is_set(java_string)) {
 552     return java_string->int_field(_hash_offset);
 553   }
 554 
 555   typeArrayOop value = java_lang_String::value(java_string);
 556   int         length = java_lang_String::length(java_string, value);
 557   bool     is_latin1 = java_lang_String::is_latin1(java_string);
 558 
 559   unsigned int hash = 0;
 560   if (length > 0) {
 561     if (is_latin1) {
 562       hash = java_lang_String::hash_code(value->byte_at_addr(0), length);
 563     } else {
 564       hash = java_lang_String::hash_code(value->char_at_addr(0), length);
 565     }
 566   }
 567 
 568   if (hash != 0) {
 569     java_string->int_field_put(_hash_offset, hash);
 570   } else {

 836         mirror()->char_field_put(fd->offset(), fd->int_initial_value());

 839         mirror()->short_field_put(fd->offset(), fd->int_initial_value());

 844       case T_FLOAT:
 845         mirror()->float_field_put(fd->offset(), fd->float_initial_value());

 848         mirror()->double_field_put(fd->offset(), fd->double_initial_value());

 876 void java_lang_Class::fixup_mirror(Klass* k, TRAPS) {
 877   assert(InstanceMirrorKlass::offset_of_static_fields() != 0, "must have been computed already");
 878 
 879   // If the offset was read from the shared archive, it was fixed up already
 880   if (!k->is_shared()) {
 881     if (k->is_instance_klass()) {
 882       // During bootstrap, java.lang.Class wasn't loaded so static field
 883       // offsets were computed without the size added it.  Go back and
 884       // update all the static field offsets to included the size.
 885       for (JavaFieldStream fs(InstanceKlass::cast(k)); !fs.done(); fs.next()) {
 886         if (fs.access_flags().is_static()) {
 887           int real_offset = fs.offset() + InstanceMirrorKlass::offset_of_static_fields();
 888           fs.set_offset(real_offset);
 889         }
 890       }
 891     }
 892   }
 893 
 894   if (k->is_shared() && k->has_raw_archived_mirror()) {
 895     if (HeapShared::open_archive_heap_region_mapped()) {
 896       bool present = restore_archived_mirror(k, Handle(), Handle(), Handle(), CHECK);
 897       assert(present, "Missing archived mirror for %s", k->external_name());
 898       return;
 899     } else {
 900       k->set_java_mirror_handle(OopHandle());
 901       k->clear_has_raw_archived_mirror();
 902     }
 903   }
 904   create_mirror(k, Handle(), Handle(), Handle(), Handle(), CHECK);
 905 }

 932     // Put the class on the fixup_module_list to patch later when the java.lang.Module

 960            (ModuleEntryTable::javabase_defined() &&

 978 void java_lang_Class::create_mirror(Klass* k, Handle class_loader,
 979                                     Handle module, Handle protection_domain,
 980                                     Handle classData, TRAPS) {
 981   assert(k != NULL, "Use create_basic_type_mirror for primitive types");
 982   assert(k->java_mirror() == NULL, "should only assign mirror once");
 983 
 984   // Use this moment of initialization to cache modifier_flags also,
 985   // to support Class.getModifiers().  Instance classes recalculate
 986   // the cached flags after the class file is parsed, but before the
 987   // class is put into the system dictionary.
 988   int computed_modifiers = k->compute_modifier_flags(CHECK);
 989   k->set_modifier_flags(computed_modifiers);
 990   // Class_klass has to be loaded because it is used to allocate
 991   // the mirror.
 992   if (SystemDictionary::Class_klass_loaded()) {
 993     // Allocate mirror (java.lang.Class instance)
 994     oop mirror_oop = InstanceMirrorKlass::cast(SystemDictionary::Class_klass())->allocate_instance(k, CHECK);
 995     Handle mirror(THREAD, mirror_oop);
 996     Handle comp_mirror;
 997 
 998     // Setup indirection from mirror->klass
 999     java_lang_Class::set_klass(mirror(), k);
1000 
1001     InstanceMirrorKlass* mk = InstanceMirrorKlass::cast(mirror->klass());
1002     assert(oop_size(mirror()) == mk->instance_size(k), "should have been set");
1003 
1004     java_lang_Class::set_static_oop_field_count(mirror(), mk->compute_static_oop_field_count(mirror()));
1005 
1006     // It might also have a component mirror.  This mirror must already exist.
1007     if (k->is_array_klass()) {
1008       if (k->is_flatArray_klass()) {
1009         Klass* element_klass = (Klass*) FlatArrayKlass::cast(k)->element_klass();
1010         assert(element_klass->is_inline_klass(), "Must be inline type component");
1011         InlineKlass* vk = InlineKlass::cast(InstanceKlass::cast(element_klass));
1012         comp_mirror = Handle(THREAD, vk->java_mirror());
1013       } else if (k->is_typeArray_klass()) {
1014         BasicType type = TypeArrayKlass::cast(k)->element_type();
1015         comp_mirror = Handle(THREAD, Universe::java_mirror(type));
1016       } else {
1017         assert(k->is_objArray_klass(), "Must be");
1018         Klass* element_klass = ObjArrayKlass::cast(k)->element_klass();
1019         assert(element_klass != NULL, "Must have an element klass");
1020         comp_mirror = Handle(THREAD, element_klass->java_mirror());
1021       }
1022       assert(comp_mirror() != NULL, "must have a mirror");
1023 
1024       // Two-way link between the array klass and its component mirror:
1025       // (array_klass) k -> mirror -> component_mirror -> array_klass -> k
1026       set_component_mirror(mirror(), comp_mirror());
1027       // See below for ordering dependencies between field array_klass in component mirror
1028       // and java_mirror in this klass.
1029     } else {
1030       assert(k->is_instance_klass(), "Must be");
1031 
1032       initialize_mirror_fields(k, mirror, protection_domain, classData, THREAD);
1033       if (HAS_PENDING_EXCEPTION) {
1034         // If any of the fields throws an exception like OOM remove the klass field
1035         // from the mirror so GC doesn't follow it after the klass has been deallocated.
1036         // This mirror looks like a primitive type, which logically it is because it
1037         // it represents no class.
1038         java_lang_Class::set_klass(mirror(), NULL);
1039         return;
1040       }
1041     }
1042 
1043     // set the classLoader field in the java_lang_Class instance
1044     assert(class_loader() == k->class_loader(), "should be same");
1045     set_class_loader(mirror(), class_loader());
1046 
1047     // Setup indirection from klass->mirror
1048     // after any exceptions can happen during allocations.
1049     k->set_java_mirror(mirror);
1050 
1051     // Set the module field in the java_lang_Class instance.  This must be done
1052     // after the mirror is set.
1053     set_mirror_module_field(k, mirror, module, THREAD);
1054 
1055     if (comp_mirror() != NULL) {
1056       // Set after k->java_mirror() is published, because compiled code running
1057       // concurrently doesn't expect a k to have a null java_mirror.
1058       release_set_array_klass(comp_mirror(), k);
1059     }
1060   } else {
1061     assert(fixup_mirror_list() != NULL, "fixup_mirror_list not initialized");
1062     fixup_mirror_list()->push(k);
1063   }
1064 }

1180 //
1181 // After the mirror object is successfully archived, the archived
1182 // klass is set with _has_archived_raw_mirror flag.
1183 //
1184 // The _has_archived_raw_mirror flag is cleared at runtime when the
1185 // archived mirror is restored. If archived java heap data cannot
1186 // be used at runtime, new mirror object is created for the shared
1187 // class. The _has_archived_raw_mirror is cleared also during the process.
1188 oop java_lang_Class::archive_mirror(Klass* k, TRAPS) {
1189   assert(HeapShared::is_heap_object_archiving_allowed(),
1190          "HeapShared::is_heap_object_archiving_allowed() must be true");
1191 
1192   // Mirror is already archived
1193   if (k->has_raw_archived_mirror()) {
1194     assert(k->archived_java_mirror_raw() != NULL, "no archived mirror");
1195     return k->archived_java_mirror_raw();
1196   }
1197 
1198   // No mirror
1199   oop mirror = k->java_mirror();
1200   if (mirror == NULL) {
1201     return NULL;
1202   }
1203 
1204   if (k->is_instance_klass()) {
1205     InstanceKlass *ik = InstanceKlass::cast(k);
1206     assert(ik->signers() == NULL, "class with signer should have been excluded");
1207 
1208     if (!(ik->is_shared_boot_class() || ik->is_shared_platform_class() ||
1209           ik->is_shared_app_class())) {
1210       // Archiving mirror for classes from non-builtin loaders is not
1211       // supported. Clear the _java_mirror within the archived class.
1212       k->set_java_mirror_handle(OopHandle());
1213       return NULL;
1214     }
1215   }
1216 
1217   if (k->is_inline_klass()) {
1218     // Inline types have a val type mirror and a ref type mirror. Don't handle this for now. TODO:CDS
1219     k->set_java_mirror_handle(OopHandle());
1220     return NULL;
1221   }
1222 
1223   // Now start archiving the mirror object
1224   oop archived_mirror = HeapShared::archive_heap_object(mirror, THREAD);
1225   if (archived_mirror == NULL) {
1226     return NULL;
1227   }
1228 
1229   archived_mirror = process_archived_mirror(k, mirror, archived_mirror, THREAD);
1230   if (archived_mirror == NULL) {
1231     return NULL;
1232   }
1233 
1234   k->set_archived_java_mirror_raw(archived_mirror);
1235 
1236   k->set_has_raw_archived_mirror();
1237 
1238   ResourceMark rm;
1239   log_trace(cds, heap, mirror)(
1240     "Archived %s mirror object from " PTR_FORMAT " ==> " PTR_FORMAT,
1241     k->external_name(), p2i(mirror), p2i(archived_mirror));
1242 
1243   return archived_mirror;
1244 }

1246 // The process is based on create_mirror().
1247 oop java_lang_Class::process_archived_mirror(Klass* k, oop mirror,
1248                                              oop archived_mirror,
1249                                              Thread *THREAD) {
1250   // Clear nonstatic fields in archived mirror. Some of the fields will be set
1251   // to archived metadata and objects below.
1252   Klass *c = archived_mirror->klass();
1253   Handle archived_mirror_h(THREAD, archived_mirror);
1254   ResetMirrorField reset(archived_mirror_h);
1255   InstanceKlass::cast(c)->do_nonstatic_fields(&reset);
1256 
1257   if (k->is_array_klass()) {
1258     oop archived_comp_mirror;
1259     if (k->is_typeArray_klass()) {
1260       // The primitive type mirrors are already archived. Get the archived mirror.
1261       oop comp_mirror = java_lang_Class::component_mirror(mirror);
1262       archived_comp_mirror = HeapShared::find_archived_heap_object(comp_mirror);
1263       assert(archived_comp_mirror != NULL, "Must be");
1264     } else {
1265       assert(k->is_objArray_klass(), "Must be");
1266       Klass* element_klass = ObjArrayKlass::cast(k)->element_klass();
1267       assert(element_klass != NULL, "Must have an element klass");
1268       archived_comp_mirror = archive_mirror(element_klass, THREAD);
1269       if (archived_comp_mirror == NULL) {
1270         return NULL;
1271       }
1272     }
1273     java_lang_Class::set_component_mirror(archived_mirror, archived_comp_mirror);
1274   } else {
1275     assert(k->is_instance_klass(), "Must be");
1276 
1277     // Reset local static fields in the mirror
1278     InstanceKlass::cast(k)->do_local_static_fields(&reset);
1279 
1280     java_lang_Class:set_init_lock(archived_mirror, NULL);
1281 
1282     set_protection_domain(archived_mirror, NULL);
1283   }
1284 
1285   // clear class loader and mirror_module_field
1286   set_class_loader(archived_mirror, NULL);
1287   set_module(archived_mirror, NULL);
1288 
1289   // The archived mirror's field at _klass_offset is still pointing to the original
1290   // klass. Updated the field in the archived mirror to point to the relocated
1291   // klass in the archive.
1292   Klass *reloc_k = MetaspaceShared::get_relocated_klass(as_Klass(mirror), true);
1293   log_debug(cds, heap, mirror)(
1294     "Relocate mirror metadata field at _klass_offset from " PTR_FORMAT " ==> " PTR_FORMAT,
1295     p2i(as_Klass(mirror)), p2i(reloc_k));
1296   archived_mirror->metadata_field_put(_klass_offset, reloc_k);
1297 
1298   // The field at _array_klass_offset is pointing to the original one dimension
1299   // higher array klass if exists. Relocate the pointer.
1300   Klass *arr = array_klass_acquire(mirror);
1301   if (arr != NULL) {
1302     Klass *reloc_arr = MetaspaceShared::get_relocated_klass(arr, true);
1303     log_debug(cds, heap, mirror)(
1304       "Relocate mirror metadata field at _array_klass_offset from " PTR_FORMAT " ==> " PTR_FORMAT,
1305       p2i(arr), p2i(reloc_arr));
1306     archived_mirror->metadata_field_put(_array_klass_offset, reloc_arr);
1307   }
1308   return archived_mirror;
1309 }

2449   if (!StackTraceInThrowable) return;

2453   // runs out of memory while allocating the stack trace

2455   // Clear lazily constructed Java level stacktrace if refilling occurs
2456   // This is unnecessary in 1.7+ but harmless
2457   clear_stacktrace(throwable());

2460   JavaThread* thread = (JavaThread*)THREAD;
2461 
2462   BacktraceBuilder bt(CHECK);
2463 
2464   // If there is no Java frame just return the method that was being called

2467     if (max_depth >= 1 && method() != NULL) {
2468       bt.push(method(), 0, CHECK);
2469       log_info(stacktrace)("%s, %d", throwable->klass()->external_name(), 1);
2470       set_depth(throwable(), 1);
2471       set_backtrace(throwable(), bt.backtrace());
2472     }
2473     return;
2474   }
2475 
2476   // Instead of using vframe directly, this version of fill_in_stack_trace
2477   // basically handles everything by hand. This significantly improved the
2478   // speed of this method call up to 28.5% on Solaris sparc. 27.1% on Windows.
2479   // See bug 6333838 for  more details.
2480   // The "ASSERT" here is to verify this method generates the exactly same stack
2481   // trace as utilizing vframe.
2482 #ifdef ASSERT
2483   vframeStream st(thread);
2484 #endif
2485   int total_count = 0;
2486   RegisterMap map(thread, false);
2487   int decode_offset = 0;
2488   CompiledMethod* nm = NULL;
2489   bool skip_fillInStackTrace_check = false;
2490   bool skip_throwableInit_check = false;
2491   bool skip_hidden = !ShowHiddenFrames;
2492 
2493   for (frame fr = thread->last_frame(); max_depth == 0 || max_depth != total_count;) {
2494     Method* method = NULL;
2495     int bci = 0;
2496 
2497     // Compiled java method case.
2498     if (decode_offset != 0) {
2499       DebugInfoReadStream stream(nm, decode_offset);
2500       decode_offset = stream.read_int();
2501       method = (Method*)nm->metadata_at(stream.read_int());
2502       bci = stream.read_bci();
2503     } else {
2504       if (fr.is_first_frame()) break;
2505       address pc = fr.pc();
2506       if (fr.is_interpreted_frame()) {
2507         address bcp = fr.interpreter_frame_bcp();
2508         method = fr.interpreter_frame_method();

2510         fr = fr.sender(&map);
2511       } else {

2514         // but non nmethod

2517           continue;
2518         }

2522           bci = 0;
2523         } else {

2525           decode_offset = pd->scope_decode_offset();
2526           // if decode_offset is not equal to 0, it will execute the
2527           // "compiled java method case" at the beginning of the loop.
2528           continue;
2529         }
2530       }
2531     }

2534            "Wrong stack trace");

2536 #endif
2537 
2538     // the format of the stacktrace will be:
2539     // - 1 or more fillInStackTrace frames for the exception class (skipped)
2540     // - 0 or more <init> methods for the exception class (skipped)
2541     // - rest of the stack
2542 
2543     if (!skip_fillInStackTrace_check) {

2546         continue;
2547       }
2548       else {
2549         skip_fillInStackTrace_check = true; // gone past them all

2551     }

2553       assert(skip_fillInStackTrace_check, "logic error in backtrace filtering");
2554 
2555       // skip <init> methods of the exception class and superclasses
2556       // This is similar to classic VM (before HotSpot).
2557       if (method->is_object_constructor() &&
2558           throwable->is_a(method->method_holder())) {
2559         continue;
2560       } else {
2561         // there are none or we've seen them all - either way stop checking
2562         skip_throwableInit_check = true;
2563       }
2564     }
2565     if (method->is_hidden()) {
2566       if (skip_hidden) {
2567         if (total_count == 0) {
2568           // The top frame will be hidden from the stack trace.
2569           bt.set_has_hidden_top_frame(CHECK);
2570         }
2571         continue;
2572       }
2573     }
2574     bt.push(method, bci, CHECK);
2575     total_count++;
2576   }
2577 
2578   log_info(stacktrace)("%s, %d", throwable->klass()->external_name(), total_count);
2579 
2580   // Put completed stack trace into throwable object
2581   set_backtrace(throwable(), bt.backtrace());

2583 }

2792                                                        const methodHandle& method,
2793                                                        int bci,
2794                                                        Symbol*& source,
2795                                                        oop& source_file,
2796                                                        int& line_number, TRAPS) {

2798   source = Backtrace::get_source_file_name(holder, version);

2800   if (source != NULL) {
2801     // Class was not redefined. We can trust its cache if set,

2804       source_file = StringTable::intern(source, CHECK);

2808     // Class was redefined. Dump the cache if it was set.
2809     if (source_file != NULL) {
2810       source_file = NULL;
2811       java_lang_Class::set_source_file(java_class(), source_file);
2812     }
2813   }

2815 }
