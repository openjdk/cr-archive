 34 // Clones the inline type to handle control flow merges involving multiple inline types.
 35 // The inputs are replaced by PhiNodes to represent the merged values for the given region.
 36 InlineTypeBaseNode* InlineTypeBaseNode::clone_with_phis(PhaseGVN* gvn, Node* region) {
 37   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
 38 
 39   // Create a PhiNode for merging the oop values
 40   const Type* phi_type = Type::get_const_type(inline_klass());
 41   PhiNode* oop = PhiNode::make(region, vt->get_oop(), phi_type);
 42   gvn->set_type(oop, phi_type);
 43   vt->set_oop(oop);
 44 
 45   // Create a PhiNode each for merging the field values
 46   for (uint i = 0; i < vt->field_count(); ++i) {
 47     ciType* type = vt->field_type(i);
 48     Node*  value = vt->field_value(i);
 49     if (value->is_InlineTypeBase()) {
 50       // Handle flattened inline type fields recursively
 51       value = value->as_InlineTypeBase()->clone_with_phis(gvn, region);
 52     } else {
 53       phi_type = Type::get_const_type(type);
 54       value = PhiNode::make(region, value, phi_type);
 55       gvn->set_type(value, phi_type);
 56     }
 57     vt->set_field_value(i, value);
 58   }
 59   gvn->set_type(vt, vt->bottom_type());
 60   return vt;
 61 }

 63 // Checks if the inputs of the InlineTypeBaseTypeNode were replaced by PhiNodes
 64 // for the given region (see InlineTypeBaseTypeNode::clone_with_phis).
 65 bool InlineTypeBaseNode::has_phi_inputs(Node* region) {
 66   // Check oop input
 67   bool result = get_oop()->is_Phi() && get_oop()->as_Phi()->region() == region;
 68 #ifdef ASSERT
 69   if (result) {
 70     // Check all field value inputs for consistency
 71     for (uint i = Oop; i < field_count(); ++i) {
 72       Node* n = in(i);
 73       if (n->is_InlineTypeBase()) {
 74         assert(n->as_InlineTypeBase()->has_phi_inputs(region), "inconsistent phi inputs");
 75       } else {
 76         assert(n->is_Phi() && n->as_Phi()->region() == region, "inconsistent phi inputs");
 77       }
 78     }
 79   }
 80 #endif
 81   return result;
 82 }

 84 // Merges 'this' with 'other' by updating the input PhiNodes added by 'clone_with_phis'
 85 InlineTypeBaseNode* InlineTypeBaseNode::merge_with(PhaseGVN* gvn, const InlineTypeBaseNode* other, int pnum, bool transform) {
 86   // Merge oop inputs
 87   PhiNode* phi = get_oop()->as_Phi();
 88   phi->set_req(pnum, other->get_oop());
 89   if (transform) {
 90     set_oop(gvn->transform(phi));
 91     gvn->record_for_igvn(phi);
 92   }
 93   // Merge field values
 94   for (uint i = 0; i < field_count(); ++i) {
 95     Node* val1 =        field_value(i);
 96     Node* val2 = other->field_value(i);
 97     if (val1->is_InlineTypeBase()) {
 98       val1->as_InlineTypeBase()->merge_with(gvn, val2->as_InlineTypeBase(), pnum, transform);
 99     } else {
100       assert(val1->is_Phi(), "must be a phi node");
101       val1->set_req(pnum, val2);
102     }
103     if (transform) {
104       set_field_value(i, gvn->transform(val1));
105       gvn->record_for_igvn(val1);
106     }
107   }
108   return this;
109 }

111 // Adds a new merge path to an inline type node with phi inputs
112 void InlineTypeBaseNode::add_new_path(Node* region) {
113   assert(has_phi_inputs(region), "must have phi inputs");
114 
115   PhiNode* phi = get_oop()->as_Phi();
116   phi->add_req(NULL);
117   assert(phi->req() == region->req(), "must be same size as region");
118 
119   for (uint i = 0; i < field_count(); ++i) {
120     Node* val = field_value(i);
121     if (val->is_InlineType()) {
122       val->as_InlineType()->add_new_path(region);
123     } else {
124       val->as_Phi()->add_req(NULL);
125       assert(val->req() == region->req(), "must be same size as region");
126     }
127   }
128 }

135 // Get the value of the field at the given offset.
136 // If 'recursive' is true, flattened inline type fields will be resolved recursively.
137 Node* InlineTypeBaseNode::field_value_by_offset(int offset, bool recursive) const {
138   // If the field at 'offset' belongs to a flattened inline type field, 'index' refers to the
139   // corresponding InlineTypeNode input and 'sub_offset' is the offset in flattened inline type.
140   int index = inline_klass()->field_index_by_offset(offset);
141   int sub_offset = offset - field_offset(index);
142   Node* value = field_value(index);
143   assert(value != NULL, "field value not found");
144   if (recursive && value->is_InlineType()) {
145     InlineTypeNode* vt = value->as_InlineType();
146     if (field_is_flattened(index)) {
147       // Flattened inline type field
148       sub_offset += vt->inline_klass()->first_field_offset(); // Add header size
149       return vt->field_value_by_offset(sub_offset, recursive);
150     } else {
151       assert(sub_offset == 0, "should not have a sub offset");
152       return vt;
153     }
154   }
155   assert(!(recursive && value->is_InlineType()), "should not be an inline type");
156   assert(sub_offset == 0, "offset mismatch");
157   return value;
158 }

193 int InlineTypeBaseNode::make_scalar_in_safepoint(PhaseIterGVN* igvn, Unique_Node_List& worklist, SafePointNode* sfpt) {
194   ciInlineKlass* vk = inline_klass();
195   uint nfields = vk->nof_nonstatic_fields();
196   JVMState* jvms = sfpt->jvms();
197   int start = jvms->debug_start();
198   int end   = jvms->debug_end();
199   // Replace safepoint edge by SafePointScalarObjectNode and add field values
200   assert(jvms != NULL, "missing JVMS");
201   uint first_ind = (sfpt->req() - jvms->scloff());
202   SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(inline_ptr(),
203 #ifdef ASSERT
204                                                                   NULL,
205 #endif
206                                                                   first_ind, nfields);
207   sobj->init_req(0, igvn->C->root());
208   // Iterate over the inline type fields in order of increasing
209   // offset and add the field values to the safepoint.
210   for (uint j = 0; j < nfields; ++j) {
211     int offset = vk->nonstatic_field_at(j)->offset();
212     Node* value = field_value_by_offset(offset, true /* include flattened inline type fields */);
213     if (value->is_InlineType()) {
214       // Add inline type field to the worklist to process later
215       worklist.push(value);
216     }
217     sfpt->add_req(value);
218   }
219   jvms->set_endoff(sfpt->req());
220   sobj = igvn->transform(sobj)->as_SafePointScalarObject();
221   igvn->rehash_node_delayed(sfpt);
222   return sfpt->replace_edges_in_range(this, sobj, start, end);
223 }

225 void InlineTypeBaseNode::make_scalar_in_safepoints(PhaseIterGVN* igvn) {
226   // Process all safepoint uses and scalarize inline type
227   Unique_Node_List worklist;
228   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
229     SafePointNode* sfpt = fast_out(i)->isa_SafePoint();
230     if (sfpt != NULL && !sfpt->is_CallLeaf() && (!sfpt->is_Call() || sfpt->as_Call()->has_debug_use(this))) {
231       int nb = 0;
232       if (is_allocated(igvn) && get_oop()->is_Con()) {
233         // Inline type is allocated with a constant oop, link it directly
234         nb = sfpt->replace_edges_in_range(this, get_oop(), sfpt->jvms()->debug_start(), sfpt->jvms()->debug_end());
235         igvn->rehash_node_delayed(sfpt);
236       } else {
237         nb = make_scalar_in_safepoint(igvn, worklist, sfpt);
238       }
239       --i; imax -= nb;
240     }
241   }
242   // Now scalarize non-flattened fields
243   for (uint i = 0; i < worklist.size(); ++i) {
244     Node* vt = worklist.at(i);
245     vt->as_InlineType()->make_scalar_in_safepoints(igvn);
246   }
247   igvn->record_for_igvn(this);
248 }

250 const TypePtr* InlineTypeBaseNode::field_adr_type(Node* base, int offset, ciInstanceKlass* holder, DecoratorSet decorators, PhaseGVN& gvn) const {
251   const TypeAryPtr* ary_type = gvn.type(base)->isa_aryptr();
252   const TypePtr* adr_type = NULL;
253   bool is_array = ary_type != NULL;
254   if ((decorators & C2_MISMATCHED) != 0) {
255     adr_type = TypeRawPtr::BOTTOM;
256   } else if (is_array) {
257     // In the case of a flattened inline type array, each field has its own slice
258     adr_type = ary_type->with_field_offset(offset)->add_offset(Type::OffsetBot);
259   } else {
260     ciField* field = holder->get_field_by_offset(offset, false);
261     assert(field != NULL, "field not found");
262     adr_type = gvn.C->alias_type(field)->adr_type();
263   }
264   return adr_type;
265 }

267 void InlineTypeBaseNode::load(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
268   // Initialize the inline type by loading its field values from
269   // memory and adding the values as input edges to the node.
270   for (uint i = 0; i < field_count(); ++i) {
271     int offset = holder_offset + field_offset(i);
272     Node* value = NULL;
273     ciType* ft = field_type(i);
274     if (field_is_flattened(i)) {
275       // Recursively load the flattened inline type field
276       value = InlineTypeNode::make_from_flattened(kit, ft->as_inline_klass(), base, ptr, holder, offset, decorators);
277     } else {
278       const TypeOopPtr* oop_ptr = kit->gvn().type(base)->isa_oopptr();
279       bool is_array = (oop_ptr->isa_aryptr() != NULL);
280       if (base->is_Con() && !is_array) {
281         // If the oop to the inline type is constant (static final field), we can
282         // also treat the fields as constants because the inline type is immutable.
283         ciObject* constant_oop = oop_ptr->const_oop();
284         ciField* field = holder->get_field_by_offset(offset, false);
285         assert(field != NULL, "field not found");
286         ciConstant constant = constant_oop->as_instance()->field_value(field);
287         const Type* con_type = Type::make_from_constant(constant, /*require_const=*/ true);
288         assert(con_type != NULL, "type not found");
289         value = kit->gvn().transform(kit->makecon(con_type));
290         // Check type of constant which might be more precise than the static field type
291         if (con_type->is_inlinetypeptr()) {
292           assert(!con_type->is_zero_type(), "Inline types are null-free");
293           ft = con_type->inline_klass();
294         }
295       } else {
296         // Load field value from memory
297         const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
298         Node* adr = kit->basic_plus_adr(base, ptr, offset);
299         BasicType bt = type2field[ft->basic_type()];
300         assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
301         const Type* val_type = Type::get_const_type(ft);
302         if (is_array) {
303           decorators |= IS_ARRAY;
304         }
305         value = kit->access_load_at(base, adr, adr_type, val_type, bt, decorators);
306       }
307       if (ft->is_inlinetype()) {
308         // Loading a non-flattened inline type from memory
309         if (ft->as_inline_klass()->is_scalarizable()) {
310           value = InlineTypeNode::make_from_oop(kit, value, ft->as_inline_klass());
311         } else {
312           value = kit->null2default(value, ft->as_inline_klass());
313         }
314       }
315     }
316     set_field_value(i, value);
317   }
318 }

320 void InlineTypeBaseNode::store_flattened(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
321   // The inline type is embedded into the object without an oop header. Subtract the
322   // offset of the first field to account for the missing header when storing the values.
323   if (holder == NULL) {
324     holder = inline_klass();
325   }
326   holder_offset -= inline_klass()->first_field_offset();
327   store(kit, base, ptr, holder, holder_offset, decorators);
328 }

330 void InlineTypeBaseNode::store(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
331   // Write field values to memory
332   for (uint i = 0; i < field_count(); ++i) {
333     int offset = holder_offset + field_offset(i);
334     Node* value = field_value(i);
335     ciType* ft = field_type(i);
336     if (field_is_flattened(i)) {
337       // Recursively store the flattened inline type field
338       if (!value->is_InlineType()) {
339         assert(!kit->gvn().type(value)->maybe_null(), "Inline types are null-free");
340         value = InlineTypeNode::make_from_oop(kit, value, ft->as_inline_klass());
341       }
342       value->as_InlineType()->store_flattened(kit, base, ptr, holder, offset, decorators);
343     } else {
344       // Store field value to memory
345       const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
346       Node* adr = kit->basic_plus_adr(base, ptr, offset);
347       BasicType bt = type2field[ft->basic_type()];
348       assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
349       const Type* val_type = Type::get_const_type(ft);
350       const TypeAryPtr* ary_type = kit->gvn().type(base)->isa_aryptr();
351       if (ary_type != NULL) {
352         decorators |= IS_ARRAY;
353       }
354       kit->access_store_at(base, adr, adr_type, value, val_type, bt, decorators);
355     }
356   }
357 }

359 InlineTypePtrNode* InlineTypeBaseNode::buffer(GraphKit* kit, bool safe_for_replace) {
360   assert(is_InlineType(), "sanity");
361   // Check if inline type is already allocated
362   Node* null_ctl = kit->top();
363   Node* not_null_oop = kit->null_check_oop(get_oop(), &null_ctl);
364   if (null_ctl->is_top()) {
365     // Inline type is allocated
366     return as_ptr(&kit->gvn());
367   }
368   assert(!is_allocated(&kit->gvn()), "should not be allocated");
369   RegionNode* region = new RegionNode(3);
370 
371   // Oop is non-NULL, use it
372   region->init_req(1, kit->control());
373   PhiNode* oop = PhiNode::make(region, not_null_oop, inline_ptr());
374   PhiNode* io  = PhiNode::make(region, kit->i_o(), Type::ABIO);
375   PhiNode* mem = PhiNode::make(region, kit->merged_memory(), Type::MEMORY, TypePtr::BOTTOM);
376 
377   int bci = kit->bci();
378   bool reexecute = kit->jvms()->should_reexecute();
379   {
380     // Oop is NULL, allocate and initialize buffer
381     PreserveJVMState pjvms(kit);
382     // Propagate re-execution state and bci
383     kit->set_bci(bci);
384     kit->jvms()->set_bci(bci);
385     kit->jvms()->set_should_reexecute(reexecute);
386     kit->set_control(null_ctl);
387     kit->kill_dead_locals();
388     ciInlineKlass* vk = inline_klass();
389     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
390     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true, this);
391     store(kit, alloc_oop, alloc_oop, vk, 0);
392 
393     // Do not let stores that initialize this buffer be reordered with a subsequent
394     // store that would make this buffer accessible by other threads.
395     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
396     assert(alloc != NULL, "must have an allocation node");
397     kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
398 
399     region->init_req(2, kit->control());
400     oop   ->init_req(2, alloc_oop);
401     io    ->init_req(2, kit->i_o());
402     mem   ->init_req(2, kit->merged_memory());
403   }
404 
405   // Update GraphKit
406   kit->set_control(kit->gvn().transform(region));
407   kit->set_i_o(kit->gvn().transform(io));
408   kit->set_all_memory(kit->gvn().transform(mem));
409   kit->record_for_igvn(region);
410   kit->record_for_igvn(oop);
411   kit->record_for_igvn(io);
412   kit->record_for_igvn(mem);
413 
414   // Use cloned InlineTypeNode to propagate oop from now on
415   Node* res_oop = kit->gvn().transform(oop);
416   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
417   vt->set_oop(res_oop);
418   vt = kit->gvn().transform(vt)->as_InlineTypeBase();
419   if (safe_for_replace) {
420     kit->replace_in_map(this, vt);
421   }
422   // InlineTypeNode::remove_redundant_allocations piggybacks on split if.
423   // Make sure it gets a chance to remove this allocation.
424   kit->C->set_has_split_ifs(true);
425   return vt->as_ptr(&kit->gvn());
426 }

428 bool InlineTypeBaseNode::is_allocated(PhaseGVN* phase) const {
429   Node* oop = get_oop();
430   const Type* oop_type = (phase != NULL) ? phase->type(oop) : oop->bottom_type();
431   return !oop_type->maybe_null();
432 }

434 InlineTypePtrNode* InlineTypeBaseNode::as_ptr(PhaseGVN* phase) const {
435   assert(is_allocated(phase), "must be allocated");
436   if (is_InlineTypePtr()) {
437     return as_InlineTypePtr();
438   }
439   return phase->transform(new InlineTypePtrNode(this))->as_InlineTypePtr();

442 // When a call returns multiple values, it has several result
443 // projections, one per field. Replacing the result of the call by a
444 // inline type node (after late inlining) requires that for each result
445 // projection, we find the corresponding inline type field.
446 void InlineTypeBaseNode::replace_call_results(GraphKit* kit, Node* call, Compile* C) {
447   ciInlineKlass* vk = inline_klass();
448   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
449     ProjNode* pn = call->fast_out(i)->as_Proj();
450     uint con = pn->_con;
451     if (con >= TypeFunc::Parms+1) {
452       uint field_nb = con - (TypeFunc::Parms+1);
453       int extra = 0;
454       for (uint j = 0; j < field_nb - extra; j++) {
455         ciField* f = vk->nonstatic_field_at(j);
456         BasicType bt = f->type()->basic_type();
457         if (bt == T_LONG || bt == T_DOUBLE) {
458           extra++;
459         }
460       }
461       ciField* f = vk->nonstatic_field_at(field_nb - extra);
462       Node* field = field_value_by_offset(f->offset(), true);
463       if (field->is_InlineType()) {
464         assert(field->as_InlineType()->is_allocated(&kit->gvn()), "must be allocated");
465         field = field->as_InlineType()->get_oop();
466       }
467       C->gvn_replace_by(pn, field);
468       C->initial_gvn()->hash_delete(pn);
469       pn->set_req(0, C->top());
470       --i; --imax;
471     }
472   }
473 }

475 Node* InlineTypeBaseNode::allocate_fields(GraphKit* kit) {
476   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
477   for (uint i = 0; i < field_count(); i++) {
478      InlineTypeNode* value = field_value(i)->isa_InlineType();
479      if (field_is_flattened(i)) {
480        // Flattened inline type field
481        vt->set_field_value(i, value->allocate_fields(kit));
482      } else if (value != NULL) {
483        // Non-flattened inline type field
484        vt->set_field_value(i, value->buffer(kit));
485      }
486   }
487   vt = kit->gvn().transform(vt)->as_InlineTypeBase();
488   kit->replace_in_map(this, vt);
489   return vt;
490 }

492 InlineTypeNode* InlineTypeNode::make_uninitialized(PhaseGVN& gvn, ciInlineKlass* vk) {
493   // Create a new InlineTypeNode with uninitialized values and NULL oop

498 Node* InlineTypeNode::default_oop(PhaseGVN& gvn, ciInlineKlass* vk) {
499   // Returns the constant oop of the default inline type allocation
500   return gvn.makecon(TypeInstPtr::make(vk->default_instance()));

503 InlineTypeNode* InlineTypeNode::make_default(PhaseGVN& gvn, ciInlineKlass* vk) {
504   // Create a new InlineTypeNode with default values
505   InlineTypeNode* vt = new InlineTypeNode(vk, default_oop(gvn, vk));
506   for (uint i = 0; i < vt->field_count(); ++i) {
507     ciType* field_type = vt->field_type(i);
508     Node* value = NULL;
509     if (field_type->is_inlinetype()) {
510       ciInlineKlass* field_klass = field_type->as_inline_klass();
511       if (field_klass->is_scalarizable()) {
512         value = InlineTypeNode::make_default(gvn, field_klass);
513       } else {
514         value = default_oop(gvn, field_klass);
515       }
516     } else {
517       value = gvn.zerocon(field_type->basic_type());
518     }
519     vt->set_field_value(i, value);
520   }
521   vt = gvn.transform(vt)->as_InlineType();
522   assert(vt->is_default(&gvn), "must be the default inline type");
523   return vt;
524 }
525 
526 bool InlineTypeNode::is_default(PhaseGVN* gvn) const {
527   for (uint i = 0; i < field_count(); ++i) {
528     Node* value = field_value(i);
529     if (!gvn->type(value)->is_zero_type() &&
530         !(value->is_InlineType() && value->as_InlineType()->is_default(gvn)) &&
531         !(field_type(i)->is_inlinetype() && value == default_oop(*gvn, field_type(i)->as_inline_klass()))) {
532       return false;
533     }
534   }
535   return true;
536 }

538 InlineTypeNode* InlineTypeNode::make_from_oop(GraphKit* kit, Node* oop, ciInlineKlass* vk) {
539   PhaseGVN& gvn = kit->gvn();
540 
541   // Create and initialize an InlineTypeNode by loading all field
542   // values from a heap-allocated version and also save the oop.
543   InlineTypeNode* vt = new InlineTypeNode(vk, oop);
544 
545   if (oop->isa_InlineTypePtr()) {
546     // Can happen with late inlining
547     InlineTypePtrNode* vtptr = oop->as_InlineTypePtr();
548     vt->set_oop(vtptr->get_oop());
549     for (uint i = Oop+1; i < vtptr->req(); ++i) {
550       vt->init_req(i, vtptr->in(i));
551     }
552   } else if (gvn.type(oop)->maybe_null()) {
553     // Add a null check because the oop may be null
554     Node* null_ctl = kit->top();
555     Node* not_null_oop = kit->null_check_oop(oop, &null_ctl);
556     if (kit->stopped()) {
557       // Constant null
558       kit->set_control(null_ctl);
559       return make_default(gvn, vk);
560     }
561     vt->set_oop(not_null_oop);
562     vt->load(kit, not_null_oop, not_null_oop, vk, /* holder_offset */ 0);
563 
564     if (null_ctl != kit->top()) {
565       // Return default inline type if oop is null
566       InlineTypeNode* def = make_default(gvn, vk);
567       Node* region = new RegionNode(3);
568       region->init_req(1, kit->control());
569       region->init_req(2, null_ctl);
570 
571       vt = vt->clone_with_phis(&gvn, region)->as_InlineType();
572       vt->merge_with(&gvn, def, 2, true);
573       kit->set_control(gvn.transform(region));
574     }
575   } else {
576     // Oop can never be null
577     Node* init_ctl = kit->control();
578     vt->load(kit, oop, oop, vk, /* holder_offset */ 0);
579     assert(init_ctl != kit->control() || !gvn.type(oop)->is_inlinetypeptr() || oop->is_Con() || oop->Opcode() == Op_InlineTypePtr ||
580            AllocateNode::Ideal_allocation(oop, &gvn) != NULL || vt->is_loaded(&gvn) == oop, "inline type should be loaded");
581   }
582 
583   assert(vt->is_allocated(&gvn), "inline type should be allocated");
584   return gvn.transform(vt)->as_InlineType();
585 }

587 // GraphKit wrapper for the 'make_from_flattened' method
588 InlineTypeNode* InlineTypeNode::make_from_flattened(GraphKit* kit, ciInlineKlass* vk, Node* obj, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
589   // Create and initialize an InlineTypeNode by loading all field values from
590   // a flattened inline type field at 'holder_offset' or from an inline type array.
591   InlineTypeNode* vt = make_uninitialized(kit->gvn(), vk);
592   // The inline type is flattened into the object without an oop header. Subtract the
593   // offset of the first field to account for the missing header when loading the values.
594   holder_offset -= vk->first_field_offset();
595   vt->load(kit, obj, ptr, holder, holder_offset, decorators);
596   assert(vt->is_loaded(&kit->gvn()) != obj, "holder oop should not be used as flattened inline type oop");
597   return kit->gvn().transform(vt)->as_InlineType();
598 }

606 InlineTypeNode* InlineTypeNode::make_larval(GraphKit* kit, bool allocate) const {
607   ciInlineKlass* vk = inline_klass();
608   InlineTypeNode* res = clone()->as_InlineType();
609   if (allocate) {
610     // Re-execute if buffering triggers deoptimization
611     PreserveReexecuteState preexecs(kit);
612     kit->jvms()->set_should_reexecute(true);
613     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
614     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, true);
615     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
616     alloc->_larval = true;
617 
618     store(kit, alloc_oop, alloc_oop, vk, 0);
619     res->set_oop(alloc_oop);
620   }
621   res->set_type(TypeInlineType::make(vk, true));
622   res = kit->gvn().transform(res)->as_InlineType();
623   assert(!allocate || res->is_allocated(&kit->gvn()), "must be allocated");
624   return res;
625 }

627 InlineTypeNode* InlineTypeNode::finish_larval(GraphKit* kit) const {
628   Node* obj = get_oop();
629   Node* mark_addr = kit->basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
630   Node* mark = kit->make_load(NULL, mark_addr, TypeX_X, TypeX_X->basic_type(), MemNode::unordered);
631   mark = kit->gvn().transform(new AndXNode(mark, kit->MakeConX(~markWord::larval_mask_in_place)));
632   kit->store_to_memory(kit->control(), mark_addr, mark, TypeX_X->basic_type(), kit->gvn().type(mark_addr)->is_ptr(), MemNode::unordered);
633 
634   // Do not let stores that initialize this buffer be reordered with a subsequent
635   // store that would make this buffer accessible by other threads.
636   AllocateNode* alloc = AllocateNode::Ideal_allocation(obj, &kit->gvn());
637   assert(alloc != NULL, "must have an allocation node");
638   kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
639 
640   ciInlineKlass* vk = inline_klass();
641   InlineTypeNode* res = clone()->as_InlineType();
642   res->set_type(TypeInlineType::make(vk, false));
643   res = kit->gvn().transform(res)->as_InlineType();
644   return res;
645 }

647 Node* InlineTypeNode::is_loaded(PhaseGVN* phase, ciInlineKlass* vk, Node* base, int holder_offset) {
648   if (vk == NULL) {
649     vk = inline_klass();
650   }
651   if (field_count() == 0) {
652     assert(is_allocated(phase), "must be allocated");
653     return get_oop();
654   }
655   for (uint i = 0; i < field_count(); ++i) {
656     int offset = holder_offset + field_offset(i);
657     Node* value = field_value(i);
658     if (value->is_InlineType()) {
659       InlineTypeNode* vt = value->as_InlineType();
660       if (field_is_flattened(i)) {
661         // Check inline type field load recursively
662         base = vt->is_loaded(phase, vk, base, offset - vt->inline_klass()->first_field_offset());
663         if (base == NULL) {
664           return NULL;
665         }
666         continue;
667       } else {
668         value = vt->get_oop();
669         if (value->Opcode() == Op_CastPP) {
670           // Skip CastPP
671           value = value->in(1);
672         }
673       }
674     }
675     if (value->isa_DecodeN()) {
676       // Skip DecodeN
677       value = value->in(1);
678     }
679     if (value->isa_Load()) {
680       // Check if base and offset of field load matches inline type layout
681       intptr_t loffset = 0;
682       Node* lbase = AddPNode::Ideal_base_and_offset(value->in(MemNode::Address), phase, loffset);
683       if (lbase == NULL || (lbase != base && base != NULL) || loffset != offset) {
684         return NULL;
685       } else if (base == NULL) {
686         // Set base and check if pointer type matches
687         base = lbase;
688         const TypeInstPtr* vtptr = phase->type(base)->isa_instptr();
689         if (vtptr == NULL || !vtptr->klass()->equals(vk)) {
690           return NULL;
691         }
692       }
693     } else {
694       return NULL;
695     }
696   }
697   return base;
698 }

700 Node* InlineTypeNode::tagged_klass(ciInlineKlass* vk, PhaseGVN& gvn) {
701   const TypeKlassPtr* tk = TypeKlassPtr::make(vk);

707 void InlineTypeNode::pass_fields(GraphKit* kit, Node* n, ExtendedSignature& sig, uint& base_input, int base_offset) {
708   for (uint i = 0; i < field_count(); i++) {
709     int sig_offset = (*sig)._offset;
710     uint idx = field_index(sig_offset - base_offset);
711     Node* arg = field_value(idx);
712 
713     if (field_is_flattened(idx)) {
714       // Flattened inline type field
715       InlineTypeNode* vt = arg->as_InlineType();
716       vt->pass_fields(kit, n, sig, base_input, sig_offset - vt->inline_klass()->first_field_offset());
717     } else {
718       if (arg->is_InlineType()) {
719         // Non-flattened inline type field
720         InlineTypeNode* vt = arg->as_InlineType();
721         assert(n->Opcode() != Op_Return || vt->is_allocated(&kit->gvn()), "inline type field should be allocated on return");
722         arg = vt->buffer(kit);
723       }
724       // Initialize call/return arguments
725       BasicType bt = field_type(i)->basic_type();
726       n->init_req(base_input++, arg);
727       if (type2size[bt] == 2) {
728         n->init_req(base_input++, kit->top());
729       }
730       // Skip reserved arguments
731       while (SigEntry::next_is_reserved(sig, bt)) {
732         n->init_req(base_input++, kit->top());
733         if (type2size[bt] == 2) {
734           n->init_req(base_input++, kit->top());
735         }
736       }
737     }
738   }
739 }

741 void InlineTypeNode::initialize_fields(GraphKit* kit, MultiNode* multi, ExtendedSignature& sig, uint& base_input, int base_offset, bool in) {
742   PhaseGVN& gvn = kit->gvn();
743   for (uint i = 0; i < field_count(); i++) {
744     int sig_offset = (*sig)._offset;
745     uint idx = field_index(sig_offset - base_offset);
746     ciType* type = field_type(idx);
747 
748     Node* parm = NULL;
749     if (field_is_flattened(idx)) {
750       // Flattened inline type field
751       InlineTypeNode* vt = InlineTypeNode::make_uninitialized(gvn, type->as_inline_klass());
752       vt->initialize_fields(kit, multi, sig, base_input, sig_offset - type->as_inline_klass()->first_field_offset(), in);
753       parm = gvn.transform(vt);
754     } else {
755       if (multi->is_Start()) {
756         assert(in, "return from start?");
757         parm = gvn.transform(new ParmNode(multi->as_Start(), base_input));
758       } else if (in) {
759         parm = multi->as_Call()->in(base_input);
760       } else {
761         parm = gvn.transform(new ProjNode(multi->as_Call(), base_input));
762       }
763       if (type->is_inlinetype()) {
764         // Non-flattened inline type field
765         if (type->as_inline_klass()->is_scalarizable()) {
766           parm = InlineTypeNode::make_from_oop(kit, parm, type->as_inline_klass());
767         } else {
768           parm = kit->null2default(parm, type->as_inline_klass());
769         }
770       }
771       base_input += type2size[type->basic_type()];
772       // Skip reserved arguments
773       BasicType bt = type->basic_type();
774       while (SigEntry::next_is_reserved(sig, bt)) {
775         base_input += type2size[bt];
776       }
777     }
778     assert(parm != NULL, "should never be null");
779     set_field_value(idx, parm);
780     gvn.record_for_igvn(parm);
781   }
782 }

784 // Replace a buffer allocation by a dominating allocation
785 static void replace_allocation(PhaseIterGVN* igvn, Node* res, Node* dom) {
786   // Remove initializing stores
787   for (DUIterator_Fast imax, i = res->fast_outs(imax); i < imax; i++) {
788     AddPNode* addp = res->fast_out(i)->isa_AddP();
789     if (addp != NULL) {
790       for (DUIterator_Fast jmax, j = addp->fast_outs(jmax); j < jmax; j++) {
791         StoreNode* store = addp->fast_out(j)->isa_Store();
792         if (store != NULL) {
793           igvn->replace_in_uses(store, store->in(MemNode::Memory));
794         }
795       }
796     }
797   }
798   igvn->replace_node(res, dom);
799 }

801 Node* InlineTypeNode::Ideal(PhaseGVN* phase, bool can_reshape) {
802   Node* oop = get_oop();
803   if (is_default(phase) && (!oop->is_Con() || phase->type(oop)->is_zero_type())) {
804     // Use the pre-allocated oop for default inline types
805     set_oop(default_oop(*phase, inline_klass()));
806     assert(is_allocated(phase), "should now be allocated");
807     return this;
808   } else if (oop->isa_InlineTypePtr()) {
809     // Can happen with late inlining
810     InlineTypePtrNode* vtptr = oop->as_InlineTypePtr();
811     set_oop(vtptr->get_oop());
812     for (uint i = Oop+1; i < vtptr->req(); ++i) {
813       set_req(i, vtptr->in(i));
814     }
815     return this;
816   }
817 
818   if (!is_allocated(phase)) {
819     // Save base oop if fields are loaded from memory and the inline
820     // type is not buffered (in this case we should not use the oop).
821     Node* base = is_loaded(phase);
822     if (base != NULL) {
823       set_oop(base);
824       assert(is_allocated(phase), "should now be allocated");
825       return this;
826     }
827   }
828 
829   if (can_reshape) {
830     PhaseIterGVN* igvn = phase->is_IterGVN();
831 
832     if (is_allocated(phase)) {
833       // Search for and remove re-allocations of this inline type. Ignore scalar replaceable ones,
834       // they will be removed anyway and changing the memory chain will confuse other optimizations.
835       // This can happen with late inlining when we first allocate an inline type argument
836       // but later decide to inline the call after the callee code also triggered allocation.
837       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
838         AllocateNode* alloc = fast_out(i)->isa_Allocate();
839         if (alloc != NULL && alloc->in(AllocateNode::InlineTypeNode) == this && !alloc->_is_scalar_replaceable) {
840           // Found a re-allocation
841           Node* res = alloc->result_cast();
842           if (res != NULL && res->is_CheckCastPP()) {
843             // Replace allocation by oop and unlink AllocateNode
844             replace_allocation(igvn, res, get_oop());
845             igvn->replace_input_of(alloc, AllocateNode::InlineTypeNode, igvn->C->top());
846             --i; --imax;
847           }
848         }
849       }
850     }
851   }
852   return NULL;
853 }

855 // Search for multiple allocations of this inline type and try to replace them by dominating allocations.
856 // Then unlink the inline type node and remove it.
857 void InlineTypeNode::remove_redundant_allocations(PhaseIterGVN* igvn, PhaseIdealLoop* phase) {
858   // Search for allocations of this inline type. Ignore scalar replaceable ones, they
859   // will be removed anyway and changing the memory chain will confuse other optimizations.
860   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
861     AllocateNode* alloc = fast_out(i)->isa_Allocate();
862     if (alloc != NULL && alloc->in(AllocateNode::InlineTypeNode) == this && !alloc->_is_scalar_replaceable) {
863       Node* res = alloc->result_cast();
864       if (res == NULL || !res->is_CheckCastPP()) {
865         break; // No unique CheckCastPP
866       }
867       assert(!is_default(igvn) && !is_allocated(igvn), "re-allocation should be removed by Ideal transformation");
868       // Search for a dominating allocation of the same inline type
869       Node* res_dom = res;
870       for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
871         AllocateNode* alloc_other = fast_out(j)->isa_Allocate();
872         if (alloc_other != NULL && alloc_other->in(AllocateNode::InlineTypeNode) == this && !alloc_other->_is_scalar_replaceable) {
873           Node* res_other = alloc_other->result_cast();
874           if (res_other != NULL && res_other->is_CheckCastPP() && res_other != res_dom &&
875               phase->is_dominator(res_other->in(0), res_dom->in(0))) {
876             res_dom = res_other;
877           }
878         }
879       }
880       if (res_dom != res) {
881         // Replace allocation by dominating one.
882         replace_allocation(igvn, res, res_dom);
883         // The result of the dominated allocation is now unused and will be removed
884         // later in PhaseMacroExpand::eliminate_allocate_node to not confuse loop opts.
885         igvn->record_for_igvn(alloc);
886       }
887     }
888   }
889 
890   // Process users
891   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
892     Node* out = fast_out(i);
893     if (out->is_InlineType()) {
894       // Unlink and recursively process inline type users
895       igvn->hash_delete(out);
896       int nb = out->replace_edge(this, igvn->C->top());
897       out->as_InlineType()->remove_redundant_allocations(igvn, phase);
898       --i; imax -= nb;
899     } else if (out->isa_Allocate() != NULL) {
900       // Unlink AllocateNode
901       assert(out->in(AllocateNode::InlineTypeNode) == this, "should be linked");
902       igvn->replace_input_of(out, AllocateNode::InlineTypeNode, igvn->C->top());
903       --i; --imax;
904     } else {
905 #ifdef ASSERT
906       // The inline type should not have any other users at this time
907       out->dump();
908       assert(false, "unexpected user of inline type");
909 #endif
910     }
911   }
912   igvn->remove_dead_node(this);
913 }
