 133 void BlockListBuilder::set_entries(int osr_bci) {
 134   // generate start blocks
 135   BlockBegin* std_entry = make_block_at(0, NULL);
 136   if (scope()->caller() == NULL) {
 137     std_entry->set(BlockBegin::std_entry_flag);
 138   }
 139   if (osr_bci != -1) {
 140     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 141     osr_entry->set(BlockBegin::osr_entry_flag);
 142   }
 143 
 144   // generate exception entry blocks
 145   XHandlers* list = xhandlers();
 146   const int n = list->length();
 147   for (int i = 0; i < n; i++) {
 148     XHandler* h = list->handler_at(i);
 149     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 150     entry->set(BlockBegin::exception_entry_flag);
 151     h->set_entry_block(entry);
 152   }
 153 }

 156 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 157   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 158 
 159   BlockBegin* block = _bci2block->at(cur_bci);
 160   if (block == NULL) {
 161     block = new BlockBegin(cur_bci);
 162     block->init_stores_to_locals(method()->max_locals());
 163     _bci2block->at_put(cur_bci, block);
 164     _blocks.append(block);
 165 
 166     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 167   }
 168 
 169   if (predecessor != NULL) {
 170     if (block->is_set(BlockBegin::exception_entry_flag)) {
 171       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 172     }
 173 
 174     predecessor->add_successor(block);
 175     block->increment_total_preds();
 176   }
 177 
 178   return block;
 179 }

 182 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 183   current->stores_to_locals().set_bit(local);

 185 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 186   store_one(current, local);
 187   store_one(current, local + 1);

 191 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 192   // Draws edges from a block to its exception handlers
 193   XHandlers* list = xhandlers();
 194   const int n = list->length();
 195 
 196   for (int i = 0; i < n; i++) {
 197     XHandler* h = list->handler_at(i);
 198 
 199     if (h->covers(cur_bci)) {
 200       BlockBegin* entry = h->entry_block();
 201       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 202       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 203 
 204       // add each exception handler only once
 205       if (!current->is_successor(entry)) {
 206         current->add_successor(entry);
 207         entry->increment_total_preds();
 208       }
 209 
 210       // stop when reaching catchall
 211       if (h->catch_type() == 0) break;
 212     }
 213   }
 214 }

 216 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 217   // start a new block after jsr-bytecode and link this block into cfg
 218   make_block_at(next_bci, current);
 219 
 220   // start a new block at the subroutine entry at mark it with special flag
 221   BlockBegin* sr_block = make_block_at(sr_bci, current);
 222   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 223     sr_block->set(BlockBegin::subroutine_entry_flag);
 224   }
 225 }

 228 void BlockListBuilder::set_leaders() {
 229   bool has_xhandlers = xhandlers()->has_handlers();
 230   BlockBegin* current = NULL;
 231 
 232   // The information which bci starts a new block simplifies the analysis
 233   // Without it, backward branches could jump to a bci where no block was created
 234   // during bytecode iteration. This would require the creation of a new block at the
 235   // branch target and a modification of the successor lists.
 236   const BitMap& bci_block_start = method()->bci_block_start();
 237 
 238   ciBytecodeStream s(method());
 239   while (s.next() != ciBytecodeStream::EOBC()) {
 240     int cur_bci = s.cur_bci();
 241 
 242     if (bci_block_start.at(cur_bci)) {
 243       current = make_block_at(cur_bci, current);
 244     }
 245     assert(current != NULL, "must have current block");
 246 
 247     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 248       handle_exceptions(current, cur_bci);
 249     }
 250 
 251     switch (s.cur_bc()) {
 252       // track stores to local variables for selective creation of phi functions
 253       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 254       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 255       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 256       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 257       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 258       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 259       case Bytecodes::_istore_0: store_one(current, 0); break;
 260       case Bytecodes::_istore_1: store_one(current, 1); break;
 261       case Bytecodes::_istore_2: store_one(current, 2); break;
 262       case Bytecodes::_istore_3: store_one(current, 3); break;
 263       case Bytecodes::_lstore_0: store_two(current, 0); break;
 264       case Bytecodes::_lstore_1: store_two(current, 1); break;
 265       case Bytecodes::_lstore_2: store_two(current, 2); break;
 266       case Bytecodes::_lstore_3: store_two(current, 3); break;
 267       case Bytecodes::_fstore_0: store_one(current, 0); break;
 268       case Bytecodes::_fstore_1: store_one(current, 1); break;
 269       case Bytecodes::_fstore_2: store_one(current, 2); break;
 270       case Bytecodes::_fstore_3: store_one(current, 3); break;
 271       case Bytecodes::_dstore_0: store_two(current, 0); break;
 272       case Bytecodes::_dstore_1: store_two(current, 1); break;
 273       case Bytecodes::_dstore_2: store_two(current, 2); break;
 274       case Bytecodes::_dstore_3: store_two(current, 3); break;
 275       case Bytecodes::_astore_0: store_one(current, 0); break;
 276       case Bytecodes::_astore_1: store_one(current, 1); break;
 277       case Bytecodes::_astore_2: store_one(current, 2); break;
 278       case Bytecodes::_astore_3: store_one(current, 3); break;
 279 
 280       // track bytecodes that affect the control flow
 281       case Bytecodes::_athrow:  // fall through
 282       case Bytecodes::_ret:     // fall through
 283       case Bytecodes::_ireturn: // fall through
 284       case Bytecodes::_lreturn: // fall through
 285       case Bytecodes::_freturn: // fall through
 286       case Bytecodes::_dreturn: // fall through
 287       case Bytecodes::_areturn: // fall through
 288       case Bytecodes::_return:
 289         current = NULL;
 290         break;
 291 
 292       case Bytecodes::_ifeq:      // fall through
 293       case Bytecodes::_ifne:      // fall through
 294       case Bytecodes::_iflt:      // fall through
 295       case Bytecodes::_ifge:      // fall through
 296       case Bytecodes::_ifgt:      // fall through
 297       case Bytecodes::_ifle:      // fall through
 298       case Bytecodes::_if_icmpeq: // fall through
 299       case Bytecodes::_if_icmpne: // fall through
 300       case Bytecodes::_if_icmplt: // fall through
 301       case Bytecodes::_if_icmpge: // fall through
 302       case Bytecodes::_if_icmpgt: // fall through
 303       case Bytecodes::_if_icmple: // fall through
 304       case Bytecodes::_if_acmpeq: // fall through
 305       case Bytecodes::_if_acmpne: // fall through
 306       case Bytecodes::_ifnull:    // fall through
 307       case Bytecodes::_ifnonnull:
 308         make_block_at(s.next_bci(), current);
 309         make_block_at(s.get_dest(), current);
 310         current = NULL;
 311         break;
 312 
 313       case Bytecodes::_goto:
 314         make_block_at(s.get_dest(), current);
 315         current = NULL;
 316         break;
 317 
 318       case Bytecodes::_goto_w:
 319         make_block_at(s.get_far_dest(), current);
 320         current = NULL;
 321         break;
 322 
 323       case Bytecodes::_jsr:
 324         handle_jsr(current, s.get_dest(), s.next_bci());
 325         current = NULL;
 326         break;
 327 
 328       case Bytecodes::_jsr_w:
 329         handle_jsr(current, s.get_far_dest(), s.next_bci());
 330         current = NULL;
 331         break;
 332 
 333       case Bytecodes::_tableswitch: {
 334         // set block for each case
 335         Bytecode_tableswitch sw(&s);
 336         int l = sw.length();
 337         for (int i = 0; i < l; i++) {
 338           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 339         }
 340         make_block_at(cur_bci + sw.default_offset(), current);
 341         current = NULL;
 342         break;
 343       }
 344 
 345       case Bytecodes::_lookupswitch: {
 346         // set block for each case
 347         Bytecode_lookupswitch sw(&s);
 348         int l = sw.number_of_pairs();
 349         for (int i = 0; i < l; i++) {
 350           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 351         }
 352         make_block_at(cur_bci + sw.default_offset(), current);
 353         current = NULL;
 354         break;
 355       }
 356 
 357       default:
 358         break;
 359     }
 360   }
 361 }

 364 void BlockListBuilder::mark_loops() {
 365   ResourceMark rm;
 366 
 367   _active.initialize(BlockBegin::number_of_blocks());
 368   _visited.initialize(BlockBegin::number_of_blocks());
 369   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 370   _next_loop_index = 0;
 371   _next_block_number = _blocks.length();
 372 
 373   // recursively iterate the control flow graph
 374   mark_loops(_bci2block->at(0), false);
 375   assert(_next_block_number >= 0, "invalid block numbers");
 376 
 377   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 378   _active.resize(0);
 379   _visited.resize(0);
 380 }

 382 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 383   if (block->is_set(BlockBegin::exception_entry_flag)) {
 384     // exception edges may look like loops but don't mark them as such
 385     // since it screws up block ordering.
 386     return;
 387   }
 388   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 389     block->set(BlockBegin::parser_loop_header_flag);
 390 
 391     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 392     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 393     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 394     if (_next_loop_index < 31) _next_loop_index++;
 395   } else {
 396     // block already marked as loop header
 397     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 398   }
 399 }

 401 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 402   int block_id = block->block_id();
 403 
 404   if (_visited.at(block_id)) {
 405     if (_active.at(block_id)) {
 406       // reached block via backward branch
 407       make_loop_header(block);
 408     }
 409     // return cached loop information for this block
 410     return _loop_map.at(block_id);
 411   }
 412 
 413   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 414     in_subroutine = true;
 415   }
 416 
 417   // set active and visited bits before successors are processed
 418   _visited.set_bit(block_id);
 419   _active.set_bit(block_id);
 420 
 421   intptr_t loop_state = 0;
 422   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 423     // recursively process all successors
 424     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 425   }
 426 
 427   // clear active-bit after all successors are processed
 428   _active.clear_bit(block_id);
 429 
 430   // reverse-post-order numbering of all blocks
 431   block->set_depth_first_number(_next_block_number);
 432   _next_block_number--;
 433 
 434   if (loop_state != 0 || in_subroutine ) {
 435     // block is contained at least in one loop, so phi functions are necessary
 436     // phi functions are also necessary for all locals stored in a subroutine
 437     scope()->requires_phi_function().set_union(block->stores_to_locals());
 438   }
 439 
 440   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 441     int header_loop_state = _loop_map.at(block_id);
 442     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 443 
 444     // If the highest bit is set (i.e. when integer value is negative), the method
 445     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 446     if (header_loop_state >= 0) {
 447       clear_bits(loop_state, header_loop_state);
 448     }
 449   }
 450 
 451   // cache and return loop information for this block
 452   _loop_map.at_put(block_id, loop_state);
 453   return loop_state;
 454 }

 463 void BlockListBuilder::print() {
 464   tty->print("----- initial block list of BlockListBuilder for method ");
 465   method()->print_short_name();
 466   tty->cr();
 467 
 468   // better readability if blocks are sorted in processing order
 469   _blocks.sort(compare_depth_first);
 470 
 471   for (int i = 0; i < _blocks.length(); i++) {
 472     BlockBegin* cur = _blocks.at(i);
 473     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 474 
 475     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 476     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 477     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 478     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 479     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 480 
 481     if (cur->number_of_sux() > 0) {
 482       tty->print("    sux: ");
 483       for (int j = 0; j < cur->number_of_sux(); j++) {
 484         BlockBegin* sux = cur->sux_at(j);
 485         tty->print("B%d ", sux->block_id());
 486       }
 487     }
 488     tty->cr();
 489   }
 490 }

 495 // A simple growable array of Values indexed by ciFields
 496 class FieldBuffer: public CompilationResourceObj {
 497  private:
 498   GrowableArray<Value> _values;
 499 
 500  public:
 501   FieldBuffer() {}
 502 
 503   void kill() {
 504     _values.trunc_to(0);
 505   }
 506 
 507   Value at(ciField* field) {
 508     assert(field->holder()->is_loaded(), "must be a loaded field");
 509     int offset = field->offset();
 510     if (offset < _values.length()) {
 511       return _values.at(offset);
 512     } else {
 513       return NULL;
 514     }
 515   }
 516 
 517   void at_put(ciField* field, Value value) {
 518     assert(field->holder()->is_loaded(), "must be a loaded field");
 519     int offset = field->offset();
 520     _values.at_put_grow(offset, value, NULL);
 521   }
 522 
 523 };

 526 // MemoryBuffer is fairly simple model of the current state of memory.
 527 // It partitions memory into several pieces.  The first piece is
 528 // generic memory where little is known about the owner of the memory.
 529 // This is conceptually represented by the tuple <O, F, V> which says
 530 // that the field F of object O has value V.  This is flattened so
 531 // that F is represented by the offset of the field and the parallel
 532 // arrays _objects and _values are used for O and V.  Loads of O.F can
 533 // simply use V.  Newly allocated objects are kept in a separate list
 534 // along with a parallel array for each object which represents the
 535 // current value of its fields.  Stores of the default value to fields
 536 // which have never been stored to before are eliminated since they
 537 // are redundant.  Once newly allocated objects are stored into
 538 // another object or they are passed out of the current compile they
 539 // are treated like generic memory.
 540 
 541 class MemoryBuffer: public CompilationResourceObj {
 542  private:
 543   FieldBuffer                 _values;
 544   GrowableArray<Value>        _objects;
 545   GrowableArray<Value>        _newobjects;
 546   GrowableArray<FieldBuffer*> _fields;
 547 
 548  public:
 549   MemoryBuffer() {}
 550 
 551   StoreField* store(StoreField* st) {
 552     if (!EliminateFieldAccess) {
 553       return st;
 554     }
 555 
 556     Value object = st->obj();
 557     Value value = st->value();
 558     ciField* field = st->field();
 559     if (field->holder()->is_loaded()) {
 560       int offset = field->offset();
 561       int index = _newobjects.find(object);
 562       if (index != -1) {
 563         // newly allocated object with no other stores performed on this field
 564         FieldBuffer* buf = _fields.at(index);
 565         if (buf->at(field) == NULL && is_default_value(value)) {
 566 #ifndef PRODUCT
 567           if (PrintIRDuringConstruction && Verbose) {
 568             tty->print_cr("Eliminated store for object %d:", index);
 569             st->print_line();
 570           }
 571 #endif
 572           return NULL;
 573         } else {
 574           buf->at_put(field, value);
 575         }
 576       } else {
 577         _objects.at_put_grow(offset, object, NULL);
 578         _values.at_put(field, value);
 579       }
 580 
 581       store_value(value);
 582     } else {
 583       // if we held onto field names we could alias based on names but
 584       // we don't know what's being stored to so kill it all.
 585       kill();
 586     }
 587     return st;
 588   }
 589 
 590 
 591   // return true if this value correspond to the default value of a field.
 592   bool is_default_value(Value value) {
 593     Constant* con = value->as_Constant();
 594     if (con) {
 595       switch (con->type()->tag()) {
 596         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 597         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 598         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 599         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 600         case objectTag: return con->type() == objectNull;
 601         default:  ShouldNotReachHere();
 602       }
 603     }
 604     return false;
 605   }
 606 
 607 
 608   // return either the actual value of a load or the load itself
 609   Value load(LoadField* load) {
 610     if (!EliminateFieldAccess) {
 611       return load;
 612     }
 613 
 614     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 615 #ifdef IA32
 616       if (UseSSE < 2) {
 617         // can't skip load since value might get rounded as a side effect
 618         return load;
 619       }
 620 #else
 621       Unimplemented();
 622 #endif // IA32
 623     }
 624 
 625     ciField* field = load->field();
 626     Value object   = load->obj();
 627     if (field->holder()->is_loaded() && !field->is_volatile()) {
 628       int offset = field->offset();
 629       Value result = NULL;
 630       int index = _newobjects.find(object);
 631       if (index != -1) {
 632         result = _fields.at(index)->at(field);
 633       } else if (_objects.at_grow(offset, NULL) == object) {
 634         result = _values.at(field);
 635       }
 636       if (result != NULL) {
 637 #ifndef PRODUCT
 638         if (PrintIRDuringConstruction && Verbose) {
 639           tty->print_cr("Eliminated load: ");
 640           load->print_line();
 641         }
 642 #endif
 643         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 644         return result;
 645       }
 646     }
 647     return load;
 648   }
 649 
 650   // Record this newly allocated object
 651   void new_instance(NewInstance* object) {
 652     int index = _newobjects.length();
 653     _newobjects.append(object);
 654     if (_fields.at_grow(index, NULL) == NULL) {
 655       _fields.at_put(index, new FieldBuffer());
 656     } else {
 657       _fields.at(index)->kill();
 658     }
 659   }
 660 
 661   // Record this newly allocated object
 662   void new_instance(NewInlineTypeInstance* object) {
 663     int index = _newobjects.length();
 664     _newobjects.append(object);
 665     if (_fields.at_grow(index, NULL) == NULL) {
 666       _fields.at_put(index, new FieldBuffer());
 667     } else {
 668       _fields.at(index)->kill();
 669     }
 670   }
 671 
 672   void store_value(Value value) {
 673     int index = _newobjects.find(value);
 674     if (index != -1) {
 675       // stored a newly allocated object into another object.
 676       // Assume we've lost track of it as separate slice of memory.
 677       // We could do better by keeping track of whether individual
 678       // fields could alias each other.
 679       _newobjects.remove_at(index);
 680       // pull out the field info and store it at the end up the list
 681       // of field info list to be reused later.
 682       _fields.append(_fields.at(index));
 683       _fields.remove_at(index);
 684     }
 685   }
 686 
 687   void kill() {
 688     _newobjects.trunc_to(0);
 689     _objects.trunc_to(0);
 690     _values.kill();
 691   }
 692 };

 733 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 734   if (parsing_jsr()) {
 735     // It is necessary to clone all blocks associated with a
 736     // subroutine, including those for exception handlers in the scope
 737     // of the method containing the jsr (because those exception
 738     // handlers may contain ret instructions in some cases).
 739     BlockBegin* block = bci2block()->at(bci);
 740     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 741       BlockBegin* new_block = new BlockBegin(block->bci());
 742       if (PrintInitialBlockList) {
 743         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 744                       block->block_id(), block->bci(), new_block->block_id());
 745       }
 746       // copy data from cloned blocked
 747       new_block->set_depth_first_number(block->depth_first_number());
 748       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 749       // Preserve certain flags for assertion checking
 750       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 751       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 752 
 753       // copy was_visited_flag to allow early detection of bailouts
 754       // if a block that is used in a jsr has already been visited before,
 755       // it is shared between the normal control flow and a subroutine
 756       // BlockBegin::try_merge returns false when the flag is set, this leads
 757       // to a compilation bailout
 758       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 759 
 760       bci2block()->at_put(bci, new_block);
 761       block = new_block;
 762     }
 763     return block;
 764   } else {
 765     return bci2block()->at(bci);
 766   }

 824 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 825   assert(worklist->top() == top, "");
 826   // sort block descending into work list
 827   const int dfn = top->depth_first_number();
 828   assert(dfn != -1, "unknown depth first number");
 829   int i = worklist->length()-2;
 830   while (i >= 0) {
 831     BlockBegin* b = worklist->at(i);
 832     if (b->depth_first_number() < dfn) {
 833       worklist->at_put(i+1, b);
 834     } else {
 835       break;
 836     }
 837     i --;
 838   }
 839   if (i >= -1) worklist->at_put(i + 1, top);
 840 }

 897 void GraphBuilder::load_constant() {
 898   ciConstant con = stream()->get_constant();
 899   if (con.basic_type() == T_ILLEGAL) {
 900     // FIXME: an unresolved Dynamic constant can get here,
 901     // and that should not terminate the whole compilation.
 902     BAILOUT("could not resolve a constant");
 903   } else {
 904     ValueType* t = illegalType;
 905     ValueStack* patch_state = NULL;
 906     switch (con.basic_type()) {
 907       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 908       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 909       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 910       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 911       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 912       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 913       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 914       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 915       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 916       case T_OBJECT :
 917        {
 918         ciObject* obj = con.as_object();
 919         if (!obj->is_loaded()
 920             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 921           // A Class, MethodType, MethodHandle, or String.
 922           // Unloaded condy nodes show up as T_ILLEGAL, above.
 923           patch_state = copy_state_before();
 924           t = new ObjectConstant(obj);
 925         } else {
 926           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 927           // result, which might turn out to be an array.
 928           if (obj->is_null_object())
 929             t = objectNull;
 930           else if (obj->is_array())
 931             t = new ArrayConstant(obj->as_array());
 932           else
 933             t = new InstanceConstant(obj->as_instance());
 934         }
 935         break;
 936        }
 937       default       : ShouldNotReachHere();
 938     }
 939     Value x;
 940     if (patch_state != NULL) {
 941       x = new Constant(t, patch_state);
 942     } else {
 943       x = new Constant(t);
 944     }
 945     push(t, append(x));
 946   }
 947 }

 992 void GraphBuilder::load_indexed(BasicType type) {
 993   // In case of in block code motion in range check elimination
 994   ValueStack* state_before = NULL;
 995   int array_idx = state()->stack_size() - 2;
 996   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 997     // Save the entire state and re-execute on deopt when accessing flattened arrays
 998     state_before = copy_state_before();
 999     state_before->set_should_reexecute(true);
1000   } else {
1001     state_before = copy_state_indexed_access();
1002   }
1003   compilation()->set_has_access_indexed(true);
1004   Value index = ipop();
1005   Value array = apop();
1006   Value length = NULL;
1007   if (CSEArrayLength ||
1008       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1009       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1010     length = append(new ArrayLength(array, state_before));
1011   }
1012 
1013   LoadIndexed* load_indexed = NULL;
1014   Instruction* result = NULL;
1015   if (array->is_loaded_flattened_array()) {
1016     ciType* array_type = array->declared_type();
1017     ciInlineKlass* elem_klass = array_type->as_flat_array_klass()->element_klass()->as_inline_klass();
1018     NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(elem_klass, state_before, false);
1019     _memory->new_instance(new_instance);
1020     apush(append_split(new_instance));
1021     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1022     load_indexed->set_vt(new_instance);
1023   } else {
1024     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1025   }
1026   if (profile_array_accesses() && is_reference_type(type)) {
1027     compilation()->set_would_profile(true);
1028     load_indexed->set_should_profile(true);
1029     load_indexed->set_profiled_method(method());
1030     load_indexed->set_profiled_bci(bci());
1031   }
1032   result = append(load_indexed);
1033   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1034   if (!array->is_loaded_flattened_array()) {
1035     push(as_ValueType(type), result);
1036   }

1191 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1192   Value s = ipop();
1193   Value x = pop(type);
1194   // try to simplify
1195   // Note: This code should go into the canonicalizer as soon as it can
1196   //       can handle canonicalized forms that contain more than one node.
1197   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1198     // pattern: x >>> s
1199     IntConstant* s1 = s->type()->as_IntConstant();
1200     if (s1 != NULL) {
1201       // pattern: x >>> s1, with s1 constant
1202       ShiftOp* l = x->as_ShiftOp();
1203       if (l != NULL && l->op() == Bytecodes::_ishl) {
1204         // pattern: (a << b) >>> s1
1205         IntConstant* s0 = l->y()->type()->as_IntConstant();
1206         if (s0 != NULL) {
1207           // pattern: (a << s0) >>> s1
1208           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1209           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1210           if (s0c == s1c) {
1211             if (s0c == 0) {
1212               // pattern: (a << 0) >>> 0 => simplify to: a
1213               ipush(l->x());
1214             } else {
1215               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1216               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1217               const int m = (1 << (BitsPerInt - s0c)) - 1;
1218               Value s = append(new Constant(new IntConstant(m)));
1219               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1220             }
1221             return;
1222           }
1223         }
1224       }
1225     }
1226   }
1227   // could not simplify
1228   push(type, append(new ShiftOp(code, x, s)));
1229 }

1276 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1277   BlockBegin* tsux = block_at(stream()->get_dest());
1278   BlockBegin* fsux = block_at(stream()->next_bci());
1279   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1280 
1281   bool subst_check = false;
1282   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne)) {
1283     ValueType* left_vt = x->type();
1284     ValueType* right_vt = y->type();
1285     if (left_vt->is_object()) {
1286       assert(right_vt->is_object(), "must be");
1287       ciKlass* left_klass = x->as_loaded_klass_or_null();
1288       ciKlass* right_klass = y->as_loaded_klass_or_null();
1289 
1290       if (left_klass == NULL || right_klass == NULL) {
1291         // The klass is still unloaded, or came from a Phi node. Go slow case;
1292         subst_check = true;
1293       } else if (left_klass->can_be_inline_klass() || right_klass->can_be_inline_klass()) {
1294         // Either operand may be a value object, but we're not sure. Go slow case;
1295         subst_check = true;
1296       } else {
1297         // No need to do substitutability check
1298       }
1299     }
1300   }
1301 
1302   // In case of loop invariant code motion or predicate insertion
1303   // before the body of a loop the state is needed
1304   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1305 
1306   assert(i->as_Goto() == NULL ||
1307          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1308          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1309          "safepoint state of Goto returned by canonicalizer incorrect");
1310 
1311   if (is_profiling()) {
1312     If* if_node = i->as_If();
1313     if (if_node != NULL) {
1314       // Note that we'd collect profile data in this method if we wanted it.
1315       compilation()->set_would_profile(true);
1316       // At level 2 we need the proper bci to count backedges
1317       if_node->set_profiled_bci(bci());
1318       if (profile_branches()) {
1319         // Successors can be rotated by the canonicalizer, check for this case.
1320         if_node->set_profiled_method(method());
1321         if_node->set_should_profile(true);
1322         if (if_node->tsux() == fsux) {
1323           if_node->set_swapped(true);
1324         }
1325       }
1326       return;
1327     }
1328 
1329     // Check if this If was reduced to Goto.
1330     Goto *goto_node = i->as_Goto();
1331     if (goto_node != NULL) {
1332       compilation()->set_would_profile(true);
1333       goto_node->set_profiled_bci(bci());
1334       if (profile_branches()) {
1335         goto_node->set_profiled_method(method());
1336         goto_node->set_should_profile(true);
1337         // Find out which successor is used.
1338         if (goto_node->default_sux() == tsux) {
1339           goto_node->set_direction(Goto::taken);
1340         } else if (goto_node->default_sux() == fsux) {
1341           goto_node->set_direction(Goto::not_taken);
1342         } else {
1343           ShouldNotReachHere();
1344         }
1345       }
1346       return;
1347     }
1348   }
1349 }

1408 void GraphBuilder::table_switch() {
1409   Bytecode_tableswitch sw(stream());
1410   const int l = sw.length();
1411   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1412     // total of 2 successors => use If instead of switch
1413     // Note: This code should go into the canonicalizer as soon as it can
1414     //       can handle canonicalized forms that contain more than one node.
1415     Value key = append(new Constant(new IntConstant(sw.low_key())));
1416     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1417     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1418     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1419     // In case of loop invariant code motion or predicate insertion
1420     // before the body of a loop the state is needed
1421     ValueStack* state_before = copy_state_if_bb(is_bb);
1422     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1423   } else {
1424     // collect successors
1425     BlockList* sux = new BlockList(l + 1, NULL);
1426     int i;
1427     bool has_bb = false;
1428     for (i = 0; i < l; i++) {
1429       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1430       if (sw.dest_offset_at(i) < 0) has_bb = true;
1431     }
1432     // add default successor
1433     if (sw.default_offset() < 0) has_bb = true;
1434     sux->at_put(i, block_at(bci() + sw.default_offset()));
1435     // In case of loop invariant code motion or predicate insertion
1436     // before the body of a loop the state is needed
1437     ValueStack* state_before = copy_state_if_bb(has_bb);
1438     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1439 #ifdef ASSERT
1440     if (res->as_Goto()) {
1441       for (i = 0; i < l; i++) {
1442         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1443           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1444         }
1445       }
1446     }
1447 #endif
1448   }
1449 }

1452 void GraphBuilder::lookup_switch() {
1453   Bytecode_lookupswitch sw(stream());
1454   const int l = sw.number_of_pairs();
1455   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1456     // total of 2 successors => use If instead of switch
1457     // Note: This code should go into the canonicalizer as soon as it can
1458     //       can handle canonicalized forms that contain more than one node.
1459     // simplify to If
1460     LookupswitchPair pair = sw.pair_at(0);
1461     Value key = append(new Constant(new IntConstant(pair.match())));
1462     BlockBegin* tsux = block_at(bci() + pair.offset());
1463     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1464     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1465     // In case of loop invariant code motion or predicate insertion
1466     // before the body of a loop the state is needed
1467     ValueStack* state_before = copy_state_if_bb(is_bb);;
1468     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1469   } else {
1470     // collect successors & keys
1471     BlockList* sux = new BlockList(l + 1, NULL);
1472     intArray* keys = new intArray(l, l, 0);
1473     int i;
1474     bool has_bb = false;
1475     for (i = 0; i < l; i++) {
1476       LookupswitchPair pair = sw.pair_at(i);
1477       if (pair.offset() < 0) has_bb = true;
1478       sux->at_put(i, block_at(bci() + pair.offset()));
1479       keys->at_put(i, pair.match());
1480     }
1481     // add default successor
1482     if (sw.default_offset() < 0) has_bb = true;
1483     sux->at_put(i, block_at(bci() + sw.default_offset()));
1484     // In case of loop invariant code motion or predicate insertion
1485     // before the body of a loop the state is needed
1486     ValueStack* state_before = copy_state_if_bb(has_bb);
1487     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1488 #ifdef ASSERT
1489     if (res->as_Goto()) {
1490       for (i = 0; i < l; i++) {
1491         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1492           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1493         }
1494       }
1495     }
1496 #endif
1497   }
1498 }

1500 void GraphBuilder::call_register_finalizer() {
1501   // If the receiver requires finalization then emit code to perform
1502   // the registration on return.
1503 
1504   // Gather some type information about the receiver
1505   Value receiver = state()->local_at(0);
1506   assert(receiver != NULL, "must have a receiver");
1507   ciType* declared_type = receiver->declared_type();
1508   ciType* exact_type = receiver->exact_type();
1509   if (exact_type == NULL &&
1510       receiver->as_Local() &&
1511       receiver->as_Local()->java_index() == 0) {
1512     ciInstanceKlass* ik = compilation()->method()->holder();
1513     if (ik->is_final()) {
1514       exact_type = ik;
1515     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1516       // test class is leaf class
1517       compilation()->dependency_recorder()->assert_leaf_type(ik);
1518       exact_type = ik;
1519     } else {
1520       declared_type = ik;
1521     }
1522   }
1523 
1524   // see if we know statically that registration isn't required
1525   bool needs_check = true;
1526   if (exact_type != NULL) {
1527     needs_check = exact_type->as_instance_klass()->has_finalizer();
1528   } else if (declared_type != NULL) {
1529     ciInstanceKlass* ik = declared_type->as_instance_klass();
1530     if (!Dependencies::has_finalizable_subclass(ik)) {
1531       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1532       needs_check = false;
1533     }
1534   }
1535 
1536   if (needs_check) {
1537     // Perform the registration of finalizable objects.
1538     ValueStack* state_before = copy_state_for_exception();
1539     load_local(objectType, 0);
1540     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1541                                state()->pop_arguments(1),
1542                                true, state_before, true));
1543   }
1544 }

1547 void GraphBuilder::method_return(Value x, bool ignore_return) {
1548   if (RegisterFinalizersAtInit &&
1549       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1550     call_register_finalizer();
1551   }
1552 
1553   // The conditions for a memory barrier are described in Parse::do_exits().
1554   bool need_mem_bar = false;
1555   if (method()->is_object_constructor() &&
1556        (scope()->wrote_final() ||
1557          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1558          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1559     need_mem_bar = true;
1560   }
1561 
1562   BasicType bt = method()->return_type()->basic_type();
1563   switch (bt) {
1564     case T_BYTE:
1565     {
1566       Value shift = append(new Constant(new IntConstant(24)));
1567       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1568       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1569       break;
1570     }
1571     case T_SHORT:
1572     {
1573       Value shift = append(new Constant(new IntConstant(16)));
1574       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1575       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1576       break;
1577     }
1578     case T_CHAR:
1579     {
1580       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1581       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1582       break;
1583     }
1584     case T_BOOLEAN:
1585     {
1586       Value mask = append(new Constant(new IntConstant(1)));
1587       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1588       break;
1589     }
1590     default:
1591       break;
1592   }
1593 
1594   // Check to see whether we are inlining. If so, Return
1595   // instructions become Gotos to the continuation point.
1596   if (continuation() != NULL) {
1597 
1598     int invoke_bci = state()->caller_state()->bci();
1599 
1600     if (x != NULL  && !ignore_return) {
1601       ciMethod* caller = state()->scope()->caller()->method();
1602       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1603       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1604         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1605         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1606             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1607           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1608         }
1609       }
1610     }
1611 
1612     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1613 
1614     if (compilation()->env()->dtrace_method_probes()) {
1615       // Report exit from inline methods
1616       Values* args = new Values(1);
1617       args->push(append(new Constant(new MethodConstant(method()))));
1618       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1619     }
1620 
1621     // If the inlined method is synchronized, the monitor must be
1622     // released before we jump to the continuation block.
1623     if (method()->is_synchronized()) {
1624       assert(state()->locks_size() == 1, "receiver must be locked here");
1625       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1626     }
1627 
1628     if (need_mem_bar) {
1629       append(new MemBar(lir_membar_storestore));
1630     }
1631 
1632     // State at end of inlined method is the state of the caller
1633     // without the method parameters on stack, including the
1634     // return value, if any, of the inlined method on operand stack.
1635     set_state(state()->caller_state()->copy_for_parsing());
1636     if (x != NULL) {
1637       if (!ignore_return) {
1638         state()->push(x->type(), x);
1639       }
1640       if (profile_return() && x->type()->is_object_kind()) {
1641         ciMethod* caller = state()->scope()->method();
1642         profile_return_type(x, method(), caller, invoke_bci);
1643       }
1644     }
1645     Goto* goto_callee = new Goto(continuation(), false);
1646 
1647     // See whether this is the first return; if so, store off some
1648     // of the state for later examination
1649     if (num_returns() == 0) {
1650       set_inline_cleanup_info();
1651     }
1652 
1653     // The current bci() is in the wrong scope, so use the bci() of
1654     // the continuation point.
1655     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1656     incr_num_returns();
1657     return;
1658   }
1659 
1660   state()->truncate_stack(0);
1661   if (method()->is_synchronized()) {
1662     // perform the unlocking before exiting the method
1663     Value receiver;
1664     if (!method()->is_static()) {
1665       receiver = _initial_state->local_at(0);
1666     } else {
1667       receiver = append(new Constant(new ClassConstant(method()->holder())));
1668     }
1669     append_split(new MonitorExit(receiver, state()->unlock()));
1670   }
1671 
1672   if (need_mem_bar) {
1673       append(new MemBar(lir_membar_storestore));
1674   }
1675 
1676   assert(!ignore_return, "Ignoring return value works only for inlining");
1677   append(new Return(x));
1678 }

1757     }
1758   }
1759 
1760   const int offset = !needs_patching ? field->offset() : -1;
1761   switch (code) {
1762     case Bytecodes::_getstatic: {
1763       // check for compile-time constants, i.e., initialized static final fields
1764       Value constant = NULL;
1765       if (field->is_static_constant() && !PatchALot) {
1766         ciConstant field_value = field->constant_value();
1767         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1768                "stable static w/ default value shouldn't be a constant");
1769         constant = make_constant(field_value, field);
1770       }
1771       if (constant != NULL) {
1772         push(type, append(constant));
1773       } else {
1774         if (state_before == NULL) {
1775           state_before = copy_state_for_exception();
1776         }
1777         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1778                                         state_before, needs_patching);
1779         push(type, append(load_field));
1780       }
1781       break;
1782     }
1783     case Bytecodes::_putstatic: {
1784       Value val = pop(type);
1785       val->set_escaped();
1786       if (state_before == NULL) {
1787         state_before = copy_state_for_exception();
1788       }
1789       if (field_type == T_BOOLEAN) {
1790         Value mask = append(new Constant(new IntConstant(1)));
1791         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1792       }
1793       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1794       break;
1795     }
1796     case Bytecodes::_getfield: {
1797       // Check for compile-time constants, i.e., trusted final non-static fields.
1798       Value constant = NULL;
1799       if (state_before == NULL && field->is_flattened()) {
1800         // Save the entire state and re-execute on deopt when accessing flattened fields
1801         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1802         state_before = copy_state_before();
1803       }
1804       obj = apop();
1805       ObjectType* obj_type = obj->type()->as_ObjectType();
1806       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1807         ciObject* const_oop = obj_type->constant_value();
1808         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1809           ciConstant field_value = field->constant_value_of(const_oop);
1810           if (field_value.is_valid()) {
1811             if (field->signature()->is_Q_signature() && field_value.is_null_or_zero()) {
1812               // Non-flattened inline type field. Replace null by the default value.
1813               constant = new Constant(new InstanceConstant(field->type()->as_inline_klass()->default_instance()));
1814             } else {
1815               constant = make_constant(field_value, field);
1816             }
1817             // For CallSite objects add a dependency for invalidation of the optimization.
1818             if (field->is_call_site_target()) {
1819               ciCallSite* call_site = const_oop->as_call_site();
1820               if (!call_site->is_fully_initialized_constant_call_site()) {
1821                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1822                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1823               }
1824             }
1825           }
1826         }
1827       }
1828       if (constant != NULL) {
1829         push(type, append(constant));
1830       } else {
1831         if (state_before == NULL) {
1832           state_before = copy_state_for_exception();
1833         }
1834         if (!field->is_flattened()) {
1835           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1836           Value replacement = !needs_patching ? _memory->load(load) : load;
1837           if (replacement != load) {
1838             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1839             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1840             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1841             BasicType bt = field->type()->basic_type();
1842             switch (bt) {
1843             case T_BOOLEAN:
1844             case T_BYTE:
1845               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1846               break;
1847             case T_CHAR:
1848               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1849               break;
1850             case T_SHORT:
1851               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1852               break;
1853             default:
1854               break;
1855             }
1856             push(type, replacement);
1857           } else {
1858             push(type, append(load));
1859           }
1860         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1861           assert(field->type()->is_inlinetype(), "Sanity check");
1862           ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1863           int flattening_offset = field->offset() - inline_klass->first_field_offset();
1864           assert(field->type()->is_inlinetype(), "Sanity check");
1865           scope()->set_wrote_final();
1866           scope()->set_wrote_fields();
1867           NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(inline_klass, state_before, false);
1868           _memory->new_instance(new_instance);
1869           apush(append_split(new_instance));
1870           copy_inline_content(inline_klass, obj, field->offset(), new_instance, inline_klass->first_field_offset(),
1871                        state_before, needs_patching);
1872         }
1873       }
1874       break;
1875     }
1876     case Bytecodes::_withfield:
1877     case Bytecodes::_putfield: {
1878       Value val = pop(type);
1879       val->set_escaped();
1880       obj = apop();
1881       if (state_before == NULL) {
1882         state_before = copy_state_for_exception();
1883       }
1884       if (field->type()->basic_type() == T_BOOLEAN) {
1885         Value mask = append(new Constant(new IntConstant(1)));
1886         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1887       }
1888 
1889       if (!field->is_flattened()) {
1890         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1891         if (!needs_patching) store = _memory->store(store);
1892         if (store != NULL) {
1893           append(store);
1894         }
1895       } else {
1896         assert(field->type()->is_inlinetype(), "Sanity check");
1897         ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1898         int flattening_offset = field->offset() - inline_klass->first_field_offset();
1899         copy_inline_content(inline_klass, val, inline_klass->first_field_offset(), obj, field->offset(),
1900                    state_before, needs_patching);
1901       }
1902       break;
1903     }
1904     default:
1905       ShouldNotReachHere();
1906       break;
1907   }
1908 }

1911 void GraphBuilder::withfield(int field_index)
1912 {
1913   bool will_link;
1914   ciField* field_modify = stream()->get_field(will_link);
1915   ciInstanceKlass* holder = field_modify->holder();
1916   BasicType field_type = field_modify->type()->basic_type();
1917   ValueType* type = as_ValueType(field_type);
1918 
1919   // call will_link again to determine if the field is valid.
1920   const bool needs_patching = !holder->is_loaded() ||
1921                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1922                               PatchALot;
1923 
1924 
1925   scope()->set_wrote_final();
1926   scope()->set_wrote_fields();
1927 
1928   const int offset = !needs_patching ? field_modify->offset() : -1;
1929 
1930   if (!holder->is_loaded()
1931       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1932       ) {
1933     ValueStack* state_before = copy_state_before();
1934     Value val = pop(type);
1935     Value obj = apop();
1936     apush(append_split(new WithField(state_before)));
1937     return;
1938   }
1939   ValueStack* state_before = copy_state_before();
1940 
1941   Value val = pop(type);
1942   Value obj = apop();
1943 
1944   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1945     int astore_index;
1946     ciBytecodeStream s(method());
1947     s.force_bci(bci());
1948     s.next();
1949     switch (s.cur_bc()) {
1950     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1951     case Bytecodes::_astore_0:  astore_index = 0; break;
1952     case Bytecodes::_astore_1:  astore_index = 1; break;
1953     case Bytecodes::_astore_2:  astore_index = 2; break;
1954     case Bytecodes::_astore_3:  astore_index = 3; break;
1955     default: astore_index = -1;
1956     }
1957 
1958     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1959       // We have a sequence like this, where we load a value object from a local slot,
1960       // and overwrite the same local slot with a modified copy of the inline object.
1961       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1962       //      astore 9
1963       //      ...
1964       //      iload_0
1965       //      aload 9
1966       //      swap
1967       //      withfield #7 // Field x:I
1968       //      astore 9
1969       // If this object was created by defaultvalue, and has not escaped, and is not stored
1970       // in any other local slots, we can effectively treat the withfield/astore
1971       // sequence as a single putfield bytecode.
1972       push(objectType, obj);
1973       push(type, val);
1974       access_field(Bytecodes::_withfield);
1975       stream()->next(); // skip the next astore/astore_n bytecode.
1976       return;
1977     }
1978   }
1979 
1980   assert(holder->is_inlinetype(), "must be an inline klass");
1981   // Save the entire state and re-execute on deopt when executing withfield
1982   state_before->set_should_reexecute(true);
1983   NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(holder->as_inline_klass(), state_before, false);
1984   _memory->new_instance(new_instance);
1985   apush(append_split(new_instance));





1986 
1987   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1988     ciField* field = holder->nonstatic_field_at(i);
1989     int off = field->offset();
1990 
1991     if (field->offset() != offset) {
1992       if (field->is_flattened()) {
1993         assert(field->type()->is_inlinetype(), "Sanity check");
1994         assert(field->type()->is_inlinetype(), "Only inline types can be flattened");
1995         ciInlineKlass* vk = field->type()->as_inline_klass();
1996         copy_inline_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
1997       } else {
1998         // Only load those fields who are not modified
1999         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2000         Value replacement = append(load);
2001         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2002         append(store);

2003       }
2004     }
2005   }
2006 
2007   // Field to modify
2008   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2009     Value mask = append(new Constant(new IntConstant(1)));
2010     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2011   }
2012   if (field_modify->is_flattened()) {
2013     assert(field_modify->type()->is_inlinetype(), "Only inline types can be flattened");
2014     ciInlineKlass* vk = field_modify->type()->as_inline_klass();
2015     copy_inline_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2016   } else {
2017     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2018     append(store);
2019   }
2020 }

2029   int n = 0;
2030   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));

2035       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2036     }
2037   }

2041       // The receiver is profiled on method entry so it's included in
2042       // the number of parameters but here we're only interested in
2043       // actual arguments.
2044       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2045     }
2046   }
2047   if (n > 0) {
2048     return new Values(n);
2049   }
2050   return NULL;

2260       // dest_method here, as opposed to the actual receiver, which may

2270       && !patch_for_appendix) {

2274         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2275         code == Bytecodes::_invokedynamic) {
2276       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2277       // static binding => check if callee is ok
2278       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2279 
2280       CHECK_BAILOUT();
2281       clear_inline_bailout();
2282 
2283       if (success) {
2284         // Register dependence if JVMTI has either breakpoint
2285         // setting or hotswapping of methods capabilities since they may
2286         // cause deoptimization.
2287         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2288           dependency_recorder()->assert_evol_method(inline_target);
2289         }
2290         return;
2291       }
2292     } else {
2293       print_inlining(target, "no static binding", /*success*/ false);
2294     }
2295   } else {
2296     print_inlining(target, "not inlineable", /*success*/ false);
2297   }
2298 
2299   // If we attempted an inline which did not succeed because of a
2300   // bailout during construction of the callee graph, the entire
2301   // compilation has to be aborted. This is fairly rare and currently
2302   // seems to only occur for jasm-generated classes which contain
2303   // jsr/ret pairs which are not associated with finally clauses and
2304   // do not have exception handlers in the containing method, and are
2305   // therefore not caught early enough to abort the inlining without
2306   // corrupting the graph. (We currently bail out with a non-empty
2307   // stack at a ret in these situations.)
2308   CHECK_BAILOUT();
2309 
2310   // inlining not successful => standard invoke
2311   ValueType* result_type = as_ValueType(declared_signature->return_type());
2312   ValueStack* state_before = copy_state_exhandling();
2313 
2314   // The bytecode (code) might change in this method so we are checking this very late.
2315   const bool has_receiver =
2316     code == Bytecodes::_invokespecial   ||
2317     code == Bytecodes::_invokevirtual   ||
2318     code == Bytecodes::_invokeinterface;
2319   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2320   Value recv = has_receiver ? apop() : NULL;
2321   int vtable_index = Method::invalid_vtable_index;
2322 
2323   // A null check is required here (when there is a receiver) for any of the following cases
2324   // - invokespecial, always need a null check.
2325   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2326   //   and require null checking. If the target is loaded a null check is emitted here.
2327   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2328   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2329   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2330   //   potentially fail, and can't have the null check before the resolution.)
2331   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2332   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2333   //
2334   // Normal invokevirtual will perform the null check during lookup
2335 
2336   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2337       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2338 
2339   if (need_null_check) {
2340     if (recv != NULL) {
2341       null_check(recv);
2342     }
2343 
2344     if (is_profiling()) {
2345       // Note that we'd collect profile data in this method if we wanted it.
2346       compilation()->set_would_profile(true);
2347 
2348       if (profile_calls()) {
2349         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2350         ciKlass* target_klass = NULL;
2351         if (cha_monomorphic_target != NULL) {
2352           target_klass = cha_monomorphic_target->holder();
2353         } else if (exact_target != NULL) {
2354           target_klass = exact_target->holder();
2355         }
2356         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2357       }
2358     }
2359   }
2360 
2361   if (recv != NULL) {
2362     recv->set_escaped();
2363   }
2364   for (int i=0; i<args->length(); i++) {
2365     args->at(0)->set_escaped();
2366   }
2367 
2368   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2369                               declared_signature->return_type()->is_inlinetype());
2370   // push result
2371   append_split(result);
2372 
2373   if (result_type != voidType) {
2374     if (method()->is_strict()) {
2375       push(result_type, round_fp(result));
2376     } else {
2377       push(result_type, result);
2378     }
2379   }
2380   if (profile_return() && result_type->is_object_kind()) {
2381     profile_return_type(result, target);
2382   }
2383 }

2566 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2567   Canonicalizer canon(compilation(), instr, bci);
2568   Instruction* i1 = canon.canonical();
2569   if (i1->is_linked() || !i1->can_be_linked()) {
2570     // Canonicalizer returned an instruction which was already
2571     // appended so simply return it.
2572     return i1;
2573   }
2574 
2575   if (UseLocalValueNumbering) {
2576     // Lookup the instruction in the ValueMap and add it to the map if
2577     // it's not found.
2578     Instruction* i2 = vmap()->find_insert(i1);
2579     if (i2 != i1) {
2580       // found an entry in the value map, so just return it.
2581       assert(i2->is_linked(), "should already be linked");
2582       return i2;
2583     }
2584     ValueNumberingEffects vne(vmap());
2585     i1->visit(&vne);
2586   }
2587 
2588   // i1 was not eliminated => append it
2589   assert(i1->next() == NULL, "shouldn't already be linked");
2590   _last = _last->set_next(i1, canon.bci());
2591 
2592   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2593     // set the bailout state but complete normal processing.  We
2594     // might do a little more work before noticing the bailout so we
2595     // want processing to continue normally until it's noticed.
2596     bailout("Method and/or inlining is too large");
2597   }
2598 
2599 #ifndef PRODUCT
2600   if (PrintIRDuringConstruction) {
2601     InstructionPrinter ip;
2602     ip.print_line(i1);
2603     if (Verbose) {
2604       state()->print();
2605     }
2606   }
2607 #endif
2608 
2609   // save state after modification of operand stack for StateSplit instructions
2610   StateSplit* s = i1->as_StateSplit();
2611   if (s != NULL) {
2612     if (EliminateFieldAccess) {
2613       Intrinsic* intrinsic = s->as_Intrinsic();
2614       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2615         _memory->kill();
2616       }
2617     }
2618     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2619   }
2620 
2621   // set up exception handlers for this instruction if necessary
2622   if (i1->can_trap()) {
2623     i1->set_exception_handlers(handle_exception(i1));
2624     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2625   }
2626   return i1;
2627 }

2661 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2662   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2663     assert(instruction->exception_state() == NULL
2664            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2665            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2666            "exception_state should be of exception kind");
2667     return new XHandlers();
2668   }
2669 
2670   XHandlers*  exception_handlers = new XHandlers();
2671   ScopeData*  cur_scope_data = scope_data();
2672   ValueStack* cur_state = instruction->state_before();
2673   ValueStack* prev_state = NULL;
2674   int scope_count = 0;
2675 
2676   assert(cur_state != NULL, "state_before must be set");
2677   do {
2678     int cur_bci = cur_state->bci();
2679     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2680     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2681 
2682     // join with all potential exception handlers
2683     XHandlers* list = cur_scope_data->xhandlers();
2684     const int n = list->length();
2685     for (int i = 0; i < n; i++) {
2686       XHandler* h = list->handler_at(i);
2687       if (h->covers(cur_bci)) {
2688         // h is a potential exception handler => join it
2689         compilation()->set_has_exception_handlers(true);
2690 
2691         BlockBegin* entry = h->entry_block();
2692         if (entry == block()) {
2693           // It's acceptable for an exception handler to cover itself
2694           // but we don't handle that in the parser currently.  It's
2695           // very rare so we bailout instead of trying to handle it.
2696           BAILOUT_("exception handler covers itself", exception_handlers);
2697         }
2698         assert(entry->bci() == h->handler_bci(), "must match");
2699         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2700 
2701         // previously this was a BAILOUT, but this is not necessary
2702         // now because asynchronous exceptions are not handled this way.
2703         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2704 
2705         // xhandler start with an empty expression stack
2706         if (cur_state->stack_size() != 0) {
2707           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2708         }
2709         if (instruction->exception_state() == NULL) {
2710           instruction->set_exception_state(cur_state);
2711         }
2712 
2713         // Note: Usually this join must work. However, very
2714         // complicated jsr-ret structures where we don't ret from
2715         // the subroutine can cause the objects on the monitor
2716         // stacks to not match because blocks can be parsed twice.
2717         // The only test case we've seen so far which exhibits this
2718         // problem is caught by the infinite recursion test in
2719         // GraphBuilder::jsr() if the join doesn't work.
2720         if (!entry->try_merge(cur_state)) {
2721           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2722         }
2723 
2724         // add current state for correct handling of phi functions at begin of xhandler
2725         int phi_operand = entry->add_exception_state(cur_state);
2726 
2727         // add entry to the list of xhandlers of this block
2728         _block->add_exception_handler(entry);
2729 
2730         // add back-edge from xhandler entry to this block
2731         if (!entry->is_predecessor(_block)) {
2732           entry->add_predecessor(_block);
2733         }
2734 
2735         // clone XHandler because phi_operand and scope_count can not be shared
2736         XHandler* new_xhandler = new XHandler(h);
2737         new_xhandler->set_phi_operand(phi_operand);
2738         new_xhandler->set_scope_count(scope_count);
2739         exception_handlers->append(new_xhandler);
2740 
2741         // fill in exception handler subgraph lazily
2742         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2743         cur_scope_data->add_to_work_list(entry);
2744 
2745         // stop when reaching catchall
2746         if (h->catch_type() == 0) {
2747           return exception_handlers;
2748         }
2749       }
2750     }
2751 
2752     if (exception_handlers->length() == 0) {
2753       // This scope and all callees do not handle exceptions, so the local
2754       // variables of this scope are not needed. However, the scope itself is
2755       // required for a correct exception stack trace -> clear out the locals.
2756       if (_compilation->env()->should_retain_local_variables()) {
2757         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2758       } else {
2759         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2760       }
2761       if (prev_state != NULL) {
2762         prev_state->set_caller_state(cur_state);
2763       }
2764       if (instruction->exception_state() == NULL) {
2765         instruction->set_exception_state(cur_state);
2766       }
2767     }
2768 
2769     // Set up iteration for next time.
2770     // If parsing a jsr, do not grab exception handlers from the
2771     // parent scopes for this method (already got them, and they
2772     // needed to be cloned)
2773 
2774     while (cur_scope_data->parsing_jsr()) {
2775       cur_scope_data = cur_scope_data->parent();
2776     }
2777 
2778     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2779     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2780 
2781     prev_state = cur_state;
2782     cur_state = cur_state->caller_state();
2783     cur_scope_data = cur_scope_data->parent();
2784     scope_count++;
2785   } while (cur_scope_data != NULL);
2786 
2787   return exception_handlers;
2788 }

2808 
2809 Value PhiSimplifier::simplify(Value v) {
2810   Phi* phi = v->as_Phi();
2811 
2812   if (phi == NULL) {
2813     // no phi function
2814     return v;
2815   } else if (v->has_subst()) {
2816     // already substituted; subst can be phi itself -> simplify
2817     return simplify(v->subst());
2818   } else if (phi->is_set(Phi::cannot_simplify)) {
2819     // already tried to simplify phi before
2820     return phi;
2821   } else if (phi->is_set(Phi::visited)) {
2822     // break cycles in phi functions
2823     return phi;
2824   } else if (phi->type()->is_illegal()) {
2825     // illegal phi functions are ignored anyway
2826     return phi;
2827 
2828   } else {
2829     // mark phi function as processed to break cycles in phi functions
2830     phi->set(Phi::visited);
2831 
2832     // simplify x = [y, x] and x = [y, y] to y
2833     Value subst = NULL;
2834     int opd_count = phi->operand_count();
2835     for (int i = 0; i < opd_count; i++) {
2836       Value opd = phi->operand_at(i);
2837       assert(opd != NULL, "Operand must exist!");
2838 
2839       if (opd->type()->is_illegal()) {
2840         // if one operand is illegal, the entire phi function is illegal
2841         phi->make_illegal();
2842         phi->clear(Phi::visited);
2843         return phi;
2844       }
2845 
2846       Value new_opd = simplify(opd);

3328 BlockBegin* GraphBuilder::setup_start_block(int osr_bci, BlockBegin* std_entry, BlockBegin* osr_entry, ValueStack* state) {
3329   BlockBegin* start = new BlockBegin(0);
3330 
3331   // This code eliminates the empty start block at the beginning of
3332   // each method.  Previously, each method started with the
3333   // start-block created below, and this block was followed by the
3334   // header block that was always empty.  This header block is only
3335   // necesary if std_entry is also a backward branch target because
3336   // then phi functions may be necessary in the header block.  It's
3337   // also necessary when profiling so that there's a single block that
3338   // can increment the interpreter_invocation_count.
3339   BlockBegin* new_header_block;
3340   if (std_entry->number_of_preds() > 0 || count_invocations() || count_backedges()) {
3341     new_header_block = header_block(std_entry, BlockBegin::std_entry_flag, state);
3342   } else {
3343     new_header_block = std_entry;
3344   }
3345 
3346   // setup start block (root for the IR graph)
3347   Base* base =
3348     new Base(
3349       new_header_block,
3350       osr_entry
3351     );
3352   start->set_next(base, 0);
3353   start->set_end(base);
3354   // create & setup state for start block
3355   start->set_state(state->copy(ValueStack::StateAfter, std_entry->bci()));
3356   base->set_state(state->copy(ValueStack::StateAfter, std_entry->bci()));
3357 
3358   if (base->std_entry()->state() == NULL) {
3359     // setup states for header blocks
3360     base->std_entry()->merge(state);
3361   }
3362 
3363   assert(base->std_entry()->state() != NULL, "");
3364   return start;

3368 void GraphBuilder::setup_osr_entry_block() {
3369   assert(compilation()->is_osr_compile(), "only for osrs");
3370 
3371   int osr_bci = compilation()->osr_bci();
3372   ciBytecodeStream s(method());
3373   s.reset_to_bci(osr_bci);
3374   s.next();
3375   scope_data()->set_stream(&s);
3376 
3377   // create a new block to be the osr setup code
3378   _osr_entry = new BlockBegin(osr_bci);
3379   _osr_entry->set(BlockBegin::osr_entry_flag);
3380   _osr_entry->set_depth_first_number(0);
3381   BlockBegin* target = bci2block()->at(osr_bci);
3382   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3383   // the osr entry has no values for locals
3384   ValueStack* state = target->state()->copy();
3385   _osr_entry->set_state(state);
3386 
3387   kill_all();
3388   _block = _osr_entry;
3389   _state = _osr_entry->state()->copy();
3390   assert(_state->bci() == osr_bci, "mismatch");
3391   _last  = _osr_entry;
3392   Value e = append(new OsrEntry());
3393   e->set_needs_null_check(false);
3394 
3395   // OSR buffer is
3396   //
3397   // locals[nlocals-1..0]
3398   // monitors[number_of_locks-1..0]
3399   //
3400   // locals is a direct copy of the interpreter frame so in the osr buffer
3401   // so first slot in the local array is the last local from the interpreter
3402   // and last slot is local[0] (receiver) from the interpreter
3403   //
3404   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3405   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3406   // in the interpreter frame (the method lock if a sync method)
3407 
3408   // Initialize monitors in the compiled activation.
3409 
3410   int index;
3411   Value local;
3412 
3413   // find all the locals that the interpreter thinks contain live oops
3414   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3415 
3416   // compute the offset into the locals so that we can treat the buffer
3417   // as if the locals were still in the interpreter frame
3418   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3419   for_each_local_value(state, index, local) {
3420     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3421     Value get;
3422     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3423       // The interpreter thinks this local is dead but the compiler
3424       // doesn't so pretend that the interpreter passed in null.
3425       get = append(new Constant(objectNull));
3426     } else {
3427       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3428                                     append(new Constant(new IntConstant(offset))),
3429                                     0,
3430                                     true /*unaligned*/, true /*wide*/));
3431     }
3432     _state->store_local(index, get);
3433   }
3434 
3435   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3436 
3437   assert(state->caller_state() == NULL, "should be top scope");
3438   state->clear_locals();
3439   Goto* g = new Goto(target, false);
3440   append(g);
3441   _osr_entry->set_end(g);
3442   target->merge(_osr_entry->end()->state());
3443 
3444   scope_data()->set_stream(NULL);
3445 }

3448 ValueStack* GraphBuilder::state_at_entry() {
3449   ValueStack* state = new ValueStack(scope(), NULL);
3450 
3451   // Set up locals for receiver
3452   int idx = 0;
3453   if (!method()->is_static()) {
3454     // we should always see the receiver
3455     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3456              /*receiver*/ true, /*null_free*/ method()->holder()->is_flat_array_klass()));
3457     idx = 1;
3458   }
3459 
3460   // Set up locals for incoming arguments
3461   ciSignature* sig = method()->signature();
3462   for (int i = 0; i < sig->count(); i++) {
3463     ciType* type = sig->type_at(i);
3464     BasicType basic_type = type->basic_type();
3465     // don't allow T_ARRAY to propagate into locals types
3466     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3467     ValueType* vt = as_ValueType(basic_type);
3468     state->store_local(idx, new Local(type, vt, idx, false, type->is_inlinetype()));
3469     idx += type->size();
3470   }
3471 
3472   // lock synchronized method
3473   if (method()->is_synchronized()) {
3474     state->lock(NULL);
3475   }
3476 
3477   return state;
3478 }

3481 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3482   : _scope_data(NULL)
3483   , _compilation(compilation)
3484   , _memory(new MemoryBuffer())
3485   , _inline_bailout_msg(NULL)
3486   , _instruction_count(0)
3487   , _osr_entry(NULL)
3488 {
3489   int osr_bci = compilation->osr_bci();
3490 
3491   // determine entry points and bci2block mapping
3492   BlockListBuilder blm(compilation, scope, osr_bci);
3493   CHECK_BAILOUT();
3494 
3495   BlockList* bci2block = blm.bci2block();
3496   BlockBegin* start_block = bci2block->at(0);
3497 
3498   push_root_scope(scope, bci2block, start_block);
3499 
3500   // setup state for std entry
3501   _initial_state = state_at_entry();
3502   start_block->merge(_initial_state);
3503 
3504   // complete graph
3505   _vmap        = new ValueMap();
3506   switch (scope->method()->intrinsic_id()) {
3507   case vmIntrinsics::_dabs          : // fall through
3508   case vmIntrinsics::_dsqrt         : // fall through
3509   case vmIntrinsics::_dsin          : // fall through
3510   case vmIntrinsics::_dcos          : // fall through
3511   case vmIntrinsics::_dtan          : // fall through
3512   case vmIntrinsics::_dlog          : // fall through
3513   case vmIntrinsics::_dlog10        : // fall through
3514   case vmIntrinsics::_dexp          : // fall through
3515   case vmIntrinsics::_dpow          : // fall through
3516     {
3517       // Compiles where the root method is an intrinsic need a special
3518       // compilation environment because the bytecodes for the method
3519       // shouldn't be parsed during the compilation, only the special
3520       // Intrinsic node should be emitted.  If this isn't done the the
3521       // code for the inlined version will be different than the root
3522       // compiled version which could lead to monotonicity problems on
3523       // intel.
3524       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3525         BAILOUT("failed to inline intrinsic, method not annotated");
3526       }
3527 
3528       // Set up a stream so that appending instructions works properly.
3529       ciBytecodeStream s(scope->method());
3530       s.reset_to_bci(0);
3531       scope_data()->set_stream(&s);
3532       s.next();
3533 
3534       // setup the initial block state
3535       _block = start_block;
3536       _state = start_block->state()->copy_for_parsing();
3537       _last  = start_block;
3538       load_local(doubleType, 0);
3539       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3540         load_local(doubleType, 2);
3541       }
3542 
3543       // Emit the intrinsic node.
3544       bool result = try_inline_intrinsics(scope->method());
3545       if (!result) BAILOUT("failed to inline intrinsic");
3546       method_return(dpop());
3547 
3548       // connect the begin and end blocks and we're all done.
3549       BlockEnd* end = last()->as_BlockEnd();
3550       block()->set_end(end);
3551       break;
3552     }
3553 
3554   case vmIntrinsics::_Reference_get:
3555     {
3556       {
3557         // With java.lang.ref.reference.get() we must go through the
3558         // intrinsic - when G1 is enabled - even when get() is the root
3559         // method of the compile so that, if necessary, the value in
3560         // the referent field of the reference object gets recorded by
3561         // the pre-barrier code.
3562         // Specifically, if G1 is enabled, the value in the referent
3563         // field is recorded by the G1 SATB pre barrier. This will
3564         // result in the referent being marked live and the reference
3565         // object removed from the list of discovered references during
3566         // reference processing.
3567         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3568           BAILOUT("failed to inline intrinsic, method not annotated");
3569         }
3570 
3571         // Also we need intrinsic to prevent commoning reads from this field
3572         // across safepoint since GC can change its value.
3573 
3574         // Set up a stream so that appending instructions works properly.
3575         ciBytecodeStream s(scope->method());
3576         s.reset_to_bci(0);
3577         scope_data()->set_stream(&s);
3578         s.next();
3579 
3580         // setup the initial block state
3581         _block = start_block;
3582         _state = start_block->state()->copy_for_parsing();
3583         _last  = start_block;
3584         load_local(objectType, 0);
3585 
3586         // Emit the intrinsic node.
3587         bool result = try_inline_intrinsics(scope->method());
3588         if (!result) BAILOUT("failed to inline intrinsic");
3589         method_return(apop());
3590 
3591         // connect the begin and end blocks and we're all done.
3592         BlockEnd* end = last()->as_BlockEnd();
3593         block()->set_end(end);
3594         break;
3595       }
3596       // Otherwise, fall thru
3597     }
3598 
3599   default:
3600     scope_data()->add_to_work_list(start_block);
3601     iterate_all_blocks();
3602     break;
3603   }
3604   CHECK_BAILOUT();
3605 
3606   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3607 
3608   eliminate_redundant_phis(_start);
3609 
3610   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3611   // for osr compile, bailout if some requirements are not fulfilled
3612   if (osr_bci != -1) {
3613     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3614     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3615       BAILOUT("osr entry must have been visited for osr compile");
3616     }
3617 
3618     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3619     if (!osr_block->state()->stack_is_empty()) {
3620       BAILOUT("stack not empty at OSR entry point");
3621     }
3622   }
3623 #ifndef PRODUCT
3624   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3625 #endif
3626 }

3673 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3674   const char* msg = NULL;
3675 
3676   // clear out any existing inline bailout condition
3677   clear_inline_bailout();
3678 
3679   // exclude methods we don't want to inline
3680   msg = should_not_inline(callee);
3681   if (msg != NULL) {
3682     print_inlining(callee, msg, /*success*/ false);
3683     return false;
3684   }
3685 
3686   // method handle invokes
3687   if (callee->is_method_handle_intrinsic()) {
3688     if (try_method_handle_inline(callee, ignore_return)) {
3689       if (callee->has_reserved_stack_access()) {
3690         compilation()->set_has_reserved_stack_access(true);
3691       }
3692       return true;
3693     }
3694     return false;
3695   }
3696 
3697   // handle intrinsics
3698   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3699       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3700     if (try_inline_intrinsics(callee, ignore_return)) {
3701       print_inlining(callee, "intrinsic");
3702       if (callee->has_reserved_stack_access()) {
3703         compilation()->set_has_reserved_stack_access(true);
3704       }
3705       return true;
3706     }
3707     // try normal inlining
3708   }
3709 
3710   // certain methods cannot be parsed at all
3711   msg = check_can_parse(callee);
3712   if (msg != NULL) {
3713     print_inlining(callee, msg, /*success*/ false);
3714     return false;
3715   }
3716 
3717   // If bytecode not set use the current one.
3718   if (bc == Bytecodes::_illegal) {
3719     bc = code();
3720   }
3721   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3722     if (callee->has_reserved_stack_access()) {
3723       compilation()->set_has_reserved_stack_access(true);
3724     }
3725     return true;
3726   }
3727 
3728   // Entire compilation could fail during try_inline_full call.
3729   // In that case printing inlining decision info is useless.
3730   if (!bailed_out())
3731     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3732 
3733   return false;
3734 }

3883   // will be replaced with Gotos to this point.

3888   // pick up the state from the Ret instructions.

3891   push_scope_for_jsr(cont, jsr_dest_bci);

3894   // (only using the bci of this stream)

3913   // Ready to resume parsing in subroutine

3921   // iterate_bytecodes_for_block()/ret() and we should not touch the

4296     // iterate_bytecodes_for_block will stop when we return.
4297     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4298       // add continuation to work list instead of parsing it immediately
4299       assert(_last && _last->as_BlockEnd(), "");
4300       scope_data()->parent()->add_to_work_list(continuation());
4301       _skip_block = true;
4302     }
4303   }
4304 
4305   // Fill the exception handler for synchronized methods with instructions
4306   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4307     fill_sync_handler(lock, sync_handler);
4308   } else {
4309     pop_scope();
4310   }
4311 
4312   compilation()->notice_inlined_method(callee);
4313 
4314   return true;
4315 }

4318 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4319   ValueStack* state_before = copy_state_before();
4320   vmIntrinsics::ID iid = callee->intrinsic_id();
4321   switch (iid) {
4322   case vmIntrinsics::_invokeBasic:
4323     {
4324       // get MethodHandle receiver
4325       const int args_base = state()->stack_size() - callee->arg_size();
4326       ValueType* type = state()->stack_at(args_base)->type();
4327       if (type->is_constant()) {
4328         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4329         // We don't do CHA here so only inline static and statically bindable methods.
4330         if (target->is_static() || target->can_be_statically_bound()) {
4331           if (ciMethod::is_consistent_info(callee, target)) {
4332             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4333             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4334             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4335               return true;
4336             }
4337           } else {
4338             print_inlining(target, "signatures mismatch", /*success*/ false);
4339           }
4340         } else {
4341           print_inlining(target, "not static or statically bindable", /*success*/ false);
4342         }
4343       } else {
4344         print_inlining(callee, "receiver not constant", /*success*/ false);
4345       }
4346     }
4347     break;
4348 
4349   case vmIntrinsics::_linkToVirtual:
4350   case vmIntrinsics::_linkToStatic:
4351   case vmIntrinsics::_linkToSpecial:
4352   case vmIntrinsics::_linkToInterface:
4353     {
4354       // pop MemberName argument
4355       const int args_base = state()->stack_size() - callee->arg_size();
4356       ValueType* type = apop()->type();
4357       if (type->is_constant()) {
4358         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4359         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4360         // If the target is another method handle invoke, try to recursively get
4361         // a better target.
4362         if (target->is_method_handle_intrinsic()) {
4363           if (try_method_handle_inline(target, ignore_return)) {
4364             return true;
4365           }
4366         } else if (!ciMethod::is_consistent_info(callee, target)) {
4367           print_inlining(target, "signatures mismatch", /*success*/ false);
4368         } else {
4369           ciSignature* signature = target->signature();
4370           const int receiver_skip = target->is_static() ? 0 : 1;
4371           // Cast receiver to its type.
4372           if (!target->is_static()) {
4373             ciKlass* tk = signature->accessing_klass();
4374             Value obj = state()->stack_at(args_base);
4375             if (obj->exact_type() == NULL &&
4376                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4377               TypeCast* c = new TypeCast(tk, obj, state_before);
4378               append(c);
4379               state()->stack_at_put(args_base, c);
4380             }
4381           }
4382           // Cast reference arguments to its type.
4383           for (int i = 0, j = 0; i < signature->count(); i++) {
4384             ciType* t = signature->type_at(i);
4385             if (t->is_klass()) {
4386               ciKlass* tk = t->as_klass();
4387               Value obj = state()->stack_at(args_base + receiver_skip + j);
4388               if (obj->exact_type() == NULL &&
4389                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4390                 TypeCast* c = new TypeCast(t, obj, state_before);
4391                 append(c);
4392                 state()->stack_at_put(args_base + receiver_skip + j, c);
4393               }
4394             }
4395             j += t->size();  // long and double take two slots
4396           }
4397           // We don't do CHA here so only inline static and statically bindable methods.
4398           if (target->is_static() || target->can_be_statically_bound()) {
4399             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4400             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4401               return true;
4402             }
4403           } else {
4404             print_inlining(target, "not static or statically bindable", /*success*/ false);
4405           }
4406         }
4407       } else {
4408         print_inlining(callee, "MemberName not constant", /*success*/ false);
4409       }
4410     }
4411     break;
4412 
4413   default:
4414     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4415     break;
4416   }
4417   set_state(state_before->copy_for_parsing());
4418   return false;
4419 }
