 133 void BlockListBuilder::set_entries(int osr_bci) {
 134   // generate start blocks
 135   BlockBegin* std_entry = make_block_at(0, NULL);
 136   if (scope()->caller() == NULL) {
 137     std_entry->set(BlockBegin::std_entry_flag);
 138   }
 139   if (osr_bci != -1) {
 140     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 141     osr_entry->set(BlockBegin::osr_entry_flag);
 142   }
 143 
 144   // generate exception entry blocks
 145   XHandlers* list = xhandlers();
 146   const int n = list->length();
 147   for (int i = 0; i < n; i++) {
 148     XHandler* h = list->handler_at(i);
 149     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 150     entry->set(BlockBegin::exception_entry_flag);
 151     h->set_entry_block(entry);
 152   }
 153 }

 156 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 157   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 158 
 159   BlockBegin* block = _bci2block->at(cur_bci);
 160   if (block == NULL) {
 161     block = new BlockBegin(cur_bci);
 162     block->init_stores_to_locals(method()->max_locals());
 163     _bci2block->at_put(cur_bci, block);
 164     _blocks.append(block);
 165 
 166     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 167   }
 168 
 169   if (predecessor != NULL) {
 170     if (block->is_set(BlockBegin::exception_entry_flag)) {
 171       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 172     }
 173 
 174     predecessor->add_successor(block);
 175     block->increment_total_preds();
 176   }
 177 
 178   return block;
 179 }

 182 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 183   current->stores_to_locals().set_bit(local);

 185 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 186   store_one(current, local);
 187   store_one(current, local + 1);

 191 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 192   // Draws edges from a block to its exception handlers
 193   XHandlers* list = xhandlers();
 194   const int n = list->length();
 195 
 196   for (int i = 0; i < n; i++) {
 197     XHandler* h = list->handler_at(i);
 198 
 199     if (h->covers(cur_bci)) {
 200       BlockBegin* entry = h->entry_block();
 201       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 202       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 203 
 204       // add each exception handler only once
 205       if (!current->is_successor(entry)) {
 206         current->add_successor(entry);
 207         entry->increment_total_preds();
 208       }
 209 
 210       // stop when reaching catchall
 211       if (h->catch_type() == 0) break;
 212     }
 213   }
 214 }

 216 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 217   // start a new block after jsr-bytecode and link this block into cfg
 218   make_block_at(next_bci, current);
 219 
 220   // start a new block at the subroutine entry at mark it with special flag
 221   BlockBegin* sr_block = make_block_at(sr_bci, current);
 222   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 223     sr_block->set(BlockBegin::subroutine_entry_flag);
 224   }
 225 }

 228 void BlockListBuilder::set_leaders() {
 229   bool has_xhandlers = xhandlers()->has_handlers();
 230   BlockBegin* current = NULL;
 231 
 232   // The information which bci starts a new block simplifies the analysis
 233   // Without it, backward branches could jump to a bci where no block was created
 234   // during bytecode iteration. This would require the creation of a new block at the
 235   // branch target and a modification of the successor lists.
 236   const BitMap& bci_block_start = method()->bci_block_start();
 237 
 238   ciBytecodeStream s(method());
 239   while (s.next() != ciBytecodeStream::EOBC()) {
 240     int cur_bci = s.cur_bci();
 241 
 242     if (bci_block_start.at(cur_bci)) {
 243       current = make_block_at(cur_bci, current);
 244     }
 245     assert(current != NULL, "must have current block");
 246 
 247     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 248       handle_exceptions(current, cur_bci);
 249     }
 250 
 251     switch (s.cur_bc()) {
 252       // track stores to local variables for selective creation of phi functions
 253       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 254       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 255       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 256       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 257       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 258       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 259       case Bytecodes::_istore_0: store_one(current, 0); break;
 260       case Bytecodes::_istore_1: store_one(current, 1); break;
 261       case Bytecodes::_istore_2: store_one(current, 2); break;
 262       case Bytecodes::_istore_3: store_one(current, 3); break;
 263       case Bytecodes::_lstore_0: store_two(current, 0); break;
 264       case Bytecodes::_lstore_1: store_two(current, 1); break;
 265       case Bytecodes::_lstore_2: store_two(current, 2); break;
 266       case Bytecodes::_lstore_3: store_two(current, 3); break;
 267       case Bytecodes::_fstore_0: store_one(current, 0); break;
 268       case Bytecodes::_fstore_1: store_one(current, 1); break;
 269       case Bytecodes::_fstore_2: store_one(current, 2); break;
 270       case Bytecodes::_fstore_3: store_one(current, 3); break;
 271       case Bytecodes::_dstore_0: store_two(current, 0); break;
 272       case Bytecodes::_dstore_1: store_two(current, 1); break;
 273       case Bytecodes::_dstore_2: store_two(current, 2); break;
 274       case Bytecodes::_dstore_3: store_two(current, 3); break;
 275       case Bytecodes::_astore_0: store_one(current, 0); break;
 276       case Bytecodes::_astore_1: store_one(current, 1); break;
 277       case Bytecodes::_astore_2: store_one(current, 2); break;
 278       case Bytecodes::_astore_3: store_one(current, 3); break;
 279 
 280       // track bytecodes that affect the control flow
 281       case Bytecodes::_athrow:  // fall through
 282       case Bytecodes::_ret:     // fall through
 283       case Bytecodes::_ireturn: // fall through
 284       case Bytecodes::_lreturn: // fall through
 285       case Bytecodes::_freturn: // fall through
 286       case Bytecodes::_dreturn: // fall through
 287       case Bytecodes::_areturn: // fall through
 288       case Bytecodes::_return:
 289         current = NULL;
 290         break;
 291 
 292       case Bytecodes::_ifeq:      // fall through
 293       case Bytecodes::_ifne:      // fall through
 294       case Bytecodes::_iflt:      // fall through
 295       case Bytecodes::_ifge:      // fall through
 296       case Bytecodes::_ifgt:      // fall through
 297       case Bytecodes::_ifle:      // fall through
 298       case Bytecodes::_if_icmpeq: // fall through
 299       case Bytecodes::_if_icmpne: // fall through
 300       case Bytecodes::_if_icmplt: // fall through
 301       case Bytecodes::_if_icmpge: // fall through
 302       case Bytecodes::_if_icmpgt: // fall through
 303       case Bytecodes::_if_icmple: // fall through
 304       case Bytecodes::_if_acmpeq: // fall through
 305       case Bytecodes::_if_acmpne: // fall through
 306       case Bytecodes::_ifnull:    // fall through
 307       case Bytecodes::_ifnonnull:
 308         make_block_at(s.next_bci(), current);
 309         make_block_at(s.get_dest(), current);
 310         current = NULL;
 311         break;
 312 
 313       case Bytecodes::_goto:
 314         make_block_at(s.get_dest(), current);
 315         current = NULL;
 316         break;
 317 
 318       case Bytecodes::_goto_w:
 319         make_block_at(s.get_far_dest(), current);
 320         current = NULL;
 321         break;
 322 
 323       case Bytecodes::_jsr:
 324         handle_jsr(current, s.get_dest(), s.next_bci());
 325         current = NULL;
 326         break;
 327 
 328       case Bytecodes::_jsr_w:
 329         handle_jsr(current, s.get_far_dest(), s.next_bci());
 330         current = NULL;
 331         break;
 332 
 333       case Bytecodes::_tableswitch: {
 334         // set block for each case
 335         Bytecode_tableswitch sw(&s);
 336         int l = sw.length();
 337         for (int i = 0; i < l; i++) {
 338           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 339         }
 340         make_block_at(cur_bci + sw.default_offset(), current);
 341         current = NULL;
 342         break;
 343       }
 344 
 345       case Bytecodes::_lookupswitch: {
 346         // set block for each case
 347         Bytecode_lookupswitch sw(&s);
 348         int l = sw.number_of_pairs();
 349         for (int i = 0; i < l; i++) {
 350           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 351         }
 352         make_block_at(cur_bci + sw.default_offset(), current);
 353         current = NULL;
 354         break;
 355       }
 356 
 357       default:
 358         break;
 359     }
 360   }
 361 }

 364 void BlockListBuilder::mark_loops() {
 365   ResourceMark rm;
 366 
 367   _active.initialize(BlockBegin::number_of_blocks());
 368   _visited.initialize(BlockBegin::number_of_blocks());
 369   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 370   _next_loop_index = 0;
 371   _next_block_number = _blocks.length();
 372 
 373   // recursively iterate the control flow graph
 374   mark_loops(_bci2block->at(0), false);
 375   assert(_next_block_number >= 0, "invalid block numbers");
 376 
 377   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 378   _active.resize(0);
 379   _visited.resize(0);
 380 }

 382 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 383   if (block->is_set(BlockBegin::exception_entry_flag)) {
 384     // exception edges may look like loops but don't mark them as such
 385     // since it screws up block ordering.
 386     return;
 387   }
 388   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 389     block->set(BlockBegin::parser_loop_header_flag);
 390 
 391     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 392     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 393     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 394     if (_next_loop_index < 31) _next_loop_index++;
 395   } else {
 396     // block already marked as loop header
 397     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 398   }
 399 }

 401 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 402   int block_id = block->block_id();
 403 
 404   if (_visited.at(block_id)) {
 405     if (_active.at(block_id)) {
 406       // reached block via backward branch
 407       make_loop_header(block);
 408     }
 409     // return cached loop information for this block
 410     return _loop_map.at(block_id);
 411   }
 412 
 413   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 414     in_subroutine = true;
 415   }
 416 
 417   // set active and visited bits before successors are processed
 418   _visited.set_bit(block_id);
 419   _active.set_bit(block_id);
 420 
 421   intptr_t loop_state = 0;
 422   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 423     // recursively process all successors
 424     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 425   }
 426 
 427   // clear active-bit after all successors are processed
 428   _active.clear_bit(block_id);
 429 
 430   // reverse-post-order numbering of all blocks
 431   block->set_depth_first_number(_next_block_number);
 432   _next_block_number--;
 433 
 434   if (loop_state != 0 || in_subroutine ) {
 435     // block is contained at least in one loop, so phi functions are necessary
 436     // phi functions are also necessary for all locals stored in a subroutine
 437     scope()->requires_phi_function().set_union(block->stores_to_locals());
 438   }
 439 
 440   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 441     int header_loop_state = _loop_map.at(block_id);
 442     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 443 
 444     // If the highest bit is set (i.e. when integer value is negative), the method
 445     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 446     if (header_loop_state >= 0) {
 447       clear_bits(loop_state, header_loop_state);
 448     }
 449   }
 450 
 451   // cache and return loop information for this block
 452   _loop_map.at_put(block_id, loop_state);
 453   return loop_state;
 454 }

 463 void BlockListBuilder::print() {
 464   tty->print("----- initial block list of BlockListBuilder for method ");
 465   method()->print_short_name();
 466   tty->cr();
 467 
 468   // better readability if blocks are sorted in processing order
 469   _blocks.sort(compare_depth_first);
 470 
 471   for (int i = 0; i < _blocks.length(); i++) {
 472     BlockBegin* cur = _blocks.at(i);
 473     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 474 
 475     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 476     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 477     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 478     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 479     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 480 
 481     if (cur->number_of_sux() > 0) {
 482       tty->print("    sux: ");
 483       for (int j = 0; j < cur->number_of_sux(); j++) {
 484         BlockBegin* sux = cur->sux_at(j);
 485         tty->print("B%d ", sux->block_id());
 486       }
 487     }
 488     tty->cr();
 489   }
 490 }

 495 // A simple growable array of Values indexed by ciFields
 496 class FieldBuffer: public CompilationResourceObj {
 497  private:
 498   GrowableArray<Value> _values;
 499 
 500  public:
 501   FieldBuffer() {}
 502 
 503   void kill() {
 504     _values.trunc_to(0);
 505   }
 506 
 507   Value at(ciField* field) {
 508     assert(field->holder()->is_loaded(), "must be a loaded field");
 509     int offset = field->offset();
 510     if (offset < _values.length()) {
 511       return _values.at(offset);
 512     } else {
 513       return NULL;
 514     }
 515   }
 516 
 517   void at_put(ciField* field, Value value) {
 518     assert(field->holder()->is_loaded(), "must be a loaded field");
 519     int offset = field->offset();
 520     _values.at_put_grow(offset, value, NULL);
 521   }
 522 
 523 };

 526 // MemoryBuffer is fairly simple model of the current state of memory.
 527 // It partitions memory into several pieces.  The first piece is
 528 // generic memory where little is known about the owner of the memory.
 529 // This is conceptually represented by the tuple <O, F, V> which says
 530 // that the field F of object O has value V.  This is flattened so
 531 // that F is represented by the offset of the field and the parallel
 532 // arrays _objects and _values are used for O and V.  Loads of O.F can
 533 // simply use V.  Newly allocated objects are kept in a separate list
 534 // along with a parallel array for each object which represents the
 535 // current value of its fields.  Stores of the default value to fields
 536 // which have never been stored to before are eliminated since they
 537 // are redundant.  Once newly allocated objects are stored into
 538 // another object or they are passed out of the current compile they
 539 // are treated like generic memory.
 540 
 541 class MemoryBuffer: public CompilationResourceObj {
 542  private:
 543   FieldBuffer                 _values;
 544   GrowableArray<Value>        _objects;
 545   GrowableArray<Value>        _newobjects;
 546   GrowableArray<FieldBuffer*> _fields;
 547 
 548  public:
 549   MemoryBuffer() {}
 550 
 551   StoreField* store(StoreField* st) {
 552     if (!EliminateFieldAccess) {
 553       return st;
 554     }
 555 
 556     Value object = st->obj();
 557     Value value = st->value();
 558     ciField* field = st->field();
 559     if (field->holder()->is_loaded()) {
 560       int offset = field->offset();
 561       int index = _newobjects.find(object);
 562       if (index != -1) {
 563         // newly allocated object with no other stores performed on this field
 564         FieldBuffer* buf = _fields.at(index);
 565         if (buf->at(field) == NULL && is_default_value(value)) {
 566 #ifndef PRODUCT
 567           if (PrintIRDuringConstruction && Verbose) {
 568             tty->print_cr("Eliminated store for object %d:", index);
 569             st->print_line();
 570           }
 571 #endif
 572           return NULL;
 573         } else {
 574           buf->at_put(field, value);
 575         }
 576       } else {
 577         _objects.at_put_grow(offset, object, NULL);
 578         _values.at_put(field, value);
 579       }
 580 
 581       store_value(value);
 582     } else {
 583       // if we held onto field names we could alias based on names but
 584       // we don't know what's being stored to so kill it all.
 585       kill();
 586     }
 587     return st;
 588   }
 589 
 590 
 591   // return true if this value correspond to the default value of a field.
 592   bool is_default_value(Value value) {
 593     Constant* con = value->as_Constant();
 594     if (con) {
 595       switch (con->type()->tag()) {
 596         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 597         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 598         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 599         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 600         case objectTag: return con->type() == objectNull;
 601         default:  ShouldNotReachHere();
 602       }
 603     }
 604     return false;
 605   }
 606 
 607 
 608   // return either the actual value of a load or the load itself
 609   Value load(LoadField* load) {
 610     if (!EliminateFieldAccess) {
 611       return load;
 612     }
 613 
 614     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 615 #ifdef IA32
 616       if (UseSSE < 2) {
 617         // can't skip load since value might get rounded as a side effect
 618         return load;
 619       }
 620 #else
 621       Unimplemented();
 622 #endif // IA32
 623     }
 624 
 625     ciField* field = load->field();
 626     Value object   = load->obj();
 627     if (field->holder()->is_loaded() && !field->is_volatile()) {
 628       int offset = field->offset();
 629       Value result = NULL;
 630       int index = _newobjects.find(object);
 631       if (index != -1) {
 632         result = _fields.at(index)->at(field);
 633       } else if (_objects.at_grow(offset, NULL) == object) {
 634         result = _values.at(field);
 635       }
 636       if (result != NULL) {
 637 #ifndef PRODUCT
 638         if (PrintIRDuringConstruction && Verbose) {
 639           tty->print_cr("Eliminated load: ");
 640           load->print_line();
 641         }
 642 #endif
 643         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 644         return result;
 645       }
 646     }
 647     return load;
 648   }
 649 
 650   // Record this newly allocated object
 651   void new_instance(NewInstance* object) {
 652     int index = _newobjects.length();
 653     _newobjects.append(object);
 654     if (_fields.at_grow(index, NULL) == NULL) {
 655       _fields.at_put(index, new FieldBuffer());
 656     } else {
 657       _fields.at(index)->kill();
 658     }
 659   }
 660 
 661   // Record this newly allocated object
 662   void new_instance(NewInlineTypeInstance* object) {
 663     int index = _newobjects.length();
 664     _newobjects.append(object);
 665     if (_fields.at_grow(index, NULL) == NULL) {
 666       _fields.at_put(index, new FieldBuffer());
 667     } else {
 668       _fields.at(index)->kill();
 669     }
 670   }
 671 
 672   void store_value(Value value) {
 673     int index = _newobjects.find(value);
 674     if (index != -1) {
 675       // stored a newly allocated object into another object.
 676       // Assume we've lost track of it as separate slice of memory.
 677       // We could do better by keeping track of whether individual
 678       // fields could alias each other.
 679       _newobjects.remove_at(index);
 680       // pull out the field info and store it at the end up the list
 681       // of field info list to be reused later.
 682       _fields.append(_fields.at(index));
 683       _fields.remove_at(index);
 684     }
 685   }
 686 
 687   void kill() {
 688     _newobjects.trunc_to(0);
 689     _objects.trunc_to(0);
 690     _values.kill();
 691   }
 692 };

 733 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 734   if (parsing_jsr()) {
 735     // It is necessary to clone all blocks associated with a
 736     // subroutine, including those for exception handlers in the scope
 737     // of the method containing the jsr (because those exception
 738     // handlers may contain ret instructions in some cases).
 739     BlockBegin* block = bci2block()->at(bci);
 740     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 741       BlockBegin* new_block = new BlockBegin(block->bci());
 742       if (PrintInitialBlockList) {
 743         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 744                       block->block_id(), block->bci(), new_block->block_id());
 745       }
 746       // copy data from cloned blocked
 747       new_block->set_depth_first_number(block->depth_first_number());
 748       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 749       // Preserve certain flags for assertion checking
 750       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 751       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 752 
 753       // copy was_visited_flag to allow early detection of bailouts
 754       // if a block that is used in a jsr has already been visited before,
 755       // it is shared between the normal control flow and a subroutine
 756       // BlockBegin::try_merge returns false when the flag is set, this leads
 757       // to a compilation bailout
 758       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 759 
 760       bci2block()->at_put(bci, new_block);
 761       block = new_block;
 762     }
 763     return block;
 764   } else {
 765     return bci2block()->at(bci);
 766   }
 767 }

 824 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 825   assert(worklist->top() == top, "");
 826   // sort block descending into work list
 827   const int dfn = top->depth_first_number();
 828   assert(dfn != -1, "unknown depth first number");
 829   int i = worklist->length()-2;
 830   while (i >= 0) {
 831     BlockBegin* b = worklist->at(i);
 832     if (b->depth_first_number() < dfn) {
 833       worklist->at_put(i+1, b);
 834     } else {
 835       break;
 836     }
 837     i --;
 838   }
 839   if (i >= -1) worklist->at_put(i + 1, top);
 840 }

 897 void GraphBuilder::load_constant() {
 898   ciConstant con = stream()->get_constant();
 899   if (con.basic_type() == T_ILLEGAL) {
 900     // FIXME: an unresolved Dynamic constant can get here,
 901     // and that should not terminate the whole compilation.
 902     BAILOUT("could not resolve a constant");
 903   } else {
 904     ValueType* t = illegalType;
 905     ValueStack* patch_state = NULL;
 906     switch (con.basic_type()) {
 907       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 908       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 909       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 910       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 911       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 912       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 913       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 914       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 915       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 916       case T_OBJECT :
 917        {
 918         ciObject* obj = con.as_object();
 919         if (!obj->is_loaded()
 920             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 921           // A Class, MethodType, MethodHandle, or String.
 922           // Unloaded condy nodes show up as T_ILLEGAL, above.
 923           patch_state = copy_state_before();
 924           t = new ObjectConstant(obj);
 925         } else {
 926           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 927           // result, which might turn out to be an array.
 928           if (obj->is_null_object())
 929             t = objectNull;
 930           else if (obj->is_array())
 931             t = new ArrayConstant(obj->as_array());
 932           else
 933             t = new InstanceConstant(obj->as_instance());
 934         }
 935         break;
 936        }
 937       default       : ShouldNotReachHere();
 938     }
 939     Value x;
 940     if (patch_state != NULL) {
 941       x = new Constant(t, patch_state);
 942     } else {
 943       x = new Constant(t);
 944     }
 945     push(t, append(x));
 946   }
 947 }

 974 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 975   if (parsing_jsr()) {
 976     // We need to do additional tracking of the location of the return
 977     // address for jsrs since we don't handle arbitrary jsr/ret
 978     // constructs. Here we are figuring out in which circumstances we
 979     // need to bail out.
 980     if (x->type()->is_address()) {
 981       scope_data()->set_jsr_return_address_local(index);
 982 
 983       // Also check parent jsrs (if any) at this time to see whether
 984       // they are using this local. We don't handle skipping over a
 985       // ret.
 986       for (ScopeData* cur_scope_data = scope_data()->parent();
 987            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 988            cur_scope_data = cur_scope_data->parent()) {
 989         if (cur_scope_data->jsr_return_address_local() == index) {
 990           BAILOUT("subroutine overwrites return address from previous subroutine");
 991         }
 992       }
 993     } else if (index == scope_data()->jsr_return_address_local()) {
 994       scope_data()->set_jsr_return_address_local(-1);
 995     }
 996   }
 997 
 998   x->set_local_index(index);
 999   state->store_local(index, round_fp(x));
1000   if (x->as_NewInlineTypeInstance() != NULL) {
1001     x->as_NewInlineTypeInstance()->set_local_index(index);
1002     x->as_NewInlineTypeInstance()->decrement_on_stack_count();
1003   }
1004 }

1007 void GraphBuilder::load_indexed(BasicType type) {
1008   // In case of in block code motion in range check elimination
1009   ValueStack* state_before = NULL;
1010   int array_idx = state()->stack_size() - 2;
1011   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
1012     // Save the entire state and re-execute on deopt when accessing flattened arrays
1013     state_before = copy_state_before();
1014     state_before->set_should_reexecute(true);
1015   } else {
1016     state_before = copy_state_indexed_access();
1017   }
1018   compilation()->set_has_access_indexed(true);
1019   Value index = ipop();
1020   Value array = apop();
1021   Value length = NULL;
1022   if (CSEArrayLength ||
1023       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1024       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1025     length = append(new ArrayLength(array, state_before));
1026   }
1027 
1028   LoadIndexed* load_indexed = NULL;
1029   Instruction* result = NULL;
1030   if (array->is_loaded_flattened_array()) {
1031     ciType* array_type = array->declared_type();
1032     ciInlineKlass* elem_klass = array_type->as_flat_array_klass()->element_klass()->as_inline_klass();
1033     NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(elem_klass, state_before, false);
1034     _memory->new_instance(new_instance);
1035     apush(append_split(new_instance));
1036     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1037     load_indexed->set_vt(new_instance);
1038   } else {
1039     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1040   }
1041   if (profile_array_accesses() && is_reference_type(type)) {
1042     compilation()->set_would_profile(true);
1043     load_indexed->set_should_profile(true);
1044     load_indexed->set_profiled_method(method());
1045     load_indexed->set_profiled_bci(bci());
1046   }
1047   result = append(load_indexed);
1048   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1049   if (!array->is_loaded_flattened_array()) {
1050     push(as_ValueType(type), result);
1051   }
1052 }

1103 void GraphBuilder::stack_op(Bytecodes::Code code) {
1104   switch (code) {
1105     case Bytecodes::_pop:
1106       { Value w = state()->raw_pop();
1107         update_larva_stack_count(w);
1108       }
1109       break;
1110     case Bytecodes::_pop2:
1111       { Value w1 = state()->raw_pop();
1112         Value w2 = state()->raw_pop();
1113         update_larva_stack_count(w1);
1114         update_larva_stack_count(w2);
1115       }
1116       break;
1117     case Bytecodes::_dup:
1118       { Value w = state()->raw_pop();
1119         update_larval_state(w);
1120         state()->raw_push(w);
1121         state()->raw_push(w);
1122       }
1123       break;
1124     case Bytecodes::_dup_x1:
1125       { Value w1 = state()->raw_pop();
1126         Value w2 = state()->raw_pop();
1127         update_larval_state(w1);
1128         state()->raw_push(w1);
1129         state()->raw_push(w2);
1130         state()->raw_push(w1);
1131       }
1132       break;
1133     case Bytecodes::_dup_x2:
1134       { Value w1 = state()->raw_pop();
1135         Value w2 = state()->raw_pop();
1136         Value w3 = state()->raw_pop();
1137         // special handling for the dup_x2/pop sequence (see JDK-8251046)
1138         if (w1 != NULL && w1->as_NewInlineTypeInstance() != NULL) {
1139           ciBytecodeStream s(method());
1140           s.force_bci(bci());
1141           s.next();
1142           if (s.cur_bc() != Bytecodes::_pop) {
1143             w1->as_NewInlineTypeInstance()->set_not_larva_anymore();
1144           }  else {
1145             w1->as_NewInlineTypeInstance()->increment_on_stack_count();
1146            }
1147         }
1148         state()->raw_push(w1);
1149         state()->raw_push(w3);
1150         state()->raw_push(w2);
1151         state()->raw_push(w1);
1152       }
1153       break;
1154     case Bytecodes::_dup2:
1155       { Value w1 = state()->raw_pop();
1156         Value w2 = state()->raw_pop();
1157         update_larval_state(w1);
1158         update_larval_state(w2);
1159         state()->raw_push(w2);
1160         state()->raw_push(w1);
1161         state()->raw_push(w2);
1162         state()->raw_push(w1);
1163       }
1164       break;
1165     case Bytecodes::_dup2_x1:
1166       { Value w1 = state()->raw_pop();
1167         Value w2 = state()->raw_pop();
1168         Value w3 = state()->raw_pop();
1169         update_larval_state(w1);
1170         update_larval_state(w2);
1171         state()->raw_push(w2);
1172         state()->raw_push(w1);
1173         state()->raw_push(w3);
1174         state()->raw_push(w2);
1175         state()->raw_push(w1);
1176       }
1177       break;
1178     case Bytecodes::_dup2_x2:
1179       { Value w1 = state()->raw_pop();
1180         Value w2 = state()->raw_pop();
1181         Value w3 = state()->raw_pop();
1182         Value w4 = state()->raw_pop();
1183         update_larval_state(w1);
1184         update_larval_state(w2);
1185         state()->raw_push(w2);
1186         state()->raw_push(w1);
1187         state()->raw_push(w4);
1188         state()->raw_push(w3);
1189         state()->raw_push(w2);
1190         state()->raw_push(w1);
1191       }
1192       break;
1193     case Bytecodes::_swap:
1194       { Value w1 = state()->raw_pop();
1195         Value w2 = state()->raw_pop();
1196         state()->raw_push(w1);
1197         state()->raw_push(w2);
1198       }
1199       break;
1200     default:
1201       ShouldNotReachHere();
1202       break;
1203   }
1204 }

1216     res = round_fp(res);
1217   }

1227 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1228   Value s = ipop();
1229   Value x = pop(type);
1230   // try to simplify
1231   // Note: This code should go into the canonicalizer as soon as it can
1232   //       can handle canonicalized forms that contain more than one node.
1233   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1234     // pattern: x >>> s
1235     IntConstant* s1 = s->type()->as_IntConstant();
1236     if (s1 != NULL) {
1237       // pattern: x >>> s1, with s1 constant
1238       ShiftOp* l = x->as_ShiftOp();
1239       if (l != NULL && l->op() == Bytecodes::_ishl) {
1240         // pattern: (a << b) >>> s1
1241         IntConstant* s0 = l->y()->type()->as_IntConstant();
1242         if (s0 != NULL) {
1243           // pattern: (a << s0) >>> s1
1244           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1245           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1246           if (s0c == s1c) {
1247             if (s0c == 0) {
1248               // pattern: (a << 0) >>> 0 => simplify to: a
1249               ipush(l->x());
1250             } else {
1251               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1252               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1253               const int m = (1 << (BitsPerInt - s0c)) - 1;
1254               Value s = append(new Constant(new IntConstant(m)));
1255               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1256             }
1257             return;
1258           }
1259         }
1260       }
1261     }
1262   }
1263   // could not simplify
1264   push(type, append(new ShiftOp(code, x, s)));
1265 }

1312 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1313   BlockBegin* tsux = block_at(stream()->get_dest());
1314   BlockBegin* fsux = block_at(stream()->next_bci());
1315   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1316 
1317   bool subst_check = false;
1318   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne)) {
1319     ValueType* left_vt = x->type();
1320     ValueType* right_vt = y->type();
1321     if (left_vt->is_object()) {
1322       assert(right_vt->is_object(), "must be");
1323       ciKlass* left_klass = x->as_loaded_klass_or_null();
1324       ciKlass* right_klass = y->as_loaded_klass_or_null();
1325 
1326       if (left_klass == NULL || right_klass == NULL) {
1327         // The klass is still unloaded, or came from a Phi node. Go slow case;
1328         subst_check = true;
1329       } else if (left_klass->can_be_inline_klass() || right_klass->can_be_inline_klass()) {
1330         // Either operand may be a value object, but we're not sure. Go slow case;
1331         subst_check = true;
1332       } else {
1333         // No need to do substitutability check
1334       }
1335     }
1336   }
1337 
1338   // In case of loop invariant code motion or predicate insertion
1339   // before the body of a loop the state is needed
1340   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1341 
1342   assert(i->as_Goto() == NULL ||
1343          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1344          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1345          "safepoint state of Goto returned by canonicalizer incorrect");
1346 
1347   if (is_profiling()) {
1348     If* if_node = i->as_If();
1349     if (if_node != NULL) {
1350       // Note that we'd collect profile data in this method if we wanted it.
1351       compilation()->set_would_profile(true);
1352       // At level 2 we need the proper bci to count backedges
1353       if_node->set_profiled_bci(bci());
1354       if (profile_branches()) {
1355         // Successors can be rotated by the canonicalizer, check for this case.
1356         if_node->set_profiled_method(method());
1357         if_node->set_should_profile(true);
1358         if (if_node->tsux() == fsux) {
1359           if_node->set_swapped(true);
1360         }
1361       }
1362       return;
1363     }
1364 
1365     // Check if this If was reduced to Goto.
1366     Goto *goto_node = i->as_Goto();
1367     if (goto_node != NULL) {
1368       compilation()->set_would_profile(true);
1369       goto_node->set_profiled_bci(bci());
1370       if (profile_branches()) {
1371         goto_node->set_profiled_method(method());
1372         goto_node->set_should_profile(true);
1373         // Find out which successor is used.
1374         if (goto_node->default_sux() == tsux) {
1375           goto_node->set_direction(Goto::taken);
1376         } else if (goto_node->default_sux() == fsux) {
1377           goto_node->set_direction(Goto::not_taken);
1378         } else {
1379           ShouldNotReachHere();
1380         }
1381       }
1382       return;
1383     }
1384   }
1385 }

1414   // If the bytecodes are strange (jumping out of a jsr block) then we

1444 void GraphBuilder::table_switch() {
1445   Bytecode_tableswitch sw(stream());
1446   const int l = sw.length();
1447   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1448     // total of 2 successors => use If instead of switch
1449     // Note: This code should go into the canonicalizer as soon as it can
1450     //       can handle canonicalized forms that contain more than one node.
1451     Value key = append(new Constant(new IntConstant(sw.low_key())));
1452     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1453     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1454     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1455     // In case of loop invariant code motion or predicate insertion
1456     // before the body of a loop the state is needed
1457     ValueStack* state_before = copy_state_if_bb(is_bb);
1458     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1459   } else {
1460     // collect successors
1461     BlockList* sux = new BlockList(l + 1, NULL);
1462     int i;
1463     bool has_bb = false;
1464     for (i = 0; i < l; i++) {
1465       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1466       if (sw.dest_offset_at(i) < 0) has_bb = true;
1467     }
1468     // add default successor
1469     if (sw.default_offset() < 0) has_bb = true;
1470     sux->at_put(i, block_at(bci() + sw.default_offset()));
1471     // In case of loop invariant code motion or predicate insertion
1472     // before the body of a loop the state is needed
1473     ValueStack* state_before = copy_state_if_bb(has_bb);
1474     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1475 #ifdef ASSERT
1476     if (res->as_Goto()) {
1477       for (i = 0; i < l; i++) {
1478         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1479           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1480         }
1481       }
1482     }
1483 #endif
1484   }
1485 }

1488 void GraphBuilder::lookup_switch() {
1489   Bytecode_lookupswitch sw(stream());
1490   const int l = sw.number_of_pairs();
1491   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1492     // total of 2 successors => use If instead of switch
1493     // Note: This code should go into the canonicalizer as soon as it can
1494     //       can handle canonicalized forms that contain more than one node.
1495     // simplify to If
1496     LookupswitchPair pair = sw.pair_at(0);
1497     Value key = append(new Constant(new IntConstant(pair.match())));
1498     BlockBegin* tsux = block_at(bci() + pair.offset());
1499     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1500     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1501     // In case of loop invariant code motion or predicate insertion
1502     // before the body of a loop the state is needed
1503     ValueStack* state_before = copy_state_if_bb(is_bb);;
1504     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1505   } else {
1506     // collect successors & keys
1507     BlockList* sux = new BlockList(l + 1, NULL);
1508     intArray* keys = new intArray(l, l, 0);
1509     int i;
1510     bool has_bb = false;
1511     for (i = 0; i < l; i++) {
1512       LookupswitchPair pair = sw.pair_at(i);
1513       if (pair.offset() < 0) has_bb = true;
1514       sux->at_put(i, block_at(bci() + pair.offset()));
1515       keys->at_put(i, pair.match());
1516     }
1517     // add default successor
1518     if (sw.default_offset() < 0) has_bb = true;
1519     sux->at_put(i, block_at(bci() + sw.default_offset()));
1520     // In case of loop invariant code motion or predicate insertion
1521     // before the body of a loop the state is needed
1522     ValueStack* state_before = copy_state_if_bb(has_bb);
1523     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1524 #ifdef ASSERT
1525     if (res->as_Goto()) {
1526       for (i = 0; i < l; i++) {
1527         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1528           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1529         }
1530       }
1531     }
1532 #endif
1533   }
1534 }

1536 void GraphBuilder::call_register_finalizer() {
1537   // If the receiver requires finalization then emit code to perform
1538   // the registration on return.
1539 
1540   // Gather some type information about the receiver
1541   Value receiver = state()->local_at(0);
1542   assert(receiver != NULL, "must have a receiver");
1543   ciType* declared_type = receiver->declared_type();
1544   ciType* exact_type = receiver->exact_type();
1545   if (exact_type == NULL &&
1546       receiver->as_Local() &&
1547       receiver->as_Local()->java_index() == 0) {
1548     ciInstanceKlass* ik = compilation()->method()->holder();
1549     if (ik->is_final()) {
1550       exact_type = ik;
1551     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1552       // test class is leaf class
1553       compilation()->dependency_recorder()->assert_leaf_type(ik);
1554       exact_type = ik;
1555     } else {
1556       declared_type = ik;
1557     }
1558   }
1559 
1560   // see if we know statically that registration isn't required
1561   bool needs_check = true;
1562   if (exact_type != NULL) {
1563     needs_check = exact_type->as_instance_klass()->has_finalizer();
1564   } else if (declared_type != NULL) {
1565     ciInstanceKlass* ik = declared_type->as_instance_klass();
1566     if (!Dependencies::has_finalizable_subclass(ik)) {
1567       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1568       needs_check = false;
1569     }
1570   }
1571 
1572   if (needs_check) {
1573     // Perform the registration of finalizable objects.
1574     ValueStack* state_before = copy_state_for_exception();
1575     load_local(objectType, 0);
1576     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1577                                state()->pop_arguments(1),
1578                                true, state_before, true));
1579   }
1580 }

1583 void GraphBuilder::method_return(Value x, bool ignore_return) {
1584   if (RegisterFinalizersAtInit &&
1585       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1586     call_register_finalizer();
1587   }
1588 
1589   // The conditions for a memory barrier are described in Parse::do_exits().
1590   bool need_mem_bar = false;
1591   if (method()->is_object_constructor() &&
1592        (scope()->wrote_final() ||
1593          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1594          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1595     need_mem_bar = true;
1596   }
1597 
1598   BasicType bt = method()->return_type()->basic_type();
1599   switch (bt) {
1600     case T_BYTE:
1601     {
1602       Value shift = append(new Constant(new IntConstant(24)));
1603       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1604       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1605       break;
1606     }
1607     case T_SHORT:
1608     {
1609       Value shift = append(new Constant(new IntConstant(16)));
1610       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1611       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1612       break;
1613     }
1614     case T_CHAR:
1615     {
1616       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1617       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1618       break;
1619     }
1620     case T_BOOLEAN:
1621     {
1622       Value mask = append(new Constant(new IntConstant(1)));
1623       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1624       break;
1625     }
1626     default:
1627       break;
1628   }
1629 
1630   // Check to see whether we are inlining. If so, Return
1631   // instructions become Gotos to the continuation point.
1632   if (continuation() != NULL) {
1633 
1634     int invoke_bci = state()->caller_state()->bci();
1635 
1636     if (x != NULL  && !ignore_return) {
1637       ciMethod* caller = state()->scope()->caller()->method();
1638       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1639       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1640         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1641         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1642             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1643           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1644         }
1645       }
1646     }
1647 
1648     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1649 
1650     if (compilation()->env()->dtrace_method_probes()) {
1651       // Report exit from inline methods
1652       Values* args = new Values(1);
1653       args->push(append(new Constant(new MethodConstant(method()))));
1654       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1655     }
1656 
1657     // If the inlined method is synchronized, the monitor must be
1658     // released before we jump to the continuation block.
1659     if (method()->is_synchronized()) {
1660       assert(state()->locks_size() == 1, "receiver must be locked here");
1661       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1662     }
1663 
1664     if (need_mem_bar) {
1665       append(new MemBar(lir_membar_storestore));
1666     }
1667 
1668     // State at end of inlined method is the state of the caller
1669     // without the method parameters on stack, including the
1670     // return value, if any, of the inlined method on operand stack.
1671     set_state(state()->caller_state()->copy_for_parsing());
1672     if (x != NULL) {
1673       if (!ignore_return) {
1674         state()->push(x->type(), x);
1675       }
1676       if (profile_return() && x->type()->is_object_kind()) {
1677         ciMethod* caller = state()->scope()->method();
1678         profile_return_type(x, method(), caller, invoke_bci);
1679       }
1680     }
1681     Goto* goto_callee = new Goto(continuation(), false);
1682 
1683     // See whether this is the first return; if so, store off some
1684     // of the state for later examination
1685     if (num_returns() == 0) {
1686       set_inline_cleanup_info();
1687     }
1688 
1689     // The current bci() is in the wrong scope, so use the bci() of
1690     // the continuation point.
1691     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1692     incr_num_returns();
1693     return;
1694   }
1695 
1696   state()->truncate_stack(0);
1697   if (method()->is_synchronized()) {
1698     // perform the unlocking before exiting the method
1699     Value receiver;
1700     if (!method()->is_static()) {
1701       receiver = _initial_state->local_at(0);
1702     } else {
1703       receiver = append(new Constant(new ClassConstant(method()->holder())));
1704     }
1705     append_split(new MonitorExit(receiver, state()->unlock()));
1706   }
1707 
1708   if (need_mem_bar) {
1709       append(new MemBar(lir_membar_storestore));
1710   }
1711 
1712   assert(!ignore_return, "Ignoring return value works only for inlining");
1713   append(new Return(x));
1714 }

1747     assert(!inner_field->is_flattened(), "the iteration over nested fields is handled by the loop itself");
1748     int off = inner_field->offset() - vk->first_field_offset();
1749     LoadField* load = new LoadField(src, src_off + off, inner_field, false, state_before, needs_patching);
1750     Value replacement = append(load);
1751     StoreField* store = new StoreField(dest, dest_off + off, inner_field, replacement, false, state_before, needs_patching);
1752     append(store);
1753   }

1756 void GraphBuilder::access_field(Bytecodes::Code code) {
1757   bool will_link;
1758   ciField* field = stream()->get_field(will_link);
1759   ciInstanceKlass* holder = field->holder();
1760   BasicType field_type = field->type()->basic_type();
1761   ValueType* type = as_ValueType(field_type);
1762 
1763   // call will_link again to determine if the field is valid.
1764   const bool needs_patching = !holder->is_loaded() ||
1765                               !field->will_link(method(), code) ||
1766                               PatchALot;
1767 
1768   ValueStack* state_before = NULL;
1769   if (!holder->is_initialized() || needs_patching) {
1770     // save state before instruction for debug info when
1771     // deoptimization happens during patching
1772     state_before = copy_state_before();
1773   }
1774 
1775   Value obj = NULL;
1776   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1777     if (state_before != NULL) {
1778       // build a patching constant
1779       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1780     } else {
1781       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1782     }
1783   }
1784 
1785   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1786     scope()->set_wrote_final();
1787   }
1788 
1789   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1790     scope()->set_wrote_fields();
1791     if (field->is_volatile()) {
1792       scope()->set_wrote_volatile();
1793     }
1794   }
1795 
1796   const int offset = !needs_patching ? field->offset() : -1;
1797   switch (code) {
1798     case Bytecodes::_getstatic: {
1799       // check for compile-time constants, i.e., initialized static final fields
1800       Value constant = NULL;
1801       if (field->is_static_constant() && !PatchALot) {
1802         ciConstant field_value = field->constant_value();
1803         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1804                "stable static w/ default value shouldn't be a constant");
1805         constant = make_constant(field_value, field);
1806       }
1807       if (constant != NULL) {
1808         push(type, append(constant));
1809       } else {
1810         if (state_before == NULL) {
1811           state_before = copy_state_for_exception();
1812         }
1813         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1814                                         state_before, needs_patching);
1815         push(type, append(load_field));
1816       }
1817       break;
1818     }
1819     case Bytecodes::_putstatic: {
1820       Value val = pop(type);
1821       val->set_escaped();
1822       if (state_before == NULL) {
1823         state_before = copy_state_for_exception();
1824       }
1825       if (field_type == T_BOOLEAN) {
1826         Value mask = append(new Constant(new IntConstant(1)));
1827         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1828       }
1829       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1830       break;
1831     }
1832     case Bytecodes::_getfield: {
1833       // Check for compile-time constants, i.e., trusted final non-static fields.
1834       Value constant = NULL;
1835       if (state_before == NULL && field->is_flattened()) {
1836         // Save the entire state and re-execute on deopt when accessing flattened fields
1837         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1838         state_before = copy_state_before();
1839       }
1840       obj = apop();
1841       ObjectType* obj_type = obj->type()->as_ObjectType();
1842       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1843         ciObject* const_oop = obj_type->constant_value();
1844         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1845           ciConstant field_value = field->constant_value_of(const_oop);
1846           if (field_value.is_valid()) {
1847             if (field->signature()->is_Q_signature() && field_value.is_null_or_zero()) {
1848               // Non-flattened inline type field. Replace null by the default value.
1849               constant = new Constant(new InstanceConstant(field->type()->as_inline_klass()->default_instance()));
1850             } else {
1851               constant = make_constant(field_value, field);
1852             }
1853             // For CallSite objects add a dependency for invalidation of the optimization.
1854             if (field->is_call_site_target()) {
1855               ciCallSite* call_site = const_oop->as_call_site();
1856               if (!call_site->is_fully_initialized_constant_call_site()) {
1857                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1858                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1859               }
1860             }
1861           }
1862         }
1863       }
1864       if (constant != NULL) {
1865         push(type, append(constant));
1866       } else {
1867         if (state_before == NULL) {
1868           state_before = copy_state_for_exception();
1869         }
1870         if (!field->is_flattened()) {
1871           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1872           Value replacement = !needs_patching ? _memory->load(load) : load;
1873           if (replacement != load) {
1874             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1875             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1876             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1877             BasicType bt = field->type()->basic_type();
1878             switch (bt) {
1879             case T_BOOLEAN:
1880             case T_BYTE:
1881               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1882               break;
1883             case T_CHAR:
1884               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1885               break;
1886             case T_SHORT:
1887               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1888               break;
1889             default:
1890               break;
1891             }
1892             push(type, replacement);
1893           } else {
1894             push(type, append(load));
1895           }
1896         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1897           assert(field->type()->is_inlinetype(), "Sanity check");
1898           ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1899           int flattening_offset = field->offset() - inline_klass->first_field_offset();
1900           assert(field->type()->is_inlinetype(), "Sanity check");
1901           scope()->set_wrote_final();
1902           scope()->set_wrote_fields();
1903           NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(inline_klass, state_before, false);
1904           _memory->new_instance(new_instance);
1905           apush(append_split(new_instance));
1906           copy_inline_content(inline_klass, obj, field->offset(), new_instance, inline_klass->first_field_offset(),
1907                        state_before, needs_patching);
1908         }
1909       }
1910       break;
1911     }
1912     case Bytecodes::_withfield:
1913     case Bytecodes::_putfield: {
1914       Value val = pop(type);
1915       val->set_escaped();
1916       obj = apop();
1917       if (state_before == NULL) {
1918         state_before = copy_state_for_exception();
1919       }
1920       if (field->type()->basic_type() == T_BOOLEAN) {
1921         Value mask = append(new Constant(new IntConstant(1)));
1922         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1923       }
1924 
1925       if (!field->is_flattened()) {
1926         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1927         if (!needs_patching) store = _memory->store(store);
1928         if (store != NULL) {
1929           append(store);
1930         }
1931       } else {
1932         assert(field->type()->is_inlinetype(), "Sanity check");
1933         ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1934         int flattening_offset = field->offset() - inline_klass->first_field_offset();
1935         copy_inline_content(inline_klass, val, inline_klass->first_field_offset(), obj, field->offset(),
1936                    state_before, needs_patching);
1937       }
1938       break;
1939     }
1940     default:
1941       ShouldNotReachHere();
1942       break;
1943   }
1944 }

1946 // Baseline version of withfield, allocate every time
1947 void GraphBuilder::withfield(int field_index)
1948 {
1949   bool will_link;
1950   ciField* field_modify = stream()->get_field(will_link);
1951   ciInstanceKlass* holder = field_modify->holder();
1952   BasicType field_type = field_modify->type()->basic_type();
1953   ValueType* type = as_ValueType(field_type);
1954 
1955   // call will_link again to determine if the field is valid.
1956   const bool needs_patching = !holder->is_loaded() ||
1957                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1958                               PatchALot;
1959 
1960 
1961   scope()->set_wrote_final();
1962   scope()->set_wrote_fields();
1963 
1964   const int offset = !needs_patching ? field_modify->offset() : -1;
1965 
1966   if (!holder->is_loaded()
1967       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1968       ) {
1969     ValueStack* state_before = copy_state_before();
1970     Value val = pop(type);
1971     Value obj = apop();
1972     apush(append_split(new WithField(state_before)));
1973     return;
1974   }
1975   ValueStack* state_before = copy_state_before();
1976 
1977   Value val = pop(type);
1978   Value obj = apop();
1979 
1980   assert(holder->is_inlinetype(), "must be a value klass");
1981   // Save the entire state and re-execute on deopt when executing withfield
1982   state_before->set_should_reexecute(true);
1983   NewInlineTypeInstance* new_instance;
1984   if (obj->as_NewInlineTypeInstance() != NULL && obj->as_NewInlineTypeInstance()->in_larval_state()) {
1985     new_instance = obj->as_NewInlineTypeInstance();
1986     apush(append_split(new_instance));
1987   } else {
1988     new_instance = new NewInlineTypeInstance(holder->as_inline_klass(), state_before, false);
1989     _memory->new_instance(new_instance);
1990     apush(append_split(new_instance));
1991 
1992     for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1993       ciField* field = holder->nonstatic_field_at(i);
1994       int off = field->offset();
1995 
1996       if (field->offset() != offset) {
1997         if (field->is_flattened()) {
1998           assert(field->type()->is_inlinetype(), "Sanity check");
1999           assert(field->type()->is_inlinetype(), "Only inline types can be flattened");
2000           ciInlineKlass* vk = field->type()->as_inline_klass();
2001           copy_inline_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2002         } else {
2003           // Only load those fields who are not modified
2004           LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2005           Value replacement = append(load);
2006           StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2007           append(store);
2008         }
2009       }
2010     }
2011   }
2012 
2013   // Field to modify
2014   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2015     Value mask = append(new Constant(new IntConstant(1)));
2016     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2017   }
2018   if (field_modify->is_flattened()) {
2019     assert(field_modify->type()->is_inlinetype(), "Only inline types can be flattened");
2020     ciInlineKlass* vk = field_modify->type()->as_inline_klass();
2021     copy_inline_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2022   } else {
2023     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2024     append(store);
2025   }
2026 }

2034 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2035   int n = 0;
2036   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2037   start = has_receiver ? 1 : 0;
2038   if (profile_arguments()) {
2039     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2040     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2041       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2042     }
2043   }
2044   // If we are inlining then we need to collect arguments to profile parameters for the target
2045   if (profile_parameters() && target != NULL) {
2046     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2047       // The receiver is profiled on method entry so it's included in
2048       // the number of parameters but here we're only interested in
2049       // actual arguments.
2050       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2051     }
2052   }
2053   if (n > 0) {
2054     return new Values(n);
2055   }
2056   return NULL;
2057 }

2088 void GraphBuilder::invoke(Bytecodes::Code code) {
2089   bool will_link;
2090   ciSignature* declared_signature = NULL;
2091   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2092   ciKlass*              holder = stream()->get_declared_method_holder();
2093   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2094   assert(declared_signature != NULL, "cannot be null");
2095   assert(will_link == target->is_loaded(), "");
2096 
2097   ciInstanceKlass* klass = target->holder();
2098   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2099 
2100   // check if CHA possible: if so, change the code to invoke_special
2101   ciInstanceKlass* calling_klass = method()->holder();
2102   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2103   ciInstanceKlass* actual_recv = callee_holder;
2104 
2105   CompileLog* log = compilation()->log();
2106   if (log != NULL)
2107       log->elem("call method='%d' instr='%s'",
2108                 log->identify(target),
2109                 Bytecodes::name(code));
2110 
2111   // invoke-special-super
2112   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2113     ciInstanceKlass* sender_klass =
2114           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2115                                                  calling_klass;
2116     if (sender_klass->is_interface()) {
2117       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2118       Value receiver = state()->stack_at(index);
2119       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2120       c->set_invokespecial_receiver_check();
2121       state()->stack_at_put(index, append_split(c));
2122     }
2123   }
2124 
2125   // Some methods are obviously bindable without any type checks so
2126   // convert them directly to an invokespecial or invokestatic.
2127   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2128     switch (bc_raw) {
2129     case Bytecodes::_invokevirtual:
2130       code = Bytecodes::_invokespecial;
2131       break;
2132     case Bytecodes::_invokehandle:
2133       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2134       break;
2135     default:
2136       break;
2137     }
2138   } else {
2139     if (bc_raw == Bytecodes::_invokehandle) {
2140       assert(!will_link, "should come here only for unlinked call");
2141       code = Bytecodes::_invokespecial;
2142     }
2143   }
2144 
2145   // Push appendix argument (MethodType, CallSite, etc.), if one.
2146   bool patch_for_appendix = false;
2147   int patching_appendix_arg = 0;
2148   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2149     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2150     apush(arg);
2151     patch_for_appendix = true;
2152     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2153   } else if (stream()->has_appendix()) {
2154     ciObject* appendix = stream()->get_appendix();
2155     Value arg = append(new Constant(new ObjectConstant(appendix)));
2156     apush(arg);
2157   }
2158 
2159   ciMethod* cha_monomorphic_target = NULL;
2160   ciMethod* exact_target = NULL;
2161   Value better_receiver = NULL;
2162   if (UseCHA && DeoptC1 && target->is_loaded() &&
2163       !(// %%% FIXME: Are both of these relevant?
2164         target->is_method_handle_intrinsic() ||
2165         target->is_compiled_lambda_form()) &&
2166       !patch_for_appendix) {
2167     Value receiver = NULL;
2168     ciInstanceKlass* receiver_klass = NULL;
2169     bool type_is_exact = false;
2170     // try to find a precise receiver type
2171     if (will_link && !target->is_static()) {
2172       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2173       receiver = state()->stack_at(index);
2174       ciType* type = receiver->exact_type();
2175       if (type != NULL && type->is_loaded() &&
2176           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2177         receiver_klass = (ciInstanceKlass*) type;
2178         type_is_exact = true;
2179       }
2180       if (type == NULL) {
2181         type = receiver->declared_type();
2182         if (type != NULL && type->is_loaded() &&
2183             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2184           receiver_klass = (ciInstanceKlass*) type;
2185           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2186             // Insert a dependency on this type since
2187             // find_monomorphic_target may assume it's already done.
2188             dependency_recorder()->assert_leaf_type(receiver_klass);
2189             type_is_exact = true;
2190           }
2191         }
2192       }
2193     }
2194     if (receiver_klass != NULL && type_is_exact &&
2195         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2196       // If we have the exact receiver type we can bind directly to
2197       // the method to call.
2198       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2199       if (exact_target != NULL) {
2200         target = exact_target;
2201         code = Bytecodes::_invokespecial;
2202       }
2203     }
2204     if (receiver_klass != NULL &&
2205         receiver_klass->is_subtype_of(actual_recv) &&
2206         actual_recv->is_initialized()) {
2207       actual_recv = receiver_klass;
2208     }
2209 
2210     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2211         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2212       // Use CHA on the receiver to select a more precise method.
2213       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2214     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2215       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2216       // If there is only one implementor of this interface then we
2217       // may be able bind this invoke directly to the implementing
2218       // klass but we need both a dependence on the single interface
2219       // and on the method we bind to.  Additionally since all we know
2220       // about the receiver type is the it's supposed to implement the
2221       // interface we have to insert a check that it's the class we
2222       // expect.  Interface types are not checked by the verifier so
2223       // they are roughly equivalent to Object.
2224       // The number of implementors for declared_interface is less or
2225       // equal to the number of implementors for target->holder() so
2226       // if number of implementors of target->holder() == 1 then
2227       // number of implementors for decl_interface is 0 or 1. If
2228       // it's 0 then no class implements decl_interface and there's
2229       // no point in inlining.
2230       ciInstanceKlass* declared_interface = callee_holder;
2231       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2232       if (singleton != NULL &&
2233           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2234         assert(singleton != declared_interface, "not a unique implementor");
2235         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2236         if (cha_monomorphic_target != NULL) {
2237           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2238             // If CHA is able to bind this invoke then update the class
2239             // to match that class, otherwise klass will refer to the
2240             // interface.
2241             klass = cha_monomorphic_target->holder();
2242             actual_recv = declared_interface;
2243 
2244             // insert a check it's really the expected class.
2245             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2246             c->set_incompatible_class_change_check();
2247             c->set_direct_compare(klass->is_final());
2248             // pass the result of the checkcast so that the compiler has
2249             // more accurate type info in the inlinee
2250             better_receiver = append_split(c);
2251           } else {
2252             cha_monomorphic_target = NULL; // subtype check against Object is useless
2253           }
2254         }
2255       }
2256     }
2257   }
2258 
2259   if (cha_monomorphic_target != NULL) {
2260     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2261     assert(!cha_monomorphic_target->is_abstract(), "");
2262     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2263       // If we inlined because CHA revealed only a single target method,
2264       // then we are dependent on that target method not getting overridden
2265       // by dynamic class loading.  Be sure to test the "static" receiver
2266       // dest_method here, as opposed to the actual receiver, which may
2267       // falsely lead us to believe that the receiver is final or private.
2268       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2269     }
2270     code = Bytecodes::_invokespecial;
2271   }
2272 
2273   // check if we could do inlining
2274   if (!PatchALot && Inline && target->is_loaded() &&
2275       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2276       && !patch_for_appendix) {
2277     // callee is known => check if we have static binding
2278     if (code == Bytecodes::_invokestatic  ||
2279         code == Bytecodes::_invokespecial ||
2280         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2281         code == Bytecodes::_invokedynamic) {
2282       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2283       // static binding => check if callee is ok
2284       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2285 
2286       CHECK_BAILOUT();
2287       clear_inline_bailout();
2288 
2289       if (success) {
2290         // Register dependence if JVMTI has either breakpoint
2291         // setting or hotswapping of methods capabilities since they may
2292         // cause deoptimization.
2293         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2294           dependency_recorder()->assert_evol_method(inline_target);
2295         }
2296         return;
2297       }
2298     } else {
2299       print_inlining(target, "no static binding", /*success*/ false);
2300     }
2301   } else {
2302     print_inlining(target, "not inlineable", /*success*/ false);
2303   }
2304 
2305   // If we attempted an inline which did not succeed because of a
2306   // bailout during construction of the callee graph, the entire
2307   // compilation has to be aborted. This is fairly rare and currently
2308   // seems to only occur for jasm-generated classes which contain
2309   // jsr/ret pairs which are not associated with finally clauses and
2310   // do not have exception handlers in the containing method, and are
2311   // therefore not caught early enough to abort the inlining without
2312   // corrupting the graph. (We currently bail out with a non-empty
2313   // stack at a ret in these situations.)
2314   CHECK_BAILOUT();
2315 
2316   // inlining not successful => standard invoke
2317   ValueType* result_type = as_ValueType(declared_signature->return_type());
2318   ValueStack* state_before = copy_state_exhandling();
2319 
2320   // The bytecode (code) might change in this method so we are checking this very late.
2321   const bool has_receiver =
2322     code == Bytecodes::_invokespecial   ||
2323     code == Bytecodes::_invokevirtual   ||
2324     code == Bytecodes::_invokeinterface;
2325   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2326   Value recv = has_receiver ? apop() : NULL;
2327   int vtable_index = Method::invalid_vtable_index;
2328 
2329   // A null check is required here (when there is a receiver) for any of the following cases
2330   // - invokespecial, always need a null check.
2331   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2332   //   and require null checking. If the target is loaded a null check is emitted here.
2333   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2334   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2335   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2336   //   potentially fail, and can't have the null check before the resolution.)
2337   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2338   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2339   //
2340   // Normal invokevirtual will perform the null check during lookup
2341 
2342   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2343       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2344 
2345   if (need_null_check) {
2346     if (recv != NULL) {
2347       null_check(recv);
2348     }
2349 
2350     if (is_profiling()) {
2351       // Note that we'd collect profile data in this method if we wanted it.
2352       compilation()->set_would_profile(true);
2353 
2354       if (profile_calls()) {
2355         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2356         ciKlass* target_klass = NULL;
2357         if (cha_monomorphic_target != NULL) {
2358           target_klass = cha_monomorphic_target->holder();
2359         } else if (exact_target != NULL) {
2360           target_klass = exact_target->holder();
2361         }
2362         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2363       }
2364     }
2365   }
2366 
2367   if (recv != NULL) {
2368     recv->set_escaped();
2369   }
2370   for (int i=0; i<args->length(); i++) {
2371     args->at(0)->set_escaped();
2372   }
2373 
2374   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2375                               declared_signature->return_type()->is_inlinetype());
2376   // push result
2377   append_split(result);
2378 
2379   if (result_type != voidType) {
2380     if (method()->is_strict()) {
2381       push(result_type, round_fp(result));
2382     } else {
2383       push(result_type, result);
2384     }
2385   }
2386   if (profile_return() && result_type->is_object_kind()) {
2387     profile_return_type(result, target);
2388   }
2389 }

2572 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2573   Canonicalizer canon(compilation(), instr, bci);
2574   Instruction* i1 = canon.canonical();
2575   if (i1->is_linked() || !i1->can_be_linked()) {
2576     // Canonicalizer returned an instruction which was already
2577     // appended so simply return it.
2578     return i1;
2579   }
2580 
2581   if (UseLocalValueNumbering) {
2582     // Lookup the instruction in the ValueMap and add it to the map if
2583     // it's not found.
2584     Instruction* i2 = vmap()->find_insert(i1);
2585     if (i2 != i1) {
2586       // found an entry in the value map, so just return it.
2587       assert(i2->is_linked(), "should already be linked");
2588       return i2;
2589     }
2590     ValueNumberingEffects vne(vmap());
2591     i1->visit(&vne);
2592   }
2593 
2594   // i1 was not eliminated => append it
2595   assert(i1->next() == NULL, "shouldn't already be linked");
2596   _last = _last->set_next(i1, canon.bci());
2597 
2598   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2599     // set the bailout state but complete normal processing.  We
2600     // might do a little more work before noticing the bailout so we
2601     // want processing to continue normally until it's noticed.
2602     bailout("Method and/or inlining is too large");
2603   }
2604 
2605 #ifndef PRODUCT
2606   if (PrintIRDuringConstruction) {
2607     InstructionPrinter ip;
2608     ip.print_line(i1);
2609     if (Verbose) {
2610       state()->print();
2611     }
2612   }
2613 #endif
2614 
2615   // save state after modification of operand stack for StateSplit instructions
2616   StateSplit* s = i1->as_StateSplit();
2617   if (s != NULL) {
2618     if (EliminateFieldAccess) {
2619       Intrinsic* intrinsic = s->as_Intrinsic();
2620       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2621         _memory->kill();
2622       }
2623     }
2624     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2625   }
2626 
2627   // set up exception handlers for this instruction if necessary
2628   if (i1->can_trap()) {
2629     i1->set_exception_handlers(handle_exception(i1));
2630     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2631   }
2632   return i1;
2633 }

2667 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2668   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2669     assert(instruction->exception_state() == NULL
2670            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2671            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2672            "exception_state should be of exception kind");
2673     return new XHandlers();
2674   }
2675 
2676   XHandlers*  exception_handlers = new XHandlers();
2677   ScopeData*  cur_scope_data = scope_data();
2678   ValueStack* cur_state = instruction->state_before();
2679   ValueStack* prev_state = NULL;
2680   int scope_count = 0;
2681 
2682   assert(cur_state != NULL, "state_before must be set");
2683   do {
2684     int cur_bci = cur_state->bci();
2685     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2686     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2687 
2688     // join with all potential exception handlers
2689     XHandlers* list = cur_scope_data->xhandlers();
2690     const int n = list->length();
2691     for (int i = 0; i < n; i++) {
2692       XHandler* h = list->handler_at(i);
2693       if (h->covers(cur_bci)) {
2694         // h is a potential exception handler => join it
2695         compilation()->set_has_exception_handlers(true);
2696 
2697         BlockBegin* entry = h->entry_block();
2698         if (entry == block()) {
2699           // It's acceptable for an exception handler to cover itself
2700           // but we don't handle that in the parser currently.  It's
2701           // very rare so we bailout instead of trying to handle it.
2702           BAILOUT_("exception handler covers itself", exception_handlers);
2703         }
2704         assert(entry->bci() == h->handler_bci(), "must match");
2705         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2706 
2707         // previously this was a BAILOUT, but this is not necessary
2708         // now because asynchronous exceptions are not handled this way.
2709         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2710 
2711         // xhandler start with an empty expression stack
2712         if (cur_state->stack_size() != 0) {
2713           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2714         }
2715         if (instruction->exception_state() == NULL) {
2716           instruction->set_exception_state(cur_state);
2717         }
2718 
2719         // Note: Usually this join must work. However, very
2720         // complicated jsr-ret structures where we don't ret from
2721         // the subroutine can cause the objects on the monitor
2722         // stacks to not match because blocks can be parsed twice.
2723         // The only test case we've seen so far which exhibits this
2724         // problem is caught by the infinite recursion test in
2725         // GraphBuilder::jsr() if the join doesn't work.
2726         if (!entry->try_merge(cur_state)) {
2727           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2728         }
2729 
2730         // add current state for correct handling of phi functions at begin of xhandler
2731         int phi_operand = entry->add_exception_state(cur_state);
2732 
2733         // add entry to the list of xhandlers of this block
2734         _block->add_exception_handler(entry);
2735 
2736         // add back-edge from xhandler entry to this block
2737         if (!entry->is_predecessor(_block)) {
2738           entry->add_predecessor(_block);
2739         }
2740 
2741         // clone XHandler because phi_operand and scope_count can not be shared
2742         XHandler* new_xhandler = new XHandler(h);
2743         new_xhandler->set_phi_operand(phi_operand);
2744         new_xhandler->set_scope_count(scope_count);
2745         exception_handlers->append(new_xhandler);
2746 
2747         // fill in exception handler subgraph lazily
2748         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2749         cur_scope_data->add_to_work_list(entry);
2750 
2751         // stop when reaching catchall
2752         if (h->catch_type() == 0) {
2753           return exception_handlers;
2754         }
2755       }
2756     }
2757 
2758     if (exception_handlers->length() == 0) {
2759       // This scope and all callees do not handle exceptions, so the local
2760       // variables of this scope are not needed. However, the scope itself is
2761       // required for a correct exception stack trace -> clear out the locals.
2762       if (_compilation->env()->should_retain_local_variables()) {
2763         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2764       } else {
2765         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2766       }
2767       if (prev_state != NULL) {
2768         prev_state->set_caller_state(cur_state);
2769       }
2770       if (instruction->exception_state() == NULL) {
2771         instruction->set_exception_state(cur_state);
2772       }
2773     }
2774 
2775     // Set up iteration for next time.
2776     // If parsing a jsr, do not grab exception handlers from the
2777     // parent scopes for this method (already got them, and they
2778     // needed to be cloned)
2779 
2780     while (cur_scope_data->parsing_jsr()) {
2781       cur_scope_data = cur_scope_data->parent();
2782     }
2783 
2784     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2785     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2786 
2787     prev_state = cur_state;
2788     cur_state = cur_state->caller_state();
2789     cur_scope_data = cur_scope_data->parent();
2790     scope_count++;
2791   } while (cur_scope_data != NULL);
2792 
2793   return exception_handlers;
2794 }

2815 Value PhiSimplifier::simplify(Value v) {
2816   Phi* phi = v->as_Phi();
2817 
2818   if (phi == NULL) {
2819     // no phi function
2820     return v;
2821   } else if (v->has_subst()) {
2822     // already substituted; subst can be phi itself -> simplify
2823     return simplify(v->subst());
2824   } else if (phi->is_set(Phi::cannot_simplify)) {
2825     // already tried to simplify phi before
2826     return phi;
2827   } else if (phi->is_set(Phi::visited)) {
2828     // break cycles in phi functions
2829     return phi;
2830   } else if (phi->type()->is_illegal()) {
2831     // illegal phi functions are ignored anyway
2832     return phi;
2833 
2834   } else {
2835     // mark phi function as processed to break cycles in phi functions
2836     phi->set(Phi::visited);

2843       assert(opd != NULL, "Operand must exist!");

2847         phi->make_illegal();

2849         return phi;
2850       }
2851 
2852       Value new_opd = simplify(opd);
2853       assert(new_opd != NULL, "Simplified operand must exist!");
2854 
2855       if (new_opd != phi && new_opd != subst) {
2856         if (subst == NULL) {
2857           subst = new_opd;
2858         } else {
2859           // no simplification possible
2860           phi->set(Phi::cannot_simplify);
2861           phi->clear(Phi::visited);
2862           return phi;
2863         }
2864       }
2865     }
2866

2868     assert(subst != NULL, "illegal phi function");

2871     phi->set_subst(subst);
2872

2874     if (PrintPhiFunctions) {

2878 
2879     return subst;

3236             // we setup the OSR edge leading into this block so that
3237             // Phis get setup correctly.
3238             setup_osr_entry_block();

3240             b->clear(BlockBegin::osr_entry_flag);
3241           }
3242           b->set(BlockBegin::was_visited_flag);
3243           connect_to_end(b);
3244         }
3245       }
3246     }
3247   } while (!bailed_out() && !scope_data()->is_work_list_empty());

3374 void GraphBuilder::setup_osr_entry_block() {
3375   assert(compilation()->is_osr_compile(), "only for osrs");
3376 
3377   int osr_bci = compilation()->osr_bci();
3378   ciBytecodeStream s(method());
3379   s.reset_to_bci(osr_bci);
3380   s.next();
3381   scope_data()->set_stream(&s);
3382 
3383   // create a new block to be the osr setup code
3384   _osr_entry = new BlockBegin(osr_bci);
3385   _osr_entry->set(BlockBegin::osr_entry_flag);
3386   _osr_entry->set_depth_first_number(0);
3387   BlockBegin* target = bci2block()->at(osr_bci);
3388   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3389   // the osr entry has no values for locals
3390   ValueStack* state = target->state()->copy();
3391   _osr_entry->set_state(state);
3392 
3393   kill_all();
3394   _block = _osr_entry;
3395   _state = _osr_entry->state()->copy();
3396   assert(_state->bci() == osr_bci, "mismatch");
3397   _last  = _osr_entry;
3398   Value e = append(new OsrEntry());
3399   e->set_needs_null_check(false);
3400 
3401   // OSR buffer is
3402   //
3403   // locals[nlocals-1..0]
3404   // monitors[number_of_locks-1..0]
3405   //
3406   // locals is a direct copy of the interpreter frame so in the osr buffer
3407   // so first slot in the local array is the last local from the interpreter
3408   // and last slot is local[0] (receiver) from the interpreter
3409   //
3410   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3411   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3412   // in the interpreter frame (the method lock if a sync method)
3413 
3414   // Initialize monitors in the compiled activation.
3415 
3416   int index;
3417   Value local;
3418 
3419   // find all the locals that the interpreter thinks contain live oops
3420   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3421 
3422   // compute the offset into the locals so that we can treat the buffer
3423   // as if the locals were still in the interpreter frame
3424   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3425   for_each_local_value(state, index, local) {
3426     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3427     Value get;
3428     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3429       // The interpreter thinks this local is dead but the compiler
3430       // doesn't so pretend that the interpreter passed in null.
3431       get = append(new Constant(objectNull));
3432     } else {
3433       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3434                                     append(new Constant(new IntConstant(offset))),
3435                                     0,
3436                                     true /*unaligned*/, true /*wide*/));
3437     }
3438     _state->store_local(index, get);
3439   }
3440 
3441   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3442 
3443   assert(state->caller_state() == NULL, "should be top scope");
3444   state->clear_locals();
3445   Goto* g = new Goto(target, false);
3446   append(g);
3447   _osr_entry->set_end(g);
3448   target->merge(_osr_entry->end()->state());
3449 
3450   scope_data()->set_stream(NULL);
3451 }

3456 
3457   // Set up locals for receiver
3458   int idx = 0;

3461     state->store_local(idx, new Local(method()->holder(), objectType, idx,

3496

3498   BlockListBuilder blm(compilation, scope, osr_bci);

3500

3515   case vmIntrinsics::_dsin          : // fall through
3516   case vmIntrinsics::_dcos          : // fall through
3517   case vmIntrinsics::_dtan          : // fall through

3524       // compilation environment because the bytecodes for the method
3525       // shouldn't be parsed during the compilation, only the special
3526       // Intrinsic node should be emitted.  If this isn't done the the
3527       // code for the inlined version will be different than the root
3528       // compiled version which could lead to monotonicity problems on
3529       // intel.

3531         BAILOUT("failed to inline intrinsic, method not annotated");

3566         // the referent field of the reference object gets recorded by
3567         // the pre-barrier code.

3578         // across safepoint since GC can change its value.
3579 
3580         // Set up a stream so that appending instructions works properly.
3581         ciBytecodeStream s(scope->method());
3582         s.reset_to_bci(0);
3583         scope_data()->set_stream(&s);
3584         s.next();
3585 
3586         // setup the initial block state
3587         _block = start_block;
3588         _state = start_block->state()->copy_for_parsing();
3589         _last  = start_block;
3590         load_local(objectType, 0);
3591 
3592         // Emit the intrinsic node.
3593         bool result = try_inline_intrinsics(scope->method());

3597         // connect the begin and end blocks and we're all done.

3615 
3616   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());

3618   if (osr_bci != -1) {
3619     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3620     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {

3627     }

3679 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3680   const char* msg = NULL;
3681 
3682   // clear out any existing inline bailout condition
3683   clear_inline_bailout();
3684 
3685   // exclude methods we don't want to inline
3686   msg = should_not_inline(callee);
3687   if (msg != NULL) {
3688     print_inlining(callee, msg, /*success*/ false);
3689     return false;
3690   }
3691 
3692   // method handle invokes
3693   if (callee->is_method_handle_intrinsic()) {
3694     if (try_method_handle_inline(callee, ignore_return)) {
3695       if (callee->has_reserved_stack_access()) {
3696         compilation()->set_has_reserved_stack_access(true);
3697       }
3698       return true;
3699     }
3700     return false;
3701   }
3702 
3703   // handle intrinsics
3704   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3705       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3706     if (try_inline_intrinsics(callee, ignore_return)) {
3707       print_inlining(callee, "intrinsic");
3708       if (callee->has_reserved_stack_access()) {
3709         compilation()->set_has_reserved_stack_access(true);
3710       }
3711       return true;
3712     }
3713     // try normal inlining
3714   }
3715 
3716   // certain methods cannot be parsed at all
3717   msg = check_can_parse(callee);
3718   if (msg != NULL) {
3719     print_inlining(callee, msg, /*success*/ false);
3720     return false;
3721   }
3722 
3723   // If bytecode not set use the current one.
3724   if (bc == Bytecodes::_illegal) {
3725     bc = code();
3726   }
3727   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3728     if (callee->has_reserved_stack_access()) {
3729       compilation()->set_has_reserved_stack_access(true);
3730     }
3731     return true;
3732   }
3733 
3734   // Entire compilation could fail during try_inline_full call.
3735   // In that case printing inlining decision info is useless.
3736   if (!bailed_out())
3737     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3738 
3739   return false;
3740 }

3887 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3888   // Introduce a new callee continuation point - all Ret instructions
3889   // will be replaced with Gotos to this point.
3890   BlockBegin* cont = block_at(next_bci());
3891   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3892 
3893   // Note: can not assign state to continuation yet, as we have to
3894   // pick up the state from the Ret instructions.
3895 
3896   // Push callee scope
3897   push_scope_for_jsr(cont, jsr_dest_bci);
3898 
3899   // Temporarily set up bytecode stream so we can append instructions
3900   // (only using the bci of this stream)
3901   scope_data()->set_stream(scope_data()->parent()->stream());
3902 
3903   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3904   assert(jsr_start_block != NULL, "jsr start block must exist");
3905   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3906   Goto* goto_sub = new Goto(jsr_start_block, false);
3907   // Must copy state to avoid wrong sharing when parsing bytecodes
3908   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3909   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3910   append(goto_sub);
3911   _block->set_end(goto_sub);
3912   _last = _block = jsr_start_block;
3913 
3914   // Clear out bytecode stream
3915   scope_data()->set_stream(NULL);
3916 
3917   scope_data()->add_to_work_list(jsr_start_block);
3918 
3919   // Ready to resume parsing in subroutine
3920   iterate_all_blocks();
3921 
3922   // If we bailed out during parsing, return immediately (this is bad news)
3923   CHECK_BAILOUT_(false);
3924 
3925   // Detect whether the continuation can actually be reached. If not,
3926   // it has not had state set by the join() operations in
3927   // iterate_bytecodes_for_block()/ret() and we should not touch the
3928   // iteration state. The calling activation of
3929   // iterate_bytecodes_for_block will then complete normally.
3930   if (cont->state() != NULL) {
3931     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3932       // add continuation to work list instead of parsing it immediately
3933       scope_data()->parent()->add_to_work_list(cont);
3934     }
3935   }
3936 
3937   assert(jsr_continuation() == cont, "continuation must not have changed");

3940          "continuation can only be visited in case of backward branches");
3941   assert(_last && _last->as_BlockEnd(), "block must have end");
3942 
3943   // continuation is in work list, so end iteration of current block
3944   _skip_block = true;
3945   pop_scope_for_jsr();
3946 
3947   return true;
3948 }

4037 bool GraphBuilder::try_inline_full(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {

4039   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4040     INLINE_BAILOUT("inlining prohibited by policy");
4041   }
4042   // first perform tests of things it's not possible to inline
4043   if (callee->has_exception_handlers() &&
4044       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4045   if (callee->is_synchronized() &&
4046       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4047   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4048   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4049 
4050   // Proper inlining of methods with jsrs requires a little more work.
4051   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4052 
4053   if (strict_fp_requires_explicit_rounding &&
4054       method()->is_strict() != callee->is_strict()) {
4055 #ifdef IA32
4056     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4057     // When SSE2 is present, no special handling is needed.
4058     if (UseSSE < 2) {
4059       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4060     }
4061 #else
4062     Unimplemented();
4063 #endif // IA32
4064   }
4065 
4066   if (is_profiling() && !callee->ensure_method_data()) {
4067     INLINE_BAILOUT("mdo allocation failed");
4068   }
4069 
4070   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4071   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4072 
4073   const int args_base = state()->stack_size() - callee->arg_size();
4074   assert(args_base >= 0, "stack underflow during inlining");
4075 
4076   Value recv = NULL;
4077   if (has_receiver) {
4078     assert(!callee->is_static(), "callee must not be static");
4079     assert(callee->arg_size() > 0, "must have at least a receiver");
4080 
4081     recv = state()->stack_at(args_base);
4082     if (recv->is_null_obj()) {
4083       INLINE_BAILOUT("receiver is always null");
4084     }
4085   }
4086 
4087   // now perform tests that are based on flag settings
4088   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4089   if (callee->force_inline() || inlinee_by_directive) {
4090     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4091     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4092 
4093     const char* msg = "";
4094     if (callee->force_inline())  msg = "force inline by annotation";
4095     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4096     print_inlining(callee, msg);
4097   } else {
4098     // use heuristic controls on inlining
4099     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4100     int callee_recursive_level = recursive_inline_level(callee);
4101     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4102     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4103     // Additional condition to limit stack usage for non-recursive calls.
4104     if ((callee_recursive_level == 0) &&
4105         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4106       INLINE_BAILOUT("callee uses too much stack");
4107     }
4108 
4109     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4110     if (callee->name() == ciSymbol::object_initializer_name() &&
4111         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4112       // Throwable constructor call
4113       IRScope* top = scope();
4114       while (top->caller() != NULL) {
4115         top = top->caller();
4116       }
4117       if (!top->method()->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4118         INLINE_BAILOUT("don't inline Throwable constructors");
4119       }
4120     }
4121

4123       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4124     }

4130   // GraphBuilder level, do not cause bailout just of the inlining but
4131   // in fact of the entire compilation.

4134 
4135   // Insert null check if necessary
4136   if (has_receiver) {
4137     // note: null check must happen even if first instruction of callee does
4138     //       an implicit null check since the callee is in a different scope
4139     //       and we must make sure exception handling does the right thing
4140     null_check(recv);
4141   }

4144     // Note that we'd collect profile data in this method if we wanted it.

4147

4197   // (only using the bci of this stream)

4201   // note: this will also ensure that all arguments are computed before being passed

4205     const int arg_no = i - args_base;

4211   // Note that we preserve locals state in case we can use it later
4212   // (see use of pop_scope() below)
4213   caller_state->truncate_stack(args_base);
4214   assert(callee_state->stack_size() == 0, "callee stack must be empty");
4215

4217   BlockBegin* sync_handler = NULL;
4218

4221     lock = callee->is_static() ? append(new Constant(new InstanceConstant(callee->holder()->java_mirror())))

4238   if (callee_start_block != NULL) {

4241     // The state for this goto is in the scope of the callee, so use
4242     // the entry bci for the callee instead of the call site bci.
4243     append_with_bci(goto_callee, 0);
4244     _block->set_end(goto_callee);

4270   // practice, we have only traversed the continuation if we are
4271   // inlining into a subroutine

4277   // the caller back in the GraphBuilder. The only thing we want to do
4278   // first is an optimization: during parsing of the callee we
4279   // generated at least one Goto to the continuation block. If we
4280   // generated exactly one, and if the inlined method spanned exactly
4281   // one block (and we didn't have to Goto its entry), then we snip
4282   // off the Goto to the continuation, allowing control to fall
4283   // through back into the caller block and effectively performing
4284   // block merging. This allows load elimination and CSE to take place
4285   // across multiple callee scopes if they are relatively simple, and
4286   // is currently essential to making inlining profitable.
4287   if (num_returns() == 1
4288       && block() == orig_block
4289       && block() == inline_cleanup_block()) {
4290     _last  = inline_cleanup_return_prev();
4291     _state = inline_cleanup_state();
4292   } else if (continuation_preds == cont->number_of_preds()) {
4293     // Inlining caused that the instructions after the invoke in the
4294     // caller are not reachable any more. So skip filling this block
4295     // with instructions!
4296     assert(cont == continuation(), "");
4297     assert(_last && _last->as_BlockEnd(), "");
4298     _skip_block = true;
4299   } else {
4300     // Resume parsing in continuation block unless it was already parsed.
4301     // Note that if we don't change _last here, iteration in
4302     // iterate_bytecodes_for_block will stop when we return.
4303     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4304       // add continuation to work list instead of parsing it immediately
4305       assert(_last && _last->as_BlockEnd(), "");
4306       scope_data()->parent()->add_to_work_list(continuation());
4307       _skip_block = true;
4308     }
4309   }
4310 
4311   // Fill the exception handler for synchronized methods with instructions
4312   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4313     fill_sync_handler(lock, sync_handler);
4314   } else {
4315     pop_scope();
4316   }
4317 
4318   compilation()->notice_inlined_method(callee);
4319 
4320   return true;
4321 }

4324 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4325   ValueStack* state_before = copy_state_before();
4326   vmIntrinsics::ID iid = callee->intrinsic_id();
4327   switch (iid) {
4328   case vmIntrinsics::_invokeBasic:
4329     {
4330       // get MethodHandle receiver
4331       const int args_base = state()->stack_size() - callee->arg_size();
4332       ValueType* type = state()->stack_at(args_base)->type();
4333       if (type->is_constant()) {
4334         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4335         // We don't do CHA here so only inline static and statically bindable methods.
4336         if (target->is_static() || target->can_be_statically_bound()) {
4337           if (ciMethod::is_consistent_info(callee, target)) {
4338             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4339             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4340             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4341               return true;
4342             }
4343           } else {
4344             print_inlining(target, "signatures mismatch", /*success*/ false);
4345           }
4346         } else {
4347           print_inlining(target, "not static or statically bindable", /*success*/ false);
4348         }
4349       } else {
4350         print_inlining(callee, "receiver not constant", /*success*/ false);
4351       }
4352     }
4353     break;
4354 
4355   case vmIntrinsics::_linkToVirtual:
4356   case vmIntrinsics::_linkToStatic:
4357   case vmIntrinsics::_linkToSpecial:
4358   case vmIntrinsics::_linkToInterface:
4359     {
4360       // pop MemberName argument
4361       const int args_base = state()->stack_size() - callee->arg_size();
4362       ValueType* type = apop()->type();
4363       if (type->is_constant()) {
4364         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4365         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4366         // If the target is another method handle invoke, try to recursively get
4367         // a better target.
4368         if (target->is_method_handle_intrinsic()) {
4369           if (try_method_handle_inline(target, ignore_return)) {
4370             return true;
4371           }
4372         } else if (!ciMethod::is_consistent_info(callee, target)) {
4373           print_inlining(target, "signatures mismatch", /*success*/ false);
4374         } else {
4375           ciSignature* signature = target->signature();
4376           const int receiver_skip = target->is_static() ? 0 : 1;
4377           // Cast receiver to its type.
4378           if (!target->is_static()) {
4379             ciKlass* tk = signature->accessing_klass();
4380             Value obj = state()->stack_at(args_base);
4381             if (obj->exact_type() == NULL &&
4382                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4383               TypeCast* c = new TypeCast(tk, obj, state_before);
4384               append(c);
4385               state()->stack_at_put(args_base, c);
4386             }
4387           }
4388           // Cast reference arguments to its type.
4389           for (int i = 0, j = 0; i < signature->count(); i++) {
4390             ciType* t = signature->type_at(i);
4391             if (t->is_klass()) {
4392               ciKlass* tk = t->as_klass();
4393               Value obj = state()->stack_at(args_base + receiver_skip + j);
4394               if (obj->exact_type() == NULL &&
4395                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4396                 TypeCast* c = new TypeCast(t, obj, state_before);
4397                 append(c);
4398                 state()->stack_at_put(args_base + receiver_skip + j, c);
4399               }
4400             }
4401             j += t->size();  // long and double take two slots
4402           }
4403           // We don't do CHA here so only inline static and statically bindable methods.
4404           if (target->is_static() || target->can_be_statically_bound()) {
4405             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4406             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4407               return true;
4408             }
4409           } else {
4410             print_inlining(target, "not static or statically bindable", /*success*/ false);
4411           }
4412         }
4413       } else {
4414         print_inlining(callee, "MemberName not constant", /*success*/ false);
4415       }
4416     }
4417     break;
4418 
4419   default:
4420     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4421     break;
4422   }
4423   set_state(state_before->copy_for_parsing());
4424   return false;
4425 }
