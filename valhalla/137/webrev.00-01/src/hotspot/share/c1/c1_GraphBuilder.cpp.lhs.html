 133 void BlockListBuilder::set_entries(int osr_bci) {
 134   // generate start blocks
 135   BlockBegin* std_entry = make_block_at(0, NULL);
 136   if (scope()->caller() == NULL) {
 137     std_entry->set(BlockBegin::std_entry_flag);
 138   }
 139   if (osr_bci != -1) {
 140     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 141     osr_entry->set(BlockBegin::osr_entry_flag);
 142   }
 143 
 144   // generate exception entry blocks
 145   XHandlers* list = xhandlers();
 146   const int n = list->length();
 147   for (int i = 0; i < n; i++) {
 148     XHandler* h = list->handler_at(i);
 149     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 150     entry->set(BlockBegin::exception_entry_flag);
 151     h->set_entry_block(entry);
 152   }
 153 }

 156 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 157   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 158 
 159   BlockBegin* block = _bci2block->at(cur_bci);
 160   if (block == NULL) {
 161     block = new BlockBegin(cur_bci);
 162     block->init_stores_to_locals(method()->max_locals());
 163     _bci2block->at_put(cur_bci, block);
 164     _blocks.append(block);
 165 
 166     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 167   }
 168 
 169   if (predecessor != NULL) {
 170     if (block->is_set(BlockBegin::exception_entry_flag)) {
 171       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 172     }
 173 
 174     predecessor->add_successor(block);
 175     block->increment_total_preds();
 176   }
 177 
 178   return block;
 179 }

 182 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 183   current->stores_to_locals().set_bit(local);

 185 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 186   store_one(current, local);
 187   store_one(current, local + 1);

 191 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 192   // Draws edges from a block to its exception handlers
 193   XHandlers* list = xhandlers();
 194   const int n = list->length();
 195 
 196   for (int i = 0; i < n; i++) {
 197     XHandler* h = list->handler_at(i);
 198 
 199     if (h->covers(cur_bci)) {
 200       BlockBegin* entry = h->entry_block();
 201       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 202       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 203 
 204       // add each exception handler only once
 205       if (!current->is_successor(entry)) {
 206         current->add_successor(entry);
 207         entry->increment_total_preds();
 208       }
 209 
 210       // stop when reaching catchall
 211       if (h->catch_type() == 0) break;
 212     }
 213   }
 214 }

 216 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 217   // start a new block after jsr-bytecode and link this block into cfg
 218   make_block_at(next_bci, current);
 219 
 220   // start a new block at the subroutine entry at mark it with special flag
 221   BlockBegin* sr_block = make_block_at(sr_bci, current);
 222   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 223     sr_block->set(BlockBegin::subroutine_entry_flag);
 224   }
 225 }

 228 void BlockListBuilder::set_leaders() {
 229   bool has_xhandlers = xhandlers()->has_handlers();
 230   BlockBegin* current = NULL;
 231 
 232   // The information which bci starts a new block simplifies the analysis
 233   // Without it, backward branches could jump to a bci where no block was created
 234   // during bytecode iteration. This would require the creation of a new block at the
 235   // branch target and a modification of the successor lists.
 236   const BitMap& bci_block_start = method()->bci_block_start();
 237 
 238   ciBytecodeStream s(method());
 239   while (s.next() != ciBytecodeStream::EOBC()) {
 240     int cur_bci = s.cur_bci();
 241 
 242     if (bci_block_start.at(cur_bci)) {
 243       current = make_block_at(cur_bci, current);
 244     }
 245     assert(current != NULL, "must have current block");
 246 
 247     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 248       handle_exceptions(current, cur_bci);
 249     }
 250 
 251     switch (s.cur_bc()) {
 252       // track stores to local variables for selective creation of phi functions
 253       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 254       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 255       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 256       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 257       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 258       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 259       case Bytecodes::_istore_0: store_one(current, 0); break;
 260       case Bytecodes::_istore_1: store_one(current, 1); break;
 261       case Bytecodes::_istore_2: store_one(current, 2); break;
 262       case Bytecodes::_istore_3: store_one(current, 3); break;
 263       case Bytecodes::_lstore_0: store_two(current, 0); break;
 264       case Bytecodes::_lstore_1: store_two(current, 1); break;
 265       case Bytecodes::_lstore_2: store_two(current, 2); break;
 266       case Bytecodes::_lstore_3: store_two(current, 3); break;
 267       case Bytecodes::_fstore_0: store_one(current, 0); break;
 268       case Bytecodes::_fstore_1: store_one(current, 1); break;
 269       case Bytecodes::_fstore_2: store_one(current, 2); break;
 270       case Bytecodes::_fstore_3: store_one(current, 3); break;
 271       case Bytecodes::_dstore_0: store_two(current, 0); break;
 272       case Bytecodes::_dstore_1: store_two(current, 1); break;
 273       case Bytecodes::_dstore_2: store_two(current, 2); break;
 274       case Bytecodes::_dstore_3: store_two(current, 3); break;
 275       case Bytecodes::_astore_0: store_one(current, 0); break;
 276       case Bytecodes::_astore_1: store_one(current, 1); break;
 277       case Bytecodes::_astore_2: store_one(current, 2); break;
 278       case Bytecodes::_astore_3: store_one(current, 3); break;
 279 
 280       // track bytecodes that affect the control flow
 281       case Bytecodes::_athrow:  // fall through
 282       case Bytecodes::_ret:     // fall through
 283       case Bytecodes::_ireturn: // fall through
 284       case Bytecodes::_lreturn: // fall through
 285       case Bytecodes::_freturn: // fall through
 286       case Bytecodes::_dreturn: // fall through
 287       case Bytecodes::_areturn: // fall through
 288       case Bytecodes::_return:
 289         current = NULL;
 290         break;
 291 
 292       case Bytecodes::_ifeq:      // fall through
 293       case Bytecodes::_ifne:      // fall through
 294       case Bytecodes::_iflt:      // fall through
 295       case Bytecodes::_ifge:      // fall through
 296       case Bytecodes::_ifgt:      // fall through
 297       case Bytecodes::_ifle:      // fall through
 298       case Bytecodes::_if_icmpeq: // fall through
 299       case Bytecodes::_if_icmpne: // fall through
 300       case Bytecodes::_if_icmplt: // fall through
 301       case Bytecodes::_if_icmpge: // fall through
 302       case Bytecodes::_if_icmpgt: // fall through
 303       case Bytecodes::_if_icmple: // fall through
 304       case Bytecodes::_if_acmpeq: // fall through
 305       case Bytecodes::_if_acmpne: // fall through
 306       case Bytecodes::_ifnull:    // fall through
 307       case Bytecodes::_ifnonnull:
 308         make_block_at(s.next_bci(), current);
 309         make_block_at(s.get_dest(), current);
 310         current = NULL;
 311         break;
 312 
 313       case Bytecodes::_goto:
 314         make_block_at(s.get_dest(), current);
 315         current = NULL;
 316         break;
 317 
 318       case Bytecodes::_goto_w:
 319         make_block_at(s.get_far_dest(), current);
 320         current = NULL;
 321         break;
 322 
 323       case Bytecodes::_jsr:
 324         handle_jsr(current, s.get_dest(), s.next_bci());
 325         current = NULL;
 326         break;
 327 
 328       case Bytecodes::_jsr_w:
 329         handle_jsr(current, s.get_far_dest(), s.next_bci());
 330         current = NULL;
 331         break;
 332 
 333       case Bytecodes::_tableswitch: {
 334         // set block for each case
 335         Bytecode_tableswitch sw(&s);
 336         int l = sw.length();
 337         for (int i = 0; i < l; i++) {
 338           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 339         }
 340         make_block_at(cur_bci + sw.default_offset(), current);
 341         current = NULL;
 342         break;
 343       }
 344 
 345       case Bytecodes::_lookupswitch: {
 346         // set block for each case
 347         Bytecode_lookupswitch sw(&s);
 348         int l = sw.number_of_pairs();
 349         for (int i = 0; i < l; i++) {
 350           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 351         }
 352         make_block_at(cur_bci + sw.default_offset(), current);
 353         current = NULL;
 354         break;
 355       }
 356 
 357       default:
 358         break;
 359     }
 360   }
 361 }

 364 void BlockListBuilder::mark_loops() {
 365   ResourceMark rm;
 366 
 367   _active.initialize(BlockBegin::number_of_blocks());
 368   _visited.initialize(BlockBegin::number_of_blocks());
 369   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 370   _next_loop_index = 0;
 371   _next_block_number = _blocks.length();
 372 
 373   // recursively iterate the control flow graph
 374   mark_loops(_bci2block->at(0), false);
 375   assert(_next_block_number >= 0, "invalid block numbers");
 376 
 377   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 378   _active.resize(0);
 379   _visited.resize(0);
 380 }

 382 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 383   if (block->is_set(BlockBegin::exception_entry_flag)) {
 384     // exception edges may look like loops but don't mark them as such
 385     // since it screws up block ordering.
 386     return;
 387   }
 388   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 389     block->set(BlockBegin::parser_loop_header_flag);
 390 
 391     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 392     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 393     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 394     if (_next_loop_index < 31) _next_loop_index++;
 395   } else {
 396     // block already marked as loop header
 397     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 398   }
 399 }

 401 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 402   int block_id = block->block_id();
 403 
 404   if (_visited.at(block_id)) {
 405     if (_active.at(block_id)) {
 406       // reached block via backward branch
 407       make_loop_header(block);
 408     }
 409     // return cached loop information for this block
 410     return _loop_map.at(block_id);
 411   }
 412 
 413   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 414     in_subroutine = true;
 415   }
 416 
 417   // set active and visited bits before successors are processed
 418   _visited.set_bit(block_id);
 419   _active.set_bit(block_id);
 420 
 421   intptr_t loop_state = 0;
 422   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 423     // recursively process all successors
 424     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 425   }
 426 
 427   // clear active-bit after all successors are processed
 428   _active.clear_bit(block_id);
 429 
 430   // reverse-post-order numbering of all blocks
 431   block->set_depth_first_number(_next_block_number);
 432   _next_block_number--;
 433 
 434   if (loop_state != 0 || in_subroutine ) {
 435     // block is contained at least in one loop, so phi functions are necessary
 436     // phi functions are also necessary for all locals stored in a subroutine
 437     scope()->requires_phi_function().set_union(block->stores_to_locals());
 438   }
 439 
 440   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 441     int header_loop_state = _loop_map.at(block_id);
 442     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 443 
 444     // If the highest bit is set (i.e. when integer value is negative), the method
 445     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 446     if (header_loop_state >= 0) {
 447       clear_bits(loop_state, header_loop_state);
 448     }
 449   }
 450 
 451   // cache and return loop information for this block
 452   _loop_map.at_put(block_id, loop_state);
 453   return loop_state;
 454 }

 463 void BlockListBuilder::print() {
 464   tty->print("----- initial block list of BlockListBuilder for method ");
 465   method()->print_short_name();
 466   tty->cr();
 467 
 468   // better readability if blocks are sorted in processing order
 469   _blocks.sort(compare_depth_first);
 470 
 471   for (int i = 0; i < _blocks.length(); i++) {
 472     BlockBegin* cur = _blocks.at(i);
 473     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 474 
 475     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 476     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 477     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 478     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 479     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 480 
 481     if (cur->number_of_sux() > 0) {
 482       tty->print("    sux: ");
 483       for (int j = 0; j < cur->number_of_sux(); j++) {
 484         BlockBegin* sux = cur->sux_at(j);
 485         tty->print("B%d ", sux->block_id());
 486       }
 487     }
 488     tty->cr();
 489   }
 490 }

 495 // A simple growable array of Values indexed by ciFields
 496 class FieldBuffer: public CompilationResourceObj {
 497  private:
 498   GrowableArray<Value> _values;
 499 
 500  public:
 501   FieldBuffer() {}
 502 
 503   void kill() {
 504     _values.trunc_to(0);
 505   }
 506 
 507   Value at(ciField* field) {
 508     assert(field->holder()->is_loaded(), "must be a loaded field");
 509     int offset = field->offset();
 510     if (offset < _values.length()) {
 511       return _values.at(offset);
 512     } else {
 513       return NULL;
 514     }
 515   }
 516 
 517   void at_put(ciField* field, Value value) {
 518     assert(field->holder()->is_loaded(), "must be a loaded field");
 519     int offset = field->offset();
 520     _values.at_put_grow(offset, value, NULL);
 521   }
 522 
 523 };

 526 // MemoryBuffer is fairly simple model of the current state of memory.
 527 // It partitions memory into several pieces.  The first piece is
 528 // generic memory where little is known about the owner of the memory.
 529 // This is conceptually represented by the tuple <O, F, V> which says
 530 // that the field F of object O has value V.  This is flattened so
 531 // that F is represented by the offset of the field and the parallel
 532 // arrays _objects and _values are used for O and V.  Loads of O.F can
 533 // simply use V.  Newly allocated objects are kept in a separate list
 534 // along with a parallel array for each object which represents the
 535 // current value of its fields.  Stores of the default value to fields
 536 // which have never been stored to before are eliminated since they
 537 // are redundant.  Once newly allocated objects are stored into
 538 // another object or they are passed out of the current compile they
 539 // are treated like generic memory.
 540 
 541 class MemoryBuffer: public CompilationResourceObj {
 542  private:
 543   FieldBuffer                 _values;
 544   GrowableArray<Value>        _objects;
 545   GrowableArray<Value>        _newobjects;
 546   GrowableArray<FieldBuffer*> _fields;
 547 
 548  public:
 549   MemoryBuffer() {}
 550 
 551   StoreField* store(StoreField* st) {
 552     if (!EliminateFieldAccess) {
 553       return st;
 554     }
 555 
 556     Value object = st->obj();
 557     Value value = st->value();
 558     ciField* field = st->field();
 559     if (field->holder()->is_loaded()) {
 560       int offset = field->offset();
 561       int index = _newobjects.find(object);
 562       if (index != -1) {
 563         // newly allocated object with no other stores performed on this field
 564         FieldBuffer* buf = _fields.at(index);
 565         if (buf->at(field) == NULL && is_default_value(value)) {
 566 #ifndef PRODUCT
 567           if (PrintIRDuringConstruction && Verbose) {
 568             tty->print_cr("Eliminated store for object %d:", index);
 569             st->print_line();
 570           }
 571 #endif
 572           return NULL;
 573         } else {
 574           buf->at_put(field, value);
 575         }
 576       } else {
 577         _objects.at_put_grow(offset, object, NULL);
 578         _values.at_put(field, value);
 579       }
 580 
 581       store_value(value);
 582     } else {
 583       // if we held onto field names we could alias based on names but
 584       // we don't know what's being stored to so kill it all.
 585       kill();
 586     }
 587     return st;
 588   }
 589 
 590 
 591   // return true if this value correspond to the default value of a field.
 592   bool is_default_value(Value value) {
 593     Constant* con = value->as_Constant();
 594     if (con) {
 595       switch (con->type()->tag()) {
 596         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 597         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 598         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 599         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 600         case objectTag: return con->type() == objectNull;
 601         default:  ShouldNotReachHere();
 602       }
 603     }
 604     return false;
 605   }
 606 
 607 
 608   // return either the actual value of a load or the load itself
 609   Value load(LoadField* load) {
 610     if (!EliminateFieldAccess) {
 611       return load;
 612     }
 613 
 614     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 615 #ifdef IA32
 616       if (UseSSE < 2) {
 617         // can't skip load since value might get rounded as a side effect
 618         return load;
 619       }
 620 #else
 621       Unimplemented();
 622 #endif // IA32
 623     }
 624 
 625     ciField* field = load->field();
 626     Value object   = load->obj();
 627     if (field->holder()->is_loaded() && !field->is_volatile()) {
 628       int offset = field->offset();
 629       Value result = NULL;
 630       int index = _newobjects.find(object);
 631       if (index != -1) {
 632         result = _fields.at(index)->at(field);
 633       } else if (_objects.at_grow(offset, NULL) == object) {
 634         result = _values.at(field);
 635       }
 636       if (result != NULL) {
 637 #ifndef PRODUCT
 638         if (PrintIRDuringConstruction && Verbose) {
 639           tty->print_cr("Eliminated load: ");
 640           load->print_line();
 641         }
 642 #endif
 643         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 644         return result;
 645       }
 646     }
 647     return load;
 648   }
 649 
 650   // Record this newly allocated object
 651   void new_instance(NewInstance* object) {
 652     int index = _newobjects.length();
 653     _newobjects.append(object);
 654     if (_fields.at_grow(index, NULL) == NULL) {
 655       _fields.at_put(index, new FieldBuffer());
 656     } else {
 657       _fields.at(index)->kill();
 658     }
 659   }
 660 
 661   // Record this newly allocated object
 662   void new_instance(NewInlineTypeInstance* object) {
 663     int index = _newobjects.length();
 664     _newobjects.append(object);
 665     if (_fields.at_grow(index, NULL) == NULL) {
 666       _fields.at_put(index, new FieldBuffer());
 667     } else {
 668       _fields.at(index)->kill();
 669     }
 670   }
 671 
 672   void store_value(Value value) {
 673     int index = _newobjects.find(value);
 674     if (index != -1) {
 675       // stored a newly allocated object into another object.
 676       // Assume we've lost track of it as separate slice of memory.
 677       // We could do better by keeping track of whether individual
 678       // fields could alias each other.
 679       _newobjects.remove_at(index);
 680       // pull out the field info and store it at the end up the list
 681       // of field info list to be reused later.
 682       _fields.append(_fields.at(index));
 683       _fields.remove_at(index);
 684     }
 685   }
 686 
 687   void kill() {
 688     _newobjects.trunc_to(0);
 689     _objects.trunc_to(0);
 690     _values.kill();
 691   }
 692 };

 733 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 734   if (parsing_jsr()) {
 735     // It is necessary to clone all blocks associated with a
 736     // subroutine, including those for exception handlers in the scope
 737     // of the method containing the jsr (because those exception
 738     // handlers may contain ret instructions in some cases).
 739     BlockBegin* block = bci2block()->at(bci);
 740     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 741       BlockBegin* new_block = new BlockBegin(block->bci());
 742       if (PrintInitialBlockList) {
 743         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 744                       block->block_id(), block->bci(), new_block->block_id());
 745       }
 746       // copy data from cloned blocked
 747       new_block->set_depth_first_number(block->depth_first_number());
 748       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 749       // Preserve certain flags for assertion checking
 750       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 751       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 752 
 753       // copy was_visited_flag to allow early detection of bailouts
 754       // if a block that is used in a jsr has already been visited before,
 755       // it is shared between the normal control flow and a subroutine
 756       // BlockBegin::try_merge returns false when the flag is set, this leads
 757       // to a compilation bailout
 758       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 759 
 760       bci2block()->at_put(bci, new_block);
 761       block = new_block;
 762     }
 763     return block;
 764   } else {
 765     return bci2block()->at(bci);
 766   }

 799 void GraphBuilder::ScopeData::add_to_work_list(BlockBegin* block) {
 800   if (_work_list == NULL) {
 801     _work_list = new BlockList();
 802   }
 803 
 804   if (!block->is_set(BlockBegin::is_on_work_list_flag)) {
 805     // Do not start parsing the continuation block while in a
 806     // sub-scope
 807     if (parsing_jsr()) {
 808       if (block == jsr_continuation()) {
 809         return;
 810       }
 811     } else {
 812       if (block == continuation()) {
 813         return;
 814       }
 815     }
 816     block->set(BlockBegin::is_on_work_list_flag);
 817     _work_list->push(block);
 818 
 819     sort_top_into_worklist(_work_list, block);
 820   }

 824 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 825   assert(worklist->top() == top, "");
 826   // sort block descending into work list
 827   const int dfn = top->depth_first_number();
 828   assert(dfn != -1, "unknown depth first number");
 829   int i = worklist->length()-2;
 830   while (i >= 0) {
 831     BlockBegin* b = worklist->at(i);
 832     if (b->depth_first_number() < dfn) {
 833       worklist->at_put(i+1, b);
 834     } else {
 835       break;
 836     }
 837     i --;
 838   }
 839   if (i >= -1) worklist->at_put(i + 1, top);
 840 }

 856 void GraphBuilder::ScopeData::setup_jsr_xhandlers() {
 857   assert(parsing_jsr(), "");
 858   // clone all the exception handlers from the scope
 859   XHandlers* handlers = new XHandlers(scope()->xhandlers());
 860   const int n = handlers->length();

 870   }

 897 void GraphBuilder::load_constant() {
 898   ciConstant con = stream()->get_constant();
 899   if (con.basic_type() == T_ILLEGAL) {
 900     // FIXME: an unresolved Dynamic constant can get here,
 901     // and that should not terminate the whole compilation.
 902     BAILOUT("could not resolve a constant");
 903   } else {
 904     ValueType* t = illegalType;
 905     ValueStack* patch_state = NULL;
 906     switch (con.basic_type()) {
 907       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 908       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 909       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 910       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 911       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 912       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 913       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 914       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 915       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 916       case T_OBJECT :
 917        {
 918         ciObject* obj = con.as_object();
 919         if (!obj->is_loaded()
 920             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 921           // A Class, MethodType, MethodHandle, or String.
 922           // Unloaded condy nodes show up as T_ILLEGAL, above.
 923           patch_state = copy_state_before();
 924           t = new ObjectConstant(obj);
 925         } else {
 926           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 927           // result, which might turn out to be an array.
 928           if (obj->is_null_object())
 929             t = objectNull;
 930           else if (obj->is_array())
 931             t = new ArrayConstant(obj->as_array());
 932           else
 933             t = new InstanceConstant(obj->as_instance());
 934         }
 935         break;
 936        }
 937       default       : ShouldNotReachHere();
 938     }
 939     Value x;
 940     if (patch_state != NULL) {
 941       x = new Constant(t, patch_state);
 942     } else {
 943       x = new Constant(t);
 944     }
 945     push(t, append(x));
 946   }
 947 }

 974 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 975   if (parsing_jsr()) {
 976     // We need to do additional tracking of the location of the return
 977     // address for jsrs since we don't handle arbitrary jsr/ret
 978     // constructs. Here we are figuring out in which circumstances we
 979     // need to bail out.
 980     if (x->type()->is_address()) {
 981       scope_data()->set_jsr_return_address_local(index);
 982 
 983       // Also check parent jsrs (if any) at this time to see whether
 984       // they are using this local. We don't handle skipping over a
 985       // ret.
 986       for (ScopeData* cur_scope_data = scope_data()->parent();
 987            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 988            cur_scope_data = cur_scope_data->parent()) {
 989         if (cur_scope_data->jsr_return_address_local() == index) {
 990           BAILOUT("subroutine overwrites return address from previous subroutine");
 991         }
 992       }
 993     } else if (index == scope_data()->jsr_return_address_local()) {
 994       scope_data()->set_jsr_return_address_local(-1);
 995     }
 996   }
 997 
 998   x->set_local_index(index);
 999   state->store_local(index, round_fp(x));
1000   if (x->as_NewInlineTypeInstance() != NULL) {
1001     x->as_NewInlineTypeInstance()->set_local_index(index);
1002     x->as_NewInlineTypeInstance()->decrement_on_stack_count();
1003   }
1004 }

1007 void GraphBuilder::load_indexed(BasicType type) {
1008   // In case of in block code motion in range check elimination
1009   ValueStack* state_before = NULL;
1010   int array_idx = state()->stack_size() - 2;
1011   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
1012     // Save the entire state and re-execute on deopt when accessing flattened arrays
1013     state_before = copy_state_before();
1014     state_before->set_should_reexecute(true);
1015   } else {
1016     state_before = copy_state_indexed_access();
1017   }
1018   compilation()->set_has_access_indexed(true);
1019   Value index = ipop();
1020   Value array = apop();
1021   Value length = NULL;
1022   if (CSEArrayLength ||
1023       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1024       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1025     length = append(new ArrayLength(array, state_before));
1026   }
1027 
1028   LoadIndexed* load_indexed = NULL;
1029   Instruction* result = NULL;
1030   if (array->is_loaded_flattened_array()) {
1031     ciType* array_type = array->declared_type();
1032     ciInlineKlass* elem_klass = array_type->as_flat_array_klass()->element_klass()->as_inline_klass();
1033     NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(elem_klass, state_before, false);
1034     _memory->new_instance(new_instance);
1035     apush(append_split(new_instance));
1036     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1037     load_indexed->set_vt(new_instance);
1038   } else {
1039     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1040   }
1041   if (profile_array_accesses() && is_reference_type(type)) {
1042     compilation()->set_would_profile(true);
1043     load_indexed->set_should_profile(true);
1044     load_indexed->set_profiled_method(method());
1045     load_indexed->set_profiled_bci(bci());
1046   }
1047   result = append(load_indexed);
1048   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1049   if (!array->is_loaded_flattened_array()) {
1050     push(as_ValueType(type), result);
1051   }
1052 }

1098   assert(!store_indexed->should_profile() || store_indexed == result, "should not be optimized out");
1099   _memory->store_value(value);

1103 void GraphBuilder::stack_op(Bytecodes::Code code) {
1104   switch (code) {
1105     case Bytecodes::_pop:
1106       { Value w = state()->raw_pop();
1107         NewInlineTypeInstance::update_stack_count(w);
1108       }
1109       break;
1110     case Bytecodes::_pop2:
1111       { Value w1 = state()->raw_pop();
1112         Value w2 = state()->raw_pop();
1113         NewInlineTypeInstance::update_stack_count(w1);
1114         NewInlineTypeInstance::update_stack_count(w2);
1115       }
1116       break;
1117     case Bytecodes::_dup:
1118       { Value w = state()->raw_pop();
1119         NewInlineTypeInstance::update_larval_state(w);
1120         state()->raw_push(w);
1121         state()->raw_push(w);
1122       }
1123       break;
1124     case Bytecodes::_dup_x1:
1125       { Value w1 = state()->raw_pop();
1126         Value w2 = state()->raw_pop();
1127         NewInlineTypeInstance::update_larval_state(w1);
1128         state()->raw_push(w1);
1129         state()->raw_push(w2);
1130         state()->raw_push(w1);
1131       }
1132       break;
1133     case Bytecodes::_dup_x2:
1134       { Value w1 = state()->raw_pop();
1135         Value w2 = state()->raw_pop();
1136         Value w3 = state()->raw_pop();

1137         if (w1 != NULL && w1->as_NewInlineTypeInstance() != NULL) {
1138           ciBytecodeStream s(method());
1139           s.force_bci(bci());
1140           s.next();
1141           if (s.cur_bc() != Bytecodes::_pop) {
1142             w1->as_NewInlineTypeInstance()->set_not_larva_anymore();
1143           }  else {
1144             w1->as_NewInlineTypeInstance()->increment_on_stack_count();
1145            }
1146         }
1147         state()->raw_push(w1);
1148         state()->raw_push(w3);
1149         state()->raw_push(w2);
1150         state()->raw_push(w1);
1151       }
1152       break;
1153     case Bytecodes::_dup2:
1154       { Value w1 = state()->raw_pop();
1155         Value w2 = state()->raw_pop();
1156         NewInlineTypeInstance::update_larval_state(w1);
1157         NewInlineTypeInstance::update_larval_state(w2);
1158         state()->raw_push(w2);
1159         state()->raw_push(w1);
1160         state()->raw_push(w2);
1161         state()->raw_push(w1);
1162       }
1163       break;
1164     case Bytecodes::_dup2_x1:
1165       { Value w1 = state()->raw_pop();
1166         Value w2 = state()->raw_pop();
1167         Value w3 = state()->raw_pop();
1168         NewInlineTypeInstance::update_larval_state(w1);
1169         NewInlineTypeInstance::update_larval_state(w2);
1170         state()->raw_push(w2);
1171         state()->raw_push(w1);
1172         state()->raw_push(w3);
1173         state()->raw_push(w2);
1174         state()->raw_push(w1);
1175       }
1176       break;
1177     case Bytecodes::_dup2_x2:
1178       { Value w1 = state()->raw_pop();
1179         Value w2 = state()->raw_pop();
1180         Value w3 = state()->raw_pop();
1181         Value w4 = state()->raw_pop();
1182         NewInlineTypeInstance::update_larval_state(w1);
1183         NewInlineTypeInstance::update_larval_state(w2);
1184         state()->raw_push(w2);
1185         state()->raw_push(w1);
1186         state()->raw_push(w4);
1187         state()->raw_push(w3);
1188         state()->raw_push(w2);
1189         state()->raw_push(w1);
1190       }
1191       break;
1192     case Bytecodes::_swap:
1193       { Value w1 = state()->raw_pop();
1194         Value w2 = state()->raw_pop();
1195         state()->raw_push(w1);
1196         state()->raw_push(w2);
1197       }
1198       break;
1199     default:
1200       ShouldNotReachHere();
1201       break;
1202   }
1203 }

1226 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1227   Value s = ipop();
1228   Value x = pop(type);
1229   // try to simplify
1230   // Note: This code should go into the canonicalizer as soon as it can
1231   //       can handle canonicalized forms that contain more than one node.
1232   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1233     // pattern: x >>> s
1234     IntConstant* s1 = s->type()->as_IntConstant();
1235     if (s1 != NULL) {
1236       // pattern: x >>> s1, with s1 constant
1237       ShiftOp* l = x->as_ShiftOp();
1238       if (l != NULL && l->op() == Bytecodes::_ishl) {
1239         // pattern: (a << b) >>> s1
1240         IntConstant* s0 = l->y()->type()->as_IntConstant();
1241         if (s0 != NULL) {
1242           // pattern: (a << s0) >>> s1
1243           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1244           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1245           if (s0c == s1c) {
1246             if (s0c == 0) {
1247               // pattern: (a << 0) >>> 0 => simplify to: a
1248               ipush(l->x());
1249             } else {
1250               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1251               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1252               const int m = (1 << (BitsPerInt - s0c)) - 1;
1253               Value s = append(new Constant(new IntConstant(m)));
1254               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1255             }
1256             return;
1257           }
1258         }
1259       }
1260     }
1261   }
1262   // could not simplify
1263   push(type, append(new ShiftOp(code, x, s)));
1264 }

1311 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1312   BlockBegin* tsux = block_at(stream()->get_dest());
1313   BlockBegin* fsux = block_at(stream()->next_bci());
1314   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1315 
1316   bool subst_check = false;
1317   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne)) {
1318     ValueType* left_vt = x->type();
1319     ValueType* right_vt = y->type();
1320     if (left_vt->is_object()) {
1321       assert(right_vt->is_object(), "must be");
1322       ciKlass* left_klass = x->as_loaded_klass_or_null();
1323       ciKlass* right_klass = y->as_loaded_klass_or_null();
1324 
1325       if (left_klass == NULL || right_klass == NULL) {
1326         // The klass is still unloaded, or came from a Phi node. Go slow case;
1327         subst_check = true;
1328       } else if (left_klass->can_be_inline_klass() || right_klass->can_be_inline_klass()) {
1329         // Either operand may be a value object, but we're not sure. Go slow case;
1330         subst_check = true;
1331       } else {
1332         // No need to do substitutability check
1333       }
1334     }
1335   }
1336 
1337   // In case of loop invariant code motion or predicate insertion
1338   // before the body of a loop the state is needed
1339   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1340 
1341   assert(i->as_Goto() == NULL ||
1342          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1343          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1344          "safepoint state of Goto returned by canonicalizer incorrect");
1345 
1346   if (is_profiling()) {
1347     If* if_node = i->as_If();
1348     if (if_node != NULL) {
1349       // Note that we'd collect profile data in this method if we wanted it.
1350       compilation()->set_would_profile(true);
1351       // At level 2 we need the proper bci to count backedges
1352       if_node->set_profiled_bci(bci());
1353       if (profile_branches()) {
1354         // Successors can be rotated by the canonicalizer, check for this case.
1355         if_node->set_profiled_method(method());
1356         if_node->set_should_profile(true);
1357         if (if_node->tsux() == fsux) {
1358           if_node->set_swapped(true);
1359         }
1360       }
1361       return;
1362     }
1363 
1364     // Check if this If was reduced to Goto.
1365     Goto *goto_node = i->as_Goto();
1366     if (goto_node != NULL) {
1367       compilation()->set_would_profile(true);
1368       goto_node->set_profiled_bci(bci());
1369       if (profile_branches()) {
1370         goto_node->set_profiled_method(method());
1371         goto_node->set_should_profile(true);
1372         // Find out which successor is used.
1373         if (goto_node->default_sux() == tsux) {
1374           goto_node->set_direction(Goto::taken);
1375         } else if (goto_node->default_sux() == fsux) {
1376           goto_node->set_direction(Goto::not_taken);
1377         } else {
1378           ShouldNotReachHere();
1379         }
1380       }
1381       return;
1382     }
1383   }
1384 }

1443 void GraphBuilder::table_switch() {
1444   Bytecode_tableswitch sw(stream());
1445   const int l = sw.length();
1446   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1447     // total of 2 successors => use If instead of switch
1448     // Note: This code should go into the canonicalizer as soon as it can
1449     //       can handle canonicalized forms that contain more than one node.
1450     Value key = append(new Constant(new IntConstant(sw.low_key())));
1451     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1452     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1453     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1454     // In case of loop invariant code motion or predicate insertion
1455     // before the body of a loop the state is needed
1456     ValueStack* state_before = copy_state_if_bb(is_bb);
1457     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1458   } else {
1459     // collect successors
1460     BlockList* sux = new BlockList(l + 1, NULL);
1461     int i;
1462     bool has_bb = false;
1463     for (i = 0; i < l; i++) {
1464       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1465       if (sw.dest_offset_at(i) < 0) has_bb = true;
1466     }
1467     // add default successor
1468     if (sw.default_offset() < 0) has_bb = true;
1469     sux->at_put(i, block_at(bci() + sw.default_offset()));
1470     // In case of loop invariant code motion or predicate insertion
1471     // before the body of a loop the state is needed
1472     ValueStack* state_before = copy_state_if_bb(has_bb);
1473     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1474 #ifdef ASSERT
1475     if (res->as_Goto()) {
1476       for (i = 0; i < l; i++) {
1477         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1478           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1479         }
1480       }
1481     }
1482 #endif
1483   }
1484 }

1487 void GraphBuilder::lookup_switch() {
1488   Bytecode_lookupswitch sw(stream());
1489   const int l = sw.number_of_pairs();
1490   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1491     // total of 2 successors => use If instead of switch
1492     // Note: This code should go into the canonicalizer as soon as it can
1493     //       can handle canonicalized forms that contain more than one node.
1494     // simplify to If
1495     LookupswitchPair pair = sw.pair_at(0);
1496     Value key = append(new Constant(new IntConstant(pair.match())));
1497     BlockBegin* tsux = block_at(bci() + pair.offset());
1498     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1499     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1500     // In case of loop invariant code motion or predicate insertion
1501     // before the body of a loop the state is needed
1502     ValueStack* state_before = copy_state_if_bb(is_bb);;
1503     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1504   } else {
1505     // collect successors & keys
1506     BlockList* sux = new BlockList(l + 1, NULL);
1507     intArray* keys = new intArray(l, l, 0);
1508     int i;
1509     bool has_bb = false;
1510     for (i = 0; i < l; i++) {
1511       LookupswitchPair pair = sw.pair_at(i);
1512       if (pair.offset() < 0) has_bb = true;
1513       sux->at_put(i, block_at(bci() + pair.offset()));
1514       keys->at_put(i, pair.match());
1515     }
1516     // add default successor
1517     if (sw.default_offset() < 0) has_bb = true;
1518     sux->at_put(i, block_at(bci() + sw.default_offset()));
1519     // In case of loop invariant code motion or predicate insertion
1520     // before the body of a loop the state is needed
1521     ValueStack* state_before = copy_state_if_bb(has_bb);
1522     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1523 #ifdef ASSERT
1524     if (res->as_Goto()) {
1525       for (i = 0; i < l; i++) {
1526         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1527           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1528         }
1529       }
1530     }
1531 #endif
1532   }
1533 }

1535 void GraphBuilder::call_register_finalizer() {
1536   // If the receiver requires finalization then emit code to perform
1537   // the registration on return.
1538 
1539   // Gather some type information about the receiver
1540   Value receiver = state()->local_at(0);
1541   assert(receiver != NULL, "must have a receiver");
1542   ciType* declared_type = receiver->declared_type();
1543   ciType* exact_type = receiver->exact_type();
1544   if (exact_type == NULL &&
1545       receiver->as_Local() &&
1546       receiver->as_Local()->java_index() == 0) {
1547     ciInstanceKlass* ik = compilation()->method()->holder();
1548     if (ik->is_final()) {
1549       exact_type = ik;
1550     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1551       // test class is leaf class
1552       compilation()->dependency_recorder()->assert_leaf_type(ik);
1553       exact_type = ik;
1554     } else {
1555       declared_type = ik;
1556     }
1557   }
1558 
1559   // see if we know statically that registration isn't required
1560   bool needs_check = true;
1561   if (exact_type != NULL) {
1562     needs_check = exact_type->as_instance_klass()->has_finalizer();
1563   } else if (declared_type != NULL) {
1564     ciInstanceKlass* ik = declared_type->as_instance_klass();
1565     if (!Dependencies::has_finalizable_subclass(ik)) {
1566       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1567       needs_check = false;
1568     }
1569   }
1570 
1571   if (needs_check) {
1572     // Perform the registration of finalizable objects.
1573     ValueStack* state_before = copy_state_for_exception();
1574     load_local(objectType, 0);
1575     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1576                                state()->pop_arguments(1),
1577                                true, state_before, true));
1578   }
1579 }

1582 void GraphBuilder::method_return(Value x, bool ignore_return) {
1583   if (RegisterFinalizersAtInit &&
1584       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1585     call_register_finalizer();
1586   }
1587 
1588   // The conditions for a memory barrier are described in Parse::do_exits().
1589   bool need_mem_bar = false;
1590   if (method()->is_object_constructor() &&
1591        (scope()->wrote_final() ||
1592          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1593          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1594     need_mem_bar = true;
1595   }
1596 
1597   BasicType bt = method()->return_type()->basic_type();
1598   switch (bt) {
1599     case T_BYTE:
1600     {
1601       Value shift = append(new Constant(new IntConstant(24)));
1602       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1603       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1604       break;
1605     }
1606     case T_SHORT:
1607     {
1608       Value shift = append(new Constant(new IntConstant(16)));
1609       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1610       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1611       break;
1612     }
1613     case T_CHAR:
1614     {
1615       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1616       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1617       break;
1618     }
1619     case T_BOOLEAN:
1620     {
1621       Value mask = append(new Constant(new IntConstant(1)));
1622       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1623       break;
1624     }
1625     default:
1626       break;
1627   }
1628 
1629   // Check to see whether we are inlining. If so, Return
1630   // instructions become Gotos to the continuation point.
1631   if (continuation() != NULL) {
1632 
1633     int invoke_bci = state()->caller_state()->bci();
1634 
1635     if (x != NULL  && !ignore_return) {
1636       ciMethod* caller = state()->scope()->caller()->method();
1637       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1638       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1639         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1640         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1641             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1642           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1643         }
1644       }
1645     }
1646 
1647     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1648 
1649     if (compilation()->env()->dtrace_method_probes()) {
1650       // Report exit from inline methods
1651       Values* args = new Values(1);
1652       args->push(append(new Constant(new MethodConstant(method()))));
1653       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1654     }
1655 
1656     // If the inlined method is synchronized, the monitor must be
1657     // released before we jump to the continuation block.
1658     if (method()->is_synchronized()) {
1659       assert(state()->locks_size() == 1, "receiver must be locked here");
1660       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1661     }
1662 
1663     if (need_mem_bar) {
1664       append(new MemBar(lir_membar_storestore));
1665     }
1666 
1667     // State at end of inlined method is the state of the caller
1668     // without the method parameters on stack, including the
1669     // return value, if any, of the inlined method on operand stack.
1670     set_state(state()->caller_state()->copy_for_parsing());
1671     if (x != NULL) {
1672       if (!ignore_return) {
1673         state()->push(x->type(), x);
1674       }
1675       if (profile_return() && x->type()->is_object_kind()) {
1676         ciMethod* caller = state()->scope()->method();
1677         profile_return_type(x, method(), caller, invoke_bci);
1678       }
1679     }
1680     Goto* goto_callee = new Goto(continuation(), false);
1681 
1682     // See whether this is the first return; if so, store off some
1683     // of the state for later examination
1684     if (num_returns() == 0) {
1685       set_inline_cleanup_info();
1686     }
1687 
1688     // The current bci() is in the wrong scope, so use the bci() of
1689     // the continuation point.
1690     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1691     incr_num_returns();
1692     return;
1693   }
1694 
1695   state()->truncate_stack(0);
1696   if (method()->is_synchronized()) {
1697     // perform the unlocking before exiting the method
1698     Value receiver;
1699     if (!method()->is_static()) {
1700       receiver = _initial_state->local_at(0);
1701     } else {
1702       receiver = append(new Constant(new ClassConstant(method()->holder())));
1703     }
1704     append_split(new MonitorExit(receiver, state()->unlock()));
1705   }
1706 
1707   if (need_mem_bar) {
1708       append(new MemBar(lir_membar_storestore));
1709   }
1710 
1711   assert(!ignore_return, "Ignoring return value works only for inlining");
1712   append(new Return(x));
1713 }

1744   for (int i = 0; i < vk->nof_nonstatic_fields(); i++) {
1745     ciField* inner_field = vk->nonstatic_field_at(i);
1746     assert(!inner_field->is_flattened(), "the iteration over nested fields is handled by the loop itself");
1747     int off = inner_field->offset() - vk->first_field_offset();
1748     LoadField* load = new LoadField(src, src_off + off, inner_field, false, state_before, needs_patching);
1749     Value replacement = append(load);
1750     StoreField* store = new StoreField(dest, dest_off + off, inner_field, replacement, false, state_before, needs_patching);
1751     append(store);
1752   }

1755 void GraphBuilder::access_field(Bytecodes::Code code) {
1756   bool will_link;
1757   ciField* field = stream()->get_field(will_link);
1758   ciInstanceKlass* holder = field->holder();
1759   BasicType field_type = field->type()->basic_type();
1760   ValueType* type = as_ValueType(field_type);
1761 
1762   // call will_link again to determine if the field is valid.
1763   const bool needs_patching = !holder->is_loaded() ||
1764                               !field->will_link(method(), code) ||
1765                               PatchALot;
1766 
1767   ValueStack* state_before = NULL;
1768   if (!holder->is_initialized() || needs_patching) {
1769     // save state before instruction for debug info when
1770     // deoptimization happens during patching
1771     state_before = copy_state_before();
1772   }
1773 
1774   Value obj = NULL;
1775   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1776     if (state_before != NULL) {
1777       // build a patching constant
1778       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1779     } else {
1780       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1781     }
1782   }
1783 
1784   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1785     scope()->set_wrote_final();
1786   }
1787 
1788   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1789     scope()->set_wrote_fields();
1790     if (field->is_volatile()) {
1791       scope()->set_wrote_volatile();
1792     }
1793   }
1794 
1795   const int offset = !needs_patching ? field->offset() : -1;
1796   switch (code) {
1797     case Bytecodes::_getstatic: {
1798       // check for compile-time constants, i.e., initialized static final fields
1799       Value constant = NULL;
1800       if (field->is_static_constant() && !PatchALot) {
1801         ciConstant field_value = field->constant_value();
1802         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1803                "stable static w/ default value shouldn't be a constant");
1804         constant = make_constant(field_value, field);
1805       }
1806       if (constant != NULL) {
1807         push(type, append(constant));
1808       } else {
1809         if (state_before == NULL) {
1810           state_before = copy_state_for_exception();
1811         }
1812         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1813                                         state_before, needs_patching);
1814         push(type, append(load_field));
1815       }
1816       break;
1817     }
1818     case Bytecodes::_putstatic: {
1819       Value val = pop(type);
1820       val->set_escaped();
1821       if (state_before == NULL) {
1822         state_before = copy_state_for_exception();
1823       }
1824       if (field_type == T_BOOLEAN) {
1825         Value mask = append(new Constant(new IntConstant(1)));
1826         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1827       }
1828       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1829       break;
1830     }
1831     case Bytecodes::_getfield: {
1832       // Check for compile-time constants, i.e., trusted final non-static fields.
1833       Value constant = NULL;
1834       if (state_before == NULL && field->is_flattened()) {
1835         // Save the entire state and re-execute on deopt when accessing flattened fields
1836         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1837         state_before = copy_state_before();
1838       }
1839       obj = apop();
1840       ObjectType* obj_type = obj->type()->as_ObjectType();
1841       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1842         ciObject* const_oop = obj_type->constant_value();
1843         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1844           ciConstant field_value = field->constant_value_of(const_oop);
1845           if (field_value.is_valid()) {
1846             if (field->signature()->is_Q_signature() && field_value.is_null_or_zero()) {
1847               // Non-flattened inline type field. Replace null by the default value.
1848               constant = new Constant(new InstanceConstant(field->type()->as_inline_klass()->default_instance()));
1849             } else {
1850               constant = make_constant(field_value, field);
1851             }
1852             // For CallSite objects add a dependency for invalidation of the optimization.
1853             if (field->is_call_site_target()) {
1854               ciCallSite* call_site = const_oop->as_call_site();
1855               if (!call_site->is_fully_initialized_constant_call_site()) {
1856                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1857                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1858               }
1859             }
1860           }
1861         }
1862       }
1863       if (constant != NULL) {
1864         push(type, append(constant));
1865       } else {
1866         if (state_before == NULL) {
1867           state_before = copy_state_for_exception();
1868         }
1869         if (!field->is_flattened()) {
1870           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1871           Value replacement = !needs_patching ? _memory->load(load) : load;
1872           if (replacement != load) {
1873             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1874             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1875             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1876             BasicType bt = field->type()->basic_type();
1877             switch (bt) {
1878             case T_BOOLEAN:
1879             case T_BYTE:
1880               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1881               break;
1882             case T_CHAR:
1883               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1884               break;
1885             case T_SHORT:
1886               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1887               break;
1888             default:
1889               break;
1890             }
1891             push(type, replacement);
1892           } else {
1893             push(type, append(load));
1894           }
1895         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1896           assert(field->type()->is_inlinetype(), "Sanity check");
1897           ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1898           int flattening_offset = field->offset() - inline_klass->first_field_offset();
1899           assert(field->type()->is_inlinetype(), "Sanity check");
1900           scope()->set_wrote_final();
1901           scope()->set_wrote_fields();
1902           NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(inline_klass, state_before, false);
1903           _memory->new_instance(new_instance);
1904           apush(append_split(new_instance));
1905           copy_inline_content(inline_klass, obj, field->offset(), new_instance, inline_klass->first_field_offset(),
1906                        state_before, needs_patching);
1907         }
1908       }
1909       break;
1910     }
1911     case Bytecodes::_withfield:
1912     case Bytecodes::_putfield: {
1913       Value val = pop(type);
1914       val->set_escaped();
1915       obj = apop();
1916       if (state_before == NULL) {
1917         state_before = copy_state_for_exception();
1918       }
1919       if (field->type()->basic_type() == T_BOOLEAN) {
1920         Value mask = append(new Constant(new IntConstant(1)));
1921         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1922       }
1923 
1924       if (!field->is_flattened()) {
1925         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1926         if (!needs_patching) store = _memory->store(store);
1927         if (store != NULL) {
1928           append(store);
1929         }
1930       } else {
1931         assert(field->type()->is_inlinetype(), "Sanity check");
1932         ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1933         int flattening_offset = field->offset() - inline_klass->first_field_offset();
1934         copy_inline_content(inline_klass, val, inline_klass->first_field_offset(), obj, field->offset(),
1935                    state_before, needs_patching);
1936       }
1937       break;
1938     }
1939     default:
1940       ShouldNotReachHere();
1941       break;
1942   }
1943 }

1945 // Baseline version of withfield, allocate every time
1946 void GraphBuilder::withfield(int field_index)
1947 {
1948   bool will_link;
1949   ciField* field_modify = stream()->get_field(will_link);
1950   ciInstanceKlass* holder = field_modify->holder();
1951   BasicType field_type = field_modify->type()->basic_type();
1952   ValueType* type = as_ValueType(field_type);
1953 
1954   // call will_link again to determine if the field is valid.
1955   const bool needs_patching = !holder->is_loaded() ||
1956                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1957                               PatchALot;
1958 
1959 
1960   scope()->set_wrote_final();
1961   scope()->set_wrote_fields();
1962 
1963   const int offset = !needs_patching ? field_modify->offset() : -1;
1964 
1965   if (!holder->is_loaded()
1966       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1967       ) {
1968     ValueStack* state_before = copy_state_before();
1969     Value val = pop(type);
1970     Value obj = apop();
1971     apush(append_split(new WithField(state_before)));
1972     return;
1973   }
1974   ValueStack* state_before = copy_state_before();
1975 
1976   Value val = pop(type);
1977   Value obj = apop();
1978 
1979   assert(holder->is_inlinetype(), "must be a value klass");
1980   // Save the entire state and re-execute on deopt when executing withfield
1981   state_before->set_should_reexecute(true);
1982   NewInlineTypeInstance* new_instance;
1983   if (obj->as_NewInlineTypeInstance() != NULL && obj->as_NewInlineTypeInstance()->in_larval_state()) {
1984     new_instance = obj->as_NewInlineTypeInstance();
1985     apush(append_split(new_instance));
1986   } else {
1987     new_instance = new NewInlineTypeInstance(holder->as_inline_klass(), state_before, false);
1988     _memory->new_instance(new_instance);
1989     apush(append_split(new_instance));
1990 
1991     for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1992       ciField* field = holder->nonstatic_field_at(i);
1993       int off = field->offset();
1994 
1995       if (field->offset() != offset) {
1996         if (field->is_flattened()) {
1997           assert(field->type()->is_inlinetype(), "Sanity check");
1998           assert(field->type()->is_inlinetype(), "Only inline types can be flattened");
1999           ciInlineKlass* vk = field->type()->as_inline_klass();
2000           copy_inline_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2001         } else {
2002           // Only load those fields who are not modified
2003           LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2004           Value replacement = append(load);
2005           StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2006           append(store);
2007         }
2008       }
2009     }
2010   }
2011 
2012   // Field to modify
2013   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2014     Value mask = append(new Constant(new IntConstant(1)));
2015     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2016   }
2017   if (field_modify->is_flattened()) {
2018     assert(field_modify->type()->is_inlinetype(), "Only inline types can be flattened");
2019     ciInlineKlass* vk = field_modify->type()->as_inline_klass();
2020     copy_inline_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2021   } else {
2022     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2023     append(store);
2024   }
2025 }

2033 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2034   int n = 0;
2035   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2036   start = has_receiver ? 1 : 0;
2037   if (profile_arguments()) {
2038     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2039     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2040       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2041     }
2042   }
2043   // If we are inlining then we need to collect arguments to profile parameters for the target
2044   if (profile_parameters() && target != NULL) {
2045     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2046       // The receiver is profiled on method entry so it's included in
2047       // the number of parameters but here we're only interested in
2048       // actual arguments.
2049       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2050     }
2051   }
2052   if (n > 0) {
2053     return new Values(n);
2054   }
2055   return NULL;
2056 }

2087 void GraphBuilder::invoke(Bytecodes::Code code) {
2088   bool will_link;
2089   ciSignature* declared_signature = NULL;
2090   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2091   ciKlass*              holder = stream()->get_declared_method_holder();
2092   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2093   assert(declared_signature != NULL, "cannot be null");
2094   assert(will_link == target->is_loaded(), "");
2095 
2096   ciInstanceKlass* klass = target->holder();
2097   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2098 
2099   // check if CHA possible: if so, change the code to invoke_special
2100   ciInstanceKlass* calling_klass = method()->holder();
2101   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2102   ciInstanceKlass* actual_recv = callee_holder;
2103 
2104   CompileLog* log = compilation()->log();
2105   if (log != NULL)
2106       log->elem("call method='%d' instr='%s'",
2107                 log->identify(target),
2108                 Bytecodes::name(code));
2109 
2110   // invoke-special-super
2111   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2112     ciInstanceKlass* sender_klass =
2113           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2114                                                  calling_klass;
2115     if (sender_klass->is_interface()) {
2116       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2117       Value receiver = state()->stack_at(index);
2118       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2119       c->set_invokespecial_receiver_check();
2120       state()->stack_at_put(index, append_split(c));
2121     }
2122   }
2123 
2124   // Some methods are obviously bindable without any type checks so
2125   // convert them directly to an invokespecial or invokestatic.
2126   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2127     switch (bc_raw) {
2128     case Bytecodes::_invokevirtual:
2129       code = Bytecodes::_invokespecial;
2130       break;
2131     case Bytecodes::_invokehandle:
2132       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2133       break;
2134     default:
2135       break;
2136     }
2137   } else {
2138     if (bc_raw == Bytecodes::_invokehandle) {
2139       assert(!will_link, "should come here only for unlinked call");
2140       code = Bytecodes::_invokespecial;
2141     }
2142   }
2143 
2144   // Push appendix argument (MethodType, CallSite, etc.), if one.
2145   bool patch_for_appendix = false;
2146   int patching_appendix_arg = 0;
2147   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2148     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2149     apush(arg);
2150     patch_for_appendix = true;
2151     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2152   } else if (stream()->has_appendix()) {
2153     ciObject* appendix = stream()->get_appendix();
2154     Value arg = append(new Constant(new ObjectConstant(appendix)));
2155     apush(arg);
2156   }
2157 
2158   ciMethod* cha_monomorphic_target = NULL;
2159   ciMethod* exact_target = NULL;
2160   Value better_receiver = NULL;
2161   if (UseCHA && DeoptC1 && target->is_loaded() &&
2162       !(// %%% FIXME: Are both of these relevant?
2163         target->is_method_handle_intrinsic() ||
2164         target->is_compiled_lambda_form()) &&
2165       !patch_for_appendix) {
2166     Value receiver = NULL;
2167     ciInstanceKlass* receiver_klass = NULL;
2168     bool type_is_exact = false;
2169     // try to find a precise receiver type
2170     if (will_link && !target->is_static()) {
2171       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2172       receiver = state()->stack_at(index);
2173       ciType* type = receiver->exact_type();
2174       if (type != NULL && type->is_loaded() &&
2175           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2176         receiver_klass = (ciInstanceKlass*) type;
2177         type_is_exact = true;
2178       }
2179       if (type == NULL) {
2180         type = receiver->declared_type();
2181         if (type != NULL && type->is_loaded() &&
2182             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2183           receiver_klass = (ciInstanceKlass*) type;
2184           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2185             // Insert a dependency on this type since
2186             // find_monomorphic_target may assume it's already done.
2187             dependency_recorder()->assert_leaf_type(receiver_klass);
2188             type_is_exact = true;
2189           }
2190         }
2191       }
2192     }
2193     if (receiver_klass != NULL && type_is_exact &&
2194         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2195       // If we have the exact receiver type we can bind directly to
2196       // the method to call.
2197       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2198       if (exact_target != NULL) {
2199         target = exact_target;
2200         code = Bytecodes::_invokespecial;
2201       }
2202     }
2203     if (receiver_klass != NULL &&
2204         receiver_klass->is_subtype_of(actual_recv) &&
2205         actual_recv->is_initialized()) {
2206       actual_recv = receiver_klass;
2207     }
2208 
2209     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2210         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2211       // Use CHA on the receiver to select a more precise method.
2212       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2213     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2214       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2215       // If there is only one implementor of this interface then we
2216       // may be able bind this invoke directly to the implementing
2217       // klass but we need both a dependence on the single interface
2218       // and on the method we bind to.  Additionally since all we know
2219       // about the receiver type is the it's supposed to implement the
2220       // interface we have to insert a check that it's the class we
2221       // expect.  Interface types are not checked by the verifier so
2222       // they are roughly equivalent to Object.
2223       // The number of implementors for declared_interface is less or
2224       // equal to the number of implementors for target->holder() so
2225       // if number of implementors of target->holder() == 1 then
2226       // number of implementors for decl_interface is 0 or 1. If
2227       // it's 0 then no class implements decl_interface and there's
2228       // no point in inlining.
2229       ciInstanceKlass* declared_interface = callee_holder;
2230       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2231       if (singleton != NULL &&
2232           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2233         assert(singleton != declared_interface, "not a unique implementor");
2234         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2235         if (cha_monomorphic_target != NULL) {
2236           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2237             // If CHA is able to bind this invoke then update the class
2238             // to match that class, otherwise klass will refer to the
2239             // interface.
2240             klass = cha_monomorphic_target->holder();
2241             actual_recv = declared_interface;
2242 
2243             // insert a check it's really the expected class.
2244             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2245             c->set_incompatible_class_change_check();
2246             c->set_direct_compare(klass->is_final());
2247             // pass the result of the checkcast so that the compiler has
2248             // more accurate type info in the inlinee
2249             better_receiver = append_split(c);
2250           } else {
2251             cha_monomorphic_target = NULL; // subtype check against Object is useless
2252           }
2253         }
2254       }
2255     }
2256   }
2257 
2258   if (cha_monomorphic_target != NULL) {
2259     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2260     assert(!cha_monomorphic_target->is_abstract(), "");
2261     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2262       // If we inlined because CHA revealed only a single target method,
2263       // then we are dependent on that target method not getting overridden
2264       // by dynamic class loading.  Be sure to test the "static" receiver
2265       // dest_method here, as opposed to the actual receiver, which may
2266       // falsely lead us to believe that the receiver is final or private.
2267       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2268     }
2269     code = Bytecodes::_invokespecial;
2270   }
2271 
2272   // check if we could do inlining
2273   if (!PatchALot && Inline && target->is_loaded() &&
2274       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2275       && !patch_for_appendix) {
2276     // callee is known => check if we have static binding
2277     if (code == Bytecodes::_invokestatic  ||
2278         code == Bytecodes::_invokespecial ||
2279         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2280         code == Bytecodes::_invokedynamic) {
2281       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2282       // static binding => check if callee is ok
2283       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2284 
2285       CHECK_BAILOUT();
2286       clear_inline_bailout();
2287 
2288       if (success) {
2289         // Register dependence if JVMTI has either breakpoint
2290         // setting or hotswapping of methods capabilities since they may
2291         // cause deoptimization.
2292         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2293           dependency_recorder()->assert_evol_method(inline_target);
2294         }
2295         return;
2296       }
2297     } else {
2298       print_inlining(target, "no static binding", /*success*/ false);
2299     }
2300   } else {
2301     print_inlining(target, "not inlineable", /*success*/ false);
2302   }
2303 
2304   // If we attempted an inline which did not succeed because of a
2305   // bailout during construction of the callee graph, the entire
2306   // compilation has to be aborted. This is fairly rare and currently
2307   // seems to only occur for jasm-generated classes which contain
2308   // jsr/ret pairs which are not associated with finally clauses and
2309   // do not have exception handlers in the containing method, and are
2310   // therefore not caught early enough to abort the inlining without
2311   // corrupting the graph. (We currently bail out with a non-empty
2312   // stack at a ret in these situations.)
2313   CHECK_BAILOUT();
2314 
2315   // inlining not successful => standard invoke
2316   ValueType* result_type = as_ValueType(declared_signature->return_type());
2317   ValueStack* state_before = copy_state_exhandling();
2318 
2319   // The bytecode (code) might change in this method so we are checking this very late.
2320   const bool has_receiver =
2321     code == Bytecodes::_invokespecial   ||
2322     code == Bytecodes::_invokevirtual   ||
2323     code == Bytecodes::_invokeinterface;
2324   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2325   Value recv = has_receiver ? apop() : NULL;
2326   int vtable_index = Method::invalid_vtable_index;
2327 
2328   // A null check is required here (when there is a receiver) for any of the following cases
2329   // - invokespecial, always need a null check.
2330   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2331   //   and require null checking. If the target is loaded a null check is emitted here.
2332   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2333   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2334   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2335   //   potentially fail, and can't have the null check before the resolution.)
2336   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2337   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2338   //
2339   // Normal invokevirtual will perform the null check during lookup
2340 
2341   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2342       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2343 
2344   if (need_null_check) {
2345     if (recv != NULL) {
2346       null_check(recv);
2347     }
2348 
2349     if (is_profiling()) {
2350       // Note that we'd collect profile data in this method if we wanted it.
2351       compilation()->set_would_profile(true);
2352 
2353       if (profile_calls()) {
2354         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2355         ciKlass* target_klass = NULL;
2356         if (cha_monomorphic_target != NULL) {
2357           target_klass = cha_monomorphic_target->holder();
2358         } else if (exact_target != NULL) {
2359           target_klass = exact_target->holder();
2360         }
2361         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2362       }
2363     }
2364   }
2365 
2366   if (recv != NULL) {
2367     recv->set_escaped();
2368   }
2369   for (int i=0; i<args->length(); i++) {
2370     args->at(0)->set_escaped();
2371   }
2372 
2373   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2374                               declared_signature->return_type()->is_inlinetype());
2375   // push result
2376   append_split(result);
2377 
2378   if (result_type != voidType) {
2379     if (method()->is_strict()) {
2380       push(result_type, round_fp(result));
2381     } else {
2382       push(result_type, result);
2383     }
2384   }
2385   if (profile_return() && result_type->is_object_kind()) {
2386     profile_return_type(result, target);
2387   }
2388 }

2571 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2572   Canonicalizer canon(compilation(), instr, bci);
2573   Instruction* i1 = canon.canonical();
2574   if (i1->is_linked() || !i1->can_be_linked()) {
2575     // Canonicalizer returned an instruction which was already
2576     // appended so simply return it.
2577     return i1;
2578   }
2579 
2580   if (UseLocalValueNumbering) {
2581     // Lookup the instruction in the ValueMap and add it to the map if
2582     // it's not found.
2583     Instruction* i2 = vmap()->find_insert(i1);
2584     if (i2 != i1) {
2585       // found an entry in the value map, so just return it.
2586       assert(i2->is_linked(), "should already be linked");
2587       return i2;
2588     }
2589     ValueNumberingEffects vne(vmap());
2590     i1->visit(&vne);
2591   }
2592 
2593   // i1 was not eliminated => append it
2594   assert(i1->next() == NULL, "shouldn't already be linked");
2595   _last = _last->set_next(i1, canon.bci());
2596 
2597   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2598     // set the bailout state but complete normal processing.  We
2599     // might do a little more work before noticing the bailout so we
2600     // want processing to continue normally until it's noticed.
2601     bailout("Method and/or inlining is too large");
2602   }
2603 
2604 #ifndef PRODUCT
2605   if (PrintIRDuringConstruction) {
2606     InstructionPrinter ip;
2607     ip.print_line(i1);
2608     if (Verbose) {
2609       state()->print();
2610     }
2611   }
2612 #endif
2613 
2614   // save state after modification of operand stack for StateSplit instructions
2615   StateSplit* s = i1->as_StateSplit();
2616   if (s != NULL) {
2617     if (EliminateFieldAccess) {
2618       Intrinsic* intrinsic = s->as_Intrinsic();
2619       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2620         _memory->kill();
2621       }
2622     }
2623     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2624   }
2625 
2626   // set up exception handlers for this instruction if necessary
2627   if (i1->can_trap()) {
2628     i1->set_exception_handlers(handle_exception(i1));
2629     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2630   }
2631   return i1;
2632 }

2666 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2667   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2668     assert(instruction->exception_state() == NULL
2669            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2670            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2671            "exception_state should be of exception kind");
2672     return new XHandlers();
2673   }
2674 
2675   XHandlers*  exception_handlers = new XHandlers();
2676   ScopeData*  cur_scope_data = scope_data();
2677   ValueStack* cur_state = instruction->state_before();
2678   ValueStack* prev_state = NULL;
2679   int scope_count = 0;
2680 
2681   assert(cur_state != NULL, "state_before must be set");
2682   do {
2683     int cur_bci = cur_state->bci();
2684     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2685     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2686 
2687     // join with all potential exception handlers
2688     XHandlers* list = cur_scope_data->xhandlers();
2689     const int n = list->length();
2690     for (int i = 0; i < n; i++) {
2691       XHandler* h = list->handler_at(i);
2692       if (h->covers(cur_bci)) {
2693         // h is a potential exception handler => join it
2694         compilation()->set_has_exception_handlers(true);
2695 
2696         BlockBegin* entry = h->entry_block();
2697         if (entry == block()) {
2698           // It's acceptable for an exception handler to cover itself
2699           // but we don't handle that in the parser currently.  It's
2700           // very rare so we bailout instead of trying to handle it.
2701           BAILOUT_("exception handler covers itself", exception_handlers);
2702         }
2703         assert(entry->bci() == h->handler_bci(), "must match");
2704         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2705 
2706         // previously this was a BAILOUT, but this is not necessary
2707         // now because asynchronous exceptions are not handled this way.
2708         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2709 
2710         // xhandler start with an empty expression stack
2711         if (cur_state->stack_size() != 0) {
2712           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2713         }
2714         if (instruction->exception_state() == NULL) {
2715           instruction->set_exception_state(cur_state);
2716         }
2717 
2718         // Note: Usually this join must work. However, very
2719         // complicated jsr-ret structures where we don't ret from
2720         // the subroutine can cause the objects on the monitor
2721         // stacks to not match because blocks can be parsed twice.
2722         // The only test case we've seen so far which exhibits this
2723         // problem is caught by the infinite recursion test in
2724         // GraphBuilder::jsr() if the join doesn't work.
2725         if (!entry->try_merge(cur_state)) {
2726           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2727         }
2728 
2729         // add current state for correct handling of phi functions at begin of xhandler
2730         int phi_operand = entry->add_exception_state(cur_state);
2731 
2732         // add entry to the list of xhandlers of this block
2733         _block->add_exception_handler(entry);
2734 
2735         // add back-edge from xhandler entry to this block
2736         if (!entry->is_predecessor(_block)) {
2737           entry->add_predecessor(_block);
2738         }
2739 
2740         // clone XHandler because phi_operand and scope_count can not be shared
2741         XHandler* new_xhandler = new XHandler(h);
2742         new_xhandler->set_phi_operand(phi_operand);
2743         new_xhandler->set_scope_count(scope_count);
2744         exception_handlers->append(new_xhandler);
2745 
2746         // fill in exception handler subgraph lazily
2747         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2748         cur_scope_data->add_to_work_list(entry);
2749 
2750         // stop when reaching catchall
2751         if (h->catch_type() == 0) {
2752           return exception_handlers;
2753         }
2754       }
2755     }
2756 
2757     if (exception_handlers->length() == 0) {
2758       // This scope and all callees do not handle exceptions, so the local
2759       // variables of this scope are not needed. However, the scope itself is
2760       // required for a correct exception stack trace -> clear out the locals.
2761       if (_compilation->env()->should_retain_local_variables()) {
2762         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2763       } else {
2764         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2765       }
2766       if (prev_state != NULL) {
2767         prev_state->set_caller_state(cur_state);
2768       }
2769       if (instruction->exception_state() == NULL) {
2770         instruction->set_exception_state(cur_state);
2771       }
2772     }
2773 
2774     // Set up iteration for next time.
2775     // If parsing a jsr, do not grab exception handlers from the
2776     // parent scopes for this method (already got them, and they
2777     // needed to be cloned)
2778 
2779     while (cur_scope_data->parsing_jsr()) {
2780       cur_scope_data = cur_scope_data->parent();
2781     }
2782 
2783     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2784     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2785 
2786     prev_state = cur_state;
2787     cur_state = cur_state->caller_state();
2788     cur_scope_data = cur_scope_data->parent();
2789     scope_count++;
2790   } while (cur_scope_data != NULL);
2791 
2792   return exception_handlers;
2793 }

2814 Value PhiSimplifier::simplify(Value v) {
2815   Phi* phi = v->as_Phi();
2816 
2817   if (phi == NULL) {
2818     // no phi function
2819     return v;
2820   } else if (v->has_subst()) {
2821     // already substituted; subst can be phi itself -> simplify
2822     return simplify(v->subst());

2829   } else if (phi->type()->is_illegal()) {
2830     // illegal phi functions are ignored anyway
2831     return phi;
2832 
2833   } else {
2834     // mark phi function as processed to break cycles in phi functions
2835     phi->set(Phi::visited);
2836 
2837     // simplify x = [y, x] and x = [y, y] to y
2838     Value subst = NULL;
2839     int opd_count = phi->operand_count();
2840     for (int i = 0; i < opd_count; i++) {
2841       Value opd = phi->operand_at(i);
2842       assert(opd != NULL, "Operand must exist!");
2843 
2844       if (opd->type()->is_illegal()) {
2845         // if one operand is illegal, the entire phi function is illegal
2846         phi->make_illegal();
2847         phi->clear(Phi::visited);
2848         return phi;
2849       }
2850 
2851       Value new_opd = simplify(opd);
2852       assert(new_opd != NULL, "Simplified operand must exist!");
2853 
2854       if (new_opd != phi && new_opd != subst) {
2855         if (subst == NULL) {
2856           subst = new_opd;
2857         } else {
2858           // no simplification possible
2859           phi->set(Phi::cannot_simplify);
2860           phi->clear(Phi::visited);
2861           return phi;
2862         }
2863       }
2864     }
2865 
2866     // sucessfully simplified phi function
2867     assert(subst != NULL, "illegal phi function");
2868     _has_substitutions = true;
2869     phi->clear(Phi::visited);
2870     phi->set_subst(subst);
2871 
2872 #ifndef PRODUCT
2873     if (PrintPhiFunctions) {
2874       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());
2875     }
2876 #endif
2877 
2878     return subst;
2879   }
2880 }

3228       start_in_current_block_for_inlining = false;
3229     } else {
3230       BlockBegin* b;
3231       while ((b = scope_data()->remove_from_work_list()) != NULL) {
3232         if (!b->is_set(BlockBegin::was_visited_flag)) {
3233           if (b->is_set(BlockBegin::osr_entry_flag)) {
3234             // we're about to parse the osr entry block, so make sure
3235             // we setup the OSR edge leading into this block so that
3236             // Phis get setup correctly.
3237             setup_osr_entry_block();
3238             // this is no longer the osr entry block, so clear it.
3239             b->clear(BlockBegin::osr_entry_flag);
3240           }
3241           b->set(BlockBegin::was_visited_flag);
3242           connect_to_end(b);
3243         }
3244       }
3245     }
3246   } while (!bailed_out() && !scope_data()->is_work_list_empty());
3247 }

3373 void GraphBuilder::setup_osr_entry_block() {
3374   assert(compilation()->is_osr_compile(), "only for osrs");
3375 
3376   int osr_bci = compilation()->osr_bci();
3377   ciBytecodeStream s(method());
3378   s.reset_to_bci(osr_bci);
3379   s.next();
3380   scope_data()->set_stream(&s);
3381 
3382   // create a new block to be the osr setup code
3383   _osr_entry = new BlockBegin(osr_bci);
3384   _osr_entry->set(BlockBegin::osr_entry_flag);
3385   _osr_entry->set_depth_first_number(0);
3386   BlockBegin* target = bci2block()->at(osr_bci);
3387   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3388   // the osr entry has no values for locals
3389   ValueStack* state = target->state()->copy();
3390   _osr_entry->set_state(state);
3391 
3392   kill_all();
3393   _block = _osr_entry;
3394   _state = _osr_entry->state()->copy();
3395   assert(_state->bci() == osr_bci, "mismatch");
3396   _last  = _osr_entry;
3397   Value e = append(new OsrEntry());
3398   e->set_needs_null_check(false);
3399 
3400   // OSR buffer is
3401   //
3402   // locals[nlocals-1..0]
3403   // monitors[number_of_locks-1..0]
3404   //
3405   // locals is a direct copy of the interpreter frame so in the osr buffer
3406   // so first slot in the local array is the last local from the interpreter
3407   // and last slot is local[0] (receiver) from the interpreter
3408   //
3409   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3410   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3411   // in the interpreter frame (the method lock if a sync method)
3412 
3413   // Initialize monitors in the compiled activation.
3414 
3415   int index;
3416   Value local;
3417 
3418   // find all the locals that the interpreter thinks contain live oops
3419   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3420 
3421   // compute the offset into the locals so that we can treat the buffer
3422   // as if the locals were still in the interpreter frame
3423   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3424   for_each_local_value(state, index, local) {
3425     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3426     Value get;
3427     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3428       // The interpreter thinks this local is dead but the compiler
3429       // doesn't so pretend that the interpreter passed in null.
3430       get = append(new Constant(objectNull));
3431     } else {
3432       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3433                                     append(new Constant(new IntConstant(offset))),
3434                                     0,
3435                                     true /*unaligned*/, true /*wide*/));
3436     }
3437     _state->store_local(index, get);
3438   }
3439 
3440   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3441 
3442   assert(state->caller_state() == NULL, "should be top scope");
3443   state->clear_locals();
3444   Goto* g = new Goto(target, false);
3445   append(g);
3446   _osr_entry->set_end(g);
3447   target->merge(_osr_entry->end()->state());
3448 
3449   scope_data()->set_stream(NULL);
3450 }

3453 ValueStack* GraphBuilder::state_at_entry() {
3454   ValueStack* state = new ValueStack(scope(), NULL);
3455 
3456   // Set up locals for receiver
3457   int idx = 0;
3458   if (!method()->is_static()) {
3459     // we should always see the receiver
3460     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3461              /*receiver*/ true, /*null_free*/ method()->holder()->is_flat_array_klass()));
3462     idx = 1;
3463   }
3464 
3465   // Set up locals for incoming arguments
3466   ciSignature* sig = method()->signature();
3467   for (int i = 0; i < sig->count(); i++) {
3468     ciType* type = sig->type_at(i);
3469     BasicType basic_type = type->basic_type();
3470     // don't allow T_ARRAY to propagate into locals types
3471     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3472     ValueType* vt = as_ValueType(basic_type);
3473     state->store_local(idx, new Local(type, vt, idx, false, type->is_inlinetype()));
3474     idx += type->size();
3475   }
3476 
3477   // lock synchronized method
3478   if (method()->is_synchronized()) {
3479     state->lock(NULL);
3480   }
3481 
3482   return state;
3483 }

3487   : _scope_data(NULL)
3488   , _compilation(compilation)
3489   , _memory(new MemoryBuffer())
3490   , _inline_bailout_msg(NULL)
3491   , _instruction_count(0)
3492   , _osr_entry(NULL)
3493 {
3494   int osr_bci = compilation->osr_bci();
3495 
3496   // determine entry points and bci2block mapping
3497   BlockListBuilder blm(compilation, scope, osr_bci);
3498   CHECK_BAILOUT();
3499 
3500   BlockList* bci2block = blm.bci2block();
3501   BlockBegin* start_block = bci2block->at(0);
3502 
3503   push_root_scope(scope, bci2block, start_block);
3504 
3505   // setup state for std entry
3506   _initial_state = state_at_entry();

3508 
3509   // complete graph
3510   _vmap        = new ValueMap();
3511   switch (scope->method()->intrinsic_id()) {
3512   case vmIntrinsics::_dabs          : // fall through
3513   case vmIntrinsics::_dsqrt         : // fall through
3514   case vmIntrinsics::_dsin          : // fall through
3515   case vmIntrinsics::_dcos          : // fall through
3516   case vmIntrinsics::_dtan          : // fall through
3517   case vmIntrinsics::_dlog          : // fall through
3518   case vmIntrinsics::_dlog10        : // fall through
3519   case vmIntrinsics::_dexp          : // fall through
3520   case vmIntrinsics::_dpow          : // fall through
3521     {
3522       // Compiles where the root method is an intrinsic need a special
3523       // compilation environment because the bytecodes for the method
3524       // shouldn't be parsed during the compilation, only the special
3525       // Intrinsic node should be emitted.  If this isn't done the the
3526       // code for the inlined version will be different than the root
3527       // compiled version which could lead to monotonicity problems on
3528       // intel.
3529       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3530         BAILOUT("failed to inline intrinsic, method not annotated");
3531       }
3532 
3533       // Set up a stream so that appending instructions works properly.
3534       ciBytecodeStream s(scope->method());
3535       s.reset_to_bci(0);
3536       scope_data()->set_stream(&s);
3537       s.next();
3538 
3539       // setup the initial block state
3540       _block = start_block;
3541       _state = start_block->state()->copy_for_parsing();
3542       _last  = start_block;
3543       load_local(doubleType, 0);

3547

3549       bool result = try_inline_intrinsics(scope->method());

3551       method_return(dpop());
3552 
3553       // connect the begin and end blocks and we're all done.
3554       BlockEnd* end = last()->as_BlockEnd();

3556       break;
3557     }
3558 
3559   case vmIntrinsics::_Reference_get:
3560     {
3561       {
3562         // With java.lang.ref.reference.get() we must go through the
3563         // intrinsic - when G1 is enabled - even when get() is the root
3564         // method of the compile so that, if necessary, the value in
3565         // the referent field of the reference object gets recorded by
3566         // the pre-barrier code.
3567         // Specifically, if G1 is enabled, the value in the referent
3568         // field is recorded by the G1 SATB pre barrier. This will
3569         // result in the referent being marked live and the reference
3570         // object removed from the list of discovered references during
3571         // reference processing.
3572         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3573           BAILOUT("failed to inline intrinsic, method not annotated");
3574         }
3575 
3576         // Also we need intrinsic to prevent commoning reads from this field
3577         // across safepoint since GC can change its value.
3578 
3579         // Set up a stream so that appending instructions works properly.
3580         ciBytecodeStream s(scope->method());
3581         s.reset_to_bci(0);
3582         scope_data()->set_stream(&s);
3583         s.next();
3584 
3585         // setup the initial block state
3586         _block = start_block;
3587         _state = start_block->state()->copy_for_parsing();
3588         _last  = start_block;
3589         load_local(objectType, 0);
3590 
3591         // Emit the intrinsic node.
3592         bool result = try_inline_intrinsics(scope->method());
3593         if (!result) BAILOUT("failed to inline intrinsic");
3594         method_return(apop());
3595 
3596         // connect the begin and end blocks and we're all done.
3597         BlockEnd* end = last()->as_BlockEnd();
3598         block()->set_end(end);
3599         break;
3600       }
3601       // Otherwise, fall thru
3602     }
3603 
3604   default:
3605     scope_data()->add_to_work_list(start_block);
3606     iterate_all_blocks();
3607     break;
3608   }
3609   CHECK_BAILOUT();
3610 
3611   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3612 
3613   eliminate_redundant_phis(_start);
3614 
3615   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3616   // for osr compile, bailout if some requirements are not fulfilled
3617   if (osr_bci != -1) {
3618     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3619     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3620       BAILOUT("osr entry must have been visited for osr compile");
3621     }
3622 
3623     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3624     if (!osr_block->state()->stack_is_empty()) {
3625       BAILOUT("stack not empty at OSR entry point");
3626     }
3627   }
3628 #ifndef PRODUCT
3629   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3630 #endif

3678 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3679   const char* msg = NULL;
3680 
3681   // clear out any existing inline bailout condition
3682   clear_inline_bailout();
3683 
3684   // exclude methods we don't want to inline
3685   msg = should_not_inline(callee);
3686   if (msg != NULL) {
3687     print_inlining(callee, msg, /*success*/ false);
3688     return false;
3689   }
3690 
3691   // method handle invokes
3692   if (callee->is_method_handle_intrinsic()) {
3693     if (try_method_handle_inline(callee, ignore_return)) {
3694       if (callee->has_reserved_stack_access()) {
3695         compilation()->set_has_reserved_stack_access(true);
3696       }
3697       return true;
3698     }
3699     return false;
3700   }
3701 
3702   // handle intrinsics
3703   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3704       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3705     if (try_inline_intrinsics(callee, ignore_return)) {
3706       print_inlining(callee, "intrinsic");
3707       if (callee->has_reserved_stack_access()) {
3708         compilation()->set_has_reserved_stack_access(true);
3709       }
3710       return true;
3711     }
3712     // try normal inlining
3713   }
3714 
3715   // certain methods cannot be parsed at all
3716   msg = check_can_parse(callee);
3717   if (msg != NULL) {
3718     print_inlining(callee, msg, /*success*/ false);
3719     return false;
3720   }
3721 
3722   // If bytecode not set use the current one.
3723   if (bc == Bytecodes::_illegal) {
3724     bc = code();
3725   }
3726   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3727     if (callee->has_reserved_stack_access()) {
3728       compilation()->set_has_reserved_stack_access(true);
3729     }
3730     return true;
3731   }
3732 
3733   // Entire compilation could fail during try_inline_full call.
3734   // In that case printing inlining decision info is useless.
3735   if (!bailed_out())
3736     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3737 
3738   return false;
3739 }

3886 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3887   // Introduce a new callee continuation point - all Ret instructions
3888   // will be replaced with Gotos to this point.
3889   BlockBegin* cont = block_at(next_bci());
3890   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3891 
3892   // Note: can not assign state to continuation yet, as we have to
3893   // pick up the state from the Ret instructions.
3894 
3895   // Push callee scope
3896   push_scope_for_jsr(cont, jsr_dest_bci);
3897 
3898   // Temporarily set up bytecode stream so we can append instructions
3899   // (only using the bci of this stream)
3900   scope_data()->set_stream(scope_data()->parent()->stream());
3901 
3902   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3903   assert(jsr_start_block != NULL, "jsr start block must exist");
3904   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3905   Goto* goto_sub = new Goto(jsr_start_block, false);
3906   // Must copy state to avoid wrong sharing when parsing bytecodes
3907   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3908   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3909   append(goto_sub);
3910   _block->set_end(goto_sub);
3911   _last = _block = jsr_start_block;
3912 
3913   // Clear out bytecode stream
3914   scope_data()->set_stream(NULL);
3915 
3916   scope_data()->add_to_work_list(jsr_start_block);
3917 
3918   // Ready to resume parsing in subroutine
3919   iterate_all_blocks();
3920 
3921   // If we bailed out during parsing, return immediately (this is bad news)
3922   CHECK_BAILOUT_(false);
3923 
3924   // Detect whether the continuation can actually be reached. If not,
3925   // it has not had state set by the join() operations in
3926   // iterate_bytecodes_for_block()/ret() and we should not touch the
3927   // iteration state. The calling activation of
3928   // iterate_bytecodes_for_block will then complete normally.
3929   if (cont->state() != NULL) {
3930     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3931       // add continuation to work list instead of parsing it immediately
3932       scope_data()->parent()->add_to_work_list(cont);
3933     }
3934   }
3935 
3936   assert(jsr_continuation() == cont, "continuation must not have changed");
3937   assert(!jsr_continuation()->is_set(BlockBegin::was_visited_flag) ||
3938          jsr_continuation()->is_set(BlockBegin::parser_loop_header_flag),
3939          "continuation can only be visited in case of backward branches");
3940   assert(_last && _last->as_BlockEnd(), "block must have end");
3941 
3942   // continuation is in work list, so end iteration of current block
3943   _skip_block = true;
3944   pop_scope_for_jsr();
3945 
3946   return true;
3947 }

4038   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4039     INLINE_BAILOUT("inlining prohibited by policy");
4040   }
4041   // first perform tests of things it's not possible to inline
4042   if (callee->has_exception_handlers() &&
4043       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4044   if (callee->is_synchronized() &&
4045       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4046   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4047   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4048 
4049   // Proper inlining of methods with jsrs requires a little more work.
4050   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4051 
4052   if (strict_fp_requires_explicit_rounding &&
4053       method()->is_strict() != callee->is_strict()) {
4054 #ifdef IA32
4055     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4056     // When SSE2 is present, no special handling is needed.
4057     if (UseSSE < 2) {
4058       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4059     }
4060 #else
4061     Unimplemented();
4062 #endif // IA32
4063   }
4064 
4065   if (is_profiling() && !callee->ensure_method_data()) {
4066     INLINE_BAILOUT("mdo allocation failed");
4067   }
4068 
4069   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4070   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4071 
4072   const int args_base = state()->stack_size() - callee->arg_size();
4073   assert(args_base >= 0, "stack underflow during inlining");
4074 
4075   Value recv = NULL;
4076   if (has_receiver) {
4077     assert(!callee->is_static(), "callee must not be static");
4078     assert(callee->arg_size() > 0, "must have at least a receiver");
4079 
4080     recv = state()->stack_at(args_base);
4081     if (recv->is_null_obj()) {
4082       INLINE_BAILOUT("receiver is always null");
4083     }
4084   }
4085 
4086   // now perform tests that are based on flag settings
4087   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4088   if (callee->force_inline() || inlinee_by_directive) {
4089     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4090     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4091 
4092     const char* msg = "";
4093     if (callee->force_inline())  msg = "force inline by annotation";
4094     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4095     print_inlining(callee, msg);
4096   } else {
4097     // use heuristic controls on inlining
4098     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4099     int callee_recursive_level = recursive_inline_level(callee);
4100     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4101     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4102     // Additional condition to limit stack usage for non-recursive calls.
4103     if ((callee_recursive_level == 0) &&
4104         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4105       INLINE_BAILOUT("callee uses too much stack");
4106     }
4107 
4108     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4109     if (callee->name() == ciSymbol::object_initializer_name() &&
4110         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4111       // Throwable constructor call
4112       IRScope* top = scope();
4113       while (top->caller() != NULL) {
4114         top = top->caller();
4115       }
4116       if (!top->method()->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4117         INLINE_BAILOUT("don't inline Throwable constructors");
4118       }
4119     }

4122       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4123     }

4129   // GraphBuilder level, do not cause bailout just of the inlining but
4130   // in fact of the entire compilation.

4133 
4134   // Insert null check if necessary
4135   if (has_receiver) {
4136     // note: null check must happen even if first instruction of callee does
4137     //       an implicit null check since the callee is in a different scope

4140   }

4143     // Note that we'd collect profile data in this method if we wanted it.

4146

4196   // (only using the bci of this stream)

4200   // note: this will also ensure that all arguments are computed before being passed

4204     const int arg_no = i - args_base;

4210   // Note that we preserve locals state in case we can use it later
4211   // (see use of pop_scope() below)
4212   caller_state->truncate_stack(args_base);
4213   assert(callee_state->stack_size() == 0, "callee stack must be empty");
4214

4216   BlockBegin* sync_handler = NULL;
4217

4220     lock = callee->is_static() ? append(new Constant(new InstanceConstant(callee->holder()->java_mirror())))
4221                                : state()->local_at(0);

4230   }

4234   }

4237   if (callee_start_block != NULL) {
4238     assert(callee_start_block->is_set(BlockBegin::parser_loop_header_flag), "must be loop header");
4239     Goto* goto_callee = new Goto(callee_start_block, false);
4240     // The state for this goto is in the scope of the callee, so use
4241     // the entry bci for the callee instead of the call site bci.
4242     append_with_bci(goto_callee, 0);
4243     _block->set_end(goto_callee);
4244     callee_start_block->merge(callee_state);

4261 
4262   if (log != NULL) log->done("parse");
4263 
4264   // If we bailed out during parsing, return immediately (this is bad news)
4265   if (bailed_out())

4269   // practice, we have only traversed the continuation if we are
4270   // inlining into a subroutine

4272          !continuation()->is_set(BlockBegin::was_visited_flag),

4276   // the caller back in the GraphBuilder. The only thing we want to do
4277   // first is an optimization: during parsing of the callee we
4278   // generated at least one Goto to the continuation block. If we
4279   // generated exactly one, and if the inlined method spanned exactly
4280   // one block (and we didn't have to Goto its entry), then we snip
4281   // off the Goto to the continuation, allowing control to fall
4282   // through back into the caller block and effectively performing
4283   // block merging. This allows load elimination and CSE to take place
4284   // across multiple callee scopes if they are relatively simple, and
4285   // is currently essential to making inlining profitable.
4286   if (num_returns() == 1
4287       && block() == orig_block
4288       && block() == inline_cleanup_block()) {
4289     _last  = inline_cleanup_return_prev();
4290     _state = inline_cleanup_state();
4291   } else if (continuation_preds == cont->number_of_preds()) {
4292     // Inlining caused that the instructions after the invoke in the
4293     // caller are not reachable any more. So skip filling this block
4294     // with instructions!
4295     assert(cont == continuation(), "");
4296     assert(_last && _last->as_BlockEnd(), "");
4297     _skip_block = true;
4298   } else {
4299     // Resume parsing in continuation block unless it was already parsed.
4300     // Note that if we don't change _last here, iteration in
4301     // iterate_bytecodes_for_block will stop when we return.
4302     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4303       // add continuation to work list instead of parsing it immediately
4304       assert(_last && _last->as_BlockEnd(), "");
4305       scope_data()->parent()->add_to_work_list(continuation());
4306       _skip_block = true;
4307     }
4308   }
4309 
4310   // Fill the exception handler for synchronized methods with instructions
4311   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4312     fill_sync_handler(lock, sync_handler);
4313   } else {
4314     pop_scope();
4315   }
4316 
4317   compilation()->notice_inlined_method(callee);
4318 
4319   return true;
4320 }

4323 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4324   ValueStack* state_before = copy_state_before();
4325   vmIntrinsics::ID iid = callee->intrinsic_id();
4326   switch (iid) {
4327   case vmIntrinsics::_invokeBasic:
4328     {
4329       // get MethodHandle receiver
4330       const int args_base = state()->stack_size() - callee->arg_size();
4331       ValueType* type = state()->stack_at(args_base)->type();
4332       if (type->is_constant()) {
4333         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4334         // We don't do CHA here so only inline static and statically bindable methods.
4335         if (target->is_static() || target->can_be_statically_bound()) {
4336           if (ciMethod::is_consistent_info(callee, target)) {
4337             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4338             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4339             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4340               return true;
4341             }
4342           } else {
4343             print_inlining(target, "signatures mismatch", /*success*/ false);
4344           }
4345         } else {
4346           print_inlining(target, "not static or statically bindable", /*success*/ false);
4347         }
4348       } else {
4349         print_inlining(callee, "receiver not constant", /*success*/ false);
4350       }
4351     }
4352     break;
4353 
4354   case vmIntrinsics::_linkToVirtual:
4355   case vmIntrinsics::_linkToStatic:
4356   case vmIntrinsics::_linkToSpecial:
4357   case vmIntrinsics::_linkToInterface:
4358     {
4359       // pop MemberName argument
4360       const int args_base = state()->stack_size() - callee->arg_size();
4361       ValueType* type = apop()->type();
4362       if (type->is_constant()) {
4363         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4364         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4365         // If the target is another method handle invoke, try to recursively get
4366         // a better target.
4367         if (target->is_method_handle_intrinsic()) {
4368           if (try_method_handle_inline(target, ignore_return)) {
4369             return true;
4370           }
4371         } else if (!ciMethod::is_consistent_info(callee, target)) {
4372           print_inlining(target, "signatures mismatch", /*success*/ false);
4373         } else {
4374           ciSignature* signature = target->signature();
4375           const int receiver_skip = target->is_static() ? 0 : 1;
4376           // Cast receiver to its type.
4377           if (!target->is_static()) {
4378             ciKlass* tk = signature->accessing_klass();
4379             Value obj = state()->stack_at(args_base);
4380             if (obj->exact_type() == NULL &&
4381                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4382               TypeCast* c = new TypeCast(tk, obj, state_before);
4383               append(c);
4384               state()->stack_at_put(args_base, c);
4385             }
4386           }
4387           // Cast reference arguments to its type.
4388           for (int i = 0, j = 0; i < signature->count(); i++) {
4389             ciType* t = signature->type_at(i);
4390             if (t->is_klass()) {
4391               ciKlass* tk = t->as_klass();
4392               Value obj = state()->stack_at(args_base + receiver_skip + j);
4393               if (obj->exact_type() == NULL &&
4394                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4395                 TypeCast* c = new TypeCast(t, obj, state_before);
4396                 append(c);
4397                 state()->stack_at_put(args_base + receiver_skip + j, c);
4398               }
4399             }
4400             j += t->size();  // long and double take two slots
4401           }
4402           // We don't do CHA here so only inline static and statically bindable methods.
4403           if (target->is_static() || target->can_be_statically_bound()) {
4404             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4405             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4406               return true;
4407             }
4408           } else {
4409             print_inlining(target, "not static or statically bindable", /*success*/ false);
4410           }
4411         }
4412       } else {
4413         print_inlining(callee, "MemberName not constant", /*success*/ false);
4414       }
4415     }
4416     break;
4417 
4418   default:
4419     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4420     break;
4421   }
4422   set_state(state_before->copy_for_parsing());
4423   return false;
4424 }
