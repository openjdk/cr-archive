  49 //------------------------------Identity---------------------------------------
  50 // If right input is a constant 0, return the left input.
  51 Node* SubNode::Identity(PhaseGVN* phase) {
  52   assert(in(1) != this, "Must already have called Value");
  53   assert(in(2) != this, "Must already have called Value");
  54 
  55   // Remove double negation
  56   const Type *zero = add_id();
  57   if( phase->type( in(1) )->higher_equal( zero ) &&
  58       in(2)->Opcode() == Opcode() &&
  59       phase->type( in(2)->in(1) )->higher_equal( zero ) ) {
  60     return in(2)->in(2);
  61   }
  62 
  63   // Convert "(X+Y) - Y" into X and "(X+Y) - X" into Y
  64   if( in(1)->Opcode() == Op_AddI ) {
  65     if( phase->eqv(in(1)->in(2),in(2)) )
  66       return in(1)->in(1);
  67     if (phase->eqv(in(1)->in(1),in(2)))
  68       return in(1)->in(2);
  69 
  70     // Also catch: "(X + Opaque2(Y)) - Y".  In this case, 'Y' is a loop-varying
  71     // trip counter and X is likely to be loop-invariant (that's how O2 Nodes
  72     // are originally used, although the optimizer sometimes jiggers things).
  73     // This folding through an O2 removes a loop-exit use of a loop-varying
  74     // value and generally lowers register pressure in and around the loop.
  75     if( in(1)->in(2)->Opcode() == Op_Opaque2 &&
  76         phase->eqv(in(1)->in(2)->in(1),in(2)) )
  77       return in(1)->in(1);
  78   }
  79 
  80   return ( phase->type( in(2) )->higher_equal( zero ) ) ? in(1) : this;
  81 }

  83 //------------------------------Value------------------------------------------
  84 // A subtract node differences it's two inputs.
  85 const Type* SubNode::Value_common(PhaseTransform *phase) const {
  86   const Node* in1 = in(1);
  87   const Node* in2 = in(2);
  88   // Either input is TOP ==> the result is TOP
  89   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
  90   if( t1 == Type::TOP ) return Type::TOP;
  91   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
  92   if( t2 == Type::TOP ) return Type::TOP;
  93 
  94   // Not correct for SubFnode and AddFNode (must check for infinity)
  95   // Equal?  Subtract is zero
  96   if (in1->eqv_uncast(in2))  return add_id();
  97 
  98   // Either input is BOTTOM ==> the result is the local BOTTOM
  99   if( t1 == Type::BOTTOM || t2 == Type::BOTTOM )
 100     return bottom_type();
 101 
 102   return NULL;
 103 }

 105 const Type* SubNode::Value(PhaseGVN* phase) const {
 106   const Type* t = Value_common(phase);
 107   if (t != NULL) {
 108     return t;
 109   }
 110   const Type* t1 = phase->type(in(1));
 111   const Type* t2 = phase->type(in(2));
 112   return sub(t1,t2);            // Local flavor of type subtraction
 113 
 114 }

 119 static bool is_cloop_increment(Node* inc) {
 120   precond(inc->Opcode() == Op_AddI || inc->Opcode() == Op_AddL);
 121 
 122   if (!inc->in(1)->is_Phi()) {
 123     return false;
 124   }
 125   const PhiNode* phi = inc->in(1)->as_Phi();
 126 
 127   if (phi->is_copy() || !phi->region()->is_CountedLoop()) {
 128     return false;
 129   }
 130 
 131   return inc == phi->region()->as_CountedLoop()->incr();
 132 }
 133 
 134 // Given the expression '(x + C) - v', or
 135 //                      'v - (x + C)', we examine nodes '+' and 'v':
 136 //
 137 //  1. Do not convert if '+' is a counted-loop increment, because the '-' is
 138 //     loop invariant and converting extends the live-range of 'x' to overlap
 139 //     with the '+', forcing another register to be used in the loop.
 140 //
 141 //  2. Do not convert if 'v' is a counted-loop induction variable, because
 142 //     'x' might be invariant.
 143 //
 144 static bool ok_to_convert(Node* inc, Node* var) {
 145   return !(is_cloop_increment(inc) || var->is_cloop_ind_var());
 146 }

 148 //------------------------------Ideal------------------------------------------
 149 Node *SubINode::Ideal(PhaseGVN *phase, bool can_reshape){
 150   Node *in1 = in(1);
 151   Node *in2 = in(2);
 152   uint op1 = in1->Opcode();
 153   uint op2 = in2->Opcode();
 154 
 155 #ifdef ASSERT
 156   // Check for dead loop
 157   if( phase->eqv( in1, this ) || phase->eqv( in2, this ) ||
 158       ( ( op1 == Op_AddI || op1 == Op_SubI ) &&
 159         ( phase->eqv( in1->in(1), this ) || phase->eqv( in1->in(2), this ) ||
 160           phase->eqv( in1->in(1), in1  ) || phase->eqv( in1->in(2), in1 ) ) ) )
 161     assert(false, "dead loop in SubINode::Ideal");
 162 #endif
 163 
 164   const Type *t2 = phase->type( in2 );
 165   if( t2 == Type::TOP ) return NULL;
 166   // Convert "x-c0" into "x+ -c0".
 167   if( t2->base() == Type::Int ){        // Might be bottom or top...
 168     const TypeInt *i = t2->is_int();
 169     if( i->is_con() )
 170       return new AddINode(in1, phase->intcon(-i->get_con()));
 171   }
 172 
 173   // Convert "(x+c0) - y" into (x-y) + c0"
 174   // Do not collapse (x+c0)-y if "+" is a loop increment or
 175   // if "y" is a loop induction variable.
 176   if( op1 == Op_AddI && ok_to_convert(in1, in2) ) {
 177     const Type *tadd = phase->type( in1->in(2) );
 178     if( tadd->singleton() && tadd != Type::TOP ) {
 179       Node *sub2 = phase->transform( new SubINode( in1->in(1), in2 ));
 180       return new AddINode( sub2, in1->in(2) );
 181     }
 182   }
 183 
 184 
 185   // Convert "x - (y+c0)" into "(x-y) - c0"
 186   // Need the same check as in above optimization but reversed.
 187   if (op2 == Op_AddI && ok_to_convert(in2, in1)) {
 188     Node* in21 = in2->in(1);
 189     Node* in22 = in2->in(2);
 190     const TypeInt* tcon = phase->type(in22)->isa_int();
 191     if (tcon != NULL && tcon->is_con()) {
 192       Node* sub2 = phase->transform( new SubINode(in1, in21) );
 193       Node* neg_c0 = phase->intcon(- tcon->get_con());
 194       return new AddINode(sub2, neg_c0);
 195     }
 196   }
 197 
 198   const Type *t1 = phase->type( in1 );
 199   if( t1 == Type::TOP ) return NULL;
 200 
 201 #ifdef ASSERT
 202   // Check for dead loop
 203   if( ( op2 == Op_AddI || op2 == Op_SubI ) &&
 204       ( phase->eqv( in2->in(1), this ) || phase->eqv( in2->in(2), this ) ||
 205         phase->eqv( in2->in(1), in2  ) || phase->eqv( in2->in(2), in2  ) ) )
 206     assert(false, "dead loop in SubINode::Ideal");
 207 #endif
 208 
 209   // Convert "x - (x+y)" into "-y"
 210   if( op2 == Op_AddI &&
 211       phase->eqv( in1, in2->in(1) ) )
 212     return new SubINode( phase->intcon(0),in2->in(2));
 213   // Convert "(x-y) - x" into "-y"
 214   if( op1 == Op_SubI &&
 215       phase->eqv( in1->in(1), in2 ) )
 216     return new SubINode( phase->intcon(0),in1->in(2));
 217   // Convert "x - (y+x)" into "-y"
 218   if( op2 == Op_AddI &&
 219       phase->eqv( in1, in2->in(2) ) )
 220     return new SubINode( phase->intcon(0),in2->in(1));
 221 
 222   // Convert "0 - (x-y)" into "y-x"
 223   if( t1 == TypeInt::ZERO && op2 == Op_SubI )
 224     return new SubINode( in2->in(2), in2->in(1) );
 225 
 226   // Convert "0 - (x+con)" into "-con-x"
 227   jint con;
 228   if( t1 == TypeInt::ZERO && op2 == Op_AddI &&
 229       (con = in2->in(2)->find_int_con(0)) != 0 )
 230     return new SubINode( phase->intcon(-con), in2->in(1) );
 231 
 232   // Convert "(X+A) - (X+B)" into "A - B"
 233   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(1) == in2->in(1) )
 234     return new SubINode( in1->in(2), in2->in(2) );
 235 
 236   // Convert "(A+X) - (B+X)" into "A - B"
 237   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(2) == in2->in(2) )
 238     return new SubINode( in1->in(1), in2->in(1) );
 239 
 240   // Convert "(A+X) - (X+B)" into "A - B"
 241   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(2) == in2->in(1) )
 242     return new SubINode( in1->in(1), in2->in(2) );
 243 
 244   // Convert "(X+A) - (B+X)" into "A - B"
 245   if( op1 == Op_AddI && op2 == Op_AddI && in1->in(1) == in2->in(2) )
 246     return new SubINode( in1->in(2), in2->in(1) );
 247 
 248   // Convert "A-(B-C)" into (A+C)-B", since add is commutative and generally
 249   // nicer to optimize than subtract.
 250   if( op2 == Op_SubI && in2->outcnt() == 1) {
 251     Node *add1 = phase->transform( new AddINode( in1, in2->in(2) ) );
 252     return new SubINode( add1, in2->in(1) );
 253   }
 254 
 255   // Convert "0-(A>>31)" into "(A>>>31)"
 256   if ( op2 == Op_RShiftI ) {
 257     Node *in21 = in2->in(1);
 258     Node *in22 = in2->in(2);
 259     const TypeInt *zero = phase->type(in1)->isa_int();
 260     const TypeInt *t21 = phase->type(in21)->isa_int();
 261     const TypeInt *t22 = phase->type(in22)->isa_int();
 262     if ( t21 && t22 && zero == TypeInt::ZERO && t22->is_con(31) ) {
 263       return new URShiftINode(in21, in22);
 264     }
 265   }
 266 
 267   return NULL;
 268 }

 270 //------------------------------sub--------------------------------------------
 271 // A subtract node differences it's two inputs.
 272 const Type *SubINode::sub( const Type *t1, const Type *t2 ) const {
 273   const TypeInt *r0 = t1->is_int(); // Handy access
 274   const TypeInt *r1 = t2->is_int();
 275   int32_t lo = java_subtract(r0->_lo, r1->_hi);
 276   int32_t hi = java_subtract(r0->_hi, r1->_lo);
 277 
 278   // We next check for 32-bit overflow.
 279   // If that happens, we just assume all integers are possible.
 280   if( (((r0->_lo ^ r1->_hi) >= 0) ||    // lo ends have same signs OR
 281        ((r0->_lo ^      lo) >= 0)) &&   // lo results have same signs AND
 282       (((r0->_hi ^ r1->_lo) >= 0) ||    // hi ends have same signs OR
 283        ((r0->_hi ^      hi) >= 0)) )    // hi results have same signs
 284     return TypeInt::make(lo,hi,MAX2(r0->_widen,r1->_widen));
 285   else                          // Overflow; assume all integers
 286     return TypeInt::INT;
 287 }

 290 //------------------------------Ideal------------------------------------------
 291 Node *SubLNode::Ideal(PhaseGVN *phase, bool can_reshape) {
 292   Node *in1 = in(1);
 293   Node *in2 = in(2);
 294   uint op1 = in1->Opcode();
 295   uint op2 = in2->Opcode();
 296 
 297 #ifdef ASSERT
 298   // Check for dead loop
 299   if( phase->eqv( in1, this ) || phase->eqv( in2, this ) ||
 300       ( ( op1 == Op_AddL || op1 == Op_SubL ) &&
 301         ( phase->eqv( in1->in(1), this ) || phase->eqv( in1->in(2), this ) ||
 302           phase->eqv( in1->in(1), in1  ) || phase->eqv( in1->in(2), in1  ) ) ) )
 303     assert(false, "dead loop in SubLNode::Ideal");
 304 #endif
 305 
 306   if( phase->type( in2 ) == Type::TOP ) return NULL;
 307   const TypeLong *i = phase->type( in2 )->isa_long();
 308   // Convert "x-c0" into "x+ -c0".
 309   if( i &&                      // Might be bottom or top...
 310       i->is_con() )
 311     return new AddLNode(in1, phase->longcon(-i->get_con()));
 312 
 313   // Convert "(x+c0) - y" into (x-y) + c0"
 314   // Do not collapse (x+c0)-y if "+" is a loop increment or
 315   // if "y" is a loop induction variable.
 316   if( op1 == Op_AddL && ok_to_convert(in1, in2) ) {
 317     Node *in11 = in1->in(1);
 318     const Type *tadd = phase->type( in1->in(2) );
 319     if( tadd->singleton() && tadd != Type::TOP ) {
 320       Node *sub2 = phase->transform( new SubLNode( in11, in2 ));
 321       return new AddLNode( sub2, in1->in(2) );
 322     }
 323   }
 324 
 325   // Convert "x - (y+c0)" into "(x-y) - c0"
 326   // Need the same check as in above optimization but reversed.
 327   if (op2 == Op_AddL && ok_to_convert(in2, in1)) {
 328     Node* in21 = in2->in(1);
 329     Node* in22 = in2->in(2);
 330     const TypeLong* tcon = phase->type(in22)->isa_long();
 331     if (tcon != NULL && tcon->is_con()) {
 332       Node* sub2 = phase->transform( new SubLNode(in1, in21) );
 333       Node* neg_c0 = phase->longcon(- tcon->get_con());
 334       return new AddLNode(sub2, neg_c0);
 335     }
 336   }
 337 
 338   const Type *t1 = phase->type( in1 );
 339   if( t1 == Type::TOP ) return NULL;
 340 
 341 #ifdef ASSERT
 342   // Check for dead loop
 343   if( ( op2 == Op_AddL || op2 == Op_SubL ) &&
 344       ( phase->eqv( in2->in(1), this ) || phase->eqv( in2->in(2), this ) ||
 345         phase->eqv( in2->in(1), in2  ) || phase->eqv( in2->in(2), in2  ) ) )
 346     assert(false, "dead loop in SubLNode::Ideal");
 347 #endif
 348 
 349   // Convert "x - (x+y)" into "-y"
 350   if( op2 == Op_AddL &&
 351       phase->eqv( in1, in2->in(1) ) )
 352     return new SubLNode( phase->makecon(TypeLong::ZERO), in2->in(2));
 353   // Convert "x - (y+x)" into "-y"
 354   if( op2 == Op_AddL &&
 355       phase->eqv( in1, in2->in(2) ) )
 356     return new SubLNode( phase->makecon(TypeLong::ZERO),in2->in(1));
 357 
 358   // Convert "0 - (x-y)" into "y-x"
 359   if( phase->type( in1 ) == TypeLong::ZERO && op2 == Op_SubL )
 360     return new SubLNode( in2->in(2), in2->in(1) );
 361 
 362   // Convert "(X+A) - (X+B)" into "A - B"
 363   if( op1 == Op_AddL && op2 == Op_AddL && in1->in(1) == in2->in(1) )
 364     return new SubLNode( in1->in(2), in2->in(2) );
 365 
 366   // Convert "(A+X) - (B+X)" into "A - B"
 367   if( op1 == Op_AddL && op2 == Op_AddL && in1->in(2) == in2->in(2) )
 368     return new SubLNode( in1->in(1), in2->in(1) );
 369 
 370   // Convert "A-(B-C)" into (A+C)-B"
 371   if( op2 == Op_SubL && in2->outcnt() == 1) {
 372     Node *add1 = phase->transform( new AddLNode( in1, in2->in(2) ) );
 373     return new SubLNode( add1, in2->in(1) );
 374   }
 375 
 376   // Convert "0L-(A>>63)" into "(A>>>63)"
 377   if ( op2 == Op_RShiftL ) {
 378     Node *in21 = in2->in(1);
 379     Node *in22 = in2->in(2);
 380     const TypeLong *zero = phase->type(in1)->isa_long();
 381     const TypeLong *t21 = phase->type(in21)->isa_long();
 382     const TypeInt *t22 = phase->type(in22)->isa_int();
 383     if ( t21 && t22 && zero == TypeLong::ZERO && t22->is_con(63) ) {
 384       return new URShiftLNode(in21, in22);
 385     }
 386   }
 387 
 388   return NULL;
 389 }

 411 //------------------------------Value------------------------------------------
 412 // A subtract node differences its two inputs.
 413 const Type* SubFPNode::Value(PhaseGVN* phase) const {
 414   const Node* in1 = in(1);
 415   const Node* in2 = in(2);
 416   // Either input is TOP ==> the result is TOP
 417   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
 418   if( t1 == Type::TOP ) return Type::TOP;
 419   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
 420   if( t2 == Type::TOP ) return Type::TOP;
 421 
 422   // if both operands are infinity of same sign, the result is NaN; do
 423   // not replace with zero
 424   if( (t1->is_finite() && t2->is_finite()) ) {
 425     if( phase->eqv(in1, in2) ) return add_id();
 426   }
 427 
 428   // Either input is BOTTOM ==> the result is the local BOTTOM
 429   const Type *bot = bottom_type();
 430   if( (t1 == bot) || (t2 == bot) ||
 431       (t1 == Type::BOTTOM) || (t2 == Type::BOTTOM) )
 432     return bot;
 433 
 434   return sub(t1,t2);            // Local flavor of type subtraction
 435 }

 463 //------------------------------sub--------------------------------------------
 464 // A subtract node differences its two inputs.
 465 const Type *SubFNode::sub( const Type *t1, const Type *t2 ) const {
 466   // no folding if one of operands is infinity or NaN, do not do constant folding
 467   if( g_isfinite(t1->getf()) && g_isfinite(t2->getf()) ) {
 468     return TypeF::make( t1->getf() - t2->getf() );
 469   }
 470   else if( g_isnan(t1->getf()) ) {
 471     return t1;
 472   }
 473   else if( g_isnan(t2->getf()) ) {
 474     return t2;
 475   }
 476   else {
 477     return Type::FLOAT;
 478   }
 479 }

 566 // Simplify a CmpI (compare 2 integers) node, based on local information.
 567 // If both inputs are constants, compare them.
 568 const Type *CmpINode::sub( const Type *t1, const Type *t2 ) const {
 569   const TypeInt *r0 = t1->is_int(); // Handy access
 570   const TypeInt *r1 = t2->is_int();
 571 
 572   if( r0->_hi < r1->_lo )       // Range is always low?
 573     return TypeInt::CC_LT;
 574   else if( r0->_lo > r1->_hi )  // Range is always high?
 575     return TypeInt::CC_GT;
 576 
 577   else if( r0->is_con() && r1->is_con() ) { // comparing constants?
 578     assert(r0->get_con() == r1->get_con(), "must be equal");
 579     return TypeInt::CC_EQ;      // Equal results.
 580   } else if( r0->_hi == r1->_lo ) // Range is never high?
 581     return TypeInt::CC_LE;
 582   else if( r0->_lo == r1->_hi ) // Range is never low?
 583     return TypeInt::CC_GE;
 584   return TypeInt::CC;           // else use worst case results
 585 }

 587 // Simplify a CmpU (compare 2 integers) node, based on local information.
 588 // If both inputs are constants, compare them.
 589 const Type *CmpUNode::sub( const Type *t1, const Type *t2 ) const {
 590   assert(!t1->isa_ptr(), "obsolete usage of CmpU");
 591 
 592   // comparing two unsigned ints
 593   const TypeInt *r0 = t1->is_int();   // Handy access
 594   const TypeInt *r1 = t2->is_int();
 595 
 596   // Current installed version
 597   // Compare ranges for non-overlap
 598   juint lo0 = r0->_lo;
 599   juint hi0 = r0->_hi;
 600   juint lo1 = r1->_lo;
 601   juint hi1 = r1->_hi;
 602 
 603   // If either one has both negative and positive values,
 604   // it therefore contains both 0 and -1, and since [0..-1] is the
 605   // full unsigned range, the type must act as an unsigned bottom.
 606   bool bot0 = ((jint)(lo0 ^ hi0) < 0);
 607   bool bot1 = ((jint)(lo1 ^ hi1) < 0);
 608 
 609   if (bot0 || bot1) {
 610     // All unsigned values are LE -1 and GE 0.
 611     if (lo0 == 0 && hi0 == 0) {
 612       return TypeInt::CC_LE;            //   0 <= bot
 613     } else if ((jint)lo0 == -1 && (jint)hi0 == -1) {
 614       return TypeInt::CC_GE;            // -1 >= bot
 615     } else if (lo1 == 0 && hi1 == 0) {
 616       return TypeInt::CC_GE;            // bot >= 0
 617     } else if ((jint)lo1 == -1 && (jint)hi1 == -1) {
 618       return TypeInt::CC_LE;            // bot <= -1
 619     }
 620   } else {
 621     // We can use ranges of the form [lo..hi] if signs are the same.
 622     assert(lo0 <= hi0 && lo1 <= hi1, "unsigned ranges are valid");
 623     // results are reversed, '-' > '+' for unsigned compare
 624     if (hi0 < lo1) {
 625       return TypeInt::CC_LT;            // smaller
 626     } else if (lo0 > hi1) {
 627       return TypeInt::CC_GT;            // greater
 628     } else if (hi0 == lo1 && lo0 == hi1) {
 629       return TypeInt::CC_EQ;            // Equal results
 630     } else if (lo0 >= hi1) {
 631       return TypeInt::CC_GE;
 632     } else if (hi0 <= lo1) {
 633       // Check for special case in Hashtable::get.  (See below.)
 634       if ((jint)lo0 >= 0 && (jint)lo1 >= 0 && is_index_range_check())
 635         return TypeInt::CC_LT;
 636       return TypeInt::CC_LE;
 637     }
 638   }
 639   // Check for special case in Hashtable::get - the hash index is
 640   // mod'ed to the table size so the following range check is useless.
 641   // Check for: (X Mod Y) CmpU Y, where the mod result and Y both have
 642   // to be positive.
 643   // (This is a gross hack, since the sub method never
 644   // looks at the structure of the node in any other case.)
 645   if ((jint)lo0 >= 0 && (jint)lo1 >= 0 && is_index_range_check())
 646     return TypeInt::CC_LT;
 647   return TypeInt::CC;                   // else use worst case results
 648 }

 650 const Type* CmpUNode::Value(PhaseGVN* phase) const {
 651   const Type* t = SubNode::Value_common(phase);
 652   if (t != NULL) {
 653     return t;
 654   }
 655   const Node* in1 = in(1);
 656   const Node* in2 = in(2);
 657   const Type* t1 = phase->type(in1);
 658   const Type* t2 = phase->type(in2);
 659   assert(t1->isa_int(), "CmpU has only Int type inputs");
 660   if (t2 == TypeInt::INT) { // Compare to bottom?
 661     return bottom_type();
 662   }
 663   uint in1_op = in1->Opcode();
 664   if (in1_op == Op_AddI || in1_op == Op_SubI) {
 665     // The problem rise when result of AddI(SubI) may overflow
 666     // signed integer value. Let say the input type is
 667     // [256, maxint] then +128 will create 2 ranges due to
 668     // overflow: [minint, minint+127] and [384, maxint].
 669     // But C2 type system keep only 1 type range and as result
 670     // it use general [minint, maxint] for this case which we
 671     // can't optimize.
 672     //
 673     // Make 2 separate type ranges based on types of AddI(SubI) inputs
 674     // and compare results of their compare. If results are the same
 675     // CmpU node can be optimized.
 676     const Node* in11 = in1->in(1);
 677     const Node* in12 = in1->in(2);
 678     const Type* t11 = (in11 == in1) ? Type::TOP : phase->type(in11);
 679     const Type* t12 = (in12 == in1) ? Type::TOP : phase->type(in12);
 680     // Skip cases when input types are top or bottom.
 681     if ((t11 != Type::TOP) && (t11 != TypeInt::INT) &&
 682         (t12 != Type::TOP) && (t12 != TypeInt::INT)) {
 683       const TypeInt *r0 = t11->is_int();
 684       const TypeInt *r1 = t12->is_int();
 685       jlong lo_r0 = r0->_lo;
 686       jlong hi_r0 = r0->_hi;
 687       jlong lo_r1 = r1->_lo;
 688       jlong hi_r1 = r1->_hi;
 689       if (in1_op == Op_SubI) {
 690         jlong tmp = hi_r1;
 691         hi_r1 = -lo_r1;
 692         lo_r1 = -tmp;
 693         // Note, for substructing [minint,x] type range
 694         // long arithmetic provides correct overflow answer.
 695         // The confusion come from the fact that in 32-bit
 696         // -minint == minint but in 64-bit -minint == maxint+1.
 697       }
 698       jlong lo_long = lo_r0 + lo_r1;
 699       jlong hi_long = hi_r0 + hi_r1;
 700       int lo_tr1 = min_jint;
 701       int hi_tr1 = (int)hi_long;
 702       int lo_tr2 = (int)lo_long;
 703       int hi_tr2 = max_jint;
 704       bool underflow = lo_long != (jlong)lo_tr2;
 705       bool overflow  = hi_long != (jlong)hi_tr1;
 706       // Use sub(t1, t2) when there is no overflow (one type range)
 707       // or when both overflow and underflow (too complex).
 708       if ((underflow != overflow) && (hi_tr1 < lo_tr2)) {
 709         // Overflow only on one boundary, compare 2 separate type ranges.
 710         int w = MAX2(r0->_widen, r1->_widen); // _widen does not matter here
 711         const TypeInt* tr1 = TypeInt::make(lo_tr1, hi_tr1, w);
 712         const TypeInt* tr2 = TypeInt::make(lo_tr2, hi_tr2, w);
 713         const Type* cmp1 = sub(tr1, t2);
 714         const Type* cmp2 = sub(tr2, t2);
 715         if (cmp1 == cmp2) {
 716           return cmp1; // Hit!
 717         }
 718       }
 719     }
 720   }
 721 
 722   return sub(t1, t2);            // Local flavor of type subtraction
 723 }

 746   }
 747   return NULL;                  // No change

 863 //------------------------------sub--------------------------------------------
 864 // Simplify an CmpP (compare 2 pointers) node, based on local information.
 865 // If both inputs are constants, compare them.
 866 const Type *CmpPNode::sub( const Type *t1, const Type *t2 ) const {
 867   const TypePtr *r0 = t1->is_ptr(); // Handy access
 868   const TypePtr *r1 = t2->is_ptr();
 869 
 870   // Undefined inputs makes for an undefined result
 871   if( TypePtr::above_centerline(r0->_ptr) ||
 872       TypePtr::above_centerline(r1->_ptr) )
 873     return Type::TOP;
 874 
 875   if (r0 == r1 && r0->singleton()) {
 876     // Equal pointer constants (klasses, nulls, etc.)
 877     return TypeInt::CC_EQ;
 878   }
 879 
 880   // See if it is 2 unrelated classes.
 881   const TypeOopPtr* oop_p0 = r0->isa_oopptr();
 882   const TypeOopPtr* oop_p1 = r1->isa_oopptr();
 883   bool both_oop_ptr = oop_p0 && oop_p1;
 884 
 885   if (both_oop_ptr) {
 886     Node* in1 = in(1)->uncast();
 887     Node* in2 = in(2)->uncast();
 888     AllocateNode* alloc1 = AllocateNode::Ideal_allocation(in1, NULL);
 889     AllocateNode* alloc2 = AllocateNode::Ideal_allocation(in2, NULL);
 890     if (MemNode::detect_ptr_independence(in1, alloc1, in2, alloc2, NULL)) {
 891       return TypeInt::CC_GT;  // different pointers
 892     }
 893   }
 894 
 895   const TypeKlassPtr* klass_p0 = r0->isa_klassptr();
 896   const TypeKlassPtr* klass_p1 = r1->isa_klassptr();
 897 
 898   if (both_oop_ptr || (klass_p0 && klass_p1)) { // both or neither are klass pointers
 899     ciKlass* klass0 = NULL;
 900     bool    xklass0 = false;
 901     ciKlass* klass1 = NULL;
 902     bool    xklass1 = false;
 903 
 904     if (oop_p0) {
 905       klass0 = oop_p0->klass();
 906       xklass0 = oop_p0->klass_is_exact();
 907     } else {
 908       assert(klass_p0, "must be non-null if oop_p0 is null");
 909       klass0 = klass_p0->klass();
 910       xklass0 = klass_p0->klass_is_exact();
 911     }
 912 
 913     if (oop_p1) {
 914       klass1 = oop_p1->klass();
 915       xklass1 = oop_p1->klass_is_exact();
 916     } else {
 917       assert(klass_p1, "must be non-null if oop_p1 is null");
 918       klass1 = klass_p1->klass();
 919       xklass1 = klass_p1->klass_is_exact();
 920     }
 921 
 922     if (klass0 && klass1 &&
 923         klass0->is_loaded() && !klass0->is_interface() && // do not trust interfaces
 924         klass1->is_loaded() && !klass1->is_interface() &&
 925         (!klass0->is_obj_array_klass() ||
 926          !klass0->as_obj_array_klass()->base_element_klass()->is_interface()) &&
 927         (!klass1->is_obj_array_klass() ||
 928          !klass1->as_obj_array_klass()->base_element_klass()->is_interface())) {
 929       bool unrelated_classes = false;
 930       // See if neither subclasses the other, or if the class on top
 931       // is precise.  In either of these cases, the compare is known
 932       // to fail if at least one of the pointers is provably not null.
 933       if (klass0->equals(klass1)) {  // if types are unequal but klasses are equal
 934         // Do nothing; we know nothing for imprecise types
 935       } else if (klass0->is_subtype_of(klass1)) {
 936         // If klass1's type is PRECISE, then classes are unrelated.
 937         unrelated_classes = xklass1;
 938       } else if (klass1->is_subtype_of(klass0)) {
 939         // If klass0's type is PRECISE, then classes are unrelated.
 940         unrelated_classes = xklass0;
 941       } else {                  // Neither subtypes the other
 942         unrelated_classes = true;
 943       }
 944       if (!unrelated_classes) {
 945         // Handle inline type arrays
 946         if ((r0->flatten_array() && (!r1->can_be_inline_type() || (klass1->is_inlinetype() && !klass1->flatten_array()))) ||
 947             (r1->flatten_array() && (!r0->can_be_inline_type() || (klass0->is_inlinetype() && !klass0->flatten_array())))) {
 948           // One type is flattened in arrays but the other type is not. Must be unrelated.
 949           unrelated_classes = true;
 950         }
 951       }
 952       if (unrelated_classes) {
 953         // The oops classes are known to be unrelated. If the joined PTRs of
 954         // two oops is not Null and not Bottom, then we are sure that one
 955         // of the two oops is non-null, and the comparison will always fail.
 956         TypePtr::PTR jp = r0->join_ptr(r1->_ptr);
 957         if (jp != TypePtr::Null && jp != TypePtr::BotPTR) {
 958           return TypeInt::CC_GT;
 959         }
 960       }
 961     }
 962   }
 963 
 964   // Known constants can be compared exactly
 965   // Null can be distinguished from any NotNull pointers
 966   // Unknown inputs makes an unknown result
 967   if( r0->singleton() ) {
 968     intptr_t bits0 = r0->get_con();
 969     if( r1->singleton() )
 970       return bits0 == r1->get_con() ? TypeInt::CC_EQ : TypeInt::CC_GT;
 971     return ( r1->_ptr == TypePtr::NotNull && bits0==0 ) ? TypeInt::CC_GT : TypeInt::CC;
 972   } else if( r1->singleton() ) {
 973     intptr_t bits1 = r1->get_con();
 974     return ( r0->_ptr == TypePtr::NotNull && bits1==0 ) ? TypeInt::CC_GT : TypeInt::CC;
 975   } else
 976     return TypeInt::CC;
 977 }

 979 static inline Node* isa_java_mirror_load(PhaseGVN* phase, Node* n) {
 980   // Return the klass node for (indirect load from OopHandle)
 981   //   LoadBarrier?(LoadP(LoadP(AddP(foo:Klass, #java_mirror))))
 982   //   or NULL if not matching.
 983   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 984     n = bs->step_over_gc_barrier(n);
 985 
 986   if (n->Opcode() != Op_LoadP) return NULL;
 987 
 988   const TypeInstPtr* tp = phase->type(n)->isa_instptr();
 989   if (!tp || tp->klass() != phase->C->env()->Class_klass()) return NULL;
 990 
 991   Node* adr = n->in(MemNode::Address);
 992   // First load from OopHandle: ((OopHandle)mirror)->resolve(); may need barrier.
 993   if (adr->Opcode() != Op_LoadP || !phase->type(adr)->isa_rawptr()) return NULL;
 994   adr = adr->in(MemNode::Address);
 995 
 996   intptr_t off = 0;
 997   Node* k = AddPNode::Ideal_base_and_offset(adr, phase, off);
 998   if (k == NULL)  return NULL;
 999   const TypeKlassPtr* tkp = phase->type(k)->isa_klassptr();
1000   if (!tkp || off != in_bytes(Klass::java_mirror_offset())) return NULL;
1001 
1002   // We've found the klass node of a Java mirror load.
1003   return k;
1004 }

1006 static inline Node* isa_const_java_mirror(PhaseGVN* phase, Node* n) {
1007   // for ConP(Foo.class) return ConP(Foo.klass)
1008   // otherwise return NULL
1009   if (!n->is_Con()) return NULL;
1010 
1011   const TypeInstPtr* tp = phase->type(n)->isa_instptr();
1012   if (!tp) return NULL;
1013 
1014   ciType* mirror_type = tp->java_mirror_type();
1015   // TypeInstPtr::java_mirror_type() returns non-NULL for compile-
1016   // time Class constants only.
1017   if (!mirror_type) return NULL;
1018 
1019   // x.getClass() == int.class can never be true (for all primitive types)
1020   // Return a ConP(NULL) node for this case.
1021   if (mirror_type->is_classless()) {
1022     return phase->makecon(TypePtr::NULL_PTR);

1030 //------------------------------Ideal------------------------------------------
1031 // Normalize comparisons between Java mirror loads to compare the klass instead.
1032 //
1033 // Also check for the case of comparing an unknown klass loaded from the primary
1034 // super-type array vs a known klass with no subtypes.  This amounts to
1035 // checking to see an unknown klass subtypes a known klass with no subtypes;
1036 // this only happens on an exact match.  We can shorten this test by 1 load.
1037 Node* CmpPNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1038   // Normalize comparisons between Java mirrors into comparisons of the low-
1039   // level klass, where a dependent load could be shortened.
1040   //
1041   // The new pattern has a nice effect of matching the same pattern used in the
1042   // fast path of instanceof/checkcast/Class.isInstance(), which allows
1043   // redundant exact type check be optimized away by GVN.
1044   // For example, in
1045   //   if (x.getClass() == Foo.class) {
1046   //     Foo foo = (Foo) x;
1047   //     // ... use a ...
1048   //   }
1049   // a CmpPNode could be shared between if_acmpne and checkcast
1050   {
1051     Node* k1 = isa_java_mirror_load(phase, in(1));
1052     Node* k2 = isa_java_mirror_load(phase, in(2));
1053     Node* conk2 = isa_const_java_mirror(phase, in(2));
1054 
1055     if (k1 && (k2 || conk2)) {
1056       Node* lhs = k1;
1057       Node* rhs = (k2 != NULL) ? k2 : conk2;
1058       PhaseIterGVN* igvn = phase->is_IterGVN();
1059       if (igvn != NULL) {
1060         set_req_X(1, lhs, igvn);
1061         set_req_X(2, rhs, igvn);
1062       } else {
1063         set_req(1, lhs);
1064         set_req(2, rhs);
1065       }
1066       return this;
1067     }
1068   }
1069 
1070   // Constant pointer on right?
1071   const TypeKlassPtr* t2 = phase->type(in(2))->isa_klassptr();
1072   if (t2 == NULL || !t2->klass_is_exact())
1073     return NULL;
1074   // Get the constant klass we are comparing to.
1075   ciKlass* superklass = t2->klass();
1076 
1077   // Now check for LoadKlass on left.
1078   Node* ldk1 = in(1);
1079   if (ldk1->is_DecodeNKlass()) {
1080     ldk1 = ldk1->in(1);
1081     if (ldk1->Opcode() != Op_LoadNKlass )
1082       return NULL;
1083   } else if (ldk1->Opcode() != Op_LoadKlass )
1084     return NULL;
1085   // Take apart the address of the LoadKlass:
1086   Node* adr1 = ldk1->in(MemNode::Address);
1087   intptr_t con2 = 0;
1088   Node* ldk2 = AddPNode::Ideal_base_and_offset(adr1, phase, con2);
1089   if (ldk2 == NULL)
1090     return NULL;
1091   if (con2 == oopDesc::klass_offset_in_bytes()) {
1092     // We are inspecting an object's concrete class.
1093     // Short-circuit the check if the query is abstract.
1094     if (superklass->is_interface() ||
1095         superklass->is_abstract()) {
1096       // Make it come out always false:
1097       this->set_req(2, phase->makecon(TypePtr::NULL_PTR));
1098       return this;
1099     }
1100   }
1101 
1102   // Check for a LoadKlass from primary supertype array.
1103   // Any nested loadklass from loadklass+con must be from the p.s. array.
1104   if (ldk2->is_DecodeNKlass()) {
1105     // Keep ldk2 as DecodeN since it could be used in CmpP below.
1106     if (ldk2->in(1)->Opcode() != Op_LoadNKlass )
1107       return NULL;
1108   } else if (ldk2->Opcode() != Op_LoadKlass)
1109     return NULL;
1110 
1111   // Verify that we understand the situation
1112   if (con2 != (intptr_t) superklass->super_check_offset())
1113     return NULL;                // Might be element-klass loading from array klass
1114 
1115   // If 'superklass' has no subklasses and is not an interface, then we are
1116   // assured that the only input which will pass the type check is
1117   // 'superklass' itself.
1118   //
1119   // We could be more liberal here, and allow the optimization on interfaces
1120   // which have a single implementor.  This would require us to increase the
1121   // expressiveness of the add_dependency() mechanism.
1122   // %%% Do this after we fix TypeOopPtr:  Deps are expressive enough now.
1123 
1124   // Object arrays must have their base element have no subtypes
1125   while (superklass->is_obj_array_klass()) {
1126     ciType* elem = superklass->as_obj_array_klass()->element_type();
1127     superklass = elem->as_klass();
1128   }
1129   if (superklass->is_instance_klass()) {
1130     ciInstanceKlass* ik = superklass->as_instance_klass();
1131     if (ik->has_subklass() || ik->is_interface())  return NULL;
1132     // Add a dependency if there is a chance that a subclass will be added later.
1133     if (!ik->is_final()) {
1134       phase->C->dependencies()->assert_leaf_type(ik);
1135     }
1136   }
1137 
1138   // Bypass the dependent load, and compare directly
1139   this->set_req(1,ldk2);
1140 
1141   return this;
1142 }

1160 // Simplify an CmpF (compare 2 floats ) node, based on local information.
1161 // If both inputs are constants, compare them.
1162 const Type* CmpFNode::Value(PhaseGVN* phase) const {
1163   const Node* in1 = in(1);
1164   const Node* in2 = in(2);
1165   // Either input is TOP ==> the result is TOP
1166   const Type* t1 = (in1 == this) ? Type::TOP : phase->type(in1);
1167   if( t1 == Type::TOP ) return Type::TOP;
1168   const Type* t2 = (in2 == this) ? Type::TOP : phase->type(in2);
1169   if( t2 == Type::TOP ) return Type::TOP;
1170 
1171   // Not constants?  Don't know squat - even if they are the same
1172   // value!  If they are NaN's they compare to LT instead of EQ.
1173   const TypeF *tf1 = t1->isa_float_constant();
1174   const TypeF *tf2 = t2->isa_float_constant();
1175   if( !tf1 || !tf2 ) return TypeInt::CC;
1176 
1177   // This implements the Java bytecode fcmpl, so unordered returns -1.
1178   if( tf1->is_nan() || tf2->is_nan() )
1179     return TypeInt::CC_LT;
1180 
1181   if( tf1->_f < tf2->_f ) return TypeInt::CC_LT;
1182   if( tf1->_f > tf2->_f ) return TypeInt::CC_GT;
1183   assert( tf1->_f == tf2->_f, "do not understand FP behavior" );
1184   return TypeInt::CC_EQ;
1185 }

1217 //------------------------------Ideal------------------------------------------
1218 Node *CmpDNode::Ideal(PhaseGVN *phase, bool can_reshape){
1219   // Check if we can change this to a CmpF and remove a ConvD2F operation.
1220   // Change  (CMPD (F2D (float)) (ConD value))
1221   // To      (CMPF      (float)  (ConF value))
1222   // Valid when 'value' does not lose precision as a float.
1223   // Benefits: eliminates conversion, does not require 24-bit mode
1224 
1225   // NaNs prevent commuting operands.  This transform works regardless of the
1226   // order of ConD and ConvF2D inputs by preserving the original order.
1227   int idx_f2d = 1;              // ConvF2D on left side?
1228   if( in(idx_f2d)->Opcode() != Op_ConvF2D )
1229     idx_f2d = 2;                // No, swap to check for reversed args
1230   int idx_con = 3-idx_f2d;      // Check for the constant on other input
1231 
1232   if( ConvertCmpD2CmpF &&
1233       in(idx_f2d)->Opcode() == Op_ConvF2D &&
1234       in(idx_con)->Opcode() == Op_ConD ) {
1235     const TypeD *t2 = in(idx_con)->bottom_type()->is_double_constant();
1236     double t2_value_as_double = t2->_d;
1237     float  t2_value_as_float  = (float)t2_value_as_double;
1238     if( t2_value_as_double == (double)t2_value_as_float ) {
1239       // Test value can be represented as a float
1240       // Eliminate the conversion to double and create new comparison
1241       Node *new_in1 = in(idx_f2d)->in(1);
1242       Node *new_in2 = phase->makecon( TypeF::make(t2_value_as_float) );
1243       if( idx_f2d != 1 ) {      // Must flip args to match original order
1244         Node *tmp = new_in1;
1245         new_in1 = new_in2;
1246         new_in2 = tmp;
1247       }
1248       CmpFNode *new_cmp = (Opcode() == Op_CmpD3)
1249         ? new CmpF3Node( new_in1, new_in2 )
1250         : new CmpFNode ( new_in1, new_in2 ) ;
1251       return new_cmp;           // Changed to CmpFNode
1252     }
1253     // Testing value required the precision of a double
1254   }
1255   return NULL;                  // No change
1256 }

1260 //------------------------------cc2logical-------------------------------------
1261 // Convert a condition code type to a logical type
1262 const Type *BoolTest::cc2logical( const Type *CC ) const {
1263   if( CC == Type::TOP ) return Type::TOP;
1264   if( CC->base() != Type::Int ) return TypeInt::BOOL; // Bottom or worse
1265   const TypeInt *ti = CC->is_int();
1266   if( ti->is_con() ) {          // Only 1 kind of condition codes set?
1267     // Match low order 2 bits
1268     int tmp = ((ti->get_con()&3) == (_test&3)) ? 1 : 0;
1269     if( _test & 4 ) tmp = 1-tmp;     // Optionally complement result
1270     return TypeInt::make(tmp);       // Boolean result
1271   }
1272 
1273   if( CC == TypeInt::CC_GE ) {
1274     if( _test == ge ) return TypeInt::ONE;
1275     if( _test == lt ) return TypeInt::ZERO;
1276   }
1277   if( CC == TypeInt::CC_LE ) {
1278     if( _test == le ) return TypeInt::ONE;
1279     if( _test == gt ) return TypeInt::ZERO;
1280   }
1281 
1282   return TypeInt::BOOL;
1283 }

1357 // Change "bool eq/ne (cmp (add/sub A B) C)" into false/true if add/sub
1358 // overflows and we can prove that C is not in the two resulting ranges.
1359 // This optimization is similar to the one performed by CmpUNode::Value().
1360 Node* BoolNode::fold_cmpI(PhaseGVN* phase, SubNode* cmp, Node* cmp1, int cmp_op,
1361                           int cmp1_op, const TypeInt* cmp2_type) {
1362   // Only optimize eq/ne integer comparison of add/sub
1363   if((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1364      (cmp_op == Op_CmpI) && (cmp1_op == Op_AddI || cmp1_op == Op_SubI)) {
1365     // Skip cases were inputs of add/sub are not integers or of bottom type
1366     const TypeInt* r0 = phase->type(cmp1->in(1))->isa_int();
1367     const TypeInt* r1 = phase->type(cmp1->in(2))->isa_int();
1368     if ((r0 != NULL) && (r0 != TypeInt::INT) &&
1369         (r1 != NULL) && (r1 != TypeInt::INT) &&
1370         (cmp2_type != TypeInt::INT)) {
1371       // Compute exact (long) type range of add/sub result
1372       jlong lo_long = r0->_lo;
1373       jlong hi_long = r0->_hi;
1374       if (cmp1_op == Op_AddI) {
1375         lo_long += r1->_lo;
1376         hi_long += r1->_hi;
1377       } else {
1378         lo_long -= r1->_hi;
1379         hi_long -= r1->_lo;
1380       }
1381       // Check for over-/underflow by casting to integer
1382       int lo_int = (int)lo_long;
1383       int hi_int = (int)hi_long;
1384       bool underflow = lo_long != (jlong)lo_int;
1385       bool overflow  = hi_long != (jlong)hi_int;
1386       if ((underflow != overflow) && (hi_int < lo_int)) {
1387         // Overflow on one boundary, compute resulting type ranges:
1388         // tr1 [MIN_INT, hi_int] and tr2 [lo_int, MAX_INT]
1389         int w = MAX2(r0->_widen, r1->_widen); // _widen does not matter here
1390         const TypeInt* tr1 = TypeInt::make(min_jint, hi_int, w);
1391         const TypeInt* tr2 = TypeInt::make(lo_int, max_jint, w);
1392         // Compare second input of cmp to both type ranges
1393         const Type* sub_tr1 = cmp->sub(tr1, cmp2_type);
1394         const Type* sub_tr2 = cmp->sub(tr2, cmp2_type);
1395         if (sub_tr1 == TypeInt::CC_LT && sub_tr2 == TypeInt::CC_GT) {
1396           // The result of the add/sub will never equal cmp2. Replace BoolNode
1397           // by false (0) if it tests for equality and by true (1) otherwise.
1398           return ConINode::make((_test._test == BoolTest::eq) ? 0 : 1);
1399         }
1400       }
1401     }
1402   }
1403   return NULL;
1404 }

1415 //------------------------------Ideal------------------------------------------
1416 Node *BoolNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1417   // Change "bool tst (cmp con x)" into "bool ~tst (cmp x con)".
1418   // This moves the constant to the right.  Helps value-numbering.
1419   Node *cmp = in(1);
1420   if( !cmp->is_Sub() ) return NULL;
1421   int cop = cmp->Opcode();
1422   if( cop == Op_FastLock || cop == Op_FastUnlock || cmp->is_SubTypeCheck()) return NULL;
1423   Node *cmp1 = cmp->in(1);
1424   Node *cmp2 = cmp->in(2);
1425   if( !cmp1 ) return NULL;
1426 
1427   if (_test._test == BoolTest::overflow || _test._test == BoolTest::no_overflow) {
1428     return NULL;
1429   }
1430 
1431   // Constant on left?
1432   Node *con = cmp1;
1433   uint op2 = cmp2->Opcode();
1434   // Move constants to the right of compare's to canonicalize.
1435   // Do not muck with Opaque1 nodes, as this indicates a loop
1436   // guard that cannot change shape.
1437   if( con->is_Con() && !cmp2->is_Con() && op2 != Op_Opaque1 &&
1438       // Because of NaN's, CmpD and CmpF are not commutative
1439       cop != Op_CmpD && cop != Op_CmpF &&
1440       // Protect against swapping inputs to a compare when it is used by a
1441       // counted loop exit, which requires maintaining the loop-limit as in(2)
1442       !is_counted_loop_exit_test() ) {
1443     // Ok, commute the constant to the right of the cmp node.
1444     // Clone the Node, getting a new Node of the same class
1445     cmp = cmp->clone();
1446     // Swap inputs to the clone
1447     cmp->swap_edges(1, 2);
1448     cmp = phase->transform( cmp );
1449     return new BoolNode( cmp, _test.commute() );
1450   }
1451 
1452   // Change "bool eq/ne (cmp (and X 16) 16)" into "bool ne/eq (cmp (and X 16) 0)".
1453   if (cop == Op_CmpI &&
1454       (_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1455       cmp1->Opcode() == Op_AndI && cmp2->Opcode() == Op_ConI &&
1456       cmp1->in(2)->Opcode() == Op_ConI) {
1457     const TypeInt *t12 = phase->type(cmp2)->isa_int();
1458     const TypeInt *t112 = phase->type(cmp1->in(2))->isa_int();
1459     if (t12 && t12->is_con() && t112 && t112->is_con() &&
1460         t12->get_con() == t112->get_con() && is_power_of_2(t12->get_con())) {
1461       Node *ncmp = phase->transform(new CmpINode(cmp1, phase->intcon(0)));
1462       return new BoolNode(ncmp, _test.negate());
1463     }
1464   }
1465 
1466   // Same for long type: change "bool eq/ne (cmp (and X 16) 16)" into "bool ne/eq (cmp (and X 16) 0)".
1467   if (cop == Op_CmpL &&
1468       (_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1469       cmp1->Opcode() == Op_AndL && cmp2->Opcode() == Op_ConL &&
1470       cmp1->in(2)->Opcode() == Op_ConL) {
1471     const TypeLong *t12 = phase->type(cmp2)->isa_long();
1472     const TypeLong *t112 = phase->type(cmp1->in(2))->isa_long();
1473     if (t12 && t12->is_con() && t112 && t112->is_con() &&
1474         t12->get_con() == t112->get_con() && is_power_of_2(t12->get_con())) {
1475       Node *ncmp = phase->transform(new CmpLNode(cmp1, phase->longcon(0)));
1476       return new BoolNode(ncmp, _test.negate());
1477     }
1478   }
1479 
1480   // Change "bool eq/ne (cmp (xor X 1) 0)" into "bool ne/eq (cmp X 0)".
1481   // The XOR-1 is an idiom used to flip the sense of a bool.  We flip the
1482   // test instead.
1483   int cmp1_op = cmp1->Opcode();
1484   const TypeInt* cmp2_type = phase->type(cmp2)->isa_int();
1485   if (cmp2_type == NULL)  return NULL;
1486   Node* j_xor = cmp1;
1487   if( cmp2_type == TypeInt::ZERO &&
1488       cmp1_op == Op_XorI &&
1489       j_xor->in(1) != j_xor &&          // An xor of itself is dead
1490       phase->type( j_xor->in(1) ) == TypeInt::BOOL &&
1491       phase->type( j_xor->in(2) ) == TypeInt::ONE &&
1492       (_test._test == BoolTest::eq ||
1493        _test._test == BoolTest::ne) ) {
1494     Node *ncmp = phase->transform(new CmpINode(j_xor->in(1),cmp2));
1495     return new BoolNode( ncmp, _test.negate() );
1496   }
1497 
1498   // Change ((x & m) u<= m) or ((m & x) u<= m) to always true
1499   // Same with ((x & m) u< m+1) and ((m & x) u< m+1)
1500   if (cop == Op_CmpU &&
1501       cmp1_op == Op_AndI) {
1502     Node* bound = NULL;
1503     if (_test._test == BoolTest::le) {
1504       bound = cmp2;
1505     } else if (_test._test == BoolTest::lt &&
1506                cmp2->Opcode() == Op_AddI &&
1507                cmp2->in(2)->find_int_con(0) == 1) {
1508       bound = cmp2->in(1);
1509     }
1510     if (cmp1->in(2) == bound || cmp1->in(1) == bound) {
1511       return ConINode::make(1);
1512     }
1513   }
1514 
1515   // Change ((x & (m - 1)) u< m) into (m > 0)
1516   // This is the off-by-one variant of the above
1517   if (cop == Op_CmpU &&
1518       _test._test == BoolTest::lt &&
1519       cmp1_op == Op_AndI) {
1520     Node* l = cmp1->in(1);
1521     Node* r = cmp1->in(2);
1522     for (int repeat = 0; repeat < 2; repeat++) {
1523       bool match = r->Opcode() == Op_AddI && r->in(2)->find_int_con(0) == -1 &&
1524                    r->in(1) == cmp2;
1525       if (match) {
1526         // arraylength known to be non-negative, so a (arraylength != 0) is sufficient,
1527         // but to be compatible with the array range check pattern, use (arraylength u> 0)
1528         Node* ncmp = cmp2->Opcode() == Op_LoadRange
1529                      ? phase->transform(new CmpUNode(cmp2, phase->intcon(0)))
1530                      : phase->transform(new CmpINode(cmp2, phase->intcon(0)));
1531         return new BoolNode(ncmp, BoolTest::gt);
1532       } else {
1533         // commute and try again
1534         l = cmp1->in(2);
1535         r = cmp1->in(1);
1536       }
1537     }
1538   }
1539 
1540   // Change x u< 1 or x u<= 0 to x == 0
1541   if (cop == Op_CmpU &&
1542       cmp1_op != Op_LoadRange &&
1543       ((_test._test == BoolTest::lt &&
1544         cmp2->find_int_con(-1) == 1) ||
1545        (_test._test == BoolTest::le &&
1546         cmp2->find_int_con(-1) == 0))) {
1547     Node* ncmp = phase->transform(new CmpINode(cmp1, phase->intcon(0)));
1548     return new BoolNode(ncmp, BoolTest::eq);
1549   }
1550 
1551   // Change (arraylength <= 0) or (arraylength == 0)
1552   //   into (arraylength u<= 0)
1553   // Also change (arraylength != 0) into (arraylength u> 0)
1554   // The latter version matches the code pattern generated for
1555   // array range checks, which will more likely be optimized later.
1556   if (cop == Op_CmpI &&
1557       cmp1_op == Op_LoadRange &&
1558       cmp2->find_int_con(-1) == 0) {
1559     if (_test._test == BoolTest::le || _test._test == BoolTest::eq) {
1560       Node* ncmp = phase->transform(new CmpUNode(cmp1, cmp2));
1561       return new BoolNode(ncmp, BoolTest::le);
1562     } else if (_test._test == BoolTest::ne) {
1563       Node* ncmp = phase->transform(new CmpUNode(cmp1, cmp2));
1564       return new BoolNode(ncmp, BoolTest::gt);
1565     }
1566   }
1567 
1568   // Change "bool eq/ne (cmp (Conv2B X) 0)" into "bool eq/ne (cmp X 0)".
1569   // This is a standard idiom for branching on a boolean value.
1570   Node *c2b = cmp1;
1571   if( cmp2_type == TypeInt::ZERO &&
1572       cmp1_op == Op_Conv2B &&
1573       (_test._test == BoolTest::eq ||
1574        _test._test == BoolTest::ne) ) {
1575     Node *ncmp = phase->transform(phase->type(c2b->in(1))->isa_int()
1576        ? (Node*)new CmpINode(c2b->in(1),cmp2)
1577        : (Node*)new CmpPNode(c2b->in(1),phase->makecon(TypePtr::NULL_PTR))
1578     );
1579     return new BoolNode( ncmp, _test._test );
1580   }
1581 
1582   // Comparing a SubI against a zero is equal to comparing the SubI
1583   // arguments directly.  This only works for eq and ne comparisons
1584   // due to possible integer overflow.
1585   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1586         (cop == Op_CmpI) &&
1587         (cmp1_op == Op_SubI) &&
1588         ( cmp2_type == TypeInt::ZERO ) ) {
1589     Node *ncmp = phase->transform( new CmpINode(cmp1->in(1),cmp1->in(2)));
1590     return new BoolNode( ncmp, _test._test );
1591   }
1592 
1593   // Same as above but with and AddI of a constant
1594   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1595       cop == Op_CmpI &&
1596       cmp1_op == Op_AddI &&
1597       cmp1->in(2) != NULL &&
1598       phase->type(cmp1->in(2))->isa_int() &&
1599       phase->type(cmp1->in(2))->is_int()->is_con() &&
1600       cmp2_type == TypeInt::ZERO &&
1601       !is_counted_loop_cmp(cmp) // modifying the exit test of a counted loop messes the counted loop shape
1602       ) {
1603     const TypeInt* cmp1_in2 = phase->type(cmp1->in(2))->is_int();
1604     Node *ncmp = phase->transform( new CmpINode(cmp1->in(1),phase->intcon(-cmp1_in2->_hi)));
1605     return new BoolNode( ncmp, _test._test );
1606   }
1607 
1608   // Change "bool eq/ne (cmp (phi (X -X) 0))" into "bool eq/ne (cmp X 0)"
1609   // since zero check of conditional negation of an integer is equal to
1610   // zero check of the integer directly.
1611   if ((_test._test == BoolTest::eq || _test._test == BoolTest::ne) &&
1612       (cop == Op_CmpI) &&
1613       (cmp2_type == TypeInt::ZERO) &&
1614       (cmp1_op == Op_Phi)) {
1615     // There should be a diamond phi with true path at index 1 or 2
1616     PhiNode *phi = cmp1->as_Phi();
1617     int idx_true = phi->is_diamond_phi();
1618     if (idx_true != 0) {
1619       // True input is in(idx_true) while false input is in(3 - idx_true)
1620       Node *tin = phi->in(idx_true);
1621       Node *fin = phi->in(3 - idx_true);
1622       if ((tin->Opcode() == Op_SubI) &&
1623           (phase->type(tin->in(1)) == TypeInt::ZERO) &&
1624           (tin->in(2) == fin)) {
1625         // Found conditional negation at true path, create a new CmpINode without that
1626         Node *ncmp = phase->transform(new CmpINode(fin, cmp2));
1627         return new BoolNode(ncmp, _test._test);
1628       }
1629       if ((fin->Opcode() == Op_SubI) &&
1630           (phase->type(fin->in(1)) == TypeInt::ZERO) &&
1631           (fin->in(2) == tin)) {
1632         // Found conditional negation at false path, create a new CmpINode without that
1633         Node *ncmp = phase->transform(new CmpINode(tin, cmp2));
1634         return new BoolNode(ncmp, _test._test);
1635       }
1636     }
1637   }
1638 
1639   // Change (-A vs 0) into (A vs 0) by commuting the test.  Disallow in the
1640   // most general case because negating 0x80000000 does nothing.  Needed for
1641   // the CmpF3/SubI/CmpI idiom.
1642   if( cop == Op_CmpI &&
1643       cmp1_op == Op_SubI &&
1644       cmp2_type == TypeInt::ZERO &&
1645       phase->type( cmp1->in(1) ) == TypeInt::ZERO &&
1646       phase->type( cmp1->in(2) )->higher_equal(TypeInt::SYMINT) ) {
1647     Node *ncmp = phase->transform( new CmpINode(cmp1->in(2),cmp2));
1648     return new BoolNode( ncmp, _test.commute() );
1649   }
1650 
1651   // Try to optimize signed integer comparison
1652   return fold_cmpI(phase, cmp->as_Sub(), cmp1, cop, cmp1_op, cmp2_type);
1653 
1654   //  The transformation below is not valid for either signed or unsigned
1655   //  comparisons due to wraparound concerns at MAX_VALUE and MIN_VALUE.
1656   //  This transformation can be resurrected when we are able to
1657   //  make inferences about the range of values being subtracted from
1658   //  (or added to) relative to the wraparound point.
1659   //
1660   //    // Remove +/-1's if possible.
1661   //    // "X <= Y-1" becomes "X <  Y"
1662   //    // "X+1 <= Y" becomes "X <  Y"
1663   //    // "X <  Y+1" becomes "X <= Y"
1664   //    // "X-1 <  Y" becomes "X <= Y"
1665   //    // Do not this to compares off of the counted-loop-end.  These guys are
1666   //    // checking the trip counter and they want to use the post-incremented
1667   //    // counter.  If they use the PRE-incremented counter, then the counter has
1668   //    // to be incremented in a private block on a loop backedge.
1669   //    if( du && du->cnt(this) && du->out(this)[0]->Opcode() == Op_CountedLoopEnd )
1670   //      return NULL;
1671   //  #ifndef PRODUCT
1672   //    // Do not do this in a wash GVN pass during verification.
1673   //    // Gets triggered by too many simple optimizations to be bothered with
1674   //    // re-trying it again and again.
1675   //    if( !phase->allow_progress() ) return NULL;

1687   //      else if( cmp2_op == Op_AddI &&
