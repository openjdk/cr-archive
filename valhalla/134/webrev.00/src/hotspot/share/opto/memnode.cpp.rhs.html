 108 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 109   st->print(" @");
 110   if (adr_type == NULL) {
 111     st->print("NULL");
 112   } else {
 113     adr_type->dump_on(st);
 114     Compile* C = Compile::current();
 115     Compile::AliasType* atp = NULL;
 116     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 117     if (atp == NULL)
 118       st->print(", idx=?\?;");
 119     else if (atp->index() == Compile::AliasIdxBot)
 120       st->print(", idx=Bot;");
 121     else if (atp->index() == Compile::AliasIdxTop)
 122       st->print(", idx=Top;");
 123     else if (atp->index() == Compile::AliasIdxRaw)
 124       st->print(", idx=Raw;");
 125     else {
 126       ciField* field = atp->field();
 127       if (field) {
 128         st->print(", name=");
 129         field->print_name_on(st);
 130       }
 131       st->print(", idx=%d;", atp->index());
 132     }
 133   }
 134 }

 140 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 141   assert((t_oop != NULL), "sanity");
 142   bool is_instance = t_oop->is_known_instance_field();
 143   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 144                              (load != NULL) && load->is_Load() &&
 145                              (phase->is_IterGVN() != NULL);
 146   if (!(is_instance || is_boxed_value_load))
 147     return mchain;  // don't try to optimize non-instance types
 148   uint instance_id = t_oop->instance_id();
 149   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 150   Node *prev = NULL;
 151   Node *result = mchain;
 152   while (prev != result) {
 153     prev = result;
 154     if (result == start_mem)
 155       break;  // hit one of our sentinels
 156     // skip over a call which does not affect this memory slice
 157     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 158       Node *proj_in = result->in(0);
 159       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 160         break;  // hit one of our sentinels
 161       } else if (proj_in->is_Call()) {
 162         // ArrayCopyNodes processed here as well
 163         CallNode *call = proj_in->as_Call();
 164         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 165           result = call->in(TypeFunc::Memory);
 166         }
 167       } else if (proj_in->is_Initialize()) {
 168         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 169         // Stop if this is the initialization for the object instance which
 170         // contains this memory slice, otherwise skip over it.
 171         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 172           break;
 173         }
 174         if (is_instance) {
 175           result = proj_in->in(TypeFunc::Memory);
 176         } else if (is_boxed_value_load) {
 177           Node* klass = alloc->in(AllocateNode::KlassNode);
 178           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 179           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 180             result = proj_in->in(TypeFunc::Memory); // not related allocation
 181           }
 182         }
 183       } else if (proj_in->is_MemBar()) {
 184         ArrayCopyNode* ac = NULL;
 185         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 186           break;
 187         }
 188         result = proj_in->in(TypeFunc::Memory);
 189       } else {
 190         assert(false, "unexpected projection");
 191       }
 192     } else if (result->is_ClearArray()) {
 193       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 194         // Can not bypass initialization of the instance
 195         // we are looking for.
 196         break;
 197       }
 198       // Otherwise skip it (the call updated 'result' value).
 199     } else if (result->is_MergeMem()) {
 200       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 201     }
 202   }
 203   return result;
 204 }

 206 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 207   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 208   if (t_oop == NULL)
 209     return mchain;  // don't try to optimize non-oop types
 210   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 211   bool is_instance = t_oop->is_known_instance_field();
 212   PhaseIterGVN *igvn = phase->is_IterGVN();
 213   if (is_instance && igvn != NULL && result->is_Phi()) {
 214     PhiNode *mphi = result->as_Phi();
 215     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 216     const TypePtr *t = mphi->adr_type();
 217     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 218         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 219          t->is_oopptr()->cast_to_exactness(true)
 220            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 221             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 222       // clone the Phi with our address type
 223       result = mphi->split_out_instance(t_adr, igvn);
 224     } else {
 225       if (t->isa_aryptr()) {
 226         // In the case of a flattened inline type array, each field has its own slice.
 227         // TODO This should be re-evaluated with JDK-8251039
 228         t = t->is_aryptr()->with_field_offset(t_adr->is_aryptr()->field_offset().get());
 229       }
 230       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 231     }
 232   }
 233   return result;
 234 }

 236 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 237   uint alias_idx = phase->C->get_alias_index(tp);
 238   Node *mem = mmem;
 239 #ifdef ASSERT
 240   {
 241     // Check that current type is consistent with the alias index used during graph construction
 242     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 243     bool consistent =  adr_check == NULL || adr_check->empty() ||
 244                        phase->C->must_alias(adr_check, alias_idx );
 245     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 246     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 247         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 248         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 249         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 250           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 251           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 252       // don't assert if it is dead code.
 253       consistent = true;
 254     }
 255     if( !consistent ) {
 256       st->print("alias_idx==%d, adr_check==", alias_idx);
 257       if( adr_check == NULL ) {
 258         st->print("NULL");
 259       } else {
 260         adr_check->dump();
 261       }
 262       st->cr();
 263       print_alias_types();
 264       assert(consistent, "adr_check must match alias idx");
 265     }
 266   }
 267 #endif
 268   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 269   // means an array I have not precisely typed yet.  Do not do any
 270   // alias stuff with it any time soon.
 271   const TypeOopPtr *toop = tp->isa_oopptr();
 272   if( tp->base() != Type::AnyPtr &&
 273       !(toop &&
 274         toop->klass() != NULL &&
 275         toop->klass()->is_java_lang_Object() &&
 276         toop->offset() == Type::OffsetBot) ) {
 277     // compress paths and change unreachable cycles to TOP
 278     // If not, we can update the input infinitely along a MergeMem cycle
 279     // Equivalent code in PhiNode::Ideal
 280     Node* m  = phase->transform(mmem);
 281     // If transformed to a MergeMem, get the desired slice
 282     // Otherwise the returned node represents memory for every slice
 283     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 284     // Update input if it is progress over what we have now
 285   }
 286   return mem;
 287 }

 289 //--------------------------Ideal_common---------------------------------------
 290 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 291 // Unhook non-raw memories from complete (macro-expanded) initializations.
 292 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 293   // If our control input is a dead region, kill all below the region
 294   Node *ctl = in(MemNode::Control);
 295   if (ctl && remove_dead_region(phase, can_reshape))
 296     return this;
 297   ctl = in(MemNode::Control);
 298   // Don't bother trying to transform a dead node
 299   if (ctl && ctl->is_top())  return NodeSentinel;
 300 
 301   PhaseIterGVN *igvn = phase->is_IterGVN();
 302   // Wait if control on the worklist.
 303   if (ctl && can_reshape && igvn != NULL) {
 304     Node* bol = NULL;
 305     Node* cmp = NULL;
 306     if (ctl->in(0)->is_If()) {
 307       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 308       bol = ctl->in(0)->in(1);
 309       if (bol->is_Bool())
 310         cmp = ctl->in(0)->in(1)->in(1);
 311     }
 312     if (igvn->_worklist.member(ctl) ||
 313         (bol != NULL && igvn->_worklist.member(bol)) ||
 314         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 315       // This control path may be dead.
 316       // Delay this memory node transformation until the control is processed.
 317       phase->is_IterGVN()->_worklist.push(this);
 318       return NodeSentinel; // caller will return NULL
 319     }
 320   }
 321   // Ignore if memory is dead, or self-loop
 322   Node *mem = in(MemNode::Memory);
 323   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 324   assert(mem != this, "dead loop in MemNode::Ideal");
 325 
 326   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 327     // This memory slice may be dead.
 328     // Delay this mem node transformation until the memory is processed.
 329     phase->is_IterGVN()->_worklist.push(this);
 330     return NodeSentinel; // caller will return NULL
 331   }
 332 
 333   Node *address = in(MemNode::Address);
 334   const Type *t_adr = phase->type(address);
 335   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 336 
 337   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 338     // Unsafe off-heap access with zero address. Remove access and other control users
 339     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 340     assert(ctl != NULL, "unsafe accesses should be control dependent");
 341     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 342       Node* u = ctl->fast_out(i);
 343       if (u != ctl) {
 344         igvn->rehash_node_delayed(u);
 345         int nb = u->replace_edge(ctl, phase->C->top());
 346         --i, imax -= nb;
 347       }
 348     }
 349     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 350     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 351     phase->C->root()->add_req(halt);
 352     return this;
 353   }
 354 
 355   if (can_reshape && igvn != NULL &&
 356       (igvn->_worklist.member(address) ||
 357        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 358     // The address's base and type may change when the address is processed.
 359     // Delay this mem node transformation until the address is processed.
 360     phase->is_IterGVN()->_worklist.push(this);
 361     return NodeSentinel; // caller will return NULL
 362   }
 363 
 364   // Do NOT remove or optimize the next lines: ensure a new alias index
 365   // is allocated for an oop pointer type before Escape Analysis.
 366   // Note: C++ will not remove it since the call has side effect.
 367   if (t_adr->isa_oopptr()) {
 368     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 369   }
 370 
 371   Node* base = NULL;
 372   if (address->is_AddP()) {
 373     base = address->in(AddPNode::Base);
 374   }
 375   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 376       !t_adr->isa_rawptr()) {
 377     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 378     // Skip this node optimization if its address has TOP base.
 379     return NodeSentinel; // caller will return NULL
 380   }
 381 
 382   // Avoid independent memory operations
 383   Node* old_mem = mem;
 384 
 385   // The code which unhooks non-raw memories from complete (macro-expanded)
 386   // initializations was removed. After macro-expansion all stores catched
 387   // by Initialize node became raw stores and there is no information
 388   // which memory slices they modify. So it is unsafe to move any memory
 389   // operation above these stores. Also in most cases hooked non-raw memories
 390   // were already unhooked by using information from detect_ptr_independence()
 391   // and find_previous_store().
 392 
 393   if (mem->is_MergeMem()) {
 394     MergeMemNode* mmem = mem->as_MergeMem();
 395     const TypePtr *tp = t_adr->is_ptr();
 396 
 397     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 398   }
 399 
 400   if (mem != old_mem) {
 401     set_req(MemNode::Memory, mem);
 402     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 403       igvn->_worklist.push(old_mem);
 404     }
 405     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 406     return this;
 407   }
 408 
 409   // let the subclass continue analyzing...
 410   return NULL;
 411 }

 413 // Helper function for proving some simple control dominations.
 414 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 415 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 416 // is not a constant (dominated by the method's StartNode).
 417 // Used by MemNode::find_previous_store to prove that the
 418 // control input of a memory operation predates (dominates)
 419 // an allocation it wants to look past.
 420 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 421   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 422     return false; // Conservative answer for dead code
 423 
 424   // Check 'dom'. Skip Proj and CatchProj nodes.
 425   dom = dom->find_exact_control(dom);
 426   if (dom == NULL || dom->is_top())
 427     return false; // Conservative answer for dead code
 428 
 429   if (dom == sub) {
 430     // For the case when, for example, 'sub' is Initialize and the original
 431     // 'dom' is Proj node of the 'sub'.
 432     return false;
 433   }
 434 
 435   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 436     return true;
 437 
 438   // 'dom' dominates 'sub' if its control edge and control edges
 439   // of all its inputs dominate or equal to sub's control edge.
 440 
 441   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 442   // Or Region for the check in LoadNode::Ideal();
 443   // 'sub' should have sub->in(0) != NULL.
 444   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 445          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 446 
 447   // Get control edge of 'sub'.
 448   Node* orig_sub = sub;
 449   sub = sub->find_exact_control(sub->in(0));
 450   if (sub == NULL || sub->is_top())
 451     return false; // Conservative answer for dead code
 452 
 453   assert(sub->is_CFG(), "expecting control");
 454 
 455   if (sub == dom)
 456     return true;
 457 
 458   if (sub->is_Start() || sub->is_Root())
 459     return false;
 460 
 461   {
 462     // Check all control edges of 'dom'.
 463 
 464     ResourceMark rm;
 465     Arena* arena = Thread::current()->resource_area();
 466     Node_List nlist(arena);
 467     Unique_Node_List dom_list(arena);
 468 
 469     dom_list.push(dom);
 470     bool only_dominating_controls = false;
 471 
 472     for (uint next = 0; next < dom_list.size(); next++) {
 473       Node* n = dom_list.at(next);
 474       if (n == orig_sub)
 475         return false; // One of dom's inputs dominated by sub.
 476       if (!n->is_CFG() && n->pinned()) {
 477         // Check only own control edge for pinned non-control nodes.
 478         n = n->find_exact_control(n->in(0));
 479         if (n == NULL || n->is_top())
 480           return false; // Conservative answer for dead code
 481         assert(n->is_CFG(), "expecting control");
 482         dom_list.push(n);
 483       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 484         only_dominating_controls = true;
 485       } else if (n->is_CFG()) {
 486         if (n->dominates(sub, nlist))
 487           only_dominating_controls = true;
 488         else
 489           return false;
 490       } else {
 491         // First, own control edge.
 492         Node* m = n->find_exact_control(n->in(0));
 493         if (m != NULL) {
 494           if (m->is_top())
 495             return false; // Conservative answer for dead code
 496           dom_list.push(m);
 497         }
 498         // Now, the rest of edges.
 499         uint cnt = n->req();
 500         for (uint i = 1; i < cnt; i++) {
 501           m = n->find_exact_control(n->in(i));
 502           if (m == NULL || m->is_top())
 503             continue;
 504           dom_list.push(m);
 505         }
 506       }
 507     }
 508     return only_dominating_controls;
 509   }
 510 }

 512 //---------------------detect_ptr_independence---------------------------------
 513 // Used by MemNode::find_previous_store to prove that two base
 514 // pointers are never equal.
 515 // The pointers are accompanied by their associated allocations,
 516 // if any, which have been previously discovered by the caller.
 517 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 518                                       Node* p2, AllocateNode* a2,
 519                                       PhaseTransform* phase) {
 520   // Attempt to prove that these two pointers cannot be aliased.
 521   // They may both manifestly be allocations, and they should differ.
 522   // Or, if they are not both allocations, they can be distinct constants.
 523   // Otherwise, one is an allocation and the other a pre-existing value.
 524   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 525     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 526   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 527     return (a1 != a2);
 528   } else if (a1 != NULL) {                  // one allocation a1
 529     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 530     return all_controls_dominate(p2, a1);
 531   } else { //(a2 != NULL)                   // one allocation a2
 532     return all_controls_dominate(p1, a2);
 533   }
 534   return false;
 535 }

 538 // Find an arraycopy that must have set (can_see_stored_value=true) or
 539 // could have set (can_see_stored_value=false) the value for this load
 540 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 541   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 542                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 543     Node* mb = mem->in(0);
 544     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 545         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 546       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 547       if (ac->is_clonebasic()) {
 548         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 549         if (alloc != NULL && alloc == ld_alloc) {
 550           return ac;
 551         }
 552       }
 553     }
 554   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 555     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 556 
 557     if (ac->is_arraycopy_validated() ||
 558         ac->is_copyof_validated() ||
 559         ac->is_copyofrange_validated()) {
 560       Node* ld_addp = in(MemNode::Address);
 561       if (ld_addp->is_AddP()) {
 562         Node* ld_base = ld_addp->in(AddPNode::Address);
 563         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 564 
 565         Node* dest = ac->in(ArrayCopyNode::Dest);
 566 
 567         if (dest == ld_base) {
 568           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 569           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 570             return ac;
 571           }
 572           if (!can_see_stored_value) {
 573             mem = ac->in(TypeFunc::Memory);
 574           }
 575         }
 576       }
 577     }
 578   }
 579   return NULL;
 580 }

 582 // The logic for reordering loads and stores uses four steps:
 583 // (a) Walk carefully past stores and initializations which we
 584 //     can prove are independent of this load.
 585 // (b) Observe that the next memory state makes an exact match
 586 //     with self (load or store), and locate the relevant store.
 587 // (c) Ensure that, if we were to wire self directly to the store,
 588 //     the optimizer would fold it up somehow.
 589 // (d) Do the rewiring, and return, depending on some other part of
 590 //     the optimizer to fold up the load.
 591 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 592 // specific to loads and stores, so they are handled by the callers.
 593 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 594 //
 595 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 596   Node*         ctrl   = in(MemNode::Control);
 597   Node*         adr    = in(MemNode::Address);
 598   intptr_t      offset = 0;
 599   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 600   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 601 
 602   if (offset == Type::OffsetBot)
 603     return NULL;            // cannot unalias unless there are precise offsets
 604 
 605   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 606   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 607 
 608   intptr_t size_in_bytes = memory_size();
 609 
 610   Node* mem = in(MemNode::Memory);   // start searching here...
 611 
 612   int cnt = 50;             // Cycle limiter
 613   for (;;) {                // While we can dance past unrelated stores...
 614     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 615 
 616     Node* prev = mem;
 617     if (mem->is_Store()) {
 618       Node* st_adr = mem->in(MemNode::Address);
 619       intptr_t st_offset = 0;
 620       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 621       if (st_base == NULL)
 622         break;              // inscrutable pointer
 623 
 624       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 625       // We need the bases to be the equal in order for the offset check to make sense.
 626       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 627         break;
 628       }
 629 
 630       if (st_offset != offset && st_offset != Type::OffsetBot) {
 631         const int MAX_STORE = BytesPerLong;
 632         if (st_offset >= offset + size_in_bytes ||
 633             st_offset <= offset - MAX_STORE ||
 634             st_offset <= offset - mem->as_Store()->memory_size()) {
 635           // Success:  The offsets are provably independent.
 636           // (You may ask, why not just test st_offset != offset and be done?
 637           // The answer is that stores of different sizes can co-exist
 638           // in the same sequence of RawMem effects.  We sometimes initialize
 639           // a whole 'tile' of array elements with a single jint or jlong.)
 640           mem = mem->in(MemNode::Memory);
 641           continue;           // (a) advance through independent store memory
 642         }
 643       }
 644       if (st_base != base &&
 645           detect_ptr_independence(base, alloc,
 646                                   st_base,
 647                                   AllocateNode::Ideal_allocation(st_base, phase),
 648                                   phase)) {
 649         // Success:  The bases are provably independent.
 650         mem = mem->in(MemNode::Memory);
 651         continue;           // (a) advance through independent store memory
 652       }
 653 
 654       // (b) At this point, if the bases or offsets do not agree, we lose,
 655       // since we have not managed to prove 'this' and 'mem' independent.
 656       if (st_base == base && st_offset == offset) {
 657         return mem;         // let caller handle steps (c), (d)
 658       }
 659 
 660     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 661       InitializeNode* st_init = mem->in(0)->as_Initialize();
 662       AllocateNode*  st_alloc = st_init->allocation();
 663       if (st_alloc == NULL)
 664         break;              // something degenerated
 665       bool known_identical = false;
 666       bool known_independent = false;
 667       if (alloc == st_alloc)
 668         known_identical = true;
 669       else if (alloc != NULL)
 670         known_independent = true;
 671       else if (all_controls_dominate(this, st_alloc))
 672         known_independent = true;
 673 
 674       if (known_independent) {
 675         // The bases are provably independent: Either they are
 676         // manifestly distinct allocations, or else the control
 677         // of this load dominates the store's allocation.
 678         int alias_idx = phase->C->get_alias_index(adr_type());
 679         if (alias_idx == Compile::AliasIdxRaw) {
 680           mem = st_alloc->in(TypeFunc::Memory);
 681         } else {
 682           mem = st_init->memory(alias_idx);
 683         }
 684         continue;           // (a) advance through independent store memory
 685       }
 686 
 687       // (b) at this point, if we are not looking at a store initializing
 688       // the same allocation we are loading from, we lose.
 689       if (known_identical) {
 690         // From caller, can_see_stored_value will consult find_captured_store.
 691         return mem;         // let caller handle steps (c), (d)
 692       }
 693 
 694     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 695       if (prev != mem) {
 696         // Found an arraycopy but it doesn't affect that load
 697         continue;
 698       }
 699       // Found an arraycopy that may affect that load
 700       return mem;
 701     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 702       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 703       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 704         // ArrayCopyNodes processed here as well.
 705         CallNode *call = mem->in(0)->as_Call();
 706         if (!call->may_modify(addr_t, phase)) {
 707           mem = call->in(TypeFunc::Memory);
 708           continue;         // (a) advance through independent call memory
 709         }
 710       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 711         ArrayCopyNode* ac = NULL;
 712         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 713           break;
 714         }
 715         mem = mem->in(0)->in(TypeFunc::Memory);
 716         continue;           // (a) advance through independent MemBar memory
 717       } else if (mem->is_ClearArray()) {
 718         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 719           // (the call updated 'mem' value)
 720           continue;         // (a) advance through independent allocation memory
 721         } else {
 722           // Can not bypass initialization of the instance
 723           // we are looking for.
 724           return mem;
 725         }
 726       } else if (mem->is_MergeMem()) {
 727         int alias_idx = phase->C->get_alias_index(adr_type());
 728         mem = mem->as_MergeMem()->memory_at(alias_idx);
 729         continue;           // (a) advance through independent MergeMem memory
 730       }
 731     }
 732 
 733     // Unless there is an explicit 'continue', we must bail out here,
 734     // because 'mem' is an inscrutable memory state (e.g., a call).
 735     break;
 736   }
 737 
 738   return NULL;              // bail out
 739 }

 742 // Helper function.  Notices when the given type of address hits top or bottom.
 743 // Also, asserts a cross-check of the type against the expected address type.
 744 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 745   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 746   #ifdef ASSERT
 747   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 748   #endif

 750   if (tp == NULL) {
 751     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 752     return TypePtr::BOTTOM;           // touches lots of memory
 753   } else {
 754     #ifdef ASSERT
 755     // %%%% [phh] We don't check the alias index if cross_check is
 756     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 757     if (cross_check != NULL &&
 758         cross_check != TypePtr::BOTTOM &&
 759         cross_check != TypeRawPtr::BOTTOM) {
 760       // Recheck the alias index, to see if it has changed (due to a bug).

 769              "real address must not escape from expected memory type");
 770     }
 771     #endif
 772     return tp;
 773   }

 932 // Is the value loaded previously stored by an arraycopy? If so return
 933 // a load node that reads from the source array so we may be able to
 934 // optimize out the ArrayCopy node later.
 935 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 936   Node* ld_adr = in(MemNode::Address);
 937   intptr_t ld_off = 0;
 938   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 939   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 940   if (ac != NULL) {
 941     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 942 
 943     Node* mem = ac->in(TypeFunc::Memory);
 944     Node* ctl = ac->in(0);
 945     Node* src = ac->in(ArrayCopyNode::Src);
 946 
 947     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 948       return NULL;
 949     }

 951     LoadNode* ld = clone()->as_Load();
 952     Node* addp = in(MemNode::Address)->clone();
 953     if (ac->as_ArrayCopy()->is_clonebasic()) {
 954       assert(ld_alloc != NULL, "need an alloc");
 955       assert(addp->is_AddP(), "address must be addp");
 956       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 957       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 958       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 959       addp->set_req(AddPNode::Base, src);
 960       addp->set_req(AddPNode::Address, src);
 961     } else {
 962       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 963              ac->as_ArrayCopy()->is_copyof_validated() ||
 964              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 965       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 966       addp->set_req(AddPNode::Base, src);
 967       addp->set_req(AddPNode::Address, src);
 968 
 969       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 970       BasicType ary_elem = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 971       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 972       uint shift  = exact_log2(type2aelembytes(ary_elem));
 973       if (ary_t->klass()->is_flat_array_klass()) {
 974         ciFlatArrayKlass* vak = ary_t->klass()->as_flat_array_klass();
 975         shift = vak->log2_element_size();
 976       }
 977 
 978       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 979 #ifdef _LP64
 980       diff = phase->transform(new ConvI2LNode(diff));
 981 #endif
 982       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 983 
 984       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 985       addp->set_req(AddPNode::Offset, offset);
 986     }
 987     addp = phase->transform(addp);
 988 #ifdef ASSERT
 989     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 990     ld->_adr_type = adr_type;
 991 #endif
 992     ld->set_req(MemNode::Address, addp);
 993     ld->set_req(0, ctl);
 994     ld->set_req(MemNode::Memory, mem);
 995     // load depends on the tests that validate the arraycopy
 996     ld->_control_dependency = UnknownControl;
 997     return ld;
 998   }
 999   return NULL;

1003 //---------------------------can_see_stored_value------------------------------
1004 // This routine exists to make sure this set of tests is done the same
1005 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1006 // will change the graph shape in a way which makes memory alive twice at the
1007 // same time (uses the Oracle model of aliasing), then some
1008 // LoadXNode::Identity will fold things back to the equivalence-class model
1009 // of aliasing.
1010 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1011   Node* ld_adr = in(MemNode::Address);
1012   intptr_t ld_off = 0;
1013   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1014   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1015   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1016   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1017   // This is more general than load from boxing objects.
1018   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1019     uint alias_idx = atp->index();
1020     bool final = !atp->is_rewritable();
1021     Node* result = NULL;
1022     Node* current = st;
1023     // Skip through chains of MemBarNodes checking the MergeMems for
1024     // new states for the slice of this load.  Stop once any other
1025     // kind of node is encountered.  Loads from final memory can skip
1026     // through any kind of MemBar but normal loads shouldn't skip
1027     // through MemBarAcquire since the could allow them to move out of
1028     // a synchronized region.
1029     while (current->is_Proj()) {
1030       int opc = current->in(0)->Opcode();
1031       if ((final && (opc == Op_MemBarAcquire ||
1032                      opc == Op_MemBarAcquireLock ||
1033                      opc == Op_LoadFence)) ||
1034           opc == Op_MemBarRelease ||
1035           opc == Op_StoreFence ||
1036           opc == Op_MemBarReleaseLock ||
1037           opc == Op_MemBarStoreStore ||
1038           opc == Op_MemBarCPUOrder) {
1039         Node* mem = current->in(0)->in(TypeFunc::Memory);
1040         if (mem->is_MergeMem()) {
1041           MergeMemNode* merge = mem->as_MergeMem();
1042           Node* new_st = merge->memory_at(alias_idx);
1043           if (new_st == merge->base_memory()) {
1044             // Keep searching
1045             current = new_st;
1046             continue;
1047           }
1048           // Save the new memory state for the slice and fall through
1049           // to exit.
1050           result = new_st;
1051         }
1052       }
1053       break;
1054     }
1055     if (result != NULL) {
1056       st = result;
1057     }
1058   }
1059 
1060   // Loop around twice in the case Load -> Initialize -> Store.
1061   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1062   for (int trip = 0; trip <= 1; trip++) {
1063 
1064     if (st->is_Store()) {
1065       Node* st_adr = st->in(MemNode::Address);
1066       if (!phase->eqv(st_adr, ld_adr)) {
1067         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1068         intptr_t st_off = 0;
1069         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1070         if (ld_base == NULL)                                   return NULL;
1071         if (st_base == NULL)                                   return NULL;
1072         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1073         if (ld_off != st_off)                                  return NULL;
1074         if (ld_off == Type::OffsetBot)                         return NULL;
1075         // Same base, same offset.
1076         // Possible improvement for arrays: check index value instead of absolute offset.
1077 
1078         // At this point we have proven something like this setup:
1079         //   B = << base >>
1080         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1081         //   S = StoreQ(AddP(             B , #Off), V)
1082         // (Actually, we haven't yet proven the Q's are the same.)
1083         // In other words, we are loading from a casted version of
1084         // the same pointer-and-offset that we stored to.
1085         // Casted version may carry a dependency and it is respected.
1086         // Thus, we are able to replace L by V.
1087       }
1088       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1089       if (store_Opcode() != st->Opcode())
1090         return NULL;
1091       return st->in(MemNode::ValueIn);
1092     }
1093 
1094     // A load from a freshly-created object always returns zero.
1095     // (This can happen after LoadNode::Ideal resets the load's memory input
1096     // to find_captured_store, which returned InitializeNode::zero_memory.)
1097     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1098         (st->in(0) == ld_alloc) &&
1099         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1100       // return a zero value for the load's basic type
1101       // (This is one of the few places where a generic PhaseTransform
1102       // can create new nodes.  Think of it as lazily manifesting
1103       // virtually pre-existing constants.)
1104       assert(memory_type() != T_INLINE_TYPE, "should not be used for inline types");
1105       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1106       if (default_value != NULL) {
1107         return default_value;
1108       }
1109       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1110       return phase->zerocon(memory_type());
1111     }
1112 
1113     // A load from an initialization barrier can match a captured store.
1114     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1115       InitializeNode* init = st->in(0)->as_Initialize();
1116       AllocateNode* alloc = init->allocation();
1117       if ((alloc != NULL) && (alloc == ld_alloc)) {
1118         // examine a captured store value
1119         st = init->find_captured_store(ld_off, memory_size(), phase);
1120         if (st != NULL) {
1121           continue;             // take one more trip around
1122         }
1123       }
1124     }
1125 
1126     // Load boxed value from result of valueOf() call is input parameter.
1127     if (this->is_Load() && ld_adr->is_AddP() &&
1128         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1129       intptr_t ignore = 0;
1130       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1131       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1132       base = bs->step_over_gc_barrier(base);
1133       if (base != NULL && base->is_Proj() &&
1134           base->as_Proj()->_con == TypeFunc::Parms &&
1135           base->in(0)->is_CallStaticJava() &&
1136           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1137         return base->in(0)->in(TypeFunc::Parms);
1138       }
1139     }
1140 
1141     break;
1142   }
1143 
1144   return NULL;
1145 }

1164 //------------------------------Identity---------------------------------------
1165 // Loads are identity if previous store is to same address
1166 Node* LoadNode::Identity(PhaseGVN* phase) {
1167   // Loading from an InlineTypePtr? The InlineTypePtr has the values of
1168   // all fields as input. Look for the field with matching offset.
1169   Node* addr = in(Address);
1170   intptr_t offset;
1171   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1172   if (base != NULL && base->is_InlineTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1173     Node* value = base->as_InlineTypePtr()->field_value_by_offset((int)offset, true);
1174     if (value->is_InlineType()) {
1175       // Non-flattened inline type field
1176       InlineTypeNode* vt = value->as_InlineType();
1177       if (vt->is_allocated(phase)) {
1178         value = vt->get_oop();
1179       } else {
1180         // Not yet allocated, bail out
1181         value = NULL;
1182       }
1183     }
1184     if (value != NULL) {
1185       if (Opcode() == Op_LoadN) {
1186         // Encode oop value if we are loading a narrow oop
1187         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1188         value = phase->transform(new EncodePNode(value, bottom_type()));
1189       }
1190       return value;
1191     }
1192   }
1193 
1194   // If the previous store-maker is the right kind of Store, and the store is
1195   // to the same address, then we are equal to the value stored.
1196   Node* mem = in(Memory);
1197   Node* value = can_see_stored_value(mem, phase);
1198   if( value ) {
1199     // byte, short & char stores truncate naturally.
1200     // A load has to load the truncated value which requires
1201     // some sort of masking operation and that requires an
1202     // Ideal call instead of an Identity call.
1203     if (memory_size() < BytesPerInt) {
1204       // If the input to the store does not fit with the load's result type,
1205       // it must be truncated via an Ideal call.
1206       if (!phase->type(value)->higher_equal(phase->type(this)))
1207         return this;
1208     }
1209     // (This works even when value is a Con, but LoadNode::Value
1210     // usually runs first, producing the singleton type of the Con.)
1211     return value;
1212   }
1213 
1214   // Search for an existing data phi which was generated before for the same
1215   // instance's field to avoid infinite generation of phis in a loop.
1216   Node *region = mem->in(0);
1217   if (is_instance_field_load_with_local_phi(region)) {
1218     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1219     int this_index  = phase->C->get_alias_index(addr_t);
1220     int this_offset = addr_t->offset();
1221     int this_iid    = addr_t->instance_id();
1222     if (!addr_t->is_known_instance() &&
1223          addr_t->is_ptr_to_boxed_value()) {
1224       // Use _idx of address base (could be Phi node) for boxed values.
1225       intptr_t   ignore = 0;
1226       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1227       if (base == NULL) {
1228         return this;
1229       }
1230       this_iid = base->_idx;
1231     }
1232     const Type* this_type = bottom_type();
1233     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1234       Node* phi = region->fast_out(i);
1235       if (phi->is_Phi() && phi != mem &&
1236           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1237         return phi;
1238       }
1239     }
1240   }
1241 
1242   return this;
1243 }

1283 // We're loading from an object which has autobox behaviour.
1284 // If this object is result of a valueOf call we'll have a phi
1285 // merging a newly allocated object and a load from the cache.
1286 // We want to replace this load with the original incoming
1287 // argument to the valueOf call.
1288 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1289   assert(phase->C->eliminate_boxing(), "sanity");
1290   intptr_t ignore = 0;
1291   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1292   if ((base == NULL) || base->is_Phi()) {
1293     // Push the loads from the phi that comes from valueOf up
1294     // through it to allow elimination of the loads and the recovery
1295     // of the original value. It is done in split_through_phi().
1296     return NULL;
1297   } else if (base->is_Load() ||
1298              (base->is_DecodeN() && base->in(1)->is_Load())) {
1299     // Eliminate the load of boxed value for integer types from the cache
1300     // array by deriving the value from the index into the array.
1301     // Capture the offset of the load and then reverse the computation.
1302 
1303     // Get LoadN node which loads a boxing object from 'cache' array.
1304     if (base->is_DecodeN()) {
1305       base = base->in(1);
1306     }
1307     if (!base->in(Address)->is_AddP()) {
1308       return NULL; // Complex address
1309     }
1310     AddPNode* address = base->in(Address)->as_AddP();
1311     Node* cache_base = address->in(AddPNode::Base);
1312     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1313       // Get ConP node which is static 'cache' field.
1314       cache_base = cache_base->in(1);
1315     }
1316     if ((cache_base != NULL) && cache_base->is_Con()) {
1317       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1318       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1319         Node* elements[4];
1320         int shift = exact_log2(type2aelembytes(T_OBJECT));
1321         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1322         if (count > 0 && elements[0]->is_Con() &&
1323             (count == 1 ||
1324              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1325                             elements[1]->in(2) == phase->intcon(shift)))) {
1326           ciObjArray* array = base_type->const_oop()->as_obj_array();
1327           // Fetch the box object cache[0] at the base of the array and get its value
1328           ciInstance* box = array->obj_at(0)->as_instance();
1329           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1330           assert(ik->is_box_klass(), "sanity");
1331           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1332           if (ik->nof_nonstatic_fields() == 1) {
1333             // This should be true nonstatic_field_at requires calling
1334             // nof_nonstatic_fields so check it anyway
1335             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1336             BasicType bt = c.basic_type();
1337             // Only integer types have boxing cache.
1338             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1339                    bt == T_BYTE    || bt == T_SHORT ||
1340                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1341             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1342             if (cache_low != (int)cache_low) {
1343               return NULL; // should not happen since cache is array indexed by value
1344             }
1345             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1346             if (offset != (int)offset) {
1347               return NULL; // should not happen since cache is array indexed by value
1348             }
1349            // Add up all the offsets making of the address of the load
1350             Node* result = elements[0];
1351             for (int i = 1; i < count; i++) {
1352               result = phase->transform(new AddXNode(result, elements[i]));
1353             }
1354             // Remove the constant offset from the address and then
1355             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1356             // remove the scaling of the offset to recover the original index.
1357             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1358               // Peel the shift off directly but wrap it in a dummy node
1359               // since Ideal can't return existing nodes
1360               result = new RShiftXNode(result->in(1), phase->intcon(0));
1361             } else if (result->is_Add() && result->in(2)->is_Con() &&
1362                        result->in(1)->Opcode() == Op_LShiftX &&
1363                        result->in(1)->in(2) == phase->intcon(shift)) {
1364               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1365               // but for boxing cache access we know that X<<Z will not overflow
1366               // (there is range check) so we do this optimizatrion by hand here.
1367               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1368               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1369             } else {
1370               result = new RShiftXNode(result, phase->intcon(shift));
1371             }
1372 #ifdef _LP64
1373             if (bt != T_LONG) {
1374               result = new ConvL2INode(phase->transform(result));
1375             }
1376 #else
1377             if (bt == T_LONG) {
1378               result = new ConvI2LNode(phase->transform(result));
1379             }
1380 #endif
1381             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1382             // Need to preserve unboxing load type if it is unsigned.
1383             switch(this->Opcode()) {
1384               case Op_LoadUB:
1385                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1386                 break;
1387               case Op_LoadUS:
1388                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1389                 break;
1390             }
1391             return result;
1392           }
1393         }
1394       }
1395     }
1396   }
1397   return NULL;
1398 }

1400 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1401   Node* region = phi->in(0);
1402   if (region == NULL) {
1403     return false; // Wait stable graph
1404   }
1405   uint cnt = phi->req();
1406   for (uint i = 1; i < cnt; i++) {
1407     Node* rc = region->in(i);
1408     if (rc == NULL || phase->type(rc) == Type::TOP)
1409       return false; // Wait stable graph
1410     Node* in = phi->in(i);
1411     if (in == NULL || phase->type(in) == Type::TOP)
1412       return false; // Wait stable graph
1413   }
1414   return true;

1416 //------------------------------split_through_phi------------------------------
1417 // Split instance or boxed field load through Phi.
1418 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1419   Node* mem     = in(Memory);
1420   Node* address = in(Address);
1421   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1422 
1423   assert((t_oop != NULL) &&
1424          (t_oop->is_known_instance_field() ||
1425           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1426 
1427   Compile* C = phase->C;
1428   intptr_t ignore = 0;
1429   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1430   bool base_is_phi = (base != NULL) && base->is_Phi();
1431   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1432                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1433                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1434 
1435   if (!((mem->is_Phi() || base_is_phi) &&
1436         (load_boxed_values || t_oop->is_known_instance_field()))) {
1437     return NULL; // memory is not Phi
1438   }
1439 
1440   if (mem->is_Phi()) {
1441     if (!stable_phi(mem->as_Phi(), phase)) {
1442       return NULL; // Wait stable graph
1443     }
1444     uint cnt = mem->req();
1445     // Check for loop invariant memory.
1446     if (cnt == 3) {
1447       for (uint i = 1; i < cnt; i++) {
1448         Node* in = mem->in(i);
1449         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1450         if (m == mem) {
1451           if (i == 1) {
1452             // if the first edge was a loop, check second edge too.
1453             // If both are replaceable - we are in an infinite loop
1454             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1455             if (n == mem) {
1456               break;
1457             }
1458           }
1459           set_req(Memory, mem->in(cnt - i));
1460           return this; // made change
1461         }
1462       }
1463     }
1464   }
1465   if (base_is_phi) {
1466     if (!stable_phi(base->as_Phi(), phase)) {
1467       return NULL; // Wait stable graph
1468     }
1469     uint cnt = base->req();
1470     // Check for loop invariant memory.
1471     if (cnt == 3) {
1472       for (uint i = 1; i < cnt; i++) {
1473         if (base->in(i) == base) {
1474           return NULL; // Wait stable graph
1475         }
1476       }
1477     }
1478   }
1479 
1480   // Split through Phi (see original code in loopopts.cpp).
1481   assert(C->have_alias_type(t_oop), "instance should have alias type");
1482 
1483   // Do nothing here if Identity will find a value
1484   // (to avoid infinite chain of value phis generation).
1485   if (!phase->eqv(this, this->Identity(phase))) {
1486     return NULL;
1487   }
1488 
1489   // Select Region to split through.
1490   Node* region;
1491   if (!base_is_phi) {
1492     assert(mem->is_Phi(), "sanity");
1493     region = mem->in(0);
1494     // Skip if the region dominates some control edge of the address.
1495     if (!MemNode::all_controls_dominate(address, region))
1496       return NULL;
1497   } else if (!mem->is_Phi()) {
1498     assert(base_is_phi, "sanity");
1499     region = base->in(0);
1500     // Skip if the region dominates some control edge of the memory.
1501     if (!MemNode::all_controls_dominate(mem, region))
1502       return NULL;
1503   } else if (base->in(0) != mem->in(0)) {
1504     assert(base_is_phi && mem->is_Phi(), "sanity");
1505     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1506       region = base->in(0);
1507     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1508       region = mem->in(0);
1509     } else {
1510       return NULL; // complex graph
1511     }
1512   } else {
1513     assert(base->in(0) == mem->in(0), "sanity");
1514     region = mem->in(0);
1515   }
1516 
1517   const Type* this_type = this->bottom_type();
1518   int this_index  = C->get_alias_index(t_oop);
1519   int this_offset = t_oop->offset();
1520   int this_iid    = t_oop->instance_id();
1521   if (!t_oop->is_known_instance() && load_boxed_values) {
1522     // Use _idx of address base for boxed values.
1523     this_iid = base->_idx;
1524   }
1525   PhaseIterGVN* igvn = phase->is_IterGVN();
1526   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1527   for (uint i = 1; i < region->req(); i++) {
1528     Node* x;
1529     Node* the_clone = NULL;
1530     Node* in = region->in(i);
1531     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1532         in != NULL && in->is_OuterStripMinedLoop()) {
1533       // No node should go in the outer strip mined loop
1534       in = in->in(LoopNode::EntryControl);
1535     }
1536     if (in == NULL || in == C->top()) {
1537       x = C->top();      // Dead path?  Use a dead data op
1538     } else {
1539       x = this->clone();        // Else clone up the data op
1540       the_clone = x;            // Remember for possible deletion.
1541       // Alter data node to use pre-phi inputs
1542       if (this->in(0) == region) {
1543         x->set_req(0, in);
1544       } else {
1545         x->set_req(0, NULL);
1546       }
1547       if (mem->is_Phi() && (mem->in(0) == region)) {
1548         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1549       }
1550       if (address->is_Phi() && address->in(0) == region) {
1551         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1552       }
1553       if (base_is_phi && (base->in(0) == region)) {
1554         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1555         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1556         x->set_req(Address, adr_x);
1557       }
1558     }
1559     // Check for a 'win' on some paths
1560     const Type *t = x->Value(igvn);
1561 
1562     bool singleton = t->singleton();
1563 
1564     // See comments in PhaseIdealLoop::split_thru_phi().
1565     if (singleton && t == Type::TOP) {
1566       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1567     }
1568 
1569     if (singleton) {
1570       x = igvn->makecon(t);
1571     } else {
1572       // We now call Identity to try to simplify the cloned node.
1573       // Note that some Identity methods call phase->type(this).
1574       // Make sure that the type array is big enough for
1575       // our new node, even though we may throw the node away.
1576       // (This tweaking with igvn only works because x is a new node.)
1577       igvn->set_type(x, t);
1578       // If x is a TypeNode, capture any more-precise type permanently into Node
1579       // otherwise it will be not updated during igvn->transform since
1580       // igvn->type(x) is set to x->Value() already.
1581       x->raise_bottom_type(t);
1582       Node* y = x->Identity(igvn);
1583       if (y != x) {
1584         x = y;
1585       } else {
1586         y = igvn->hash_find_insert(x);
1587         if (y) {
1588           x = y;
1589         } else {
1590           // Else x is a new node we are keeping
1591           // We do not need register_new_node_with_optimizer
1592           // because set_type has already been called.
1593           igvn->_worklist.push(x);
1594         }
1595       }
1596     }
1597     if (x != the_clone && the_clone != NULL) {
1598       igvn->remove_dead_node(the_clone);
1599     }
1600     phi->set_req(i, x);
1601   }
1602   // Record Phi
1603   igvn->register_new_node_with_optimizer(phi);
1604   return phi;
1605 }

1623 //------------------------------Ideal------------------------------------------
1624 // If the load is from Field memory and the pointer is non-null, it might be possible to
1625 // zero out the control input.
1626 // If the offset is constant and the base is an object allocation,
1627 // try to hook me up to the exact initializing store.
1628 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1629   Node* p = MemNode::Ideal_common(phase, can_reshape);
1630   if (p)  return (p == NodeSentinel) ? NULL : p;
1631 
1632   Node* ctrl    = in(MemNode::Control);
1633   Node* address = in(MemNode::Address);
1634   bool progress = false;
1635 
1636   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1637          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1638 
1639   // Skip up past a SafePoint control.  Cannot do this for Stores because
1640   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1641   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1642       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1643       !addr_mark &&
1644       (depends_only_on_test() || has_unknown_control_dependency())) {
1645     ctrl = ctrl->in(0);
1646     set_req(MemNode::Control,ctrl);
1647     progress = true;
1648   }
1649 
1650   intptr_t ignore = 0;
1651   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1652   if (base != NULL
1653       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1654     // Check for useless control edge in some common special cases
1655     if (in(MemNode::Control) != NULL
1656         && can_remove_control()
1657         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1658         && all_controls_dominate(base, phase->C->start())) {
1659       // A method-invariant, non-null address (constant or 'this' argument).
1660       set_req(MemNode::Control, NULL);
1661       progress = true;
1662     }
1663   }
1664 
1665   Node* mem = in(MemNode::Memory);
1666   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1667 
1668   if (can_reshape && (addr_t != NULL)) {
1669     // try to optimize our memory input
1670     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1671     if (opt_mem != mem) {
1672       set_req(MemNode::Memory, opt_mem);
1673       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1674       return this;
1675     }
1676     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1677     if ((t_oop != NULL) &&
1678         (t_oop->is_known_instance_field() ||
1679          t_oop->is_ptr_to_boxed_value())) {
1680       PhaseIterGVN *igvn = phase->is_IterGVN();
1681       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1682         // Delay this transformation until memory Phi is processed.
1683         phase->is_IterGVN()->_worklist.push(this);
1684         return NULL;
1685       }
1686       // Split instance field load through Phi.
1687       Node* result = split_through_phi(phase);
1688       if (result != NULL) return result;
1689 
1690       if (t_oop->is_ptr_to_boxed_value()) {
1691         Node* result = eliminate_autobox(phase);
1692         if (result != NULL) return result;
1693       }
1694     }
1695   }
1696 
1697   // Is there a dominating load that loads the same value?  Leave
1698   // anything that is not a load of a field/array element (like
1699   // barriers etc.) alone
1700   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1701     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1702       Node *use = mem->fast_out(i);
1703       if (use != this &&
1704           use->Opcode() == Opcode() &&
1705           use->in(0) != NULL &&
1706           use->in(0) != in(0) &&
1707           use->in(Address) == in(Address)) {
1708         Node* ctl = in(0);
1709         for (int i = 0; i < 10 && ctl != NULL; i++) {
1710           ctl = IfNode::up_one_dom(ctl);
1711           if (ctl == use->in(0)) {
1712             set_req(0, use->in(0));
1713             return this;
1714           }
1715         }
1716       }
1717     }
1718   }
1719 
1720   // Check for prior store with a different base or offset; make Load
1721   // independent.  Skip through any number of them.  Bail out if the stores
1722   // are in an endless dead cycle and report no progress.  This is a key
1723   // transform for Reflection.  However, if after skipping through the Stores
1724   // we can't then fold up against a prior store do NOT do the transform as
1725   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1726   // array memory alive twice: once for the hoisted Load and again after the
1727   // bypassed Store.  This situation only works if EVERYBODY who does
1728   // anti-dependence work knows how to bypass.  I.e. we need all
1729   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1730   // the alias index stuff.  So instead, peek through Stores and IFF we can
1731   // fold up, do so.
1732   Node* prev_mem = find_previous_store(phase);
1733   if (prev_mem != NULL) {
1734     Node* value = can_see_arraycopy_value(prev_mem, phase);
1735     if (value != NULL) {
1736       return value;
1737     }
1738   }
1739   // Steps (a), (b):  Walk past independent stores to find an exact match.
1740   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1741     // (c) See if we can fold up on the spot, but don't fold up here.
1742     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1743     // just return a prior value, which is done by Identity calls.
1744     if (can_see_stored_value(prev_mem, phase)) {
1745       // Make ready for step (d):
1746       set_req(MemNode::Memory, prev_mem);
1747       return this;
1748     }
1749   }
1750 
1751   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1752   if (alloc != NULL && mem->is_Proj() &&
1753       mem->in(0) != NULL &&
1754       mem->in(0) == alloc->initialization() &&
1755       Opcode() == Op_LoadX &&
1756       alloc->initialization()->proj_out_or_null(0) != NULL) {
1757     InitializeNode* init = alloc->initialization();
1758     Node* control = init->proj_out(0);
1759     return alloc->make_ideal_mark(phase, control, mem);
1760   }
1761 
1762   return progress ? this : NULL;
1763 }

1765 // Helper to recognize certain Klass fields which are invariant across
1766 // some group of array types (e.g., int[] or all T[] where T < Object).
1767 const Type*
1768 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1769                                  ciKlass* klass) const {
1770   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1771     // The field is Klass::_modifier_flags.  Return its (constant) value.
1772     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1773     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1774     return TypeInt::make(klass->modifier_flags());
1775   }
1776   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1777     // The field is Klass::_access_flags.  Return its (constant) value.
1778     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1779     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1780     return TypeInt::make(klass->access_flags());
1781   }
1782   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1783     // The field is Klass::_layout_helper.  Return its constant value if known.
1784     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1785     return TypeInt::make(klass->layout_helper());
1786   }
1787 
1788   // No match.
1789   return NULL;
1790 }

1792 //------------------------------Value-----------------------------------------
1793 const Type* LoadNode::Value(PhaseGVN* phase) const {
1794   // Either input is TOP ==> the result is TOP
1795   Node* mem = in(MemNode::Memory);
1796   const Type *t1 = phase->type(mem);
1797   if (t1 == Type::TOP)  return Type::TOP;
1798   Node* adr = in(MemNode::Address);
1799   const TypePtr* tp = phase->type(adr)->isa_ptr();
1800   if (tp == NULL || tp->empty())  return Type::TOP;
1801   int off = tp->offset();
1802   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1803   Compile* C = phase->C;
1804 
1805   // Try to guess loaded type from pointer type
1806   if (tp->isa_aryptr()) {
1807     const TypeAryPtr* ary = tp->is_aryptr();
1808     const Type* t = ary->elem();
1809 
1810     // Determine whether the reference is beyond the header or not, by comparing
1811     // the offset against the offset of the start of the array's data.
1812     // Different array types begin at slightly different offsets (12 vs. 16).
1813     // We choose T_BYTE as an example base type that is least restrictive
1814     // as to alignment, which will therefore produce the smallest
1815     // possible base offset.
1816     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1817     const bool off_beyond_header = (off >= min_base_off);
1818 
1819     // Try to constant-fold a stable array element.
1820     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1821       // Make sure the reference is not into the header and the offset is constant
1822       ciObject* aobj = ary->const_oop();
1823       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1824         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1825         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1826                                                                       stable_dimension,
1827                                                                       memory_type(), is_unsigned());
1828         if (con_type != NULL) {
1829           return con_type;
1830         }
1831       }
1832     }
1833 
1834     // Don't do this for integer types. There is only potential profit if
1835     // the element type t is lower than _type; that is, for int types, if _type is
1836     // more restrictive than t.  This only happens here if one is short and the other
1837     // char (both 16 bits), and in those cases we've made an intentional decision
1838     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1839     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1840     //
1841     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1842     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1843     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1844     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1845     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1846     // In fact, that could have been the original type of p1, and p1 could have
1847     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1848     // expression (LShiftL quux 3) independently optimized to the constant 8.
1849     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1850         && (_type->isa_vect() == NULL)
1851         && t->isa_inlinetype() == NULL
1852         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1853       // t might actually be lower than _type, if _type is a unique
1854       // concrete subclass of abstract class t.
1855       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1856         const Type* jt = t->join_speculative(_type);
1857         // In any case, do not allow the join, per se, to empty out the type.
1858         if (jt->empty() && !t->empty()) {
1859           // This can happen if a interface-typed array narrows to a class type.
1860           jt = _type;
1861         }
1862 #ifdef ASSERT
1863         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1864           // The pointers in the autobox arrays are always non-null
1865           Node* base = adr->in(AddPNode::Base);
1866           if ((base != NULL) && base->is_DecodeN()) {
1867             // Get LoadN node which loads IntegerCache.cache field
1868             base = base->in(1);
1869           }
1870           if ((base != NULL) && base->is_Con()) {
1871             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1872             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1873               // It could be narrow oop
1874               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1875             }
1876           }
1877         }
1878 #endif
1879         return jt;
1880       }
1881     }
1882   } else if (tp->base() == Type::InstPtr) {
1883     assert( off != Type::OffsetBot ||
1884             // arrays can be cast to Objects
1885             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1886             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1887             // unsafe field access may not have a constant offset
1888             C->has_unsafe_access(),
1889             "Field accesses must be precise" );
1890     // For oop loads, we expect the _type to be precise.
1891 
1892     const TypeInstPtr* tinst = tp->is_instptr();
1893     BasicType bt = memory_type();
1894 
1895     // Optimize loads from constant fields.
1896     ciObject* const_oop = tinst->const_oop();
1897     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1898       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1899       if (mirror_type != NULL && mirror_type->is_inlinetype()) {
1900         ciInlineKlass* vk = mirror_type->as_inline_klass();
1901         if (off == vk->default_value_offset()) {
1902           // Loading a special hidden field that contains the oop of the default inline type
1903           const Type* const_oop = TypeInstPtr::make(vk->default_instance());
1904           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1905         }
1906       }
1907       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1908       if (con_type != NULL) {
1909         return con_type;
1910       }
1911     }
1912   } else if (tp->base() == Type::KlassPtr) {
1913     assert( off != Type::OffsetBot ||
1914             // arrays can be cast to Objects
1915             tp->is_klassptr()->klass() == NULL ||
1916             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1917             // also allow array-loading from the primary supertype
1918             // array during subtype checks
1919             Opcode() == Op_LoadKlass,
1920             "Field accesses must be precise" );
1921     // For klass/static loads, we expect the _type to be precise
1922   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1923     if (adr->is_Load() && off == 0) {
1924       /* With mirrors being an indirect in the Klass*
1925        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1926        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1927        *
1928        * So check the type and klass of the node before the LoadP.
1929        */
1930       Node* adr2 = adr->in(MemNode::Address);
1931       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1932       if (tkls != NULL) {
1933         ciKlass* klass = tkls->klass();
1934         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1935           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1936           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1937           return TypeInstPtr::make(klass->java_mirror());
1938         }
1939       }
1940     } else {
1941       // Check for a load of the default value offset from the InlineKlassFixedBlock:
1942       // LoadI(LoadP(inline_klass, adr_inlineklass_fixed_block_offset), default_value_offset_offset)
1943       intptr_t offset = 0;
1944       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1945       if (base != NULL && base->is_Load() && offset == in_bytes(InlineKlass::default_value_offset_offset())) {
1946         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1947         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_inlinetype() &&
1948             tkls->offset() == in_bytes(InstanceKlass::adr_inlineklass_fixed_block_offset())) {
1949           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1950           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1951           return TypeInt::make(tkls->klass()->as_inline_klass()->default_value_offset());
1952         }
1953       }
1954     }
1955   }
1956 
1957   const TypeKlassPtr *tkls = tp->isa_klassptr();
1958   if (tkls != NULL && !StressReflectiveCode) {
1959     ciKlass* klass = tkls->klass();
1960     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1961       // We are loading a field from a Klass metaobject whose identity
1962       // is known at compile time (the type is "exact" or "precise").
1963       // Check for fields we know are maintained as constants by the VM.
1964       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1965         // The field is Klass::_super_check_offset.  Return its (constant) value.
1966         // (Folds up type checking code.)
1967         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1968         return TypeInt::make(klass->super_check_offset());
1969       }
1970       // Compute index into primary_supers array
1971       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1972       // Check for overflowing; use unsigned compare to handle the negative case.
1973       if( depth < ciKlass::primary_super_limit() ) {
1974         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1975         // (Folds up type checking code.)
1976         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1977         ciKlass *ss = klass->super_of_depth(depth);
1978         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1979       }
1980       const Type* aift = load_array_final_field(tkls, klass);
1981       if (aift != NULL)  return aift;
1982     }
1983 
1984     // We can still check if we are loading from the primary_supers array at a
1985     // shallow enough depth.  Even though the klass is not exact, entries less
1986     // than or equal to its super depth are correct.
1987     if (tkls->is_loaded()) {
1988       ciType *inner = klass;
1989       while( inner->is_obj_array_klass() )
1990         inner = inner->as_obj_array_klass()->base_element_type();
1991       if( inner->is_instance_klass() &&
1992           !inner->as_instance_klass()->flags().is_interface() ) {
1993         // Compute index into primary_supers array
1994         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1995         // Check for overflowing; use unsigned compare to handle the negative case.
1996         if( depth < ciKlass::primary_super_limit() &&
1997             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1998           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1999           // (Folds up type checking code.)
2000           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
2001           ciKlass *ss = klass->super_of_depth(depth);
2002           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
2003         }
2004       }
2005     }
2006 
2007     // If the type is enough to determine that the thing is not an array,
2008     // we can give the layout_helper a positive interval type.
2009     // This will help short-circuit some reflective code.
2010     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2011         && !klass->is_array_klass() // not directly typed as an array
2012         && !klass->is_interface()  // specifically not Serializable & Cloneable
2013         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2014         ) {
2015       // Note:  When interfaces are reliable, we can narrow the interface
2016       // test to (klass != Serializable && klass != Cloneable).
2017       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2018       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2019       // The key property of this type is that it folds up tests
2020       // for array-ness, since it proves that the layout_helper is positive.
2021       // Thus, a generic value like the basic object layout helper works fine.
2022       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2023     }
2024   }
2025 
2026   // If we are loading from a freshly-allocated object, produce a zero,
2027   // if the load is provably beyond the header of the object.
2028   // (Also allow a variable load from a fresh array to produce zero.)
2029   const TypeOopPtr *tinst = tp->isa_oopptr();
2030   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2031   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
2032   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
2033     Node* value = can_see_stored_value(mem,phase);
2034     if (value != NULL && value->is_Con()) {
2035       assert(value->bottom_type()->higher_equal(_type),"sanity");
2036       return value->bottom_type();
2037     }
2038   }
2039 
2040   if (is_instance) {
2041     // If we have an instance type and our memory input is the
2042     // programs's initial memory state, there is no matching store,
2043     // so just return a zero of the appropriate type
2044     Node *mem = in(MemNode::Memory);
2045     if (mem->is_Parm() && mem->in(0)->is_Start()) {
2046       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
2047       return Type::get_zero_type(_type->basic_type());
2048     }
2049   }

2228       tp->ptr() == TypePtr::Null)  return Type::TOP;
2229

2231   const TypeInstPtr *tinst = tp->isa_instptr();

2233     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
2234     int offset = tinst->offset();
2235     if (ik == phase->C->env()->Class_klass()
2236         && (offset == java_lang_Class::klass_offset() ||
2237             offset == java_lang_Class::array_klass_offset())) {
2238       // We are loading a special hidden field from a Class mirror object,
2239       // the field which points to the VM's Klass metaobject.
2240       ciType* t = tinst->java_mirror_type();
2241       // java_mirror_type returns non-null for compile-time Class constants.
2242       if (t != NULL) {
2243         // constant oop => constant klass
2244         if (offset == java_lang_Class::array_klass_offset()) {
2245           if (t->is_void()) {
2246             // We cannot create a void array.  Since void is a primitive type return null

2250           return TypeKlassPtr::make(ciArrayKlass::make(t));
2251         }

2255         }

2257         return TypeKlassPtr::make(t->as_klass());
2258       }
2259       // non-constant mirror, so we can't tell what's going on
2260     }

2262       return _type;             // Bail out if not loaded
2263     if (offset == oopDesc::klass_offset_in_bytes()) {
2264       if (tinst->klass_is_exact()) {

2273           // %%% should use stronger assert_unique_concrete_subtype instead
2274           phase->C->dependencies()->assert_leaf_type(ik);

2277         return TypeKlassPtr::make(ik);
2278       }

2282     }
2283   }
2284

2286   const TypeAryPtr *tary = tp->isa_aryptr();

2293       }

2296       // array component klass.

2303           if (!ik->is_interface() && !ik->has_subklass()) {
2304             //assert(!UseExactTypes, "this code should be useless with exact types");
2305             // Add a dependence; if any subclass added we need to recompile
2306             if (!ik->is_final()) {
2307               phase->C->dependencies()->assert_leaf_type(ik);
2308             }

2310             return TypeKlassPtr::make(ak);

2312         }

2314       } else if (ak->is_type_array_klass()) {
2315         //assert(!UseExactTypes, "this code should be useless with exact types");

2317       }

2332       // // e.g., element type could be object and array may contain strings
2333       // return TypeKlassPtr::make(TypePtr::Constant, elem, 0);
2334

2336       // according to the element type's subclassing.

2339                tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2340       ciKlass* elem = klass->as_flat_array_klass()->element_klass();

2363 Node* LoadNode::klass_identity_common(PhaseGVN* phase) {
2364   Node* x = LoadNode::Identity(phase);
2365   if (x != this)  return x;
2366 
2367   // Take apart the address into an oop and and offset.
2368   // Return 'this' if we cannot.
2369   Node*    adr    = in(MemNode::Address);
2370   intptr_t offset = 0;
2371   Node*    base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
2372   if (base == NULL)     return this;
2373   const TypeOopPtr* toop = phase->type(adr)->isa_oopptr();
2374   if (toop == NULL)     return this;
2375 
2376   // Step over potential GC barrier for OopHandle resolve
2377   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2378   if (bs->is_gc_barrier_node(base)) {
2379     base = bs->step_over_gc_barrier(base);
2380   }
2381 
2382   // We can fetch the klass directly through an AllocateNode.
2383   // This works even if the klass is not constant (clone or newArray).
2384   if (offset == oopDesc::klass_offset_in_bytes()) {
2385     Node* allocated_klass = AllocateNode::Ideal_klass(base, phase);
2386     if (allocated_klass != NULL) {
2387       return allocated_klass;
2388     }
2389   }
2390 
2391   // Simplify k.java_mirror.as_klass to plain k, where k is a Klass*.
2392   // See inline_native_Class_query for occurrences of these patterns.
2393   // Java Example:  x.getClass().isAssignableFrom(y)
2394   //
2395   // This improves reflective code, often making the Class
2396   // mirror go completely dead.  (Current exception:  Class
2397   // mirrors may appear in debug info, but we could clean them out by
2398   // introducing a new debug info operator for Klass.java_mirror).
2399 
2400   if (toop->isa_instptr() && toop->klass() == phase->C->env()->Class_klass()
2401       && offset == java_lang_Class::klass_offset()) {
2402     if (base->is_Load()) {
2403       Node* base2 = base->in(MemNode::Address);
2404       if (base2->is_Load()) { /* direct load of a load which is the OopHandle */
2405         Node* adr2 = base2->in(MemNode::Address);
2406         const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
2407         if (tkls != NULL && !tkls->empty()
2408             && (tkls->klass()->is_instance_klass() ||
2409               tkls->klass()->is_array_klass())
2410             && adr2->is_AddP()
2411            ) {
2412           int mirror_field = in_bytes(Klass::java_mirror_offset());
2413           if (tkls->offset() == mirror_field) {
2414             return adr2->in(AddPNode::Base);
2415           }
2416         }
2417       }
2418     }
2419   }
2420 
2421   return this;
2422 }

2596 //------------------------------Ideal------------------------------------------
2597 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2598 // When a store immediately follows a relevant allocation/initialization,
2599 // try to capture it into the initialization, or hoist it above.
2600 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2601   Node* p = MemNode::Ideal_common(phase, can_reshape);
2602   if (p)  return (p == NodeSentinel) ? NULL : p;
2603 
2604   Node* mem     = in(MemNode::Memory);
2605   Node* address = in(MemNode::Address);
2606   // Back-to-back stores to same address?  Fold em up.  Generally
2607   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2608   // since they must follow each StoreP operation.  Redundant StoreCMs
2609   // are eliminated just before matching in final_graph_reshape.
2610   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::INLINES) {
2611     Node* st = mem;
2612     // If Store 'st' has more than one use, we cannot fold 'st' away.
2613     // For example, 'st' might be the final state at a conditional
2614     // return.  Or, 'st' might be used by some node which is live at
2615     // the same time 'st' is live, which might be unschedulable.  So,
2616     // require exactly ONE user until such time as we clone 'mem' for
2617     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2618     // true).
2619     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2620       // Looking at a dead closed cycle of memory?
2621       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2622       assert(Opcode() == st->Opcode() ||
2623              st->Opcode() == Op_StoreVector ||
2624              Opcode() == Op_StoreVector ||
2625              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2626              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2627              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2628              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2629              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2630              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2631 
2632       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2633           st->as_Store()->memory_size() <= this->memory_size()) {
2634         Node* use = st->raw_out(0);
2635         phase->igvn_rehash_node_delayed(use);
2636         if (can_reshape) {
2637           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2638         } else {
2639           // It's OK to do this in the parser, since DU info is always accurate,
2640           // and the parser always refers to nodes via SafePointNode maps.
2641           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2642         }
2643         return this;
2644       }
2645       st = st->in(MemNode::Memory);
2646     }
2647   }
2648 
2649 
2650   // Capture an unaliased, unconditional, simple store into an initializer.
2651   // Or, if it is independent of the allocation, hoist it above the allocation.
2652   if (ReduceFieldZeroing && /*can_reshape &&*/
2653       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2654     InitializeNode* init = mem->in(0)->as_Initialize();
2655     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2656     if (offset > 0) {
2657       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2658       // If the InitializeNode captured me, it made a raw copy of me,
2659       // and I need to disappear.
2660       if (moved != NULL) {
2661         // %%% hack to ensure that Ideal returns a new node:
2662         mem = MergeMemNode::make(mem);
2663         return mem;             // fold me away
2664       }
2665     }
2666   }
2667 
2668   return NULL;                  // No further progress
2669 }

2683 //------------------------------Identity---------------------------------------
2684 // Remove redundant stores:
2685 //   Store(m, p, Load(m, p)) changes to m.
2686 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2687 Node* StoreNode::Identity(PhaseGVN* phase) {
2688   Node* mem = in(MemNode::Memory);
2689   Node* adr = in(MemNode::Address);
2690   Node* val = in(MemNode::ValueIn);
2691 
2692   Node* result = this;
2693 
2694   // Load then Store?  Then the Store is useless
2695   if (val->is_Load() &&
2696       val->in(MemNode::Address)->eqv_uncast(adr) &&
2697       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2698       val->as_Load()->store_Opcode() == Opcode()) {
2699     result = mem;
2700   }
2701 
2702   // Two stores in a row of the same value?
2703   if (result == this &&
2704       mem->is_Store() &&
2705       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2706       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2707       mem->Opcode() == Opcode()) {
2708     result = mem;
2709   }
2710 
2711   // Store of zero anywhere into a freshly-allocated object?
2712   // Then the store is useless.
2713   // (It must already have been captured by the InitializeNode.)
2714   if (result == this && ReduceFieldZeroing) {
2715     // a newly allocated object is already all-zeroes everywhere
2716     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2717         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2718       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to inline type array is forbidden");
2719       result = mem;
2720     }
2721 
2722     if (result == this) {
2723       // the store may also apply to zero-bits in an earlier object
2724       Node* prev_mem = find_previous_store(phase);
2725       // Steps (a), (b):  Walk past independent stores to find an exact match.
2726       if (prev_mem != NULL) {
2727         Node* prev_val = can_see_stored_value(prev_mem, phase);
2728         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2729           // prev_val and val might differ by a cast; it would be good
2730           // to keep the more informative of the two.
2731           if (phase->type(val)->is_zero_type()) {
2732             result = mem;
2733           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2734             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2735             AllocateNode* alloc = init->allocation();
2736             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2737               result = mem;
2738             }
2739           }
2740         }
2741       }
2742     }
2743   }
2744 
2745   if (result != this && phase->is_IterGVN() != NULL) {
2746     MemBarNode* trailing = trailing_membar();
2747     if (trailing != NULL) {
2748 #ifdef ASSERT
2749       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2750       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2751 #endif
2752       PhaseIterGVN* igvn = phase->is_IterGVN();
2753       trailing->remove(igvn);
2754     }
2755   }
2756 
2757   return result;
2758 }

3027 //------------------------------Idealize---------------------------------------
3028 // Clearing a short array is faster with stores
3029 Node *ClearArrayNode::Ideal(PhaseGVN *phase, bool can_reshape) {
3030   // Already know this is a large node, do not try to ideal it
3031   if (!IdealizeClearArrayNode || _is_large) return NULL;
3032 
3033   const int unit = BytesPerLong;
3034   const TypeX* t = phase->type(in(2))->isa_intptr_t();
3035   if (!t)  return NULL;
3036   if (!t->is_con())  return NULL;
3037   intptr_t raw_count = t->get_con();
3038   intptr_t size = raw_count;
3039   if (!Matcher::init_array_count_is_in_bytes) size *= unit;
3040   // Clearing nothing uses the Identity call.
3041   // Negative clears are possible on dead ClearArrays
3042   // (see jck test stmt114.stmt11402.val).
3043   if (size <= 0 || size % unit != 0)  return NULL;
3044   intptr_t count = size / unit;
3045   // Length too long; communicate this to matchers and assemblers.
3046   // Assemblers are responsible to produce fast hardware clears for it.
3047   if (size > InitArrayShortSize) {
3048     return new ClearArrayNode(in(0), in(1), in(2), in(3), in(4), true);
3049   }
3050   Node *mem = in(1);
3051   if( phase->type(mem)==Type::TOP ) return NULL;
3052   Node *adr = in(3);
3053   const Type* at = phase->type(adr);
3054   if( at==Type::TOP ) return NULL;
3055   const TypePtr* atp = at->isa_ptr();
3056   // adjust atp to be the correct array element address type
3057   if (atp == NULL)  atp = TypePtr::BOTTOM;
3058   else              atp = atp->add_offset(Type::OffsetBot);
3059   // Get base for derived pointer purposes
3060   if( adr->Opcode() != Op_AddP ) Unimplemented();
3061   Node *base = adr->in(1);
3062 
3063   Node *val = in(4);
3064   Node *off  = phase->MakeConX(BytesPerLong);
3065   mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3066   count--;
3067   while( count-- ) {
3068     mem = phase->transform(mem);
3069     adr = phase->transform(new AddPNode(base,adr,off));
3070     mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3071   }
3072   return mem;
3073 }

3075 //----------------------------step_through----------------------------------
3076 // Return allocation input memory edge if it is different instance
3077 // or itself if it is the one we are looking for.
3078 bool ClearArrayNode::step_through(Node** np, uint instance_id, PhaseTransform* phase) {
3079   Node* n = *np;
3080   assert(n->is_ClearArray(), "sanity");
3081   intptr_t offset;
3082   AllocateNode* alloc = AllocateNode::Ideal_allocation(n->in(3), phase, offset);
3083   // This method is called only before Allocate nodes are expanded
3084   // during macro nodes expansion. Before that ClearArray nodes are
3085   // only generated in PhaseMacroExpand::generate_arraycopy() (before
3086   // Allocate nodes are expanded) which follows allocations.
3087   assert(alloc != NULL, "should have allocation");
3088   if (alloc->_idx == instance_id) {
3089     // Can not bypass initialization of the instance we are looking for.
3090     return false;
3091   }
3092   // Otherwise skip it.
3093   InitializeNode* init = alloc->initialization();
3094   if (init != NULL)

3472 //===========================InitializeNode====================================
3473 // SUMMARY:
3474 // This node acts as a memory barrier on raw memory, after some raw stores.
3475 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3476 // The Initialize can 'capture' suitably constrained stores as raw inits.
3477 // It can coalesce related raw stores into larger units (called 'tiles').
3478 // It can avoid zeroing new storage for memory units which have raw inits.
3479 // At macro-expansion, it is marked 'complete', and does not optimize further.
3480 //
3481 // EXAMPLE:
3482 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3483 //   ctl = incoming control; mem* = incoming memory
3484 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3485 // First allocate uninitialized memory and fill in the header:
3486 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3487 //   ctl := alloc.Control; mem* := alloc.Memory*
3488 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3489 // Then initialize to zero the non-header parts of the raw memory block:
3490 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3491 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3492 // After the initialize node executes, the object is ready for service:
3493 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3494 // Suppose its body is immediately initialized as {1,2}:
3495 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3496 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3497 //   mem.SLICE(#short[*]) := store2
3498 //
3499 // DETAILS:
3500 // An InitializeNode collects and isolates object initialization after
3501 // an AllocateNode and before the next possible safepoint.  As a
3502 // memory barrier (MemBarNode), it keeps critical stores from drifting
3503 // down past any safepoint or any publication of the allocation.
3504 // Before this barrier, a newly-allocated object may have uninitialized bits.
3505 // After this barrier, it may be treated as a real oop, and GC is allowed.
3506 //
3507 // The semantics of the InitializeNode include an implicit zeroing of
3508 // the new object from object header to the end of the object.
3509 // (The object header and end are determined by the AllocateNode.)
3510 //
3511 // Certain stores may be added as direct inputs to the InitializeNode.
3512 // These stores must update raw memory, and they must be to addresses
3513 // derived from the raw address produced by AllocateNode, and with
3514 // a constant offset.  They must be ordered by increasing offset.
3515 // The first one is at in(RawStores), the last at in(req()-1).
3516 // Unlike most memory operations, they are not linked in a chain,
3517 // but are displayed in parallel as users of the rawmem output of
3518 // the allocation.
3519 //
3520 // (See comments in InitializeNode::capture_store, which continue
3521 // the example given above.)
3522 //
3523 // When the associated Allocate is macro-expanded, the InitializeNode
3524 // may be rewritten to optimize collected stores.  A ClearArrayNode
3525 // may also be created at that point to represent any required zeroing.
3526 // The InitializeNode is then marked 'complete', prohibiting further
3527 // capturing of nearby memory operations.
3528 //
3529 // During macro-expansion, all captured initializations which store
3530 // constant values of 32 bits or smaller are coalesced (if advantageous)
3531 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3532 // initialized in fewer memory operations.  Memory words which are
3533 // covered by neither tiles nor non-constant stores are pre-zeroed
3534 // by explicit stores of zero.  (The code shape happens to do all
3535 // zeroing first, then all other stores, with both sequences occurring
3536 // in order of ascending offsets.)
3537 //
3538 // Alternatively, code may be inserted between an AllocateNode and its
3539 // InitializeNode, to perform arbitrary initialization of the new object.
3540 // E.g., the object copying intrinsics insert complex data transfers here.
3541 // The initialization must then be marked as 'complete' disable the
3542 // built-in zeroing semantics and the collection of initializing stores.
3543 //
3544 // While an InitializeNode is incomplete, reads from the memory state
3545 // produced by it are optimizable if they match the control edge and
3546 // new oop address associated with the allocation/initialization.
3547 // They return a stored value (if the offset matches) or else zero.
3548 // A write to the memory state, if it matches control and address,
3549 // and if it is to a constant offset, may be 'captured' by the
3550 // InitializeNode.  It is cloned as a raw memory operation and rewired
3551 // inside the initialization, to the raw oop produced by the allocation.
3552 // Operations on addresses which are provably distinct (e.g., to
3553 // other AllocateNodes) are allowed to bypass the initialization.
3554 //
3555 // The effect of all this is to consolidate object initialization
3556 // (both arrays and non-arrays, both piecewise and bulk) into a
3557 // single location, where it can be optimized as a unit.
3558 //
3559 // Only stores with an offset less than TrackedInitializationLimit words
3560 // will be considered for capture by an InitializeNode.  This puts a
3561 // reasonable limit on the complexity of optimized initializations.
3562

3655 // Helper for proving that an initialization expression is
3656 // "simple enough" to be folded into an object initialization.
3657 // Attempts to prove that a store's initial value 'n' can be captured
3658 // within the initialization without creating a vicious cycle, such as:
3659 //     { Foo p = new Foo(); p.next = p; }
3660 // True for constants and parameters and small combinations thereof.
3661 bool InitializeNode::detect_init_independence(Node* value, PhaseGVN* phase) {
3662   ResourceMark rm;
3663   Unique_Node_List worklist;
3664   worklist.push(value);
3665 
3666   uint complexity_limit = 20;
3667   for (uint j = 0; j < worklist.size(); j++) {
3668     if (j >= complexity_limit) {
3669       return false;  // Bail out if processed too many nodes
3670     }
3671 
3672     Node* n = worklist.at(j);
3673     if (n == NULL)      continue;   // (can this really happen?)
3674     if (n->is_Proj())   n = n->in(0);
3675     if (n == this)      return false;  // found a cycle
3676     if (n->is_Con())    continue;
3677     if (n->is_Start())  continue;   // params, etc., are OK
3678     if (n->is_Root())   continue;   // even better
3679 
3680     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3681     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3682       continue;
3683     }
3684 
3685     Node* ctl = n->in(0);
3686     if (ctl != NULL && !ctl->is_top()) {
3687       if (ctl->is_Proj())  ctl = ctl->in(0);
3688       if (ctl == this)  return false;
3689 
3690       // If we already know that the enclosing memory op is pinned right after
3691       // the init, then any control flow that the store has picked up
3692       // must have preceded the init, or else be equal to the init.
3693       // Even after loop optimizations (which might change control edges)
3694       // a store is never pinned *before* the availability of its inputs.
3695       if (!MemNode::all_controls_dominate(n, this))
3696         return false;                  // failed to prove a good control
3697     }
3698 
3699     // Check data edges for possible dependencies on 'this'.
3700     for (uint i = 1; i < n->req(); i++) {
3701       Node* m = n->in(i);
3702       if (m == NULL || m == n || m->is_top())  continue;
3703 
3704       // Only process data inputs once
3705       worklist.push(m);
3706     }
3707   }
3708 
3709   return true;
3710 }

3712 // Here are all the checks a Store must pass before it can be moved into
3713 // an initialization.  Returns zero if a check fails.
3714 // On success, returns the (constant) offset to which the store applies,
3715 // within the initialized memory.
3716 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3717   const int FAIL = 0;
3718   if (st->req() != MemNode::ValueIn + 1)
3719     return FAIL;                // an inscrutable StoreNode (card mark?)
3720   Node* ctl = st->in(MemNode::Control);
3721   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3722     return FAIL;                // must be unconditional after the initialization
3723   Node* mem = st->in(MemNode::Memory);
3724   if (!(mem->is_Proj() && mem->in(0) == this))
3725     return FAIL;                // must not be preceded by other stores
3726   Node* adr = st->in(MemNode::Address);
3727   intptr_t offset;
3728   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3729   if (alloc == NULL)
3730     return FAIL;                // inscrutable address
3731   if (alloc != allocation())
3732     return FAIL;                // wrong allocation!  (store needs to float up)
3733   int size_in_bytes = st->memory_size();
3734   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3735     return FAIL;                // mismatched access
3736   }
3737   Node* val = st->in(MemNode::ValueIn);
3738 
3739   if (!detect_init_independence(val, phase))
3740     return FAIL;                // stored value must be 'simple enough'
3741 
3742   // The Store can be captured only if nothing after the allocation
3743   // and before the Store is using the memory location that the store
3744   // overwrites.
3745   bool failed = false;
3746   // If is_complete_with_arraycopy() is true the shape of the graph is
3747   // well defined and is safe so no need for extra checks.
3748   if (!is_complete_with_arraycopy()) {
3749     // We are going to look at each use of the memory state following
3750     // the allocation to make sure nothing reads the memory that the
3751     // Store writes.
3752     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3753     int alias_idx = phase->C->get_alias_index(t_adr);
3754     ResourceMark rm;
3755     Unique_Node_List mems;
3756     mems.push(mem);
3757     Node* unique_merge = NULL;
3758     for (uint next = 0; next < mems.size(); ++next) {
3759       Node *m  = mems.at(next);
3760       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3761         Node *n = m->fast_out(j);
3762         if (n->outcnt() == 0) {
3763           continue;
3764         }
3765         if (n == st) {
3766           continue;
3767         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3768           // If the control of this use is different from the control
3769           // of the Store which is right after the InitializeNode then
3770           // this node cannot be between the InitializeNode and the
3771           // Store.
3772           continue;
3773         } else if (n->is_MergeMem()) {
3774           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3775             // We can hit a MergeMemNode (that will likely go away
3776             // later) that is a direct use of the memory state
3777             // following the InitializeNode on the same slice as the
3778             // store node that we'd like to capture. We need to check
3779             // the uses of the MergeMemNode.
3780             mems.push(n);
3781           }
3782         } else if (n->is_Mem()) {
3783           Node* other_adr = n->in(MemNode::Address);
3784           if (other_adr == adr) {
3785             failed = true;
3786             break;
3787           } else {
3788             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3789             if (other_t_adr != NULL) {
3790               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3791               if (other_alias_idx == alias_idx) {
3792                 // A load from the same memory slice as the store right
3793                 // after the InitializeNode. We check the control of the
3794                 // object/array that is loaded from. If it's the same as
3795                 // the store control then we cannot capture the store.
3796                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3797                 Node* base = other_adr;
3798                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3799                 base = base->in(AddPNode::Base);
3800                 if (base != NULL) {
3801                   base = base->uncast();
3802                   if (base->is_Proj() && base->in(0) == alloc) {
3803                     failed = true;
3804                     break;
3805                   }
3806                 }
3807               }
3808             }
3809           }
3810         } else {
3811           failed = true;
3812           break;
3813         }
3814       }
3815     }
3816   }
3817   if (failed) {
3818     if (!can_reshape) {
3819       // We decided we couldn't capture the store during parsing. We
3820       // should try again during the next IGVN once the graph is
3821       // cleaner.
3822       phase->C->record_for_igvn(st);
3823     }
3824     return FAIL;
3825   }
3826 
3827   return offset;                // success
3828 }

3830 // Find the captured store in(i) which corresponds to the range
3831 // [start..start+size) in the initialized object.
3832 // If there is one, return its index i.  If there isn't, return the
3833 // negative of the index where it should be inserted.
3834 // Return 0 if the queried range overlaps an initialization boundary
3835 // or if dead code is encountered.
3836 // If size_in_bytes is zero, do not bother with overlap checks.
3837 int InitializeNode::captured_store_insertion_point(intptr_t start,
3838                                                    int size_in_bytes,
3839                                                    PhaseTransform* phase) {
3840   const int FAIL = 0, MAX_STORE = BytesPerLong;
3841 
3842   if (is_complete())
3843     return FAIL;                // arraycopy got here first; punt
3844 
3845   assert(allocation() != NULL, "must be present");
3846 
3847   // no negatives, no header fields:
3848   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3849 
3850   // after a certain size, we bail out on tracking all the stores:
3851   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3852   if (start >= ti_limit)  return FAIL;
3853 
3854   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3855     if (i >= limit)  return -(int)i; // not found; here is where to put it
3856 
3857     Node*    st     = in(i);
3858     intptr_t st_off = get_store_offset(st, phase);
3859     if (st_off < 0) {
3860       if (st != zero_memory()) {
3861         return FAIL;            // bail out if there is dead garbage
3862       }
3863     } else if (st_off > start) {
3864       // ...we are done, since stores are ordered
3865       if (st_off < start + size_in_bytes) {
3866         return FAIL;            // the next store overlaps
3867       }
3868       return -(int)i;           // not found; here is where to put it
3869     } else if (st_off < start) {
3870       if (size_in_bytes != 0 &&
3871           start < st_off + MAX_STORE &&
3872           start < st_off + st->as_Store()->memory_size()) {
3873         return FAIL;            // the previous store overlaps
3874       }
3875     } else {
3876       if (size_in_bytes != 0 &&
3877           st->as_Store()->memory_size() != size_in_bytes) {
3878         return FAIL;            // mismatched store size
3879       }
3880       return i;
3881     }
3882 
3883     ++i;
3884   }
3885 }

3887 // Look for a captured store which initializes at the offset 'start'
3888 // with the given size.  If there is no such store, and no other
3889 // initialization interferes, then return zero_memory (the memory
3890 // projection of the AllocateNode).
3891 Node* InitializeNode::find_captured_store(intptr_t start, int size_in_bytes,
3892                                           PhaseTransform* phase) {

3899   } else {

3901     assert(get_store_offset(st->as_Store(), phase) == start, "sanity");
3902     return st;

3918 // Clone the given store, converting it into a raw store
3919 // initializing a field or element of my new object.
3920 // Caller is responsible for retiring the original store,
3921 // with subsume_node or the like.
3922 //
3923 // From the example above InitializeNode::InitializeNode,
3924 // here are the old stores to be captured:
3925 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3926 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3927 //
3928 // Here is the changed code; note the extra edges on init:
3929 //   alloc = (Allocate ...)
3930 //   rawoop = alloc.RawAddress
3931 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3932 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3933 //   init = (Initialize alloc.Control alloc.Memory rawoop
3934 //                      rawstore1 rawstore2)
3935 //
3936 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3937                                     PhaseGVN* phase, bool can_reshape) {
3938   assert(stores_are_sane(phase), "");
3939 
3940   if (start < 0)  return NULL;
3941   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3942 
3943   Compile* C = phase->C;
3944   int size_in_bytes = st->memory_size();
3945   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3946   if (i == 0)  return NULL;     // bail out
3947   Node* prev_mem = NULL;        // raw memory for the captured store
3948   if (i > 0) {
3949     prev_mem = in(i);           // there is a pre-existing store under this one
3950     set_req(i, C->top());       // temporarily disconnect it
3951     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3952   } else {
3953     i = -i;                     // no pre-existing store
3954     prev_mem = zero_memory();   // a slice of the newly allocated object
3955     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3956       set_req(--i, C->top());   // reuse this edge; it has been folded away
3957     else
3958       ins_req(i, C->top());     // build a new edge
3959   }
3960   Node* new_st = st->clone();
3961   new_st->set_req(MemNode::Control, in(Control));
3962   new_st->set_req(MemNode::Memory,  prev_mem);
3963   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3964   new_st = phase->transform(new_st);
3965 
3966   // At this point, new_st might have swallowed a pre-existing store
3967   // at the same offset, or perhaps new_st might have disappeared,
3968   // if it redundantly stored the same value (or zero to fresh memory).
3969 
3970   // In any case, wire it in:
3971   phase->igvn_rehash_node_delayed(this);
3972   set_req(i, new_st);
3973 
3974   // The caller may now kill the old guy.
3975   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3976   assert(check_st == new_st || check_st == NULL, "must be findable");
3977   assert(!is_complete(), "");
3978   return new_st;
3979 }

3981 static bool store_constant(jlong* tiles, int num_tiles,
3982                            intptr_t st_off, int st_size,
3983                            jlong con) {
3984   if ((st_off & (st_size-1)) != 0)
3985     return false;               // strange store offset (assume size==2**N)
3986   address addr = (address)tiles + st_off;
3987   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
3988   switch (st_size) {
3989   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
3990   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
3991   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
3992   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3993   default: return false;        // strange store size (detect size!=2**N here)
3994   }
3995   return true;                  // return success to caller
3996 }

3998 // Coalesce subword constants into int constants and possibly
3999 // into long constants.  The goal, if the CPU permits,
4000 // is to initialize the object with a small number of 64-bit tiles.
4001 // Also, convert floating-point constants to bit patterns.
4002 // Non-constants are not relevant to this pass.
4003 //
4004 // In terms of the running example on InitializeNode::InitializeNode
4005 // and InitializeNode::capture_store, here is the transformation
4006 // of rawstore1 and rawstore2 into rawstore12:
4007 //   alloc = (Allocate ...)
4008 //   rawoop = alloc.RawAddress
4009 //   tile12 = 0x00010002
4010 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4011 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4012 //
4013 void
4014 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4015                                         Node* size_in_bytes,
4016                                         PhaseGVN* phase) {
4017   Compile* C = phase->C;
4018 
4019   assert(stores_are_sane(phase), "");
4020   // Note:  After this pass, they are not completely sane,
4021   // since there may be some overlaps.
4022 
4023   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4024 
4025   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4026   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4027   size_limit = MIN2(size_limit, ti_limit);
4028   size_limit = align_up(size_limit, BytesPerLong);
4029   int num_tiles = size_limit / BytesPerLong;
4030 
4031   // allocate space for the tile map:
4032   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4033   jlong  tiles_buf[small_len];
4034   Node*  nodes_buf[small_len];
4035   jlong  inits_buf[small_len];
4036   jlong* tiles = ((num_tiles <= small_len) ? &tiles_buf[0]
4037                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4038   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
4039                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4040   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4041                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4042   // tiles: exact bitwise model of all primitive constants
4043   // nodes: last constant-storing node subsumed into the tiles model
4044   // inits: which bytes (in each tile) are touched by any initializations
4045 
4046   //// Pass A: Fill in the tile model with any relevant stores.
4047 
4048   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
4049   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
4050   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
4051   Node* zmem = zero_memory(); // initially zero memory state
4052   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4053     Node* st = in(i);
4054     intptr_t st_off = get_store_offset(st, phase);
4055 
4056     // Figure out the store's offset and constant value:
4057     if (st_off < header_size)             continue; //skip (ignore header)
4058     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
4059     int st_size = st->as_Store()->memory_size();
4060     if (st_off + st_size > size_limit)    break;
4061 
4062     // Record which bytes are touched, whether by constant or not.
4063     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
4064       continue;                 // skip (strange store size)
4065 
4066     const Type* val = phase->type(st->in(MemNode::ValueIn));
4067     if (!val->singleton())                continue; //skip (non-con store)
4068     BasicType type = val->basic_type();
4069 
4070     jlong con = 0;
4071     switch (type) {
4072     case T_INT:    con = val->is_int()->get_con();  break;
4073     case T_LONG:   con = val->is_long()->get_con(); break;
4074     case T_FLOAT:  con = jint_cast(val->getf());    break;
4075     case T_DOUBLE: con = jlong_cast(val->getd());   break;
4076     default:                              continue; //skip (odd store type)
4077     }
4078 
4079     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
4080         st->Opcode() == Op_StoreL) {
4081       continue;                 // This StoreL is already optimal.
4082     }
4083 
4084     // Store down the constant.
4085     store_constant(tiles, num_tiles, st_off, st_size, con);
4086 
4087     intptr_t j = st_off >> LogBytesPerLong;
4088 
4089     if (type == T_INT && st_size == BytesPerInt
4090         && (st_off & BytesPerInt) == BytesPerInt) {
4091       jlong lcon = tiles[j];
4092       if (!Matcher::isSimpleConstant64(lcon) &&
4093           st->Opcode() == Op_StoreI) {
4094         // This StoreI is already optimal by itself.
4095         jint* intcon = (jint*) &tiles[j];
4096         intcon[1] = 0;  // undo the store_constant()
4097 
4098         // If the previous store is also optimal by itself, back up and
4099         // undo the action of the previous loop iteration... if we can.
4100         // But if we can't, just let the previous half take care of itself.
4101         st = nodes[j];
4102         st_off -= BytesPerInt;
4103         con = intcon[0];
4104         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
4105           assert(st_off >= header_size, "still ignoring header");
4106           assert(get_store_offset(st, phase) == st_off, "must be");
4107           assert(in(i-1) == zmem, "must be");
4108           DEBUG_ONLY(const Type* tcon = phase->type(st->in(MemNode::ValueIn)));
4109           assert(con == tcon->is_int()->get_con(), "must be");
4110           // Undo the effects of the previous loop trip, which swallowed st:
4111           intcon[0] = 0;        // undo store_constant()
4112           set_req(i-1, st);     // undo set_req(i, zmem)
4113           nodes[j] = NULL;      // undo nodes[j] = st
4114           --old_subword;        // undo ++old_subword
4115         }
4116         continue;               // This StoreI is already optimal.
4117       }
4118     }
4119 
4120     // This store is not needed.
4121     set_req(i, zmem);
4122     nodes[j] = st;              // record for the moment
4123     if (st_size < BytesPerLong) // something has changed
4124           ++old_subword;        // includes int/float, but who's counting...
4125     else  ++old_long;
4126   }
4127 
4128   if ((old_subword + old_long) == 0)
4129     return;                     // nothing more to do
4130 
4131   //// Pass B: Convert any non-zero tiles into optimal constant stores.
4132   // Be sure to insert them before overlapping non-constant stores.
4133   // (E.g., byte[] x = { 1,2,y,4 }  =>  x[int 0] = 0x01020004, x[2]=y.)
4134   for (int j = 0; j < num_tiles; j++) {
4135     jlong con  = tiles[j];
4136     jlong init = inits[j];
4137     if (con == 0)  continue;
4138     jint con0,  con1;           // split the constant, address-wise
4139     jint init0, init1;          // split the init map, address-wise
4140     { union { jlong con; jint intcon[2]; } u;
4141       u.con = con;
4142       con0  = u.intcon[0];
4143       con1  = u.intcon[1];
4144       u.con = init;
4145       init0 = u.intcon[0];
4146       init1 = u.intcon[1];
4147     }
4148 
4149     Node* old = nodes[j];
4150     assert(old != NULL, "need the prior store");
4151     intptr_t offset = (j * BytesPerLong);
4152 
4153     bool split = !Matcher::isSimpleConstant64(con);
4154 
4155     if (offset < header_size) {
4156       assert(offset + BytesPerInt >= header_size, "second int counts");
4157       assert(*(jint*)&tiles[j] == 0, "junk in header");
4158       split = true;             // only the second word counts
4159       // Example:  int a[] = { 42 ... }
4160     } else if (con0 == 0 && init0 == -1) {
4161       split = true;             // first word is covered by full inits
4162       // Example:  int a[] = { ... foo(), 42 ... }
4163     } else if (con1 == 0 && init1 == -1) {
4164       split = true;             // second word is covered by full inits
4165       // Example:  int a[] = { ... 42, foo() ... }
4166     }
4167 
4168     // Here's a case where init0 is neither 0 nor -1:
4169     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4170     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4171     // In this case the tile is not split; it is (jlong)42.
4172     // The big tile is stored down, and then the foo() value is inserted.

4235 // Explore forward from in(start) to find the first fully initialized
4236 // word, and return its offset.  Skip groups of subword stores which
4237 // together initialize full words.  If in(start) is itself part of a
4238 // fully initialized word, return the offset of in(start).  If there
4239 // are no following full-word stores, or if something is fishy, return
4240 // a negative value.
4241 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4242   int       int_map = 0;
4243   intptr_t  int_map_off = 0;
4244   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4245 
4246   for (uint i = start, limit = req(); i < limit; i++) {
4247     Node* st = in(i);
4248 
4249     intptr_t st_off = get_store_offset(st, phase);
4250     if (st_off < 0)  break;  // return conservative answer
4251 
4252     int st_size = st->as_Store()->memory_size();
4253     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4254       return st_off;            // we found a complete word init
4255     }
4256 
4257     // update the map:
4258 
4259     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4260     if (this_int_off != int_map_off) {
4261       // reset the map:
4262       int_map = 0;
4263       int_map_off = this_int_off;
4264     }
4265 
4266     int subword_off = st_off - this_int_off;
4267     int_map |= right_n_bits(st_size) << subword_off;
4268     if ((int_map & FULL_MAP) == FULL_MAP) {
4269       return this_int_off;      // we found a complete word init
4270     }
4271 
4272     // Did this store hit or cross the word boundary?
4273     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4274     if (next_int_off == this_int_off + BytesPerInt) {
4275       // We passed the current int, without fully initializing it.
4276       int_map_off = next_int_off;
4277       int_map >>= BytesPerInt;
4278     } else if (next_int_off > this_int_off + BytesPerInt) {
4279       // We passed the current and next int.
4280       return this_int_off + BytesPerInt;
4281     }
4282   }
4283 
4284   return -1;
4285 }

4291 // activity as coherent as possible.
4292 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4293                                       intptr_t header_size,
4294                                       Node* size_in_bytes,
4295                                       PhaseIterGVN* phase) {
4296   assert(!is_complete(), "not already complete");
4297   assert(stores_are_sane(phase), "");
4298   assert(allocation() != NULL, "must be present");
4299 
4300   remove_extra_zeroes();
4301 
4302   if (ReduceFieldZeroing || ReduceBulkZeroing)
4303     // reduce instruction count for common initialization patterns
4304     coalesce_subword_stores(header_size, size_in_bytes, phase);
4305 
4306   Node* zmem = zero_memory();   // initially zero memory state
4307   Node* inits = zmem;           // accumulating a linearized chain of inits
4308   #ifdef ASSERT
4309   intptr_t first_offset = allocation()->minimum_header_size();
4310   intptr_t last_init_off = first_offset;  // previous init offset
4311   intptr_t last_init_end = first_offset;  // previous init offset+size
4312   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4313   #endif
4314   intptr_t zeroes_done = header_size;
4315 
4316   bool do_zeroing = true;       // we might give up if inits are very sparse
4317   int  big_init_gaps = 0;       // how many large gaps have we seen?
4318 
4319   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4320   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4321 
4322   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4323     Node* st = in(i);
4324     intptr_t st_off = get_store_offset(st, phase);
4325     if (st_off < 0)
4326       break;                    // unknown junk in the inits
4327     if (st->in(MemNode::Memory) != zmem)
4328       break;                    // complicated store chains somehow in list
4329 
4330     int st_size = st->as_Store()->memory_size();
4331     intptr_t next_init_off = st_off + st_size;
4332 
4333     if (do_zeroing && zeroes_done < next_init_off) {
4334       // See if this store needs a zero before it or under it.
4335       intptr_t zeroes_needed = st_off;
4336 
4337       if (st_size < BytesPerInt) {
4338         // Look for subword stores which only partially initialize words.
4339         // If we find some, we must lay down some word-level zeroes first,
4340         // underneath the subword stores.
4341         //
4342         // Examples:
4343         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4344         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4345         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4346         //
4347         // Note:  coalesce_subword_stores may have already done this,
4348         // if it was prompted by constant non-zero subword initializers.
4349         // But this case can still arise with non-constant stores.
4350 
4351         intptr_t next_full_store = find_next_fullword_store(i, phase);
4352 
4353         // In the examples above:
4354         //   in(i)          p   q   r   s     x   y     z
4355         //   st_off        12  13  14  15    12  13    14
4356         //   st_size        1   1   1   1     1   1     1
4357         //   next_full_s.  12  16  16  16    16  16    16
4358         //   z's_done      12  16  16  16    12  16    12
4359         //   z's_needed    12  16  16  16    16  16    16
4360         //   zsize          0   0   0   0     4   0     4
4361         if (next_full_store < 0) {
4362           // Conservative tack:  Zero to end of current word.
4363           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4364         } else {
4365           // Zero to beginning of next fully initialized word.
4366           // Or, don't zero at all, if we are already in that word.
4367           assert(next_full_store >= zeroes_needed, "must go forward");
4368           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4369           zeroes_needed = next_full_store;
4370         }
4371       }
4372 
4373       if (zeroes_needed > zeroes_done) {
4374         intptr_t zsize = zeroes_needed - zeroes_done;
4375         // Do some incremental zeroing on rawmem, in parallel with inits.
4376         zeroes_done = align_down(zeroes_done, BytesPerInt);
4377         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4378                                               allocation()->in(AllocateNode::DefaultValue),
4379                                               allocation()->in(AllocateNode::RawDefaultValue),
4380                                               zeroes_done, zeroes_needed,
4381                                               phase);
4382         zeroes_done = zeroes_needed;
4383         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4384           do_zeroing = false;   // leave the hole, next time
4385       }
4386     }
4387 
4388     // Collect the store and move on:
4389     phase->replace_input_of(st, MemNode::Memory, inits);
4390     inits = st;                 // put it on the linearized chain
4391     set_req(i, zmem);           // unhook from previous position
4392 
4393     if (zeroes_done == st_off)
4394       zeroes_done = next_init_off;
4395 
4396     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4397 
4398     #ifdef ASSERT
4399     // Various order invariants.  Weaker than stores_are_sane because
4400     // a large constant tile can be filled in by smaller non-constant stores.
4401     assert(st_off >= last_init_off, "inits do not reverse");
4402     last_init_off = st_off;
4403     const Type* val = NULL;
4404     if (st_size >= BytesPerInt &&
4405         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4406         (int)val->basic_type() < (int)T_OBJECT) {
4407       assert(st_off >= last_tile_end, "tiles do not overlap");
4408       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4409       last_tile_end = MAX2(last_tile_end, next_init_off);
4410     } else {
4411       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4412       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4413       assert(st_off      >= last_init_end, "inits do not overlap");
4414       last_init_end = next_init_off;  // it's a non-tile
4415     }
4416     #endif //ASSERT
4417   }
4418 
4419   remove_extra_zeroes();        // clear out all the zmems left over
4420   add_req(inits);
4421 
4422   if (!(UseTLAB && ZeroTLAB)) {
4423     // If anything remains to be zeroed, zero it all now.
4424     zeroes_done = align_down(zeroes_done, BytesPerInt);
4425     // if it is the last unused 4 bytes of an instance, forget about it
4426     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4427     if (zeroes_done + BytesPerLong >= size_limit) {
4428       AllocateNode* alloc = allocation();
4429       assert(alloc != NULL, "must be present");
4430       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4431         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4432         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4433         if (zeroes_done == k->layout_helper())
4434           zeroes_done = size_limit;
4435       }
4436     }
4437     if (zeroes_done < size_limit) {
4438       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4439                                             allocation()->in(AllocateNode::DefaultValue),
4440                                             allocation()->in(AllocateNode::RawDefaultValue),
4441                                             zeroes_done, size_in_bytes, phase);
4442     }
4443   }
4444 
4445   set_complete(phase);
4446   return rawmem;
4447 }

4475 //============================MergeMemNode=====================================
4476 //
4477 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4478 // contributing store or call operations.  Each contributor provides the memory
4479 // state for a particular "alias type" (see Compile::alias_type).  For example,
4480 // if a MergeMem has an input X for alias category #6, then any memory reference
4481 // to alias category #6 may use X as its memory state input, as an exact equivalent
4482 // to using the MergeMem as a whole.
4483 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4484 //
4485 // (Here, the <N> notation gives the index of the relevant adr_type.)
4486 //
4487 // In one special case (and more cases in the future), alias categories overlap.
4488 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4489 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4490 // it is exactly equivalent to that state W:
4491 //   MergeMem(<Bot>: W) <==> W
4492 //
4493 // Usually, the merge has more than one input.  In that case, where inputs
4494 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4495 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4496 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4497 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4498 //
4499 // A merge can take a "wide" memory state as one of its narrow inputs.
4500 // This simply means that the merge observes out only the relevant parts of
4501 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4502 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4503 //
4504 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4505 // and that memory slices "leak through":
4506 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4507 //
4508 // But, in such a cascade, repeated memory slices can "block the leak":
4509 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4510 //
4511 // In the last example, Y is not part of the combined memory state of the
4512 // outermost MergeMem.  The system must, of course, prevent unschedulable
4513 // memory states from arising, so you can be sure that the state Y is somehow
4514 // a precursor to state Y'.
4515 //
4516 //
4517 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4518 // of each MergeMemNode array are exactly the numerical alias indexes, including
4519 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4520 // Compile::alias_type (and kin) produce and manage these indexes.
4521 //
4522 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4523 // (Note that this provides quick access to the top node inside MergeMem methods,
4524 // without the need to reach out via TLS to Compile::current.)
4525 //
4526 // As a consequence of what was just described, a MergeMem that represents a full
4527 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4528 // containing all alias categories.
4529 //
4530 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4531 //
4532 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4533 // a memory state for the alias type <N>, or else the top node, meaning that
4534 // there is no particular input for that alias type.  Note that the length of
4535 // a MergeMem is variable, and may be extended at any time to accommodate new
4536 // memory states at larger alias indexes.  When merges grow, they are of course
4537 // filled with "top" in the unused in() positions.
4538 //
4539 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4540 // (Top was chosen because it works smoothly with passes like GCM.)
4541 //
4542 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4543 // the type of random VM bits like TLS references.)  Since it is always the
4544 // first non-Bot memory slice, some low-level loops use it to initialize an
4545 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4546 //
4547 //
4548 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4549 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4550 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4551 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4552 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4553 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4554 //
4555 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4556 // really that different from the other memory inputs.  An abbreviation called
4557 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4558 //
4559 //
4560 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4561 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4562 // that "emerges though" the base memory will be marked as excluding the alias types
4563 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4564 //
4565 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4566 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4567 //
4568 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4569 // (It is currently unimplemented.)  As you can see, the resulting merge is
4570 // actually a disjoint union of memory states, rather than an overlay.
4571 //

4634 // This method is invoked recursively on chains of MergeMem nodes
4635 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4636   // Remove chain'd MergeMems
4637   //
4638   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4639   // relative to the "in(Bot)".  Since we are patching both at the same time,
4640   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4641   // but rewrite each "in(i)" relative to the new "in(Bot)".
4642   Node *progress = NULL;
4643 
4644 
4645   Node* old_base = base_memory();
4646   Node* empty_mem = empty_memory();
4647   if (old_base == empty_mem)
4648     return NULL; // Dead memory path.
4649 
4650   MergeMemNode* old_mbase;
4651   if (old_base != NULL && old_base->is_MergeMem())
4652     old_mbase = old_base->as_MergeMem();
4653   else
4654     old_mbase = NULL;
4655   Node* new_base = old_base;
4656 
4657   // simplify stacked MergeMems in base memory
4658   if (old_mbase)  new_base = old_mbase->base_memory();
4659 
4660   // the base memory might contribute new slices beyond my req()
4661   if (old_mbase)  grow_to_match(old_mbase);
4662 
4663   // Look carefully at the base node if it is a phi.
4664   PhiNode* phi_base;
4665   if (new_base != NULL && new_base->is_Phi())
4666     phi_base = new_base->as_Phi();
4667   else
4668     phi_base = NULL;
4669 
4670   Node*    phi_reg = NULL;
4671   uint     phi_len = (uint)-1;
4672   if (phi_base != NULL && !phi_base->is_copy()) {
4673     // do not examine phi if degraded to a copy
4674     phi_reg = phi_base->region();
4675     phi_len = phi_base->req();
4676     // see if the phi is unfinished
4677     for (uint i = 1; i < phi_len; i++) {
4678       if (phi_base->in(i) == NULL) {
4679         // incomplete phi; do not look at it yet!
4680         phi_reg = NULL;
4681         phi_len = (uint)-1;
4682         break;
4683       }
4684     }
4685   }
4686 
4687   // Note:  We do not call verify_sparse on entry, because inputs
4688   // can normalize to the base_memory via subsume_node or similar
4689   // mechanisms.  This method repairs that damage.
4690 
4691   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4692 
4693   // Look at each slice.
4694   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4695     Node* old_in = in(i);
4696     // calculate the old memory value
4697     Node* old_mem = old_in;
4698     if (old_mem == empty_mem)  old_mem = old_base;
4699     assert(old_mem == memory_at(i), "");
4700 
4701     // maybe update (reslice) the old memory value
4702 
4703     // simplify stacked MergeMems
4704     Node* new_mem = old_mem;
4705     MergeMemNode* old_mmem;
4706     if (old_mem != NULL && old_mem->is_MergeMem())
4707       old_mmem = old_mem->as_MergeMem();
4708     else
4709       old_mmem = NULL;
4710     if (old_mmem == this) {
4711       // This can happen if loops break up and safepoints disappear.
4712       // A merge of BotPtr (default) with a RawPtr memory derived from a
4713       // safepoint can be rewritten to a merge of the same BotPtr with
4714       // the BotPtr phi coming into the loop.  If that phi disappears
4715       // also, we can end up with a self-loop of the mergemem.
4716       // In general, if loops degenerate and memory effects disappear,
4717       // a mergemem can be left looking at itself.  This simply means
4718       // that the mergemem's default should be used, since there is
4719       // no longer any apparent effect on this slice.
4720       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4721       //       from start.  Update the input to TOP.
4722       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4723     }
4724     else if (old_mmem != NULL) {
4725       new_mem = old_mmem->memory_at(i);
4726     }
4727     // else preceding memory was not a MergeMem
4728 
4729     // replace equivalent phis (unfortunately, they do not GVN together)
4730     if (new_mem != NULL && new_mem != new_base &&
4731         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4732       if (new_mem->is_Phi()) {
4733         PhiNode* phi_mem = new_mem->as_Phi();
4734         for (uint i = 1; i < phi_len; i++) {
4735           if (phi_base->in(i) != phi_mem->in(i)) {
4736             phi_mem = NULL;
4737             break;
4738           }
4739         }
4740         if (phi_mem != NULL) {
4741           // equivalent phi nodes; revert to the def
4742           new_mem = new_base;
4743         }
4744       }
4745     }
4746 
4747     // maybe store down a new value
4748     Node* new_in = new_mem;
4749     if (new_in == new_base)  new_in = empty_mem;
4750 
4751     if (new_in != old_in) {
4752       // Warning:  Do not combine this "if" with the previous "if"
4753       // A memory slice might have be be rewritten even if it is semantically
4754       // unchanged, if the base_memory value has changed.
4755       set_req(i, new_in);
4756       progress = this;          // Report progress
4757     }
4758   }
4759 
4760   if (new_base != old_base) {
4761     set_req(Compile::AliasIdxBot, new_base);
4762     // Don't use set_base_memory(new_base), because we need to update du.
4763     assert(base_memory() == new_base, "");
4764     progress = this;
4765   }
4766 
4767   if( base_memory() == this ) {
4768     // a self cycle indicates this memory path is dead
4769     set_req(Compile::AliasIdxBot, empty_mem);
4770   }
4771 
4772   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4773   // Recursion must occur after the self cycle check above
4774   if( base_memory()->is_MergeMem() ) {
4775     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4776     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4777     if( m != NULL &&
4778         (m->is_top() ||
4779          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4780       // propagate rollup of dead cycle to self
4781       set_req(Compile::AliasIdxBot, empty_mem);
4782     }
4783   }
4784 
4785   if( base_memory() == empty_mem ) {
4786     progress = this;
4787     // Cut inputs during Parse phase only.
4788     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4789     if( !can_reshape ) {
4790       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4791         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4792       }
4793     }
4794   }
4795 
4796   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4797     // Check if PhiNode::Ideal's "Split phis through memory merges"
4798     // transform should be attempted. Look for this->phi->this cycle.
4799     uint merge_width = req();
4800     if (merge_width > Compile::AliasIdxRaw) {
4801       PhiNode* phi = base_memory()->as_Phi();
4802       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4803         if (phi->in(i) == this) {
4804           phase->is_IterGVN()->_worklist.push(phi);
4805           break;
4806         }
4807       }
4808     }
4809   }
4810 
4811   assert(progress || verify_sparse(), "please, no dups of base");
4812   return progress;
4813 }

4857 // verify a narrow slice (either incoming or outgoing)
4858 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4859   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4860   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4861   if (Node::in_dump())               return;  // muzzle asserts when printing
4862   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4863   assert(n != NULL, "");
4864   // Elide intervening MergeMem's
4865   while (n->is_MergeMem()) {
4866     n = n->as_MergeMem()->memory_at(alias_idx);
4867   }
4868   Compile* C = Compile::current();
4869   const TypePtr* n_adr_type = n->adr_type();
4870   if (n == m->empty_memory()) {
4871     // Implicit copy of base_memory()
4872   } else if (n_adr_type != TypePtr::BOTTOM) {
4873     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4874     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4875   } else {
4876     // A few places like make_runtime_call "know" that VM calls are narrow,
4877     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4878     bool expected_wide_mem = false;
4879     if (n == m->base_memory()) {
4880       expected_wide_mem = true;
4881     } else if (alias_idx == Compile::AliasIdxRaw ||
4882                n == m->memory_at(Compile::AliasIdxRaw)) {
4883       expected_wide_mem = true;
4884     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4885       // memory can "leak through" calls on channels that
4886       // are write-once.  Allow this also.
4887       expected_wide_mem = true;
4888     }
4889     assert(expected_wide_mem, "expected narrow slice replacement");
4890   }
4891 }

4897 //-----------------------------memory_at---------------------------------------
4898 Node* MergeMemNode::memory_at(uint alias_idx) const {
4899   assert(alias_idx >= Compile::AliasIdxRaw ||
4900          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4901          "must avoid base_memory and AliasIdxTop");
4902 
4903   // Otherwise, it is a narrow slice.
4904   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4905   Compile *C = Compile::current();
4906   if (is_empty_memory(n)) {
4907     // the array is sparse; empty slots are the "top" node
4908     n = base_memory();
4909     assert(Node::in_dump()
4910            || n == NULL || n->bottom_type() == Type::TOP
4911            || n->adr_type() == NULL // address is TOP
4912            || n->adr_type() == TypePtr::BOTTOM
4913            || n->adr_type() == TypeRawPtr::BOTTOM
4914            || Compile::current()->AliasLevel() == 0,
4915            "must be a wide memory");
4916     // AliasLevel == 0 if we are organizing the memory states manually.
4917     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.
4918   } else {
4919     // make sure the stored slice is sane
4920     #ifdef ASSERT
4921     if (VMError::is_error_reported() || Node::in_dump()) {
4922     } else if (might_be_same(n, base_memory())) {
4923       // Give it a pass:  It is a mostly harmless repetition of the base.
4924       // This can arise normally from node subsumption during optimization.
4925     } else {
4926       verify_memory_slice(this, alias_idx, n);
4927     }
4928     #endif
4929   }
4930   return n;
4931 }

4934 void MergeMemNode::set_memory_at(uint alias_idx, Node *n) {
4935   verify_memory_slice(this, alias_idx, n);
4936   Node* empty_mem = empty_memory();
4937   if (n == base_memory())  n = empty_mem;  // collapse default
4938   uint need_req = alias_idx+1;
4939   if (req() < need_req) {
4940     if (n == empty_mem)  return;  // already the default, so do not grow me
4941     // grow the sparse array
4942     do {
4943       add_req(empty_mem);
4944     } while (req() < need_req);
4945   }
4946   set_req( alias_idx, n );

4952 void MergeMemNode::iteration_setup(const MergeMemNode* other) {
4953   if (other != NULL) {
4954     grow_to_match(other);
4955     // invariant:  the finite support of mm2 is within mm->req()
4956     #ifdef ASSERT
4957     for (uint i = req(); i < other->req(); i++) {
4958       assert(other->is_empty_memory(other->in(i)), "slice left uncovered");
4959     }
4960     #endif
4961   }
4962   // Replace spurious copies of base_memory by top.
4963   Node* base_mem = base_memory();
4964   if (base_mem != NULL && !base_mem->is_top()) {
4965     for (uint i = Compile::AliasIdxBot+1, imax = req(); i < imax; i++) {
4966       if (in(i) == base_mem)
4967         set_req(i, empty_memory());
4968     }
4969   }

4972 //---------------------------grow_to_match-------------------------------------
4973 void MergeMemNode::grow_to_match(const MergeMemNode* other) {
4974   Node* empty_mem = empty_memory();
4975   assert(other->is_empty_memory(empty_mem), "consistent sentinels");
4976   // look for the finite support of the other memory
4977   for (uint i = other->req(); --i >= req(); ) {
4978     if (other->in(i) != empty_mem) {
4979       uint new_len = i+1;
4980       while (req() < new_len)  add_req(empty_mem);
4981       break;
4982     }
4983   }
