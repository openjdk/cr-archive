  84 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  85                                          int  caller_adjustment,
  86                                          int  caller_actual_parameters,
  87                                          int  number_of_frames,
  88                                          intptr_t* frame_sizes,
  89                                          address* frame_pcs,
  90                                          BasicType return_type,
  91                                          int exec_mode) {
  92   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  93   _caller_adjustment         = caller_adjustment;
  94   _caller_actual_parameters  = caller_actual_parameters;
  95   _number_of_frames          = number_of_frames;
  96   _frame_sizes               = frame_sizes;
  97   _frame_pcs                 = frame_pcs;
  98   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
  99   _return_type               = return_type;
 100   _initial_info              = 0;
 101   // PD (x86 only)
 102   _counter_temp              = 0;
 103   _unpack_kind               = exec_mode;
 104   _sender_sp_temp            = 0;
 105 
 106   _total_frame_sizes         = size_of_frames();
 107   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 108 }

 125 int Deoptimization::UnrollBlock::size_of_frames() const {
 126   // Acount first for the adjustment of the initial frame
 127   int result = _caller_adjustment;
 128   for (int index = 0; index < number_of_frames(); index++) {
 129     result += frame_sizes()[index];
 130   }
 131   return result;
 132 }

 147 // In order to make fetch_unroll_info work properly with escape
 148 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 149 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 150 // of previously eliminated objects occurs in realloc_objects, which is
 151 // called from the method fetch_unroll_info_helper below.
 152 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 153   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 154   // but makes the entry a little slower. There is however a little dance we have to
 155   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 156 
 157   // fetch_unroll_info() is called at the beginning of the deoptimization
 158   // handler. Note this fact before we start generating temporary frames
 159   // that can confuse an asynchronous stack walker. This counter is
 160   // decremented at the end of unpack_frames().
 161   if (TraceDeoptimization) {
 162     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 163   }
 164   thread->inc_in_deopt_handler();
 165 
 166   return fetch_unroll_info_helper(thread, exec_mode);
 167 JRT_END

 169 #if COMPILER2_OR_JVMCI
 170 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 171                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 172   bool realloc_failures = false;
 173   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 174 
 175   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 176 
 177   // The flag return_oop() indicates call sites which return oop
 178   // in compiled code. Such sites include java method calls,
 179   // runtime calls (for example, used to allocate new objects/arrays
 180   // on slow code path) and any other calls generated in compiled code.
 181   // It is not guaranteed that we can get such information here only
 182   // by analyzing bytecode in deoptimized frames. This is why this flag
 183   // is set during method compilation (see Compile::Process_OopMap_Node()).
 184   // If the previous frame was popped or if we are dispatching an exception,
 185   // we don't have an oop result.
 186   ScopeDesc* scope = chunk->at(0)->scope();
 187   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 188   // In case of the return of multiple values, we must take care
 189   // of all oop return values.
 190   GrowableArray<Handle> return_oops;
 191   InlineKlass* vk = NULL;
 192   if (save_oop_result && scope->return_vt()) {
 193     vk = InlineKlass::returned_inline_klass(map);
 194     if (vk != NULL) {
 195       vk->save_oop_fields(map, return_oops);
 196       save_oop_result = false;
 197     }
 198   }
 199   if (save_oop_result) {
 200     // Reallocation may trigger GC. If deoptimization happened on return from
 201     // call which returns oop we need to save it since it is not in oopmap.
 202     oop result = deoptee.saved_oop_result(&map);
 203     assert(oopDesc::is_oop_or_null(result), "must be oop");
 204     return_oops.push(Handle(thread, result));
 205     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 206     if (TraceDeoptimization) {
 207       ttyLocker ttyl;
 208       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 209     }
 210   }
 211   if (objects != NULL || vk != NULL) {
 212     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 213     JRT_BLOCK
 214       if (vk != NULL) {
 215         realloc_failures = Deoptimization::realloc_inline_type_result(vk, map, return_oops, THREAD);
 216       }
 217       if (objects != NULL) {
 218         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 219         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 220       }
 221     JRT_END
 222 #ifndef PRODUCT
 223     if (TraceDeoptimization) {
 224       ttyLocker ttyl;
 225       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 226       if (objects != NULL) {
 227         Deoptimization::print_objects(objects, realloc_failures);
 228       } else {
 229         Handle obj = realloc_failures ? Handle() : return_oops.first();
 230         Deoptimization::print_object(vk, obj, realloc_failures);
 231       }
 232     }
 233 #endif
 234   }
 235   if (save_oop_result || vk != NULL) {
 236     // Restore result.
 237     assert(return_oops.length() == 1, "no inline type");
 238     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 239   }
 240   return realloc_failures;
 241 }

 243 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 244 #ifndef PRODUCT
 245   bool first = true;
 246 #endif
 247   for (int i = 0; i < chunk->length(); i++) {
 248     compiledVFrame* cvf = chunk->at(i);
 249     assert (cvf->scope() != NULL,"expect only compiled java frames");
 250     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 251     if (monitors->is_nonempty()) {
 252       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 253 #ifndef PRODUCT
 254       if (PrintDeoptimizationDetails) {
 255         ttyLocker ttyl;
 256         for (int j = 0; j < monitors->length(); j++) {
 257           MonitorInfo* mi = monitors->at(j);
 258           if (mi->eliminated()) {
 259             if (first) {
 260               first = false;
 261               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 262             }
 263             if (mi->owner_is_scalar_replaced()) {
 264               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 265               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 266             } else {
 267               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 268             }
 269           }
 270         }
 271       }
 272 #endif // !PRODUCT
 273     }
 274   }
 275 }
 276 #endif // COMPILER2_OR_JVMCI

 278 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 279 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 280 
 281   // Note: there is a safepoint safety issue here. No matter whether we enter
 282   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 283   // the vframeArray is created.
 284   //
 285 
 286   // Allocate our special deoptimization ResourceMark
 287   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 288   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 289   thread->set_deopt_mark(dmark);
 290 
 291   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 292   RegisterMap map(thread, true);
 293   RegisterMap dummy_map(thread, false);
 294   // Now get the deoptee with a valid map
 295   frame deoptee = stub_frame.sender(&map);
 296   // Set the deoptee nmethod
 297   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 298   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 299   thread->set_deopt_compiled_method(cm);
 300 
 301   if (VerifyStack) {
 302     thread->validate_frame_layout();
 303   }
 304 
 305   // Create a growable array of VFrames where each VFrame represents an inlined
 306   // Java frame.  This storage is allocated with the usual system arena.
 307   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 308   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 309   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 310   while (!vf->is_top()) {
 311     assert(vf->is_compiled_frame(), "Wrong frame type");
 312     chunk->push(compiledVFrame::cast(vf));
 313     vf = vf->sender();
 314   }
 315   assert(vf->is_compiled_frame(), "Wrong frame type");
 316   chunk->push(compiledVFrame::cast(vf));
 317 
 318   bool realloc_failures = false;
 319 
 320 #if COMPILER2_OR_JVMCI
 321 #if INCLUDE_JVMCI
 322   bool jvmci_enabled = true;
 323 #else
 324   bool jvmci_enabled = false;
 325 #endif
 326 
 327   // Reallocate the non-escaping objects and restore their fields. Then
 328   // relock objects if synchronization on them was eliminated.
 329   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 330     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 331   }
 332 #endif // COMPILER2_OR_JVMCI
 333 
 334   // Revoke biases, done with in java state.
 335   // No safepoints allowed after this
 336   revoke_from_deopt_handler(thread, deoptee, &map);
 337 
 338   // Ensure that no safepoint is taken after pointers have been stored
 339   // in fields of rematerialized objects.  If a safepoint occurs from here on
 340   // out the java state residing in the vframeArray will be missed.
 341   // Locks may be rebaised in a safepoint.
 342   NoSafepointVerifier no_safepoint;
 343 
 344 #if COMPILER2_OR_JVMCI
 345   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 346     eliminate_locks(thread, chunk, realloc_failures);
 347   }
 348 #endif // COMPILER2_OR_JVMCI
 349 
 350   ScopeDesc* trap_scope = chunk->at(0)->scope();
 351   Handle exceptionObject;
 352   if (trap_scope->rethrow_exception()) {
 353     if (PrintDeoptimizationDetails) {
 354       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 355     }
 356     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 357     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 358     ScopeValue* topOfStack = expressions->top();
 359     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 360     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 361   }
 362 
 363   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 364 #if COMPILER2_OR_JVMCI
 365   if (realloc_failures) {
 366     pop_frames_failed_reallocs(thread, array);
 367   }
 368 #endif
 369 
 370   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 371   thread->set_vframe_array_head(array);
 372 
 373   // Now that the vframeArray has been created if we have any deferred local writes
 374   // added by jvmti then we can free up that structure as the data is now in the
 375   // vframeArray
 376 
 377   if (thread->deferred_locals() != NULL) {
 378     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 379     int i = 0;
 380     do {
 381       // Because of inlining we could have multiple vframes for a single frame
 382       // and several of the vframes could have deferred writes. Find them all.
 383       if (list->at(i)->id() == array->original().id()) {
 384         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 385         list->remove_at(i);
 386         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 387         delete dlv;
 388       } else {
 389         i++;
 390       }
 391     } while ( i < list->length() );
 392     if (list->length() == 0) {
 393       thread->set_deferred_locals(NULL);
 394       // free the list and elements back to C heap.
 395       delete list;
 396     }
 397 
 398   }
 399 
 400   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 401   CodeBlob* cb = stub_frame.cb();
 402   // Verify we have the right vframeArray
 403   assert(cb->frame_size() >= 0, "Unexpected frame size");
 404   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 405 
 406   // If the deopt call site is a MethodHandle invoke call site we have
 407   // to adjust the unpack_sp.
 408   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 409   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 410     unpack_sp = deoptee.unextended_sp();
 411 
 412 #ifdef ASSERT
 413   assert(cb->is_deoptimization_stub() ||
 414          cb->is_uncommon_trap_stub() ||
 415          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 416          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 417          "unexpected code blob: %s", cb->name());
 418 #endif
 419 
 420   // This is a guarantee instead of an assert because if vframe doesn't match
 421   // we will unpack the wrong deoptimized frame and wind up in strange places
 422   // where it will be very difficult to figure out what went wrong. Better
 423   // to die an early death here than some very obscure death later when the
 424   // trail is cold.
 425   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 426   // in that it will fail to detect a problem when there is one. This needs
 427   // more work in tiger timeframe.
 428   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 429 
 430   int number_of_frames = array->frames();
 431 
 432   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 433   // virtual activation, which is the reverse of the elements in the vframes array.
 434   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 435   // +1 because we always have an interpreter return address for the final slot.
 436   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 437   int popframe_extra_args = 0;
 438   // Create an interpreter return address for the stub to use as its return
 439   // address so the skeletal frames are perfectly walkable
 440   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 441 
 442   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 443   // activation be put back on the expression stack of the caller for reexecution
 444   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 445     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 446   }
 447 
 448   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 449   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 450   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 451   //
 452   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 453   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 454 
 455   // It's possible that the number of parameters at the call site is
 456   // different than number of arguments in the callee when method
 457   // handles are used.  If the caller is interpreted get the real
 458   // value so that the proper amount of space can be added to it's
 459   // frame.
 460   bool caller_was_method_handle = false;
 461   if (deopt_sender.is_interpreted_frame()) {
 462     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 463     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 464     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 465       // Method handle invokes may involve fairly arbitrary chains of
 466       // calls so it's impossible to know how much actual space the
 467       // caller has for locals.
 468       caller_was_method_handle = true;
 469     }
 470   }
 471 
 472   //
 473   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 474   // frame_sizes/frame_pcs[1] next oldest frame (int)
 475   // frame_sizes/frame_pcs[n] youngest frame (int)
 476   //
 477   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 478   // owns the space for the return address to it's caller).  Confusing ain't it.
 479   //
 480   // The vframe array can address vframes with indices running from
 481   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 482   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 483   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 484   // so things look a little strange in this loop.
 485   //
 486   int callee_parameters = 0;
 487   int callee_locals = 0;
 488   for (int index = 0; index < array->frames(); index++ ) {
 489     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 490     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 491     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 492     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 493                                                                                                     callee_locals,
 494                                                                                                     index == 0,
 495                                                                                                     popframe_extra_args);
 496     // This pc doesn't have to be perfect just good enough to identify the frame
 497     // as interpreted so the skeleton frame will be walkable
 498     // The correct pc will be set when the skeleton frame is completely filled out
 499     // The final pc we store in the loop is wrong and will be overwritten below
 500     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 501 
 502     callee_parameters = array->element(index)->method()->size_of_parameters();
 503     callee_locals = array->element(index)->method()->max_locals();
 504     popframe_extra_args = 0;
 505   }
 506 
 507   // Compute whether the root vframe returns a float or double value.
 508   BasicType return_type;
 509   {
 510     methodHandle method(thread, array->element(0)->method());
 511     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 512     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 513   }
 514 
 515   // Compute information for handling adapters and adjusting the frame size of the caller.
 516   int caller_adjustment = 0;
 517 
 518   // Compute the amount the oldest interpreter frame will have to adjust
 519   // its caller's stack by. If the caller is a compiled frame then
 520   // we pretend that the callee has no parameters so that the
 521   // extension counts for the full amount of locals and not just
 522   // locals-parms. This is because without a c2i adapter the parm
 523   // area as created by the compiled frame will not be usable by
 524   // the interpreter. (Depending on the calling convention there
 525   // may not even be enough space).
 526 
 527   // QQQ I'd rather see this pushed down into last_frame_adjust
 528   // and have it take the sender (aka caller).
 529 
 530   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 531     caller_adjustment = last_frame_adjust(0, callee_locals);
 532   } else if (callee_locals > callee_parameters) {
 533     // The caller frame may need extending to accommodate
 534     // non-parameter locals of the first unpacked interpreted frame.
 535     // Compute that adjustment.
 536     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 537   }
 538 
 539   // If the sender is deoptimized we must retrieve the address of the handler
 540   // since the frame will "magically" show the original pc before the deopt
 541   // and we'd undo the deopt.
 542 
 543   frame_pcs[0] = deopt_sender.raw_pc();
 544 
 545   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 546 
 547 #if INCLUDE_JVMCI
 548   if (exceptionObject() != NULL) {
 549     thread->set_exception_oop(exceptionObject());
 550     exec_mode = Unpack_exception;
 551   }
 552 #endif
 553 
 554   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 555     assert(thread->has_pending_exception(), "should have thrown OOME");
 556     thread->set_exception_oop(thread->pending_exception());
 557     thread->clear_pending_exception();
 558     exec_mode = Unpack_exception;
 559   }
 560 
 561 #if INCLUDE_JVMCI
 562   if (thread->frames_to_pop_failed_realloc() > 0) {
 563     thread->set_pending_monitorenter(false);
 564   }
 565 #endif
 566 
 567   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 568                                       caller_adjustment * BytesPerWord,
 569                                       caller_was_method_handle ? 0 : callee_parameters,
 570                                       number_of_frames,
 571                                       frame_sizes,
 572                                       frame_pcs,
 573                                       return_type,
 574                                       exec_mode);
 575   // On some platforms, we need a way to pass some platform dependent
 576   // information to the unpacking code so the skeletal frames come out
 577   // correct (initial fp value, unextended sp, ...)
 578   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 579 
 580   if (array->frames() > 1) {
 581     if (VerifyStack && TraceDeoptimization) {
 582       ttyLocker ttyl;
 583       tty->print_cr("Deoptimizing method containing inlining");
 584     }
 585   }
 586 
 587   array->set_unroll_block(info);
 588   return info;
 589 }

 594                                         vframeArray *array) {
 595 
 596   // Get array if coming from exception
 597   if (array == NULL) {
 598     array = thread->vframe_array_head();
 599   }

 601

 604   thread->set_vframe_array_last(array);

 636 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 637 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 638 
 639   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 640   // the days we had adapter frames. When we deoptimize a situation where a
 641   // compiled caller calls a compiled caller will have registers it expects
 642   // to survive the call to the callee. If we deoptimize the callee the only
 643   // way we can restore these registers is to have the oldest interpreter
 644   // frame that we create restore these values. That is what this routine
 645   // will accomplish.
 646 
 647   // At the moment we have modified c2 to not have any callee save registers
 648   // so this problem does not exist and this routine is just a place holder.
 649 
 650   assert(f->is_interpreted_frame(), "must be interpreted");

 653 // Return BasicType of value being returned
 654 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 655 
 656   // We are already active in the special DeoptResourceMark any ResourceObj's we
 657   // allocate will be freed at the end of the routine.
 658 
 659   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 660   // but makes the entry a little slower. There is however a little dance we have to
 661   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 662   ResetNoHandleMark rnhm; // No-op in release/product versions
 663   HandleMark hm;
 664 
 665   frame stub_frame = thread->last_frame();
 666 
 667   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 668   // must point to the vframeArray for the unpack frame.
 669   vframeArray* array = thread->vframe_array_head();
 670 
 671 #ifndef PRODUCT
 672   if (TraceDeoptimization) {
 673     ttyLocker ttyl;
 674     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 675                   p2i(thread), p2i(array), exec_mode);
 676   }
 677 #endif
 678   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 679               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 680 
 681   UnrollBlock* info = array->unroll_block();
 682 
 683   // We set the last_Java frame. But the stack isn't really parsable here. So we
 684   // clear it to make sure JFR understands not to try and walk stacks from events
 685   // in here.
 686   intptr_t* sp = thread->frame_anchor()->last_Java_sp();
 687   thread->frame_anchor()->set_last_Java_sp(NULL);
 688 
 689   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 690   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 691 
 692   thread->frame_anchor()->set_last_Java_sp(sp);
 693 
 694   BasicType bt = info->return_type();
 695 
 696   // If we have an exception pending, claim that the return type is an oop
 697   // so the deopt_blob does not overwrite the exception_oop.
 698 
 699   if (exec_mode == Unpack_exception)
 700     bt = T_OBJECT;
 701 
 702   // Cleanup thread deopt data
 703   cleanup_deopt_info(thread, array);
 704 
 705 #ifndef PRODUCT
 706   if (VerifyStack) {
 707     ResourceMark res_mark;
 708     // Clear pending exception to not break verification code (restored afterwards)
 709     PRESERVE_EXCEPTION_MARK;
 710 
 711     thread->validate_frame_layout();
 712 
 713     // Verify that the just-unpacked frames match the interpreter's
 714     // notions of expression stack and locals
 715     vframeArray* cur_array = thread->vframe_array_last();
 716     RegisterMap rm(thread, false);
 717     rm.set_include_argument_oops(false);
 718     bool is_top_frame = true;
 719     int callee_size_of_parameters = 0;
 720     int callee_max_locals = 0;
 721     for (int i = 0; i < cur_array->frames(); i++) {
 722       vframeArrayElement* el = cur_array->element(i);
 723       frame* iframe = el->iframe();
 724       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 725 
 726       // Get the oop map for this bci
 727       InterpreterOopMap mask;
 728       int cur_invoke_parameter_size = 0;
 729       bool try_next_mask = false;
 730       int next_mask_expression_stack_size = -1;
 731       int top_frame_expression_stack_adjustment = 0;
 732       methodHandle mh(thread, iframe->interpreter_frame_method());
 733       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 734       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 735       int max_bci = mh->code_size();
 736       // Get to the next bytecode if possible
 737       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 738       // Check to see if we can grab the number of outgoing arguments
 739       // at an uncommon trap for an invoke (where the compiler
 740       // generates debug info before the invoke has executed)
 741       Bytecodes::Code cur_code = str.next();
 742       if (Bytecodes::is_invoke(cur_code)) {
 743         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 744         cur_invoke_parameter_size = invoke.size_of_parameters();
 745         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 746           callee_size_of_parameters++;
 747         }
 748       }
 749       if (str.bci() < max_bci) {
 750         Bytecodes::Code next_code = str.next();
 751         if (next_code >= 0) {
 752           // The interpreter oop map generator reports results before
 753           // the current bytecode has executed except in the case of
 754           // calls. It seems to be hard to tell whether the compiler
 755           // has emitted debug information matching the "state before"
 756           // a given bytecode or the state after, so we try both
 757           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 758             // Get expression stack size for the next bytecode
 759             InterpreterOopMap next_mask;
 760             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 761             next_mask_expression_stack_size = next_mask.expression_stack_size();
 762             if (Bytecodes::is_invoke(next_code)) {
 763               Bytecode_invoke invoke(mh, str.bci());
 764               next_mask_expression_stack_size += invoke.size_of_parameters();
 765             }
 766             // Need to subtract off the size of the result type of
 767             // the bytecode because this is not described in the
 768             // debug info but returned to the interpreter in the TOS
 769             // caching register
 770             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 771             if (bytecode_result_type != T_ILLEGAL) {
 772               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 773             }
 774             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 775             try_next_mask = true;
 776           }
 777         }
 778       }
 779 
 780       // Verify stack depth and oops in frame
 781       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 782       if (!(
 783             /* SPARC */
 784             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 785             /* x86 */
 786             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 787             (try_next_mask &&
 788              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 789                                                                     top_frame_expression_stack_adjustment))) ||
 790             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 791             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 792              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 793             )) {
 794         {
 795           ttyLocker ttyl;
 796 
 797           // Print out some information that will help us debug the problem
 798           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 799           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 800           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 801                         iframe->interpreter_frame_expression_stack_size());
 802           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 803           tty->print_cr("  try_next_mask = %d", try_next_mask);
 804           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 805           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 806           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 807           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 808           tty->print_cr("  exec_mode = %d", exec_mode);
 809           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 810           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 811           tty->print_cr("  Interpreted frames:");
 812           for (int k = 0; k < cur_array->frames(); k++) {
 813             vframeArrayElement* el = cur_array->element(k);
 814             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 815           }
 816           cur_array->print_on_2(tty);
 817         } // release tty lock before calling guarantee
 818         guarantee(false, "wrong number of expression stack elements during deopt");
 819       }
 820       VerifyOopClosure verify;
 821       iframe->oops_interpreted_do(&verify, &rm, false);
 822       callee_size_of_parameters = mh->size_of_parameters();
 823       callee_max_locals = mh->max_locals();
 824       is_top_frame = false;
 825     }
 826   }
 827 #endif /* !PRODUCT */
 828 
 829   return bt;
 830 JRT_END

 920       jlong value = (jlong)raw_value;
 921       return lookup(value);

 923     PrimitiveType value = (PrimitiveType)*((jint*)&raw_value);

 925   }

1026       InstanceKlass* ik = InstanceKlass::cast(k);

1032       // Inline type array must be zeroed because not all memory is reassigned
1033       obj = ak->allocate(sv->field_size(), THREAD);
1034     } else if (k->is_typeArray_klass()) {
1035       TypeArrayKlass* ak = TypeArrayKlass::cast(k);
1036       assert(sv->field_size() % type2size[ak->element_type()] == 0, "non-integral array length");
1037       int len = sv->field_size() / type2size[ak->element_type()];
1038       obj = ak->allocate(len, THREAD);
1039     } else if (k->is_objArray_klass()) {
1040       ObjArrayKlass* ak = ObjArrayKlass::cast(k);
1041       obj = ak->allocate(sv->field_size(), THREAD);
1042     }
1043 
1044     if (obj == NULL) {
1045       failures = true;
1046     }
1047 
1048     assert(sv->value().is_null(), "redundant reallocation");
1049     assert(obj != NULL || HAS_PENDING_EXCEPTION, "allocation should succeed or we should get an exception");
1050     CLEAR_PENDING_EXCEPTION;
1051     sv->set_value(obj);
1052   }
1053 
1054   if (failures) {
1055     THROW_OOP_(Universe::out_of_memory_error_realloc_objects(), failures);
1056   } else if (pending_exception.not_null()) {
1057     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1058   }
1059 
1060   return failures;
1061 }

1079 /**
1080  * For primitive types whose kind gets "erased" at runtime (shorts become stack ints),
1081  * we need to somehow be able to recover the actual kind to be able to write the correct
1082  * amount of bytes.
1083  * For that purpose, this method assumes that, for an entry spanning n bytes at index i,
1084  * the entries at index n + 1 to n + i are 'markers'.
1085  * For example, if we were writing a short at index 4 of a byte array of size 8, the
1086  * expected form of the array would be:
1087  *
1088  * {b0, b1, b2, b3, INT, marker, b6, b7}
1089  *
1090  * Thus, in order to get back the size of the entry, we simply need to count the number
1091  * of marked entries
1092  *
1093  * @param virtualArray the virtualized byte array
1094  * @param i index of the virtual entry we are recovering
1095  * @return The number of bytes the entry spans
1096  */
1097 static int count_number_of_bytes_for_entry(ObjectValue *virtualArray, int i) {
1098   int index = i;
1099   while (++index < virtualArray->field_size() &&
1100            virtualArray->field_at(index)->is_marker()) {}
1101   return index - i;
1102 }

1104 /**
1105  * If there was a guarantee for byte array to always start aligned to a long, we could
1106  * do a simple check on the parity of the index. Unfortunately, that is not always the
1107  * case. Thus, we check alignment of the actual address we are writing to.
1108  * In the unlikely case index 0 is 5-aligned for example, it would then be possible to
1109  * write a long to index 3.
1110  */
1111 static jbyte* check_alignment_get_addr(typeArrayOop obj, int index, int expected_alignment) {
1112     jbyte* res = obj->byte_at_addr(index);
1113     assert((((intptr_t) res) % expected_alignment) == 0, "Non-aligned write");
1114     return res;
1115 }

1121       break;
1122     case 2:

1124       break;
1125     case 4:
1126       *((jint *) check_alignment_get_addr(obj, index, 4)) = (jint) *((jint *) &val);
1127       break;
1128     case 8:

1138 // restore elements of an eliminated type array
1139 void Deoptimization::reassign_type_array_elements(frame* fr, RegisterMap* reg_map, ObjectValue* sv, typeArrayOop obj, BasicType type) {
1140   int index = 0;
1141   intptr_t val;
1142 
1143   for (int i = 0; i < sv->field_size(); i++) {
1144     StackValue* value = StackValue::create_stack_value(fr, reg_map, sv->field_at(i));
1145     switch(type) {
1146     case T_LONG: case T_DOUBLE: {
1147       assert(value->type() == T_INT, "Agreement.");
1148       StackValue* low =
1149         StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1150 #ifdef _LP64
1151       jlong res = (jlong)low->get_int();
1152 #else
1153       jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1154 #endif
1155       obj->long_at_put(index, res);
1156       break;
1157     }
1158 
1159     // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1160     case T_INT: case T_FLOAT: { // 4 bytes.
1161       assert(value->type() == T_INT, "Agreement.");
1162       bool big_value = false;
1163       if (i + 1 < sv->field_size() && type == T_INT) {
1164         if (sv->field_at(i)->is_location()) {
1165           Location::Type type = ((LocationValue*) sv->field_at(i))->location().type();
1166           if (type == Location::dbl || type == Location::lng) {
1167             big_value = true;
1168           }
1169         } else if (sv->field_at(i)->is_constant_int()) {
1170           ScopeValue* next_scope_field = sv->field_at(i + 1);
1171           if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1172             big_value = true;
1173           }
1174         }
1175       }
1176 
1177       if (big_value) {
1178         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1179   #ifdef _LP64
1180         jlong res = (jlong)low->get_int();
1181   #else
1182         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1183   #endif
1184         obj->int_at_put(index, (jint)*((jint*)&res));
1185         obj->int_at_put(++index, (jint)*(((jint*)&res) + 1));
1186       } else {
1187         val = value->get_int();
1188         obj->int_at_put(index, (jint)*((jint*)&val));
1189       }
1190       break;
1191     }
1192 
1193     case T_SHORT:
1194       assert(value->type() == T_INT, "Agreement.");
1195       val = value->get_int();
1196       obj->short_at_put(index, (jshort)*((jint*)&val));
1197       break;
1198 
1199     case T_CHAR:
1200       assert(value->type() == T_INT, "Agreement.");
1201       val = value->get_int();
1202       obj->char_at_put(index, (jchar)*((jint*)&val));
1203       break;
1204 
1205     case T_BYTE: {
1206       assert(value->type() == T_INT, "Agreement.");
1207       // The value we get is erased as a regular int. We will need to find its actual byte count 'by hand'.
1208       val = value->get_int();
1209 #if INCLUDE_JVMCI
1210       int byte_count = count_number_of_bytes_for_entry(sv, i);
1211       byte_array_put(obj, val, index, byte_count);
1212       // According to byte_count contract, the values from i + 1 to i + byte_count are illegal values. Skip.
1213       i += byte_count - 1; // Balance the loop counter.
1214       index += byte_count;
1215       // index has been updated so continue at top of loop
1216       continue;
1217 #else
1218       obj->byte_at_put(index, (jbyte)*((jint*)&val));
1219       break;
1220 #endif // INCLUDE_JVMCI
1221     }
1222 
1223     case T_BOOLEAN: {
1224       assert(value->type() == T_INT, "Agreement.");
1225       val = value->get_int();
1226       obj->bool_at_put(index, (jboolean)*((jint*)&val));
1227       break;
1228     }
1229 
1230       default:
1231         ShouldNotReachHere();
1232     }
1233     index++;
1234   }
1235 }

1263 // Restore fields of an eliminated instance object using the same field order
1264 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1265 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1266 
1267   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1268   InstanceKlass* ik = klass;
1269   while (ik != NULL) {
1270     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1271       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1272         ReassignedField field;
1273         field._offset = fs.offset();
1274         field._type = Signature::basic_type(fs.signature());
1275         if (field._type == T_INLINE_TYPE) {
1276           field._type = T_OBJECT;
1277         }
1278         if (fs.is_inlined()) {
1279           // Resolve klass of flattened inline type field
1280           Klass* vk = klass->get_inline_type_field_klass(fs.index());
1281           field._klass = InlineKlass::cast(vk);
1282           field._type = T_INLINE_TYPE;
1283         }
1284         fields->append(field);
1285       }
1286     }
1287     ik = ik->superklass();
1288   }
1289   fields->sort(compare);
1290   for (int i = 0; i < fields->length(); i++) {
1291     intptr_t val;
1292     ScopeValue* scope_field = sv->field_at(svIndex);
1293     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1294     int offset = base_offset + fields->at(i)._offset;
1295     BasicType type = fields->at(i)._type;
1296     switch (type) {
1297       case T_OBJECT:
1298       case T_ARRAY:
1299         assert(value->type() == T_OBJECT, "Agreement.");
1300         obj->obj_field_put(offset, value->get_obj()());
1301         break;
1302 
1303       case T_INLINE_TYPE: {
1304         // Recursively re-assign flattened inline type fields
1305         InstanceKlass* vk = fields->at(i)._klass;
1306         assert(vk != NULL, "must be resolved");
1307         offset -= InlineKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1308         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1309         continue; // Continue because we don't need to increment svIndex
1310       }
1311 
1312       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1313       case T_INT: case T_FLOAT: { // 4 bytes.
1314         assert(value->type() == T_INT, "Agreement.");
1315         bool big_value = false;
1316         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1317           if (scope_field->is_location()) {
1318             Location::Type type = ((LocationValue*) scope_field)->location().type();
1319             if (type == Location::dbl || type == Location::lng) {
1320               big_value = true;
1321             }
1322           }
1323           if (scope_field->is_constant_int()) {
1324             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1325             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1326               big_value = true;
1327             }
1328           }
1329         }
1330 
1331         if (big_value) {
1332           i++;
1333           assert(i < fields->length(), "second T_INT field needed");
1334           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1335         } else {
1336           val = value->get_int();
1337           obj->int_field_put(offset, (jint)*((jint*)&val));
1338           break;
1339         }
1340       }
1341         /* no break */
1342 
1343       case T_LONG: case T_DOUBLE: {
1344         assert(value->type() == T_INT, "Agreement.");
1345         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1346 #ifdef _LP64
1347         jlong res = (jlong)low->get_int();
1348 #else
1349         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1350 #endif
1351         obj->long_field_put(offset, res);
1352         break;
1353       }
1354 
1355       case T_SHORT:
1356         assert(value->type() == T_INT, "Agreement.");
1357         val = value->get_int();
1358         obj->short_field_put(offset, (jshort)*((jint*)&val));
1359         break;
1360 
1361       case T_CHAR:
1362         assert(value->type() == T_INT, "Agreement.");
1363         val = value->get_int();
1364         obj->char_field_put(offset, (jchar)*((jint*)&val));
1365         break;
1366 
1367       case T_BYTE:
1368         assert(value->type() == T_INT, "Agreement.");
1369         val = value->get_int();
1370         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1371         break;
1372 
1373       case T_BOOLEAN:
1374         assert(value->type() == T_INT, "Agreement.");
1375         val = value->get_int();
1376         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1377         break;
1378 
1379       default:
1380         ShouldNotReachHere();
1381     }
1382     svIndex++;
1383   }
1384   return svIndex;
1385 }

1436 // relock objects for which synchronization was eliminated
1437 void Deoptimization::relock_objects(GrowableArray<MonitorInfo*>* monitors, JavaThread* thread, bool realloc_failures) {
1438   for (int i = 0; i < monitors->length(); i++) {
1439     MonitorInfo* mon_info = monitors->at(i);

1441       assert(!mon_info->owner_is_scalar_replaced() || realloc_failures, "reallocation was missed");

1443         Handle obj(thread, mon_info->owner());
1444         markWord mark = obj->mark();
1445         if (UseBiasedLocking && mark.has_bias_pattern()) {
1446           // New allocated objects may have the mark set to anonymously biased.
1447           // Also the deoptimized method may called methods with synchronization
1448           // where the thread-local object is bias locked to the current thread.
1449           assert(mark.is_biased_anonymously() ||
1450                  mark.biased_locker() == thread, "should be locked to current thread");
1451           // Reset mark word to unbiased prototype.
1452           markWord unbiased_prototype = markWord::prototype().set_age(mark.age());
1453           obj->set_mark(unbiased_prototype);
1454         }
1455         BasicLock* lock = mon_info->lock();
1456         ObjectSynchronizer::enter(obj, lock, thread);
1457         assert(mon_info->owner()->is_locked(), "object must be locked now");
1458       }
1459     }
1460   }

1590 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1591   // Unfortunately we don't have a RegisterMap available in most of
1592   // the places we want to call this routine so we need to walk the
1593   // stack again to update the register map.
1594   if (map == NULL || !map->update_map()) {
1595     StackFrameStream sfs(thread, true);
1596     bool found = false;
1597     while (!found && !sfs.is_done()) {
1598       frame* cur = sfs.current();
1599       sfs.next();
1600       found = cur->id() == fr.id();
1601     }
1602     assert(found, "frame to be deoptimized not found on target thread's stack");
1603     map = sfs.register_map();
1604   }
1605 
1606   vframe* vf = vframe::new_vframe(&fr, map, thread);
1607   compiledVFrame* cvf = compiledVFrame::cast(vf);
1608   // Revoke monitors' biases in all scopes
1609   while (!cvf->is_top()) {
1610     collect_monitors(cvf, objects_to_revoke);
1611     cvf = compiledVFrame::cast(cvf->sender());
1612   }
1613   collect_monitors(cvf, objects_to_revoke);
1614 }

1616 void Deoptimization::revoke_from_deopt_handler(JavaThread* thread, frame fr, RegisterMap* map) {
1617   if (!UseBiasedLocking) {
1618     return;

1621   get_monitors_from_stack(objects_to_revoke, thread, fr, map);

1854   // handler. Note this fact before we start generating temporary frames

1878

1881 #if INCLUDE_JVMCI

1896 #if INCLUDE_JVMCI
1897           , debug_id
1898 #endif
1899           );
1900     }
1901 
1902     methodHandle    trap_method(THREAD, trap_scope->method());
1903     int             trap_bci    = trap_scope->bci();
1904 #if INCLUDE_JVMCI
1905     jlong           speculation = thread->pending_failed_speculation();
1906     if (nm->is_compiled_by_jvmci() && nm->is_nmethod()) { // Exclude AOTed methods
1907       nm->as_nmethod()->update_speculation(thread);
1908     } else {
1909       assert(speculation == 0, "There should not be a speculation for methods compiled by non-JVMCI compilers");
1910     }

1913       trap_bci = 0;

1916 
1917     if (reason == Deoptimization::Reason_transfer_to_interpreter) {
1918       thread->set_pending_transfer_to_interpreter(true);
1919     }

1922     Bytecodes::Code trap_bc     = trap_method->java_code_at(trap_bci);

1924     gather_statistics(reason, action, trap_bc);
1925

1927     // Need MDO to record RTM code generation state.

1931 #if INCLUDE_JVMCI

1942       get_method_data(thread, profiled_method, create_if_missing);
1943 
1944     JFR_ONLY(post_deoptimization_event(nm, trap_method(), trap_bci, trap_bc, reason, action);)
1945

1947     Events::log_deopt_message(thread, "Uncommon trap: reason=%s action=%s pc=" INTPTR_FORMAT " method=%s @ %d %s",
1948                               trap_reason_name(reason), trap_action_name(action), p2i(fr.pc()),
1949                               trap_method->name_and_sig_as_C_string(), trap_bci, nm->compiler_name());

1951     // Print a bunch of diagnostics, if requested.
1952     if (TraceDeoptimization || LogCompilation) {
1953       ResourceMark rm;
1954       ttyLocker ttyl;
1955       char buf[100];
1956       if (xtty != NULL) {

1958                          os::current_thread_id(),

1960 #if INCLUDE_JVMCI
1961         if (speculation != 0) {

1963         }
1964 #endif
1965         nm->log_identity(xtty);
1966       }
1967       Symbol* class_name = NULL;
1968       bool unresolved = false;
1969       if (unloaded_class_index >= 0) {
1970         constantPoolHandle constants (THREAD, trap_method->constants());
1971         if (constants->tag_at(unloaded_class_index).is_unresolved_klass()) {
1972           class_name = constants->klass_name_at(unloaded_class_index);
1973           unresolved = true;
1974           if (xtty != NULL)
1975             xtty->print(" unresolved='1'");
1976         } else if (constants->tag_at(unloaded_class_index).is_symbol()) {
1977           class_name = constants->symbol_at(unloaded_class_index);
1978         }
1979         if (xtty != NULL)
1980           xtty->name(class_name);
1981       }
1982       if (xtty != NULL && trap_mdo != NULL && (int)reason < (int)MethodData::_trap_hist_limit) {
1983         // Dump the relevant MDO state.
1984         // This is the deopt count for the current reason, any previous
1985         // reasons or recompiles seen at this point.
1986         int dcnt = trap_mdo->trap_count(reason);
1987         if (dcnt != 0)
1988           xtty->print(" count='%d'", dcnt);
1989         ProfileData* pdata = trap_mdo->bci_to_data(trap_bci);
1990         int dos = (pdata == NULL)? 0: pdata->trap_state();
1991         if (dos != 0) {
1992           xtty->print(" state='%s'", format_trap_state(buf, sizeof(buf), dos));
1993           if (trap_state_is_recompiled(dos)) {
1994             int recnt2 = trap_mdo->overflow_recompile_count();
1995             if (recnt2 != 0)
1996               xtty->print(" recompiles2='%d'", recnt2);
1997           }
1998         }
1999       }
2000       if (xtty != NULL) {
2001         xtty->stamp();
2002         xtty->end_head();
2003       }
2004       if (TraceDeoptimization) {  // make noise on the tty
2005         tty->print("Uncommon trap occurred in");
2006         nm->method()->print_short_name(tty);
2007         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
2008 #if INCLUDE_JVMCI
2009         if (nm->is_nmethod()) {
2010           const char* installed_code_name = nm->as_nmethod()->jvmci_name();
2011           if (installed_code_name != NULL) {
2012             tty->print(" (JVMCI: installed code name=%s) ", installed_code_name);
2013           }
2014         }
2015 #endif
2016         tty->print(" (@" INTPTR_FORMAT ") thread=" UINTX_FORMAT " reason=%s action=%s unloaded_class_index=%d" JVMCI_ONLY(" debug_id=%d"),
2017                    p2i(fr.pc()),
2018                    os::current_thread_id(),
2019                    trap_reason_name(reason),
2020                    trap_action_name(action),
2021                    unloaded_class_index
2022 #if INCLUDE_JVMCI
2023                    , debug_id
2024 #endif
2025                    );
2026         if (class_name != NULL) {
2027           tty->print(unresolved ? " unresolved class: " : " symbol: ");
2028           class_name->print_symbol_on(tty);
2029         }
2030         tty->cr();
2031       }
2032       if (xtty != NULL) {
2033         // Log the precise location of the trap.
2034         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
2035           xtty->begin_elem("jvms bci='%d'", sd->bci());
2036           xtty->method(sd->method());
2037           xtty->end_elem();
2038           if (sd->is_top())  break;
2039         }
2040         xtty->tail("uncommon_trap");
2041       }
2042     }
2043     // (End diagnostic printout.)
2044 
2045     // Load class if necessary
2046     if (unloaded_class_index >= 0) {
2047       constantPoolHandle constants(THREAD, trap_method->constants());
2048       load_class_by_index(constants, unloaded_class_index);
2049     }
2050 
2051     // Flush the nmethod if necessary and desirable.
2052     //
2053     // We need to avoid situations where we are re-flushing the nmethod
2054     // because of a hot deoptimization site.  Repeated flushes at the same
2055     // point need to be detected by the compiler and avoided.  If the compiler
2056     // cannot avoid them (or has a bug and "refuses" to avoid them), this
2057     // module must take measures to avoid an infinite cycle of recompilation
2058     // and deoptimization.  There are several such measures:
2059     //
2060     //   1. If a recompilation is ordered a second time at some site X
2061     //   and for the same reason R, the action is adjusted to 'reinterpret',
2062     //   to give the interpreter time to exercise the method more thoroughly.
2063     //   If this happens, the method's overflow_recompile_count is incremented.
2064     //
2065     //   2. If the compiler fails to reduce the deoptimization rate, then
2066     //   the method's overflow_recompile_count will begin to exceed the set
2067     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2068     //   is adjusted to 'make_not_compilable', and the method is abandoned
2069     //   to the interpreter.  This is a performance hit for hot methods,
2070     //   but is better than a disastrous infinite cycle of recompilations.
2071     //   (Actually, only the method containing the site X is abandoned.)
2072     //
2073     //   3. In parallel with the previous measures, if the total number of
2074     //   recompilations of a method exceeds the much larger set limit
2075     //   PerMethodRecompilationCutoff, the method is abandoned.
2076     //   This should only happen if the method is very large and has
2077     //   many "lukewarm" deoptimizations.  The code which enforces this
2078     //   limit is elsewhere (class nmethod, class Method).
2079     //
2080     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2081     // to recompile at each bytecode independently of the per-BCI cutoff.
2082     //
2083     // The decision to update code is up to the compiler, and is encoded
2084     // in the Action_xxx code.  If the compiler requests Action_none
2085     // no trap state is changed, no compiled code is changed, and the
2086     // computation suffers along in the interpreter.
2087     //
2088     // The other action codes specify various tactics for decompilation
2089     // and recompilation.  Action_maybe_recompile is the loosest, and
2090     // allows the compiled code to stay around until enough traps are seen,
2091     // and until the compiler gets around to recompiling the trapping method.
2092     //
2093     // The other actions cause immediate removal of the present code.
2094 
2095     // Traps caused by injected profile shouldn't pollute trap counts.
2096     bool injected_profile_trap = trap_method->has_injected_profile() &&
2097                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2098 
2099     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;
2100     bool make_not_entrant = false;
2101     bool make_not_compilable = false;
2102     bool reprofile = false;
2103     switch (action) {
2104     case Action_none:
2105       // Keep the old code.
2106       update_trap_state = false;
2107       break;
2108     case Action_maybe_recompile:
2109       // Do not need to invalidate the present code, but we can
2110       // initiate another
2111       // Start compiler without (necessarily) invalidating the nmethod.
2112       // The system will tolerate the old code, but new code should be
2113       // generated when possible.
2114       break;
2115     case Action_reinterpret:
2116       // Go back into the interpreter for a while, and then consider
2117       // recompiling form scratch.
2118       make_not_entrant = true;
2119       // Reset invocation counter for outer most method.
2120       // This will allow the interpreter to exercise the bytecodes
2121       // for a while before recompiling.
2122       // By contrast, Action_make_not_entrant is immediate.
2123       //
2124       // Note that the compiler will track null_check, null_assert,
2125       // range_check, and class_check events and log them as if they
2126       // had been traps taken from compiled code.  This will update
2127       // the MDO trap history so that the next compilation will
2128       // properly detect hot trap sites.
2129       reprofile = true;
2130       break;
2131     case Action_make_not_entrant:
2132       // Request immediate recompilation, and get rid of the old code.
2133       // Make them not entrant, so next time they are called they get
2134       // recompiled.  Unloaded classes are loaded now so recompile before next
2135       // time they are called.  Same for uninitialized.  The interpreter will
2136       // link the missing class, if any.
2137       make_not_entrant = true;
2138       break;
2139     case Action_make_not_compilable:
2140       // Give up on compiling this method at all.
2141       make_not_entrant = true;
2142       make_not_compilable = true;
2143       break;
2144     default:
2145       ShouldNotReachHere();
2146     }
2147 
2148     // Setting +ProfileTraps fixes the following, on all platforms:
2149     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2150     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2151     // recompile relies on a MethodData* to record heroic opt failures.
2152 
2153     // Whether the interpreter is producing MDO data or not, we also need
2154     // to use the MDO to detect hot deoptimization points and control
2155     // aggressive optimization.
2156     bool inc_recompile_count = false;
2157     ProfileData* pdata = NULL;
2158     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2159       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2160       uint this_trap_count = 0;
2161       bool maybe_prior_trap = false;
2162       bool maybe_prior_recompile = false;
2163       pdata = query_update_method_data(trap_mdo, trap_bci, reason, true,
2164 #if INCLUDE_JVMCI
2165                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2166 #endif
2167                                    nm->method(),
2168                                    //outputs:
2169                                    this_trap_count,
2170                                    maybe_prior_trap,
2171                                    maybe_prior_recompile);
2172       // Because the interpreter also counts null, div0, range, and class
2173       // checks, these traps from compiled code are double-counted.
2174       // This is harmless; it just means that the PerXTrapLimit values
2175       // are in effect a little smaller than they look.
2176 
2177       DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2178       if (per_bc_reason != Reason_none) {
2179         // Now take action based on the partially known per-BCI history.
2180         if (maybe_prior_trap
2181             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2182           // If there are too many traps at this BCI, force a recompile.
2183           // This will allow the compiler to see the limit overflow, and
2184           // take corrective action, if possible.  The compiler generally
2185           // does not use the exact PerBytecodeTrapLimit value, but instead
2186           // changes its tactics if it sees any traps at all.  This provides
2187           // a little hysteresis, delaying a recompile until a trap happens
2188           // several times.
2189           //
2190           // Actually, since there is only one bit of counter per BCI,
2191           // the possible per-BCI counts are {0,1,(per-method count)}.
2192           // This produces accurate results if in fact there is only
2193           // one hot trap site, but begins to get fuzzy if there are
2194           // many sites.  For example, if there are ten sites each
2195           // trapping two or more times, they each get the blame for
2196           // all of their traps.
2197           make_not_entrant = true;
2198         }
2199 
2200         // Detect repeated recompilation at the same BCI, and enforce a limit.
2201         if (make_not_entrant && maybe_prior_recompile) {
2202           // More than one recompile at this point.
2203           inc_recompile_count = maybe_prior_trap;
2204         }
2205       } else {
2206         // For reasons which are not recorded per-bytecode, we simply
2207         // force recompiles unconditionally.
2208         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2209         make_not_entrant = true;
2210       }
2211 
2212       // Go back to the compiler if there are too many traps in this method.
2213       if (this_trap_count >= per_method_trap_limit(reason)) {
2214         // If there are too many traps in this method, force a recompile.
2215         // This will allow the compiler to see the limit overflow, and
2216         // take corrective action, if possible.
2217         // (This condition is an unlikely backstop only, because the
2218         // PerBytecodeTrapLimit is more likely to take effect first,
2219         // if it is applicable.)
2220         make_not_entrant = true;
2221       }
2222 
2223       // Here's more hysteresis:  If there has been a recompile at
2224       // this trap point already, run the method in the interpreter
2225       // for a while to exercise it more thoroughly.
2226       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2227         reprofile = true;
2228       }
2229     }
2230 
2231     // Take requested actions on the method:
2232 
2233     // Recompile
2234     if (make_not_entrant) {
2235       if (!nm->make_not_entrant()) {
2236         return; // the call did not change nmethod's state
2237       }
2238 
2239       if (pdata != NULL) {
2240         // Record the recompilation event, if any.
2241         int tstate0 = pdata->trap_state();
2242         int tstate1 = trap_state_set_recompiled(tstate0, true);
2243         if (tstate1 != tstate0)
2244           pdata->set_trap_state(tstate1);
2245       }
2246 
2247 #if INCLUDE_RTM_OPT
2248       // Restart collecting RTM locking abort statistic if the method
2249       // is recompiled for a reason other than RTM state change.
2250       // Assume that in new recompiled code the statistic could be different,
2251       // for example, due to different inlining.
2252       if ((reason != Reason_rtm_state_change) && (trap_mdo != NULL) &&
2253           UseRTMDeopt && (nm->as_nmethod()->rtm_state() != ProfileRTM)) {
2254         trap_mdo->atomic_set_rtm_state(ProfileRTM);
2255       }
2256 #endif
2257       // For code aging we count traps separately here, using make_not_entrant()
2258       // as a guard against simultaneous deopts in multiple threads.
2259       if (reason == Reason_tenured && trap_mdo != NULL) {
2260         trap_mdo->inc_tenure_traps();
2261       }
2262     }
2263 
2264     if (inc_recompile_count) {
2265       trap_mdo->inc_overflow_recompile_count();
2266       if ((uint)trap_mdo->overflow_recompile_count() >
2267           (uint)PerBytecodeRecompilationCutoff) {
2268         // Give up on the method containing the bad BCI.
2269         if (trap_method() == nm->method()) {
2270           make_not_compilable = true;
2271         } else {
2272           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2273           // But give grace to the enclosing nm->method().
2274         }
2275       }
2276     }
2277 
2278     // Reprofile
2279     if (reprofile) {
2280       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2281     }
2282 
2283     // Give up compiling
2284     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2285       assert(make_not_entrant, "consistent");
2286       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2287     }
2288 
2289   } // Free marked resources
2290 
2291 }

2295 Deoptimization::query_update_method_data(MethodData* trap_mdo,
2296                                          int trap_bci,
2297                                          Deoptimization::DeoptReason reason,
2298                                          bool update_total_trap_count,
2299 #if INCLUDE_JVMCI
2300                                          bool is_osr,
2301 #endif
2302                                          Method* compiled_method,
2303                                          //outputs:
2304                                          uint& ret_this_trap_count,
2305                                          bool& ret_maybe_prior_trap,
2306                                          bool& ret_maybe_prior_recompile) {
2307   bool maybe_prior_trap = false;
2308   bool maybe_prior_recompile = false;
2309   uint this_trap_count = 0;
2310   if (update_total_trap_count) {
2311     uint idx = reason;
2312 #if INCLUDE_JVMCI
2313     if (is_osr) {
2314       idx += Reason_LIMIT;
2315     }
2316 #endif
2317     uint prior_trap_count = trap_mdo->trap_count(idx);
2318     this_trap_count  = trap_mdo->inc_trap_count(idx);
2319 
2320     // If the runtime cannot find a place to store trap history,
2321     // it is estimated based on the general condition of the method.
2322     // If the method has ever been recompiled, or has ever incurred
2323     // a trap with the present reason , then this BCI is assumed
2324     // (pessimistically) to be the culprit.
2325     maybe_prior_trap      = (prior_trap_count != 0);
2326     maybe_prior_recompile = (trap_mdo->decompile_count() != 0);
2327   }
2328   ProfileData* pdata = NULL;
2329 
2330 
2331   // For reasons which are recorded per bytecode, we check per-BCI data.
2332   DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2333   assert(per_bc_reason != Reason_none || update_total_trap_count, "must be");
2334   if (per_bc_reason != Reason_none) {
2335     // Find the profile data for this BCI.  If there isn't one,
2336     // try to allocate one from the MDO's set of spares.
2337     // This will let us detect a repeated trap at this point.
2338     pdata = trap_mdo->allocate_bci_to_data(trap_bci, reason_is_speculate(reason) ? compiled_method : NULL);
2339 
2340     if (pdata != NULL) {
2341       if (reason_is_speculate(reason) && !pdata->is_SpeculativeTrapData()) {
2342         if (LogCompilation && xtty != NULL) {
2343           ttyLocker ttyl;
2344           // no more room for speculative traps in this MDO
2345           xtty->elem("speculative_traps_oom");
2346         }
2347       }
2348       // Query the trap state of this profile datum.
2349       int tstate0 = pdata->trap_state();
2350       if (!trap_state_has_reason(tstate0, per_bc_reason))
2351         maybe_prior_trap = false;
2352       if (!trap_state_is_recompiled(tstate0))
2353         maybe_prior_recompile = false;
2354 
2355       // Update the trap state of this profile datum.
2356       int tstate1 = tstate0;
2357       // Record the reason.
2358       tstate1 = trap_state_add_reason(tstate1, per_bc_reason);
2359       // Store the updated state on the MDO, for next time.
2360       if (tstate1 != tstate0)
2361         pdata->set_trap_state(tstate1);
2362     } else {
2363       if (LogCompilation && xtty != NULL) {
2364         ttyLocker ttyl;
2365         // Missing MDP?  Leave a small complaint in the log.
2366         xtty->elem("missing_mdp bci='%d'", trap_bci);
2367       }
2368     }
2369   }
2370 
2371   // Return results:
2372   ret_this_trap_count = this_trap_count;
2373   ret_maybe_prior_trap = maybe_prior_trap;
2374   ret_maybe_prior_recompile = maybe_prior_recompile;
2375   return pdata;
2376 }

2412 // Local derived constants.
2413 // Further breakdown of DataLayout::trap_state, as promised by DataLayout.
2414 const int DS_REASON_MASK   = ((uint)DataLayout::trap_mask) >> 1;
2415 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;
2416 
2417 //---------------------------trap_state_reason---------------------------------
2418 Deoptimization::DeoptReason
2419 Deoptimization::trap_state_reason(int trap_state) {
2420   // This assert provides the link between the width of DataLayout::trap_bits
2421   // and the encoding of "recorded" reasons.  It ensures there are enough
2422   // bits to store all needed reasons in the per-BCI MDO profile.
2423   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2424   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2425   trap_state -= recompile_bit;
2426   if (trap_state == DS_REASON_MASK) {
2427     return Reason_many;
2428   } else {
2429     assert((int)Reason_none == 0, "state=0 => Reason_none");
2430     return (DeoptReason)trap_state;
2431   }
2432 }
2433 //-------------------------trap_state_has_reason-------------------------------
2434 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2435   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2436   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2437   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2438   trap_state -= recompile_bit;
2439   if (trap_state == DS_REASON_MASK) {
2440     return -1;  // true, unspecifically (bottom of state lattice)
2441   } else if (trap_state == reason) {
2442     return 1;   // true, definitely
2443   } else if (trap_state == 0) {
2444     return 0;   // false, definitely (top of state lattice)
2445   } else {
2446     return 0;   // false, definitely
2447   }
2448 }
2449 //-------------------------trap_state_add_reason-------------------------------
2450 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2451   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2452   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2453   trap_state -= recompile_bit;
2454   if (trap_state == DS_REASON_MASK) {
2455     return trap_state + recompile_bit;     // already at state lattice bottom
2456   } else if (trap_state == reason) {
2457     return trap_state + recompile_bit;     // the condition is already true
2458   } else if (trap_state == 0) {
2459     return reason + recompile_bit;          // no condition has yet been true
2460   } else {
2461     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2462   }
2463 }
2464 //-----------------------trap_state_is_recompiled------------------------------
2465 bool Deoptimization::trap_state_is_recompiled(int trap_state) {
2466   return (trap_state & DS_RECOMPILE_BIT) != 0;
2467 }
2468 //-----------------------trap_state_set_recompiled-----------------------------
2469 int Deoptimization::trap_state_set_recompiled(int trap_state, bool z) {
2470   if (z)  return trap_state |  DS_RECOMPILE_BIT;
2471   else    return trap_state & ~DS_RECOMPILE_BIT;
2472 }

2609 void Deoptimization::gather_statistics(DeoptReason reason, DeoptAction action,
2610                                        Bytecodes::Code bc) {
2611   assert(reason >= 0 && reason < Reason_LIMIT, "oob");
2612   assert(action >= 0 && action < Action_LIMIT, "oob");
2613   _deoptimization_hist[Reason_none][0][0] += 1;  // total
2614   _deoptimization_hist[reason][0][0]      += 1;  // per-reason total
2615   juint* cases = _deoptimization_hist[reason][1+action];
2616   juint* bc_counter_addr = NULL;
2617   juint  bc_counter      = 0;
2618   // Look for an unused counter, or an exact match to this BC.
2619   if (bc != Bytecodes::_illegal) {
2620     for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2621       juint* counter_addr = &cases[bc_case];
2622       juint  counter = *counter_addr;
2623       if ((counter == 0 && bc_counter_addr == NULL)
2624           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2625         // this counter is either free or is already devoted to this BC
2626         bc_counter_addr = counter_addr;
2627         bc_counter = counter | bc;
2628       }
2629     }
2630   }
2631   if (bc_counter_addr == NULL) {
2632     // Overflow, or no given bytecode.
2633     bc_counter_addr = &cases[BC_CASE_LIMIT-1];
2634     bc_counter = (*bc_counter_addr & ~LSB_MASK);  // clear LSB
2635   }

2643 void Deoptimization::print_statistics() {
2644   juint total = total_deoptimization_count();
2645   juint account = total;
2646   if (total != 0) {
2647     ttyLocker ttyl;
2648     if (xtty != NULL)  xtty->head("statistics type='deoptimization'");
2649     tty->print_cr("Deoptimization traps recorded:");
2650     #define PRINT_STAT_LINE(name, r) \
2651       tty->print_cr("  %4d (%4.1f%%) %s", (int)(r), ((r) * 100.0) / total, name);
2652     PRINT_STAT_LINE("total", total);
2653     // For each non-zero entry in the histogram, print the reason,
2654     // the action, and (if specifically known) the type of bytecode.
2655     for (int reason = 0; reason < Reason_LIMIT; reason++) {
2656       for (int action = 0; action < Action_LIMIT; action++) {
2657         juint* cases = _deoptimization_hist[reason][1+action];
2658         for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2659           juint counter = cases[bc_case];
2660           if (counter != 0) {
2661             char name[1*K];
2662             Bytecodes::Code bc = (Bytecodes::Code)(counter & LSB_MASK);
2663             if (bc_case == BC_CASE_LIMIT && (int)bc == 0)
2664               bc = Bytecodes::_illegal;
2665             sprintf(name, "%s/%s/%s",
2666                     trap_reason_name(reason),
2667                     trap_action_name(action),
2668                     Bytecodes::is_defined(bc)? Bytecodes::name(bc): "other");
2669             juint r = counter >> LSB_BITS;
2670             tty->print_cr("  %40s: " UINT32_FORMAT " (%.1f%%)", name, r, (r * 100.0) / total);
2671             account -= r;
2672           }
2673         }
2674       }
2675     }
2676     if (account != 0) {
2677       PRINT_STAT_LINE("unaccounted", account);
2678     }
2679     #undef PRINT_STAT_LINE
2680     if (xtty != NULL)  xtty->tail("statistics");
2681   }
