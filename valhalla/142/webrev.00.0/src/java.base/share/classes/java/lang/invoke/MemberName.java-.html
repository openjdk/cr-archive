  46 /**
  47  * A {@code MemberName} is a compact symbolic datum which fully characterizes
  48  * a method or field reference.
  49  * A member name refers to a field, method, constructor, or member type.
  50  * Every member name has a simple name (a string) and a type (either a Class or MethodType).
  51  * A member name may also have a non-null declaring class, or it may be simply
  52  * a naked name/type pair.
  53  * A member name may also have non-zero modifier flags.
  54  * Finally, a member name may be either resolved or unresolved.
  55  * If it is resolved, the existence of the named member has been determined by the JVM.
  56  * <p>
  57  * Whether resolved or not, a member name provides no access rights or
  58  * invocation capability to its possessor.  It is merely a compact
  59  * representation of all symbolic information necessary to link to
  60  * and properly use the named member.
  61  * <p>
  62  * When resolved, a member name's internal implementation may include references to JVM metadata.
  63  * This representation is stateless and only descriptive.
  64  * It provides no private information and no capability to use the member.
  65  * <p>
  66  * By contrast, a {@linkplain java.lang.reflect.Method} contains fuller information
  67  * about the internals of a method (except its bytecodes) and also
  68  * allows invocation.  A MemberName is much lighter than a Method,
  69  * since it contains about 7 fields to the 16 of Method (plus its sub-arrays),
  70  * and those seven fields omit much of the information in Method.
  71  * @author jrose
  72  */
  73 /*non-public*/
  74 final class ResolvedMethodName {
  75     //@Injected JVM_Method* vmtarget;
  76     //@Injected Class<?>    vmholder;
  77 };

 126     /** Return the declared type of this member, which
 127      *  must be a method or constructor.
 128      */
 129     public MethodType getMethodType() {
 130         if (type == null) {
 131             expandFromVM();
 132             if (type == null) {
 133                 return null;
 134             }
 135         }
 136         if (!isInvocable()) {
 137             throw newIllegalArgumentException("not invocable, no method type");
 138         }
 139 
 140         {
 141             // Get a snapshot of type which doesn't get changed by racing threads.
 142             final Object type = this.type;
 143             if (type instanceof MethodType) {
 144                 return (MethodType) type;
 145             }
 146         }
 147 
 148         // type is not a MethodType yet.  Convert it thread-safely.
 149         synchronized (this) {
 150             if (type instanceof String) {
 151                 String sig = (String) type;
 152                 MethodType res = MethodType.fromDescriptor(sig, getClassLoader());
 153                 type = res;
 154             } else if (type instanceof Object[]) {
 155                 Object[] typeInfo = (Object[]) type;
 156                 Class<?>[] ptypes = (Class<?>[]) typeInfo[1];
 157                 Class<?> rtype = (Class<?>) typeInfo[0];
 158                 MethodType res = MethodType.makeImpl(rtype, ptypes, true);
 159                 type = res;
 160             }
 161             // Make sure type is a MethodType for racing threads.
 162             assert type instanceof MethodType : "bad method type " + type;
 163         }
 164         return (MethodType) type;
 165     }

 213     /** Return the declared type of this member, which
 214      *  must be a field or type.
 215      *  If it is a type member, that type itself is returned.
 216      */
 217     public Class<?> getFieldType() {
 218         if (type == null) {
 219             expandFromVM();
 220             if (type == null) {
 221                 return null;
 222             }
 223         }
 224         if (isInvocable()) {
 225             throw newIllegalArgumentException("not a field or nested class, no simple type");
 226         }
 227 
 228         {
 229             // Get a snapshot of type which doesn't get changed by racing threads.
 230             final Object type = this.type;
 231             if (type instanceof Class<?>) {
 232                 return (Class<?>) type;
 233             }
 234         }
 235 
 236         // type is not a Class yet.  Convert it thread-safely.
 237         synchronized (this) {
 238             if (type instanceof String) {
 239                 String sig = (String) type;
 240                 MethodType mtype = MethodType.fromDescriptor("()"+sig, getClassLoader());
 241                 Class<?> res = mtype.returnType();
 242                 type = res;
 243             }
 244             // Make sure type is a Class for racing threads.
 245             assert type instanceof Class<?> : "bad field type " + type;
 246         }
 247         return (Class<?>) type;
 248     }

 283     private boolean referenceKindIsConsistent() {
 284         byte refKind = getReferenceKind();
 285         if (refKind == REF_NONE)  return isType();
 286         if (isField()) {
 287             assert(staticIsConsistent());
 288             assert(MethodHandleNatives.refKindIsField(refKind));
 289         } else if (isObjectConstructor()) {
 290             assert(refKind == REF_newInvokeSpecial || refKind == REF_invokeSpecial);
 291         } else if (isMethod()) {
 292             assert(staticIsConsistent());
 293             assert(MethodHandleNatives.refKindIsMethod(refKind));
 294             if (clazz.isInterface())
 295                 assert(refKind == REF_invokeInterface ||
 296                        refKind == REF_invokeStatic    ||
 297                        refKind == REF_invokeSpecial   ||
 298                        refKind == REF_invokeVirtual && isObjectPublicMethod());
 299         } else {
 300             assert(false);
 301         }
 302         return true;
 303     }
 304     private boolean isObjectPublicMethod() {
 305         if (clazz == Object.class)  return true;
 306         MethodType mtype = getMethodType();
 307         if (name.equals("toString") && mtype.returnType() == String.class && mtype.parameterCount() == 0)
 308             return true;
 309         if (name.equals("hashCode") && mtype.returnType() == int.class && mtype.parameterCount() == 0)
 310             return true;
 311         if (name.equals("equals") && mtype.returnType() == boolean.class && mtype.parameterCount() == 1 && mtype.parameterType(0) == Object.class)
 312             return true;
 313         return false;
 314     }

 316     /*non-public*/
 317     boolean referenceKindIsConsistentWith(int originalRefKind) {
 318         int refKind = getReferenceKind();
 319         if (refKind == originalRefKind)  return true;
 320         switch (originalRefKind) {
 321         case REF_invokeInterface:
 322             // Looking up an interface method, can get (e.g.) Object.hashCode
 323             assert(refKind == REF_invokeVirtual ||
 324                    refKind == REF_invokeSpecial) : this;
 325             return true;
 326         case REF_invokeVirtual:
 327         case REF_newInvokeSpecial:
 328             // Looked up a virtual, can get (e.g.) final String.hashCode.
 329             assert(refKind == REF_invokeSpecial) : this;
 330             return true;
 331         }
 332         assert(false) : this+" != "+MethodHandleNatives.refKindName((byte)originalRefKind);
 333         return true;
 334     }
 335     private boolean staticIsConsistent() {
 336         byte refKind = getReferenceKind();
 337         return MethodHandleNatives.refKindIsStatic(refKind) == isStatic() || getModifiers() == 0;
 338     }
 339     private boolean vminfoIsConsistent() {
 340         byte refKind = getReferenceKind();
 341         assert(isResolved());  // else don't call
 342         Object vminfo = MethodHandleNatives.getMemberVMInfo(this);
 343         assert(vminfo instanceof Object[]);
 344         long vmindex = (Long) ((Object[])vminfo)[0];
 345         Object vmtarget = ((Object[])vminfo)[1];
 346         if (MethodHandleNatives.refKindIsField(refKind)) {
 347             assert(vmindex >= 0) : vmindex + ":" + this;
 348             assert(vmtarget instanceof Class);
 349         } else {
 350             if (MethodHandleNatives.refKindDoesDispatch(refKind))
 351                 assert(vmindex >= 0) : vmindex + ":" + this;
 352             else
 353                 assert(vmindex < 0) : vmindex;
 354             assert(vmtarget instanceof MemberName) : vmtarget + " in " + this;
 355         }
 356         return true;
 357     }

 359     private MemberName changeReferenceKind(byte refKind, byte oldKind) {
 360         assert(getReferenceKind() == oldKind);
 361         assert(MethodHandleNatives.refKindIsValid(refKind));
 362         flags += (((int)refKind - oldKind) << MN_REFERENCE_KIND_SHIFT);
 363         return this;
 364     }

 376     /** Utility method to query if this member is a method handle invocation (invoke or invokeExact).
 377      */
 378     public boolean isMethodHandleInvoke() {
 379         final int bits = MH_INVOKE_MODS &~ Modifier.PUBLIC;
 380         final int negs = Modifier.STATIC;
 381         if (testFlags(bits | negs, bits) &&
 382             clazz == MethodHandle.class) {
 383             return isMethodHandleInvokeName(name);
 384         }
 385         return false;
 386     }

 390         case "invokeExact":
 391             return true;

 393             return false;

 587     // Capturing information from the Core Reflection API:
 588     private static int flagsMods(int flags, int mods, byte refKind) {
 589         assert((flags & RECOGNIZED_MODIFIERS) == 0);
 590         assert((mods & ~RECOGNIZED_MODIFIERS) == 0);
 591         assert((refKind & ~MN_REFERENCE_KIND_MASK) == 0);
 592         return flags | mods | (refKind << MN_REFERENCE_KIND_SHIFT);
 593     }
 594     /** Create a name for the given reflected method.  The resulting name will be in a resolved state. */
 595     public MemberName(Method m) {
 596         this(m, false);
 597     }
 598     @SuppressWarnings("LeakingThisInConstructor")
 599     public MemberName(Method m, boolean wantSpecial) {
 600         Objects.requireNonNull(m);
 601         // fill in vmtarget, vmindex while we have m in hand:
 602         MethodHandleNatives.init(this, m);
 603         if (clazz == null) {  // MHN.init failed
 604             if (m.getDeclaringClass() == MethodHandle.class &&
 605                 isMethodHandleInvokeName(m.getName())) {
 606                 // The JVM did not reify this signature-polymorphic instance.
 607                 // Need a special case here.
 608                 // See comments on MethodHandleNatives.linkMethod.
 609                 MethodType type = MethodType.methodType(m.getReturnType(), m.getParameterTypes());
 610                 int flags = flagsMods(IS_METHOD, m.getModifiers(), REF_invokeVirtual);
 611                 init(MethodHandle.class, m.getName(), type, flags);
 612                 if (isMethodHandleInvoke())
 613                     return;
 614             }
 615             if (m.getDeclaringClass() == VarHandle.class &&
 616                 isVarHandleMethodInvokeName(m.getName())) {
 617                 // The JVM did not reify this signature-polymorphic instance.
 618                 // Need a special case here.
 619                 // See comments on MethodHandleNatives.linkMethod.
 620                 MethodType type = MethodType.methodType(m.getReturnType(), m.getParameterTypes());
 621                 int flags = flagsMods(IS_METHOD, m.getModifiers(), REF_invokeVirtual);
 622                 init(VarHandle.class, m.getName(), type, flags);
 623                 if (isVarHandleMethodInvoke())
 624                     return;
 625             }
 626             throw new LinkageError(m.toString());
 627         }
 628         assert(isResolved() && this.clazz != null);
 629         this.name = m.getName();
 630         if (this.type == null)
 631             this.type = new Object[] { m.getReturnType(), m.getParameterTypes() };
 632         if (wantSpecial) {
 633             if (isAbstract())
 634                 throw new AbstractMethodError(this.toString());
 635             if (getReferenceKind() == REF_invokeVirtual)
 636                 changeReferenceKind(REF_invokeSpecial, REF_invokeVirtual);

 661     /** If this MN is a REF_invokeSpecial, return a clone with the "normal" kind
 662      *  REF_invokeVirtual; also switch either to REF_invokeInterface if clazz.isInterface.
 663      *  The end result is to get a fully virtualized version of the MN.
 664      *  (Note that resolving in the JVM will sometimes devirtualize, changing
 665      *  REF_invokeVirtual of a final to REF_invokeSpecial, and REF_invokeInterface
 666      *  in some corner cases to either of the previous two; this transform
 667      *  undoes that change under the assumption that it occurred.)
 668      */
 669     public MemberName asNormalOriginal() {
 670         byte normalVirtual = clazz.isInterface() ? REF_invokeInterface : REF_invokeVirtual;
 671         byte refKind = getReferenceKind();
 672         byte newRefKind = refKind;
 673         MemberName result = this;
 674         switch (refKind) {
 675         case REF_invokeInterface:
 676         case REF_invokeVirtual:
 677         case REF_invokeSpecial:
 678             newRefKind = normalVirtual;
 679             break;
 680         }
 681         if (newRefKind == refKind)
 682             return this;
 683         result = clone().changeReferenceKind(newRefKind, refKind);
 684         assert(this.referenceKindIsConsistentWith(result.getReferenceKind()));
 685         return result;
 686     }

 894         if (isInvocable()) {
 895             MethodType type;
 896             if (this.type instanceof MethodType)
 897                 type = (MethodType) this.type;

 915     /** Produce a string form of this member name.
 916      *  For types, it is simply the type's own string (as reported by {@code toString}).
 917      *  For fields, it is {@code "DeclaringClass.name/type"}.
 918      *  For methods and constructors, it is {@code "DeclaringClass.name(ptype...)rtype"}.
 919      *  If the declaring class is null, the prefix {@code "DeclaringClass."} is omitted.
 920      *  If the member is unresolved, a prefix {@code "*."} is prepended.
 921      */
 922     @SuppressWarnings("LocalVariableHidesMemberVariable")
 923     @Override
 924     public String toString() {
 925         if (isType())
 926             return type.toString();  // class java.lang.String
 927         // else it is a field, method, or constructor
 928         StringBuilder buf = new StringBuilder();
 929         if (getDeclaringClass() != null) {
 930             buf.append(getName(clazz));
 931             buf.append('.');
 932         }
 933         String name = this.name; // avoid expanding from VM
 934         buf.append(name == null ? "*" : name);
 935         Object type = this.type; // avoid expanding from VM
 936         if (!isInvocable()) {
 937             buf.append('/');
 938             buf.append(type == null ? "*" : getName(type));
 939         } else {
 940             buf.append(type == null ? "(*)*" : getName(type));
 941         }
 942         byte refKind = getReferenceKind();
 943         if (refKind != REF_NONE) {
 944             buf.append('/');
 945             buf.append(MethodHandleNatives.refKindName(refKind));
 946         }
 947         //buf.append("#").append(System.identityHashCode(this));
 948         return buf.toString();
 949     }

 956     public IllegalAccessException makeAccessException(String message, Object from) {
 957         message = message + ": "+ toString();
 958         if (from != null)  {
 959             if (from == MethodHandles.publicLookup()) {
 960                 message += ", from public Lookup";
 961             } else {
 962                 Module m;
 963                 Class<?> plc;
 964                 if (from instanceof MethodHandles.Lookup) {
 965                     MethodHandles.Lookup lookup = (MethodHandles.Lookup)from;
 966                     from = lookup.lookupClass();
 967                     m = lookup.lookupClass().getModule();
 968                     plc = lookup.previousLookupClass();
 969                 } else {
 970                     m = ((Class<?>)from).getModule();
 971                     plc = null;
 972                 }
 973                 message += ", from " + from + " (" + m + ")";
 974                 if (plc != null) {
 975                     message += ", previous lookup " +
 976                         plc.getName() + " (" + plc.getModule() + ")";
 977                 }
 978             }
 979         }
 980         return new IllegalAccessException(message);
 981     }
 982     private String message() {
 983         if (isResolved())
 984             return "no access";
 985         else if (isObjectConstructor())
 986             return "no such constructor";
 987         else if (isMethod())
 988             return "no such method";
 989         else
 990             return "no such field";
 991     }
 992     public ReflectiveOperationException makeAccessException() {
 993         String message = message() + ": "+ toString();
 994         ReflectiveOperationException ex;
 995         if (isResolved() || !(resolution instanceof NoSuchMethodError ||
 996                               resolution instanceof NoSuchFieldError))
 997             ex = new IllegalAccessException(message);
 998         else if (isObjectConstructor())
 999             ex = new NoSuchMethodException(message);
1000         else if (isMethod())
1001             ex = new NoSuchMethodException(message);
1002         else
1003             ex = new NoSuchFieldException(message);
1004         if (resolution instanceof Throwable)
1005             ex.initCause((Throwable) resolution);
1006         return ex;
1007     }

1014     /** A factory type for resolving member names with the help of the VM.
1015      *  TBD: Define access-safe public constructors for this factory.
1016      */
1017     /*non-public*/
1018     static class Factory {
1019         private Factory() { } // singleton pattern
1020         static Factory INSTANCE = new Factory();

1022         private static int ALLOWED_FLAGS = ALL_KINDS;
1023 
1024         /// Queries
1025         List<MemberName> getMembers(Class<?> defc,
1026                 String matchName, Object matchType,
1027                 int matchFlags, Class<?> lookupClass) {
1028             matchFlags &= ALLOWED_FLAGS;
1029             String matchSig = null;
1030             if (matchType != null) {
1031                 matchSig = BytecodeDescriptor.unparse(matchType);
1032                 if (matchSig.startsWith("("))
1033                     matchFlags &= ~(ALL_KINDS & ~IS_INVOCABLE);
1034                 else
1035                     matchFlags &= ~(ALL_KINDS & ~IS_FIELD);
1036             }
1037             final int BUF_MAX = 0x2000;
1038             int len1 = matchName == null ? 10 : matchType == null ? 4 : 1;
1039             MemberName[] buf = newMemberBuffer(len1);
1040             int totalCount = 0;
1041             ArrayList<MemberName[]> bufs = null;
1042             int bufCount = 0;
1043             for (;;) {
1044                 bufCount = MethodHandleNatives.getMembers(defc,
1045                         matchName, matchSig, matchFlags,
1046                         lookupClass,
1047                         totalCount, buf);
1048                 if (bufCount <= buf.length) {
1049                     if (bufCount < 0)  bufCount = 0;
1050                     totalCount += bufCount;
1051                     break;
1052                 }
1053                 // JVM returned to us with an intentional overflow!
1054                 totalCount += buf.length;
1055                 int excess = bufCount - buf.length;
1056                 if (bufs == null)  bufs = new ArrayList<>(1);
1057                 bufs.add(buf);
1058                 int len2 = buf.length;
1059                 len2 = Math.max(len2, excess);
1060                 len2 = Math.max(len2, totalCount / 4);
1061                 buf = newMemberBuffer(Math.min(BUF_MAX, len2));
1062             }
1063             ArrayList<MemberName> result = new ArrayList<>(totalCount);
1064             if (bufs != null) {
1065                 for (MemberName[] buf0 : bufs) {
1066                     Collections.addAll(result, buf0);
1067                 }
1068             }
1069             for (int i = 0; i < bufCount; i++) {
1070                 result.add(buf[i]);
1071             }
1072             // Signature matching is not the same as type matching, since
1073             // one signature might correspond to several types.
1074             // So if matchType is a Class or MethodType, refilter the results.
1075             if (matchType != null && matchType != matchSig) {
1076                 for (Iterator<MemberName> it = result.iterator(); it.hasNext();) {
1077                     MemberName m = it.next();
1078                     if (!matchType.equals(m.getType()))
1079                         it.remove();
1080                 }
1081             }
1082             return result;
1083         }
1084         /** Produce a resolved version of the given member.
1085          *  Super types are searched (for inherited members) if {@code searchSupers} is true.
1086          *  Access checking is performed on behalf of the given {@code lookupClass}.
1087          *  If lookup fails or access is not permitted, null is returned.
1088          *  Otherwise a fresh copy of the given member is returned, with modifier bits filled in.
1089          */
1090         private MemberName resolve(byte refKind, MemberName ref, Class<?> lookupClass,
1091                                    boolean speculativeResolve) {
1092             MemberName m = ref.clone();  // JVM will side-effect the ref
1093             assert(refKind == m.getReferenceKind());
1094             try {
1095                 // There are 4 entities in play here:
1096                 //   * LC: lookupClass
1097                 //   * REFC: symbolic reference class (MN.clazz before resolution);
1098                 //   * DEFC: resolved method holder (MN.clazz after resolution);
1099                 //   * PTYPES: parameter types (MN.type)
1100                 //
1101                 // What we care about when resolving a MemberName is consistency between DEFC and PTYPES.
1102                 // We do type alias (TA) checks on DEFC to ensure that. DEFC is not known until the JVM
1103                 // finishes the resolution, so do TA checks right after MHN.resolve() is over.
1104                 //
1105                 // All parameters passed by a caller are checked against MH type (PTYPES) on every invocation,
1106                 // so it is safe to call a MH from any context.
1107                 //
1108                 // REFC view on PTYPES doesn't matter, since it is used only as a starting point for resolution and doesn't
1109                 // participate in method selection.
1110                 m = MethodHandleNatives.resolve(m, lookupClass, speculativeResolve);
1111                 if (m == null && speculativeResolve) {
1112                     return null;
1113                 }
1114                 m.checkForTypeAlias(m.getDeclaringClass());
1115                 m.resolution = null;
1116             } catch (ClassNotFoundException | LinkageError ex) {
1117                 // JVM reports that the "bytecode behavior" would get an error
1118                 assert(!m.isResolved());
1119                 m.resolution = ex;
1120                 return m;
1121             }
1122             assert(m.referenceKindIsConsistent());
1123             m.initResolved(true);
1124             assert(m.vminfoIsConsistent());
1125             return m;
1126         }
1127         /** Produce a resolved version of the given member.
1128          *  Super types are searched (for inherited members) if {@code searchSupers} is true.
1129          *  Access checking is performed on behalf of the given {@code lookupClass}.
1130          *  If lookup fails or access is not permitted, a {@linkplain ReflectiveOperationException} is thrown.
1131          *  Otherwise a fresh copy of the given member is returned, with modifier bits filled in.
1132          */
1133         public <NoSuchMemberException extends ReflectiveOperationException>
1134                 MemberName resolveOrFail(byte refKind, MemberName m, Class<?> lookupClass,
