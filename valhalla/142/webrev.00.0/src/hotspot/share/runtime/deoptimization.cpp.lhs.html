  84 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  85                                          int  caller_adjustment,
  86                                          int  caller_actual_parameters,
  87                                          int  number_of_frames,
  88                                          intptr_t* frame_sizes,
  89                                          address* frame_pcs,
  90                                          BasicType return_type,
  91                                          int exec_mode) {
  92   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  93   _caller_adjustment         = caller_adjustment;
  94   _caller_actual_parameters  = caller_actual_parameters;
  95   _number_of_frames          = number_of_frames;
  96   _frame_sizes               = frame_sizes;
  97   _frame_pcs                 = frame_pcs;
  98   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
  99   _return_type               = return_type;
 100   _initial_info              = 0;
 101   // PD (x86 only)
 102   _counter_temp              = 0;
 103   _unpack_kind               = exec_mode;
 104   _sender_sp_temp            = 0;
 105 
 106   _total_frame_sizes         = size_of_frames();
 107   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 108 }

 125 int Deoptimization::UnrollBlock::size_of_frames() const {
 126   // Acount first for the adjustment of the initial frame
 127   int result = _caller_adjustment;
 128   for (int index = 0; index < number_of_frames(); index++) {
 129     result += frame_sizes()[index];
 130   }
 131   return result;
 132 }

 147 // In order to make fetch_unroll_info work properly with escape
 148 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 149 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 150 // of previously eliminated objects occurs in realloc_objects, which is
 151 // called from the method fetch_unroll_info_helper below.
 152 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 153   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 154   // but makes the entry a little slower. There is however a little dance we have to
 155   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 156 
 157   // fetch_unroll_info() is called at the beginning of the deoptimization
 158   // handler. Note this fact before we start generating temporary frames
 159   // that can confuse an asynchronous stack walker. This counter is
 160   // decremented at the end of unpack_frames().
 161   if (TraceDeoptimization) {
 162     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 163   }
 164   thread->inc_in_deopt_handler();
 165 
 166   return fetch_unroll_info_helper(thread, exec_mode);
 167 JRT_END

 169 #if COMPILER2_OR_JVMCI
 170 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 171                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 172   bool realloc_failures = false;
 173   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 174 
 175   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 176 
 177   // The flag return_oop() indicates call sites which return oop
 178   // in compiled code. Such sites include java method calls,
 179   // runtime calls (for example, used to allocate new objects/arrays
 180   // on slow code path) and any other calls generated in compiled code.
 181   // It is not guaranteed that we can get such information here only
 182   // by analyzing bytecode in deoptimized frames. This is why this flag
 183   // is set during method compilation (see Compile::Process_OopMap_Node()).
 184   // If the previous frame was popped or if we are dispatching an exception,
 185   // we don't have an oop result.
 186   ScopeDesc* scope = chunk->at(0)->scope();
 187   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 188   // In case of the return of multiple values, we must take care
 189   // of all oop return values.
 190   GrowableArray<Handle> return_oops;
 191   InlineKlass* vk = NULL;
 192   if (save_oop_result && scope->return_vt()) {
 193     vk = InlineKlass::returned_inline_klass(map);
 194     if (vk != NULL) {
 195       vk->save_oop_fields(map, return_oops);
 196       save_oop_result = false;
 197     }
 198   }
 199   if (save_oop_result) {
 200     // Reallocation may trigger GC. If deoptimization happened on return from
 201     // call which returns oop we need to save it since it is not in oopmap.
 202     oop result = deoptee.saved_oop_result(&map);
 203     assert(oopDesc::is_oop_or_null(result), "must be oop");
 204     return_oops.push(Handle(thread, result));
 205     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 206     if (TraceDeoptimization) {
 207       ttyLocker ttyl;
 208       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 209     }
 210   }
 211   if (objects != NULL || vk != NULL) {
 212     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 213     JRT_BLOCK
 214       if (vk != NULL) {
 215         realloc_failures = Deoptimization::realloc_inline_type_result(vk, map, return_oops, THREAD);
 216       }
 217       if (objects != NULL) {
 218         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 219         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 220       }
 221     JRT_END
 222 #ifndef PRODUCT
 223     if (TraceDeoptimization) {
 224       ttyLocker ttyl;
 225       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 226       if (objects != NULL) {
 227         Deoptimization::print_objects(objects, realloc_failures);
 228       } else {
 229         Handle obj = realloc_failures ? Handle() : return_oops.first();
 230         Deoptimization::print_object(vk, obj, realloc_failures);
 231       }
 232     }
 233 #endif
 234   }
 235   if (save_oop_result || vk != NULL) {
 236     // Restore result.
 237     assert(return_oops.length() == 1, "no inline type");
 238     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 239   }
 240   return realloc_failures;
 241 }

 243 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 244 #ifndef PRODUCT
 245   bool first = true;
 246 #endif
 247   for (int i = 0; i < chunk->length(); i++) {
 248     compiledVFrame* cvf = chunk->at(i);
 249     assert (cvf->scope() != NULL,"expect only compiled java frames");
 250     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 251     if (monitors->is_nonempty()) {
 252       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 253 #ifndef PRODUCT
 254       if (PrintDeoptimizationDetails) {
 255         ttyLocker ttyl;
 256         for (int j = 0; j < monitors->length(); j++) {
 257           MonitorInfo* mi = monitors->at(j);
 258           if (mi->eliminated()) {
 259             if (first) {
 260               first = false;
 261               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 262             }
 263             if (mi->owner_is_scalar_replaced()) {
 264               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 265               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 266             } else {
 267               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 268             }
 269           }
 270         }
 271       }
 272 #endif // !PRODUCT
 273     }
 274   }
 275 }
 276 #endif // COMPILER2_OR_JVMCI

 278 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 279 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 280 
 281   // Note: there is a safepoint safety issue here. No matter whether we enter
 282   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 283   // the vframeArray is created.
 284   //
 285 
 286   // Allocate our special deoptimization ResourceMark
 287   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 288   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 289   thread->set_deopt_mark(dmark);
 290 
 291   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 292   RegisterMap map(thread, true);
 293   RegisterMap dummy_map(thread, false);
 294   // Now get the deoptee with a valid map
 295   frame deoptee = stub_frame.sender(&map);
 296   // Set the deoptee nmethod
 297   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 298   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 299   thread->set_deopt_compiled_method(cm);
 300 
 301   if (VerifyStack) {
 302     thread->validate_frame_layout();
 303   }
 304 
 305   // Create a growable array of VFrames where each VFrame represents an inlined
 306   // Java frame.  This storage is allocated with the usual system arena.
 307   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 308   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 309   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 310   while (!vf->is_top()) {
 311     assert(vf->is_compiled_frame(), "Wrong frame type");
 312     chunk->push(compiledVFrame::cast(vf));
 313     vf = vf->sender();
 314   }
 315   assert(vf->is_compiled_frame(), "Wrong frame type");
 316   chunk->push(compiledVFrame::cast(vf));
 317 
 318   bool realloc_failures = false;
 319 
 320 #if COMPILER2_OR_JVMCI
 321 #if INCLUDE_JVMCI
 322   bool jvmci_enabled = true;
 323 #else
 324   bool jvmci_enabled = false;
 325 #endif
 326 
 327   // Reallocate the non-escaping objects and restore their fields. Then
 328   // relock objects if synchronization on them was eliminated.
 329   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 330     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 331   }
 332 #endif // COMPILER2_OR_JVMCI
 333 
 334   // Revoke biases, done with in java state.
 335   // No safepoints allowed after this
 336   revoke_from_deopt_handler(thread, deoptee, &map);
 337 
 338   // Ensure that no safepoint is taken after pointers have been stored
 339   // in fields of rematerialized objects.  If a safepoint occurs from here on
 340   // out the java state residing in the vframeArray will be missed.
 341   // Locks may be rebaised in a safepoint.
 342   NoSafepointVerifier no_safepoint;
 343 
 344 #if COMPILER2_OR_JVMCI
 345   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 346     eliminate_locks(thread, chunk, realloc_failures);
 347   }
 348 #endif // COMPILER2_OR_JVMCI
 349 
 350   ScopeDesc* trap_scope = chunk->at(0)->scope();
 351   Handle exceptionObject;
 352   if (trap_scope->rethrow_exception()) {
 353     if (PrintDeoptimizationDetails) {
 354       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 355     }
 356     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 357     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 358     ScopeValue* topOfStack = expressions->top();
 359     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 360     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 361   }
 362 
 363   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 364 #if COMPILER2_OR_JVMCI
 365   if (realloc_failures) {
 366     pop_frames_failed_reallocs(thread, array);
 367   }
 368 #endif
 369 
 370   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 371   thread->set_vframe_array_head(array);
 372 
 373   // Now that the vframeArray has been created if we have any deferred local writes
 374   // added by jvmti then we can free up that structure as the data is now in the
 375   // vframeArray
 376 
 377   if (thread->deferred_locals() != NULL) {
 378     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 379     int i = 0;
 380     do {
 381       // Because of inlining we could have multiple vframes for a single frame
 382       // and several of the vframes could have deferred writes. Find them all.
 383       if (list->at(i)->id() == array->original().id()) {
 384         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 385         list->remove_at(i);
 386         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 387         delete dlv;
 388       } else {
 389         i++;
 390       }
 391     } while ( i < list->length() );
 392     if (list->length() == 0) {
 393       thread->set_deferred_locals(NULL);
 394       // free the list and elements back to C heap.
 395       delete list;
 396     }
 397 
 398   }
 399 
 400   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 401   CodeBlob* cb = stub_frame.cb();
 402   // Verify we have the right vframeArray
 403   assert(cb->frame_size() >= 0, "Unexpected frame size");
 404   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 405 
 406   // If the deopt call site is a MethodHandle invoke call site we have
 407   // to adjust the unpack_sp.
 408   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 409   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 410     unpack_sp = deoptee.unextended_sp();
 411 
 412 #ifdef ASSERT
 413   assert(cb->is_deoptimization_stub() ||
 414          cb->is_uncommon_trap_stub() ||
 415          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 416          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 417          "unexpected code blob: %s", cb->name());
 418 #endif
 419 
 420   // This is a guarantee instead of an assert because if vframe doesn't match
 421   // we will unpack the wrong deoptimized frame and wind up in strange places
 422   // where it will be very difficult to figure out what went wrong. Better
 423   // to die an early death here than some very obscure death later when the
 424   // trail is cold.
 425   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 426   // in that it will fail to detect a problem when there is one. This needs
 427   // more work in tiger timeframe.
 428   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 429 
 430   int number_of_frames = array->frames();
 431 
 432   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 433   // virtual activation, which is the reverse of the elements in the vframes array.
 434   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 435   // +1 because we always have an interpreter return address for the final slot.
 436   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 437   int popframe_extra_args = 0;
 438   // Create an interpreter return address for the stub to use as its return
 439   // address so the skeletal frames are perfectly walkable
 440   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 441 
 442   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 443   // activation be put back on the expression stack of the caller for reexecution
 444   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 445     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 446   }
 447 
 448   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 449   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 450   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 451   //
 452   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 453   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 454 
 455   // It's possible that the number of parameters at the call site is
 456   // different than number of arguments in the callee when method
 457   // handles are used.  If the caller is interpreted get the real
 458   // value so that the proper amount of space can be added to it's
 459   // frame.
 460   bool caller_was_method_handle = false;
 461   if (deopt_sender.is_interpreted_frame()) {
 462     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 463     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 464     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 465       // Method handle invokes may involve fairly arbitrary chains of
 466       // calls so it's impossible to know how much actual space the
 467       // caller has for locals.
 468       caller_was_method_handle = true;
 469     }
 470   }
 471 
 472   //
 473   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 474   // frame_sizes/frame_pcs[1] next oldest frame (int)
 475   // frame_sizes/frame_pcs[n] youngest frame (int)
 476   //
 477   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 478   // owns the space for the return address to it's caller).  Confusing ain't it.
 479   //
 480   // The vframe array can address vframes with indices running from
 481   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 482   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 483   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 484   // so things look a little strange in this loop.
 485   //
 486   int callee_parameters = 0;
 487   int callee_locals = 0;
 488   for (int index = 0; index < array->frames(); index++ ) {
 489     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 490     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 491     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 492     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 493                                                                                                     callee_locals,
 494                                                                                                     index == 0,
 495                                                                                                     popframe_extra_args);
 496     // This pc doesn't have to be perfect just good enough to identify the frame
 497     // as interpreted so the skeleton frame will be walkable
 498     // The correct pc will be set when the skeleton frame is completely filled out
 499     // The final pc we store in the loop is wrong and will be overwritten below
 500     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 501 
 502     callee_parameters = array->element(index)->method()->size_of_parameters();
 503     callee_locals = array->element(index)->method()->max_locals();
 504     popframe_extra_args = 0;
 505   }
 506 
 507   // Compute whether the root vframe returns a float or double value.
 508   BasicType return_type;
 509   {
 510     methodHandle method(thread, array->element(0)->method());
 511     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 512     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 513   }
 514 
 515   // Compute information for handling adapters and adjusting the frame size of the caller.
 516   int caller_adjustment = 0;
 517 
 518   // Compute the amount the oldest interpreter frame will have to adjust
 519   // its caller's stack by. If the caller is a compiled frame then
 520   // we pretend that the callee has no parameters so that the
 521   // extension counts for the full amount of locals and not just
 522   // locals-parms. This is because without a c2i adapter the parm
 523   // area as created by the compiled frame will not be usable by
 524   // the interpreter. (Depending on the calling convention there
 525   // may not even be enough space).
 526 
 527   // QQQ I'd rather see this pushed down into last_frame_adjust
 528   // and have it take the sender (aka caller).
 529 
 530   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 531     caller_adjustment = last_frame_adjust(0, callee_locals);
 532   } else if (callee_locals > callee_parameters) {
 533     // The caller frame may need extending to accommodate
 534     // non-parameter locals of the first unpacked interpreted frame.
 535     // Compute that adjustment.
 536     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 537   }
 538 
 539   // If the sender is deoptimized we must retrieve the address of the handler
 540   // since the frame will "magically" show the original pc before the deopt
 541   // and we'd undo the deopt.
 542 
 543   frame_pcs[0] = deopt_sender.raw_pc();
 544 
 545   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 546 
 547 #if INCLUDE_JVMCI
 548   if (exceptionObject() != NULL) {
 549     thread->set_exception_oop(exceptionObject());
 550     exec_mode = Unpack_exception;
 551   }
 552 #endif
 553 
 554   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 555     assert(thread->has_pending_exception(), "should have thrown OOME");
 556     thread->set_exception_oop(thread->pending_exception());
 557     thread->clear_pending_exception();
 558     exec_mode = Unpack_exception;
 559   }
 560 
 561 #if INCLUDE_JVMCI
 562   if (thread->frames_to_pop_failed_realloc() > 0) {
 563     thread->set_pending_monitorenter(false);
 564   }
 565 #endif
 566 
 567   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 568                                       caller_adjustment * BytesPerWord,
 569                                       caller_was_method_handle ? 0 : callee_parameters,
 570                                       number_of_frames,
 571                                       frame_sizes,
 572                                       frame_pcs,
 573                                       return_type,
 574                                       exec_mode);
 575   // On some platforms, we need a way to pass some platform dependent
 576   // information to the unpacking code so the skeletal frames come out
 577   // correct (initial fp value, unextended sp, ...)
 578   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 579 
 580   if (array->frames() > 1) {
 581     if (VerifyStack && TraceDeoptimization) {
 582       ttyLocker ttyl;
 583       tty->print_cr("Deoptimizing method containing inlining");
 584     }
 585   }
 586 
 587   array->set_unroll_block(info);
 588   return info;
 589 }

 593 void Deoptimization::cleanup_deopt_info(JavaThread *thread,
 594                                         vframeArray *array) {
 595 
 596   // Get array if coming from exception
 597   if (array == NULL) {
 598     array = thread->vframe_array_head();
 599   }
 600   thread->set_vframe_array_head(NULL);
 601 
 602   // Free the previous UnrollBlock
 603   vframeArray* old_array = thread->vframe_array_last();
 604   thread->set_vframe_array_last(array);
 605 
 606   if (old_array != NULL) {
 607     UnrollBlock* old_info = old_array->unroll_block();

 609     delete old_info;

 611   }

 614   // inside the vframeArray (StackValueCollections)

 624     // popframe condition bit set, we should always clear it now
 625     thread->clear_popframe_condition();

 638   // the beginning of uncommon_trap().
 639   thread->dec_in_deopt_handler();

 643 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 644 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 645 
 646   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 647   // the days we had adapter frames. When we deoptimize a situation where a
 648   // compiled caller calls a compiled caller will have registers it expects
 649   // to survive the call to the callee. If we deoptimize the callee the only
 650   // way we can restore these registers is to have the oldest interpreter
 651   // frame that we create restore these values. That is what this routine
 652   // will accomplish.
 653 
 654   // At the moment we have modified c2 to not have any callee save registers
 655   // so this problem does not exist and this routine is just a place holder.
 656 
 657   assert(f->is_interpreted_frame(), "must be interpreted");
 658 }

 660 // Return BasicType of value being returned
 661 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 662 
 663   // We are already active in the special DeoptResourceMark any ResourceObj's we
 664   // allocate will be freed at the end of the routine.
 665 
 666   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 667   // but makes the entry a little slower. There is however a little dance we have to
 668   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 669   ResetNoHandleMark rnhm; // No-op in release/product versions
 670   HandleMark hm;
 671 
 672   frame stub_frame = thread->last_frame();
 673 
 674   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 675   // must point to the vframeArray for the unpack frame.
 676   vframeArray* array = thread->vframe_array_head();
 677 
 678 #ifndef PRODUCT
 679   if (TraceDeoptimization) {
 680     ttyLocker ttyl;
 681     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 682                   p2i(thread), p2i(array), exec_mode);
 683   }
 684 #endif
 685   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 686               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 687 
 688   UnrollBlock* info = array->unroll_block();
 689 
 690   // We set the last_Java frame. But the stack isn't really parsable here. So we
 691   // clear it to make sure JFR understands not to try and walk stacks from events
 692   // in here.
 693   intptr_t* sp = thread->frame_anchor()->last_Java_sp();
 694   thread->frame_anchor()->set_last_Java_sp(NULL);
 695 
 696   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 697   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 698 
 699   thread->frame_anchor()->set_last_Java_sp(sp);
 700 
 701   BasicType bt = info->return_type();
 702 
 703   // If we have an exception pending, claim that the return type is an oop
 704   // so the deopt_blob does not overwrite the exception_oop.
 705 
 706   if (exec_mode == Unpack_exception)
 707     bt = T_OBJECT;
 708 
 709   // Cleanup thread deopt data
 710   cleanup_deopt_info(thread, array);
 711 
 712 #ifndef PRODUCT
 713   if (VerifyStack) {
 714     ResourceMark res_mark;
 715     // Clear pending exception to not break verification code (restored afterwards)
 716     PRESERVE_EXCEPTION_MARK;
 717 
 718     thread->validate_frame_layout();
 719 
 720     // Verify that the just-unpacked frames match the interpreter's
 721     // notions of expression stack and locals
 722     vframeArray* cur_array = thread->vframe_array_last();
 723     RegisterMap rm(thread, false);
 724     rm.set_include_argument_oops(false);
 725     bool is_top_frame = true;
 726     int callee_size_of_parameters = 0;
 727     int callee_max_locals = 0;
 728     for (int i = 0; i < cur_array->frames(); i++) {
 729       vframeArrayElement* el = cur_array->element(i);
 730       frame* iframe = el->iframe();
 731       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 732 
 733       // Get the oop map for this bci
 734       InterpreterOopMap mask;
 735       int cur_invoke_parameter_size = 0;
 736       bool try_next_mask = false;
 737       int next_mask_expression_stack_size = -1;
 738       int top_frame_expression_stack_adjustment = 0;
 739       methodHandle mh(thread, iframe->interpreter_frame_method());
 740       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 741       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 742       int max_bci = mh->code_size();
 743       // Get to the next bytecode if possible
 744       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 745       // Check to see if we can grab the number of outgoing arguments
 746       // at an uncommon trap for an invoke (where the compiler
 747       // generates debug info before the invoke has executed)
 748       Bytecodes::Code cur_code = str.next();
 749       if (Bytecodes::is_invoke(cur_code)) {
 750         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 751         cur_invoke_parameter_size = invoke.size_of_parameters();
 752         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 753           callee_size_of_parameters++;
 754         }
 755       }
 756       if (str.bci() < max_bci) {
 757         Bytecodes::Code next_code = str.next();
 758         if (next_code >= 0) {
 759           // The interpreter oop map generator reports results before
 760           // the current bytecode has executed except in the case of
 761           // calls. It seems to be hard to tell whether the compiler
 762           // has emitted debug information matching the "state before"
 763           // a given bytecode or the state after, so we try both
 764           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 765             // Get expression stack size for the next bytecode
 766             InterpreterOopMap next_mask;
 767             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 768             next_mask_expression_stack_size = next_mask.expression_stack_size();
 769             if (Bytecodes::is_invoke(next_code)) {
 770               Bytecode_invoke invoke(mh, str.bci());
 771               next_mask_expression_stack_size += invoke.size_of_parameters();
 772             }
 773             // Need to subtract off the size of the result type of
 774             // the bytecode because this is not described in the
 775             // debug info but returned to the interpreter in the TOS
 776             // caching register
 777             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 778             if (bytecode_result_type != T_ILLEGAL) {
 779               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 780             }
 781             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 782             try_next_mask = true;
 783           }
 784         }
 785       }
 786 
 787       // Verify stack depth and oops in frame
 788       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 789       if (!(
 790             /* SPARC */
 791             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 792             /* x86 */
 793             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 794             (try_next_mask &&
 795              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 796                                                                     top_frame_expression_stack_adjustment))) ||
 797             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 798             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 799              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 800             )) {
 801         {
 802           ttyLocker ttyl;
 803 
 804           // Print out some information that will help us debug the problem
 805           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 806           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 807           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 808                         iframe->interpreter_frame_expression_stack_size());
 809           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 810           tty->print_cr("  try_next_mask = %d", try_next_mask);
 811           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 812           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 813           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 814           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 815           tty->print_cr("  exec_mode = %d", exec_mode);
 816           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 817           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 818           tty->print_cr("  Interpreted frames:");
 819           for (int k = 0; k < cur_array->frames(); k++) {
 820             vframeArrayElement* el = cur_array->element(k);
 821             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 822           }
 823           cur_array->print_on_2(tty);
 824         } // release tty lock before calling guarantee
 825         guarantee(false, "wrong number of expression stack elements during deopt");
 826       }
 827       VerifyOopClosure verify;
 828       iframe->oops_interpreted_do(&verify, &rm, false);
 829       callee_size_of_parameters = mh->size_of_parameters();
 830       callee_max_locals = mh->max_locals();
 831       is_top_frame = false;
 832     }
 833   }
 834 #endif /* !PRODUCT */
 835 
 836   return bt;
 837 JRT_END

1019     oop obj = NULL;

1022 #if INCLUDE_JVMCI || INCLUDE_AOT
1023       CompiledMethod* cm = fr->cb()->as_compiled_method_or_null();

1028           // Set the flag to indicate the box came from a cache, so that we can skip the field reassignment for it.
1029           abv->set_cached(true);
1030         }
1031       }
1032 #endif // INCLUDE_JVMCI || INCLUDE_AOT
1033       InstanceKlass* ik = InstanceKlass::cast(k);
1034       if (obj == NULL) {
1035         obj = ik->allocate_instance(THREAD);
1036       }
1037     } else if (k->is_flatArray_klass()) {
1038       FlatArrayKlass* ak = FlatArrayKlass::cast(k);
1039       // Inline type array must be zeroed because not all memory is reassigned
1040       obj = ak->allocate(sv->field_size(), THREAD);
1041     } else if (k->is_typeArray_klass()) {
1042       TypeArrayKlass* ak = TypeArrayKlass::cast(k);
1043       assert(sv->field_size() % type2size[ak->element_type()] == 0, "non-integral array length");
1044       int len = sv->field_size() / type2size[ak->element_type()];
1045       obj = ak->allocate(len, THREAD);
1046     } else if (k->is_objArray_klass()) {
1047       ObjArrayKlass* ak = ObjArrayKlass::cast(k);
1048       obj = ak->allocate(sv->field_size(), THREAD);
1049     }
1050 
1051     if (obj == NULL) {
1052       failures = true;
1053     }
1054 
1055     assert(sv->value().is_null(), "redundant reallocation");
1056     assert(obj != NULL || HAS_PENDING_EXCEPTION, "allocation should succeed or we should get an exception");
1057     CLEAR_PENDING_EXCEPTION;
1058     sv->set_value(obj);
1059   }
1060 
1061   if (failures) {
1062     THROW_OOP_(Universe::out_of_memory_error_realloc_objects(), failures);
1063   } else if (pending_exception.not_null()) {
1064     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1065   }
1066 
1067   return failures;
1068 }

1086 /**
1087  * For primitive types whose kind gets "erased" at runtime (shorts become stack ints),
1088  * we need to somehow be able to recover the actual kind to be able to write the correct
1089  * amount of bytes.
1090  * For that purpose, this method assumes that, for an entry spanning n bytes at index i,
1091  * the entries at index n + 1 to n + i are 'markers'.
1092  * For example, if we were writing a short at index 4 of a byte array of size 8, the
1093  * expected form of the array would be:
1094  *
1095  * {b0, b1, b2, b3, INT, marker, b6, b7}
1096  *
1097  * Thus, in order to get back the size of the entry, we simply need to count the number
1098  * of marked entries
1099  *
1100  * @param virtualArray the virtualized byte array
1101  * @param i index of the virtual entry we are recovering
1102  * @return The number of bytes the entry spans
1103  */
1104 static int count_number_of_bytes_for_entry(ObjectValue *virtualArray, int i) {
1105   int index = i;
1106   while (++index < virtualArray->field_size() &&
1107            virtualArray->field_at(index)->is_marker()) {}
1108   return index - i;
1109 }

1111 /**
1112  * If there was a guarantee for byte array to always start aligned to a long, we could
1113  * do a simple check on the parity of the index. Unfortunately, that is not always the
1114  * case. Thus, we check alignment of the actual address we are writing to.
1115  * In the unlikely case index 0 is 5-aligned for example, it would then be possible to
1116  * write a long to index 3.
1117  */

1119     jbyte* res = obj->byte_at_addr(index);
1120     assert((((intptr_t) res) % expected_alignment) == 0, "Non-aligned write");

1135     case 8:

1145 // restore elements of an eliminated type array
1146 void Deoptimization::reassign_type_array_elements(frame* fr, RegisterMap* reg_map, ObjectValue* sv, typeArrayOop obj, BasicType type) {
1147   int index = 0;
1148   intptr_t val;
1149 
1150   for (int i = 0; i < sv->field_size(); i++) {
1151     StackValue* value = StackValue::create_stack_value(fr, reg_map, sv->field_at(i));
1152     switch(type) {
1153     case T_LONG: case T_DOUBLE: {
1154       assert(value->type() == T_INT, "Agreement.");
1155       StackValue* low =
1156         StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1157 #ifdef _LP64
1158       jlong res = (jlong)low->get_int();
1159 #else
1160       jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1161 #endif
1162       obj->long_at_put(index, res);
1163       break;
1164     }
1165 
1166     // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1167     case T_INT: case T_FLOAT: { // 4 bytes.
1168       assert(value->type() == T_INT, "Agreement.");
1169       bool big_value = false;
1170       if (i + 1 < sv->field_size() && type == T_INT) {
1171         if (sv->field_at(i)->is_location()) {
1172           Location::Type type = ((LocationValue*) sv->field_at(i))->location().type();
1173           if (type == Location::dbl || type == Location::lng) {
1174             big_value = true;
1175           }
1176         } else if (sv->field_at(i)->is_constant_int()) {
1177           ScopeValue* next_scope_field = sv->field_at(i + 1);
1178           if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1179             big_value = true;
1180           }
1181         }
1182       }
1183 
1184       if (big_value) {
1185         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1186   #ifdef _LP64
1187         jlong res = (jlong)low->get_int();
1188   #else
1189         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1190   #endif
1191         obj->int_at_put(index, (jint)*((jint*)&res));
1192         obj->int_at_put(++index, (jint)*(((jint*)&res) + 1));
1193       } else {
1194         val = value->get_int();
1195         obj->int_at_put(index, (jint)*((jint*)&val));
1196       }
1197       break;
1198     }
1199 
1200     case T_SHORT:
1201       assert(value->type() == T_INT, "Agreement.");
1202       val = value->get_int();
1203       obj->short_at_put(index, (jshort)*((jint*)&val));
1204       break;
1205 
1206     case T_CHAR:
1207       assert(value->type() == T_INT, "Agreement.");
1208       val = value->get_int();
1209       obj->char_at_put(index, (jchar)*((jint*)&val));
1210       break;
1211 
1212     case T_BYTE: {
1213       assert(value->type() == T_INT, "Agreement.");
1214       // The value we get is erased as a regular int. We will need to find its actual byte count 'by hand'.
1215       val = value->get_int();
1216 #if INCLUDE_JVMCI
1217       int byte_count = count_number_of_bytes_for_entry(sv, i);
1218       byte_array_put(obj, val, index, byte_count);
1219       // According to byte_count contract, the values from i + 1 to i + byte_count are illegal values. Skip.
1220       i += byte_count - 1; // Balance the loop counter.
1221       index += byte_count;
1222       // index has been updated so continue at top of loop
1223       continue;
1224 #else
1225       obj->byte_at_put(index, (jbyte)*((jint*)&val));
1226       break;
1227 #endif // INCLUDE_JVMCI
1228     }
1229 
1230     case T_BOOLEAN: {
1231       assert(value->type() == T_INT, "Agreement.");
1232       val = value->get_int();
1233       obj->bool_at_put(index, (jboolean)*((jint*)&val));
1234       break;
1235     }
1236 
1237       default:
1238         ShouldNotReachHere();
1239     }
1240     index++;
1241   }
1242 }

1270 // Restore fields of an eliminated instance object using the same field order
1271 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1272 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1273 
1274   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1275   InstanceKlass* ik = klass;
1276   while (ik != NULL) {
1277     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1278       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1279         ReassignedField field;
1280         field._offset = fs.offset();
1281         field._type = Signature::basic_type(fs.signature());
1282         if (field._type == T_INLINE_TYPE) {
1283           field._type = T_OBJECT;
1284         }
1285         if (fs.is_inlined()) {
1286           // Resolve klass of flattened inline type field
1287           Klass* vk = klass->get_inline_type_field_klass(fs.index());
1288           field._klass = InlineKlass::cast(vk);
1289           field._type = T_INLINE_TYPE;
1290         }
1291         fields->append(field);
1292       }
1293     }
1294     ik = ik->superklass();
1295   }
1296   fields->sort(compare);
1297   for (int i = 0; i < fields->length(); i++) {
1298     intptr_t val;
1299     ScopeValue* scope_field = sv->field_at(svIndex);
1300     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1301     int offset = base_offset + fields->at(i)._offset;
1302     BasicType type = fields->at(i)._type;
1303     switch (type) {
1304       case T_OBJECT:
1305       case T_ARRAY:
1306         assert(value->type() == T_OBJECT, "Agreement.");
1307         obj->obj_field_put(offset, value->get_obj()());
1308         break;
1309 
1310       case T_INLINE_TYPE: {
1311         // Recursively re-assign flattened inline type fields
1312         InstanceKlass* vk = fields->at(i)._klass;
1313         assert(vk != NULL, "must be resolved");
1314         offset -= InlineKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1315         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1316         continue; // Continue because we don't need to increment svIndex
1317       }
1318 
1319       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1320       case T_INT: case T_FLOAT: { // 4 bytes.
1321         assert(value->type() == T_INT, "Agreement.");
1322         bool big_value = false;
1323         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1324           if (scope_field->is_location()) {
1325             Location::Type type = ((LocationValue*) scope_field)->location().type();
1326             if (type == Location::dbl || type == Location::lng) {
1327               big_value = true;
1328             }
1329           }
1330           if (scope_field->is_constant_int()) {
1331             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1332             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1333               big_value = true;
1334             }
1335           }
1336         }
1337 
1338         if (big_value) {
1339           i++;
1340           assert(i < fields->length(), "second T_INT field needed");
1341           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1342         } else {
1343           val = value->get_int();
1344           obj->int_field_put(offset, (jint)*((jint*)&val));
1345           break;
1346         }
1347       }
1348         /* no break */
1349 
1350       case T_LONG: case T_DOUBLE: {
1351         assert(value->type() == T_INT, "Agreement.");
1352         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1353 #ifdef _LP64
1354         jlong res = (jlong)low->get_int();
1355 #else
1356         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1357 #endif
1358         obj->long_field_put(offset, res);
1359         break;
1360       }
1361 
1362       case T_SHORT:
1363         assert(value->type() == T_INT, "Agreement.");
1364         val = value->get_int();
1365         obj->short_field_put(offset, (jshort)*((jint*)&val));
1366         break;
1367 
1368       case T_CHAR:
1369         assert(value->type() == T_INT, "Agreement.");
1370         val = value->get_int();
1371         obj->char_field_put(offset, (jchar)*((jint*)&val));
1372         break;
1373 
1374       case T_BYTE:
1375         assert(value->type() == T_INT, "Agreement.");
1376         val = value->get_int();
1377         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1378         break;
1379 
1380       case T_BOOLEAN:
1381         assert(value->type() == T_INT, "Agreement.");
1382         val = value->get_int();
1383         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1384         break;
1385 
1386       default:
1387         ShouldNotReachHere();
1388     }
1389     svIndex++;
1390   }
1391   return svIndex;
1392 }

1443 // relock objects for which synchronization was eliminated
1444 void Deoptimization::relock_objects(GrowableArray<MonitorInfo*>* monitors, JavaThread* thread, bool realloc_failures) {
1445   for (int i = 0; i < monitors->length(); i++) {
1446     MonitorInfo* mon_info = monitors->at(i);

1448       assert(!mon_info->owner_is_scalar_replaced() || realloc_failures, "reallocation was missed");

1450         Handle obj(thread, mon_info->owner());
1451         markWord mark = obj->mark();
1452         if (UseBiasedLocking && mark.has_bias_pattern()) {
1453           // New allocated objects may have the mark set to anonymously biased.
1454           // Also the deoptimized method may called methods with synchronization
1455           // where the thread-local object is bias locked to the current thread.
1456           assert(mark.is_biased_anonymously() ||
1457                  mark.biased_locker() == thread, "should be locked to current thread");

1459           markWord unbiased_prototype = markWord::prototype().set_age(mark.age());
1460           obj->set_mark(unbiased_prototype);
1461         }
1462         BasicLock* lock = mon_info->lock();
1463         ObjectSynchronizer::enter(obj, lock, thread);
1464         assert(mon_info->owner()->is_locked(), "object must be locked now");
1465       }
1466     }
1467   }

1564   thread->set_frames_to_pop_failed_realloc(array->frames());
1565   // Unlock all monitors here otherwise the interpreter will see a
1566   // mix of locked and unlocked monitors (because of failed
1567   // reallocations of synchronized objects) and be confused.
1568   for (int i = 0; i < array->frames(); i++) {
1569     MonitorChunk* monitors = array->element(i)->monitors();
1570     if (monitors != NULL) {
1571       for (int j = 0; j < monitors->number_of_monitors(); j++) {
1572         BasicObjectLock* src = monitors->at(j);
1573         if (src->obj() != NULL) {
1574           ObjectSynchronizer::exit(src->obj(), src->lock(), thread);
1575         }
1576       }
1577       array->element(i)->free_monitors(thread);
1578 #ifdef ASSERT
1579       array->element(i)->set_removed_monitors();
1580 #endif
1581     }
1582   }
1583 }

1591     if (!mon_info->eliminated() && mon_info->owner() != NULL) {
1592       objects_to_revoke->append(Handle(thread, mon_info->owner()));

1597 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1598   // Unfortunately we don't have a RegisterMap available in most of
1599   // the places we want to call this routine so we need to walk the
1600   // stack again to update the register map.
1601   if (map == NULL || !map->update_map()) {
1602     StackFrameStream sfs(thread, true);
1603     bool found = false;
1604     while (!found && !sfs.is_done()) {
1605       frame* cur = sfs.current();
1606       sfs.next();
1607       found = cur->id() == fr.id();
1608     }
1609     assert(found, "frame to be deoptimized not found on target thread's stack");
1610     map = sfs.register_map();
1611   }
1612 
1613   vframe* vf = vframe::new_vframe(&fr, map, thread);
1614   compiledVFrame* cvf = compiledVFrame::cast(vf);
1615   // Revoke monitors' biases in all scopes
1616   while (!cvf->is_top()) {
1617     collect_monitors(cvf, objects_to_revoke);
1618     cvf = compiledVFrame::cast(cvf->sender());
1619   }
1620   collect_monitors(cvf, objects_to_revoke);
1621 }

1623 void Deoptimization::revoke_from_deopt_handler(JavaThread* thread, frame fr, RegisterMap* map) {
1624   if (!UseBiasedLocking) {
1625     return;
1626   }
1627   GrowableArray<Handle>* objects_to_revoke = new GrowableArray<Handle>();
1628   get_monitors_from_stack(objects_to_revoke, thread, fr, map);

1861   // handler. Note this fact before we start generating temporary frames
1862   // that can confuse an asynchronous stack walker. This counter is
1863   // decremented at the end of unpack_frames().

1876   // before we are done with it.

1880   Events::log_deopt_message(thread, "Uncommon trap: trap_request=" PTR32_FORMAT " fr.pc=" INTPTR_FORMAT " relative=" INTPTR_FORMAT,
1881               trap_request, p2i(fr.pc()), fr.pc() - fr.cb()->code_begin());

1884     ResourceMark rm;
1885 
1886     DeoptReason reason = trap_request_reason(trap_request);
1887     DeoptAction action = trap_request_action(trap_request);
1888 #if INCLUDE_JVMCI

1890 #endif

1903 #if INCLUDE_JVMCI
1904           , debug_id
1905 #endif
1906           );
1907     }
1908 
1909     methodHandle    trap_method(THREAD, trap_scope->method());
1910     int             trap_bci    = trap_scope->bci();
1911 #if INCLUDE_JVMCI
1912     jlong           speculation = thread->pending_failed_speculation();
1913     if (nm->is_compiled_by_jvmci() && nm->is_nmethod()) { // Exclude AOTed methods
1914       nm->as_nmethod()->update_speculation(thread);
1915     } else {
1916       assert(speculation == 0, "There should not be a speculation for methods compiled by non-JVMCI compilers");
1917     }

1925       thread->set_pending_transfer_to_interpreter(true);

1934     // Need MDO to record RTM code generation state.

1938 #if INCLUDE_JVMCI

1959     if (TraceDeoptimization || LogCompilation) {
1960       ResourceMark rm;
1961       ttyLocker ttyl;
1962       char buf[100];
1963       if (xtty != NULL) {
1964         xtty->begin_head("uncommon_trap thread='" UINTX_FORMAT "' %s",
1965                          os::current_thread_id(),
1966                          format_trap_request(buf, sizeof(buf), trap_request));
1967 #if INCLUDE_JVMCI
1968         if (speculation != 0) {
1969           xtty->print(" speculation='" JLONG_FORMAT "'", speculation);
1970         }
1971 #endif
1972         nm->log_identity(xtty);
1973       }
1974       Symbol* class_name = NULL;
1975       bool unresolved = false;
1976       if (unloaded_class_index >= 0) {
1977         constantPoolHandle constants (THREAD, trap_method->constants());
1978         if (constants->tag_at(unloaded_class_index).is_unresolved_klass()) {
1979           class_name = constants->klass_name_at(unloaded_class_index);
1980           unresolved = true;
1981           if (xtty != NULL)
1982             xtty->print(" unresolved='1'");
1983         } else if (constants->tag_at(unloaded_class_index).is_symbol()) {
1984           class_name = constants->symbol_at(unloaded_class_index);
1985         }
1986         if (xtty != NULL)
1987           xtty->name(class_name);
1988       }
1989       if (xtty != NULL && trap_mdo != NULL && (int)reason < (int)MethodData::_trap_hist_limit) {
1990         // Dump the relevant MDO state.
1991         // This is the deopt count for the current reason, any previous
1992         // reasons or recompiles seen at this point.
1993         int dcnt = trap_mdo->trap_count(reason);
1994         if (dcnt != 0)
1995           xtty->print(" count='%d'", dcnt);
1996         ProfileData* pdata = trap_mdo->bci_to_data(trap_bci);
1997         int dos = (pdata == NULL)? 0: pdata->trap_state();
1998         if (dos != 0) {
1999           xtty->print(" state='%s'", format_trap_state(buf, sizeof(buf), dos));
2000           if (trap_state_is_recompiled(dos)) {
2001             int recnt2 = trap_mdo->overflow_recompile_count();
2002             if (recnt2 != 0)
2003               xtty->print(" recompiles2='%d'", recnt2);
2004           }
2005         }
2006       }
2007       if (xtty != NULL) {
2008         xtty->stamp();
2009         xtty->end_head();
2010       }
2011       if (TraceDeoptimization) {  // make noise on the tty
2012         tty->print("Uncommon trap occurred in");
2013         nm->method()->print_short_name(tty);
2014         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
2015 #if INCLUDE_JVMCI
2016         if (nm->is_nmethod()) {
2017           const char* installed_code_name = nm->as_nmethod()->jvmci_name();
2018           if (installed_code_name != NULL) {
2019             tty->print(" (JVMCI: installed code name=%s) ", installed_code_name);
2020           }
2021         }
2022 #endif
2023         tty->print(" (@" INTPTR_FORMAT ") thread=" UINTX_FORMAT " reason=%s action=%s unloaded_class_index=%d" JVMCI_ONLY(" debug_id=%d"),
2024                    p2i(fr.pc()),
2025                    os::current_thread_id(),
2026                    trap_reason_name(reason),
2027                    trap_action_name(action),
2028                    unloaded_class_index
2029 #if INCLUDE_JVMCI
2030                    , debug_id
2031 #endif
2032                    );
2033         if (class_name != NULL) {
2034           tty->print(unresolved ? " unresolved class: " : " symbol: ");
2035           class_name->print_symbol_on(tty);
2036         }
2037         tty->cr();
2038       }
2039       if (xtty != NULL) {
2040         // Log the precise location of the trap.
2041         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
2042           xtty->begin_elem("jvms bci='%d'", sd->bci());
2043           xtty->method(sd->method());
2044           xtty->end_elem();
2045           if (sd->is_top())  break;
2046         }
2047         xtty->tail("uncommon_trap");
2048       }
2049     }
2050     // (End diagnostic printout.)
2051 
2052     // Load class if necessary
2053     if (unloaded_class_index >= 0) {
2054       constantPoolHandle constants(THREAD, trap_method->constants());
2055       load_class_by_index(constants, unloaded_class_index);
2056     }
2057 
2058     // Flush the nmethod if necessary and desirable.
2059     //
2060     // We need to avoid situations where we are re-flushing the nmethod
2061     // because of a hot deoptimization site.  Repeated flushes at the same
2062     // point need to be detected by the compiler and avoided.  If the compiler
2063     // cannot avoid them (or has a bug and "refuses" to avoid them), this
2064     // module must take measures to avoid an infinite cycle of recompilation
2065     // and deoptimization.  There are several such measures:
2066     //
2067     //   1. If a recompilation is ordered a second time at some site X
2068     //   and for the same reason R, the action is adjusted to 'reinterpret',
2069     //   to give the interpreter time to exercise the method more thoroughly.
2070     //   If this happens, the method's overflow_recompile_count is incremented.
2071     //
2072     //   2. If the compiler fails to reduce the deoptimization rate, then
2073     //   the method's overflow_recompile_count will begin to exceed the set
2074     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2075     //   is adjusted to 'make_not_compilable', and the method is abandoned
2076     //   to the interpreter.  This is a performance hit for hot methods,
2077     //   but is better than a disastrous infinite cycle of recompilations.
2078     //   (Actually, only the method containing the site X is abandoned.)
2079     //
2080     //   3. In parallel with the previous measures, if the total number of
2081     //   recompilations of a method exceeds the much larger set limit
2082     //   PerMethodRecompilationCutoff, the method is abandoned.
2083     //   This should only happen if the method is very large and has
2084     //   many "lukewarm" deoptimizations.  The code which enforces this
2085     //   limit is elsewhere (class nmethod, class Method).
2086     //
2087     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2088     // to recompile at each bytecode independently of the per-BCI cutoff.
2089     //
2090     // The decision to update code is up to the compiler, and is encoded
2091     // in the Action_xxx code.  If the compiler requests Action_none
2092     // no trap state is changed, no compiled code is changed, and the
2093     // computation suffers along in the interpreter.
2094     //
2095     // The other action codes specify various tactics for decompilation
2096     // and recompilation.  Action_maybe_recompile is the loosest, and
2097     // allows the compiled code to stay around until enough traps are seen,
2098     // and until the compiler gets around to recompiling the trapping method.
2099     //
2100     // The other actions cause immediate removal of the present code.
2101 
2102     // Traps caused by injected profile shouldn't pollute trap counts.
2103     bool injected_profile_trap = trap_method->has_injected_profile() &&
2104                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2105 
2106     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;
2107     bool make_not_entrant = false;
2108     bool make_not_compilable = false;
2109     bool reprofile = false;
2110     switch (action) {
2111     case Action_none:
2112       // Keep the old code.
2113       update_trap_state = false;
2114       break;
2115     case Action_maybe_recompile:
2116       // Do not need to invalidate the present code, but we can
2117       // initiate another
2118       // Start compiler without (necessarily) invalidating the nmethod.
2119       // The system will tolerate the old code, but new code should be
2120       // generated when possible.
2121       break;
2122     case Action_reinterpret:
2123       // Go back into the interpreter for a while, and then consider
2124       // recompiling form scratch.
2125       make_not_entrant = true;
2126       // Reset invocation counter for outer most method.
2127       // This will allow the interpreter to exercise the bytecodes
2128       // for a while before recompiling.
2129       // By contrast, Action_make_not_entrant is immediate.
2130       //
2131       // Note that the compiler will track null_check, null_assert,
2132       // range_check, and class_check events and log them as if they
2133       // had been traps taken from compiled code.  This will update
2134       // the MDO trap history so that the next compilation will
2135       // properly detect hot trap sites.
2136       reprofile = true;
2137       break;
2138     case Action_make_not_entrant:
2139       // Request immediate recompilation, and get rid of the old code.
2140       // Make them not entrant, so next time they are called they get
2141       // recompiled.  Unloaded classes are loaded now so recompile before next
2142       // time they are called.  Same for uninitialized.  The interpreter will
2143       // link the missing class, if any.
2144       make_not_entrant = true;
2145       break;
2146     case Action_make_not_compilable:
2147       // Give up on compiling this method at all.
2148       make_not_entrant = true;
2149       make_not_compilable = true;
2150       break;
2151     default:
2152       ShouldNotReachHere();
2153     }
2154 
2155     // Setting +ProfileTraps fixes the following, on all platforms:
2156     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2157     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2158     // recompile relies on a MethodData* to record heroic opt failures.
2159 
2160     // Whether the interpreter is producing MDO data or not, we also need
2161     // to use the MDO to detect hot deoptimization points and control
2162     // aggressive optimization.
2163     bool inc_recompile_count = false;
2164     ProfileData* pdata = NULL;
2165     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2166       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2167       uint this_trap_count = 0;
2168       bool maybe_prior_trap = false;
2169       bool maybe_prior_recompile = false;
2170       pdata = query_update_method_data(trap_mdo, trap_bci, reason, true,
2171 #if INCLUDE_JVMCI
2172                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2173 #endif
2174                                    nm->method(),
2175                                    //outputs:
2176                                    this_trap_count,
2177                                    maybe_prior_trap,
2178                                    maybe_prior_recompile);
2179       // Because the interpreter also counts null, div0, range, and class
2180       // checks, these traps from compiled code are double-counted.
2181       // This is harmless; it just means that the PerXTrapLimit values
2182       // are in effect a little smaller than they look.
2183 
2184       DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2185       if (per_bc_reason != Reason_none) {
2186         // Now take action based on the partially known per-BCI history.
2187         if (maybe_prior_trap
2188             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2189           // If there are too many traps at this BCI, force a recompile.
2190           // This will allow the compiler to see the limit overflow, and
2191           // take corrective action, if possible.  The compiler generally
2192           // does not use the exact PerBytecodeTrapLimit value, but instead
2193           // changes its tactics if it sees any traps at all.  This provides
2194           // a little hysteresis, delaying a recompile until a trap happens
2195           // several times.
2196           //
2197           // Actually, since there is only one bit of counter per BCI,
2198           // the possible per-BCI counts are {0,1,(per-method count)}.
2199           // This produces accurate results if in fact there is only
2200           // one hot trap site, but begins to get fuzzy if there are
2201           // many sites.  For example, if there are ten sites each
2202           // trapping two or more times, they each get the blame for
2203           // all of their traps.
2204           make_not_entrant = true;
2205         }
2206 
2207         // Detect repeated recompilation at the same BCI, and enforce a limit.
2208         if (make_not_entrant && maybe_prior_recompile) {
2209           // More than one recompile at this point.
2210           inc_recompile_count = maybe_prior_trap;
2211         }
2212       } else {
2213         // For reasons which are not recorded per-bytecode, we simply
2214         // force recompiles unconditionally.
2215         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2216         make_not_entrant = true;
2217       }
2218 
2219       // Go back to the compiler if there are too many traps in this method.
2220       if (this_trap_count >= per_method_trap_limit(reason)) {
2221         // If there are too many traps in this method, force a recompile.
2222         // This will allow the compiler to see the limit overflow, and
2223         // take corrective action, if possible.
2224         // (This condition is an unlikely backstop only, because the
2225         // PerBytecodeTrapLimit is more likely to take effect first,
2226         // if it is applicable.)
2227         make_not_entrant = true;
2228       }
2229 
2230       // Here's more hysteresis:  If there has been a recompile at
2231       // this trap point already, run the method in the interpreter
2232       // for a while to exercise it more thoroughly.
2233       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2234         reprofile = true;
2235       }
2236     }
2237 
2238     // Take requested actions on the method:
2239 
2240     // Recompile
2241     if (make_not_entrant) {
2242       if (!nm->make_not_entrant()) {
2243         return; // the call did not change nmethod's state
2244       }
2245 
2246       if (pdata != NULL) {
2247         // Record the recompilation event, if any.
2248         int tstate0 = pdata->trap_state();
2249         int tstate1 = trap_state_set_recompiled(tstate0, true);
2250         if (tstate1 != tstate0)
2251           pdata->set_trap_state(tstate1);
2252       }
2253 
2254 #if INCLUDE_RTM_OPT
2255       // Restart collecting RTM locking abort statistic if the method
2256       // is recompiled for a reason other than RTM state change.
2257       // Assume that in new recompiled code the statistic could be different,
2258       // for example, due to different inlining.
2259       if ((reason != Reason_rtm_state_change) && (trap_mdo != NULL) &&
2260           UseRTMDeopt && (nm->as_nmethod()->rtm_state() != ProfileRTM)) {
2261         trap_mdo->atomic_set_rtm_state(ProfileRTM);
2262       }
2263 #endif
2264       // For code aging we count traps separately here, using make_not_entrant()
2265       // as a guard against simultaneous deopts in multiple threads.
2266       if (reason == Reason_tenured && trap_mdo != NULL) {
2267         trap_mdo->inc_tenure_traps();
2268       }
2269     }
2270 
2271     if (inc_recompile_count) {
2272       trap_mdo->inc_overflow_recompile_count();
2273       if ((uint)trap_mdo->overflow_recompile_count() >
2274           (uint)PerBytecodeRecompilationCutoff) {
2275         // Give up on the method containing the bad BCI.
2276         if (trap_method() == nm->method()) {
2277           make_not_compilable = true;
2278         } else {
2279           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2280           // But give grace to the enclosing nm->method().
2281         }
2282       }
2283     }
2284 
2285     // Reprofile
2286     if (reprofile) {
2287       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2288     }
2289 
2290     // Give up compiling
2291     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2292       assert(make_not_entrant, "consistent");
2293       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2294     }
2295 
2296   } // Free marked resources
2297 
2298 }

2302 Deoptimization::query_update_method_data(MethodData* trap_mdo,
2303                                          int trap_bci,
2304                                          Deoptimization::DeoptReason reason,
2305                                          bool update_total_trap_count,
2306 #if INCLUDE_JVMCI
2307                                          bool is_osr,
2308 #endif
2309                                          Method* compiled_method,
2310                                          //outputs:
2311                                          uint& ret_this_trap_count,
2312                                          bool& ret_maybe_prior_trap,
2313                                          bool& ret_maybe_prior_recompile) {
2314   bool maybe_prior_trap = false;
2315   bool maybe_prior_recompile = false;
2316   uint this_trap_count = 0;
2317   if (update_total_trap_count) {
2318     uint idx = reason;
2319 #if INCLUDE_JVMCI
2320     if (is_osr) {
2321       idx += Reason_LIMIT;
2322     }
2323 #endif
2324     uint prior_trap_count = trap_mdo->trap_count(idx);
2325     this_trap_count  = trap_mdo->inc_trap_count(idx);
2326 
2327     // If the runtime cannot find a place to store trap history,
2328     // it is estimated based on the general condition of the method.
2329     // If the method has ever been recompiled, or has ever incurred
2330     // a trap with the present reason , then this BCI is assumed
2331     // (pessimistically) to be the culprit.
2332     maybe_prior_trap      = (prior_trap_count != 0);
2333     maybe_prior_recompile = (trap_mdo->decompile_count() != 0);
2334   }
2335   ProfileData* pdata = NULL;
2336 
2337 
2338   // For reasons which are recorded per bytecode, we check per-BCI data.
2339   DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2340   assert(per_bc_reason != Reason_none || update_total_trap_count, "must be");
2341   if (per_bc_reason != Reason_none) {
2342     // Find the profile data for this BCI.  If there isn't one,
2343     // try to allocate one from the MDO's set of spares.
2344     // This will let us detect a repeated trap at this point.
2345     pdata = trap_mdo->allocate_bci_to_data(trap_bci, reason_is_speculate(reason) ? compiled_method : NULL);
2346 
2347     if (pdata != NULL) {
2348       if (reason_is_speculate(reason) && !pdata->is_SpeculativeTrapData()) {
2349         if (LogCompilation && xtty != NULL) {
2350           ttyLocker ttyl;
2351           // no more room for speculative traps in this MDO
2352           xtty->elem("speculative_traps_oom");
2353         }
2354       }
2355       // Query the trap state of this profile datum.
2356       int tstate0 = pdata->trap_state();
2357       if (!trap_state_has_reason(tstate0, per_bc_reason))
2358         maybe_prior_trap = false;
2359       if (!trap_state_is_recompiled(tstate0))
2360         maybe_prior_recompile = false;
2361 
2362       // Update the trap state of this profile datum.
2363       int tstate1 = tstate0;
2364       // Record the reason.
2365       tstate1 = trap_state_add_reason(tstate1, per_bc_reason);
2366       // Store the updated state on the MDO, for next time.
2367       if (tstate1 != tstate0)
2368         pdata->set_trap_state(tstate1);
2369     } else {
2370       if (LogCompilation && xtty != NULL) {
2371         ttyLocker ttyl;
2372         // Missing MDP?  Leave a small complaint in the log.
2373         xtty->elem("missing_mdp bci='%d'", trap_bci);
2374       }
2375     }
2376   }
2377 
2378   // Return results:
2379   ret_this_trap_count = this_trap_count;
2380   ret_maybe_prior_trap = maybe_prior_trap;
2381   ret_maybe_prior_recompile = maybe_prior_recompile;
2382   return pdata;
2383 }

2420 // Further breakdown of DataLayout::trap_state, as promised by DataLayout.
2421 const int DS_REASON_MASK   = ((uint)DataLayout::trap_mask) >> 1;
2422 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;

2424 //---------------------------trap_state_reason---------------------------------
2425 Deoptimization::DeoptReason
2426 Deoptimization::trap_state_reason(int trap_state) {
2427   // This assert provides the link between the width of DataLayout::trap_bits
2428   // and the encoding of "recorded" reasons.  It ensures there are enough
2429   // bits to store all needed reasons in the per-BCI MDO profile.
2430   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2431   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2432   trap_state -= recompile_bit;
2433   if (trap_state == DS_REASON_MASK) {
2434     return Reason_many;
2435   } else {
2436     assert((int)Reason_none == 0, "state=0 => Reason_none");
2437     return (DeoptReason)trap_state;
2438   }
2439 }
2440 //-------------------------trap_state_has_reason-------------------------------
2441 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2442   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2443   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2444   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2445   trap_state -= recompile_bit;
2446   if (trap_state == DS_REASON_MASK) {
2447     return -1;  // true, unspecifically (bottom of state lattice)
2448   } else if (trap_state == reason) {
2449     return 1;   // true, definitely
2450   } else if (trap_state == 0) {
2451     return 0;   // false, definitely (top of state lattice)
2452   } else {
2453     return 0;   // false, definitely
2454   }
2455 }
2456 //-------------------------trap_state_add_reason-------------------------------
2457 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2458   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2459   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2460   trap_state -= recompile_bit;
2461   if (trap_state == DS_REASON_MASK) {
2462     return trap_state + recompile_bit;     // already at state lattice bottom
2463   } else if (trap_state == reason) {
2464     return trap_state + recompile_bit;     // the condition is already true
2465   } else if (trap_state == 0) {
2466     return reason + recompile_bit;          // no condition has yet been true
2467   } else {
2468     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2469   }
2470 }
2471 //-----------------------trap_state_is_recompiled------------------------------
2472 bool Deoptimization::trap_state_is_recompiled(int trap_state) {
2473   return (trap_state & DS_RECOMPILE_BIT) != 0;
2474 }
2475 //-----------------------trap_state_set_recompiled-----------------------------
2476 int Deoptimization::trap_state_set_recompiled(int trap_state, bool z) {
2477   if (z)  return trap_state |  DS_RECOMPILE_BIT;
2478   else    return trap_state & ~DS_RECOMPILE_BIT;
2479 }

2616 void Deoptimization::gather_statistics(DeoptReason reason, DeoptAction action,

2621   _deoptimization_hist[reason][0][0]      += 1;  // per-reason total
2622   juint* cases = _deoptimization_hist[reason][1+action];
2623   juint* bc_counter_addr = NULL;
2624   juint  bc_counter      = 0;
2625   // Look for an unused counter, or an exact match to this BC.
2626   if (bc != Bytecodes::_illegal) {
2627     for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2628       juint* counter_addr = &cases[bc_case];
2629       juint  counter = *counter_addr;
2630       if ((counter == 0 && bc_counter_addr == NULL)
2631           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2632         // this counter is either free or is already devoted to this BC
2633         bc_counter_addr = counter_addr;
2634         bc_counter = counter | bc;
2635       }
2636     }
2637   }
2638   if (bc_counter_addr == NULL) {
2639     // Overflow, or no given bytecode.
2640     bc_counter_addr = &cases[BC_CASE_LIMIT-1];
2641     bc_counter = (*bc_counter_addr & ~LSB_MASK);  // clear LSB
2642   }
2643   *bc_counter_addr = bc_counter + (1 << LSB_BITS);
2644 }

2650 void Deoptimization::print_statistics() {
2651   juint total = total_deoptimization_count();
2652   juint account = total;
2653   if (total != 0) {
2654     ttyLocker ttyl;
2655     if (xtty != NULL)  xtty->head("statistics type='deoptimization'");
2656     tty->print_cr("Deoptimization traps recorded:");
2657     #define PRINT_STAT_LINE(name, r) \
2658       tty->print_cr("  %4d (%4.1f%%) %s", (int)(r), ((r) * 100.0) / total, name);
2659     PRINT_STAT_LINE("total", total);
2660     // For each non-zero entry in the histogram, print the reason,
2661     // the action, and (if specifically known) the type of bytecode.
2662     for (int reason = 0; reason < Reason_LIMIT; reason++) {
2663       for (int action = 0; action < Action_LIMIT; action++) {
2664         juint* cases = _deoptimization_hist[reason][1+action];
2665         for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2666           juint counter = cases[bc_case];
2667           if (counter != 0) {
2668             char name[1*K];
2669             Bytecodes::Code bc = (Bytecodes::Code)(counter & LSB_MASK);
2670             if (bc_case == BC_CASE_LIMIT && (int)bc == 0)
2671               bc = Bytecodes::_illegal;
2672             sprintf(name, "%s/%s/%s",
2673                     trap_reason_name(reason),
2674                     trap_action_name(action),
2675                     Bytecodes::is_defined(bc)? Bytecodes::name(bc): "other");
2676             juint r = counter >> LSB_BITS;
2677             tty->print_cr("  %40s: " UINT32_FORMAT " (%.1f%%)", name, r, (r * 100.0) / total);
2678             account -= r;
2679           }
2680         }
2681       }
2682     }
2683     if (account != 0) {
2684       PRINT_STAT_LINE("unaccounted", account);
2685     }
2686     #undef PRINT_STAT_LINE
2687     if (xtty != NULL)  xtty->tail("statistics");
2688   }
2689 }
