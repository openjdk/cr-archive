3104 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3105   {
3106     VMTypeEntry* types = origtypes;
3107     while (types->typeName != NULL) {
3108       if (strcmp(typeName, types->typeName) == 0) {
3109         // Found it
3110         return 1;
3111       }
3112       ++types;
3113     }
3114   }
3115   // Search for the base type by peeling off const and *
3116   size_t len = strlen(typeName);
3117   if (typeName[len-1] == '*') {
3118     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3119     strncpy(s, typeName, len - 1);
3120     s[len-1] = '\0';
3121     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3122     if (recursiveFindType(origtypes, s, true) == 1) {
3123       FREE_C_HEAP_ARRAY(char, s);
3124       return 1;
3125     }
3126     FREE_C_HEAP_ARRAY(char, s);
3127   }
3128   const char* start = NULL;
3129   if (strstr(typeName, "GrowableArray<") == typeName) {
3130     start = typeName + strlen("GrowableArray<");
3131   } else if (strstr(typeName, "Array<") == typeName) {
3132     start = typeName + strlen("Array<");
3133   }
3134   if (start != NULL) {
3135     const char * end = strrchr(typeName, '>');
3136     int len = end - start + 1;
3137     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3138     strncpy(s, start, len - 1);
3139     s[len-1] = '\0';
3140     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3141     if (recursiveFindType(origtypes, s, true) == 1) {
3142       FREE_C_HEAP_ARRAY(char, s);
3143       return 1;
3144     }
3145     FREE_C_HEAP_ARRAY(char, s);
3146   }
3147   if (strstr(typeName, "const ") == typeName) {
3148     const char * s = typeName + strlen("const ");
3149     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3150     if (recursiveFindType(origtypes, s, true) == 1) {
3151       return 1;
3152     }
3153   }
3154   if (strstr(typeName, " const") == typeName + len - 6) {
3155     char * s = os::strdup_check_oom(typeName);
3156     s[len - 6] = '\0';
3157     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3158     if (recursiveFindType(origtypes, s, true) == 1) {
3159       os::free(s);
3160       return 1;
3161     }
3162     os::free(s);
3163   }
3164   if (!isRecurse) {
3165     tty->print_cr("type \"%s\" not found", typeName);
3166   }
3167   return 0;
3168 }
