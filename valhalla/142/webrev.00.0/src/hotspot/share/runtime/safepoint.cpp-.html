  73 static void post_safepoint_begin_event(EventSafepointBegin& event,
  74                                        uint64_t safepoint_id,
  75                                        int thread_count,
  76                                        int critical_thread_count) {
  77   if (event.should_commit()) {
  78     event.set_safepointId(safepoint_id);
  79     event.set_totalThreadCount(thread_count);
  80     event.set_jniCriticalThreadCount(critical_thread_count);
  81     event.commit();
  82   }
  83 }
  84 
  85 static void post_safepoint_cleanup_event(EventSafepointCleanup& event, uint64_t safepoint_id) {
  86   if (event.should_commit()) {
  87     event.set_safepointId(safepoint_id);
  88     event.commit();
  89   }
  90 }
  91 
  92 static void post_safepoint_synchronize_event(EventSafepointStateSynchronization& event,
  93                                              uint64_t safepoint_id,
  94                                              int initial_number_of_threads,
  95                                              int threads_waiting_to_block,
  96                                              uint64_t iterations) {
  97   if (event.should_commit()) {
  98     event.set_safepointId(safepoint_id);
  99     event.set_initialThreadCount(initial_number_of_threads);
 100     event.set_runningThreadCount(threads_waiting_to_block);
 101     event.set_iterations(iterations);
 102     event.commit();
 103   }
 104 }

 106 static void post_safepoint_cleanup_task_event(EventSafepointCleanupTask& event,
 107                                               uint64_t safepoint_id,
 108                                               const char* name) {
 109   if (event.should_commit()) {
 110     event.set_safepointId(safepoint_id);
 111     event.set_name(name);
 112     event.commit();
 113   }
 114 }

 116 static void post_safepoint_end_event(EventSafepointEnd& event, uint64_t safepoint_id) {
 117   if (event.should_commit()) {
 118     event.set_safepointId(safepoint_id);
 119     event.commit();
 120   }
 121 }

 168 bool SafepointSynchronize::thread_not_running(ThreadSafepointState *cur_state) {
 169   if (!cur_state->is_running()) {
 170     return true;
 171   }
 172   cur_state->examine_state_of_thread(SafepointSynchronize::safepoint_counter());
 173   if (!cur_state->is_running()) {
 174     return true;
 175   }
 176   LogTarget(Trace, safepoint) lt;
 177   if (lt.is_enabled()) {
 178     ResourceMark rm;
 179     LogStream ls(lt);
 180     cur_state->print_on(&ls);
 181   }
 182   return false;
 183 }

 185 #ifdef ASSERT
 186 static void assert_list_is_valid(const ThreadSafepointState* tss_head, int still_running) {
 187   int a = 0;
 188   const ThreadSafepointState *tmp_tss = tss_head;
 189   while (tmp_tss != NULL) {
 190     ++a;
 191     assert(tmp_tss->is_running(), "Illegal initial state");
 192     tmp_tss = tmp_tss->get_next();
 193   }
 194   assert(a == still_running, "Must be the same");
 195 }
 196 #endif // ASSERT

 198 static void back_off(int64_t start_time) {
 199   // We start with fine-grained nanosleeping until a millisecond has
 200   // passed, at which point we resort to plain naked_short_sleep.
 201   if (os::javaTimeNanos() - start_time < NANOSECS_PER_MILLISEC) {
 202     os::naked_short_nanosleep(10 * (NANOUNITS / MICROUNITS));
 203   } else {
 204     os::naked_short_sleep(1);
 205   }
 206 }

 208 int SafepointSynchronize::synchronize_threads(jlong safepoint_limit_time, int nof_threads, int* initial_running)
 209 {
 210   JavaThreadIteratorWithHandle jtiwh;
 211 
 212 #ifdef ASSERT
 213   for (; JavaThread *cur = jtiwh.next(); ) {
 214     assert(cur->safepoint_state()->is_running(), "Illegal initial state");
 215   }
 216   jtiwh.rewind();
 217 #endif // ASSERT
 218 
 219   // Iterate through all threads until it has been determined how to stop them all at a safepoint.
 220   int still_running = nof_threads;
 221   ThreadSafepointState *tss_head = NULL;
 222   ThreadSafepointState **p_prev = &tss_head;
 223   for (; JavaThread *cur = jtiwh.next(); ) {
 224     ThreadSafepointState *cur_tss = cur->safepoint_state();
 225     assert(cur_tss->get_next() == NULL, "Must be NULL");
 226     if (thread_not_running(cur_tss)) {
 227       --still_running;
 228     } else {
 229       *p_prev = cur_tss;
 230       p_prev = cur_tss->next_ptr();
 231     }
 232   }
 233   *p_prev = NULL;
 234 
 235   DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 236 
 237   *initial_running = still_running;
 238 
 239   // If there is no thread still running, we are already done.
 240   if (still_running <= 0) {
 241     assert(tss_head == NULL, "Must be empty");
 242     return 1;
 243   }
 244 
 245   int iterations = 1; // The first iteration is above.
 246   int64_t start_time = os::javaTimeNanos();
 247 
 248   do {
 249     // Check if this has taken too long:
 250     if (SafepointTimeout && safepoint_limit_time < os::javaTimeNanos()) {
 251       print_safepoint_timeout();
 252     }
 253 
 254     p_prev = &tss_head;
 255     ThreadSafepointState *cur_tss = tss_head;
 256     while (cur_tss != NULL) {
 257       assert(cur_tss->is_running(), "Illegal initial state");
 258       if (thread_not_running(cur_tss)) {
 259         --still_running;
 260         *p_prev = NULL;
 261         ThreadSafepointState *tmp = cur_tss;
 262         cur_tss = cur_tss->get_next();
 263         tmp->set_next(NULL);
 264       } else {
 265         *p_prev = cur_tss;
 266         p_prev = cur_tss->next_ptr();
 267         cur_tss = cur_tss->get_next();
 268       }
 269     }
 270 
 271     DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 272 
 273     if (still_running > 0) {
 274       back_off(start_time);
 275     }
 276 
 277     iterations++;
 278   } while (still_running > 0);
 279 
 280   assert(tss_head == NULL, "Must be empty");
 281 
 282   return iterations;
 283 }

 285 void SafepointSynchronize::arm_safepoint() {
 286   // Begin the process of bringing the system to a safepoint.
 287   // Java threads can be in several different states and are
 288   // stopped by different mechanisms:
 289   //
 290   //  1. Running interpreted
 291   //     When executing branching/returning byte codes interpreter
 292   //     checks if the poll is armed, if so blocks in SS::block().
 293   //  2. Running in native code
 294   //     When returning from the native code, a Java thread must check
 295   //     the safepoint _state to see if we must block.  If the
 296   //     VM thread sees a Java thread in native, it does
 297   //     not wait for this thread to block.  The order of the memory
 298   //     writes and reads of both the safepoint state and the Java
 299   //     threads state is critical.  In order to guarantee that the
 300   //     memory writes are serialized with respect to each other,
 301   //     the VM thread issues a memory barrier instruction.
 302   //  3. Running compiled Code
 303   //     Compiled code reads the local polling page that
 304   //     is set to fault if we are trying to get to a safepoint.
 305   //  4. Blocked
 306   //     A thread which is blocked will not be allowed to return from the
 307   //     block condition until the safepoint operation is complete.
 308   //  5. In VM or Transitioning between states
 309   //     If a Java thread is currently running in the VM or transitioning
 310   //     between states, the safepointing code will poll the thread state
 311   //     until the thread blocks itself when it attempts transitions to a
 312   //     new state or locking a safepoint checked monitor.
 313 
 314   // We must never miss a thread with correct safepoint id, so we must make sure we arm
 315   // the wait barrier for the next safepoint id/counter.
 316   // Arming must be done after resetting _current_jni_active_count, _waiting_to_block.
 317   _wait_barrier->arm(static_cast<int>(_safepoint_counter + 1));
 318 
 319   assert((_safepoint_counter & 0x1) == 0, "must be even");
 320   // The store to _safepoint_counter must happen after any stores in arming.
 321   Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 322 
 323   // We are synchronizing
 324   OrderAccess::storestore(); // Ordered with _safepoint_counter
 325   _state = _synchronizing;
 326 
 327   // Arming the per thread poll while having _state != _not_synchronized means safepointing
 328   log_trace(safepoint)("Setting thread local yield flag for threads");
 329   OrderAccess::storestore(); // storestore, global state -> local state
 330   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 331     // Make sure the threads start polling, it is time to yield.
 332     SafepointMechanism::arm_local_poll(cur);
 333   }
 334 
 335   OrderAccess::fence(); // storestore|storeload, global state -> local state
 336 }

 338 // Roll all threads forward to a safepoint and suspend them all
 339 void SafepointSynchronize::begin() {
 340   assert(Thread::current()->is_VM_thread(), "Only VM thread may execute a safepoint");
 341 
 342   EventSafepointBegin begin_event;
 343   SafepointTracing::begin(VMThread::vm_op_type());
 344 
 345   Universe::heap()->safepoint_synchronize_begin();
 346 
 347   // By getting the Threads_lock, we assure that no threads are about to start or
 348   // exit. It is released again in SafepointSynchronize::end().
 349   Threads_lock->lock();
 350 
 351   assert( _state == _not_synchronized, "trying to safepoint synchronize with wrong state");
 352 
 353   int nof_threads = Threads::number_of_threads();
 354 
 355   _nof_threads_hit_polling_page = 0;
 356 
 357   log_debug(safepoint)("Safepoint synchronization initiated using %s wait barrier. (%d threads)", _wait_barrier->description(), nof_threads);
 358 
 359   // Reset the count of active JNI critical threads
 360   _current_jni_active_count = 0;
 361 
 362   // Set number of threads to wait for
 363   _waiting_to_block = nof_threads;
 364 
 365   jlong safepoint_limit_time = 0;
 366   if (SafepointTimeout) {
 367     // Set the limit time, so that it can be compared to see if this has taken
 368     // too long to complete.
 369     safepoint_limit_time = SafepointTracing::start_of_safepoint() + (jlong)SafepointTimeoutDelay * (NANOUNITS / MILLIUNITS);
 370     timeout_error_printed = false;
 371   }
 372 
 373   EventSafepointStateSynchronization sync_event;
 374   int initial_running = 0;
 375 
 376   // Arms the safepoint, _current_jni_active_count and _waiting_to_block must be set before.
 377   arm_safepoint();
 378 
 379   // Will spin until all threads are safe.
 380   int iterations = synchronize_threads(safepoint_limit_time, nof_threads, &initial_running);
 381   assert(_waiting_to_block == 0, "No thread should be running");
 382 
 383 #ifndef PRODUCT
 384   if (safepoint_limit_time != 0) {
 385     jlong current_time = os::javaTimeNanos();
 386     if (safepoint_limit_time < current_time) {
 387       log_warning(safepoint)("# SafepointSynchronize: Finished after "
 388                     INT64_FORMAT_W(6) " ms",
 389                     (int64_t)(current_time - SafepointTracing::start_of_safepoint()) / (NANOUNITS / MILLIUNITS));
 390     }
 391   }
 392 #endif
 393 
 394   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 395 
 396   // Record state
 397   _state = _synchronized;
 398 
 399   OrderAccess::fence();
 400 
 401   // Set the new id
 402   ++_safepoint_id;
 403 
 404 #ifdef ASSERT
 405   // Make sure all the threads were visited.
 406   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 407     assert(cur->was_visited_for_critical_count(_safepoint_counter), "missed a thread");
 408   }
 409 #endif // ASSERT
 410 
 411   // Update the count of active JNI critical regions
 412   GCLocker::set_jni_lock_count(_current_jni_active_count);
 413 
 414   post_safepoint_synchronize_event(sync_event,
 415                                    _safepoint_id,
 416                                    initial_running,
 417                                    _waiting_to_block, iterations);
 418 
 419   SafepointTracing::synchronized(nof_threads, initial_running, _nof_threads_hit_polling_page);
 420 
 421   // We do the safepoint cleanup first since a GC related safepoint
 422   // needs cleanup to be completed before running the GC op.
 423   EventSafepointCleanup cleanup_event;
 424   do_cleanup_tasks();
 425   post_safepoint_cleanup_event(cleanup_event, _safepoint_id);
 426 
 427   post_safepoint_begin_event(begin_event, _safepoint_id, nof_threads, _current_jni_active_count);
 428   SafepointTracing::cleanup();
 429 }

 431 void SafepointSynchronize::disarm_safepoint() {
 432   uint64_t active_safepoint_counter = _safepoint_counter;
 433   {
 434     JavaThreadIteratorWithHandle jtiwh;
 435 #ifdef ASSERT
 436     // A pending_exception cannot be installed during a safepoint.  The threads
 437     // may install an async exception after they come back from a safepoint into
 438     // pending_exception after they unblock.  But that should happen later.
 439     for (; JavaThread *cur = jtiwh.next(); ) {
 440       assert (!(cur->has_pending_exception() &&
 441                 cur->safepoint_state()->is_at_poll_safepoint()),
 442               "safepoint installed a pending exception");
 443     }
 444 #endif // ASSERT
 445 
 446     OrderAccess::fence(); // keep read and write of _state from floating up
 447     assert(_state == _synchronized, "must be synchronized before ending safepoint synchronization");
 448 
 449     // Change state first to _not_synchronized.
 450     // No threads should see _synchronized when running.
 451     _state = _not_synchronized;
 452 
 453     // Set the next dormant (even) safepoint id.
 454     assert((_safepoint_counter & 0x1) == 1, "must be odd");
 455     Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 456 
 457     OrderAccess::fence(); // Keep the local state from floating up.
 458 
 459     jtiwh.rewind();
 460     for (; JavaThread *current = jtiwh.next(); ) {
 461       // Clear the visited flag to ensure that the critical counts are collected properly.
 462       DEBUG_ONLY(current->reset_visited_for_critical_count(active_safepoint_counter);)
 463       ThreadSafepointState* cur_state = current->safepoint_state();
 464       assert(!cur_state->is_running(), "Thread not suspended at safepoint");
 465       cur_state->restart(); // TSS _running
 466       assert(cur_state->is_running(), "safepoint state has not been reset");
 467     }
 468   } // ~JavaThreadIteratorWithHandle
 469 
 470   // Release threads lock, so threads can be created/destroyed again.
 471   Threads_lock->unlock();
 472 
 473   // Wake threads after local state is correctly set.
 474   _wait_barrier->disarm();
 475 }

 477 // Wake up all threads, so they are ready to resume execution after the safepoint
 478 // operation has been carried out
 479 void SafepointSynchronize::end() {
 480   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 481   EventSafepointEnd event;
 482   assert(Thread::current()->is_VM_thread(), "Only VM thread can execute a safepoint");
 483 
 484   disarm_safepoint();
 485 
 486   Universe::heap()->safepoint_synchronize_end();
 487 
 488   SafepointTracing::end();
 489 
 490   post_safepoint_end_event(event, safepoint_id());
 491 }

 493 bool SafepointSynchronize::is_cleanup_needed() {
 494   // Need a cleanup safepoint if there are too many monitors in use
 495   // and the monitor deflation needs to be done at a safepoint.
 496   if (ObjectSynchronizer::is_safepoint_deflation_needed()) return true;
 497   // Need a safepoint if some inline cache buffers is non-empty
 498   if (!InlineCacheBuffer::is_empty()) return true;
 499   if (StringTable::needs_rehashing()) return true;
 500   if (SymbolTable::needs_rehashing()) return true;
 501   return false;
 502 }

 504 class ParallelSPCleanupThreadClosure : public ThreadClosure {
 505 private:
 506   CodeBlobClosure* _nmethod_cl;
 507   DeflateMonitorCounters* _counters;
 508 
 509 public:
 510   ParallelSPCleanupThreadClosure(DeflateMonitorCounters* counters) :
 511     _nmethod_cl(UseCodeAging ? NMethodSweeper::prepare_reset_hotness_counters() : NULL),
 512     _counters(counters) {}
 513 
 514   void do_thread(Thread* thread) {
 515     // deflate_thread_local_monitors() handles or requests deflation of
 516     // this thread's idle monitors. If !AsyncDeflateIdleMonitors or if
 517     // there is a special cleanup request, deflation is handled now.
 518     // Otherwise, async deflation is requested via a flag.
 519     ObjectSynchronizer::deflate_thread_local_monitors(thread, _counters);
 520     if (_nmethod_cl != NULL && thread->is_Java_thread() &&
 521         ! thread->is_Code_cache_sweeper_thread()) {
 522       JavaThread* jt = (JavaThread*) thread;
 523       jt->nmethods_do(_nmethod_cl);
 524     }
 525   }
 526 };

 528 class ParallelSPCleanupTask : public AbstractGangTask {
 529 private:
 530   SubTasksDone _subtasks;
 531   ParallelSPCleanupThreadClosure _cleanup_threads_cl;
 532   uint _num_workers;
 533   DeflateMonitorCounters* _counters;
 534 public:
 535   ParallelSPCleanupTask(uint num_workers, DeflateMonitorCounters* counters) :
 536     AbstractGangTask("Parallel Safepoint Cleanup"),
 537     _subtasks(SafepointSynchronize::SAFEPOINT_CLEANUP_NUM_TASKS),
 538     _cleanup_threads_cl(ParallelSPCleanupThreadClosure(counters)),
 539     _num_workers(num_workers),
 540     _counters(counters) {}
 541 
 542   void work(uint worker_id) {
 543     uint64_t safepoint_id = SafepointSynchronize::safepoint_id();
 544     // All threads deflate monitors and mark nmethods (if necessary).
 545     Threads::possibly_parallel_threads_do(true, &_cleanup_threads_cl);
 546 
 547     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_DEFLATE_MONITORS)) {
 548       const char* name = "deflating global idle monitors";
 549       EventSafepointCleanupTask event;
 550       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 551       // AsyncDeflateIdleMonitors only uses DeflateMonitorCounters
 552       // when a special cleanup has been requested.
 553       // Note: This logging output will include global idle monitor
 554       // elapsed times, but not global idle monitor deflation count.
 555       ObjectSynchronizer::do_safepoint_work(_counters);
 556 
 557       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 558     }
 559 
 560     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_UPDATE_INLINE_CACHES)) {
 561       const char* name = "updating inline caches";
 562       EventSafepointCleanupTask event;
 563       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 564       InlineCacheBuffer::update_inline_caches();
 565 
 566       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 567     }
 568 
 569     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_COMPILATION_POLICY)) {
 570       const char* name = "compilation policy safepoint handler";
 571       EventSafepointCleanupTask event;
 572       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 573       CompilationPolicy::policy()->do_safepoint_work();
 574 
 575       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 576     }
 577 
 578     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYMBOL_TABLE_REHASH)) {
 579       if (SymbolTable::needs_rehashing()) {
 580         const char* name = "rehashing symbol table";
 581         EventSafepointCleanupTask event;
 582         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 583         SymbolTable::rehash_table();
 584 
 585         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 586       }
 587     }
 588 
 589     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_STRING_TABLE_REHASH)) {
 590       if (StringTable::needs_rehashing()) {
 591         const char* name = "rehashing string table";
 592         EventSafepointCleanupTask event;
 593         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 594         StringTable::rehash_table();
 595 
 596         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 597       }
 598     }
 599 
 600     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYSTEM_DICTIONARY_RESIZE)) {
 601       if (Dictionary::does_any_dictionary_needs_resizing()) {
 602         const char* name = "resizing system dictionaries";
 603         EventSafepointCleanupTask event;
 604         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 605         ClassLoaderDataGraph::resize_dictionaries();
 606 
 607         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 608       }
 609     }
 610 
 611     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_REQUEST_OOPSTORAGE_CLEANUP)) {
 612       // Don't bother reporting event or time for this very short operation.
 613       // To have any utility we'd also want to report whether needed.
 614       OopStorage::trigger_cleanup_if_needed();
 615     }
 616 
 617     _subtasks.all_tasks_completed(_num_workers);
 618   }
 619 };

 621 // Various cleaning tasks that should be done periodically at safepoints.
 622 void SafepointSynchronize::do_cleanup_tasks() {
 623 
 624   TraceTime timer("safepoint cleanup tasks", TRACETIME_LOG(Info, safepoint, cleanup));
 625 
 626   // Prepare for monitor deflation.
 627   DeflateMonitorCounters deflate_counters;
 628   ObjectSynchronizer::prepare_deflate_idle_monitors(&deflate_counters);
 629 
 630   CollectedHeap* heap = Universe::heap();
 631   assert(heap != NULL, "heap not initialized yet?");
 632   WorkGang* cleanup_workers = heap->get_safepoint_workers();
 633   if (cleanup_workers != NULL) {
 634     // Parallel cleanup using GC provided thread pool.
 635     uint num_cleanup_workers = cleanup_workers->active_workers();
 636     ParallelSPCleanupTask cleanup(num_cleanup_workers, &deflate_counters);
 637     StrongRootsScope srs(num_cleanup_workers);
 638     cleanup_workers->run_task(&cleanup);
 639   } else {
 640     // Serial cleanup using VMThread.
 641     ParallelSPCleanupTask cleanup(1, &deflate_counters);
 642     StrongRootsScope srs(1);
 643     cleanup.work(0);
 644   }
 645 
 646   // Needs to be done single threaded by the VMThread.  This walks
 647   // the thread stacks looking for references to metadata before
 648   // deciding to remove it from the metaspaces.
 649   if (ClassLoaderDataGraph::should_clean_metaspaces_and_reset()) {
 650     const char* name = "cleanup live ClassLoaderData metaspaces";
 651     TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 652     ClassLoaderDataGraph::walk_metadata_and_clean_metaspaces();
 653   }
 654 
 655   // Finish monitor deflation.
 656   ObjectSynchronizer::finish_deflate_idle_monitors(&deflate_counters);
 657 
 658   assert(InlineCacheBuffer::is_empty(), "should have cleaned up ICBuffer");
 659 }

 661 // Methods for determining if a JavaThread is safepoint safe.
 662 
 663 // False means unsafe with undetermined state.
 664 // True means a determined state, but it may be an unsafe state.
 665 // If called from a non-safepoint context safepoint_count MUST be InactiveSafepointCounter.
 666 bool SafepointSynchronize::try_stable_load_state(JavaThreadState *state, JavaThread *thread, uint64_t safepoint_count) {
 667   assert((safepoint_count != InactiveSafepointCounter &&
 668           Thread::current() == (Thread*)VMThread::vm_thread() &&
 669           SafepointSynchronize::_state != _not_synchronized)
 670          || safepoint_count == InactiveSafepointCounter, "Invalid check");
 671 
 672   // To handle the thread_blocked state on the backedge of the WaitBarrier from
 673   // previous safepoint and reading the reset value (0/InactiveSafepointCounter) we
 674   // re-read state after we read thread safepoint id. The JavaThread changes its
 675   // thread state from thread_blocked before resetting safepoint id to 0.
 676   // This guarantees the second read will be from an updated thread state. It can
 677   // either be different state making this an unsafe state or it can see blocked
 678   // again. When we see blocked twice with a 0 safepoint id, either:
 679   // - It is normally blocked, e.g. on Mutex, TBIVM.
 680   // - It was in SS:block(), looped around to SS:block() and is blocked on the WaitBarrier.
 681   // - It was in SS:block() but now on a Mutex.
 682   // All of these cases are safe.
 683 
 684   *state = thread->thread_state();
 685   OrderAccess::loadload();
 686   uint64_t sid = thread->safepoint_state()->get_safepoint_id();  // Load acquire
 687   if (sid != InactiveSafepointCounter && sid != safepoint_count) {
 688     // In an old safepoint, state not relevant.
 689     return false;
 690   }
 691   return *state == thread->thread_state();
 692 }
 693 
 694 static bool safepoint_safe_with(JavaThread *thread, JavaThreadState state) {
 695   switch(state) {
 696   case _thread_in_native:
 697     // native threads are safe if they have no java stack or have walkable stack
 698     return !thread->has_last_Java_frame() || thread->frame_anchor()->walkable();
 699 
 700   case _thread_blocked:
 701     // On wait_barrier or blocked.
 702     // Blocked threads should already have walkable stack.
 703     assert(!thread->has_last_Java_frame() || thread->frame_anchor()->walkable(), "blocked and not walkable");
 704     return true;
 705 
 706   default:
 707     return false;
 708   }
 709 }
 710 
 711 bool SafepointSynchronize::handshake_safe(JavaThread *thread) {
 712   if (thread->is_ext_suspended() || thread->is_terminated()) {
 713     return true;
 714   }
 715   JavaThreadState stable_state;
 716   if (try_stable_load_state(&stable_state, thread, InactiveSafepointCounter)) {
 717     return safepoint_safe_with(thread, stable_state);
 718   }
 719   return false;
 720 }
 721 
 722 // See if the thread is running inside a lazy critical native and
 723 // update the thread critical count if so.  Also set a suspend flag to
 724 // cause the native wrapper to return into the JVM to do the unlock
 725 // once the native finishes.
 726 static void check_for_lazy_critical_native(JavaThread *thread, JavaThreadState state) {
 727   if (state == _thread_in_native &&
 728       thread->has_last_Java_frame() &&
 729       thread->frame_anchor()->walkable()) {
 730     // This thread might be in a critical native nmethod so look at
 731     // the top of the stack and increment the critical count if it
 732     // is.
 733     frame wrapper_frame = thread->last_frame();
 734     CodeBlob* stub_cb = wrapper_frame.cb();
 735     if (stub_cb != NULL &&
 736         stub_cb->is_nmethod() &&
 737         stub_cb->as_nmethod_or_null()->is_lazy_critical_native()) {
 738       // A thread could potentially be in a critical native across
 739       // more than one safepoint, so only update the critical state on
 740       // the first one.  When it returns it will perform the unlock.
 741       if (!thread->do_critical_native_unlock()) {
 742 #ifdef ASSERT
 743         if (!thread->in_critical()) {
 744           GCLocker::increment_debug_jni_lock_count();
 745         }
 746 #endif
 747         thread->enter_critical();
 748         // Make sure the native wrapper calls back on return to
 749         // perform the needed critical unlock.
 750         thread->set_critical_native_unlock();
 751       }
 752     }
 753   }
 754 }

 756 // -------------------------------------------------------------------------------------------------------
 757 // Implementation of Safepoint blocking point
 758 
 759 void SafepointSynchronize::block(JavaThread *thread) {
 760   assert(thread != NULL, "thread must be set");
 761   assert(thread->is_Java_thread(), "not a Java thread");
 762 
 763   // Threads shouldn't block if they are in the middle of printing, but...
 764   ttyLocker::break_tty_lock_for_safepoint(os::current_thread_id());
 765 
 766   // Only bail from the block() call if the thread is gone from the
 767   // thread list; starting to exit should still block.
 768   if (thread->is_terminated()) {
 769      // block current thread if we come here from native code when VM is gone
 770      thread->block_if_vm_exited();
 771 
 772      // otherwise do nothing
 773      return;
 774   }
 775 
 776   JavaThreadState state = thread->thread_state();
 777   thread->frame_anchor()->make_walkable(thread);
 778 
 779   uint64_t safepoint_id = SafepointSynchronize::safepoint_counter();
 780   // Check that we have a valid thread_state at this point
 781   switch(state) {
 782     case _thread_in_vm_trans:
 783     case _thread_in_Java:        // From compiled code
 784     case _thread_in_native_trans:
 785     case _thread_blocked_trans:
 786     case _thread_new_trans:
 787 
 788       // We have no idea where the VMThread is, it might even be at next safepoint.
 789       // So we can miss this poll, but stop at next.
 790 
 791       // Load dependent store, it must not pass loading of safepoint_id.
 792       thread->safepoint_state()->set_safepoint_id(safepoint_id); // Release store
 793 
 794       // This part we can skip if we notice we miss or are in a future safepoint.
 795       OrderAccess::storestore();
 796       // Load in wait barrier should not float up
 797       thread->set_thread_state_fence(_thread_blocked);
 798 
 799       _wait_barrier->wait(static_cast<int>(safepoint_id));
 800       assert(_state != _synchronized, "Can't be");
 801 
 802       // If barrier is disarmed stop store from floating above loads in barrier.
 803       OrderAccess::loadstore();
 804       thread->set_thread_state(state);
 805 
 806       // Then we reset the safepoint id to inactive.
 807       thread->safepoint_state()->reset_safepoint_id(); // Release store
 808 
 809       OrderAccess::fence();
 810 
 811       break;
 812 
 813     default:
 814      fatal("Illegal threadstate encountered: %d", state);
 815   }
 816   guarantee(thread->safepoint_state()->get_safepoint_id() == InactiveSafepointCounter,
 817             "The safepoint id should be set only in block path");
 818 
 819   // Check for pending. async. exceptions or suspends - except if the
 820   // thread was blocked inside the VM. has_special_runtime_exit_condition()
 821   // is called last since it grabs a lock and we only want to do that when
 822   // we must.
 823   //
 824   // Note: we never deliver an async exception at a polling point as the
 825   // compiler may not have an exception handler for it. The polling
 826   // code will notice the async and deoptimize and the exception will
 827   // be delivered. (Polling at a return point is ok though). Sure is
 828   // a lot of bother for a deprecated feature...
 829   //
 830   // We don't deliver an async exception if the thread state is
 831   // _thread_in_native_trans so JNI functions won't be called with
 832   // a surprising pending exception. If the thread state is going back to java,
 833   // async exception is checked in check_special_condition_for_native_trans().
 834 
 835   if (state != _thread_blocked_trans &&
 836       state != _thread_in_vm_trans &&
 837       thread->has_special_runtime_exit_condition()) {
 838     thread->handle_special_runtime_exit_condition(
 839       !thread->is_at_poll_safepoint() && (state != _thread_in_native_trans));
 840   }
 841 
 842   // cross_modify_fence is done by SafepointMechanism::block_if_requested_slow
 843   // which is the only caller here.
 844 }

 939 void ThreadSafepointState::examine_state_of_thread(uint64_t safepoint_count) {
 940   assert(is_running(), "better be running or just have hit safepoint poll");
 941 
 942   JavaThreadState stable_state;
 943   if (!SafepointSynchronize::try_stable_load_state(&stable_state, _thread, safepoint_count)) {
 944     // We could not get stable state of the JavaThread.
 945     // Consider it running and just return.
 946     return;
 947   }
 948 
 949   // Check for a thread that is suspended. Note that thread resume tries
 950   // to grab the Threads_lock which we own here, so a thread cannot be
 951   // resumed during safepoint synchronization.
 952 
 953   // We check to see if this thread is suspended without locking to
 954   // avoid deadlocking with a third thread that is waiting for this
 955   // thread to be suspended. The third thread can notice the safepoint
 956   // that we're trying to start at the beginning of its SR_lock->wait()
 957   // call. If that happens, then the third thread will block on the
 958   // safepoint while still holding the underlying SR_lock. We won't be
 959   // able to get the SR_lock and we'll deadlock.
 960   //
 961   // We don't need to grab the SR_lock here for two reasons:
 962   // 1) The suspend flags are both volatile and are set with an
 963   //    Atomic::cmpxchg() call so we should see the suspended
 964   //    state right away.
 965   // 2) We're being called from the safepoint polling loop; if
 966   //    we don't see the suspended state on this iteration, then
 967   //    we'll come around again.
 968   //
 969   bool is_suspended = _thread->is_ext_suspended();
 970   if (is_suspended) {
 971     account_safe_thread();
 972     return;
 973   }
 974 
 975   if (safepoint_safe_with(_thread, stable_state)) {
 976     check_for_lazy_critical_native(_thread, stable_state);
 977     account_safe_thread();
 978     return;
 979   }
 980 
 981   // All other thread states will continue to run until they
 982   // transition and self-block in state _blocked
 983   // Safepoint polling in compiled code causes the Java threads to do the same.
 984   // Note: new threads may require a malloc so they must be allowed to finish
 985 
 986   assert(is_running(), "examine_state_of_thread on non-running thread");
 987   return;
 988 }

 990 void ThreadSafepointState::account_safe_thread() {
 991   SafepointSynchronize::decrement_waiting_to_block();
 992   if (_thread->in_critical()) {
 993     // Notice that this thread is in a critical section
 994     SafepointSynchronize::increment_jni_active_count();
 995   }
 996   DEBUG_ONLY(_thread->set_visited_for_critical_count(SafepointSynchronize::safepoint_counter());)
 997   assert(!_safepoint_safe, "Must be unsafe before safe");
 998   _safepoint_safe = true;
 999 }

1018 // Block the thread at poll or poll return for safepoint/handshake.
1019 void ThreadSafepointState::handle_polling_page_exception() {
1020 
1021   // Step 1: Find the nmethod from the return address
1022   address real_return_addr = thread()->saved_exception_pc();
1023 
1024   CodeBlob *cb = CodeCache::find_blob(real_return_addr);
1025   assert(cb != NULL && cb->is_compiled(), "return address should be in nmethod");
1026   CompiledMethod* nm = (CompiledMethod*)cb;
1027 
1028   // Find frame of caller
1029   frame stub_fr = thread()->last_frame();
1030   CodeBlob* stub_cb = stub_fr.cb();
1031   assert(stub_cb->is_safepoint_stub(), "must be a safepoint stub");
1032   RegisterMap map(thread(), true);
1033   frame caller_fr = stub_fr.sender(&map);
1034 
1035   // Should only be poll_return or poll
1036   assert( nm->is_at_poll_or_poll_return(real_return_addr), "should not be at call" );
1037 
1038   // This is a poll immediately before a return. The exception handling code
1039   // has already had the effect of causing the return to occur, so the execution
1040   // will continue immediately after the call. In addition, the oopmap at the
1041   // return point does not mark the return value as an oop (if it is), so
1042   // it needs a handle here to be updated.
1043   if( nm->is_at_poll_return(real_return_addr) ) {
1044     ResourceMark rm;
1045     // See if return type is an oop.
1046     Method* method = nm->method();
1047     bool return_oop = method->is_returning_oop();
1048 
1049     GrowableArray<Handle> return_values;
1050     InlineKlass* vk = NULL;
1051 
1052     if (return_oop && InlineTypeReturnedAsFields) {
1053       SignatureStream ss(method->signature());
1054       while (!ss.at_return_type()) {
1055         ss.next();
1056       }
1057       if (ss.type() == T_INLINE_TYPE) {
1058         // Check if inline type is returned as fields
1059         vk = InlineKlass::returned_inline_klass(map);
1060         if (vk != NULL) {
1061           // We're at a safepoint at the return of a method that returns
1062           // multiple values. We must make sure we preserve the oop values
1063           // across the safepoint.
1064           assert(vk == method->returned_inline_type(thread()), "bad inline klass");
1065           vk->save_oop_fields(map, return_values);
1066           return_oop = false;
1067         }
1068       }
1069     }
1070 
1071     if (return_oop) {
1072       // The oop result has been saved on the stack together with all
1073       // the other registers. In order to preserve it over GCs we need
1074       // to keep it in a handle.
1075       oop result = caller_fr.saved_oop_result(&map);
1076       assert(oopDesc::is_oop_or_null(result), "must be oop");
1077       return_values.push(Handle(thread(), result));
1078       assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
1079     }
1080 
1081     // Block the thread
1082     SafepointMechanism::block_if_requested(thread());
1083 
1084     // restore oop result, if any
1085     if (return_oop) {
1086       assert(return_values.length() == 1, "only one return value");
1087       caller_fr.set_saved_oop_result(&map, return_values.pop()());
1088     } else if (vk != NULL) {
1089       vk->restore_oop_results(map, return_values);
1090     }
1091   }
1092 
1093   // This is a safepoint poll. Verify the return address and block.
1094   else {
1095     set_at_poll_safepoint(true);
1096 
1097     // verify the blob built the "return address" correctly
1098     assert(real_return_addr == caller_fr.pc(), "must match");
1099 
1100     // Block the thread
1101     SafepointMechanism::block_if_requested(thread());
1102     set_at_poll_safepoint(false);
1103 
1104     // If we have a pending async exception deoptimize the frame
1105     // as otherwise we may never deliver it.
1106     if (thread()->has_async_condition()) {
1107       ThreadInVMfromJavaNoAsyncException __tiv(thread());
1108       Deoptimization::deoptimize_frame(thread(), caller_fr.id());
1109     }
1110 
1111     // If an exception has been installed we must check for a pending deoptimization
1112     // Deoptimize frame if exception has been thrown.
1113 
1114     if (thread()->has_pending_exception() ) {
1115       RegisterMap map(thread(), true);
1116       frame caller_fr = stub_fr.sender(&map);
1117       if (caller_fr.is_deoptimized_frame()) {
1118         // The exception patch will destroy registers that are still
1119         // live and will be needed during deoptimization. Defer the
1120         // Async exception should have deferred the exception until the
1121         // next safepoint which will be detected when we get into
1122         // the interpreter so if we have an exception now things
1123         // are messed up.
1124 
1125         fatal("Exception installed and deoptimization is pending");
1126       }
1127     }
1128   }
1129 }

1167 void SafepointTracing::statistics_log() {
1168   LogTarget(Info, safepoint, stats) lt;
1169   assert (lt.is_enabled(), "should only be called when printing statistics is enabled");
1170   LogStream ls(lt);
1171 
1172   static int _cur_stat_index = 0;
1173 
1174   // Print header every 30 entries
1175   if ((_cur_stat_index % 30) == 0) {
1176     print_header(&ls);
1177     _cur_stat_index = 1;  // wrap
1178   } else {
1179     _cur_stat_index++;
1180   }
1181 
1182   ls.print("%-28s [       "
1183            INT32_FORMAT_W(8) "        " INT32_FORMAT_W(8) " "
1184            "]",
1185            VM_Operation::name(_current_type),
1186            _nof_threads,
1187            _nof_running);
1188   ls.print("[       "
1189            INT64_FORMAT_W(10) " " INT64_FORMAT_W(10) " "
1190            INT64_FORMAT_W(10) " " INT64_FORMAT_W(10) " ]",
1191            (int64_t)(_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns),
1192            (int64_t)(_last_safepoint_cleanup_time_ns - _last_safepoint_sync_time_ns),
1193            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns),
1194            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_begin_time_ns));
1195 
1196   ls.print_cr(INT32_FORMAT_W(16), _page_trap);
1197 }

1199 // This method will be called when VM exits. This tries to summarize the sampling.
1200 // Current thread may already be deleted, so don't use ResourceMark.
1201 void SafepointTracing::statistics_exit_log() {
1202   if (!log_is_enabled(Info, safepoint, stats)) {
1203     return;
1204   }
1205   for (int index = 0; index < VM_Operation::VMOp_Terminating; index++) {
1206     if (_op_count[index] != 0) {
1207       log_info(safepoint, stats)("%-28s" UINT64_FORMAT_W(10), VM_Operation::name(index),
1208                _op_count[index]);
1209     }
1210   }
1211 
1212   log_info(safepoint, stats)("VM operations coalesced during safepoint " INT64_FORMAT,
1213                               VMThread::get_coalesced_count());
1214   log_info(safepoint, stats)("Maximum sync time  " INT64_FORMAT" ns",
1215                               (int64_t)(_max_sync_time));
1216   log_info(safepoint, stats)("Maximum vm operation time (except for Exit VM operation)  "
1217                               INT64_FORMAT " ns",
1218                               (int64_t)(_max_vmop_time));
1219 }

1221 void SafepointTracing::begin(VM_Operation::VMOp_Type type) {
1222   _op_count[type]++;
1223   _current_type = type;
1224 
1225   // update the time stamp to begin recording safepoint time
1226   _last_safepoint_begin_time_ns = os::javaTimeNanos();
1227   _last_safepoint_sync_time_ns = 0;
1228   _last_safepoint_cleanup_time_ns = 0;
1229 
1230   _last_app_time_ns = _last_safepoint_begin_time_ns - _last_safepoint_end_time_ns;
1231   _last_safepoint_end_time_ns = 0;
1232 
1233   RuntimeService::record_safepoint_begin(_last_app_time_ns);
1234 }

1236 void SafepointTracing::synchronized(int nof_threads, int nof_running, int traps) {
1237   _last_safepoint_sync_time_ns = os::javaTimeNanos();
1238   _nof_threads = nof_threads;
1239   _nof_running = nof_running;
1240   _page_trap   = traps;
1241   RuntimeService::record_safepoint_synchronized(_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns);
1242 }
1243 
1244 void SafepointTracing::cleanup() {
1245   _last_safepoint_cleanup_time_ns = os::javaTimeNanos();
1246 }
1247 
1248 void SafepointTracing::end() {
1249   _last_safepoint_end_time_ns = os::javaTimeNanos();
1250 
1251   if (_max_sync_time < (_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns)) {
1252     _max_sync_time = _last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns;
1253   }
1254   if (_max_vmop_time < (_last_safepoint_end_time_ns - _last_safepoint_sync_time_ns)) {
1255     _max_vmop_time = _last_safepoint_end_time_ns - _last_safepoint_sync_time_ns;
1256   }
1257   if (log_is_enabled(Info, safepoint, stats)) {
1258     statistics_log();
1259   }
1260 
1261   log_info(safepoint)(
1262      "Safepoint \"%s\", "
1263      "Time since last: " JLONG_FORMAT " ns, "
1264      "Reaching safepoint: " JLONG_FORMAT " ns, "
1265      "At safepoint: " JLONG_FORMAT " ns, "
1266      "Total: " JLONG_FORMAT " ns",
1267       VM_Operation::name(_current_type),
1268       _last_app_time_ns,
1269       _last_safepoint_cleanup_time_ns - _last_safepoint_begin_time_ns,
1270       _last_safepoint_end_time_ns     - _last_safepoint_cleanup_time_ns,
1271       _last_safepoint_end_time_ns     - _last_safepoint_begin_time_ns
1272      );
1273 
1274   RuntimeService::record_safepoint_end(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns);
1275 }
