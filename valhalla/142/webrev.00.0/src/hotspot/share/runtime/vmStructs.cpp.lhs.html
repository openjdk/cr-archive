3106 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3107   {
3108     VMTypeEntry* types = origtypes;
3109     while (types->typeName != NULL) {
3110       if (strcmp(typeName, types->typeName) == 0) {
3111         // Found it
3112         return 1;
3113       }
3114       ++types;
3115     }
3116   }
3117   // Search for the base type by peeling off const and *
3118   size_t len = strlen(typeName);
3119   if (typeName[len-1] == '*') {
3120     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3121     strncpy(s, typeName, len - 1);
3122     s[len-1] = '\0';
3123     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3124     if (recursiveFindType(origtypes, s, true) == 1) {
3125       FREE_C_HEAP_ARRAY(char, s);
3126       return 1;
3127     }
3128     FREE_C_HEAP_ARRAY(char, s);
3129   }
3130   const char* start = NULL;
3131   if (strstr(typeName, "GrowableArray<") == typeName) {
3132     start = typeName + strlen("GrowableArray<");
3133   } else if (strstr(typeName, "Array<") == typeName) {
3134     start = typeName + strlen("Array<");
3135   }
3136   if (start != NULL) {
3137     const char * end = strrchr(typeName, '>');
3138     int len = end - start + 1;
3139     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3140     strncpy(s, start, len - 1);
3141     s[len-1] = '\0';
3142     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3143     if (recursiveFindType(origtypes, s, true) == 1) {
3144       FREE_C_HEAP_ARRAY(char, s);
3145       return 1;
3146     }
3147     FREE_C_HEAP_ARRAY(char, s);
3148   }
3149   if (strstr(typeName, "const ") == typeName) {
3150     const char * s = typeName + strlen("const ");
3151     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3152     if (recursiveFindType(origtypes, s, true) == 1) {
3153       return 1;
3154     }
3155   }
3156   if (strstr(typeName, " const") == typeName + len - 6) {
3157     char * s = os::strdup_check_oom(typeName);
3158     s[len - 6] = '\0';
3159     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3160     if (recursiveFindType(origtypes, s, true) == 1) {
3161       os::free(s);
3162       return 1;
3163     }
3164     os::free(s);
3165   }
3166   if (!isRecurse) {
3167     tty->print_cr("type \"%s\" not found", typeName);
3168   }
3169   return 0;
3170 }
