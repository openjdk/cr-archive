 336     /** Report duplicate declaration error.
 337      */
 338     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 339         if (!sym.type.isErroneous()) {
 340             Symbol location = sym.location();
 341             if (location.kind == MTH &&
 342                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 343                 log.error(pos,
 344                           Errors.AlreadyDefinedInClinit(kindName(sym),
 345                                                         sym,
 346                                                         kindName(sym.location()),
 347                                                         kindName(sym.location().enclClass()),
 348                                                         sym.location().enclClass()));
 349             } else {
 350                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 351                  * as we should have issued an error for the duplicated fields
 352                  */
 353                 if (location.kind != MTH ||
 354                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 355                         ((sym.owner.flags_field & RECORD) == 0)) {
 356                     log.error(pos,
 357                             Errors.AlreadyDefined(kindName(sym),
 358                                     sym,
 359                                     kindName(sym.location()),
 360                                     sym.location()));
 361                 }
 362             }
 363         }
 364     }

 378     /** Check that variable does not hide variable with same name in
 379      *  immediately enclosing local scope.
 380      *  @param pos           Position for error reporting.
 381      *  @param v             The symbol.
 382      *  @param s             The scope.
 383      */
 384     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 385         for (Symbol sym : s.getSymbolsByName(v.name)) {
 386             if (sym.owner != v.owner) break;
 387             if (sym.kind == VAR &&
 388                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 389                 v.name != names.error) {
 390                 duplicateError(pos, sym);
 391                 return;
 392             }
 393         }
 394     }
 395 
 396     /** Check that a class or interface does not hide a class or
 397      *  interface with same name in immediately enclosing local scope.
 398      *  @param pos           Position for error reporting.
 399      *  @param c             The symbol.
 400      *  @param s             The scope.
 401      */
 402     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 403         for (Symbol sym : s.getSymbolsByName(c.name)) {
 404             if (sym.owner != c.owner) break;
 405             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 406                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 407                 c.name != names.error) {
 408                 duplicateError(pos, sym);
 409                 return;
 410             }
 411         }
 412     }

 414     /** Check that class does not have the same name as one of
 415      *  its enclosing classes, or as a class defined in its enclosing scope.
 416      *  return true if class is unique in its enclosing scope.
 417      *  @param pos           Position for error reporting.
 418      *  @param name          The class name.
 419      *  @param s             The enclosing scope.
 420      */
 421     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 422         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 423             if (sym.kind == TYP && sym.name != names.error) {
 424                 duplicateError(pos, sym);
 425                 return false;
 426             }
 427         }
 428         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 429             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 430                 duplicateError(pos, sym);
 431                 return true;
 432             }
 433         }
 434         return true;
 435     }

 444     /** Return name of local class.
 445      *  This is of the form   {@code <enclClass> $ n <classname> }
 446      *  where
 447      *    enclClass is the flat name of the enclosing class,
 448      *    classname is the simple name of the local class
 449      */
 450     public Name localClassName(ClassSymbol c) {
 451         Name enclFlatname = c.owner.enclClass().flatname;
 452         String enclFlatnameStr = enclFlatname.toString();
 453         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 454         Integer index = localClassNameIndexes.get(key);
 455         for (int i = (index == null) ? 1 : index; ; i++) {
 456             Name flatname = names.fromString(enclFlatnameStr
 457                     + syntheticNameChar + i + c.name);
 458             if (getCompiled(c.packge().modle, flatname) == null) {
 459                 localClassNameIndexes.put(key, i + 1);
 460                 return flatname;
 461             }
 462         }
 463     }
 464 
 465     public void clearLocalClassNameIndexes(ClassSymbol c) {
 466         if (c.owner != null && c.owner.kind != NIL) {
 467             localClassNameIndexes.remove(new Pair<>(
 468                     c.owner.enclClass().flatname, c.name));
 469         }
 470     }

 601     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 602         final InferenceContext inferenceContext = checkContext.inferenceContext();
 603         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 604             inferenceContext.addFreeTypeListener(List.of(req, found),
 605                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 606         } else {
 607             if (found.hasTag(CLASS)) {
 608                 if (inferenceContext != infer.emptyContext)
 609                     checkParameterizationWithValues(pos, found);
 610             }
 611         }
 612         if (req.hasTag(ERROR))
 613             return req;
 614         if (req.hasTag(NONE))
 615             return found;
 616         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 617             if (found.hasTag(BOT) && types.isValueBased(req)) {
 618                 log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 619             }
 620             return found;
 621         } else {
 622             if (found.isNumeric() && req.isNumeric()) {
 623                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 624                 return types.createErrorType(found);
 625             }
 626             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 627             return types.createErrorType(found);
 628         }
 629     }

 640     Type checkCastable(DiagnosticPosition pos, Type found, Type req, CheckContext checkContext) {
 641         if (types.isCastable(found, req, castWarner(pos, found, req))) {

1141     Type checkMethod(final Type mtype,
1142             final Symbol sym,
1143             final Env<AttrContext> env,
1144             final List<JCExpression> argtrees,

1147             InferenceContext inferenceContext) {

1150         // System.out.println("actuals: " + argtypes);

1153                     solvedContext -> checkMethod(solvedContext.asInstType(mtype), sym, env, argtrees, argtypes, useVarargs, solvedContext));
1154             return mtype;

1170         if (args != null) {
1171             //this is null when type-checking a method reference

1173                 JCTree arg = args.head;

1175                 assertConvertible(arg, arg.type, formals.head, warn);
1176                 args = args.tail;
1177                 formals = formals.tail;
1178                 nonInferred = nonInferred.tail;
1179             }

1181                 Type varArg = types.elemtype(last);
1182                 while (args.tail != null) {

1184                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1185                     assertConvertible(arg, arg.type, varArg, warn);
1186                     args = args.tail;
1187                 }
1188             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1189                 // non-varargs call to varargs method
1190                 Type varParam = owntype.getParameterTypes().last();
1191                 Type lastArg = argtypes.last();
1192                 if (types.isSubtypeUnchecked(lastArg, types.elemtype(varParam)) &&
1193                     !types.isSameType(types.erasure(varParam), types.erasure(lastArg)))

1195                                 Warnings.InexactNonVarargsCall(types.elemtype(varParam),varParam));
1196             }

1199             Type argtype = owntype.getParameterTypes().last();
1200             if (!types.isReifiable(argtype) &&
1201                 (sym.baseSymbol().attribute(syms.trustMeType.tsym) == null ||
1202                  !isTrustMeAllowedOnMethod(sym))) {
1203                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1204             }
1205             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1206          }
1207          return owntype;

1231         private Type firstIncompatibleTypeArg(Type type) {
1232             List<Type> formals = type.tsym.type.allparams();
1233             List<Type> actuals = type.allparams();
1234             List<Type> args = type.getTypeArguments();
1235             List<Type> forms = type.tsym.type.getTypeArguments();
1236             ListBuffer<Type> bounds_buf = new ListBuffer<>();
1237 
1238             // For matching pairs of actual argument types `a' and
1239             // formal type parameters with declared bound `b' ...
1240             while (args.nonEmpty() && forms.nonEmpty()) {
1241                 // exact type arguments needs to know their
1242                 // bounds (for upper and lower bound
1243                 // calculations).  So we create new bounds where
1244                 // type-parameters are replaced with actuals argument types.
1245                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1246                 args = args.tail;
1247                 forms = forms.tail;
1248             }
1249 
1250             args = type.getTypeArguments();
1251             List<Type> tvars_cap = types.substBounds(formals,
1252                                       formals,
1253                                       types.capture(type).allparams());
1254             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1255                 // Let the actual arguments know their bound
1256                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1257                 args = args.tail;
1258                 tvars_cap = tvars_cap.tail;
1259             }
1260 
1261             args = type.getTypeArguments();
1262             List<Type> bounds = bounds_buf.toList();
1263 
1264             while (args.nonEmpty() && bounds.nonEmpty()) {
1265                 Type actual = args.head;
1266                 if (!isTypeArgErroneous(actual) &&
1267                         !bounds.head.isErroneous() &&
1268                         !checkExtends(actual, bounds.head)) {
1269                     return args.head;
1270                 }
1271                 args = args.tail;
1272                 bounds = bounds.tail;
1273             }
1274 
1275             args = type.getTypeArguments();
1276             bounds = bounds_buf.toList();
1277 
1278             for (Type arg : types.capture(type).getTypeArguments()) {
1279                 if (arg.hasTag(TYPEVAR) &&
1280                         arg.getUpperBound().isErroneous() &&
1281                         !bounds.head.isErroneous() &&
1282                         !isTypeArgErroneous(args.head)) {
1283                     return args.head;
1284                 }
1285                 bounds = bounds.tail;
1286                 args = args.tail;
1287             }
1288 
1289             return null;
1290         }

1315     /** Check that given modifiers are legal for given symbol and
1316      *  return modifiers together with any implicit modifiers for that symbol.
1317      *  Warning: we can't use flags() here since this method
1318      *  is called during class enter, when flags() would cause a premature
1319      *  completion.
1320      *  @param pos           Position to be used for error reporting.
1321      *  @param flags         The set of modifiers given in a definition.
1322      *  @param sym           The defined symbol.
1323      */
1324     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1325         long mask;
1326         long implicit = 0;
1327 
1328         switch (sym.kind) {
1329         case VAR:
1330             if (TreeInfo.isReceiverParam(tree))
1331                 mask = ReceiverParamFlags;
1332             else if (sym.owner.kind != TYP)
1333                 mask = LocalVarFlags;
1334             else if ((sym.owner.flags_field & INTERFACE) != 0)
1335                 mask = implicit = InterfaceVarFlags;
1336             else {
1337                 mask = VarFlags;
1338                 if (types.isValue(sym.owner.type) && (flags & STATIC) == 0) {
1339                     implicit |= FINAL;
1340                 }
1341             }
1342             break;
1343         case MTH:
1344             if (sym.name == names.init) {
1345                 if ((sym.owner.flags_field & ENUM) != 0) {
1346                     // enum constructors cannot be declared public or
1347                     // protected and must be implicitly or explicitly
1348                     // private
1349                     implicit = PRIVATE;
1350                     mask = PRIVATE;
1351                 } else
1352                     mask = ConstructorFlags;
1353             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1354                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1355                     mask = AnnotationTypeElementMask;
1356                     implicit = PUBLIC | ABSTRACT;
1357                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1358                     mask = InterfaceMethodMask;
1359                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1360                     if ((flags & DEFAULT) != 0) {
1361                         implicit |= ABSTRACT;
1362                     }
1363                 } else {
1364                     mask = implicit = InterfaceMethodFlags;
1365                 }
1366             } else if ((sym.owner.flags_field & RECORD) != 0) {
1367                 mask = RecordMethodFlags;
1368             } else {
1369                 // instance methods of value types do not have a monitor associated with their `this'
1370                 mask = ((sym.owner.flags_field & VALUE) != 0 && (flags & Flags.STATIC) == 0) ?
1371                         MethodFlags & ~SYNCHRONIZED : MethodFlags;
1372             }
1373             // Imply STRICTFP if owner has STRICTFP set.
1374             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1375                 ((flags) & Flags.DEFAULT) != 0)
1376                 implicit |= sym.owner.flags_field & STRICTFP;
1377             break;
1378         case TYP:
1379             if (sym.isLocal()) {
1380                 boolean implicitlyStatic = !sym.isAnonymous() &&
1381                         ((flags & RECORD) != 0 || (flags & ENUM) != 0 || (flags & INTERFACE) != 0);
1382                 boolean staticOrImplicitlyStatic = (flags & STATIC) != 0 || implicitlyStatic;
1383                 mask = staticOrImplicitlyStatic && allowRecords ? StaticLocalFlags : LocalClassFlags;
1384                 implicit = implicitlyStatic ? STATIC : implicit;
1385                 if (staticOrImplicitlyStatic) {
1386                     if (sym.owner.kind == TYP) {
1387                         log.error(pos, Errors.StaticDeclarationNotAllowedInInnerClasses);
1388                     }
1389                 }
1390             } else if (sym.owner.kind == TYP) {
1391                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : ExtendedMemberClassFlags;
1392                 if (sym.owner.owner.kind == PCK ||
1393                     (sym.owner.flags_field & STATIC) != 0)
1394                     mask |= STATIC;
1395                 else if ((flags & ENUM) != 0 || (flags & RECORD) != 0) {
1396                     log.error(pos, Errors.StaticDeclarationNotAllowedInInnerClasses);
1397                 }
1398                 // Nested interfaces and enums are always STATIC (Spec ???)
1399                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1400             } else {
1401                 mask = ExtendedClassFlags;
1402             }
1403             // Interfaces are always ABSTRACT
1404             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1405 
1406             if ((flags & ENUM) != 0) {
1407                 // enums can't be declared abstract, final, sealed or non-sealed or value type
1408                 mask &= ~(ABSTRACT | FINAL | SEALED | NON_SEALED | VALUE);
1409                 implicit |= implicitEnumFinalFlag(tree);
1410             }
1411             if ((flags & RECORD) != 0) {
1412                 // records can't be declared abstract
1413                 mask &= ~ABSTRACT;
1414                 implicit |= FINAL;
1415             }
1416             // Imply STRICTFP if owner has STRICTFP set.
1417             implicit |= sym.owner.flags_field & STRICTFP;
1418             break;
1419         default:
1420             throw new AssertionError();
1421         }
1422         long illegal = flags & ExtendedStandardFlags & ~mask;
1423         if (illegal != 0) {
1424             if ((illegal & INTERFACE) != 0) {
1425                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1426                 mask |= INTERFACE;
1427             }
1428             else {
1429                 log.error(pos,
1430                         Errors.ModNotAllowedHere(asFlagSet(illegal)));
1431             }
1432         }
1433         else if ((sym.kind == TYP ||
1434                   // ISSUE: Disallowing abstract&private is no longer appropriate
1435                   // in the presence of inner classes. Should it be deleted here?
1436                   checkDisjoint(pos, flags,
1437                                 ABSTRACT,
1438                                 PRIVATE | STATIC | DEFAULT))
1439                  &&
1440                  checkDisjoint(pos, flags,
1441                                 STATIC | PRIVATE,
1442                                 DEFAULT)
1443                  &&
1444                  checkDisjoint(pos, flags,
1445                                ABSTRACT | INTERFACE,
1446                                FINAL | NATIVE | SYNCHRONIZED | VALUE)
1447                  &&
1448                  checkDisjoint(pos, flags,
1449                                PUBLIC,
1450                                PRIVATE | PROTECTED)
1451                  &&
1452                  checkDisjoint(pos, flags,
1453                                PRIVATE,
1454                                PUBLIC | PROTECTED)
1455                  &&
1456                  checkDisjoint(pos, (flags | implicit), // complain against volatile & implcitly final entities too.
1457                                FINAL,
1458                                VOLATILE)
1459                  &&
1460                  (sym.kind == TYP ||
1461                   checkDisjoint(pos, flags,
1462                                 ABSTRACT | NATIVE,
1463                                 STRICTFP))
1464                  && checkDisjoint(pos, flags,
1465                                 FINAL,
1466                            SEALED | NON_SEALED)
1467                  && checkDisjoint(pos, flags,
1468                                 SEALED,
1469                            FINAL | NON_SEALED)) {
1470             // skip
1471         }
1472         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1473     }

1895     /** Check that this method conforms with overridden method 'other'.
1896      *  where `origin' is the class where checking started.
1897      *  Complications:
1898      *  (1) Do not check overriding of synthetic methods
1899      *      (reason: they might be final).
1900      *      todo: check whether this is still necessary.
1901      *  (2) Admit the case where an interface proxy throws fewer exceptions
1902      *      than the method it implements. Augment the proxy methods with the
1903      *      undeclared exceptions in this case.
1904      *  (3) When generics are enabled, admit the case where an interface proxy
1905      *      has a result type
1906      *      extended by the result type of the method it implements.
1907      *      Change the proxies result type to the smaller type in this case.
1908      *
1909      *  @param tree         The tree from which positions
1910      *                      are extracted for errors.
1911      *  @param m            The overriding method.
1912      *  @param other        The overridden method.
1913      *  @param origin       The class of which the overriding method
1914      *                      is a member.
1915      */
1916     void checkOverride(JCTree tree,
1917                        MethodSymbol m,
1918                        MethodSymbol other,
1919                        ClassSymbol origin) {
1920         // Don't check overriding of synthetic methods or by bridge methods.
1921         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1922             return;
1923         }
1924 
1925         // Error if static method overrides instance method (JLS 8.4.6.2).
1926         if ((m.flags() & STATIC) != 0 &&
1927                    (other.flags() & STATIC) == 0) {
1928             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1929                       Errors.OverrideStatic(cannotOverride(m, other)));
1930             m.flags_field |= BAD_OVERRIDE;
1931             return;
1932         }
1933 
1934         // Error if instance method overrides static or final
1935         // method (JLS 8.4.6.1).
1936         if ((other.flags() & FINAL) != 0 ||
1937                  (m.flags() & STATIC) == 0 &&
1938                  (other.flags() & STATIC) != 0) {
1939             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1940                       Errors.OverrideMeth(cannotOverride(m, other),
1941                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1942             m.flags_field |= BAD_OVERRIDE;
1943             return;
1944         }
1945 
1946         if ((m.owner.flags() & ANNOTATION) != 0) {
1947             // handled in validateAnnotationMethod
1948             return;
1949         }
1950 
1951         // Error if overriding method has weaker access (JLS 8.4.6.3).
1952         if (protection(m.flags()) > protection(other.flags())) {
1953             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1954                       (other.flags() & AccessFlags) == 0 ?
1955                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1956                                                           "package") :
1957                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1958                                                           asFlagSet(other.flags() & AccessFlags)));
1959             m.flags_field |= BAD_OVERRIDE;
1960             return;
1961         }
1962 
1963         if (origin.isValue() && other.owner == syms.objectType.tsym && m.type.getParameterTypes().size() == 0) {
1964             if (m.name == names.clone || m.name == names.finalize) {
1965                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1966                         Errors.InlineClassMayNotOverride(m.name));
1967                 m.flags_field |= BAD_OVERRIDE;
1968                 return;
1969             }
1970         }
1971 
1972         Type mt = types.memberType(origin.type, m);
1973         Type ot = types.memberType(origin.type, other);
1974         // Error if overriding result type is different
1975         // (or, in the case of generics mode, not a subtype) of
1976         // overridden result type. We have to rename any type parameters
1977         // before comparing types.
1978         List<Type> mtvars = mt.getTypeArguments();
1979         List<Type> otvars = ot.getTypeArguments();
1980         Type mtres = mt.getReturnType();
1981         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1982 
1983         overrideWarner.clear();
1984         boolean resultTypesOK =
1985             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1986         if (!resultTypesOK) {
1987             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1988                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1989                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1990                                         other.location()), mtres, otres));
1991                 m.flags_field |= BAD_OVERRIDE;
1992             } else {
1993                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1994                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1995                 m.flags_field |= BAD_OVERRIDE;
1996             }
1997             return;
1998         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1999             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
2000                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
2001         }
2002 
2003         // Error if overriding method throws an exception not reported
2004         // by overridden method.
2005         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
2006         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
2007         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
2008         if (unhandledErased.nonEmpty()) {
2009             log.error(TreeInfo.diagnosticPositionFor(m, tree),
2010                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
2011             m.flags_field |= BAD_OVERRIDE;
2012             return;
2013         }
2014         else if (unhandledUnerased.nonEmpty()) {
2015             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
2016                           Warnings.OverrideUncheckedThrown(cannotOverride(m, other), unhandledUnerased.head));
2017             return;
2018         }
2019 
2020         // Optional warning if varargs don't agree
2021         if ((((m.flags() ^ other.flags()) & Flags.VARARGS) != 0)
2022             && lint.isEnabled(LintCategory.OVERRIDES)) {
2023             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2024                         ((m.flags() & Flags.VARARGS) != 0)
2025                         ? Warnings.OverrideVarargsMissing(varargsOverrides(m, other))
2026                         : Warnings.OverrideVarargsExtra(varargsOverrides(m, other)));
2027         }
2028 
2029         // Warn if instance method overrides bridge method (compiler spec ??)
2030         if ((other.flags() & BRIDGE) != 0) {
2031             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2032                         Warnings.OverrideBridge(uncheckedOverrides(m, other)));
2033         }
2034 
2035         // Warn if a deprecated method overridden by a non-deprecated one.
2036         if (!isDeprecatedOverrideIgnorable(other, origin)) {
2037             Lint prevLint = setLint(lint.augment(m));
2038             try {
2039                 checkDeprecated(() -> TreeInfo.diagnosticPositionFor(m, tree), m, other);
2040             } finally {
2041                 setLint(prevLint);
2042             }
2043         }
2044     }

2880     /**
2881       * Report warnings for potentially ambiguous method declarations. Two declarations
2882       * are potentially ambiguous if they feature two unrelated functional interface
2883       * in same argument position (in which case, a call site passing an implicit
2884       * lambda would be ambiguous).
2885       */
2886     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2887             MethodSymbol msym1, MethodSymbol msym2) {
2888         if (msym1 != msym2 &&
2889                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2890                 lint.isEnabled(LintCategory.OVERLOADS) &&
2891                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2892                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2893             Type mt1 = types.memberType(site, msym1);
2894             Type mt2 = types.memberType(site, msym2);
2895             //if both generic methods, adjust type variables
2896             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2897                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2898                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2899             }
2900             //expand varargs methods if needed
2901             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2902             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2903             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2904             //if arities don't match, exit
2905             if (args1.length() != args2.length()) return;
2906             boolean potentiallyAmbiguous = false;
2907             while (args1.nonEmpty() && args2.nonEmpty()) {
2908                 Type s = args1.head;
2909                 Type t = args2.head;
2910                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2911                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2912                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2913                             types.findDescriptorType(s).getParameterTypes().length() ==
2914                             types.findDescriptorType(t).getParameterTypes().length()) {
2915                         potentiallyAmbiguous = true;
2916                     } else {
2917                         return;
2918                     }
2919                 }
2920                 args1 = args1.tail;
2921                 args2 = args2.tail;
2922             }
2923             if (potentiallyAmbiguous) {
2924                 //we found two incompatible functional interfaces with same arity
2925                 //this means a call site passing an implicit lambda would be ambiguous
2926                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2927                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2928                 log.warning(LintCategory.OVERLOADS, pos,
2929                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2930                                                                   msym2, msym2.location()));
2931                 return;
2932             }
2933         }

3649     private boolean validateAnnotation(JCAnnotation a) {
3650         boolean isValid = true;
3651         AnnotationTypeMetadata metadata = a.annotationType.type.tsym.getAnnotationTypeMetadata();
3652 
3653         // collect an inventory of the annotation elements
3654         Set<MethodSymbol> elements = metadata.getAnnotationElements();
3655 
3656         // remove the ones that are assigned values
3657         for (JCTree arg : a.args) {
3658             if (!arg.hasTag(ASSIGN)) continue; // recovery
3659             JCAssign assign = (JCAssign)arg;
3660             Symbol m = TreeInfo.symbol(assign.lhs);
3661             if (m == null || m.type.isErroneous()) continue;
3662             if (!elements.remove(m)) {
3663                 isValid = false;
3664                 log.error(assign.lhs.pos(),
3665                           Errors.DuplicateAnnotationMemberValue(m.name, a.type));
3666             }
3667         }
3668 
3669         // all the remaining ones better have default values
3670         List<Name> missingDefaults = List.nil();
3671         Set<MethodSymbol> membersWithDefault = metadata.getAnnotationElementsWithDefault();
3672         for (MethodSymbol m : elements) {
3673             if (m.type.isErroneous())
3674                 continue;
3675 
3676             if (!membersWithDefault.contains(m))
3677                 missingDefaults = missingDefaults.append(m.name);
3678         }
3679         missingDefaults = missingDefaults.reverse();
3680         if (missingDefaults.nonEmpty()) {
3681             isValid = false;
3682             Error errorKey = (missingDefaults.size() > 1)
3683                     ? Errors.AnnotationMissingDefaultValue1(a.type, missingDefaults)
3684                     : Errors.AnnotationMissingDefaultValue(a.type, missingDefaults);
3685             log.error(a.pos(), errorKey);
3686         }
3687 
3688         return isValid && validateTargetAnnotationValue(a);
3689     }

3831      *  constructors.

3833     void checkCyclicConstructors(JCClassDecl tree) {
3834         Map<Symbol,Symbol> callMap = new HashMap<>();
3835 
3836         // enter each constructor this-call into the map
3837         for (List<JCTree> l = tree.defs; l.nonEmpty(); l = l.tail) {
3838             JCMethodInvocation app = TreeInfo.firstConstructorCall(l.head);
3839             if (app == null) continue;

3841             if (TreeInfo.name(app.meth) == names._this) {
3842                 callMap.put(meth.sym, TreeInfo.symbol(app.meth));
3843             } else {
3844                 meth.sym.flags_field |= ACYCLIC;
3845             }
3846         }
3847 
3848         // Check for cycles in the map
3849         Symbol[] ctors = new Symbol[0];
3850         ctors = callMap.keySet().toArray(ctors);
3851         for (Symbol caller : ctors) {
3852             checkCyclicConstructor(tree, caller, callMap);
3853         }
