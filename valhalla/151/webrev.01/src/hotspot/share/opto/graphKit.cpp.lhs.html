  54 //----------------------------GraphKit-----------------------------------------
  55 // Main utility constructor.
  56 GraphKit::GraphKit(JVMState* jvms, PhaseGVN* gvn)
  57   : Phase(Phase::Parser),
  58     _env(C->env()),
  59     _gvn((gvn != NULL) ? *gvn : *C->initial_gvn()),
  60     _barrier_set(BarrierSet::barrier_set()->barrier_set_c2())
  61 {
  62   assert(gvn == NULL || !gvn->is_IterGVN() || gvn->is_IterGVN()->delay_transform(), "delay transform should be enabled");
  63   _exceptions = jvms->map()->next_exception();
  64   if (_exceptions != NULL)  jvms->map()->set_next_exception(NULL);
  65   set_jvms(jvms);
  66 #ifdef ASSERT
  67   if (_gvn.is_IterGVN() != NULL) {
  68     assert(_gvn.is_IterGVN()->delay_transform(), "Transformation must be delayed if IterGVN is used");
  69     // Save the initial size of _for_igvn worklist for verification (see ~GraphKit)
  70     _worklist_size = _gvn.C->for_igvn()->size();
  71   }
  72 #endif
  73 }

  90 //---------------------------clean_stack---------------------------------------
  91 // Clear away rubbish from the stack area of the JVM state.
  92 // This destroys any arguments that may be waiting on the stack.
  93 void GraphKit::clean_stack(int from_sp) {
  94   SafePointNode* map      = this->map();
  95   JVMState*      jvms     = this->jvms();
  96   int            stk_size = jvms->stk_size();
  97   int            stkoff   = jvms->stkoff();
  98   Node*          top      = this->top();
  99   for (int i = from_sp; i < stk_size; i++) {
 100     if (map->in(stkoff + i) != top) {
 101       map->set_req(stkoff + i, top);
 102     }
 103   }
 104 }

 107 //--------------------------------sync_jvms-----------------------------------
 108 // Make sure our current jvms agrees with our parse state.
 109 JVMState* GraphKit::sync_jvms() const {
 110   JVMState* jvms = this->jvms();
 111   jvms->set_bci(bci());       // Record the new bci in the JVMState
 112   jvms->set_sp(sp());         // Record the new sp in the JVMState
 113   assert(jvms_in_sync(), "jvms is now in sync");
 114   return jvms;
 115 }

 117 //--------------------------------sync_jvms_for_reexecute---------------------
 118 // Make sure our current jvms agrees with our parse state.  This version
 119 // uses the reexecute_sp for reexecuting bytecodes.
 120 JVMState* GraphKit::sync_jvms_for_reexecute() {
 121   JVMState* jvms = this->jvms();
 122   jvms->set_bci(bci());          // Record the new bci in the JVMState
 123   jvms->set_sp(reexecute_sp());  // Record the new sp in the JVMState
 124   return jvms;
 125 }

 127 #ifdef ASSERT
 128 bool GraphKit::jvms_in_sync() const {
 129   Parse* parse = is_Parse();
 130   if (parse == NULL) {
 131     if (bci() !=      jvms()->bci())          return false;
 132     if (sp()  != (int)jvms()->sp())           return false;
 133     return true;
 134   }
 135   if (jvms()->method() != parse->method())    return false;
 136   if (jvms()->bci()    != parse->bci())       return false;
 137   int jvms_sp = jvms()->sp();
 138   if (jvms_sp          != parse->sp())        return false;
 139   int jvms_depth = jvms()->depth();
 140   if (jvms_depth       != parse->depth())     return false;
 141   return true;
 142 }

 144 // Local helper checks for special internal merge points
 145 // used to accumulate and merge exception states.
 146 // They are marked by the region's in(0) edge being the map itself.
 147 // Such merge points must never "escape" into the parser at large,
 148 // until they have been handed to gvn.transform.
 149 static bool is_hidden_merge(Node* reg) {
 150   if (reg == NULL)  return false;
 151   if (reg->is_Phi()) {
 152     reg = reg->in(0);
 153     if (reg == NULL)  return false;
 154   }
 155   return reg->is_Region() && reg->in(0) != NULL && reg->in(0)->is_Root();
 156 }

 158 void GraphKit::verify_map() const {
 159   if (map() == NULL)  return;  // null map is OK
 160   assert(map()->req() <= jvms()->endoff(), "no extra garbage on map");
 161   assert(!map()->has_exceptions(),    "call add_exception_states_from 1st");
 162   assert(!is_hidden_merge(control()), "call use_exception_state, not set_map");
 163 }

 165 void GraphKit::verify_exception_state(SafePointNode* ex_map) {
 166   assert(ex_map->next_exception() == NULL, "not already part of a chain");
 167   assert(has_saved_ex_oop(ex_map), "every exception state has an ex_oop");
 168 }

 172 // Set _map to NULL, signalling a stop to further bytecode execution.
 173 // First smash the current map's control to a constant, to mark it dead.

 178     assert(dead_map->is_killed(), "must be so marked");
 179   }

 192 //-----------------------------has_ex_handler----------------------------------
 193 // Tell if this method or any caller method has exception handlers.
 194 bool GraphKit::has_ex_handler() {
 195   for (JVMState* jvmsp = jvms(); jvmsp != NULL; jvmsp = jvmsp->caller()) {
 196     if (jvmsp->has_method() && jvmsp->method()->has_exception_handlers()) {
 197       return true;
 198     }
 199   }
 200   return false;

 203 //------------------------------save_ex_oop------------------------------------
 204 // Save an exception without blowing stack contents or other JVM state.
 205 void GraphKit::set_saved_ex_oop(SafePointNode* ex_map, Node* ex_oop) {
 206   assert(!has_saved_ex_oop(ex_map), "clear ex-oop before setting again");
 207   ex_map->add_req(ex_oop);
 208   debug_only(verify_exception_state(ex_map));
 209 }

 211 inline static Node* common_saved_ex_oop(SafePointNode* ex_map, bool clear_it) {
 212   assert(GraphKit::has_saved_ex_oop(ex_map), "ex_oop must be there");
 213   Node* ex_oop = ex_map->in(ex_map->req()-1);
 214   if (clear_it)  ex_map->del_req(ex_map->req()-1);
 215   return ex_oop;
 216 }

 238 //-------------------------make_exception_state--------------------------------
 239 // Turn the current JVM state into an exception state, appending the ex_oop.
 240 SafePointNode* GraphKit::make_exception_state(Node* ex_oop) {
 241   sync_jvms();
 242   SafePointNode* ex_map = stop();  // do not manipulate this map any more
 243   set_saved_ex_oop(ex_map, ex_oop);
 244   return ex_map;
 245 }

 248 //--------------------------add_exception_state--------------------------------
 249 // Add an exception to my list of exceptions.
 250 void GraphKit::add_exception_state(SafePointNode* ex_map) {
 251   if (ex_map == NULL || ex_map->control() == top()) {
 252     return;
 253   }
 254 #ifdef ASSERT
 255   verify_exception_state(ex_map);
 256   if (has_exceptions()) {
 257     assert(ex_map->jvms()->same_calls_as(_exceptions->jvms()), "all collected exceptions must come from the same place");
 258   }
 259 #endif
 260 
 261   // If there is already an exception of exactly this type, merge with it.
 262   // In particular, null-checks and other low-level exceptions common up here.
 263   Node*       ex_oop  = saved_ex_oop(ex_map);
 264   const Type* ex_type = _gvn.type(ex_oop);
 265   if (ex_oop == top()) {
 266     // No action needed.
 267     return;
 268   }
 269   assert(ex_type->isa_instptr(), "exception must be an instance");
 270   for (SafePointNode* e2 = _exceptions; e2 != NULL; e2 = e2->next_exception()) {
 271     const Type* ex_type2 = _gvn.type(saved_ex_oop(e2));
 272     // We check sp also because call bytecodes can generate exceptions
 273     // both before and after arguments are popped!
 274     if (ex_type2 == ex_type
 275         && e2->_jvms->sp() == ex_map->_jvms->sp()) {
 276       combine_exception_states(ex_map, e2);
 277       return;
 278     }
 279   }
 280 
 281   // No pre-existing exception of the same type.  Chain it on the list.
 282   push_exception_state(ex_map);
 283 }

 285 //-----------------------add_exception_states_from-----------------------------
 286 void GraphKit::add_exception_states_from(JVMState* jvms) {
 287   SafePointNode* ex_map = jvms->map()->next_exception();
 288   if (ex_map != NULL) {
 289     jvms->map()->set_next_exception(NULL);
 290     for (SafePointNode* next_map; ex_map != NULL; ex_map = next_map) {
 291       next_map = ex_map->next_exception();
 292       ex_map->set_next_exception(NULL);
 293       add_exception_state(ex_map);
 294     }
 295   }
 296 }

 298 //-----------------------transfer_exceptions_into_jvms-------------------------
 299 JVMState* GraphKit::transfer_exceptions_into_jvms() {
 300   if (map() == NULL) {
 301     // We need a JVMS to carry the exceptions, but the map has gone away.
 302     // Create a scratch JVMS, cloned from any of the exception states...
 303     if (has_exceptions()) {
 304       _map = _exceptions;
 305       _map = clone_map();
 306       _map->set_next_exception(NULL);
 307       clear_saved_ex_oop(_map);
 308       debug_only(verify_map());
 309     } else {
 310       // ...or created from scratch
 311       JVMState* jvms = new (C) JVMState(_method, NULL);
 312       jvms->set_bci(_bci);
 313       jvms->set_sp(_sp);
 314       jvms->set_map(new SafePointNode(TypeFunc::Parms, jvms));
 315       set_jvms(jvms);
 316       for (uint i = 0; i < map()->req(); i++)  map()->init_req(i, top());
 317       set_all_memory(top());
 318       while (map()->req() < jvms->endoff())  map()->add_req(top());
 319     }
 320     // (This is a kludge, in case you didn't notice.)
 321     set_control(top());
 322   }
 323   JVMState* jvms = sync_jvms();
 324   assert(!jvms->map()->has_exceptions(), "no exceptions on this map yet");
 325   jvms->map()->set_next_exception(_exceptions);
 326   _exceptions = NULL;   // done with this set of exceptions
 327   return jvms;
 328 }

 330 static inline void add_n_reqs(Node* dstphi, Node* srcphi) {
 331   assert(is_hidden_merge(dstphi), "must be a special merge node");
 332   assert(is_hidden_merge(srcphi), "must be a special merge node");
 333   uint limit = srcphi->req();
 334   for (uint i = PhiNode::Input; i < limit; i++) {
 335     dstphi->add_req(srcphi->in(i));
 336   }
 337 }
 338 static inline void add_one_req(Node* dstphi, Node* src) {
 339   assert(is_hidden_merge(dstphi), "must be a special merge node");
 340   assert(!is_hidden_merge(src), "must not be a special merge node");
 341   dstphi->add_req(src);
 342 }

 344 //-----------------------combine_exception_states------------------------------
 345 // This helper function combines exception states by building phis on a
 346 // specially marked state-merging region.  These regions and phis are
 347 // untransformed, and can build up gradually.  The region is marked by
 348 // having a control input of its exception map, rather than NULL.  Such
 349 // regions do not appear except in this function, and in use_exception_state.
 350 void GraphKit::combine_exception_states(SafePointNode* ex_map, SafePointNode* phi_map) {
 351   if (failing())  return;  // dying anyway...
 352   JVMState* ex_jvms = ex_map->_jvms;
 353   assert(ex_jvms->same_calls_as(phi_map->_jvms), "consistent call chains");
 354   assert(ex_jvms->stkoff() == phi_map->_jvms->stkoff(), "matching locals");
 355   assert(ex_jvms->sp() == phi_map->_jvms->sp(), "matching stack sizes");
 356   assert(ex_jvms->monoff() == phi_map->_jvms->monoff(), "matching JVMS");
 357   assert(ex_jvms->scloff() == phi_map->_jvms->scloff(), "matching scalar replaced objects");
 358   assert(ex_map->req() == phi_map->req(), "matching maps");
 359   uint tos = ex_jvms->stkoff() + ex_jvms->sp();
 360   Node*         hidden_merge_mark = root();
 361   Node*         region  = phi_map->control();
 362   MergeMemNode* phi_mem = phi_map->merged_memory();
 363   MergeMemNode* ex_mem  = ex_map->merged_memory();
 364   if (region->in(0) != hidden_merge_mark) {
 365     // The control input is not (yet) a specially-marked region in phi_map.
 366     // Make it so, and build some phis.
 367     region = new RegionNode(2);
 368     _gvn.set_type(region, Type::CONTROL);
 369     region->set_req(0, hidden_merge_mark);  // marks an internal ex-state
 370     region->init_req(1, phi_map->control());
 371     phi_map->set_control(region);
 372     Node* io_phi = PhiNode::make(region, phi_map->i_o(), Type::ABIO);
 373     record_for_igvn(io_phi);
 374     _gvn.set_type(io_phi, Type::ABIO);
 375     phi_map->set_i_o(io_phi);
 376     for (MergeMemStream mms(phi_mem); mms.next_non_empty(); ) {
 377       Node* m = mms.memory();
 378       Node* m_phi = PhiNode::make(region, m, Type::MEMORY, mms.adr_type(C));
 379       record_for_igvn(m_phi);
 380       _gvn.set_type(m_phi, Type::MEMORY);
 381       mms.set_memory(m_phi);
 382     }
 383   }
 384 
 385   // Either or both of phi_map and ex_map might already be converted into phis.
 386   Node* ex_control = ex_map->control();
 387   // if there is special marking on ex_map also, we add multiple edges from src
 388   bool add_multiple = (ex_control->in(0) == hidden_merge_mark);
 389   // how wide was the destination phi_map, originally?
 390   uint orig_width = region->req();
 391 
 392   if (add_multiple) {
 393     add_n_reqs(region, ex_control);
 394     add_n_reqs(phi_map->i_o(), ex_map->i_o());
 395   } else {
 396     // ex_map has no merges, so we just add single edges everywhere
 397     add_one_req(region, ex_control);
 398     add_one_req(phi_map->i_o(), ex_map->i_o());
 399   }
 400   for (MergeMemStream mms(phi_mem, ex_mem); mms.next_non_empty2(); ) {
 401     if (mms.is_empty()) {
 402       // get a copy of the base memory, and patch some inputs into it
 403       const TypePtr* adr_type = mms.adr_type(C);
 404       Node* phi = mms.force_memory()->as_Phi()->slice_memory(adr_type);
 405       assert(phi->as_Phi()->region() == mms.base_memory()->in(0), "");
 406       mms.set_memory(phi);
 407       // Prepare to append interesting stuff onto the newly sliced phi:
 408       while (phi->req() > orig_width)  phi->del_req(phi->req()-1);
 409     }
 410     // Append stuff from ex_map:
 411     if (add_multiple) {
 412       add_n_reqs(mms.memory(), mms.memory2());
 413     } else {
 414       add_one_req(mms.memory(), mms.memory2());
 415     }
 416   }
 417   uint limit = ex_map->req();
 418   for (uint i = TypeFunc::Parms; i < limit; i++) {
 419     // Skip everything in the JVMS after tos.  (The ex_oop follows.)
 420     if (i == tos)  i = ex_jvms->monoff();
 421     Node* src = ex_map->in(i);
 422     Node* dst = phi_map->in(i);
 423     if (src != dst) {
 424       PhiNode* phi;
 425       if (dst->in(0) != region) {
 426         dst = phi = PhiNode::make(region, dst, _gvn.type(dst));
 427         record_for_igvn(phi);
 428         _gvn.set_type(phi, phi->type());
 429         phi_map->set_req(i, dst);
 430         // Prepare to append interesting stuff onto the new phi:
 431         while (dst->req() > orig_width)  dst->del_req(dst->req()-1);
 432       } else {
 433         assert(dst->is_Phi(), "nobody else uses a hidden region");
 434         phi = dst->as_Phi();
 435       }
 436       if (add_multiple && src->in(0) == ex_control) {
 437         // Both are phis.
 438         add_n_reqs(dst, src);
 439       } else {
 440         while (dst->req() < region->req())  add_one_req(dst, src);
 441       }
 442       const Type* srctype = _gvn.type(src);
 443       if (phi->type() != srctype) {
 444         const Type* dsttype = phi->type()->meet_speculative(srctype);
 445         if (phi->type() != dsttype) {
 446           phi->set_type(dsttype);
 447           _gvn.set_type(phi, dsttype);
 448         }
 449       }
 450     }
 451   }
 452   phi_map->merge_replaced_nodes_with(ex_map);
 453 }

 455 //--------------------------use_exception_state--------------------------------
 456 Node* GraphKit::use_exception_state(SafePointNode* phi_map) {
 457   if (failing()) { stop(); return top(); }
 458   Node* region = phi_map->control();
 459   Node* hidden_merge_mark = root();
 460   assert(phi_map->jvms()->map() == phi_map, "sanity: 1-1 relation");
 461   Node* ex_oop = clear_saved_ex_oop(phi_map);
 462   if (region->in(0) == hidden_merge_mark) {
 463     // Special marking for internal ex-states.  Process the phis now.
 464     region->set_req(0, region);  // now it's an ordinary region
 465     set_jvms(phi_map->jvms());   // ...so now we can use it as a map
 466     // Note: Setting the jvms also sets the bci and sp.
 467     set_control(_gvn.transform(region));
 468     uint tos = jvms()->stkoff() + sp();
 469     for (uint i = 1; i < tos; i++) {
 470       Node* x = phi_map->in(i);
 471       if (x->in(0) == region) {
 472         assert(x->is_Phi(), "expected a special phi");
 473         phi_map->set_req(i, _gvn.transform(x));
 474       }
 475     }
 476     for (MergeMemStream mms(merged_memory()); mms.next_non_empty(); ) {
 477       Node* x = mms.memory();
 478       if (x->in(0) == region) {
 479         assert(x->is_Phi(), "nobody else uses a hidden region");
 480         mms.set_memory(_gvn.transform(x));
 481       }
 482     }
 483     if (ex_oop->in(0) == region) {
 484       assert(ex_oop->is_Phi(), "expected a special phi");
 485       ex_oop = _gvn.transform(ex_oop);
 486     }
 487   } else {
 488     set_jvms(phi_map->jvms());
 489   }
 490 
 491   assert(!is_hidden_merge(phi_map->control()), "hidden ex. states cleared");
 492   assert(!is_hidden_merge(phi_map->i_o()), "hidden ex. states cleared");
 493   return ex_oop;
 494 }

 506 void GraphKit::uncommon_trap_if_should_post_on_exceptions(Deoptimization::DeoptReason reason,
 507                                                           bool must_throw) {
 508     // if the exception capability is set, then we will generate code
 509     // to check the JavaThread.should_post_on_exceptions flag to see
 510     // if we actually need to report exception events (for this
 511     // thread).  If we don't need to report exception events, we will
 512     // take the normal fast path provided by add_exception_events.  If
 513     // exception event reporting is enabled for this thread, we will
 514     // take the uncommon_trap in the BuildCutout below.
 515 
 516     // first must access the should_post_on_exceptions_flag in this thread's JavaThread
 517     Node* jthread = _gvn.transform(new ThreadLocalNode());
 518     Node* adr = basic_plus_adr(top(), jthread, in_bytes(JavaThread::should_post_on_exceptions_flag_offset()));
 519     Node* should_post_flag = make_load(control(), adr, TypeInt::INT, T_INT, Compile::AliasIdxRaw, MemNode::unordered);
 520 
 521     // Test the should_post_on_exceptions_flag vs. 0
 522     Node* chk = _gvn.transform( new CmpINode(should_post_flag, intcon(0)) );
 523     Node* tst = _gvn.transform( new BoolNode(chk, BoolTest::eq) );
 524 
 525     // Branch to slow_path if should_post_on_exceptions_flag was true
 526     { BuildCutout unless(this, tst, PROB_MAX);
 527       // Do not try anything fancy if we're notifying the VM on every throw.
 528       // Cf. case Bytecodes::_athrow in parse2.cpp.
 529       uncommon_trap(reason, Deoptimization::Action_none,
 530                     (ciKlass*)NULL, (char*)NULL, must_throw);
 531     }
 532 
 533 }

 535 //------------------------------builtin_throw----------------------------------
 536 void GraphKit::builtin_throw(Deoptimization::DeoptReason reason, Node* arg) {
 537   bool must_throw = true;
 538 
 539   if (env()->jvmti_can_post_on_exceptions()) {
 540     // check if we must post exception events, take uncommon trap if so
 541     uncommon_trap_if_should_post_on_exceptions(reason, must_throw);
 542     // here if should_post_on_exceptions is false
 543     // continue on with the normal codegen
 544   }
 545 
 546   // If this particular condition has not yet happened at this
 547   // bytecode, then use the uncommon trap mechanism, and allow for
 548   // a future recompilation if several traps occur here.
 549   // If the throw is hot, try to use a more complicated inline mechanism
 550   // which keeps execution inside the compiled code.
 551   bool treat_throw_as_hot = false;
 552   ciMethodData* md = method()->method_data();
 553 
 554   if (ProfileTraps) {
 555     if (too_many_traps(reason)) {
 556       treat_throw_as_hot = true;
 557     }
 558     // (If there is no MDO at all, assume it is early in
 559     // execution, and that any deopts are part of the
 560     // startup transient, and don't need to be remembered.)
 561 
 562     // Also, if there is a local exception handler, treat all throws
 563     // as hot if there has been at least one in this method.
 564     if (C->trap_count(reason) != 0
 565         && method()->method_data()->trap_count(reason) != 0
 566         && has_ex_handler()) {
 567         treat_throw_as_hot = true;
 568     }
 569   }
 570 
 571   // If this throw happens frequently, an uncommon trap might cause
 572   // a performance pothole.  If there is a local exception handler,
 573   // and if this particular bytecode appears to be deoptimizing often,
 574   // let us handle the throw inline, with a preconstructed instance.
 575   // Note:   If the deopt count has blown up, the uncommon trap
 576   // runtime is going to flush this nmethod, not matter what.
 577   if (treat_throw_as_hot
 578       && (!StackTraceInThrowable || OmitStackTraceInFastThrow)) {
 579     // If the throw is local, we use a pre-existing instance and
 580     // punt on the backtrace.  This would lead to a missing backtrace
 581     // (a repeat of 4292742) if the backtrace object is ever asked
 582     // for its backtrace.
 583     // Fixing this remaining case of 4292742 requires some flavor of
 584     // escape analysis.  Leave that for the future.
 585     ciInstance* ex_obj = NULL;
 586     switch (reason) {
 587     case Deoptimization::Reason_null_check:
 588       ex_obj = env()->NullPointerException_instance();
 589       break;
 590     case Deoptimization::Reason_div0_check:
 591       ex_obj = env()->ArithmeticException_instance();
 592       break;
 593     case Deoptimization::Reason_range_check:
 594       ex_obj = env()->ArrayIndexOutOfBoundsException_instance();
 595       break;
 596     case Deoptimization::Reason_class_check:
 597       if (java_bc() == Bytecodes::_aastore) {
 598         ex_obj = env()->ArrayStoreException_instance();
 599       } else {
 600         ex_obj = env()->ClassCastException_instance();
 601       }
 602       break;
 603     default:
 604       break;
 605     }
 606     if (failing()) { stop(); return; }  // exception allocation might fail
 607     if (ex_obj != NULL) {
 608       // Cheat with a preallocated exception object.
 609       if (C->log() != NULL)
 610         C->log()->elem("hot_throw preallocated='1' reason='%s'",
 611                        Deoptimization::trap_reason_name(reason));
 612       const TypeInstPtr* ex_con  = TypeInstPtr::make(ex_obj);
 613       Node*              ex_node = _gvn.transform(ConNode::make(ex_con));
 614 
 615       // Clear the detail message of the preallocated exception object.
 616       // Weblogic sometimes mutates the detail message of exceptions
 617       // using reflection.
 618       int offset = java_lang_Throwable::get_detailMessage_offset();
 619       const TypePtr* adr_typ = ex_con->add_offset(offset);
 620 
 621       Node *adr = basic_plus_adr(ex_node, ex_node, offset);
 622       const TypeOopPtr* val_type = TypeOopPtr::make_from_klass(env()->String_klass());
 623       Node *store = access_store_at(ex_node, adr, adr_typ, null(), val_type, T_OBJECT, IN_HEAP);
 624 
 625       add_exception_state(make_exception_state(ex_node));
 626       return;
 627     }
 628   }
 629 
 630   // %%% Maybe add entry to OptoRuntime which directly throws the exc.?
 631   // It won't be much cheaper than bailing to the interp., since we'll
 632   // have to pass up all the debug-info, and the runtime will have to
 633   // create the stack trace.
 634 
 635   // Usual case:  Bail to interpreter.
 636   // Reserve the right to recompile if we haven't seen anything yet.
 637 
 638   ciMethod* m = Deoptimization::reason_is_speculate(reason) ? C->method() : NULL;
 639   Deoptimization::DeoptAction action = Deoptimization::Action_maybe_recompile;
 640   if (treat_throw_as_hot
 641       && (method()->method_data()->trap_recompiled_at(bci(), m)
 642           || C->too_many_traps(reason))) {
 643     // We cannot afford to take more traps here.  Suffer in the interpreter.
 644     if (C->log() != NULL)
 645       C->log()->elem("hot_throw preallocated='0' reason='%s' mcount='%d'",
 646                      Deoptimization::trap_reason_name(reason),
 647                      C->trap_count(reason));
 648     action = Deoptimization::Action_none;
 649   }
 650 
 651   // "must_throw" prunes the JVM state to include only the stack, if there
 652   // are no local exception handlers.  This should cut down on register
 653   // allocation time and code size, by drastically reducing the number
 654   // of in-edges on the call to the uncommon trap.
 655 
 656   uncommon_trap(reason, action, (ciKlass*)NULL, (char*)NULL, must_throw);
 657 }

 751 //----------------------------kill_dead_locals---------------------------------
 752 // Detect any locals which are known to be dead, and force them to top.
 753 void GraphKit::kill_dead_locals() {
 754   // Consult the liveness information for the locals.  If any
 755   // of them are unused, then they can be replaced by top().  This
 756   // should help register allocation time and cut down on the size
 757   // of the deoptimization information.
 758 
 759   // This call is made from many of the bytecode handling
 760   // subroutines called from the Big Switch in do_one_bytecode.
 761   // Every bytecode which might include a slow path is responsible
 762   // for killing its dead locals.  The more consistent we
 763   // are about killing deads, the fewer useless phis will be
 764   // constructed for them at various merge points.
 765 
 766   // bci can be -1 (InvocationEntryBci).  We return the entry
 767   // liveness for the method.
 768 
 769   if (method() == NULL || method()->code_size() == 0) {
 770     // We are building a graph for a call to a native method.
 771     // All locals are live.
 772     return;
 773   }
 774 
 775   ResourceMark rm;
 776 
 777   // Consult the liveness information for the locals.  If any
 778   // of them are unused, then they can be replaced by top().  This
 779   // should help register allocation time and cut down on the size
 780   // of the deoptimization information.
 781   MethodLivenessResult live_locals = method()->liveness_at_bci(bci());
 782 
 783   int len = (int)live_locals.size();
 784   assert(len <= jvms()->loc_size(), "too many live locals");
 785   for (int local = 0; local < len; local++) {
 786     if (!live_locals.at(local)) {
 787       set_local(local, top());
 788     }
 789   }
 790 }

 792 #ifdef ASSERT
 793 //-------------------------dead_locals_are_killed------------------------------
 794 // Return true if all dead locals are set to top in the map.
 795 // Used to assert "clean" debug info at various points.
 796 bool GraphKit::dead_locals_are_killed() {
 797   if (method() == NULL || method()->code_size() == 0) {
 798     // No locals need to be dead, so all is as it should be.
 799     return true;
 800   }
 801 
 802   // Make sure somebody called kill_dead_locals upstream.
 803   ResourceMark rm;
 804   for (JVMState* jvms = this->jvms(); jvms != NULL; jvms = jvms->caller()) {
 805     if (jvms->loc_size() == 0)  continue;  // no locals to consult
 806     SafePointNode* map = jvms->map();
 807     ciMethod* method = jvms->method();
 808     int       bci    = jvms->bci();
 809     if (jvms == this->jvms()) {
 810       bci = this->bci();  // it might not yet be synched
 811     }
 812     MethodLivenessResult live_locals = method->liveness_at_bci(bci);
 813     int len = (int)live_locals.size();
 814     if (!live_locals.is_valid() || len == 0)
 815       // This method is trivial, or is poisoned by a breakpoint.
 816       return true;
 817     assert(len == jvms->loc_size(), "live map consistent with locals map");
 818     for (int local = 0; local < len; local++) {
 819       if (!live_locals.at(local) && map->local(jvms, local) != top()) {
 820         if (PrintMiscellaneous && (Verbose || WizardMode)) {
 821           tty->print_cr("Zombie local %d: ", local);
 822           jvms->dump();
 823         }
 824         return false;
 825       }
 826     }
 827   }
 828   return true;
 829 }

 833 // Helper function for enforcing certain bytecodes to reexecute if
 834 // deoptimization happens
 835 static bool should_reexecute_implied_by_bytecode(JVMState *jvms, bool is_anewarray) {
 836   ciMethod* cur_method = jvms->method();
 837   int       cur_bci   = jvms->bci();
 838   if (cur_method != NULL && cur_bci != InvocationEntryBci) {
 839     Bytecodes::Code code = cur_method->java_code_at_bci(cur_bci);
 840     return Interpreter::bytecode_should_reexecute(code) ||
 841            (is_anewarray && (code == Bytecodes::_multianewarray));
 842     // Reexecute _multianewarray bytecode which was replaced with
 843     // sequence of [a]newarray. See Parse::do_multianewarray().
 844     //
 845     // Note: interpreter should not have it set since this optimization
 846     // is limited by dimensions and guarded by flag so in some cases
 847     // multianewarray() runtime calls will be generated and
 848     // the bytecode should not be reexecutes (stack will not be reset).
 849   } else {
 850     return false;
 851   }
 852 }

 854 // Helper function for adding JVMState and debug information to node
 855 void GraphKit::add_safepoint_edges(SafePointNode* call, bool must_throw) {
 856   // Add the safepoint edges to the call (or other safepoint).
 857 
 858   // Make sure dead locals are set to top.  This
 859   // should help register allocation time and cut down on the size
 860   // of the deoptimization information.
 861   assert(dead_locals_are_killed(), "garbage in debug info before safepoint");
 862 
 863   // Walk the inline list to fill in the correct set of JVMState's
 864   // Also fill in the associated edges for each JVMState.
 865 
 866   // If the bytecode needs to be reexecuted we need to put
 867   // the arguments back on the stack.
 868   const bool should_reexecute = jvms()->should_reexecute();
 869   JVMState* youngest_jvms = should_reexecute ? sync_jvms_for_reexecute() : sync_jvms();
 870 
 871   // NOTE: set_bci (called from sync_jvms) might reset the reexecute bit to
 872   // undefined if the bci is different.  This is normal for Parse but it
 873   // should not happen for LibraryCallKit because only one bci is processed.
 874   assert(!is_LibraryCallKit() || (jvms()->should_reexecute() == should_reexecute),
 875          "in LibraryCallKit the reexecute bit should not change");
 876 
 877   // If we are guaranteed to throw, we can prune everything but the
 878   // input to the current bytecode.
 879   bool can_prune_locals = false;
 880   uint stack_slots_not_pruned = 0;
 881   int inputs = 0, depth = 0;
 882   if (must_throw) {
 883     assert(method() == youngest_jvms->method(), "sanity");
 884     if (compute_stack_effects(inputs, depth)) {
 885       can_prune_locals = true;
 886       stack_slots_not_pruned = inputs;
 887     }
 888   }
 889 
 890   if (env()->should_retain_local_variables()) {
 891     // At any safepoint, this method can get breakpointed, which would
 892     // then require an immediate deoptimization.
 893     can_prune_locals = false;  // do not prune locals
 894     stack_slots_not_pruned = 0;
 895   }
 896 
 897   // do not scribble on the input jvms
 898   JVMState* out_jvms = youngest_jvms->clone_deep(C);
 899   call->set_jvms(out_jvms); // Start jvms list for call node
 900 
 901   // For a known set of bytecodes, the interpreter should reexecute them if
 902   // deoptimization happens. We set the reexecute state for them here
 903   if (out_jvms->is_reexecute_undefined() && //don't change if already specified
 904       should_reexecute_implied_by_bytecode(out_jvms, call->is_AllocateArray())) {
 905     out_jvms->set_should_reexecute(true); //NOTE: youngest_jvms not changed
 906   }
 907 
 908   // Presize the call:
 909   DEBUG_ONLY(uint non_debug_edges = call->req());
 910   call->add_req_batch(top(), youngest_jvms->debug_depth());
 911   assert(call->req() == non_debug_edges + youngest_jvms->debug_depth(), "");
 912 
 913   // Set up edges so that the call looks like this:
 914   //  Call [state:] ctl io mem fptr retadr
 915   //       [parms:] parm0 ... parmN
 916   //       [root:]  loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN
 917   //    [...mid:]   loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN [...]
 918   //       [young:] loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN
 919   // Note that caller debug info precedes callee debug info.
 920 
 921   // Fill pointer walks backwards from "young:" to "root:" in the diagram above:
 922   uint debug_ptr = call->req();
 923 
 924   // Loop over the map input edges associated with jvms, add them
 925   // to the call node, & reset all offsets to match call node array.
 926   for (JVMState* in_jvms = youngest_jvms; in_jvms != NULL; ) {
 927     uint debug_end   = debug_ptr;
 928     uint debug_start = debug_ptr - in_jvms->debug_size();
 929     debug_ptr = debug_start;  // back up the ptr
 930 
 931     uint p = debug_start;  // walks forward in [debug_start, debug_end)
 932     uint j, k, l;
 933     SafePointNode* in_map = in_jvms->map();
 934     out_jvms->set_map(call);
 935 
 936     if (can_prune_locals) {
 937       assert(in_jvms->method() == out_jvms->method(), "sanity");
 938       // If the current throw can reach an exception handler in this JVMS,
 939       // then we must keep everything live that can reach that handler.
 940       // As a quick and dirty approximation, we look for any handlers at all.
 941       if (in_jvms->method()->has_exception_handlers()) {
 942         can_prune_locals = false;
 943       }
 944     }
 945 
 946     // Add the Locals
 947     k = in_jvms->locoff();
 948     l = in_jvms->loc_size();
 949     out_jvms->set_locoff(p);
 950     if (!can_prune_locals) {
 951       for (j = 0; j < l; j++)
 952         call->set_req(p++, in_map->in(k+j));
 953     } else {
 954       p += l;  // already set to top above by add_req_batch
 955     }
 956 
 957     // Add the Expression Stack
 958     k = in_jvms->stkoff();
 959     l = in_jvms->sp();
 960     out_jvms->set_stkoff(p);
 961     if (!can_prune_locals) {
 962       for (j = 0; j < l; j++)
 963         call->set_req(p++, in_map->in(k+j));
 964     } else if (can_prune_locals && stack_slots_not_pruned != 0) {
 965       // Divide stack into {S0,...,S1}, where S0 is set to top.
 966       uint s1 = stack_slots_not_pruned;
 967       stack_slots_not_pruned = 0;  // for next iteration
 968       if (s1 > l)  s1 = l;
 969       uint s0 = l - s1;
 970       p += s0;  // skip the tops preinstalled by add_req_batch
 971       for (j = s0; j < l; j++)
 972         call->set_req(p++, in_map->in(k+j));
 973     } else {
 974       p += l;  // already set to top above by add_req_batch
 975     }
 976 
 977     // Add the Monitors
 978     k = in_jvms->monoff();
 979     l = in_jvms->mon_size();
 980     out_jvms->set_monoff(p);
 981     for (j = 0; j < l; j++)
 982       call->set_req(p++, in_map->in(k+j));
 983 
 984     // Copy any scalar object fields.
 985     k = in_jvms->scloff();
 986     l = in_jvms->scl_size();
 987     out_jvms->set_scloff(p);
 988     for (j = 0; j < l; j++)
 989       call->set_req(p++, in_map->in(k+j));
 990 
 991     // Finish the new jvms.
 992     out_jvms->set_endoff(p);
 993 
 994     assert(out_jvms->endoff()     == debug_end,             "fill ptr must match");
 995     assert(out_jvms->depth()      == in_jvms->depth(),      "depth must match");
 996     assert(out_jvms->loc_size()   == in_jvms->loc_size(),   "size must match");
 997     assert(out_jvms->mon_size()   == in_jvms->mon_size(),   "size must match");
 998     assert(out_jvms->scl_size()   == in_jvms->scl_size(),   "size must match");
 999     assert(out_jvms->debug_size() == in_jvms->debug_size(), "size must match");
1000 
1001     // Update the two tail pointers in parallel.
1002     out_jvms = out_jvms->caller();
1003     in_jvms  = in_jvms->caller();
1004   }
1005 
1006   assert(debug_ptr == non_debug_edges, "debug info must fit exactly");
1007 
1008   // Test the correctness of JVMState::debug_xxx accessors:
1009   assert(call->jvms()->debug_start() == non_debug_edges, "");
1010   assert(call->jvms()->debug_end()   == call->req(), "");
1011   assert(call->jvms()->debug_depth() == call->req() - non_debug_edges, "");
1012 }

1014 bool GraphKit::compute_stack_effects(int& inputs, int& depth) {
1015   Bytecodes::Code code = java_bc();
1016   if (code == Bytecodes::_wide) {
1017     code = method()->java_code_at_bci(bci() + 1);
1018   }
1019 
1020   BasicType rtype = T_ILLEGAL;
1021   int       rsize = 0;
1022 
1023   if (code != Bytecodes::_illegal) {
1024     depth = Bytecodes::depth(code); // checkcast=0, athrow=-1
1025     rtype = Bytecodes::result_type(code); // checkcast=P, athrow=V
1026     if (rtype < T_CONFLICT)
1027       rsize = type2size[rtype];
1028   }
1029 
1030   switch (code) {
1031   case Bytecodes::_illegal:
1032     return false;
1033 
1034   case Bytecodes::_ldc:
1035   case Bytecodes::_ldc_w:
1036   case Bytecodes::_ldc2_w:
1037     inputs = 0;
1038     break;
1039 
1040   case Bytecodes::_dup:         inputs = 1;  break;
1041   case Bytecodes::_dup_x1:      inputs = 2;  break;
1042   case Bytecodes::_dup_x2:      inputs = 3;  break;
1043   case Bytecodes::_dup2:        inputs = 2;  break;
1044   case Bytecodes::_dup2_x1:     inputs = 3;  break;
1045   case Bytecodes::_dup2_x2:     inputs = 4;  break;
1046   case Bytecodes::_swap:        inputs = 2;  break;
1047   case Bytecodes::_arraylength: inputs = 1;  break;
1048 
1049   case Bytecodes::_getstatic:
1050   case Bytecodes::_putstatic:
1051   case Bytecodes::_getfield:
1052   case Bytecodes::_putfield:
1053     {
1054       bool ignored_will_link;
1055       ciField* field = method()->get_field_at_bci(bci(), ignored_will_link);
1056       int      size  = field->type()->size();
1057       bool is_get = (depth >= 0), is_static = (depth & 1);
1058       inputs = (is_static ? 0 : 1);
1059       if (is_get) {
1060         depth = size - inputs;
1061       } else {
1062         inputs += size;        // putxxx pops the value from the stack
1063         depth = - inputs;
1064       }
1065     }
1066     break;
1067 
1068   case Bytecodes::_invokevirtual:
1069   case Bytecodes::_invokespecial:
1070   case Bytecodes::_invokestatic:
1071   case Bytecodes::_invokedynamic:
1072   case Bytecodes::_invokeinterface:
1073     {
1074       bool ignored_will_link;
1075       ciSignature* declared_signature = NULL;
1076       ciMethod* ignored_callee = method()->get_method_at_bci(bci(), ignored_will_link, &declared_signature);
1077       assert(declared_signature != NULL, "cannot be null");
1078       inputs   = declared_signature->arg_size_for_bc(code);
1079       int size = declared_signature->return_type()->size();
1080       depth = size - inputs;
1081     }
1082     break;
1083 
1084   case Bytecodes::_multianewarray:
1085     {
1086       ciBytecodeStream iter(method());
1087       iter.reset_to_bci(bci());
1088       iter.next();
1089       inputs = iter.get_dimensions();
1090       assert(rsize == 1, "");
1091       depth = rsize - inputs;
1092     }
1093     break;
1094 
1095   case Bytecodes::_withfield: {
1096     bool ignored_will_link;
1097     ciField* field = method()->get_field_at_bci(bci(), ignored_will_link);
1098     int      size  = field->type()->size();
1099     inputs = size+1;
1100     depth = rsize - inputs;
1101     break;
1102   }
1103 
1104   case Bytecodes::_ireturn:
1105   case Bytecodes::_lreturn:
1106   case Bytecodes::_freturn:
1107   case Bytecodes::_dreturn:
1108   case Bytecodes::_areturn:
1109     assert(rsize == -depth, "");
1110     inputs = rsize;
1111     break;
1112 
1113   case Bytecodes::_jsr:
1114   case Bytecodes::_jsr_w:
1115     inputs = 0;
1116     depth  = 1;                  // S.B. depth=1, not zero
1117     break;
1118 
1119   default:
1120     // bytecode produces a typed result
1121     inputs = rsize - depth;
1122     assert(inputs >= 0, "");
1123     break;
1124   }
1125 
1126 #ifdef ASSERT
1127   // spot check
1128   int outputs = depth + inputs;
1129   assert(outputs >= 0, "sanity");
1130   switch (code) {
1131   case Bytecodes::_checkcast: assert(inputs == 1 && outputs == 1, ""); break;
1132   case Bytecodes::_athrow:    assert(inputs == 1 && outputs == 0, ""); break;
1133   case Bytecodes::_aload_0:   assert(inputs == 0 && outputs == 1, ""); break;
1134   case Bytecodes::_return:    assert(inputs == 0 && outputs == 0, ""); break;
1135   case Bytecodes::_drem:      assert(inputs == 4 && outputs == 2, ""); break;
1136   default:                    break;
1137   }
1138 #endif //ASSERT
1139 
1140   return true;
1141 }

1207 //------------------------------do_null_check----------------------------------
1208 // Helper function to do a NULL pointer check.  Returned value is
1209 // the incoming address with NULL casted away.  You are allowed to use the
1210 // not-null value only if you are control dependent on the test.
1211 #ifndef PRODUCT
1212 extern int explicit_null_checks_inserted,
1213            explicit_null_checks_elided;
1214 #endif
1215 Node* GraphKit::null_check_common(Node* value, BasicType type,
1216                                   // optional arguments for variations:
1217                                   bool assert_null,
1218                                   Node* *null_control,
1219                                   bool speculative) {
1220   assert(!assert_null || null_control == NULL, "not both at once");
1221   if (stopped())  return top();
1222   NOT_PRODUCT(explicit_null_checks_inserted++);
1223 
1224   // Construct NULL check
1225   Node *chk = NULL;
1226   switch(type) {
1227     case T_LONG   : chk = new CmpLNode(value, _gvn.zerocon(T_LONG)); break;
1228     case T_INT    : chk = new CmpINode(value, _gvn.intcon(0)); break;
1229     case T_INLINE_TYPE : // fall through
1230     case T_ARRAY  : // fall through
1231       type = T_OBJECT;  // simplify further tests
1232     case T_OBJECT : {
1233       const Type *t = _gvn.type( value );
1234 
1235       const TypeOopPtr* tp = t->isa_oopptr();
1236       if (tp != NULL && tp->klass() != NULL && !tp->klass()->is_loaded()
1237           // Only for do_null_check, not any of its siblings:
1238           && !assert_null && null_control == NULL) {
1239         // Usually, any field access or invocation on an unloaded oop type
1240         // will simply fail to link, since the statically linked class is
1241         // likely also to be unloaded.  However, in -Xcomp mode, sometimes
1242         // the static class is loaded but the sharper oop type is not.
1243         // Rather than checking for this obscure case in lots of places,
1244         // we simply observe that a null check on an unloaded class
1245         // will always be followed by a nonsense operation, so we
1246         // can just issue the uncommon trap here.
1247         // Our access to the unloaded class will only be correct
1248         // after it has been loaded and initialized, which requires
1249         // a trip through the interpreter.
1250 #ifndef PRODUCT
1251         if (WizardMode) { tty->print("Null check of unloaded "); tp->klass()->print(); tty->cr(); }
1252 #endif
1253         uncommon_trap(Deoptimization::Reason_unloaded,
1254                       Deoptimization::Action_reinterpret,
1255                       tp->klass(), "!loaded");
1256         return top();
1257       }
1258 
1259       if (assert_null) {
1260         // See if the type is contained in NULL_PTR.
1261         // If so, then the value is already null.
1262         if (t->higher_equal(TypePtr::NULL_PTR)) {
1263           NOT_PRODUCT(explicit_null_checks_elided++);
1264           return value;           // Elided null assert quickly!
1265         }
1266       } else {
1267         // See if mixing in the NULL pointer changes type.
1268         // If so, then the NULL pointer was not allowed in the original
1269         // type.  In other words, "value" was not-null.
1270         if (t->meet(TypePtr::NULL_PTR) != t->remove_speculative()) {
1271           // same as: if (!TypePtr::NULL_PTR->higher_equal(t)) ...
1272           NOT_PRODUCT(explicit_null_checks_elided++);
1273           return value;           // Elided null check quickly!
1274         }
1275       }
1276       chk = new CmpPNode( value, null() );
1277       break;
1278     }
1279 
1280     default:
1281       fatal("unexpected type: %s", type2name(type));
1282   }
1283   assert(chk != NULL, "sanity check");
1284   chk = _gvn.transform(chk);
1285 
1286   BoolTest::mask btest = assert_null ? BoolTest::eq : BoolTest::ne;
1287   BoolNode *btst = new BoolNode( chk, btest);
1288   Node   *tst = _gvn.transform( btst );
1289 
1290   //-----------
1291   // if peephole optimizations occurred, a prior test existed.
1292   // If a prior test existed, maybe it dominates as we can avoid this test.
1293   if (tst != btst && type == T_OBJECT) {
1294     // At this point we want to scan up the CFG to see if we can
1295     // find an identical test (and so avoid this test altogether).
1296     Node *cfg = control();
1297     int depth = 0;
1298     while( depth < 16 ) {       // Limit search depth for speed
1299       if( cfg->Opcode() == Op_IfTrue &&
1300           cfg->in(0)->in(1) == tst ) {
1301         // Found prior test.  Use "cast_not_null" to construct an identical
1302         // CastPP (and hence hash to) as already exists for the prior test.
1303         // Return that casted value.
1304         if (assert_null) {
1305           replace_in_map(value, null());
1306           return null();  // do not issue the redundant test
1307         }
1308         Node *oldcontrol = control();
1309         set_control(cfg);
1310         Node *res = cast_not_null(value);
1311         set_control(oldcontrol);
1312         NOT_PRODUCT(explicit_null_checks_elided++);
1313         return res;
1314       }
1315       cfg = IfNode::up_one_dom(cfg, /*linear_only=*/ true);
1316       if (cfg == NULL)  break;  // Quit at region nodes
1317       depth++;
1318     }
1319   }
1320 
1321   //-----------
1322   // Branch to failure if null
1323   float ok_prob = PROB_MAX;  // a priori estimate:  nulls never happen
1324   Deoptimization::DeoptReason reason;
1325   if (assert_null) {
1326     reason = Deoptimization::reason_null_assert(speculative);
1327   } else if (type == T_OBJECT) {
1328     reason = Deoptimization::reason_null_check(speculative);
1329   } else {
1330     reason = Deoptimization::Reason_div0_check;
1331   }
1332   // %%% Since Reason_unhandled is not recorded on a per-bytecode basis,
1333   // ciMethodData::has_trap_at will return a conservative -1 if any
1334   // must-be-null assertion has failed.  This could cause performance
1335   // problems for a method after its first do_null_assert failure.
1336   // Consider using 'Reason_class_check' instead?
1337 
1338   // To cause an implicit null check, we set the not-null probability
1339   // to the maximum (PROB_MAX).  For an explicit check the probability
1340   // is set to a smaller value.
1341   if (null_control != NULL || too_many_traps(reason)) {
1342     // probability is less likely
1343     ok_prob =  PROB_LIKELY_MAG(3);
1344   } else if (!assert_null &&
1345              (ImplicitNullCheckThreshold > 0) &&
1346              method() != NULL &&
1347              (method()->method_data()->trap_count(reason)
1348               >= (uint)ImplicitNullCheckThreshold)) {
1349     ok_prob =  PROB_LIKELY_MAG(3);
1350   }
1351 
1352   if (null_control != NULL) {
1353     IfNode* iff = create_and_map_if(control(), tst, ok_prob, COUNT_UNKNOWN);
1354     Node* null_true = _gvn.transform( new IfFalseNode(iff));
1355     set_control(      _gvn.transform( new IfTrueNode(iff)));
1356 #ifndef PRODUCT
1357     if (null_true == top()) {
1358       explicit_null_checks_elided++;
1359     }
1360 #endif
1361     (*null_control) = null_true;
1362   } else {
1363     BuildCutout unless(this, tst, ok_prob);
1364     // Check for optimizer eliding test at parse time
1365     if (stopped()) {
1366       // Failure not possible; do not bother making uncommon trap.
1367       NOT_PRODUCT(explicit_null_checks_elided++);
1368     } else if (assert_null) {
1369       uncommon_trap(reason,
1370                     Deoptimization::Action_make_not_entrant,
1371                     NULL, "assert_null");
1372     } else {
1373       replace_in_map(value, zerocon(type));
1374       builtin_throw(reason);
1375     }
1376   }
1377 
1378   // Must throw exception, fall-thru not possible?
1379   if (stopped()) {
1380     return top();               // No result
1381   }
1382 
1383   if (assert_null) {
1384     // Cast obj to null on this path.
1385     replace_in_map(value, zerocon(type));
1386     return zerocon(type);
1387   }
1388 
1389   // Cast obj to not-null on this path, if there is no null_control.
1390   // (If there is a null_control, a non-null value may come back to haunt us.)
1391   if (type == T_OBJECT) {
1392     Node* cast = cast_not_null(value, false);
1393     if (null_control == NULL || (*null_control) == top())
1394       replace_in_map(value, cast);
1395     value = cast;
1396   }
1397 
1398   return value;
1399 }

1418 //------------------------------cast_not_null----------------------------------
1419 // Cast obj to not-null on this path
1420 Node* GraphKit::cast_not_null(Node* obj, bool do_replace_in_map) {
1421   if (obj->is_InlineType()) {
1422     return obj;
1423   }
1424   const Type *t = _gvn.type(obj);
1425   const Type *t_not_null = t->join_speculative(TypePtr::NOTNULL);
1426   // Object is already not-null?
1427   if( t == t_not_null ) return obj;
1428 
1429   Node *cast = new CastPPNode(obj,t_not_null);
1430   cast->init_req(0, control());
1431   cast = _gvn.transform( cast );
1432 
1433   if (t->is_inlinetypeptr() && t->inline_klass()->is_scalarizable()) {
1434     // Scalarize inline type now that we know it's non-null
1435     cast = InlineTypeNode::make_from_oop(this, cast, t->inline_klass())->as_ptr(&gvn());
1436   }
1437 
1438   // Scan for instances of 'obj' in the current JVM mapping.
1439   // These instances are known to be not-null after the test.
1440   if (do_replace_in_map)
1441     replace_in_map(obj, cast);
1442 
1443   return cast;                  // Return casted value
1444 }

1446 // Sometimes in intrinsics, we implicitly know an object is not null
1447 // (there's no actual null check) so we can cast it to not null. In
1448 // the course of optimizations, the input to the cast can become null.
1449 // In that case that data path will die and we need the control path
1450 // to become dead as well to keep the graph consistent. So we have to
1451 // add a check for null for which one branch can't be taken. It uses
1452 // an Opaque4 node that will cause the check to be removed after loop
1453 // opts so the test goes away and the compiled code doesn't execute a
1454 // useless check.
1455 Node* GraphKit::must_be_not_null(Node* value, bool do_replace_in_map) {
1456   if (!TypePtr::NULL_PTR->higher_equal(_gvn.type(value))) {
1457     return value;
1458   }
1459   Node* chk = _gvn.transform(new CmpPNode(value, null()));
1460   Node *tst = _gvn.transform(new BoolNode(chk, BoolTest::ne));
1461   Node* opaq = _gvn.transform(new Opaque4Node(C, tst, intcon(1)));
1462   IfNode *iff = new IfNode(control(), opaq, PROB_MAX, COUNT_UNKNOWN);
1463   _gvn.set_type(iff, iff->Value(&_gvn));
1464   Node *if_f = _gvn.transform(new IfFalseNode(iff));
1465   Node *frame = _gvn.transform(new ParmNode(C->start(), TypeFunc::FramePtr));
1466   Node* halt = _gvn.transform(new HaltNode(if_f, frame, "unexpected null in intrinsic"));
1467   C->root()->add_req(halt);
1468   Node *if_t = _gvn.transform(new IfTrueNode(iff));
1469   set_control(if_t);
1470   return cast_not_null(value, do_replace_in_map);
1471 }

1475 void GraphKit::replace_in_map(Node* old, Node* neww) {
1476   if (old == neww) {
1477     return;
1478   }
1479

1493     return;
1494   }
1495

1607 Node* GraphKit::access_store_at(Node* obj,
1608                                 Node* adr,
1609                                 const TypePtr* adr_type,
1610                                 Node* val,
1611                                 const Type* val_type,
1612                                 BasicType bt,

1759 //-------------------------array_element_address-------------------------
1760 Node* GraphKit::array_element_address(Node* ary, Node* idx, BasicType elembt,
1761                                       const TypeInt* sizetype, Node* ctrl) {
1762   uint shift  = exact_log2(type2aelembytes(elembt));
1763   ciKlass* arytype_klass = _gvn.type(ary)->is_aryptr()->klass();
1764   if (arytype_klass != NULL && arytype_klass->is_flat_array_klass()) {
1765     ciFlatArrayKlass* vak = arytype_klass->as_flat_array_klass();
1766     shift = vak->log2_element_size();
1767   }
1768   uint header = arrayOopDesc::base_offset_in_bytes(elembt);
1769 
1770   // short-circuit a common case (saves lots of confusing waste motion)
1771   jint idx_con = find_int_con(idx, -1);
1772   if (idx_con >= 0) {
1773     intptr_t offset = header + ((intptr_t)idx_con << shift);
1774     return basic_plus_adr(ary, offset);
1775   }
1776 
1777   // must be correct type for alignment purposes
1778   Node* base  = basic_plus_adr(ary, header);
1779   idx = Compile::conv_I2X_index(&_gvn, idx, sizetype, ctrl);
1780   Node* scale = _gvn.transform( new LShiftXNode(idx, intcon(shift)) );
1781   return basic_plus_adr(ary, base, scale);
1782 }

1797 //-------------------------set_arguments_for_java_call-------------------------
1798 // Arguments (pre-popped from the stack) are taken from the JVMS.
1799 void GraphKit::set_arguments_for_java_call(CallJavaNode* call, bool is_late_inline) {
1800   PreserveReexecuteState preexecs(this);
1801   if (EnableValhalla) {
1802     // Make sure the call is re-executed, if buffering of inline type arguments triggers deoptimization
1803     jvms()->set_should_reexecute(true);
1804     int arg_size = method()->get_declared_signature_at_bci(bci())->arg_size_for_bc(java_bc());
1805     inc_sp(arg_size);
1806   }
1807   // Add the call arguments
1808   const TypeTuple* domain = call->tf()->domain_sig();
1809   ExtendedSignature sig_cc = ExtendedSignature(call->method()->get_sig_cc(), SigEntryFilter());
1810   uint nargs = domain->cnt();
1811   for (uint i = TypeFunc::Parms, idx = TypeFunc::Parms; i < nargs; i++) {
1812     Node* arg = argument(i-TypeFunc::Parms);
1813     const Type* t = domain->field_at(i);
1814     if (call->method()->has_scalarized_args() && t->is_inlinetypeptr() && !t->maybe_null()) {
1815       // We don't pass inline type arguments by reference but instead pass each field of the inline type
1816       InlineTypeNode* vt = arg->as_InlineType();
1817       vt->pass_fields(this, call, sig_cc, idx);
1818       // If an inline type argument is passed as fields, attach the Method* to the call site
1819       // to be able to access the extended signature later via attached_method_before_pc().
1820       // For example, see CompiledMethod::preserve_callee_argument_oops().
1821       call->set_override_symbolic_info(true);
1822       continue;
1823     } else if (arg->is_InlineType()) {
1824       // Pass inline type argument via oop to callee
1825       arg = arg->as_InlineType()->buffer(this);
1826       if (!is_late_inline) {
1827         arg = arg->as_InlineTypePtr()->get_oop();
1828       }
1829     }
1830     call->init_req(idx++, arg);
1831     // Skip reserved arguments
1832     BasicType bt = t->basic_type();
1833     while (SigEntry::next_is_reserved(sig_cc, bt, true)) {
1834       call->init_req(idx++, top());
1835       if (type2size[bt] == 2) {
1836         call->init_req(idx++, top());
1837       }
1838     }
1839   }
1840 }

1910 //--------------------set_predefined_input_for_runtime_call--------------------
1911 // Reading and setting the memory state is way conservative here.
1912 // The real problem is that I am not doing real Type analysis on memory,
1913 // so I cannot distinguish card mark stores from other stores.  Across a GC
1914 // point the Store Barrier and the card mark memory has to agree.  I cannot
1915 // have a card mark store and its barrier split across the GC point from
1916 // either above or below.  Here I get that to happen by reading ALL of memory.
1917 // A better answer would be to separate out card marks from other memory.
1918 // For now, return the input memory state, so that it can be reused
1919 // after the call, if this call has restricted memory effects.
1920 Node* GraphKit::set_predefined_input_for_runtime_call(SafePointNode* call, Node* narrow_mem) {
1921   // Set fixed predefined input arguments
1922   Node* memory = reset_memory();
1923   Node* m = narrow_mem == NULL ? memory : narrow_mem;
1924   call->init_req( TypeFunc::Control,   control()  );
1925   call->init_req( TypeFunc::I_O,       top()      ); // does no i/o
1926   call->init_req( TypeFunc::Memory,    m          ); // may gc ptrs
1927   call->init_req( TypeFunc::FramePtr,  frameptr() );
1928   call->init_req( TypeFunc::ReturnAdr, top()      );
1929   return memory;

1932 //-------------------set_predefined_output_for_runtime_call--------------------
1933 // Set control and memory (not i_o) from the call.
1934 // If keep_mem is not NULL, use it for the output state,
1935 // except for the RawPtr output of the call, if hook_mem is TypeRawPtr::BOTTOM.
1936 // If hook_mem is NULL, this call produces no memory effects at all.
1937 // If hook_mem is a Java-visible memory slice (such as arraycopy operands),
1938 // then only that memory slice is taken from the call.
1939 // In the last case, we must put an appropriate memory barrier before
1940 // the call, so as to create the correct anti-dependencies on loads
1941 // preceding the call.
1942 void GraphKit::set_predefined_output_for_runtime_call(Node* call,
1943                                                       Node* keep_mem,
1944                                                       const TypePtr* hook_mem) {
1945   // no i/o
1946   set_control(_gvn.transform( new ProjNode(call,TypeFunc::Control) ));
1947   if (keep_mem) {
1948     // First clone the existing memory state
1949     set_all_memory(keep_mem);
1950     if (hook_mem != NULL) {
1951       // Make memory for the call
1952       Node* mem = _gvn.transform( new ProjNode(call, TypeFunc::Memory) );
1953       // Set the RawPtr memory state only.  This covers all the heap top/GC stuff
1954       // We also use hook_mem to extract specific effects from arraycopy stubs.
1955       set_memory(mem, hook_mem);
1956     }
1957     // ...else the call has NO memory effects.
1958 
1959     // Make sure the call advertises its memory effects precisely.
1960     // This lets us build accurate anti-dependences in gcm.cpp.
1961     assert(C->alias_type(call->adr_type()) == C->alias_type(hook_mem),
1962            "call node must be constructed correctly");
1963   } else {
1964     assert(hook_mem == NULL, "");
1965     // This is not a "slow path" call; all memory comes from the call.
1966     set_all_memory_call(call);
1967   }
1968 }

1983 // Replace the call with the current state of the kit.
1984 void GraphKit::replace_call(CallNode* call, Node* result, bool do_replaced_nodes) {
1985   JVMState* ejvms = NULL;
1986   if (has_exceptions()) {
1987     ejvms = transfer_exceptions_into_jvms();
1988   }
1989 
1990   ReplacedNodes replaced_nodes = map()->replaced_nodes();
1991   ReplacedNodes replaced_nodes_exception;
1992   Node* ex_ctl = top();
1993 
1994   SafePointNode* final_state = stop();
1995 
1996   // Find all the needed outputs of this call
1997   CallProjections* callprojs = call->extract_projections(true);
1998 
1999   Unique_Node_List wl;
2000   Node* init_mem = call->in(TypeFunc::Memory);
2001   Node* final_mem = final_state->in(TypeFunc::Memory);
2002   Node* final_ctl = final_state->in(TypeFunc::Control);
2003   Node* final_io = final_state->in(TypeFunc::I_O);
2004 
2005   // Replace all the old call edges with the edges from the inlining result
2006   if (callprojs->fallthrough_catchproj != NULL) {
2007     C->gvn_replace_by(callprojs->fallthrough_catchproj, final_ctl);
2008   }
2009   if (callprojs->fallthrough_memproj != NULL) {
2010     if (final_mem->is_MergeMem()) {
2011       // Parser's exits MergeMem was not transformed but may be optimized
2012       final_mem = _gvn.transform(final_mem);
2013     }
2014     C->gvn_replace_by(callprojs->fallthrough_memproj,   final_mem);
2015     add_mergemem_users_to_worklist(wl, final_mem);
2016   }
2017   if (callprojs->fallthrough_ioproj != NULL) {
2018     C->gvn_replace_by(callprojs->fallthrough_ioproj,    final_io);
2019   }
2020 
2021   // Replace the result with the new result if it exists and is used
2022   if (callprojs->resproj[0] != NULL && result != NULL) {
2023     assert(callprojs->nb_resproj == 1, "unexpected number of results");
2024     C->gvn_replace_by(callprojs->resproj[0], result);
2025   }
2026 
2027   if (ejvms == NULL) {
2028     // No exception edges to simply kill off those paths
2029     if (callprojs->catchall_catchproj != NULL) {
2030       C->gvn_replace_by(callprojs->catchall_catchproj, C->top());
2031     }
2032     if (callprojs->catchall_memproj != NULL) {
2033       C->gvn_replace_by(callprojs->catchall_memproj,   C->top());
2034     }
2035     if (callprojs->catchall_ioproj != NULL) {
2036       C->gvn_replace_by(callprojs->catchall_ioproj,    C->top());
2037     }
2038     // Replace the old exception object with top
2039     if (callprojs->exobj != NULL) {
2040       C->gvn_replace_by(callprojs->exobj, C->top());
2041     }
2042   } else {
2043     GraphKit ekit(ejvms);
2044 
2045     // Load my combined exception state into the kit, with all phis transformed:
2046     SafePointNode* ex_map = ekit.combine_and_pop_all_exception_states();
2047     replaced_nodes_exception = ex_map->replaced_nodes();
2048 
2049     Node* ex_oop = ekit.use_exception_state(ex_map);
2050 
2051     if (callprojs->catchall_catchproj != NULL) {
2052       C->gvn_replace_by(callprojs->catchall_catchproj, ekit.control());
2053       ex_ctl = ekit.control();
2054     }
2055     if (callprojs->catchall_memproj != NULL) {
2056       Node* ex_mem = ekit.reset_memory();
2057       C->gvn_replace_by(callprojs->catchall_memproj,   ex_mem);
2058       add_mergemem_users_to_worklist(wl, ex_mem);
2059     }
2060     if (callprojs->catchall_ioproj != NULL) {
2061       C->gvn_replace_by(callprojs->catchall_ioproj,    ekit.i_o());
2062     }
2063 
2064     // Replace the old exception object with the newly created one
2065     if (callprojs->exobj != NULL) {
2066       C->gvn_replace_by(callprojs->exobj, ex_oop);
2067     }
2068   }
2069 
2070   // Disconnect the call from the graph
2071   call->disconnect_inputs(NULL, C);
2072   C->gvn_replace_by(call, C->top());
2073 
2074   // Clean up any MergeMems that feed other MergeMems since the
2075   // optimizer doesn't like that.
2076   while (wl.size() > 0) {
2077     _gvn.transform(wl.pop());
2078   }
2079 
2080   if (callprojs->fallthrough_catchproj != NULL && !final_ctl->is_top() && do_replaced_nodes) {
2081     replaced_nodes.apply(C, final_ctl);
2082   }
2083   if (!ex_ctl->is_top() && do_replaced_nodes) {
2084     replaced_nodes_exception.apply(C, ex_ctl);
2085   }
2086 }

2106 //------------------------------uncommon_trap----------------------------------
2107 // Bail out to the interpreter in mid-method.  Implemented by calling the
2108 // uncommon_trap blob.  This helper function inserts a runtime call with the
2109 // right debug info.
2110 void GraphKit::uncommon_trap(int trap_request,
2111                              ciKlass* klass, const char* comment,
2112                              bool must_throw,
2113                              bool keep_exact_action) {
2114   if (failing())  stop();
2115   if (stopped())  return; // trap reachable?
2116 
2117   // Note:  If ProfileTraps is true, and if a deopt. actually
2118   // occurs here, the runtime will make sure an MDO exists.  There is
2119   // no need to call method()->ensure_method_data() at this point.
2120 
2121   // Set the stack pointer to the right value for reexecution:
2122   set_sp(reexecute_sp());
2123 
2124 #ifdef ASSERT
2125   if (!must_throw) {
2126     // Make sure the stack has at least enough depth to execute
2127     // the current bytecode.
2128     int inputs, ignored_depth;
2129     if (compute_stack_effects(inputs, ignored_depth)) {
2130       assert(sp() >= inputs, "must have enough JVMS stack to execute %s: sp=%d, inputs=%d",
2131              Bytecodes::name(java_bc()), sp(), inputs);
2132     }
2133   }
2134 #endif
2135 
2136   Deoptimization::DeoptReason reason = Deoptimization::trap_request_reason(trap_request);
2137   Deoptimization::DeoptAction action = Deoptimization::trap_request_action(trap_request);
2138 
2139   switch (action) {
2140   case Deoptimization::Action_maybe_recompile:

2522 // gvn to   opt_test    |   |
2523 //          /    \      |  <1>

2542 //------------------------------null_check_oop---------------------------------
2543 // Null check oop.  Set null-path control into Region in slot 3.
2544 // Make a cast-not-nullness use the other not-null control.  Return cast.
2545 Node* GraphKit::null_check_oop(Node* value, Node* *null_control,
2546                                bool never_see_null,
2547                                bool safe_for_replace,
2548                                bool speculative) {
2549   // Initial NULL check taken path
2550   (*null_control) = top();
2551   Node* cast = null_check_common(value, T_OBJECT, false, null_control, speculative);
2552 
2553   // Generate uncommon_trap:
2554   if (never_see_null && (*null_control) != top()) {
2555     // If we see an unexpected null at a check-cast we record it and force a
2556     // recompile; the offending check-cast will be compiled to handle NULLs.
2557     // If we see more than one offending BCI, then all checkcasts in the
2558     // method will be compiled to handle NULLs.
2559     PreserveJVMState pjvms(this);
2560     set_control(*null_control);
2561     replace_in_map(value, null());
2562     Deoptimization::DeoptReason reason = Deoptimization::reason_null_check(speculative);
2563     uncommon_trap(reason,
2564                   Deoptimization::Action_make_not_entrant);
2565     (*null_control) = top();    // NULL path is dead
2566   }
2567   if ((*null_control) == top() && safe_for_replace) {
2568     replace_in_map(value, cast);
2569   }
2570 
2571   // Cast away null-ness on the result

2621   // except that the memory effects of the call are restricted to AliasIdxRaw.

2761 //-------------------------------gen_subtype_check-----------------------------
2762 // Generate a subtyping check.  Takes as input the subtype and supertype.
2763 // Returns 2 values: sets the default control() to the true path and returns
2764 // the false path.  Only reads invariant memory; sets no (visible) memory.
2765 // The PartialSubtypeCheckNode sets the hidden 1-word cache in the encoding
2766 // but that's not exposed to the optimizer.  This call also doesn't take in an
2767 // Object; if you wish to check an Object you need to load the Object's class
2768 // prior to coming here.
2769 Node* Phase::gen_subtype_check(Node* subklass, Node* superklass, Node** ctrl, Node* mem, PhaseGVN& gvn) {
2770   Compile* C = gvn.C;
2771   if ((*ctrl)->is_top()) {
2772     return C->top();
2773   }
2774 
2775   // Fast check for identical types, perhaps identical constants.
2776   // The types can even be identical non-constants, in cases
2777   // involving Array.newInstance, Object.clone, etc.
2778   if (subklass == superklass)
2779     return C->top();             // false path is dead; no test needed.
2780 
2781   if (gvn.type(superklass)->singleton()) {
2782     ciKlass* superk = gvn.type(superklass)->is_klassptr()->klass();
2783     ciKlass* subk   = gvn.type(subklass)->is_klassptr()->klass();
2784 
2785     // In the common case of an exact superklass, try to fold up the
2786     // test before generating code.  You may ask, why not just generate
2787     // the code and then let it fold up?  The answer is that the generated
2788     // code will necessarily include null checks, which do not always
2789     // completely fold away.  If they are also needless, then they turn
2790     // into a performance loss.  Example:
2791     //    Foo[] fa = blah(); Foo x = fa[0]; fa[1] = x;
2792     // Here, the type of 'fa' is often exact, so the store check
2793     // of fa[1]=x will fold up, without testing the nullness of x.
2794     switch (C->static_subtype_check(superk, subk)) {
2795     case Compile::SSC_always_false:
2796       {
2797         Node* always_fail = *ctrl;
2798         *ctrl = gvn.C->top();
2799         return always_fail;
2800       }
2801     case Compile::SSC_always_true:
2802       return C->top();
2803     case Compile::SSC_easy_test:
2804       {
2805         // Just do a direct pointer compare and be done.
2806         IfNode* iff = gen_subtype_check_compare(*ctrl, subklass, superklass, BoolTest::eq, PROB_STATIC_FREQUENT, gvn, T_ADDRESS);
2807         *ctrl = gvn.transform(new IfTrueNode(iff));
2808         return gvn.transform(new IfFalseNode(iff));
2809       }
2810     case Compile::SSC_full_test:
2811       break;
2812     default:
2813       ShouldNotReachHere();
2814     }
2815   }
2816 
2817   // %%% Possible further optimization:  Even if the superklass is not exact,
2818   // if the subklass is the unique subtype of the superklass, the check
2819   // will always succeed.  We could leave a dependency behind to ensure this.
2820 
2821   // First load the super-klass's check-offset
2822   Node *p1 = gvn.transform(new AddPNode(superklass, superklass, gvn.MakeConX(in_bytes(Klass::super_check_offset_offset()))));
2823   Node* m = C->immutable_memory();
2824   Node *chk_off = gvn.transform(new LoadINode(NULL, m, p1, gvn.type(p1)->is_ptr(), TypeInt::INT, MemNode::unordered));
2825   int cacheoff_con = in_bytes(Klass::secondary_super_cache_offset());
2826   bool might_be_cache = (gvn.find_int_con(chk_off, cacheoff_con) == cacheoff_con);
2827 
2828   // Load from the sub-klass's super-class display list, or a 1-word cache of
2829   // the secondary superclass list, or a failing value with a sentinel offset
2830   // if the super-klass is an interface or exceptionally deep in the Java
2831   // hierarchy and we have to scan the secondary superclass list the hard way.
2832   // Worst-case type is a little odd: NULL is allowed as a result (usually
2833   // klass loads can never produce a NULL).
2834   Node *chk_off_X = chk_off;
2835 #ifdef _LP64
2836   chk_off_X = gvn.transform(new ConvI2LNode(chk_off_X));
2837 #endif
2838   Node *p2 = gvn.transform(new AddPNode(subklass,subklass,chk_off_X));
2839   // For some types like interfaces the following loadKlass is from a 1-word
2840   // cache which is mutable so can't use immutable memory.  Other
2841   // types load from the super-class display table which is immutable.
2842   Node *kmem = C->immutable_memory();
2843   // secondary_super_cache is not immutable but can be treated as such because:
2844   // - no ideal node writes to it in a way that could cause an
2845   //   incorrect/missed optimization of the following Load.
2846   // - it's a cache so, worse case, not reading the latest value
2847   //   wouldn't cause incorrect execution
2848   if (might_be_cache && mem != NULL) {
2849     kmem = mem->is_MergeMem() ? mem->as_MergeMem()->memory_at(C->get_alias_index(gvn.type(p2)->is_ptr())) : mem;
2850   }
2851   Node *nkls = gvn.transform(LoadKlassNode::make(gvn, NULL, kmem, p2, gvn.type(p2)->is_ptr(), TypeKlassPtr::OBJECT_OR_NULL));
2852 
2853   // Compile speed common case: ARE a subtype and we canNOT fail
2854   if( superklass == nkls )
2855     return C->top();             // false path is dead; no test needed.
2856 
2857   // See if we get an immediate positive hit.  Happens roughly 83% of the
2858   // time.  Test to see if the value loaded just previously from the subklass
2859   // is exactly the superklass.
2860   IfNode *iff1 = gen_subtype_check_compare(*ctrl, superklass, nkls, BoolTest::eq, PROB_LIKELY(0.83f), gvn, T_ADDRESS);
2861   Node *iftrue1 = gvn.transform( new IfTrueNode (iff1));
2862   *ctrl = gvn.transform(new IfFalseNode(iff1));
2863 
2864   // Compile speed common case: Check for being deterministic right now.  If
2865   // chk_off is a constant and not equal to cacheoff then we are NOT a
2866   // subklass.  In this case we need exactly the 1 test above and we can
2867   // return those results immediately.
2868   if (!might_be_cache) {
2869     Node* not_subtype_ctrl = *ctrl;
2870     *ctrl = iftrue1; // We need exactly the 1 test above
2871     return not_subtype_ctrl;
2872   }
2873 
2874   // Gather the various success & failures here
2875   RegionNode *r_ok_subtype = new RegionNode(4);
2876   gvn.record_for_igvn(r_ok_subtype);
2877   RegionNode *r_not_subtype = new RegionNode(3);
2878   gvn.record_for_igvn(r_not_subtype);
2879 
2880   r_ok_subtype->init_req(1, iftrue1);
2881 
2882   // Check for immediate negative hit.  Happens roughly 11% of the time (which
2883   // is roughly 63% of the remaining cases).  Test to see if the loaded
2884   // check-offset points into the subklass display list or the 1-element
2885   // cache.  If it points to the display (and NOT the cache) and the display
2886   // missed then it's not a subtype.
2887   Node *cacheoff = gvn.intcon(cacheoff_con);
2888   IfNode *iff2 = gen_subtype_check_compare(*ctrl, chk_off, cacheoff, BoolTest::ne, PROB_LIKELY(0.63f), gvn, T_INT);
2889   r_not_subtype->init_req(1, gvn.transform(new IfTrueNode (iff2)));
2890   *ctrl = gvn.transform(new IfFalseNode(iff2));
2891 
2892   // Check for self.  Very rare to get here, but it is taken 1/3 the time.
2893   // No performance impact (too rare) but allows sharing of secondary arrays
2894   // which has some footprint reduction.
2895   IfNode *iff3 = gen_subtype_check_compare(*ctrl, subklass, superklass, BoolTest::eq, PROB_LIKELY(0.36f), gvn, T_ADDRESS);
2896   r_ok_subtype->init_req(2, gvn.transform(new IfTrueNode(iff3)));
2897   *ctrl = gvn.transform(new IfFalseNode(iff3));
2898 
2899   // -- Roads not taken here: --
2900   // We could also have chosen to perform the self-check at the beginning
2901   // of this code sequence, as the assembler does.  This would not pay off
2902   // the same way, since the optimizer, unlike the assembler, can perform
2903   // static type analysis to fold away many successful self-checks.
2904   // Non-foldable self checks work better here in second position, because
2905   // the initial primary superclass check subsumes a self-check for most
2906   // types.  An exception would be a secondary type like array-of-interface,
2907   // which does not appear in its own primary supertype display.
2908   // Finally, we could have chosen to move the self-check into the
2909   // PartialSubtypeCheckNode, and from there out-of-line in a platform
2910   // dependent manner.  But it is worthwhile to have the check here,
2911   // where it can be perhaps be optimized.  The cost in code space is
2912   // small (register compare, branch).
2913 
2914   // Now do a linear scan of the secondary super-klass array.  Again, no real
2915   // performance impact (too rare) but it's gotta be done.
2916   // Since the code is rarely used, there is no penalty for moving it
2917   // out of line, and it can only improve I-cache density.
2918   // The decision to inline or out-of-line this final check is platform
2919   // dependent, and is found in the AD file definition of PartialSubtypeCheck.
2920   Node* psc = gvn.transform(
2921     new PartialSubtypeCheckNode(*ctrl, subklass, superklass));
2922 
2923   IfNode *iff4 = gen_subtype_check_compare(*ctrl, psc, gvn.zerocon(T_OBJECT), BoolTest::ne, PROB_FAIR, gvn, T_ADDRESS);
2924   r_not_subtype->init_req(2, gvn.transform(new IfTrueNode (iff4)));
2925   r_ok_subtype ->init_req(3, gvn.transform(new IfFalseNode(iff4)));
2926 
2927   // Return false path; set default control to true path.
2928   *ctrl = gvn.transform(r_ok_subtype);
2929   return gvn.transform(r_not_subtype);
2930 }

3009 // Use null_seen information if it is available from the profile.
3010 // If we see an unexpected null at a type check we record it and force a

3022       return true;

3025       // Edge case:  no mature data.  Be optimistic here.

3210 // Generate an instance-of idiom.  Used by both the instance-of bytecode
3211 // and the reflective instance-of call.
3212 Node* GraphKit::gen_instanceof(Node* obj, Node* superklass, bool safe_for_replace) {
3213   kill_dead_locals();           // Benefit all the uncommon traps
3214   assert( !stopped(), "dead parse path should be checked in callers" );
3215   assert(!TypePtr::NULL_PTR->higher_equal(_gvn.type(superklass)->is_klassptr()),
3216          "must check for not-null not-dead klass in callers");
3217

3219   enum { _obj_path = 1, _fail_path, _null_path, PATH_LIMIT };

3226     data = method()->method_data()->bci_to_data(bci());

3228   bool speculative_not_null = false;
3229   bool never_see_null = (ProfileDynamicTypes  // aggressive use of profile
3230                          && seems_never_null(obj, data, speculative_not_null));

3234   Node* null_ctl = top();

3236 
3237   // If not_null_obj is dead, only null-path is taken
3238   if (stopped()) {              // Doing instance-of on a NULL?
3239     set_control(null_ctl);
3240     return intcon(0);
3241   }
3242   region->init_req(_null_path, null_ctl);

3244   if (null_ctl == top()) {
3245     // Do this eagerly, so that pattern matches like is_diamond_phi
3246     // will work even during parsing.
3247     assert(_null_path == PATH_LIMIT-1, "delete last");
3248     region->del_req(_null_path);
3249     phi   ->del_req(_null_path);
3250   }
3251 
3252   // Do we know the type check always succeed?
3253   if (!is_value) {
3254     bool known_statically = false;
3255     if (_gvn.type(superklass)->singleton()) {

3267       // have a speculative type use it to perform an exact cast.

3282     }

3293   region->init_req(_fail_path, not_subtype_ctrl);

3295

3297   set_control( _gvn.transform(region) );

3301   // profiling data at this bytecode. Don't lose it, feed it to the
3302   // type system as a speculative type.
3303   if (safe_for_replace && !is_value) {
3304     Node* casted_obj = record_profiled_receiver_for_speculation(obj);
3305     replace_in_map(obj, casted_obj);
3306   }
3307

3311 //-------------------------------gen_checkcast---------------------------------
3312 // Generate a checkcast idiom.  Used by both the checkcast bytecode and the
3313 // array store bytecode.  Stack must be as-if BEFORE doing the bytecode so the
3314 // uncommon-trap paths work.  Adjust stack after this call.
3315 // If failure_control is supplied and not null, it is filled in with
3316 // the control edge for the cast failure.  Otherwise, an appropriate
3317 // uncommon trap or exception is thrown.
3318 Node* GraphKit::gen_checkcast(Node *obj, Node* superklass, Node* *failure_control) {
3319   kill_dead_locals();           // Benefit all the uncommon traps
3320   const TypeKlassPtr* tk = _gvn.type(superklass)->is_klassptr();
3321   const TypeOopPtr* toop = TypeOopPtr::make_from_klass(tk->klass());
3322 
3323   // Check if inline types are involved
3324   bool from_inline = obj->is_InlineType();
3325   bool to_inline = tk->klass()->is_inlinetype();
3326 
3327   // Fast cutout:  Check the case that the cast is vacuously true.
3328   // This detects the common cases where the test will short-circuit
3329   // away completely.  We do this before we perform the null check,
3330   // because if the test is going to turn into zero code, we don't
3331   // want a residual null check left around.  (Causes a slowdown,
3332   // for example, in some objArray manipulations, such as a[i]=a[j].)
3333   if (tk->singleton()) {
3334     ciKlass* klass = NULL;
3335     if (from_inline) {
3336       klass = _gvn.type(obj)->inline_klass();
3337     } else {
3338       const TypeOopPtr* objtp = _gvn.type(obj)->isa_oopptr();
3339       if (objtp != NULL) {
3340         klass = objtp->klass();
3341       }
3342     }
3343     if (klass != NULL) {
3344       switch (C->static_subtype_check(tk->klass(), klass)) {
3345       case Compile::SSC_always_true:
3346         // If we know the type check always succeed then we don't use
3347         // the profiling data at this bytecode. Don't lose it, feed it
3348         // to the type system as a speculative type.
3349         if (!from_inline) {
3350           obj = record_profiled_receiver_for_speculation(obj);
3351           if (to_inline) {
3352             obj = null_check(obj);
3353             if (toop->inline_klass()->is_scalarizable()) {
3354               obj = InlineTypeNode::make_from_oop(this, obj, toop->inline_klass());
3355             }
3356           }
3357         }
3358         return obj;
3359       case Compile::SSC_always_false:
3360         if (from_inline || to_inline) {
3361           if (!from_inline) {
3362             null_check(obj);
3363           }
3364           // Inline type is never null. Always throw an exception.
3365           builtin_throw(Deoptimization::Reason_class_check, makecon(TypeKlassPtr::make(klass)));
3366           return top();
3367         } else {
3368           // It needs a null check because a null will *pass* the cast check.
3369           return null_assert(obj);
3370         }
3371       }
3372     }
3373   }
3374 
3375   ciProfileData* data = NULL;
3376   bool safe_for_replace = false;
3377   if (failure_control == NULL) {        // use MDO in regular case only
3378     assert(java_bc() == Bytecodes::_aastore ||
3379            java_bc() == Bytecodes::_checkcast,
3380            "interpreter profiles type checks only for these BCs");
3381     if (method()->method_data()->is_mature()) {
3382       data = method()->method_data()->bci_to_data(bci());
3383     }
3384     safe_for_replace = true;
3385   }
3386 
3387   // Make the merge point
3388   enum { _obj_path = 1, _null_path, PATH_LIMIT };
3389   RegionNode* region = new RegionNode(PATH_LIMIT);
3390   Node*       phi    = new PhiNode(region, toop);
3391   _gvn.set_type(region, Type::CONTROL);
3392   _gvn.set_type(phi, toop);
3393 
3394   C->set_has_split_ifs(true); // Has chance for split-if optimization
3395 
3396   // Use null-cast information if it is available
3397   bool speculative_not_null = false;
3398   bool never_see_null = ((failure_control == NULL)  // regular case only
3399                          && seems_never_null(obj, data, speculative_not_null));
3400 
3401   // Null check; get casted pointer; set region slot 3
3402   Node* null_ctl = top();
3403   Node* not_null_obj = NULL;
3404   if (from_inline) {
3405     not_null_obj = obj;
3406   } else if (to_inline) {
3407     not_null_obj = null_check(obj);
3408   } else {
3409     not_null_obj = null_check_oop(obj, &null_ctl, never_see_null, safe_for_replace, speculative_not_null);
3410   }
3411 
3412   // If not_null_obj is dead, only null-path is taken
3413   if (stopped()) {              // Doing instance-of on a NULL?
3414     set_control(null_ctl);
3415     return null();
3416   }

3419   if (null_ctl == top()) {
3420     // Do this eagerly, so that pattern matches like is_diamond_phi
3421     // will work even during parsing.
3422     assert(_null_path == PATH_LIMIT-1, "delete last");
3423     region->del_req(_null_path);
3424     phi   ->del_req(_null_path);
3425   }

3429     // The following optimization tries to statically cast the speculative type of the object
3430     // (for example obtained during profiling) to the type of the superklass and then do a
3431     // dynamic check that the type of the object is what we expect. To work correctly
3432     // for checkcast and aastore the type of superklass should be exact.
3433     const TypeOopPtr* obj_type = _gvn.type(obj)->is_oopptr();

3439       if (cast_obj != NULL && cast_obj->is_InlineType()) {
3440         if (null_ctl != top()) {
3441           cast_obj = NULL; // A value that's sometimes null is not something we can optimize well
3442         } else {
3443           return cast_obj;
3444         }
3445       }
3446       if (cast_obj != NULL) {
3447         if (failure_control != NULL) // failure is now impossible
3448           (*failure_control) = top();
3449         // adjust the type of the phi to the exact klass:
3450         phi->raise_bottom_type(_gvn.type(cast_obj)->meet_speculative(TypePtr::NULL_PTR));
3451       }
3452     }
3453   }
3454 
3455   if (cast_obj == NULL) {
3456     // Generate the subtype check
3457     Node* not_subtype_ctrl = gen_subtype_check(not_null_obj, superklass);
3458 
3459     // Plug in success path into the merge
3460     cast_obj = from_inline ? not_null_obj : _gvn.transform(new CheckCastPPNode(control(), not_null_obj, toop));

3462     if (failure_control == NULL) {
3463       if (not_subtype_ctrl != top()) { // If failure is possible
3464         PreserveJVMState pjvms(this);
3465         set_control(not_subtype_ctrl);
3466         Node* obj_klass = NULL;
3467         if (from_inline) {
3468           obj_klass = makecon(TypeKlassPtr::make(_gvn.type(not_null_obj)->inline_klass()));
3469         } else {

3471         }

3473       }
3474     } else {
3475       (*failure_control) = not_subtype_ctrl;
3476     }
3477   }
3478 
3479   region->init_req(_obj_path, control());
3480   phi   ->init_req(_obj_path, cast_obj);
3481 
3482   // A merge of NULL or Casted-NotNull obj
3483   Node* res = _gvn.transform(phi);
3484 
3485   // Note I do NOT always 'replace_in_map(obj,result)' here.
3486   //  if( tk->klass()->can_be_primary_super()  )

3488     // I 'forget' that the Object is really now known to be a String.  I have to
3489     // do this because we don't have true union types for interfaces - if I store
3490     // a Baz into an array-of-Interface and then tell the optimizer it's an

3493   //  replace_in_map( obj, res );

3496   set_control( _gvn.transform(region) );
3497   record_for_igvn(region);
3498 
3499   bool not_inline = !toop->can_be_inline_type();
3500   bool not_flattened = !UseFlatArray || not_inline || (toop->is_inlinetypeptr() && !toop->inline_klass()->flatten_array());
3501   if (EnableValhalla && not_flattened) {
3502     // Check if obj has been loaded from an array
3503     obj = obj->isa_DecodeN() ? obj->in(1) : obj;

3510     } else if (obj->is_Phi()) {

3512       // TODO make this more robust (see JDK-8231346)

3523         if (!ary_t->is_not_null_free() && not_inline) {

3576 Node* GraphKit::gen_inline_array_null_guard(Node* ary, Node* val, int nargs, bool safe_for_replace) {

3601     // Since we were just successfully storing null, the array can't be null free.
3602     ary_t = ary_t->cast_to_not_null_free();

3604     if (safe_for_replace) {

3647 //-------------------------insert_mem_bar_volatile----------------------------
3648 // Memory barrier to avoid floating things around
3649 // The membar serves as a pinch point between both control and memory(alias_idx).
3650 // If you want to make a pinch point on all memory slices, do not use this
3651 // function (even with AliasIdxBot); use insert_mem_bar() instead.
3652 Node* GraphKit::insert_mem_bar_volatile(int opcode, int alias_idx, Node* precedent) {
3653   // When Parse::do_put_xxx updates a volatile field, it appends a series
3654   // of MemBarVolatile nodes, one for *each* volatile field alias category.
3655   // The first membar is on the same memory slice as the field store opcode.
3656   // This forces the membar to follow the store.  (Bug 6500685 broke this.)
3657   // All the other membars (for other volatile slices, including AliasIdxBot,
3658   // which stands for all unknown volatile slices) are control-dependent
3659   // on the first membar.  This prevents later volatile loads or stores
3660   // from sliding up past the just-emitted store.
3661 
3662   MemBarNode* mb = MemBarNode::make(C, opcode, alias_idx, precedent);
3663   mb->set_req(TypeFunc::Control,control());
3664   if (alias_idx == Compile::AliasIdxBot) {
3665     mb->set_req(TypeFunc::Memory, merged_memory()->base_memory());
3666   } else {
3667     assert(!(opcode == Op_Initialize && alias_idx != Compile::AliasIdxRaw), "fix caller");
3668     mb->set_req(TypeFunc::Memory, memory(alias_idx));
3669   }
3670   Node* membar = _gvn.transform(mb);
3671   set_control(_gvn.transform(new ProjNode(membar, TypeFunc::Control)));
3672   if (alias_idx == Compile::AliasIdxBot) {
3673     merged_memory()->set_base_memory(_gvn.transform(new ProjNode(membar, TypeFunc::Memory)));
3674   } else {
3675     set_memory(_gvn.transform(new ProjNode(membar, TypeFunc::Memory)),alias_idx);
3676   }
3677   return membar;
3678 }

3685   assert(SynchronizationEntryBCI == InvocationEntryBci, "");

3691     return NULL;

3792 //-------------------------------get_layout_helper-----------------------------
3793 // If the given klass is a constant or known to be an array,
3794 // fetch the constant layout helper value into constant_value
3795 // and return (Node*)NULL.  Otherwise, load the non-constant
3796 // layout helper value, and return the node which represents it.
3797 // This two-faced routine is useful because allocation sites
3798 // almost always feature constant types.
3799 Node* GraphKit::get_layout_helper(Node* klass_node, jint& constant_value) {
3800   const TypeKlassPtr* inst_klass = _gvn.type(klass_node)->isa_klassptr();
3801   if (!StressReflectiveCode && inst_klass != NULL) {
3802     ciKlass* klass = inst_klass->klass();
3803     assert(klass != NULL, "klass should not be NULL");
3804     bool    xklass = inst_klass->klass_is_exact();
3805     bool can_be_flattened = false;
3806     if (UseFlatArray && klass->is_obj_array_klass()) {
3807       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
3808       can_be_flattened = elem->can_be_inline_klass() && (!elem->is_inlinetype() || elem->as_inline_klass()->flatten_array());
3809     }
3810     if (xklass || (klass->is_array_klass() && !can_be_flattened)) {
3811       jint lhelper = klass->layout_helper();
3812       if (lhelper != Klass::_lh_neutral_value) {
3813         constant_value = lhelper;
3814         return (Node*) NULL;
3815       }
3816     }
3817   }
3818   constant_value = Klass::_lh_neutral_value;  // put in a known value
3819   Node* lhp = basic_plus_adr(klass_node, klass_node, in_bytes(Klass::layout_helper_offset()));
3820   return make_load(NULL, lhp, TypeInt::INT, T_INT, MemNode::unordered);
3821 }

3836 //---------------------------set_output_for_allocation-------------------------
3837 Node* GraphKit::set_output_for_allocation(AllocateNode* alloc,
3838                                           const TypeOopPtr* oop_type,
3839                                           bool deoptimize_on_exception) {
3840   int rawidx = Compile::AliasIdxRaw;
3841   alloc->set_req( TypeFunc::FramePtr, frameptr() );
3842   add_safepoint_edges(alloc);
3843   Node* allocx = _gvn.transform(alloc);
3844   set_control( _gvn.transform(new ProjNode(allocx, TypeFunc::Control) ) );

3846   set_memory ( _gvn.transform( new ProjNode(allocx, TypeFunc::Memory, true) ), rawidx );
3847   make_slow_call_ex(allocx, env()->Throwable_klass(), true, deoptimize_on_exception);
3848 
3849   // create a memory projection as for the normal control path
3850   Node* malloc = _gvn.transform(new ProjNode(allocx, TypeFunc::Memory));

3852 
3853   // a normal slow-call doesn't change i_o, but an allocation does
3854   // we create a separate i_o projection for the normal control path

3860                                                  rawoop)->as_Initialize();

3864     // Extract memory strands which may participate in the new object's
3865     // initialization, and source them from the new InitializeNode.
3866     // This will allow us to observe initializations when they occur,
3867     // and link them properly (as a group) to the InitializeNode.
3868     assert(init->in(InitializeNode::Memory) == malloc, "");
3869     MergeMemNode* minit_in = MergeMemNode::make(malloc);
3870     init->set_req(InitializeNode::Memory, minit_in);

3876     // to one of those has correct memory state
3877     set_memory(minit_out, C->get_alias_index(oop_type->add_offset(oopDesc::mark_offset_in_bytes())));
3878     set_memory(minit_out, C->get_alias_index(oop_type->add_offset(oopDesc::klass_offset_in_bytes())));
3879     if (oop_type->isa_aryptr()) {
3880       const TypeAryPtr* arytype = oop_type->is_aryptr();
3881       if (arytype->klass()->is_flat_array_klass()) {
3882         // Initially all flattened array accesses share a single slice
3883         // but that changes after parsing. Prepare the memory graph so
3884         // it can optimize flattened array accesses properly once they
3885         // don't share a single slice.
3886         assert(C->flattened_accesses_share_alias(), "should be set at parse time");

3896           int fieldidx = C->get_alias_index(adr_type, true);
3897           hook_memory_on_init(*this, fieldidx, minit_in, minit_out);

3899         C->set_flattened_accesses_share_alias(true);
3900         hook_memory_on_init(*this, C->get_alias_index(TypeAryPtr::INLINES), minit_in, minit_out);
3901       } else {
3902         const TypePtr* telemref = oop_type->add_offset(Type::OffsetBot);
3903         int            elemidx  = C->get_alias_index(telemref);
3904         hook_memory_on_init(*this, elemidx, minit_in, minit_out);
3905       }
3906     } else if (oop_type->isa_instptr()) {
3907       set_memory(minit_out, C->get_alias_index(oop_type)); // mark word

3925 
3926 #ifdef ASSERT
3927   { // Verify that the AllocateNode::Ideal_allocation recognizers work:

3929            "Ideal_allocation works");

3938       assert(alloc->in(AllocateNode::ALength)->is_top(), "no length, please");

3940   }

3942

3946 //---------------------------new_instance--------------------------------------
3947 // This routine takes a klass_node which may be constant (for a static type)
3948 // or may be non-constant (for reflective code).  It will work equally well
3949 // for either, and the graph will fold nicely if the optimizer later reduces
3950 // the type to a constant.
3951 // The optional arguments are for specialized use by intrinsics:
3952 //  - If 'extra_slow_test' if not null is an extra condition for the slow-path.
3953 //  - If 'return_size_val', report the the total object size to the caller.
3954 //  - deoptimize_on_exception controls how Java exceptions are handled (rethrow vs deoptimize)
3955 Node* GraphKit::new_instance(Node* klass_node,
3956                              Node* extra_slow_test,
3957                              Node* *return_size_val,
3958                              bool deoptimize_on_exception,
3959                              InlineTypeBaseNode* inline_type_node) {
3960   // Compute size in doublewords
3961   // The size is always an integral number of doublewords, represented
3962   // as a positive bytewise size stored in the klass's layout_helper.
3963   // The layout_helper also encodes (in a low bit) the need for a slow path.
3964   jint  layout_con = Klass::_lh_neutral_value;
3965   Node* layout_val = get_layout_helper(klass_node, layout_con);
3966   bool  layout_is_con = (layout_val == NULL);
3967 
3968   if (extra_slow_test == NULL)  extra_slow_test = intcon(0);
3969   // Generate the initial go-slow test.  It's either ALWAYS (return a
3970   // Node for 1) or NEVER (return a NULL) or perhaps (in the reflective
3971   // case) a computed value derived from the layout_helper.
3972   Node* initial_slow_test = NULL;
3973   if (layout_is_con) {
3974     assert(!StressReflectiveCode, "stress mode does not use these paths");
3975     bool must_go_slow = Klass::layout_helper_needs_slow_path(layout_con);
3976     initial_slow_test = must_go_slow ? intcon(1) : extra_slow_test;
3977   } else {   // reflective case
3978     // This reflective path is used by Unsafe.allocateInstance.
3979     // (It may be stress-tested by specifying StressReflectiveCode.)
3980     // Basically, we want to get into the VM is there's an illegal argument.
3981     Node* bit = intcon(Klass::_lh_instance_slow_path_bit);
3982     initial_slow_test = _gvn.transform( new AndINode(layout_val, bit) );
3983     if (extra_slow_test != intcon(0)) {
3984       initial_slow_test = _gvn.transform( new OrINode(initial_slow_test, extra_slow_test) );
3985     }
3986     // (Macro-expander will further convert this to a Bool, if necessary.)
3987   }
3988 
3989   // Find the size in bytes.  This is easy; it's the layout_helper.
3990   // The size value must be valid even if the slow path is taken.
3991   Node* size = NULL;
3992   if (layout_is_con) {
3993     size = MakeConX(Klass::layout_helper_size_in_bytes(layout_con));
3994   } else {   // reflective case
3995     // This reflective path is used by clone and Unsafe.allocateInstance.
3996     size = ConvI2X(layout_val);
3997 
3998     // Clear the low bits to extract layout_helper_size_in_bytes:
3999     assert((int)Klass::_lh_instance_slow_path_bit < BytesPerLong, "clear bit");
4000     Node* mask = MakeConX(~ (intptr_t)right_n_bits(LogBytesPerLong));
4001     size = _gvn.transform( new AndXNode(size, mask) );
4002   }
4003   if (return_size_val != NULL) {
4004     (*return_size_val) = size;
4005   }
4006 
4007   // This is a precise notnull oop of the klass.
4008   // (Actually, it need not be precise if this is a reflective allocation.)
4009   // It's what we cast the result to.
4010   const TypeKlassPtr* tklass = _gvn.type(klass_node)->isa_klassptr();
4011   if (!tklass)  tklass = TypeKlassPtr::OBJECT;
4012   const TypeOopPtr* oop_type = tklass->as_instance_type();
4013 
4014   // Now generate allocation code
4015 
4016   // The entire memory state is needed for slow path of the allocation
4017   // since GC and deoptimization can happen.
4018   Node *mem = reset_memory();
4019   set_all_memory(mem); // Create new memory state
4020 
4021   AllocateNode* alloc = new AllocateNode(C, AllocateNode::alloc_type(Type::TOP),
4022                                          control(), mem, i_o(),
4023                                          size, klass_node,
4024                                          initial_slow_test, inline_type_node);
4025 
4026   return set_output_for_allocation(alloc, oop_type, deoptimize_on_exception);
4027 }

4037 //-------------------------------new_array-------------------------------------
4038 // helper for newarray and anewarray
4039 // The 'length' parameter is (obviously) the length of the array.
4040 // See comments on new_instance for the meaning of the other arguments.
4041 Node* GraphKit::new_array(Node* klass_node,     // array klass (maybe variable)
4042                           Node* length,         // number of array elements
4043                           int   nargs,          // number of arguments to push back for uncommon trap
4044                           Node* *return_size_val,
4045                           bool deoptimize_on_exception) {
4046   jint  layout_con = Klass::_lh_neutral_value;
4047   Node* layout_val = get_layout_helper(klass_node, layout_con);
4048   bool  layout_is_con = (layout_val == NULL);
4049 
4050   if (!layout_is_con && !StressReflectiveCode &&
4051       !too_many_traps(Deoptimization::Reason_class_check)) {
4052     // This is a reflective array creation site.
4053     // Optimistically assume that it is a subtype of Object[],
4054     // so that we can fold up all the address arithmetic.
4055     layout_con = Klass::array_layout_helper(T_OBJECT);
4056     Node* cmp_lh = _gvn.transform( new CmpINode(layout_val, intcon(layout_con)) );
4057     Node* bol_lh = _gvn.transform( new BoolNode(cmp_lh, BoolTest::eq) );
4058     { BuildCutout unless(this, bol_lh, PROB_MAX);
4059       inc_sp(nargs);
4060       uncommon_trap(Deoptimization::Reason_class_check,
4061                     Deoptimization::Action_maybe_recompile);
4062     }
4063     layout_val = NULL;
4064     layout_is_con = true;
4065   }
4066 
4067   // Generate the initial go-slow test.  Make sure we do not overflow
4068   // if length is huge (near 2Gig) or negative!  We do not need
4069   // exact double-words here, just a close approximation of needed
4070   // double-words.  We can't add any offset or rounding bits, lest we
4071   // take a size -1 of bytes and make it positive.  Use an unsigned
4072   // compare, so negative sizes look hugely positive.
4073   int fast_size_limit = FastAllocateSizeLimit;
4074   if (layout_is_con) {
4075     assert(!StressReflectiveCode, "stress mode does not use these paths");
4076     // Increase the size limit if we have exact knowledge of array type.
4077     int log2_esize = Klass::layout_helper_log2_element_size(layout_con);
4078     fast_size_limit <<= MAX2(LogBytesPerLong - log2_esize, 0);
4079   }
4080 
4081   Node* initial_slow_cmp  = _gvn.transform( new CmpUNode( length, intcon( fast_size_limit ) ) );
4082   Node* initial_slow_test = _gvn.transform( new BoolNode( initial_slow_cmp, BoolTest::gt ) );
4083 
4084   // --- Size Computation ---
4085   // array_size = round_to_heap(array_header + (length << elem_shift));
4086   // where round_to_heap(x) == align_to(x, MinObjAlignmentInBytes)
4087   // and align_to(x, y) == ((x + y-1) & ~(y-1))
4088   // The rounding mask is strength-reduced, if possible.
4089   int round_mask = MinObjAlignmentInBytes - 1;
4090   Node* header_size = NULL;
4091   int   header_size_min  = arrayOopDesc::base_offset_in_bytes(T_BYTE);
4092   // (T_BYTE has the weakest alignment and size restrictions...)
4093   if (layout_is_con) {
4094     int       hsize  = Klass::layout_helper_header_size(layout_con);
4095     int       eshift = Klass::layout_helper_log2_element_size(layout_con);
4096     bool is_flat_array = Klass::layout_helper_is_flatArray(layout_con);

4503 Node* GraphKit::compress_string(Node* src, const TypeAryPtr* src_type, Node* dst, Node* count) {
4504   assert(Matcher::match_rule_supported(Op_StrCompressedCopy), "Intrinsic not supported");
4505   assert(src_type == TypeAryPtr::BYTES || src_type == TypeAryPtr::CHARS, "invalid source type");

4507   // the dependency between preceding and subsequent loads/stores.

4509   //  StoreB

4511   //  LoadB

4513   //  LoadB -> compress_string -> CharMem
4514   //             ... -> StoreB -> ByteMem
4515   // The intrinsic hides the dependency between LoadB and StoreB, causing
4516   // the load to read from memory not containing the result of the StoreB.
4517   // The correct memory graph should look like this:
4518   //  LoadB -> compress_string -> MergeMem(CharMem, StoreB(ByteMem))
4519   Node* mem = capture_memory(src_type, TypeAryPtr::BYTES);
4520   StrCompressedCopyNode* str = new StrCompressedCopyNode(control(), mem, src, dst, count);
4521   Node* res_mem = _gvn.transform(new SCMemProjNode(str));
4522   set_memory(res_mem, TypeAryPtr::BYTES);
