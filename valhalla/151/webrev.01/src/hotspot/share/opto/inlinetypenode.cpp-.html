 34 // Clones the inline type to handle control flow merges involving multiple inline types.
 35 // The inputs are replaced by PhiNodes to represent the merged values for the given region.
 36 InlineTypeBaseNode* InlineTypeBaseNode::clone_with_phis(PhaseGVN* gvn, Node* region) {
 37   assert(!has_phi_inputs(region), "already cloned with phis");
 38   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
 39 
 40   // Create a PhiNode for merging the oop values
 41   const Type* phi_type = Type::get_const_type(inline_klass());
 42   PhiNode* oop = PhiNode::make(region, vt->get_oop(), phi_type);
 43   gvn->set_type(oop, phi_type);
 44   vt->set_oop(oop);
 45 
 46   // Create a PhiNode each for merging the field values
 47   for (uint i = 0; i < vt->field_count(); ++i) {
 48     ciType* type = vt->field_type(i);
 49     Node*  value = vt->field_value(i);
 50     if (value->is_InlineTypeBase()) {
 51       // Handle flattened inline type fields recursively
 52       value = value->as_InlineTypeBase()->clone_with_phis(gvn, region);
 53     } else {
 54       phi_type = Type::get_const_type(type);
 55       value = PhiNode::make(region, value, phi_type);
 56       gvn->set_type(value, phi_type);
 57     }
 58     vt->set_field_value(i, value);
 59   }
 60   gvn->set_type(vt, vt->bottom_type());
 61   return vt;
 62 }

 64 // Checks if the inputs of the InlineTypeBaseTypeNode were replaced by PhiNodes
 65 // for the given region (see InlineTypeBaseTypeNode::clone_with_phis).
 66 bool InlineTypeBaseNode::has_phi_inputs(Node* region) {
 67   // Check oop input
 68   bool result = get_oop()->is_Phi() && get_oop()->as_Phi()->region() == region;
 69 #ifdef ASSERT
 70   if (result) {
 71     // Check all field value inputs for consistency
 72     for (uint i = Oop; i < field_count(); ++i) {
 73       Node* n = in(i);
 74       if (n->is_InlineTypeBase()) {
 75         assert(n->as_InlineTypeBase()->has_phi_inputs(region), "inconsistent phi inputs");
 76       } else {
 77         assert(n->is_Phi() && n->as_Phi()->region() == region, "inconsistent phi inputs");
 78       }
 79     }
 80   }
 81 #endif
 82   return result;
 83 }

 85 // Merges 'this' with 'other' by updating the input PhiNodes added by 'clone_with_phis'
 86 InlineTypeBaseNode* InlineTypeBaseNode::merge_with(PhaseGVN* gvn, const InlineTypeBaseNode* other, int pnum, bool transform) {
 87   // Merge oop inputs
 88   PhiNode* phi = get_oop()->as_Phi();
 89   phi->set_req(pnum, other->get_oop());
 90   if (transform) {
 91     set_oop(gvn->transform(phi));
 92     gvn->record_for_igvn(phi);
 93   }
 94   // Merge field values
 95   for (uint i = 0; i < field_count(); ++i) {
 96     Node* val1 =        field_value(i);
 97     Node* val2 = other->field_value(i);
 98     if (val1->is_InlineTypeBase()) {
 99       val1->as_InlineTypeBase()->merge_with(gvn, val2->as_InlineTypeBase(), pnum, transform);
100     } else {
101       assert(val1->is_Phi(), "must be a phi node");
102       assert(!val2->is_InlineType(), "inconsistent merge values");
103       val1->set_req(pnum, val2);
104     }
105     if (transform) {
106       set_field_value(i, gvn->transform(val1));
107       gvn->record_for_igvn(val1);
108     }
109   }
110   return this;
111 }

113 // Adds a new merge path to an inline type node with phi inputs
114 void InlineTypeBaseNode::add_new_path(Node* region) {
115   assert(has_phi_inputs(region), "must have phi inputs");
116 
117   PhiNode* phi = get_oop()->as_Phi();
118   phi->add_req(NULL);
119   assert(phi->req() == region->req(), "must be same size as region");
120 
121   for (uint i = 0; i < field_count(); ++i) {
122     Node* val = field_value(i);
123     if (val->is_InlineType()) {
124       val->as_InlineType()->add_new_path(region);
125     } else {
126       val->as_Phi()->add_req(NULL);
127       assert(val->req() == region->req(), "must be same size as region");
128     }
129   }
130 }

137 // Get the value of the field at the given offset.
138 // If 'recursive' is true, flattened inline type fields will be resolved recursively.
139 Node* InlineTypeBaseNode::field_value_by_offset(int offset, bool recursive) const {
140   // If the field at 'offset' belongs to a flattened inline type field, 'index' refers to the
141   // corresponding InlineTypeNode input and 'sub_offset' is the offset in flattened inline type.
142   int index = inline_klass()->field_index_by_offset(offset);
143   int sub_offset = offset - field_offset(index);
144   Node* value = field_value(index);
145   assert(value != NULL, "field value not found");
146   if (recursive && value->is_InlineType()) {
147     InlineTypeNode* vt = value->as_InlineType();
148     if (field_is_flattened(index)) {
149       // Flattened inline type field
150       sub_offset += vt->inline_klass()->first_field_offset(); // Add header size
151       return vt->field_value_by_offset(sub_offset, recursive);
152     } else {
153       assert(sub_offset == 0, "should not have a sub offset");
154       return vt;
155     }
156   }
157   assert(!(recursive && value->is_InlineType()), "should not be an inline type");
158   assert(sub_offset == 0, "offset mismatch");
159   return value;
160 }

177   uint i = 0;

188 bool InlineTypeBaseNode::field_is_flattened(uint index) const {
189   assert(index < field_count(), "index out of bounds");
190   ciField* field = inline_klass()->declared_nonstatic_field_at(index);

195 int InlineTypeBaseNode::make_scalar_in_safepoint(PhaseIterGVN* igvn, Unique_Node_List& worklist, SafePointNode* sfpt) {
196   ciInlineKlass* vk = inline_klass();
197   uint nfields = vk->nof_nonstatic_fields();
198   JVMState* jvms = sfpt->jvms();
199   int start = jvms->debug_start();
200   int end   = jvms->debug_end();
201   // Replace safepoint edge by SafePointScalarObjectNode and add field values
202   assert(jvms != NULL, "missing JVMS");
203   uint first_ind = (sfpt->req() - jvms->scloff());
204   SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(inline_ptr(),
205 #ifdef ASSERT
206                                                                   NULL,
207 #endif
208                                                                   first_ind, nfields);
209   sobj->init_req(0, igvn->C->root());
210   // Iterate over the inline type fields in order of increasing
211   // offset and add the field values to the safepoint.
212   for (uint j = 0; j < nfields; ++j) {
213     int offset = vk->nonstatic_field_at(j)->offset();
214     Node* value = field_value_by_offset(offset, true /* include flattened inline type fields */);
215     if (value->is_InlineType()) {
216       // Add inline type field to the worklist to process later
217       worklist.push(value);
218     }
219     sfpt->add_req(value);
220   }
221   jvms->set_endoff(sfpt->req());
222   sobj = igvn->transform(sobj)->as_SafePointScalarObject();
223   igvn->rehash_node_delayed(sfpt);
224   return sfpt->replace_edges_in_range(this, sobj, start, end);
225 }

227 void InlineTypeBaseNode::make_scalar_in_safepoints(PhaseIterGVN* igvn) {
228   // Process all safepoint uses and scalarize inline type
229   Unique_Node_List worklist;
230   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
231     SafePointNode* sfpt = fast_out(i)->isa_SafePoint();
232     if (sfpt != NULL && !sfpt->is_CallLeaf() && (!sfpt->is_Call() || sfpt->as_Call()->has_debug_use(this))) {
233       int nb = 0;
234       if (is_allocated(igvn) && get_oop()->is_Con()) {
235         // Inline type is allocated with a constant oop, link it directly
236         nb = sfpt->replace_edges_in_range(this, get_oop(), sfpt->jvms()->debug_start(), sfpt->jvms()->debug_end());
237         igvn->rehash_node_delayed(sfpt);
238       } else {
239         nb = make_scalar_in_safepoint(igvn, worklist, sfpt);
240       }
241       --i; imax -= nb;
242     }
243   }
244   // Now scalarize non-flattened fields
245   for (uint i = 0; i < worklist.size(); ++i) {
246     Node* vt = worklist.at(i);
247     vt->as_InlineType()->make_scalar_in_safepoints(igvn);
248   }
249   igvn->record_for_igvn(this);
250 }

252 const TypePtr* InlineTypeBaseNode::field_adr_type(Node* base, int offset, ciInstanceKlass* holder, DecoratorSet decorators, PhaseGVN& gvn) const {
253   const TypeAryPtr* ary_type = gvn.type(base)->isa_aryptr();
254   const TypePtr* adr_type = NULL;
255   bool is_array = ary_type != NULL;
256   if ((decorators & C2_MISMATCHED) != 0) {
257     adr_type = TypeRawPtr::BOTTOM;
258   } else if (is_array) {
259     // In the case of a flattened inline type array, each field has its own slice
260     adr_type = ary_type->with_field_offset(offset)->add_offset(Type::OffsetBot);
261   } else {
262     ciField* field = holder->get_field_by_offset(offset, false);
263     assert(field != NULL, "field not found");
264     adr_type = gvn.C->alias_type(field)->adr_type();
265   }
266   return adr_type;
267 }

269 void InlineTypeBaseNode::load(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
270   // Initialize the inline type by loading its field values from
271   // memory and adding the values as input edges to the node.
272   for (uint i = 0; i < field_count(); ++i) {
273     int offset = holder_offset + field_offset(i);
274     Node* value = NULL;
275     ciType* ft = field_type(i);
276     if (field_is_flattened(i)) {
277       // Recursively load the flattened inline type field
278       value = InlineTypeNode::make_from_flattened(kit, ft->as_inline_klass(), base, ptr, holder, offset, decorators);
279     } else {
280       const TypeOopPtr* oop_ptr = kit->gvn().type(base)->isa_oopptr();
281       bool is_array = (oop_ptr->isa_aryptr() != NULL);
282       if (base->is_Con() && !is_array) {
283         // If the oop to the inline type is constant (static final field), we can
284         // also treat the fields as constants because the inline type is immutable.
285         ciObject* constant_oop = oop_ptr->const_oop();
286         ciField* field = holder->get_field_by_offset(offset, false);
287         assert(field != NULL, "field not found");
288         ciConstant constant = constant_oop->as_instance()->field_value(field);
289         const Type* con_type = Type::make_from_constant(constant, /*require_const=*/ true);
290         assert(con_type != NULL, "type not found");
291         value = kit->gvn().transform(kit->makecon(con_type));
292         // Check type of constant which might be more precise than the static field type
293         if (con_type->is_inlinetypeptr()) {
294           assert(!con_type->is_zero_type(), "Inline types are null-free");
295           ft = con_type->inline_klass();
296         }
297       } else {
298         // Load field value from memory
299         const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
300         Node* adr = kit->basic_plus_adr(base, ptr, offset);
301         BasicType bt = type2field[ft->basic_type()];
302         assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
303         const Type* val_type = Type::get_const_type(ft);
304         if (is_array) {
305           decorators |= IS_ARRAY;
306         }
307         value = kit->access_load_at(base, adr, adr_type, val_type, bt, decorators);
308       }
309       if (ft->is_inlinetype()) {
310         // Loading a non-flattened inline type from memory
311         if (ft->as_inline_klass()->is_scalarizable()) {
312           value = InlineTypeNode::make_from_oop(kit, value, ft->as_inline_klass());
313         } else {
314           value = kit->null2default(value, ft->as_inline_klass());
315         }
316       }
317     }
318     set_field_value(i, value);
319   }
320 }

322 void InlineTypeBaseNode::store_flattened(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
323   // The inline type is embedded into the object without an oop header. Subtract the
324   // offset of the first field to account for the missing header when storing the values.
325   if (holder == NULL) {
326     holder = inline_klass();
327   }
328   holder_offset -= inline_klass()->first_field_offset();
329   store(kit, base, ptr, holder, holder_offset, decorators);
330 }

332 void InlineTypeBaseNode::store(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
333   // Write field values to memory
334   for (uint i = 0; i < field_count(); ++i) {
335     int offset = holder_offset + field_offset(i);
336     Node* value = field_value(i);
337     ciType* ft = field_type(i);
338     if (field_is_flattened(i)) {
339       // Recursively store the flattened inline type field
340       if (!value->is_InlineType()) {
341         assert(!kit->gvn().type(value)->maybe_null(), "Inline types are null-free");
342         value = InlineTypeNode::make_from_oop(kit, value, ft->as_inline_klass());
343       }
344       value->as_InlineType()->store_flattened(kit, base, ptr, holder, offset, decorators);
345     } else {
346       // Store field value to memory
347       const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
348       Node* adr = kit->basic_plus_adr(base, ptr, offset);
349       BasicType bt = type2field[ft->basic_type()];
350       assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
351       const Type* val_type = Type::get_const_type(ft);
352       const TypeAryPtr* ary_type = kit->gvn().type(base)->isa_aryptr();
353       if (ary_type != NULL) {
354         decorators |= IS_ARRAY;
355       }
356       kit->access_store_at(base, adr, adr_type, value, val_type, bt, decorators);
357     }
358   }
359 }

361 InlineTypePtrNode* InlineTypeBaseNode::buffer(GraphKit* kit, bool safe_for_replace) {
362   assert(is_InlineType(), "sanity");
363   // Check if inline type is already allocated
364   Node* null_ctl = kit->top();
365   Node* not_null_oop = kit->null_check_oop(get_oop(), &null_ctl);
366   if (null_ctl->is_top()) {
367     // Inline type is allocated
368     return as_ptr(&kit->gvn());
369   }
370   assert(!is_allocated(&kit->gvn()), "should not be allocated");
371   RegionNode* region = new RegionNode(3);
372 
373   // Oop is non-NULL, use it
374   region->init_req(1, kit->control());
375   PhiNode* oop = PhiNode::make(region, not_null_oop, inline_ptr());
376   PhiNode* io  = PhiNode::make(region, kit->i_o(), Type::ABIO);
377   PhiNode* mem = PhiNode::make(region, kit->merged_memory(), Type::MEMORY, TypePtr::BOTTOM);
378 
379   int bci = kit->bci();
380   bool reexecute = kit->jvms()->should_reexecute();
381   {
382     // Oop is NULL, allocate and initialize buffer
383     PreserveJVMState pjvms(kit);
384     // Propagate re-execution state and bci
385     kit->set_bci(bci);
386     kit->jvms()->set_bci(bci);
387     kit->jvms()->set_should_reexecute(reexecute);
388     kit->set_control(null_ctl);
389     kit->kill_dead_locals();
390     ciInlineKlass* vk = inline_klass();
391     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
392     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true, this);
393     store(kit, alloc_oop, alloc_oop, vk, 0);
394 
395     // Do not let stores that initialize this buffer be reordered with a subsequent
396     // store that would make this buffer accessible by other threads.
397     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
398     assert(alloc != NULL, "must have an allocation node");
399     kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
400 
401     region->init_req(2, kit->control());
402     oop   ->init_req(2, alloc_oop);
403     io    ->init_req(2, kit->i_o());
404     mem   ->init_req(2, kit->merged_memory());
405   }
406 
407   // Update GraphKit
408   kit->set_control(kit->gvn().transform(region));
409   kit->set_i_o(kit->gvn().transform(io));
410   kit->set_all_memory(kit->gvn().transform(mem));
411   kit->record_for_igvn(region);
412   kit->record_for_igvn(oop);
413   kit->record_for_igvn(io);
414   kit->record_for_igvn(mem);
415 
416   // Use cloned InlineTypeNode to propagate oop from now on
417   Node* res_oop = kit->gvn().transform(oop);
418   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
419   vt->set_oop(res_oop);
420   vt = kit->gvn().transform(vt)->as_InlineTypeBase();
421   if (safe_for_replace) {
422     kit->replace_in_map(this, vt);
423   }
424   // InlineTypeNode::remove_redundant_allocations piggybacks on split if.
425   // Make sure it gets a chance to remove this allocation.
426   kit->C->set_has_split_ifs(true);
427   return vt->as_ptr(&kit->gvn());
428 }

430 bool InlineTypeBaseNode::is_allocated(PhaseGVN* phase) const {
431   Node* oop = get_oop();
432   const Type* oop_type = (phase != NULL) ? phase->type(oop) : oop->bottom_type();
433   return !oop_type->maybe_null();
434 }

436 InlineTypePtrNode* InlineTypeBaseNode::as_ptr(PhaseGVN* phase) const {
437   assert(is_allocated(phase), "must be allocated");
438   if (is_InlineTypePtr()) {
439     return as_InlineTypePtr();
440   }
441   return phase->transform(new InlineTypePtrNode(this))->as_InlineTypePtr();
442 }

444 // When a call returns multiple values, it has several result
445 // projections, one per field. Replacing the result of the call by a
446 // inline type node (after late inlining) requires that for each result
447 // projection, we find the corresponding inline type field.
448 void InlineTypeBaseNode::replace_call_results(GraphKit* kit, Node* call, Compile* C) {
449   ciInlineKlass* vk = inline_klass();
450   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
451     ProjNode* pn = call->fast_out(i)->as_Proj();
452     uint con = pn->_con;
453     if (con >= TypeFunc::Parms+1) {
454       uint field_nb = con - (TypeFunc::Parms+1);
455       int extra = 0;
456       for (uint j = 0; j < field_nb - extra; j++) {
457         ciField* f = vk->nonstatic_field_at(j);
458         BasicType bt = f->type()->basic_type();
459         if (bt == T_LONG || bt == T_DOUBLE) {
460           extra++;
461         }
462       }
463       ciField* f = vk->nonstatic_field_at(field_nb - extra);
464       Node* field = field_value_by_offset(f->offset(), true);
465       if (field->is_InlineType()) {
466         assert(field->as_InlineType()->is_allocated(&kit->gvn()), "must be allocated");
467         field = field->as_InlineType()->get_oop();
468       }
469       C->gvn_replace_by(pn, field);
470       C->initial_gvn()->hash_delete(pn);
471       pn->set_req(0, C->top());
472       --i; --imax;
473     }
474   }
475 }

477 Node* InlineTypeBaseNode::allocate_fields(GraphKit* kit) {
478   InlineTypeBaseNode* vt = clone()->as_InlineTypeBase();
479   for (uint i = 0; i < field_count(); i++) {
480      InlineTypeNode* value = field_value(i)->isa_InlineType();
481      if (field_is_flattened(i)) {
482        // Flattened inline type field
483        vt->set_field_value(i, value->allocate_fields(kit));
484      } else if (value != NULL) {
485        // Non-flattened inline type field
486        vt->set_field_value(i, value->buffer(kit));
487      }
488   }
489   vt = kit->gvn().transform(vt)->as_InlineTypeBase();
490   kit->replace_in_map(this, vt);
491   return vt;
492 }

494 InlineTypeNode* InlineTypeNode::make_uninitialized(PhaseGVN& gvn, ciInlineKlass* vk) {
495   // Create a new InlineTypeNode with uninitialized values and NULL oop
496   Node* oop = vk->is_empty() ? default_oop(gvn, vk) : gvn.zerocon(T_INLINE_TYPE);
497   return new InlineTypeNode(vk, oop);

500 Node* InlineTypeNode::default_oop(PhaseGVN& gvn, ciInlineKlass* vk) {
501   // Returns the constant oop of the default inline type allocation
502   return gvn.makecon(TypeInstPtr::make(vk->default_instance()));
503 }

505 InlineTypeNode* InlineTypeNode::make_default(PhaseGVN& gvn, ciInlineKlass* vk) {
506   // Create a new InlineTypeNode with default values
507   InlineTypeNode* vt = new InlineTypeNode(vk, default_oop(gvn, vk));
508   for (uint i = 0; i < vt->field_count(); ++i) {
509     ciType* field_type = vt->field_type(i);
510     Node* value = NULL;
511     if (field_type->is_inlinetype()) {
512       ciInlineKlass* field_klass = field_type->as_inline_klass();
513       if (field_klass->is_scalarizable()) {
514         value = InlineTypeNode::make_default(gvn, field_klass);
515       } else {
516         value = default_oop(gvn, field_klass);
517       }
518     } else {
519       value = gvn.zerocon(field_type->basic_type());
520     }
521     vt->set_field_value(i, value);
522   }
523   vt = gvn.transform(vt)->as_InlineType();
524   assert(vt->is_default(&gvn), "must be the default inline type");
525   return vt;
526 }
527 
528 bool InlineTypeNode::is_default(PhaseGVN* gvn) const {
529   for (uint i = 0; i < field_count(); ++i) {
530     Node* value = field_value(i);
531     if (!gvn->type(value)->is_zero_type() &&
532         !(value->is_InlineType() && value->as_InlineType()->is_default(gvn)) &&
533         !(field_type(i)->is_inlinetype() && value == default_oop(*gvn, field_type(i)->as_inline_klass()))) {
534       return false;
535     }
536   }
537   return true;
538 }

540 InlineTypeNode* InlineTypeNode::make_from_oop(GraphKit* kit, Node* oop, ciInlineKlass* vk) {
541   PhaseGVN& gvn = kit->gvn();
542 
543   // Create and initialize an InlineTypeNode by loading all field
544   // values from a heap-allocated version and also save the oop.
545   InlineTypeNode* vt = new InlineTypeNode(vk, oop);
546 
547   if (oop->isa_InlineTypePtr()) {
548     // Can happen with late inlining
549     InlineTypePtrNode* vtptr = oop->as_InlineTypePtr();
550     vt->set_oop(vtptr->get_oop());
551     for (uint i = Oop+1; i < vtptr->req(); ++i) {
552       vt->init_req(i, vtptr->in(i));
553     }
554   } else if (gvn.type(oop)->maybe_null()) {
555     // Add a null check because the oop may be null
556     Node* null_ctl = kit->top();
557     Node* not_null_oop = kit->null_check_oop(oop, &null_ctl);
558     if (kit->stopped()) {
559       // Constant null
560       kit->set_control(null_ctl);
561       return make_default(gvn, vk);
562     }
563     vt->set_oop(not_null_oop);
564     vt->load(kit, not_null_oop, not_null_oop, vk, /* holder_offset */ 0);
565 
566     if (null_ctl != kit->top()) {
567       // Return default inline type if oop is null
568       InlineTypeNode* def = make_default(gvn, vk);
569       Node* region = new RegionNode(3);
570       region->init_req(1, kit->control());
571       region->init_req(2, null_ctl);
572 
573       vt = vt->clone_with_phis(&gvn, region)->as_InlineType();
574       vt->merge_with(&gvn, def, 2, true);
575       kit->set_control(gvn.transform(region));
576     }
577   } else {
578     // Oop can never be null
579     Node* init_ctl = kit->control();
580     vt->load(kit, oop, oop, vk, /* holder_offset */ 0);
581     assert(init_ctl != kit->control() || !gvn.type(oop)->is_inlinetypeptr() || oop->is_Con() || oop->Opcode() == Op_InlineTypePtr ||
582            AllocateNode::Ideal_allocation(oop, &gvn) != NULL || vt->is_loaded(&gvn) == oop, "inline type should be loaded");
583   }
584 
585   assert(vt->is_allocated(&gvn), "inline type should be allocated");
586   return gvn.transform(vt)->as_InlineType();
587 }

589 // GraphKit wrapper for the 'make_from_flattened' method
590 InlineTypeNode* InlineTypeNode::make_from_flattened(GraphKit* kit, ciInlineKlass* vk, Node* obj, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
591   // Create and initialize an InlineTypeNode by loading all field values from
592   // a flattened inline type field at 'holder_offset' or from an inline type array.
593   InlineTypeNode* vt = make_uninitialized(kit->gvn(), vk);
594   // The inline type is flattened into the object without an oop header. Subtract the
595   // offset of the first field to account for the missing header when loading the values.
596   holder_offset -= vk->first_field_offset();
597   vt->load(kit, obj, ptr, holder, holder_offset, decorators);
598   assert(vt->is_loaded(&kit->gvn()) != obj, "holder oop should not be used as flattened inline type oop");
599   return kit->gvn().transform(vt)->as_InlineType();
600 }

608 InlineTypeNode* InlineTypeNode::make_larval(GraphKit* kit, bool allocate) const {
609   ciInlineKlass* vk = inline_klass();
610   InlineTypeNode* res = clone()->as_InlineType();
611   if (allocate) {
612     // Re-execute if buffering triggers deoptimization
613     PreserveReexecuteState preexecs(kit);
614     kit->jvms()->set_should_reexecute(true);
615     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
616     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, true);
617     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
618     alloc->_larval = true;
619 
620     store(kit, alloc_oop, alloc_oop, vk, 0);
621     res->set_oop(alloc_oop);
622   }
623   res->set_type(TypeInlineType::make(vk, true));
624   res = kit->gvn().transform(res)->as_InlineType();
625   assert(!allocate || res->is_allocated(&kit->gvn()), "must be allocated");
626   return res;
627 }

629 InlineTypeNode* InlineTypeNode::finish_larval(GraphKit* kit) const {
630   Node* obj = get_oop();
631   Node* mark_addr = kit->basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
632   Node* mark = kit->make_load(NULL, mark_addr, TypeX_X, TypeX_X->basic_type(), MemNode::unordered);
633   mark = kit->gvn().transform(new AndXNode(mark, kit->MakeConX(~markWord::larval_mask_in_place)));
634   kit->store_to_memory(kit->control(), mark_addr, mark, TypeX_X->basic_type(), kit->gvn().type(mark_addr)->is_ptr(), MemNode::unordered);
635 
636   // Do not let stores that initialize this buffer be reordered with a subsequent
637   // store that would make this buffer accessible by other threads.
638   AllocateNode* alloc = AllocateNode::Ideal_allocation(obj, &kit->gvn());
639   assert(alloc != NULL, "must have an allocation node");
640   kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
641 
642   ciInlineKlass* vk = inline_klass();
643   InlineTypeNode* res = clone()->as_InlineType();
644   res->set_type(TypeInlineType::make(vk, false));
645   res = kit->gvn().transform(res)->as_InlineType();
646   return res;
647 }

649 Node* InlineTypeNode::is_loaded(PhaseGVN* phase, ciInlineKlass* vk, Node* base, int holder_offset) {
650   if (vk == NULL) {
651     vk = inline_klass();
652   }
653   if (field_count() == 0) {
654     assert(is_allocated(phase), "must be allocated");
655     return get_oop();
656   }
657   for (uint i = 0; i < field_count(); ++i) {
658     int offset = holder_offset + field_offset(i);
659     Node* value = field_value(i);
660     if (value->is_InlineType()) {
661       InlineTypeNode* vt = value->as_InlineType();
662       if (field_is_flattened(i)) {
663         // Check inline type field load recursively
664         base = vt->is_loaded(phase, vk, base, offset - vt->inline_klass()->first_field_offset());
665         if (base == NULL) {
666           return NULL;
667         }
668         continue;
669       } else {
670         value = vt->get_oop();
671         if (value->Opcode() == Op_CastPP) {
672           // Skip CastPP
673           value = value->in(1);
674         }
675       }
676     }
677     if (value->isa_DecodeN()) {
678       // Skip DecodeN
679       value = value->in(1);
680     }
681     if (value->isa_Load()) {
682       // Check if base and offset of field load matches inline type layout
683       intptr_t loffset = 0;
684       Node* lbase = AddPNode::Ideal_base_and_offset(value->in(MemNode::Address), phase, loffset);
685       if (lbase == NULL || (lbase != base && base != NULL) || loffset != offset) {
686         return NULL;
687       } else if (base == NULL) {
688         // Set base and check if pointer type matches
689         base = lbase;
690         const TypeInstPtr* vtptr = phase->type(base)->isa_instptr();
691         if (vtptr == NULL || !vtptr->klass()->equals(vk)) {
692           return NULL;
693         }
694       }
695     } else {
696       return NULL;
697     }
698   }
699   return base;
700 }

702 Node* InlineTypeNode::tagged_klass(ciInlineKlass* vk, PhaseGVN& gvn) {
703   const TypeKlassPtr* tk = TypeKlassPtr::make(vk);

709 void InlineTypeNode::pass_fields(GraphKit* kit, Node* n, ExtendedSignature& sig, uint& base_input, int base_offset) {
710   for (uint i = 0; i < field_count(); i++) {
711     int sig_offset = (*sig)._offset;
712     uint idx = field_index(sig_offset - base_offset);
713     Node* arg = field_value(idx);
714 
715     if (field_is_flattened(idx)) {
716       // Flattened inline type field
717       InlineTypeNode* vt = arg->as_InlineType();
718       vt->pass_fields(kit, n, sig, base_input, sig_offset - vt->inline_klass()->first_field_offset());
719     } else {
720       if (arg->is_InlineType()) {
721         // Non-flattened inline type field
722         InlineTypeNode* vt = arg->as_InlineType();
723         assert(n->Opcode() != Op_Return || vt->is_allocated(&kit->gvn()), "inline type field should be allocated on return");
724         arg = vt->buffer(kit);
725       }
726       // Initialize call/return arguments
727       BasicType bt = field_type(i)->basic_type();
728       n->init_req(base_input++, arg);
729       if (type2size[bt] == 2) {
730         n->init_req(base_input++, kit->top());
731       }
732       // Skip reserved arguments
733       while (SigEntry::next_is_reserved(sig, bt)) {
734         n->init_req(base_input++, kit->top());
735         if (type2size[bt] == 2) {
736           n->init_req(base_input++, kit->top());
737         }
738       }
739     }
740   }
741 }

743 void InlineTypeNode::initialize_fields(GraphKit* kit, MultiNode* multi, ExtendedSignature& sig, uint& base_input, int base_offset, bool in) {
744   PhaseGVN& gvn = kit->gvn();
745   for (uint i = 0; i < field_count(); i++) {
746     int sig_offset = (*sig)._offset;
747     uint idx = field_index(sig_offset - base_offset);
748     ciType* type = field_type(idx);
749 
750     Node* parm = NULL;
751     if (field_is_flattened(idx)) {
752       // Flattened inline type field
753       InlineTypeNode* vt = InlineTypeNode::make_uninitialized(gvn, type->as_inline_klass());
754       vt->initialize_fields(kit, multi, sig, base_input, sig_offset - type->as_inline_klass()->first_field_offset(), in);
755       parm = gvn.transform(vt);
756     } else {
757       if (multi->is_Start()) {
758         assert(in, "return from start?");
759         parm = gvn.transform(new ParmNode(multi->as_Start(), base_input));
760       } else if (in) {
761         parm = multi->as_Call()->in(base_input);
762       } else {
763         parm = gvn.transform(new ProjNode(multi->as_Call(), base_input));
764       }
765       if (type->is_inlinetype()) {
766         // Non-flattened inline type field
767         if (type->as_inline_klass()->is_scalarizable()) {
768           parm = InlineTypeNode::make_from_oop(kit, parm, type->as_inline_klass());
769         } else {
770           parm = kit->null2default(parm, type->as_inline_klass());
771         }
772       }
773       base_input += type2size[type->basic_type()];
774       // Skip reserved arguments
775       BasicType bt = type->basic_type();
776       while (SigEntry::next_is_reserved(sig, bt)) {
777         base_input += type2size[bt];
778       }
779     }
780     assert(parm != NULL, "should never be null");
781     set_field_value(idx, parm);
782     gvn.record_for_igvn(parm);
783   }
784 }

786 // Replace a buffer allocation by a dominating allocation
787 static void replace_allocation(PhaseIterGVN* igvn, Node* res, Node* dom) {
788   // Remove initializing stores
789   for (DUIterator_Fast imax, i = res->fast_outs(imax); i < imax; i++) {
790     AddPNode* addp = res->fast_out(i)->isa_AddP();
791     if (addp != NULL) {
792       for (DUIterator_Fast jmax, j = addp->fast_outs(jmax); j < jmax; j++) {
793         StoreNode* store = addp->fast_out(j)->isa_Store();
794         if (store != NULL) {
795           igvn->replace_in_uses(store, store->in(MemNode::Memory));
796         }
797       }
798     }
799   }
800   igvn->replace_node(res, dom);
801 }

803 Node* InlineTypeNode::Ideal(PhaseGVN* phase, bool can_reshape) {
804   Node* oop = get_oop();
805   if (is_default(phase) && (!oop->is_Con() || phase->type(oop)->is_zero_type())) {
806     // Use the pre-allocated oop for default inline types
807     set_oop(default_oop(*phase, inline_klass()));
808     assert(is_allocated(phase), "should now be allocated");
809     return this;
810   } else if (oop->isa_InlineTypePtr()) {
811     // Can happen with late inlining
812     InlineTypePtrNode* vtptr = oop->as_InlineTypePtr();
813     set_oop(vtptr->get_oop());
814     for (uint i = Oop+1; i < vtptr->req(); ++i) {
815       set_req(i, vtptr->in(i));
816     }
817     return this;
818   }
819 
820   if (!is_allocated(phase)) {
821     // Save base oop if fields are loaded from memory and the inline
822     // type is not buffered (in this case we should not use the oop).
823     Node* base = is_loaded(phase);
824     if (base != NULL) {
825       set_oop(base);
826       assert(is_allocated(phase), "should now be allocated");
827       return this;
828     }
829   }
830 
831   if (can_reshape) {
832     PhaseIterGVN* igvn = phase->is_IterGVN();
833 
834     if (is_allocated(phase)) {
835       // Search for and remove re-allocations of this inline type. Ignore scalar replaceable ones,
836       // they will be removed anyway and changing the memory chain will confuse other optimizations.
837       // This can happen with late inlining when we first allocate an inline type argument
838       // but later decide to inline the call after the callee code also triggered allocation.
839       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
840         AllocateNode* alloc = fast_out(i)->isa_Allocate();
841         if (alloc != NULL && alloc->in(AllocateNode::InlineTypeNode) == this && !alloc->_is_scalar_replaceable) {
842           // Found a re-allocation
843           Node* res = alloc->result_cast();
844           if (res != NULL && res->is_CheckCastPP()) {
845             // Replace allocation by oop and unlink AllocateNode
846             replace_allocation(igvn, res, get_oop());
847             igvn->replace_input_of(alloc, AllocateNode::InlineTypeNode, igvn->C->top());
848             --i; --imax;
849           }
850         }
851       }
852     }
853   }
854   return NULL;
855 }

857 // Search for multiple allocations of this inline type and try to replace them by dominating allocations.
858 // Then unlink the inline type node and remove it.
859 void InlineTypeNode::remove_redundant_allocations(PhaseIterGVN* igvn, PhaseIdealLoop* phase) {
860   // Search for allocations of this inline type. Ignore scalar replaceable ones, they
861   // will be removed anyway and changing the memory chain will confuse other optimizations.
862   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
863     AllocateNode* alloc = fast_out(i)->isa_Allocate();
864     if (alloc != NULL && alloc->in(AllocateNode::InlineTypeNode) == this && !alloc->_is_scalar_replaceable) {
865       Node* res = alloc->result_cast();
866       if (res == NULL || !res->is_CheckCastPP()) {
867         break; // No unique CheckCastPP
868       }
869       assert(!is_default(igvn) && !is_allocated(igvn), "re-allocation should be removed by Ideal transformation");
870       // Search for a dominating allocation of the same inline type
871       Node* res_dom = res;
872       for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
873         AllocateNode* alloc_other = fast_out(j)->isa_Allocate();
874         if (alloc_other != NULL && alloc_other->in(AllocateNode::InlineTypeNode) == this && !alloc_other->_is_scalar_replaceable) {
875           Node* res_other = alloc_other->result_cast();
876           if (res_other != NULL && res_other->is_CheckCastPP() && res_other != res_dom &&
877               phase->is_dominator(res_other->in(0), res_dom->in(0))) {
878             res_dom = res_other;
879           }
880         }
881       }
882       if (res_dom != res) {
883         // Replace allocation by dominating one.
884         replace_allocation(igvn, res, res_dom);
885         // The result of the dominated allocation is now unused and will be removed
886         // later in PhaseMacroExpand::eliminate_allocate_node to not confuse loop opts.
887         igvn->record_for_igvn(alloc);
888       }
889     }
890   }
891 
892   // Process users
893   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
894     Node* out = fast_out(i);
895     if (out->is_InlineType()) {
896       // Unlink and recursively process inline type users
897       igvn->hash_delete(out);
898       int nb = out->replace_edge(this, igvn->C->top());
899       out->as_InlineType()->remove_redundant_allocations(igvn, phase);
900       --i; imax -= nb;
901     } else if (out->isa_Allocate() != NULL) {
902       // Unlink AllocateNode
903       assert(out->in(AllocateNode::InlineTypeNode) == this, "should be linked");
904       igvn->replace_input_of(out, AllocateNode::InlineTypeNode, igvn->C->top());
905       --i; --imax;
906     } else {
907 #ifdef ASSERT
908       // The inline type should not have any other users at this time
909       out->dump();
910       assert(false, "unexpected user of inline type");
911 #endif
912     }
913   }
914   igvn->remove_dead_node(this);
915 }
