226     public static String[] concat(String prefix[], String... extra) {
227         ArrayList<String> list = new ArrayList<String>();
228         if (prefix != null) {
229             for (String s : prefix) {
230                 list.add(s);
231             }
232         }
233         if (extra != null) {
234             for (String s : extra) {
235                 list.add(s);
236             }
237         }
238 
239         return list.toArray(new String[list.size()]);
240     }

326     public static void main(String[] args) throws Throwable {
327         if (args.length != 1) {
328             throw new RuntimeException("Usage: @run main/othervm/timeout=120 -Xbootclasspath/a:." +
329                                        " -XX:+IgnoreUnrecognizedVMOptions -XX:+UnlockDiagnosticVMOptions" +
330                                        " -XX:+UnlockExperimentalVMOptions -XX:+WhiteBoxAPI" +
331                                        " compiler.valhalla.inlinetypes.InlineTypeTest <YourTestMainClass>");
332         }
333         String testMainClassName = args[0];
334         Class testMainClass = Class.forName(testMainClassName);
335         InlineTypeTest test = (InlineTypeTest)testMainClass.newInstance();
336         List<String> scenarios = null;
337         if (!SCENARIOS.isEmpty()) {
338            scenarios = Arrays.asList(SCENARIOS.split(","));
339         }
340         for (int i=0; i<test.getNumScenarios(); i++) {
341             String reason;
342             if ((reason = test.isScenarioIgnored(i)) != null) {
343                 System.out.println("Scenario #" + i + " is ignored: " + reason);
344             } else if (scenarios != null && !scenarios.contains(Integer.toString(i))) {
345                 System.out.println("Scenario #" + i + " is skipped due to -Dscenarios=" + SCENARIOS);
346             } else {
347                 System.out.println("Scenario #" + i + " -------- ");
348                 String[] cmds = InputArguments.getVmInputArgs();
349                 if (RUN_SCENARIOS_WITH_XCOMP) {
350                     cmds = concat(cmds, "-Xcomp");
351                 }
352                 cmds = concat(cmds, test.getVMParameters(i));
353                 cmds = concat(cmds, test.getExtraVMParameters(i));
354                 cmds = concat(cmds, testMainClassName);
355 
356                 OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
357                 String output = oa.getOutput();
358                 oa.shouldHaveExitValue(0);
359                 System.out.println(output);
360             }
361         }
362     }

364     // To exclude test cases, use -DExclude=<case1>,<case2>,...
365     // Each case can be just the method name, or can be <class>.<method>. The latter form is useful
366     // when you are running several tests at the same time.
367     //
368     // jtreg -DExclude=test12 TestArrays.java
369     // jtreg -DExclude=test34 TestLWorld.java
370     // -- or --
371     // jtreg -DExclude=TestArrays.test12,TestLWorld.test34 TestArrays.java TestLWorld.java
372     //
373     private List<String> buildExcludeList() {
374         List<String> exclude = null;
375         String classPrefix = getClass().getSimpleName() + ".";
376         if (!EXCLUDELIST.isEmpty()) {
377             exclude = new ArrayList(Arrays.asList(EXCLUDELIST.split(",")));
378             for (int i = exclude.size() - 1; i >= 0; i--) {
379                 String ex = exclude.get(i);
380                 if (ex.indexOf(".") > 0) {
381                     if (ex.startsWith(classPrefix)) {
382                         ex = ex.substring(classPrefix.length());
383                         exclude.set(i, ex);
384                     } else {
385                         exclude.remove(i);
386                     }
387                 }
388             }
389         }
390         return exclude;
391     }

393     protected InlineTypeTest() {
394         List<String> list = null;
395         if (!TESTLIST.isEmpty()) {
396            list = Arrays.asList(TESTLIST.split(","));
397         }
398         List<String> exclude = buildExcludeList();
399 
400         // Gather all test methods and put them in Hashtable
401         for (Method m : getClass().getDeclaredMethods()) {
402             Test[] annos = m.getAnnotationsByType(Test.class);
403             if (annos.length != 0 &&
404                 ((list == null || list.contains(m.getName())) && (exclude == null || !exclude.contains(m.getName())))) {
405                 tests.put(getClass().getSimpleName() + "::" + m.getName(), m);
406             }
407         }
408     }

410     protected void run(String[] args, Class<?>... classes) throws Throwable {
411         if (args.length == 0) {
412             // Spawn a new VM instance
413             execute_vm();
414         } else {
415             // Execute tests in the VM spawned by the above code.
416             Asserts.assertTrue(args.length == 1 && args[0].equals("run"), "must be");
417             run(classes);
418         }
419     }

421     private void execute_vm() throws Throwable {
422         Asserts.assertFalse(tests.isEmpty(), "no tests to execute");
423         String[] vmInputArgs = InputArguments.getVmInputArgs();
424         for (String arg : vmInputArgs) {
425             if (arg.startsWith("-XX:CompileThreshold")) {
426                 // Disable IR verification if non-default CompileThreshold is set
427                 VERIFY_IR = false;
428             }
429         }
430         // Each VM is launched with flags in this order, so the later ones can override the earlier one:
431         //     VERIFY_IR/VERIFY_VM flags specified below
432         //     vmInputArgs, which consists of:
433         //        @run options
434         //        getVMParameters()
435         //        getExtraVMParameters()
436         //     defaultFlags
437         //     compileCommandFlags
438         String cmds[] = null;
439         if (VERIFY_IR) {
440             // Add print flags for IR verification
441             cmds = concat(cmds, printFlags);
442             // Always trap for exception throwing to not confuse IR verification
443             cmds = concat(cmds, "-XX:-OmitStackTraceInFastThrow");
444         }
445         if (VERIFY_VM) {
446             cmds = concat(cmds, verifyFlags);
447         }
448         cmds = concat(cmds, vmInputArgs);
449         cmds = concat(cmds, defaultFlags);
450         if (COMPILE_COMMANDS) {
451           cmds = concat(cmds, compileCommandFlags);
452         }
453 
454         // Run tests in own process and verify output
455         cmds = concat(cmds, getClass().getName(), "run");
456         OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
457         // If ideal graph printing is enabled/supported, verify output
458         String output = oa.getOutput();
459         oa.shouldHaveExitValue(0);
460         if (VERIFY_IR) {
461             if (output.contains("PrintIdeal enabled")) {
462                 parseOutput(output);
463             } else {
464                 System.out.println(output);
465                 System.out.println("WARNING: IR verification failed! Running with -Xint, -Xcomp or release build?");
466             }
467         }
468     }

470     static final class TestAnnotation {
471         private final int flag;
472         private final BooleanSupplier predicate;
473 
474         private static final TestAnnotation testAnnotations[] = {
475             new TestAnnotation(InlineTypePassFieldsAsArgsOn, () -> InlineTypePassFieldsAsArgs),
476             new TestAnnotation(InlineTypePassFieldsAsArgsOff, () -> !InlineTypePassFieldsAsArgs),
477             new TestAnnotation(InlineTypeArrayFlattenOn, () -> InlineTypeArrayFlatten),
478             new TestAnnotation(InlineTypeArrayFlattenOff, () -> !InlineTypeArrayFlatten),
479             new TestAnnotation(InlineTypeReturnedAsFieldsOn, () -> InlineTypeReturnedAsFields),
480             new TestAnnotation(InlineTypeReturnedAsFieldsOff, () -> !InlineTypeReturnedAsFields),
481             new TestAnnotation(AlwaysIncrementalInlineOn, () -> AlwaysIncrementalInline),
482             new TestAnnotation(AlwaysIncrementalInlineOff, () -> !AlwaysIncrementalInline),
483             new TestAnnotation(G1GCOn, () -> G1GC),
484             new TestAnnotation(G1GCOff, () -> !G1GC),
485             new TestAnnotation(ZGCOn, () -> ZGC),
486             new TestAnnotation(ZGCOff, () -> !ZGC),
487             new TestAnnotation(ArrayLoadStoreProfileOn, () -> UseArrayLoadStoreProfile),
488             new TestAnnotation(ArrayLoadStoreProfileOff, () -> !UseArrayLoadStoreProfile),
489             new TestAnnotation(TypeProfileOn, () -> TypeProfileLevel == 222),
490             new TestAnnotation(TypeProfileOff, () -> TypeProfileLevel == 0),
491         };
492 
493         private TestAnnotation(int flag, BooleanSupplier predicate) {
494             this.flag = flag;
495             this.predicate = predicate;
496         }
497 
498         private boolean match(Test a) {
499             return (a.valid() & flag) != 0 && predicate.getAsBoolean();
500         }
501 
502         static boolean find(Test a) {
503             Stream<TestAnnotation> s = Arrays.stream(testAnnotations).filter(t -> t.match(a));
504             long c = s.count();
505             if (c > 1) {
506                 throw new RuntimeException("At most one Test annotation should match");
507             }
508             return c > 0;
509         }
510     }

512     private void parseOutput(String output) throws Exception {
513         Pattern comp_re = Pattern.compile("\\n\\s+\\d+\\s+\\d+\\s+(%| )(s| )(!| )b(n| )\\s+\\d?\\s+\\S+\\.(?<name>[^.]+::\\S+)\\s+(?<osr>@ \\d+\\s+)?[(]\\d+ bytes[)]");
514         Matcher m = comp_re.matcher(output);
515         Map<String,String> compilations = new LinkedHashMap<>();
516         int prev = 0;
517         String methodName = null;
518         while (m.find()) {
519             if (prev == 0) {
520                 // Print header
521                 System.out.print(output.substring(0, m.start()+1));
522             } else if (methodName != null) {
523                 compilations.put(methodName, output.substring(prev, m.start()+1));
524             }
525             if (m.group("osr") != null) {
526                 methodName = null;
527             } else {
528                 methodName = m.group("name");
529             }
530             prev = m.end();
531         }
532         if (prev == 0) {
533             // Print header
534             System.out.print(output);
535         } else if (methodName != null) {
536             compilations.put(methodName, output.substring(prev));
537         }
538         // Iterate over compilation output
539         for (String testName : compilations.keySet()) {
540             Method test = tests.get(testName);
541             if (test == null) {
542                 // Skip helper methods
543                 continue;
544             }
545             String graph = compilations.get(testName);
546             if (PRINT_GRAPH) {
547                 System.out.println("\nGraph for " + testName + "\n" + graph);
548             }
549             // Parse graph using regular expressions to determine if it contains forbidden nodes
550             Test[] annos = test.getAnnotationsByType(Test.class);
551             Test anno = Arrays.stream(annos).filter(TestAnnotation::find).findFirst().orElse(null);
552             if (anno == null) {
553                 Object[] res = Arrays.stream(annos).filter(a -> a.valid() == 0).toArray();
554                 if (res.length != 1) {
555                     throw new RuntimeException("Only one Test annotation should match");
556                 }
557                 anno = (Test)res[0];
558             }
559             String regexFail = anno.failOn();
560             if (!regexFail.isEmpty()) {
561                 Pattern pattern = Pattern.compile(regexFail.substring(0, regexFail.length()-1));
562                 Matcher matcher = pattern.matcher(graph);
563                 boolean found = matcher.find();
564                 Asserts.assertFalse(found, "Graph for '" + testName + "' contains forbidden node:\n" + (found ? matcher.group() : ""));
565             }
566             String[] regexMatch = anno.match();
567             int[] matchCount = anno.matchCount();
568             for (int i = 0; i < regexMatch.length; ++i) {
569                 Pattern pattern = Pattern.compile(regexMatch[i].substring(0, regexMatch[i].length()-1));
570                 Matcher matcher = pattern.matcher(graph);
571                 int count = 0;
572                 String nodes = "";
573                 while (matcher.find()) {
574                     count++;
575                     nodes += matcher.group() + "\n";
576                 }
577                 if (matchCount[i] < 0) {
578                     Asserts.assertLTE(Math.abs(matchCount[i]), count, "Graph for '" + testName + "' contains different number of match nodes (expected <= " + matchCount[i] + " but got " + count + "):\n" + nodes);
579                 } else {
580                     Asserts.assertEQ(matchCount[i], count, "Graph for '" + testName + "' contains different number of match nodes (expected " + matchCount[i] + " but got " + count + "):\n" + nodes);
581                 }
582             }
583             tests.remove(testName);
584             System.out.println(testName + " passed");
585         }
586         // Check if all tests were compiled
587         if (tests.size() != 0) {
588             for (String name : tests.keySet()) {
589                 System.out.println("Test '" + name + "' not compiled!");
590             }
591             throw new RuntimeException("Not all tests were compiled");
592         }
593     }

595     private void setup(Class<?> clazz) {
596         if (XCOMP) {
597             // Don't control compilation if -Xcomp is enabled
598             return;
599         }
600         if (DUMP_REPLAY) {
601             // Generate replay compilation files
602             String directive = "[{ match: \"*.*\", DumpReplay: true }]";
603             if (WHITE_BOX.addCompilerDirective(directive) != 1) {
604                 throw new RuntimeException("Failed to add compiler directive");
605             }
606         }
607 
608         Method[] methods = clazz.getDeclaredMethods();
609         for (Method m : methods) {
610             if (m.isAnnotationPresent(Test.class)) {
611                 // Don't inline tests
612                 WHITE_BOX.testSetDontInlineMethod(m, true);
613             }
614             if (m.isAnnotationPresent(DontCompile.class)) {
615                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, true);
616                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, false);
617                 WHITE_BOX.testSetDontInlineMethod(m, true);
618             }
619             if (m.isAnnotationPresent(ForceInline.class)) {
620                 Asserts.assertFalse(m.isAnnotationPresent(DontInline.class), "Method " + m.getName() + " has contradicting DontInline annotation");
621                 WHITE_BOX.testSetForceInlineMethod(m, true);
622             }
623             if (m.isAnnotationPresent(DontInline.class)) {
624                 Asserts.assertFalse(m.isAnnotationPresent(ForceInline.class), "Method " + m.getName() + " has contradicting ForceInline annotation");
625                 WHITE_BOX.testSetDontInlineMethod(m, true);
626             }
627             if (STRESS_CC) {
628                 // Exclude some methods from compilation with C2 to stress test the calling convention
629                 if (Utils.getRandomInstance().nextBoolean()) {
630                     System.out.println("Excluding from C2 compilation: " + m);
631                     WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_FULL_OPTIMIZATION, false);
632                 }
633             }
634         }
635         // Only force compilation now because above annotations affect inlining
636         for (Method m : methods) {
637             if (m.isAnnotationPresent(ForceCompile.class)) {
638                 Asserts.assertFalse(m.isAnnotationPresent(DontCompile.class), "Method " + m.getName() + " has contradicting DontCompile annotation");
639                 int compLevel = getCompLevel(m.getAnnotation(ForceCompile.class));
640                 enqueueMethodForCompilation(m, compLevel);
641             }
642         }
643         // Compile class initializers
644         int compLevel = getCompLevel(null);
645         WHITE_BOX.enqueueInitializerForCompilation(clazz, compLevel);
646     }

648     private void run(Class<?>... classes) throws Exception {
649         if (USE_COMPILER && PRINT_IDEAL && !XCOMP && !STRESS_CC) {
650             System.out.println("PrintIdeal enabled");
651         }
652         System.out.format("rI = %d, rL = %d\n", rI, rL);
653 
654         setup(getClass());
655         for (Class<?> clazz : classes) {
656             setup(clazz);
657         }
658 
659         // Execute tests
660         TreeMap<Long, String> durations = (PRINT_TIMES || VERBOSE) ? new TreeMap<Long, String>() : null;
661         for (Method test : tests.values()) {
662             if (VERBOSE) {
663                 System.out.println("Starting " + test.getName());
664             }
665             TempSkipForC1 c1skip = test.getAnnotation(TempSkipForC1.class);
666             if (TEST_C1 && c1skip != null) {
667                 System.out.println("Skipped " + test.getName() + " for C1 testing: " + c1skip.reason());
668                 continue;
669             }
670             long startTime = System.nanoTime();
671             Method verifier = getClass().getMethod(test.getName() + "_verifier", boolean.class);
672             // Warmup using verifier method
673             Warmup anno = test.getAnnotation(Warmup.class);
674             int warmup = anno == null ? WARMUP : anno.value();
675             for (int i = 0; i < warmup; ++i) {
676                 verifier.invoke(this, true);
677             }
678             boolean osrOnly = (test.getAnnotation(OSRCompileOnly.class) != null);

679 
680             // C1 generates a lot of code when VerifyOops is enabled and may run out of space (for a small
681             // number of test cases).
682             boolean maybeCodeBufferOverflow = (TEST_C1 && VerifyOops);
683 
684             if (osrOnly) {
685                 long started = System.currentTimeMillis();
686                 boolean stateCleared = false;
687                 for (;;) {
688                     long elapsed = System.currentTimeMillis() - started;
689                     if (maybeCodeBufferOverflow && elapsed > 5000 && !WHITE_BOX.isMethodCompiled(test, false)) {

690                         System.out.println("Temporarily disabling VerifyOops");
691                         try {
692                             WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
693                             if (!stateCleared) {
694                                 WHITE_BOX.clearMethodState(test);
695                                 stateCleared = true;
696                             }
697                             verifier.invoke(this, false);
698                         } finally {
699                             WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
700                             System.out.println("Re-enabled VerifyOops");
701                         }
702                     } else {
703                         verifier.invoke(this, false);
704                     }
705 
706                     boolean b = WHITE_BOX.isMethodCompiled(test, false);
707                     if (VERBOSE) {
708                         System.out.println("Is " + test.getName() + " compiled? " + b);
709                     }
710                     if (b || XCOMP || STRESS_CC || !USE_COMPILER) {
711                         // Don't control compilation if -Xcomp is enabled, or if compiler is disabled
712                         break;
713                     }
714                     Asserts.assertTrue(OSR_TEST_TIMEOUT < 0 || elapsed < OSR_TEST_TIMEOUT, test + " not compiled after " + OSR_TEST_TIMEOUT + " ms");
715                 }
716                 Asserts.assertTrue(XCOMP || STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");
717             } else {
718                 int compLevel = getCompLevel(test.getAnnotation(Test.class));
719                 // Trigger compilation
720                 enqueueMethodForCompilation(test, compLevel);
721                 if (maybeCodeBufferOverflow && !WHITE_BOX.isMethodCompiled(test, false)) {
722                     // Let's disable VerifyOops temporarily and retry.
723                     WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
724                     WHITE_BOX.clearMethodState(test);
725                     enqueueMethodForCompilation(test, compLevel);
726                     WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
727                 }
728                 Asserts.assertTrue(STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");




729                 // Check result
730                 verifier.invoke(this, false);
731             }
732             if (PRINT_TIMES || VERBOSE) {
733                 long endTime = System.nanoTime();
734                 long duration = (endTime - startTime);
735                 durations.put(duration, test.getName());
736                 if (VERBOSE) {
737                     System.out.println("Done " + test.getName() + ": " + duration + " ns = " + (duration / 1000000) + " ms");
738                 }
739             }
740             if (GC_AFTER) {
741                 System.out.println("doing GC");
742                 System.gc();
743             }
744         }
745 
746         // Print execution times
747         if (PRINT_TIMES) {
748           System.out.println("\n\nTest execution times:");
749           for (Map.Entry<Long, String> entry : durations.entrySet()) {
750               System.out.format("%-10s%15d ns\n", entry.getValue() + ":", entry.getKey());
751           }
752         }
753     }

755     // Get the appropriate compilation level for a method, according to the
756     // given annotation, as well as the current test scenario and VM options.
757     //
758     private int getCompLevel(Object annotation) {
759         int compLevel;
760         if (annotation == null) {
761             compLevel = COMP_LEVEL_ANY;
762         } else if (annotation instanceof Test) {
763             compLevel = ((Test)annotation).compLevel();
764         } else {
765             compLevel = ((ForceCompile)annotation).compLevel();
766         }
767 
768         return restrictCompLevel(compLevel);
769     }
770 
771     // Get the appropriate level as permitted by the test scenario and VM options.
772     private static int restrictCompLevel(int compLevel) {
773         if (compLevel == COMP_LEVEL_ANY) {
774             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
775         }
776         if (FLIP_C1_C2) {
777             // Effectively treat all (compLevel = C1) as (compLevel = C2), and
778             //                       (compLevel = C2) as (compLevel = C1).
779             if (compLevel == COMP_LEVEL_SIMPLE) {
780                 compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
781             } else if (compLevel == COMP_LEVEL_FULL_OPTIMIZATION) {
782                 compLevel = COMP_LEVEL_SIMPLE;
783             }
784         }
785         if (!TEST_C1 && compLevel < COMP_LEVEL_FULL_OPTIMIZATION) {
786             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
787         }
788         if (compLevel > (int)TieredStopAtLevel) {
789             compLevel = (int)TieredStopAtLevel;
790         }
791         return compLevel;
792     }

794     public static void enqueueMethodForCompilation(Method m, int level) {
795         level = restrictCompLevel(level);
796         if (VERBOSE) {
797             System.out.println("enqueueMethodForCompilation " + m + ", level = " + level);
798         }
799         WHITE_BOX.enqueueMethodForCompilation(m, level);
800     }

802     // Unlike C2, C1 intrinsics never deoptimize System.arraycopy. Instead, we fall back to
803     // a normal method invocation when encountering flattened arrays.
804     static boolean isCompiledByC2(Method m) {
805         return USE_COMPILER && !XCOMP && WHITE_BOX.isMethodCompiled(m, false) &&
806             WHITE_BOX.getMethodCompilationLevel(m, false) >= COMP_LEVEL_FULL_OPTIMIZATION;
