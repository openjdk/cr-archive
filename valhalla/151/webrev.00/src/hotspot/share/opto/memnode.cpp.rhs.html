 108 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 109   st->print(" @");
 110   if (adr_type == NULL) {
 111     st->print("NULL");
 112   } else {
 113     adr_type->dump_on(st);
 114     Compile* C = Compile::current();
 115     Compile::AliasType* atp = NULL;
 116     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 117     if (atp == NULL)
 118       st->print(", idx=?\?;");
 119     else if (atp->index() == Compile::AliasIdxBot)
 120       st->print(", idx=Bot;");
 121     else if (atp->index() == Compile::AliasIdxTop)
 122       st->print(", idx=Top;");
 123     else if (atp->index() == Compile::AliasIdxRaw)
 124       st->print(", idx=Raw;");
 125     else {
 126       ciField* field = atp->field();
 127       if (field) {
 128         st->print(", name=");
 129         field->print_name_on(st);
 130       }
 131       st->print(", idx=%d;", atp->index());
 132     }
 133   }
 134 }

 140 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 141   assert((t_oop != NULL), "sanity");
 142   bool is_instance = t_oop->is_known_instance_field();
 143   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 144                              (load != NULL) && load->is_Load() &&
 145                              (phase->is_IterGVN() != NULL);
 146   if (!(is_instance || is_boxed_value_load))
 147     return mchain;  // don't try to optimize non-instance types
 148   uint instance_id = t_oop->instance_id();
 149   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 150   Node *prev = NULL;
 151   Node *result = mchain;
 152   while (prev != result) {
 153     prev = result;
 154     if (result == start_mem)
 155       break;  // hit one of our sentinels
 156     // skip over a call which does not affect this memory slice
 157     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 158       Node *proj_in = result->in(0);
 159       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 160         break;  // hit one of our sentinels
 161       } else if (proj_in->is_Call()) {
 162         // ArrayCopyNodes processed here as well
 163         CallNode *call = proj_in->as_Call();
 164         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 165           result = call->in(TypeFunc::Memory);
 166         }
 167       } else if (proj_in->is_Initialize()) {
 168         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 169         // Stop if this is the initialization for the object instance which
 170         // contains this memory slice, otherwise skip over it.
 171         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 172           break;
 173         }
 174         if (is_instance) {
 175           result = proj_in->in(TypeFunc::Memory);
 176         } else if (is_boxed_value_load) {
 177           Node* klass = alloc->in(AllocateNode::KlassNode);
 178           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 179           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 180             result = proj_in->in(TypeFunc::Memory); // not related allocation
 181           }
 182         }
 183       } else if (proj_in->is_MemBar()) {
 184         ArrayCopyNode* ac = NULL;
 185         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 186           break;
 187         }
 188         result = proj_in->in(TypeFunc::Memory);
 189       } else {
 190         assert(false, "unexpected projection");
 191       }
 192     } else if (result->is_ClearArray()) {
 193       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 194         // Can not bypass initialization of the instance
 195         // we are looking for.
 196         break;
 197       }
 198       // Otherwise skip it (the call updated 'result' value).
 199     } else if (result->is_MergeMem()) {
 200       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 201     }
 202   }
 203   return result;
 204 }

 206 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 207   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 208   if (t_oop == NULL)
 209     return mchain;  // don't try to optimize non-oop types
 210   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 211   bool is_instance = t_oop->is_known_instance_field();
 212   PhaseIterGVN *igvn = phase->is_IterGVN();
 213   if (is_instance && igvn != NULL && result->is_Phi()) {
 214     PhiNode *mphi = result->as_Phi();
 215     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 216     const TypePtr *t = mphi->adr_type();
 217     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 218         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 219          t->is_oopptr()->cast_to_exactness(true)
 220            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 221             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 222       // clone the Phi with our address type
 223       result = mphi->split_out_instance(t_adr, igvn);
 224     } else {





 225       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 226     }
 227   }
 228   return result;
 229 }

 231 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 232   uint alias_idx = phase->C->get_alias_index(tp);
 233   Node *mem = mmem;
 234 #ifdef ASSERT
 235   {
 236     // Check that current type is consistent with the alias index used during graph construction
 237     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 238     bool consistent =  adr_check == NULL || adr_check->empty() ||
 239                        phase->C->must_alias(adr_check, alias_idx );
 240     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 241     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 242         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 243         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 244         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 245           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 246           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 247       // don't assert if it is dead code.
 248       consistent = true;
 249     }
 250     if( !consistent ) {
 251       st->print("alias_idx==%d, adr_check==", alias_idx);
 252       if( adr_check == NULL ) {
 253         st->print("NULL");
 254       } else {
 255         adr_check->dump();
 256       }
 257       st->cr();
 258       print_alias_types();
 259       assert(consistent, "adr_check must match alias idx");
 260     }
 261   }
 262 #endif
 263   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 264   // means an array I have not precisely typed yet.  Do not do any
 265   // alias stuff with it any time soon.
 266   const TypeOopPtr *toop = tp->isa_oopptr();
 267   if( tp->base() != Type::AnyPtr &&
 268       !(toop &&
 269         toop->klass() != NULL &&
 270         toop->klass()->is_java_lang_Object() &&
 271         toop->offset() == Type::OffsetBot) ) {
 272     // compress paths and change unreachable cycles to TOP
 273     // If not, we can update the input infinitely along a MergeMem cycle
 274     // Equivalent code in PhiNode::Ideal
 275     Node* m  = phase->transform(mmem);
 276     // If transformed to a MergeMem, get the desired slice
 277     // Otherwise the returned node represents memory for every slice
 278     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 279     // Update input if it is progress over what we have now
 280   }
 281   return mem;
 282 }

 284 //--------------------------Ideal_common---------------------------------------
 285 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 286 // Unhook non-raw memories from complete (macro-expanded) initializations.
 287 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 288   // If our control input is a dead region, kill all below the region
 289   Node *ctl = in(MemNode::Control);
 290   if (ctl && remove_dead_region(phase, can_reshape))
 291     return this;
 292   ctl = in(MemNode::Control);
 293   // Don't bother trying to transform a dead node
 294   if (ctl && ctl->is_top())  return NodeSentinel;
 295 
 296   PhaseIterGVN *igvn = phase->is_IterGVN();
 297   // Wait if control on the worklist.
 298   if (ctl && can_reshape && igvn != NULL) {
 299     Node* bol = NULL;
 300     Node* cmp = NULL;
 301     if (ctl->in(0)->is_If()) {
 302       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 303       bol = ctl->in(0)->in(1);
 304       if (bol->is_Bool())
 305         cmp = ctl->in(0)->in(1)->in(1);
 306     }
 307     if (igvn->_worklist.member(ctl) ||
 308         (bol != NULL && igvn->_worklist.member(bol)) ||
 309         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 310       // This control path may be dead.
 311       // Delay this memory node transformation until the control is processed.
 312       phase->is_IterGVN()->_worklist.push(this);
 313       return NodeSentinel; // caller will return NULL
 314     }
 315   }
 316   // Ignore if memory is dead, or self-loop
 317   Node *mem = in(MemNode::Memory);
 318   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 319   assert(mem != this, "dead loop in MemNode::Ideal");
 320 
 321   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 322     // This memory slice may be dead.
 323     // Delay this mem node transformation until the memory is processed.
 324     phase->is_IterGVN()->_worklist.push(this);
 325     return NodeSentinel; // caller will return NULL
 326   }
 327 
 328   Node *address = in(MemNode::Address);
 329   const Type *t_adr = phase->type(address);
 330   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 331 
 332   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 333     // Unsafe off-heap access with zero address. Remove access and other control users
 334     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 335     assert(ctl != NULL, "unsafe accesses should be control dependent");
 336     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 337       Node* u = ctl->fast_out(i);
 338       if (u != ctl) {
 339         igvn->rehash_node_delayed(u);
 340         int nb = u->replace_edge(ctl, phase->C->top());
 341         --i, imax -= nb;
 342       }
 343     }
 344     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 345     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 346     phase->C->root()->add_req(halt);
 347     return this;
 348   }
 349 
 350   if (can_reshape && igvn != NULL &&
 351       (igvn->_worklist.member(address) ||
 352        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 353     // The address's base and type may change when the address is processed.
 354     // Delay this mem node transformation until the address is processed.
 355     phase->is_IterGVN()->_worklist.push(this);
 356     return NodeSentinel; // caller will return NULL
 357   }
 358 
 359   // Do NOT remove or optimize the next lines: ensure a new alias index
 360   // is allocated for an oop pointer type before Escape Analysis.
 361   // Note: C++ will not remove it since the call has side effect.
 362   if (t_adr->isa_oopptr()) {
 363     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 364   }
 365 
 366   Node* base = NULL;
 367   if (address->is_AddP()) {
 368     base = address->in(AddPNode::Base);
 369   }
 370   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 371       !t_adr->isa_rawptr()) {
 372     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 373     // Skip this node optimization if its address has TOP base.
 374     return NodeSentinel; // caller will return NULL
 375   }
 376 
 377   // Avoid independent memory operations
 378   Node* old_mem = mem;
 379 
 380   // The code which unhooks non-raw memories from complete (macro-expanded)
 381   // initializations was removed. After macro-expansion all stores catched
 382   // by Initialize node became raw stores and there is no information
 383   // which memory slices they modify. So it is unsafe to move any memory
 384   // operation above these stores. Also in most cases hooked non-raw memories
 385   // were already unhooked by using information from detect_ptr_independence()
 386   // and find_previous_store().
 387 
 388   if (mem->is_MergeMem()) {
 389     MergeMemNode* mmem = mem->as_MergeMem();
 390     const TypePtr *tp = t_adr->is_ptr();
 391 
 392     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 393   }
 394 
 395   if (mem != old_mem) {
 396     set_req(MemNode::Memory, mem);
 397     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 398       igvn->_worklist.push(old_mem);
 399     }
 400     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 401     return this;
 402   }
 403 
 404   // let the subclass continue analyzing...
 405   return NULL;
 406 }

 408 // Helper function for proving some simple control dominations.
 409 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 410 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 411 // is not a constant (dominated by the method's StartNode).
 412 // Used by MemNode::find_previous_store to prove that the
 413 // control input of a memory operation predates (dominates)
 414 // an allocation it wants to look past.
 415 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 416   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 417     return false; // Conservative answer for dead code
 418 
 419   // Check 'dom'. Skip Proj and CatchProj nodes.
 420   dom = dom->find_exact_control(dom);
 421   if (dom == NULL || dom->is_top())
 422     return false; // Conservative answer for dead code
 423 
 424   if (dom == sub) {
 425     // For the case when, for example, 'sub' is Initialize and the original
 426     // 'dom' is Proj node of the 'sub'.
 427     return false;
 428   }
 429 
 430   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 431     return true;
 432 
 433   // 'dom' dominates 'sub' if its control edge and control edges
 434   // of all its inputs dominate or equal to sub's control edge.
 435 
 436   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 437   // Or Region for the check in LoadNode::Ideal();
 438   // 'sub' should have sub->in(0) != NULL.
 439   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 440          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 441 
 442   // Get control edge of 'sub'.
 443   Node* orig_sub = sub;
 444   sub = sub->find_exact_control(sub->in(0));
 445   if (sub == NULL || sub->is_top())
 446     return false; // Conservative answer for dead code
 447 
 448   assert(sub->is_CFG(), "expecting control");
 449 
 450   if (sub == dom)
 451     return true;
 452 
 453   if (sub->is_Start() || sub->is_Root())
 454     return false;
 455 
 456   {
 457     // Check all control edges of 'dom'.
 458 
 459     ResourceMark rm;
 460     Node_List nlist;
 461     Unique_Node_List dom_list;
 462 
 463     dom_list.push(dom);
 464     bool only_dominating_controls = false;
 465 
 466     for (uint next = 0; next < dom_list.size(); next++) {
 467       Node* n = dom_list.at(next);
 468       if (n == orig_sub)
 469         return false; // One of dom's inputs dominated by sub.
 470       if (!n->is_CFG() && n->pinned()) {
 471         // Check only own control edge for pinned non-control nodes.
 472         n = n->find_exact_control(n->in(0));
 473         if (n == NULL || n->is_top())
 474           return false; // Conservative answer for dead code
 475         assert(n->is_CFG(), "expecting control");
 476         dom_list.push(n);
 477       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 478         only_dominating_controls = true;
 479       } else if (n->is_CFG()) {
 480         if (n->dominates(sub, nlist))
 481           only_dominating_controls = true;
 482         else
 483           return false;
 484       } else {
 485         // First, own control edge.
 486         Node* m = n->find_exact_control(n->in(0));
 487         if (m != NULL) {
 488           if (m->is_top())
 489             return false; // Conservative answer for dead code
 490           dom_list.push(m);
 491         }
 492         // Now, the rest of edges.
 493         uint cnt = n->req();
 494         for (uint i = 1; i < cnt; i++) {
 495           m = n->find_exact_control(n->in(i));
 496           if (m == NULL || m->is_top())
 497             continue;
 498           dom_list.push(m);
 499         }
 500       }
 501     }
 502     return only_dominating_controls;
 503   }
 504 }

 506 //---------------------detect_ptr_independence---------------------------------
 507 // Used by MemNode::find_previous_store to prove that two base
 508 // pointers are never equal.
 509 // The pointers are accompanied by their associated allocations,
 510 // if any, which have been previously discovered by the caller.
 511 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 512                                       Node* p2, AllocateNode* a2,
 513                                       PhaseTransform* phase) {
 514   // Attempt to prove that these two pointers cannot be aliased.
 515   // They may both manifestly be allocations, and they should differ.
 516   // Or, if they are not both allocations, they can be distinct constants.
 517   // Otherwise, one is an allocation and the other a pre-existing value.
 518   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 519     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 520   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 521     return (a1 != a2);
 522   } else if (a1 != NULL) {                  // one allocation a1
 523     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 524     return all_controls_dominate(p2, a1);
 525   } else { //(a2 != NULL)                   // one allocation a2
 526     return all_controls_dominate(p1, a2);
 527   }
 528   return false;
 529 }

 532 // Find an arraycopy that must have set (can_see_stored_value=true) or
 533 // could have set (can_see_stored_value=false) the value for this load
 534 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 535   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 536                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 537     Node* mb = mem->in(0);
 538     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 539         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 540       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 541       if (ac->is_clonebasic()) {
 542         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 543         if (alloc != NULL && alloc == ld_alloc) {
 544           return ac;
 545         }
 546       }
 547     }
 548   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 549     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 550 
 551     if (ac->is_arraycopy_validated() ||
 552         ac->is_copyof_validated() ||
 553         ac->is_copyofrange_validated()) {
 554       Node* ld_addp = in(MemNode::Address);
 555       if (ld_addp->is_AddP()) {
 556         Node* ld_base = ld_addp->in(AddPNode::Address);
 557         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 558 
 559         Node* dest = ac->in(ArrayCopyNode::Dest);
 560 
 561         if (dest == ld_base) {
 562           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 563           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 564             return ac;
 565           }
 566           if (!can_see_stored_value) {
 567             mem = ac->in(TypeFunc::Memory);
 568           }
 569         }
 570       }
 571     }
 572   }
 573   return NULL;

 576 // The logic for reordering loads and stores uses four steps:
 577 // (a) Walk carefully past stores and initializations which we
 578 //     can prove are independent of this load.
 579 // (b) Observe that the next memory state makes an exact match
 580 //     with self (load or store), and locate the relevant store.
 581 // (c) Ensure that, if we were to wire self directly to the store,
 582 //     the optimizer would fold it up somehow.
 583 // (d) Do the rewiring, and return, depending on some other part of
 584 //     the optimizer to fold up the load.
 585 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 586 // specific to loads and stores, so they are handled by the callers.
 587 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 588 //
 589 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 590   Node*         ctrl   = in(MemNode::Control);
 591   Node*         adr    = in(MemNode::Address);
 592   intptr_t      offset = 0;
 593   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 594   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 595 
 596   if (offset == Type::OffsetBot)
 597     return NULL;            // cannot unalias unless there are precise offsets
 598 
 599   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 600   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 601 
 602   intptr_t size_in_bytes = memory_size();
 603 
 604   Node* mem = in(MemNode::Memory);   // start searching here...
 605 
 606   int cnt = 50;             // Cycle limiter
 607   for (;;) {                // While we can dance past unrelated stores...
 608     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 609 
 610     Node* prev = mem;
 611     if (mem->is_Store()) {
 612       Node* st_adr = mem->in(MemNode::Address);
 613       intptr_t st_offset = 0;
 614       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 615       if (st_base == NULL)
 616         break;              // inscrutable pointer
 617 
 618       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 619       // We need the bases to be the equal in order for the offset check to make sense.
 620       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 621         break;
 622       }
 623 
 624       if (st_offset != offset && st_offset != Type::OffsetBot) {
 625         const int MAX_STORE = BytesPerLong;
 626         if (st_offset >= offset + size_in_bytes ||
 627             st_offset <= offset - MAX_STORE ||
 628             st_offset <= offset - mem->as_Store()->memory_size()) {
 629           // Success:  The offsets are provably independent.
 630           // (You may ask, why not just test st_offset != offset and be done?
 631           // The answer is that stores of different sizes can co-exist
 632           // in the same sequence of RawMem effects.  We sometimes initialize
 633           // a whole 'tile' of array elements with a single jint or jlong.)
 634           mem = mem->in(MemNode::Memory);
 635           continue;           // (a) advance through independent store memory
 636         }
 637       }
 638       if (st_base != base &&
 639           detect_ptr_independence(base, alloc,
 640                                   st_base,
 641                                   AllocateNode::Ideal_allocation(st_base, phase),
 642                                   phase)) {
 643         // Success:  The bases are provably independent.
 644         mem = mem->in(MemNode::Memory);
 645         continue;           // (a) advance through independent store memory
 646       }
 647 
 648       // (b) At this point, if the bases or offsets do not agree, we lose,
 649       // since we have not managed to prove 'this' and 'mem' independent.
 650       if (st_base == base && st_offset == offset) {
 651         return mem;         // let caller handle steps (c), (d)
 652       }
 653 
 654     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 655       InitializeNode* st_init = mem->in(0)->as_Initialize();
 656       AllocateNode*  st_alloc = st_init->allocation();
 657       if (st_alloc == NULL)
 658         break;              // something degenerated
 659       bool known_identical = false;
 660       bool known_independent = false;
 661       if (alloc == st_alloc)
 662         known_identical = true;
 663       else if (alloc != NULL)
 664         known_independent = true;
 665       else if (all_controls_dominate(this, st_alloc))
 666         known_independent = true;
 667 
 668       if (known_independent) {
 669         // The bases are provably independent: Either they are
 670         // manifestly distinct allocations, or else the control
 671         // of this load dominates the store's allocation.
 672         int alias_idx = phase->C->get_alias_index(adr_type());
 673         if (alias_idx == Compile::AliasIdxRaw) {
 674           mem = st_alloc->in(TypeFunc::Memory);
 675         } else {
 676           mem = st_init->memory(alias_idx);
 677         }
 678         continue;           // (a) advance through independent store memory
 679       }
 680 
 681       // (b) at this point, if we are not looking at a store initializing
 682       // the same allocation we are loading from, we lose.
 683       if (known_identical) {
 684         // From caller, can_see_stored_value will consult find_captured_store.
 685         return mem;         // let caller handle steps (c), (d)
 686       }
 687 
 688     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 689       if (prev != mem) {
 690         // Found an arraycopy but it doesn't affect that load
 691         continue;
 692       }
 693       // Found an arraycopy that may affect that load
 694       return mem;
 695     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 696       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 697       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 698         // ArrayCopyNodes processed here as well.
 699         CallNode *call = mem->in(0)->as_Call();
 700         if (!call->may_modify(addr_t, phase)) {
 701           mem = call->in(TypeFunc::Memory);
 702           continue;         // (a) advance through independent call memory
 703         }
 704       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 705         ArrayCopyNode* ac = NULL;
 706         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 707           break;
 708         }
 709         mem = mem->in(0)->in(TypeFunc::Memory);
 710         continue;           // (a) advance through independent MemBar memory
 711       } else if (mem->is_ClearArray()) {
 712         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 713           // (the call updated 'mem' value)
 714           continue;         // (a) advance through independent allocation memory
 715         } else {
 716           // Can not bypass initialization of the instance
 717           // we are looking for.
 718           return mem;
 719         }
 720       } else if (mem->is_MergeMem()) {
 721         int alias_idx = phase->C->get_alias_index(adr_type());
 722         mem = mem->as_MergeMem()->memory_at(alias_idx);
 723         continue;           // (a) advance through independent MergeMem memory
 724       }
 725     }
 726 
 727     // Unless there is an explicit 'continue', we must bail out here,
 728     // because 'mem' is an inscrutable memory state (e.g., a call).
 729     break;
 730   }
 731 
 732   return NULL;              // bail out
 733 }

 735 //----------------------calculate_adr_type-------------------------------------
 736 // Helper function.  Notices when the given type of address hits top or bottom.
 737 // Also, asserts a cross-check of the type against the expected address type.
 738 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 739   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 740   #ifdef ASSERT
 741   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 742   #endif
 743   const TypePtr* tp = t->isa_ptr();
 744   if (tp == NULL) {
 745     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 746     return TypePtr::BOTTOM;           // touches lots of memory
 747   } else {
 748     #ifdef ASSERT
 749     // %%%% [phh] We don't check the alias index if cross_check is
 750     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 751     if (cross_check != NULL &&
 752         cross_check != TypePtr::BOTTOM &&
 753         cross_check != TypeRawPtr::BOTTOM) {
 754       // Recheck the alias index, to see if it has changed (due to a bug).
 755       Compile* C = Compile::current();
 756       assert(C->get_alias_index(cross_check) == C->get_alias_index(tp),

 759       // disregarding "null"-ness.
 760       // (We make an exception for TypeRawPtr::BOTTOM, which is a bit bucket.)

 763              "real address must not escape from expected memory type");
 764     }
 765     #endif
 766     return tp;
 767   }

 926 // Is the value loaded previously stored by an arraycopy? If so return
 927 // a load node that reads from the source array so we may be able to
 928 // optimize out the ArrayCopy node later.
 929 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 930   Node* ld_adr = in(MemNode::Address);
 931   intptr_t ld_off = 0;
 932   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 933   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 934   if (ac != NULL) {
 935     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 936 
 937     Node* mem = ac->in(TypeFunc::Memory);
 938     Node* ctl = ac->in(0);
 939     Node* src = ac->in(ArrayCopyNode::Src);
 940 
 941     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 942       return NULL;
 943     }
 944 
 945     LoadNode* ld = clone()->as_Load();
 946     Node* addp = in(MemNode::Address)->clone();
 947     if (ac->as_ArrayCopy()->is_clonebasic()) {
 948       assert(ld_alloc != NULL, "need an alloc");
 949       assert(addp->is_AddP(), "address must be addp");
 950       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 951       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 952       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 953       addp->set_req(AddPNode::Base, src);
 954       addp->set_req(AddPNode::Address, src);
 955     } else {
 956       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 957              ac->as_ArrayCopy()->is_copyof_validated() ||
 958              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 959       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 960       addp->set_req(AddPNode::Base, src);
 961       addp->set_req(AddPNode::Address, src);
 962 
 963       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 964       BasicType ary_elem = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 965       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 966       uint shift  = exact_log2(type2aelembytes(ary_elem));
 967       if (ary_t->klass()->is_flat_array_klass()) {
 968         ciFlatArrayKlass* vak = ary_t->klass()->as_flat_array_klass();
 969         shift = vak->log2_element_size();
 970       }
 971 
 972       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 973 #ifdef _LP64
 974       diff = phase->transform(new ConvI2LNode(diff));
 975 #endif
 976       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 977 
 978       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 979       addp->set_req(AddPNode::Offset, offset);
 980     }
 981     addp = phase->transform(addp);
 982 #ifdef ASSERT
 983     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 984     ld->_adr_type = adr_type;
 985 #endif
 986     ld->set_req(MemNode::Address, addp);
 987     ld->set_req(0, ctl);
 988     ld->set_req(MemNode::Memory, mem);
 989     // load depends on the tests that validate the arraycopy
 990     ld->_control_dependency = UnknownControl;
 991     return ld;
 992   }
 993   return NULL;
 994 }

 997 //---------------------------can_see_stored_value------------------------------
 998 // This routine exists to make sure this set of tests is done the same
 999 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1000 // will change the graph shape in a way which makes memory alive twice at the
1001 // same time (uses the Oracle model of aliasing), then some
1002 // LoadXNode::Identity will fold things back to the equivalence-class model
1003 // of aliasing.
1004 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1005   Node* ld_adr = in(MemNode::Address);
1006   intptr_t ld_off = 0;
1007   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1008   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1009   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1010   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1011   // This is more general than load from boxing objects.
1012   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1013     uint alias_idx = atp->index();
1014     bool final = !atp->is_rewritable();
1015     Node* result = NULL;
1016     Node* current = st;
1017     // Skip through chains of MemBarNodes checking the MergeMems for
1018     // new states for the slice of this load.  Stop once any other
1019     // kind of node is encountered.  Loads from final memory can skip
1020     // through any kind of MemBar but normal loads shouldn't skip
1021     // through MemBarAcquire since the could allow them to move out of
1022     // a synchronized region.
1023     while (current->is_Proj()) {
1024       int opc = current->in(0)->Opcode();
1025       if ((final && (opc == Op_MemBarAcquire ||
1026                      opc == Op_MemBarAcquireLock ||
1027                      opc == Op_LoadFence)) ||
1028           opc == Op_MemBarRelease ||
1029           opc == Op_StoreFence ||
1030           opc == Op_MemBarReleaseLock ||
1031           opc == Op_MemBarStoreStore ||
1032           opc == Op_MemBarCPUOrder) {
1033         Node* mem = current->in(0)->in(TypeFunc::Memory);
1034         if (mem->is_MergeMem()) {
1035           MergeMemNode* merge = mem->as_MergeMem();
1036           Node* new_st = merge->memory_at(alias_idx);
1037           if (new_st == merge->base_memory()) {
1038             // Keep searching
1039             current = new_st;
1040             continue;
1041           }
1042           // Save the new memory state for the slice and fall through
1043           // to exit.
1044           result = new_st;
1045         }
1046       }
1047       break;
1048     }
1049     if (result != NULL) {
1050       st = result;
1051     }
1052   }
1053 
1054   // Loop around twice in the case Load -> Initialize -> Store.
1055   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1056   for (int trip = 0; trip <= 1; trip++) {
1057 
1058     if (st->is_Store()) {
1059       Node* st_adr = st->in(MemNode::Address);
1060       if (!phase->eqv(st_adr, ld_adr)) {
1061         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1062         intptr_t st_off = 0;
1063         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1064         if (ld_base == NULL)                                   return NULL;
1065         if (st_base == NULL)                                   return NULL;
1066         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1067         if (ld_off != st_off)                                  return NULL;
1068         if (ld_off == Type::OffsetBot)                         return NULL;
1069         // Same base, same offset.
1070         // Possible improvement for arrays: check index value instead of absolute offset.
1071 
1072         // At this point we have proven something like this setup:
1073         //   B = << base >>
1074         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1075         //   S = StoreQ(AddP(             B , #Off), V)
1076         // (Actually, we haven't yet proven the Q's are the same.)
1077         // In other words, we are loading from a casted version of
1078         // the same pointer-and-offset that we stored to.
1079         // Casted version may carry a dependency and it is respected.
1080         // Thus, we are able to replace L by V.
1081       }
1082       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1083       if (store_Opcode() != st->Opcode())
1084         return NULL;
1085       return st->in(MemNode::ValueIn);
1086     }
1087 
1088     // A load from a freshly-created object always returns zero.
1089     // (This can happen after LoadNode::Ideal resets the load's memory input
1090     // to find_captured_store, which returned InitializeNode::zero_memory.)
1091     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1092         (st->in(0) == ld_alloc) &&
1093         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1094       // return a zero value for the load's basic type
1095       // (This is one of the few places where a generic PhaseTransform
1096       // can create new nodes.  Think of it as lazily manifesting
1097       // virtually pre-existing constants.)
1098       assert(memory_type() != T_INLINE_TYPE, "should not be used for inline types");
1099       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1100       if (default_value != NULL) {
1101         return default_value;
1102       }
1103       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1104       return phase->zerocon(memory_type());
1105     }
1106 
1107     // A load from an initialization barrier can match a captured store.
1108     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1109       InitializeNode* init = st->in(0)->as_Initialize();
1110       AllocateNode* alloc = init->allocation();
1111       if ((alloc != NULL) && (alloc == ld_alloc)) {
1112         // examine a captured store value
1113         st = init->find_captured_store(ld_off, memory_size(), phase);
1114         if (st != NULL) {
1115           continue;             // take one more trip around
1116         }
1117       }
1118     }
1119 
1120     // Load boxed value from result of valueOf() call is input parameter.
1121     if (this->is_Load() && ld_adr->is_AddP() &&
1122         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1123       intptr_t ignore = 0;
1124       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1125       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1126       base = bs->step_over_gc_barrier(base);
1127       if (base != NULL && base->is_Proj() &&
1128           base->as_Proj()->_con == TypeFunc::Parms &&
1129           base->in(0)->is_CallStaticJava() &&
1130           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1131         return base->in(0)->in(TypeFunc::Parms);
1132       }
1133     }
1134 
1135     break;
1136   }
1137 
1138   return NULL;
1139 }

1158 //------------------------------Identity---------------------------------------
1159 // Loads are identity if previous store is to same address
1160 Node* LoadNode::Identity(PhaseGVN* phase) {
1161   // Loading from an InlineTypePtr? The InlineTypePtr has the values of
1162   // all fields as input. Look for the field with matching offset.
1163   Node* addr = in(Address);
1164   intptr_t offset;
1165   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1166   if (base != NULL && base->is_InlineTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1167     Node* value = base->as_InlineTypePtr()->field_value_by_offset((int)offset, true);
1168     if (value->is_InlineType()) {
1169       // Non-flattened inline type field
1170       InlineTypeNode* vt = value->as_InlineType();
1171       if (vt->is_allocated(phase)) {
1172         value = vt->get_oop();
1173       } else {
1174         // Not yet allocated, bail out
1175         value = NULL;
1176       }
1177     }
1178     if (value != NULL) {
1179       if (Opcode() == Op_LoadN) {
1180         // Encode oop value if we are loading a narrow oop
1181         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1182         value = phase->transform(new EncodePNode(value, bottom_type()));
1183       }
1184       return value;
1185     }
1186   }
1187 
1188   // If the previous store-maker is the right kind of Store, and the store is
1189   // to the same address, then we are equal to the value stored.
1190   Node* mem = in(Memory);
1191   Node* value = can_see_stored_value(mem, phase);
1192   if( value ) {
1193     // byte, short & char stores truncate naturally.
1194     // A load has to load the truncated value which requires
1195     // some sort of masking operation and that requires an
1196     // Ideal call instead of an Identity call.
1197     if (memory_size() < BytesPerInt) {
1198       // If the input to the store does not fit with the load's result type,
1199       // it must be truncated via an Ideal call.
1200       if (!phase->type(value)->higher_equal(phase->type(this)))
1201         return this;
1202     }
1203     // (This works even when value is a Con, but LoadNode::Value
1204     // usually runs first, producing the singleton type of the Con.)
1205     return value;
1206   }
1207 
1208   // Search for an existing data phi which was generated before for the same
1209   // instance's field to avoid infinite generation of phis in a loop.
1210   Node *region = mem->in(0);
1211   if (is_instance_field_load_with_local_phi(region)) {
1212     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1213     int this_index  = phase->C->get_alias_index(addr_t);
1214     int this_offset = addr_t->offset();
1215     int this_iid    = addr_t->instance_id();
1216     if (!addr_t->is_known_instance() &&
1217          addr_t->is_ptr_to_boxed_value()) {
1218       // Use _idx of address base (could be Phi node) for boxed values.
1219       intptr_t   ignore = 0;
1220       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1221       if (base == NULL) {
1222         return this;
1223       }
1224       this_iid = base->_idx;
1225     }
1226     const Type* this_type = bottom_type();
1227     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1228       Node* phi = region->fast_out(i);
1229       if (phi->is_Phi() && phi != mem &&
1230           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1231         return phi;
1232       }
1233     }
1234   }
1235 
1236   return this;
1237 }

1277 // We're loading from an object which has autobox behaviour.
1278 // If this object is result of a valueOf call we'll have a phi
1279 // merging a newly allocated object and a load from the cache.
1280 // We want to replace this load with the original incoming
1281 // argument to the valueOf call.
1282 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1283   assert(phase->C->eliminate_boxing(), "sanity");
1284   intptr_t ignore = 0;
1285   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1286   if ((base == NULL) || base->is_Phi()) {
1287     // Push the loads from the phi that comes from valueOf up
1288     // through it to allow elimination of the loads and the recovery
1289     // of the original value. It is done in split_through_phi().
1290     return NULL;
1291   } else if (base->is_Load() ||
1292              (base->is_DecodeN() && base->in(1)->is_Load())) {
1293     // Eliminate the load of boxed value for integer types from the cache
1294     // array by deriving the value from the index into the array.
1295     // Capture the offset of the load and then reverse the computation.
1296 
1297     // Get LoadN node which loads a boxing object from 'cache' array.
1298     if (base->is_DecodeN()) {
1299       base = base->in(1);
1300     }
1301     if (!base->in(Address)->is_AddP()) {
1302       return NULL; // Complex address
1303     }
1304     AddPNode* address = base->in(Address)->as_AddP();
1305     Node* cache_base = address->in(AddPNode::Base);
1306     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1307       // Get ConP node which is static 'cache' field.
1308       cache_base = cache_base->in(1);
1309     }
1310     if ((cache_base != NULL) && cache_base->is_Con()) {
1311       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1312       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1313         Node* elements[4];
1314         int shift = exact_log2(type2aelembytes(T_OBJECT));
1315         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1316         if (count > 0 && elements[0]->is_Con() &&
1317             (count == 1 ||
1318              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1319                             elements[1]->in(2) == phase->intcon(shift)))) {
1320           ciObjArray* array = base_type->const_oop()->as_obj_array();
1321           // Fetch the box object cache[0] at the base of the array and get its value
1322           ciInstance* box = array->obj_at(0)->as_instance();
1323           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1324           assert(ik->is_box_klass(), "sanity");
1325           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1326           if (ik->nof_nonstatic_fields() == 1) {
1327             // This should be true nonstatic_field_at requires calling
1328             // nof_nonstatic_fields so check it anyway
1329             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1330             BasicType bt = c.basic_type();
1331             // Only integer types have boxing cache.
1332             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1333                    bt == T_BYTE    || bt == T_SHORT ||
1334                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1335             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1336             if (cache_low != (int)cache_low) {
1337               return NULL; // should not happen since cache is array indexed by value
1338             }
1339             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1340             if (offset != (int)offset) {
1341               return NULL; // should not happen since cache is array indexed by value
1342             }
1343            // Add up all the offsets making of the address of the load
1344             Node* result = elements[0];
1345             for (int i = 1; i < count; i++) {
1346               result = phase->transform(new AddXNode(result, elements[i]));
1347             }
1348             // Remove the constant offset from the address and then
1349             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1350             // remove the scaling of the offset to recover the original index.
1351             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1352               // Peel the shift off directly but wrap it in a dummy node
1353               // since Ideal can't return existing nodes
1354               result = new RShiftXNode(result->in(1), phase->intcon(0));
1355             } else if (result->is_Add() && result->in(2)->is_Con() &&
1356                        result->in(1)->Opcode() == Op_LShiftX &&
1357                        result->in(1)->in(2) == phase->intcon(shift)) {
1358               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1359               // but for boxing cache access we know that X<<Z will not overflow
1360               // (there is range check) so we do this optimizatrion by hand here.
1361               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1362               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1363             } else {
1364               result = new RShiftXNode(result, phase->intcon(shift));
1365             }
1366 #ifdef _LP64
1367             if (bt != T_LONG) {
1368               result = new ConvL2INode(phase->transform(result));
1369             }
1370 #else
1371             if (bt == T_LONG) {
1372               result = new ConvI2LNode(phase->transform(result));
1373             }
1374 #endif
1375             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1376             // Need to preserve unboxing load type if it is unsigned.
1377             switch(this->Opcode()) {
1378               case Op_LoadUB:
1379                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1380                 break;
1381               case Op_LoadUS:
1382                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1383                 break;
1384             }
1385             return result;
1386           }
1387         }
1388       }
1389     }
1390   }
1391   return NULL;
1392 }

1394 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1395   Node* region = phi->in(0);
1396   if (region == NULL) {
1397     return false; // Wait stable graph
1398   }
1399   uint cnt = phi->req();
1400   for (uint i = 1; i < cnt; i++) {
1401     Node* rc = region->in(i);
1402     if (rc == NULL || phase->type(rc) == Type::TOP)
1403       return false; // Wait stable graph
1404     Node* in = phi->in(i);
1405     if (in == NULL || phase->type(in) == Type::TOP)
1406       return false; // Wait stable graph
1407   }
1408   return true;

1410 //------------------------------split_through_phi------------------------------
1411 // Split instance or boxed field load through Phi.
1412 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1413   Node* mem     = in(Memory);
1414   Node* address = in(Address);
1415   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1416 
1417   assert((t_oop != NULL) &&
1418          (t_oop->is_known_instance_field() ||
1419           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1420 
1421   Compile* C = phase->C;
1422   intptr_t ignore = 0;
1423   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1424   bool base_is_phi = (base != NULL) && base->is_Phi();
1425   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1426                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1427                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1428 
1429   if (!((mem->is_Phi() || base_is_phi) &&
1430         (load_boxed_values || t_oop->is_known_instance_field()))) {
1431     return NULL; // memory is not Phi
1432   }
1433 
1434   if (mem->is_Phi()) {
1435     if (!stable_phi(mem->as_Phi(), phase)) {
1436       return NULL; // Wait stable graph
1437     }
1438     uint cnt = mem->req();
1439     // Check for loop invariant memory.
1440     if (cnt == 3) {
1441       for (uint i = 1; i < cnt; i++) {
1442         Node* in = mem->in(i);
1443         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1444         if (m == mem) {
1445           if (i == 1) {
1446             // if the first edge was a loop, check second edge too.
1447             // If both are replaceable - we are in an infinite loop
1448             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1449             if (n == mem) {
1450               break;
1451             }
1452           }
1453           set_req(Memory, mem->in(cnt - i));
1454           return this; // made change
1455         }
1456       }
1457     }
1458   }
1459   if (base_is_phi) {
1460     if (!stable_phi(base->as_Phi(), phase)) {
1461       return NULL; // Wait stable graph
1462     }
1463     uint cnt = base->req();
1464     // Check for loop invariant memory.
1465     if (cnt == 3) {
1466       for (uint i = 1; i < cnt; i++) {
1467         if (base->in(i) == base) {
1468           return NULL; // Wait stable graph
1469         }
1470       }
1471     }
1472   }
1473 
1474   // Split through Phi (see original code in loopopts.cpp).
1475   assert(C->have_alias_type(t_oop), "instance should have alias type");
1476 
1477   // Do nothing here if Identity will find a value
1478   // (to avoid infinite chain of value phis generation).
1479   if (!phase->eqv(this, this->Identity(phase))) {
1480     return NULL;
1481   }
1482 
1483   // Select Region to split through.
1484   Node* region;
1485   if (!base_is_phi) {
1486     assert(mem->is_Phi(), "sanity");
1487     region = mem->in(0);
1488     // Skip if the region dominates some control edge of the address.
1489     if (!MemNode::all_controls_dominate(address, region))
1490       return NULL;
1491   } else if (!mem->is_Phi()) {
1492     assert(base_is_phi, "sanity");
1493     region = base->in(0);
1494     // Skip if the region dominates some control edge of the memory.
1495     if (!MemNode::all_controls_dominate(mem, region))
1496       return NULL;
1497   } else if (base->in(0) != mem->in(0)) {
1498     assert(base_is_phi && mem->is_Phi(), "sanity");
1499     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1500       region = base->in(0);
1501     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1502       region = mem->in(0);
1503     } else {
1504       return NULL; // complex graph
1505     }
1506   } else {
1507     assert(base->in(0) == mem->in(0), "sanity");
1508     region = mem->in(0);
1509   }
1510 
1511   const Type* this_type = this->bottom_type();
1512   int this_index  = C->get_alias_index(t_oop);
1513   int this_offset = t_oop->offset();
1514   int this_iid    = t_oop->instance_id();
1515   if (!t_oop->is_known_instance() && load_boxed_values) {
1516     // Use _idx of address base for boxed values.
1517     this_iid = base->_idx;
1518   }
1519   PhaseIterGVN* igvn = phase->is_IterGVN();
1520   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1521   for (uint i = 1; i < region->req(); i++) {
1522     Node* x;
1523     Node* the_clone = NULL;
1524     Node* in = region->in(i);
1525     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1526         in != NULL && in->is_OuterStripMinedLoop()) {
1527       // No node should go in the outer strip mined loop
1528       in = in->in(LoopNode::EntryControl);
1529     }
1530     if (in == NULL || in == C->top()) {
1531       x = C->top();      // Dead path?  Use a dead data op
1532     } else {
1533       x = this->clone();        // Else clone up the data op
1534       the_clone = x;            // Remember for possible deletion.
1535       // Alter data node to use pre-phi inputs
1536       if (this->in(0) == region) {
1537         x->set_req(0, in);
1538       } else {
1539         x->set_req(0, NULL);
1540       }
1541       if (mem->is_Phi() && (mem->in(0) == region)) {
1542         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1543       }
1544       if (address->is_Phi() && address->in(0) == region) {
1545         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1546       }
1547       if (base_is_phi && (base->in(0) == region)) {
1548         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1549         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1550         x->set_req(Address, adr_x);
1551       }
1552     }
1553     // Check for a 'win' on some paths
1554     const Type *t = x->Value(igvn);
1555 
1556     bool singleton = t->singleton();
1557 
1558     // See comments in PhaseIdealLoop::split_thru_phi().
1559     if (singleton && t == Type::TOP) {
1560       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1561     }
1562 
1563     if (singleton) {
1564       x = igvn->makecon(t);
1565     } else {
1566       // We now call Identity to try to simplify the cloned node.
1567       // Note that some Identity methods call phase->type(this).
1568       // Make sure that the type array is big enough for
1569       // our new node, even though we may throw the node away.
1570       // (This tweaking with igvn only works because x is a new node.)
1571       igvn->set_type(x, t);
1572       // If x is a TypeNode, capture any more-precise type permanently into Node
1573       // otherwise it will be not updated during igvn->transform since
1574       // igvn->type(x) is set to x->Value() already.
1575       x->raise_bottom_type(t);
1576       Node* y = x->Identity(igvn);
1577       if (y != x) {
1578         x = y;
1579       } else {
1580         y = igvn->hash_find_insert(x);
1581         if (y) {
1582           x = y;
1583         } else {
1584           // Else x is a new node we are keeping
1585           // We do not need register_new_node_with_optimizer
1586           // because set_type has already been called.
1587           igvn->_worklist.push(x);
1588         }
1589       }
1590     }
1591     if (x != the_clone && the_clone != NULL) {
1592       igvn->remove_dead_node(the_clone);
1593     }
1594     phi->set_req(i, x);
1595   }
1596   // Record Phi
1597   igvn->register_new_node_with_optimizer(phi);
1598   return phi;
1599 }

1617 //------------------------------Ideal------------------------------------------
1618 // If the load is from Field memory and the pointer is non-null, it might be possible to
1619 // zero out the control input.
1620 // If the offset is constant and the base is an object allocation,
1621 // try to hook me up to the exact initializing store.
1622 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1623   Node* p = MemNode::Ideal_common(phase, can_reshape);
1624   if (p)  return (p == NodeSentinel) ? NULL : p;
1625 
1626   Node* ctrl    = in(MemNode::Control);
1627   Node* address = in(MemNode::Address);
1628   bool progress = false;
1629 
1630   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1631          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1632 
1633   // Skip up past a SafePoint control.  Cannot do this for Stores because
1634   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1635   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1636       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1637       !addr_mark &&
1638       (depends_only_on_test() || has_unknown_control_dependency())) {
1639     ctrl = ctrl->in(0);
1640     set_req(MemNode::Control,ctrl);
1641     progress = true;
1642   }
1643 
1644   intptr_t ignore = 0;
1645   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1646   if (base != NULL
1647       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1648     // Check for useless control edge in some common special cases
1649     if (in(MemNode::Control) != NULL
1650         && can_remove_control()
1651         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1652         && all_controls_dominate(base, phase->C->start())) {
1653       // A method-invariant, non-null address (constant or 'this' argument).
1654       set_req(MemNode::Control, NULL);
1655       progress = true;
1656     }
1657   }
1658 
1659   Node* mem = in(MemNode::Memory);
1660   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1661 
1662   if (can_reshape && (addr_t != NULL)) {
1663     // try to optimize our memory input
1664     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1665     if (opt_mem != mem) {
1666       set_req(MemNode::Memory, opt_mem);
1667       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1668       return this;
1669     }
1670     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1671     if ((t_oop != NULL) &&
1672         (t_oop->is_known_instance_field() ||
1673          t_oop->is_ptr_to_boxed_value())) {
1674       PhaseIterGVN *igvn = phase->is_IterGVN();
1675       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1676         // Delay this transformation until memory Phi is processed.
1677         phase->is_IterGVN()->_worklist.push(this);
1678         return NULL;
1679       }
1680       // Split instance field load through Phi.
1681       Node* result = split_through_phi(phase);
1682       if (result != NULL) return result;
1683 
1684       if (t_oop->is_ptr_to_boxed_value()) {
1685         Node* result = eliminate_autobox(phase);
1686         if (result != NULL) return result;
1687       }
1688     }
1689   }
1690 
1691   // Is there a dominating load that loads the same value?  Leave
1692   // anything that is not a load of a field/array element (like
1693   // barriers etc.) alone
1694   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1695     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1696       Node *use = mem->fast_out(i);
1697       if (use != this &&
1698           use->Opcode() == Opcode() &&
1699           use->in(0) != NULL &&
1700           use->in(0) != in(0) &&
1701           use->in(Address) == in(Address)) {
1702         Node* ctl = in(0);
1703         for (int i = 0; i < 10 && ctl != NULL; i++) {
1704           ctl = IfNode::up_one_dom(ctl);
1705           if (ctl == use->in(0)) {
1706             set_req(0, use->in(0));
1707             return this;
1708           }
1709         }
1710       }
1711     }
1712   }
1713 
1714   // Check for prior store with a different base or offset; make Load
1715   // independent.  Skip through any number of them.  Bail out if the stores
1716   // are in an endless dead cycle and report no progress.  This is a key
1717   // transform for Reflection.  However, if after skipping through the Stores
1718   // we can't then fold up against a prior store do NOT do the transform as
1719   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1720   // array memory alive twice: once for the hoisted Load and again after the
1721   // bypassed Store.  This situation only works if EVERYBODY who does
1722   // anti-dependence work knows how to bypass.  I.e. we need all
1723   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1724   // the alias index stuff.  So instead, peek through Stores and IFF we can
1725   // fold up, do so.
1726   Node* prev_mem = find_previous_store(phase);
1727   if (prev_mem != NULL) {
1728     Node* value = can_see_arraycopy_value(prev_mem, phase);
1729     if (value != NULL) {
1730       return value;
1731     }
1732   }
1733   // Steps (a), (b):  Walk past independent stores to find an exact match.
1734   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1735     // (c) See if we can fold up on the spot, but don't fold up here.
1736     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1737     // just return a prior value, which is done by Identity calls.
1738     if (can_see_stored_value(prev_mem, phase)) {
1739       // Make ready for step (d):
1740       set_req(MemNode::Memory, prev_mem);
1741       return this;
1742     }
1743   }
1744 
1745   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1746   if (alloc != NULL && mem->is_Proj() &&
1747       mem->in(0) != NULL &&
1748       mem->in(0) == alloc->initialization() &&
1749       Opcode() == Op_LoadX &&
1750       alloc->initialization()->proj_out_or_null(0) != NULL) {
1751     InitializeNode* init = alloc->initialization();
1752     Node* control = init->proj_out(0);
1753     return alloc->make_ideal_mark(phase, control, mem);
1754   }
1755 
1756   return progress ? this : NULL;
1757 }

1759 // Helper to recognize certain Klass fields which are invariant across
1760 // some group of array types (e.g., int[] or all T[] where T < Object).
1761 const Type*
1762 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1763                                  ciKlass* klass) const {
1764   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1765     // The field is Klass::_modifier_flags.  Return its (constant) value.
1766     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1767     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1768     return TypeInt::make(klass->modifier_flags());
1769   }
1770   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1771     // The field is Klass::_access_flags.  Return its (constant) value.
1772     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1773     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1774     return TypeInt::make(klass->access_flags());
1775   }
1776   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1777     // The field is Klass::_layout_helper.  Return its constant value if known.
1778     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1779     return TypeInt::make(klass->layout_helper());
1780   }
1781 
1782   // No match.
1783   return NULL;
1784 }

1786 //------------------------------Value-----------------------------------------
1787 const Type* LoadNode::Value(PhaseGVN* phase) const {
1788   // Either input is TOP ==> the result is TOP
1789   Node* mem = in(MemNode::Memory);
1790   const Type *t1 = phase->type(mem);
1791   if (t1 == Type::TOP)  return Type::TOP;
1792   Node* adr = in(MemNode::Address);
1793   const TypePtr* tp = phase->type(adr)->isa_ptr();
1794   if (tp == NULL || tp->empty())  return Type::TOP;
1795   int off = tp->offset();
1796   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1797   Compile* C = phase->C;
1798 
1799   // Try to guess loaded type from pointer type
1800   if (tp->isa_aryptr()) {
1801     const TypeAryPtr* ary = tp->is_aryptr();
1802     const Type* t = ary->elem();
1803 
1804     // Determine whether the reference is beyond the header or not, by comparing
1805     // the offset against the offset of the start of the array's data.
1806     // Different array types begin at slightly different offsets (12 vs. 16).
1807     // We choose T_BYTE as an example base type that is least restrictive
1808     // as to alignment, which will therefore produce the smallest
1809     // possible base offset.
1810     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1811     const bool off_beyond_header = (off >= min_base_off);
1812 
1813     // Try to constant-fold a stable array element.
1814     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1815       // Make sure the reference is not into the header and the offset is constant
1816       ciObject* aobj = ary->const_oop();
1817       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1818         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1819         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1820                                                                       stable_dimension,
1821                                                                       memory_type(), is_unsigned());
1822         if (con_type != NULL) {
1823           return con_type;
1824         }
1825       }
1826     }
1827 
1828     // Don't do this for integer types. There is only potential profit if
1829     // the element type t is lower than _type; that is, for int types, if _type is
1830     // more restrictive than t.  This only happens here if one is short and the other
1831     // char (both 16 bits), and in those cases we've made an intentional decision
1832     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1833     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1834     //
1835     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1836     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1837     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1838     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1839     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1840     // In fact, that could have been the original type of p1, and p1 could have
1841     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1842     // expression (LShiftL quux 3) independently optimized to the constant 8.
1843     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1844         && (_type->isa_vect() == NULL)
1845         && t->isa_inlinetype() == NULL
1846         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1847       // t might actually be lower than _type, if _type is a unique
1848       // concrete subclass of abstract class t.
1849       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1850         const Type* jt = t->join_speculative(_type);
1851         // In any case, do not allow the join, per se, to empty out the type.
1852         if (jt->empty() && !t->empty()) {
1853           // This can happen if a interface-typed array narrows to a class type.
1854           jt = _type;
1855         }
1856 #ifdef ASSERT
1857         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1858           // The pointers in the autobox arrays are always non-null
1859           Node* base = adr->in(AddPNode::Base);
1860           if ((base != NULL) && base->is_DecodeN()) {
1861             // Get LoadN node which loads IntegerCache.cache field
1862             base = base->in(1);
1863           }
1864           if ((base != NULL) && base->is_Con()) {
1865             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1866             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1867               // It could be narrow oop
1868               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1869             }
1870           }
1871         }
1872 #endif
1873         return jt;
1874       }
1875     }
1876   } else if (tp->base() == Type::InstPtr) {
1877     assert( off != Type::OffsetBot ||
1878             // arrays can be cast to Objects
1879             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1880             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1881             // unsafe field access may not have a constant offset
1882             C->has_unsafe_access(),
1883             "Field accesses must be precise" );
1884     // For oop loads, we expect the _type to be precise.
1885 
1886     const TypeInstPtr* tinst = tp->is_instptr();
1887     BasicType bt = memory_type();
1888 
1889     // Optimize loads from constant fields.
1890     ciObject* const_oop = tinst->const_oop();
1891     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1892       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1893       if (mirror_type != NULL && mirror_type->is_inlinetype()) {
1894         ciInlineKlass* vk = mirror_type->as_inline_klass();
1895         if (off == vk->default_value_offset()) {
1896           // Loading a special hidden field that contains the oop of the default inline type
1897           const Type* const_oop = TypeInstPtr::make(vk->default_instance());
1898           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1899         }
1900       }
1901       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1902       if (con_type != NULL) {
1903         return con_type;
1904       }
1905     }
1906   } else if (tp->base() == Type::KlassPtr) {
1907     assert( off != Type::OffsetBot ||
1908             // arrays can be cast to Objects
1909             tp->is_klassptr()->klass() == NULL ||
1910             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1911             // also allow array-loading from the primary supertype
1912             // array during subtype checks
1913             Opcode() == Op_LoadKlass,
1914             "Field accesses must be precise" );
1915     // For klass/static loads, we expect the _type to be precise
1916   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1917     if (adr->is_Load() && off == 0) {
1918       /* With mirrors being an indirect in the Klass*
1919        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1920        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1921        *
1922        * So check the type and klass of the node before the LoadP.
1923        */
1924       Node* adr2 = adr->in(MemNode::Address);
1925       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1926       if (tkls != NULL) {
1927         ciKlass* klass = tkls->klass();
1928         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1929           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1930           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1931           return TypeInstPtr::make(klass->java_mirror());
1932         }
1933       }
1934     } else {
1935       // Check for a load of the default value offset from the InlineKlassFixedBlock:
1936       // LoadI(LoadP(inline_klass, adr_inlineklass_fixed_block_offset), default_value_offset_offset)
1937       intptr_t offset = 0;
1938       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1939       if (base != NULL && base->is_Load() && offset == in_bytes(InlineKlass::default_value_offset_offset())) {
1940         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1941         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_inlinetype() &&
1942             tkls->offset() == in_bytes(InstanceKlass::adr_inlineklass_fixed_block_offset())) {
1943           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1944           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1945           return TypeInt::make(tkls->klass()->as_inline_klass()->default_value_offset());
1946         }
1947       }
1948     }
1949   }
1950 
1951   const TypeKlassPtr *tkls = tp->isa_klassptr();
1952   if (tkls != NULL && !StressReflectiveCode) {
1953     ciKlass* klass = tkls->klass();
1954     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1955       // We are loading a field from a Klass metaobject whose identity
1956       // is known at compile time (the type is "exact" or "precise").
1957       // Check for fields we know are maintained as constants by the VM.
1958       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1959         // The field is Klass::_super_check_offset.  Return its (constant) value.
1960         // (Folds up type checking code.)
1961         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1962         return TypeInt::make(klass->super_check_offset());
1963       }
1964       // Compute index into primary_supers array
1965       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1966       // Check for overflowing; use unsigned compare to handle the negative case.
1967       if( depth < ciKlass::primary_super_limit() ) {
1968         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1969         // (Folds up type checking code.)
1970         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1971         ciKlass *ss = klass->super_of_depth(depth);
1972         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1973       }
1974       const Type* aift = load_array_final_field(tkls, klass);
1975       if (aift != NULL)  return aift;
1976     }
1977 
1978     // We can still check if we are loading from the primary_supers array at a
1979     // shallow enough depth.  Even though the klass is not exact, entries less
1980     // than or equal to its super depth are correct.
1981     if (tkls->is_loaded()) {
1982       ciType *inner = klass;
1983       while( inner->is_obj_array_klass() )
1984         inner = inner->as_obj_array_klass()->base_element_type();
1985       if( inner->is_instance_klass() &&
1986           !inner->as_instance_klass()->flags().is_interface() ) {
1987         // Compute index into primary_supers array
1988         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1989         // Check for overflowing; use unsigned compare to handle the negative case.
1990         if( depth < ciKlass::primary_super_limit() &&
1991             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1992           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1993           // (Folds up type checking code.)
1994           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1995           ciKlass *ss = klass->super_of_depth(depth);
1996           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1997         }
1998       }
1999     }
2000 
2001     // If the type is enough to determine that the thing is not an array,
2002     // we can give the layout_helper a positive interval type.
2003     // This will help short-circuit some reflective code.
2004     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2005         && !klass->is_array_klass() // not directly typed as an array
2006         && !klass->is_interface()  // specifically not Serializable & Cloneable
2007         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2008         ) {
2009       // Note:  When interfaces are reliable, we can narrow the interface
2010       // test to (klass != Serializable && klass != Cloneable).
2011       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2012       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2013       // The key property of this type is that it folds up tests
2014       // for array-ness, since it proves that the layout_helper is positive.
2015       // Thus, a generic value like the basic object layout helper works fine.
2016       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2017     }
2018   }
2019 
2020   // If we are loading from a freshly-allocated object, produce a zero,
2021   // if the load is provably beyond the header of the object.
2022   // (Also allow a variable load from a fresh array to produce zero.)
2023   const TypeOopPtr *tinst = tp->isa_oopptr();
2024   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2025   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
2026   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
2027     Node* value = can_see_stored_value(mem,phase);
2028     if (value != NULL && value->is_Con()) {
2029       assert(value->bottom_type()->higher_equal(_type),"sanity");
2030       return value->bottom_type();
2031     }
2032   }
2033 
2034   if (is_instance) {
2035     // If we have an instance type and our memory input is the
2036     // programs's initial memory state, there is no matching store,
2037     // so just return a zero of the appropriate type
2038     Node *mem = in(MemNode::Memory);
2039     if (mem->is_Parm() && mem->in(0)->is_Start()) {
2040       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
2041       return Type::get_zero_type(_type->basic_type());
2042     }
2043   }
2044 
2045   Node* alloc = is_new_object_mark_load(phase);
2046   if (alloc != NULL && !(alloc->Opcode() == Op_Allocate && UseBiasedLocking)) {
2047     return TypeX::make(markWord::prototype().value());

2587 //------------------------------Ideal------------------------------------------
2588 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2589 // When a store immediately follows a relevant allocation/initialization,
2590 // try to capture it into the initialization, or hoist it above.
2591 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2592   Node* p = MemNode::Ideal_common(phase, can_reshape);
2593   if (p)  return (p == NodeSentinel) ? NULL : p;
2594 
2595   Node* mem     = in(MemNode::Memory);
2596   Node* address = in(MemNode::Address);
2597   // Back-to-back stores to same address?  Fold em up.  Generally
2598   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2599   // since they must follow each StoreP operation.  Redundant StoreCMs
2600   // are eliminated just before matching in final_graph_reshape.
2601   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::INLINES) {
2602     Node* st = mem;
2603     // If Store 'st' has more than one use, we cannot fold 'st' away.
2604     // For example, 'st' might be the final state at a conditional
2605     // return.  Or, 'st' might be used by some node which is live at
2606     // the same time 'st' is live, which might be unschedulable.  So,
2607     // require exactly ONE user until such time as we clone 'mem' for
2608     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2609     // true).
2610     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2611       // Looking at a dead closed cycle of memory?
2612       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2613       assert(Opcode() == st->Opcode() ||
2614              st->Opcode() == Op_StoreVector ||
2615              Opcode() == Op_StoreVector ||
2616              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2617              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2618              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2619              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2620              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2621              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2622 
2623       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2624           st->as_Store()->memory_size() <= this->memory_size()) {
2625         Node* use = st->raw_out(0);
2626         phase->igvn_rehash_node_delayed(use);
2627         if (can_reshape) {
2628           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2629         } else {
2630           // It's OK to do this in the parser, since DU info is always accurate,
2631           // and the parser always refers to nodes via SafePointNode maps.
2632           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2633         }
2634         return this;
2635       }
2636       st = st->in(MemNode::Memory);
2637     }
2638   }
2639 
2640 
2641   // Capture an unaliased, unconditional, simple store into an initializer.
2642   // Or, if it is independent of the allocation, hoist it above the allocation.
2643   if (ReduceFieldZeroing && /*can_reshape &&*/
2644       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2645     InitializeNode* init = mem->in(0)->as_Initialize();
2646     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2647     if (offset > 0) {
2648       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2649       // If the InitializeNode captured me, it made a raw copy of me,
2650       // and I need to disappear.
2651       if (moved != NULL) {
2652         // %%% hack to ensure that Ideal returns a new node:
2653         mem = MergeMemNode::make(mem);
2654         return mem;             // fold me away
2655       }
2656     }
2657   }
2658 
2659   return NULL;                  // No further progress
2660 }

2674 //------------------------------Identity---------------------------------------
2675 // Remove redundant stores:
2676 //   Store(m, p, Load(m, p)) changes to m.
2677 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2678 Node* StoreNode::Identity(PhaseGVN* phase) {
2679   Node* mem = in(MemNode::Memory);
2680   Node* adr = in(MemNode::Address);
2681   Node* val = in(MemNode::ValueIn);
2682 
2683   Node* result = this;
2684 
2685   // Load then Store?  Then the Store is useless
2686   if (val->is_Load() &&
2687       val->in(MemNode::Address)->eqv_uncast(adr) &&
2688       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2689       val->as_Load()->store_Opcode() == Opcode()) {
2690     result = mem;
2691   }
2692 
2693   // Two stores in a row of the same value?
2694   if (result == this &&
2695       mem->is_Store() &&
2696       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2697       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2698       mem->Opcode() == Opcode()) {
2699     result = mem;
2700   }
2701 
2702   // Store of zero anywhere into a freshly-allocated object?
2703   // Then the store is useless.
2704   // (It must already have been captured by the InitializeNode.)
2705   if (result == this && ReduceFieldZeroing) {
2706     // a newly allocated object is already all-zeroes everywhere
2707     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2708         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2709       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to inline type array is forbidden");
2710       result = mem;
2711     }
2712 
2713     if (result == this) {
2714       // the store may also apply to zero-bits in an earlier object
2715       Node* prev_mem = find_previous_store(phase);
2716       // Steps (a), (b):  Walk past independent stores to find an exact match.
2717       if (prev_mem != NULL) {
2718         Node* prev_val = can_see_stored_value(prev_mem, phase);
2719         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2720           // prev_val and val might differ by a cast; it would be good
2721           // to keep the more informative of the two.
2722           if (phase->type(val)->is_zero_type()) {
2723             result = mem;
2724           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2725             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2726             AllocateNode* alloc = init->allocation();
2727             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2728               result = mem;
2729             }
2730           }
2731         }
2732       }
2733     }
2734   }
2735 
2736   if (result != this && phase->is_IterGVN() != NULL) {
2737     MemBarNode* trailing = trailing_membar();
2738     if (trailing != NULL) {
2739 #ifdef ASSERT
2740       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2741       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2742 #endif
2743       PhaseIterGVN* igvn = phase->is_IterGVN();
2744       trailing->remove(igvn);
2745     }
2746   }
2747 
2748   return result;
2749 }

3018 //------------------------------Idealize---------------------------------------
3019 // Clearing a short array is faster with stores
3020 Node *ClearArrayNode::Ideal(PhaseGVN *phase, bool can_reshape) {
3021   // Already know this is a large node, do not try to ideal it
3022   if (!IdealizeClearArrayNode || _is_large) return NULL;
3023 
3024   const int unit = BytesPerLong;
3025   const TypeX* t = phase->type(in(2))->isa_intptr_t();
3026   if (!t)  return NULL;
3027   if (!t->is_con())  return NULL;
3028   intptr_t raw_count = t->get_con();
3029   intptr_t size = raw_count;
3030   if (!Matcher::init_array_count_is_in_bytes) size *= unit;
3031   // Clearing nothing uses the Identity call.
3032   // Negative clears are possible on dead ClearArrays
3033   // (see jck test stmt114.stmt11402.val).
3034   if (size <= 0 || size % unit != 0)  return NULL;
3035   intptr_t count = size / unit;
3036   // Length too long; communicate this to matchers and assemblers.
3037   // Assemblers are responsible to produce fast hardware clears for it.
3038   if (size > InitArrayShortSize) {
3039     return new ClearArrayNode(in(0), in(1), in(2), in(3), in(4), true);
3040   }
3041   Node *mem = in(1);
3042   if( phase->type(mem)==Type::TOP ) return NULL;
3043   Node *adr = in(3);
3044   const Type* at = phase->type(adr);
3045   if( at==Type::TOP ) return NULL;
3046   const TypePtr* atp = at->isa_ptr();
3047   // adjust atp to be the correct array element address type
3048   if (atp == NULL)  atp = TypePtr::BOTTOM;
3049   else              atp = atp->add_offset(Type::OffsetBot);
3050   // Get base for derived pointer purposes
3051   if( adr->Opcode() != Op_AddP ) Unimplemented();
3052   Node *base = adr->in(1);
3053 
3054   Node *val = in(4);
3055   Node *off  = phase->MakeConX(BytesPerLong);
3056   mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3057   count--;
3058   while( count-- ) {
3059     mem = phase->transform(mem);
3060     adr = phase->transform(new AddPNode(base,adr,off));
3061     mem = new StoreLNode(in(0), mem, adr, atp, val, MemNode::unordered, false);
3062   }

3253   // Don't bother trying to transform a dead node

3463 //===========================InitializeNode====================================
3464 // SUMMARY:
3465 // This node acts as a memory barrier on raw memory, after some raw stores.
3466 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3467 // The Initialize can 'capture' suitably constrained stores as raw inits.
3468 // It can coalesce related raw stores into larger units (called 'tiles').
3469 // It can avoid zeroing new storage for memory units which have raw inits.
3470 // At macro-expansion, it is marked 'complete', and does not optimize further.
3471 //
3472 // EXAMPLE:
3473 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3474 //   ctl = incoming control; mem* = incoming memory
3475 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3476 // First allocate uninitialized memory and fill in the header:
3477 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3478 //   ctl := alloc.Control; mem* := alloc.Memory*
3479 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3480 // Then initialize to zero the non-header parts of the raw memory block:
3481 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3482 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3483 // After the initialize node executes, the object is ready for service:
3484 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3485 // Suppose its body is immediately initialized as {1,2}:
3486 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3487 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3488 //   mem.SLICE(#short[*]) := store2
3489 //
3490 // DETAILS:
3491 // An InitializeNode collects and isolates object initialization after
3492 // an AllocateNode and before the next possible safepoint.  As a
3493 // memory barrier (MemBarNode), it keeps critical stores from drifting
3494 // down past any safepoint or any publication of the allocation.
3495 // Before this barrier, a newly-allocated object may have uninitialized bits.
3496 // After this barrier, it may be treated as a real oop, and GC is allowed.
3497 //
3498 // The semantics of the InitializeNode include an implicit zeroing of
3499 // the new object from object header to the end of the object.
3500 // (The object header and end are determined by the AllocateNode.)
3501 //
3502 // Certain stores may be added as direct inputs to the InitializeNode.
3503 // These stores must update raw memory, and they must be to addresses
3504 // derived from the raw address produced by AllocateNode, and with
3505 // a constant offset.  They must be ordered by increasing offset.
3506 // The first one is at in(RawStores), the last at in(req()-1).
3507 // Unlike most memory operations, they are not linked in a chain,
3508 // but are displayed in parallel as users of the rawmem output of
3509 // the allocation.
3510 //
3511 // (See comments in InitializeNode::capture_store, which continue
3512 // the example given above.)
3513 //
3514 // When the associated Allocate is macro-expanded, the InitializeNode
3515 // may be rewritten to optimize collected stores.  A ClearArrayNode
3516 // may also be created at that point to represent any required zeroing.
3517 // The InitializeNode is then marked 'complete', prohibiting further
3518 // capturing of nearby memory operations.
3519 //
3520 // During macro-expansion, all captured initializations which store
3521 // constant values of 32 bits or smaller are coalesced (if advantageous)
3522 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3523 // initialized in fewer memory operations.  Memory words which are
3524 // covered by neither tiles nor non-constant stores are pre-zeroed
3525 // by explicit stores of zero.  (The code shape happens to do all
3526 // zeroing first, then all other stores, with both sequences occurring
3527 // in order of ascending offsets.)
3528 //
3529 // Alternatively, code may be inserted between an AllocateNode and its
3530 // InitializeNode, to perform arbitrary initialization of the new object.
3531 // E.g., the object copying intrinsics insert complex data transfers here.
3532 // The initialization must then be marked as 'complete' disable the
3533 // built-in zeroing semantics and the collection of initializing stores.
3534 //
3535 // While an InitializeNode is incomplete, reads from the memory state
3536 // produced by it are optimizable if they match the control edge and
3537 // new oop address associated with the allocation/initialization.
3538 // They return a stored value (if the offset matches) or else zero.
3539 // A write to the memory state, if it matches control and address,
3540 // and if it is to a constant offset, may be 'captured' by the
3541 // InitializeNode.  It is cloned as a raw memory operation and rewired
3542 // inside the initialization, to the raw oop produced by the allocation.
3543 // Operations on addresses which are provably distinct (e.g., to
3544 // other AllocateNodes) are allowed to bypass the initialization.
3545 //
3546 // The effect of all this is to consolidate object initialization
3547 // (both arrays and non-arrays, both piecewise and bulk) into a
3548 // single location, where it can be optimized as a unit.
3549 //
3550 // Only stores with an offset less than TrackedInitializationLimit words
3551 // will be considered for capture by an InitializeNode.  This puts a
3552 // reasonable limit on the complexity of optimized initializations.
3553

3657   uint complexity_limit = 20;
3658   for (uint j = 0; j < worklist.size(); j++) {
3659     if (j >= complexity_limit) {
3660       return false;  // Bail out if processed too many nodes
3661     }
3662 
3663     Node* n = worklist.at(j);
3664     if (n == NULL)      continue;   // (can this really happen?)
3665     if (n->is_Proj())   n = n->in(0);
3666     if (n == this)      return false;  // found a cycle
3667     if (n->is_Con())    continue;
3668     if (n->is_Start())  continue;   // params, etc., are OK
3669     if (n->is_Root())   continue;   // even better
3670 
3671     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3672     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3673       continue;
3674     }
3675 
3676     Node* ctl = n->in(0);
3677     if (ctl != NULL && !ctl->is_top()) {
3678       if (ctl->is_Proj())  ctl = ctl->in(0);
3679       if (ctl == this)  return false;
3680 
3681       // If we already know that the enclosing memory op is pinned right after
3682       // the init, then any control flow that the store has picked up
3683       // must have preceded the init, or else be equal to the init.
3684       // Even after loop optimizations (which might change control edges)
3685       // a store is never pinned *before* the availability of its inputs.
3686       if (!MemNode::all_controls_dominate(n, this))
3687         return false;                  // failed to prove a good control
3688     }
3689 
3690     // Check data edges for possible dependencies on 'this'.
3691     for (uint i = 1; i < n->req(); i++) {
3692       Node* m = n->in(i);
3693       if (m == NULL || m == n || m->is_top())  continue;
3694 
3695       // Only process data inputs once
3696       worklist.push(m);
3697     }
3698   }
3699 
3700   return true;
3701 }

3703 // Here are all the checks a Store must pass before it can be moved into
3704 // an initialization.  Returns zero if a check fails.
3705 // On success, returns the (constant) offset to which the store applies,
3706 // within the initialized memory.
3707 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3708   const int FAIL = 0;
3709   if (st->req() != MemNode::ValueIn + 1)
3710     return FAIL;                // an inscrutable StoreNode (card mark?)
3711   Node* ctl = st->in(MemNode::Control);
3712   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3713     return FAIL;                // must be unconditional after the initialization
3714   Node* mem = st->in(MemNode::Memory);
3715   if (!(mem->is_Proj() && mem->in(0) == this))
3716     return FAIL;                // must not be preceded by other stores
3717   Node* adr = st->in(MemNode::Address);
3718   intptr_t offset;
3719   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3720   if (alloc == NULL)
3721     return FAIL;                // inscrutable address
3722   if (alloc != allocation())
3723     return FAIL;                // wrong allocation!  (store needs to float up)
3724   int size_in_bytes = st->memory_size();
3725   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3726     return FAIL;                // mismatched access
3727   }
3728   Node* val = st->in(MemNode::ValueIn);
3729 
3730   if (!detect_init_independence(val, phase))
3731     return FAIL;                // stored value must be 'simple enough'
3732 
3733   // The Store can be captured only if nothing after the allocation
3734   // and before the Store is using the memory location that the store
3735   // overwrites.
3736   bool failed = false;
3737   // If is_complete_with_arraycopy() is true the shape of the graph is
3738   // well defined and is safe so no need for extra checks.
3739   if (!is_complete_with_arraycopy()) {
3740     // We are going to look at each use of the memory state following
3741     // the allocation to make sure nothing reads the memory that the
3742     // Store writes.
3743     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3744     int alias_idx = phase->C->get_alias_index(t_adr);
3745     ResourceMark rm;
3746     Unique_Node_List mems;
3747     mems.push(mem);
3748     Node* unique_merge = NULL;
3749     for (uint next = 0; next < mems.size(); ++next) {
3750       Node *m  = mems.at(next);
3751       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3752         Node *n = m->fast_out(j);
3753         if (n->outcnt() == 0) {
3754           continue;
3755         }
3756         if (n == st) {
3757           continue;
3758         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3759           // If the control of this use is different from the control
3760           // of the Store which is right after the InitializeNode then
3761           // this node cannot be between the InitializeNode and the
3762           // Store.
3763           continue;
3764         } else if (n->is_MergeMem()) {
3765           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3766             // We can hit a MergeMemNode (that will likely go away
3767             // later) that is a direct use of the memory state
3768             // following the InitializeNode on the same slice as the
3769             // store node that we'd like to capture. We need to check
3770             // the uses of the MergeMemNode.
3771             mems.push(n);
3772           }
3773         } else if (n->is_Mem()) {
3774           Node* other_adr = n->in(MemNode::Address);
3775           if (other_adr == adr) {
3776             failed = true;
3777             break;
3778           } else {
3779             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3780             if (other_t_adr != NULL) {
3781               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3782               if (other_alias_idx == alias_idx) {
3783                 // A load from the same memory slice as the store right
3784                 // after the InitializeNode. We check the control of the
3785                 // object/array that is loaded from. If it's the same as
3786                 // the store control then we cannot capture the store.
3787                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3788                 Node* base = other_adr;
3789                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3790                 base = base->in(AddPNode::Base);
3791                 if (base != NULL) {
3792                   base = base->uncast();
3793                   if (base->is_Proj() && base->in(0) == alloc) {
3794                     failed = true;
3795                     break;
3796                   }
3797                 }
3798               }
3799             }
3800           }
3801         } else {
3802           failed = true;
3803           break;
3804         }
3805       }
3806     }
3807   }
3808   if (failed) {
3809     if (!can_reshape) {
3810       // We decided we couldn't capture the store during parsing. We
3811       // should try again during the next IGVN once the graph is
3812       // cleaner.
3813       phase->C->record_for_igvn(st);
3814     }
3815     return FAIL;
3816   }
3817 
3818   return offset;                // success
3819 }

3821 // Find the captured store in(i) which corresponds to the range
3822 // [start..start+size) in the initialized object.
3823 // If there is one, return its index i.  If there isn't, return the
3824 // negative of the index where it should be inserted.
3825 // Return 0 if the queried range overlaps an initialization boundary
3826 // or if dead code is encountered.
3827 // If size_in_bytes is zero, do not bother with overlap checks.
3828 int InitializeNode::captured_store_insertion_point(intptr_t start,
3829                                                    int size_in_bytes,
3830                                                    PhaseTransform* phase) {
3831   const int FAIL = 0, MAX_STORE = BytesPerLong;
3832 
3833   if (is_complete())
3834     return FAIL;                // arraycopy got here first; punt
3835 
3836   assert(allocation() != NULL, "must be present");
3837 
3838   // no negatives, no header fields:
3839   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3840 
3841   // after a certain size, we bail out on tracking all the stores:
3842   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3843   if (start >= ti_limit)  return FAIL;
3844 
3845   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3846     if (i >= limit)  return -(int)i; // not found; here is where to put it
3847 
3848     Node*    st     = in(i);
3849     intptr_t st_off = get_store_offset(st, phase);
3850     if (st_off < 0) {
3851       if (st != zero_memory()) {
3852         return FAIL;            // bail out if there is dead garbage
3853       }
3854     } else if (st_off > start) {
3855       // ...we are done, since stores are ordered
3856       if (st_off < start + size_in_bytes) {
3857         return FAIL;            // the next store overlaps
3858       }
3859       return -(int)i;           // not found; here is where to put it
3860     } else if (st_off < start) {
3861       if (size_in_bytes != 0 &&
3862           start < st_off + MAX_STORE &&
3863           start < st_off + st->as_Store()->memory_size()) {
3864         return FAIL;            // the previous store overlaps
3865       }
3866     } else {
3867       if (size_in_bytes != 0 &&
3868           st->as_Store()->memory_size() != size_in_bytes) {
3869         return FAIL;            // mismatched store size
3870       }
3871       return i;
3872     }
3873 
3874     ++i;
3875   }
3876 }

3878 // Look for a captured store which initializes at the offset 'start'
3879 // with the given size.  If there is no such store, and no other
3880 // initialization interferes, then return zero_memory (the memory
3881 // projection of the AllocateNode).
3882 Node* InitializeNode::find_captured_store(intptr_t start, int size_in_bytes,

3890   } else {

3892     assert(get_store_offset(st->as_Store(), phase) == start, "sanity");

3909 // Clone the given store, converting it into a raw store
3910 // initializing a field or element of my new object.
3911 // Caller is responsible for retiring the original store,
3912 // with subsume_node or the like.
3913 //
3914 // From the example above InitializeNode::InitializeNode,
3915 // here are the old stores to be captured:
3916 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3917 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3918 //
3919 // Here is the changed code; note the extra edges on init:
3920 //   alloc = (Allocate ...)
3921 //   rawoop = alloc.RawAddress
3922 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3923 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3924 //   init = (Initialize alloc.Control alloc.Memory rawoop
3925 //                      rawstore1 rawstore2)
3926 //
3927 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3928                                     PhaseGVN* phase, bool can_reshape) {
3929   assert(stores_are_sane(phase), "");
3930 
3931   if (start < 0)  return NULL;
3932   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3933 
3934   Compile* C = phase->C;
3935   int size_in_bytes = st->memory_size();
3936   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3937   if (i == 0)  return NULL;     // bail out
3938   Node* prev_mem = NULL;        // raw memory for the captured store
3939   if (i > 0) {
3940     prev_mem = in(i);           // there is a pre-existing store under this one
3941     set_req(i, C->top());       // temporarily disconnect it
3942     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3943   } else {
3944     i = -i;                     // no pre-existing store
3945     prev_mem = zero_memory();   // a slice of the newly allocated object
3946     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3947       set_req(--i, C->top());   // reuse this edge; it has been folded away
3948     else
3949       ins_req(i, C->top());     // build a new edge
3950   }
3951   Node* new_st = st->clone();
3952   new_st->set_req(MemNode::Control, in(Control));
3953   new_st->set_req(MemNode::Memory,  prev_mem);
3954   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3955   new_st = phase->transform(new_st);
3956 
3957   // At this point, new_st might have swallowed a pre-existing store
3958   // at the same offset, or perhaps new_st might have disappeared,
3959   // if it redundantly stored the same value (or zero to fresh memory).
3960 
3961   // In any case, wire it in:
3962   phase->igvn_rehash_node_delayed(this);
3963   set_req(i, new_st);
3964 
3965   // The caller may now kill the old guy.
3966   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3967   assert(check_st == new_st || check_st == NULL, "must be findable");
3968   assert(!is_complete(), "");
3969   return new_st;
3970 }

3972 static bool store_constant(jlong* tiles, int num_tiles,
3973                            intptr_t st_off, int st_size,
3974                            jlong con) {
3975   if ((st_off & (st_size-1)) != 0)
3976     return false;               // strange store offset (assume size==2**N)
3977   address addr = (address)tiles + st_off;
3978   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
3979   switch (st_size) {
3980   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
3981   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
3982   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
3983   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3984   default: return false;        // strange store size (detect size!=2**N here)
3985   }
3986   return true;                  // return success to caller
3987 }

3989 // Coalesce subword constants into int constants and possibly
3990 // into long constants.  The goal, if the CPU permits,
3991 // is to initialize the object with a small number of 64-bit tiles.
3992 // Also, convert floating-point constants to bit patterns.
3993 // Non-constants are not relevant to this pass.
3994 //
3995 // In terms of the running example on InitializeNode::InitializeNode
3996 // and InitializeNode::capture_store, here is the transformation
3997 // of rawstore1 and rawstore2 into rawstore12:
3998 //   alloc = (Allocate ...)
3999 //   rawoop = alloc.RawAddress
4000 //   tile12 = 0x00010002
4001 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4002 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4003 //
4004 void
4005 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4006                                         Node* size_in_bytes,
4007                                         PhaseGVN* phase) {
4008   Compile* C = phase->C;
4009 
4010   assert(stores_are_sane(phase), "");
4011   // Note:  After this pass, they are not completely sane,
4012   // since there may be some overlaps.
4013 
4014   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4015 
4016   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4017   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4018   size_limit = MIN2(size_limit, ti_limit);
4019   size_limit = align_up(size_limit, BytesPerLong);
4020   int num_tiles = size_limit / BytesPerLong;
4021 
4022   // allocate space for the tile map:
4023   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4024   jlong  tiles_buf[small_len];
4025   Node*  nodes_buf[small_len];
4026   jlong  inits_buf[small_len];
4027   jlong* tiles = ((num_tiles <= small_len) ? &tiles_buf[0]
4028                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4029   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
4030                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4031   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4032                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4033   // tiles: exact bitwise model of all primitive constants
4034   // nodes: last constant-storing node subsumed into the tiles model
4035   // inits: which bytes (in each tile) are touched by any initializations
4036 
4037   //// Pass A: Fill in the tile model with any relevant stores.
4038 
4039   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
4040   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
4041   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
4042   Node* zmem = zero_memory(); // initially zero memory state
4043   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4044     Node* st = in(i);
4045     intptr_t st_off = get_store_offset(st, phase);
4046 
4047     // Figure out the store's offset and constant value:
4048     if (st_off < header_size)             continue; //skip (ignore header)
4049     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
4050     int st_size = st->as_Store()->memory_size();
4051     if (st_off + st_size > size_limit)    break;
4052 
4053     // Record which bytes are touched, whether by constant or not.
4054     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
4055       continue;                 // skip (strange store size)
4056 
4057     const Type* val = phase->type(st->in(MemNode::ValueIn));
4058     if (!val->singleton())                continue; //skip (non-con store)
4059     BasicType type = val->basic_type();
4060 
4061     jlong con = 0;
4062     switch (type) {
4063     case T_INT:    con = val->is_int()->get_con();  break;
4064     case T_LONG:   con = val->is_long()->get_con(); break;
4065     case T_FLOAT:  con = jint_cast(val->getf());    break;
4066     case T_DOUBLE: con = jlong_cast(val->getd());   break;
4067     default:                              continue; //skip (odd store type)
4068     }
4069 
4070     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
4071         st->Opcode() == Op_StoreL) {
4072       continue;                 // This StoreL is already optimal.
4073     }
4074 
4075     // Store down the constant.
4076     store_constant(tiles, num_tiles, st_off, st_size, con);
4077 
4078     intptr_t j = st_off >> LogBytesPerLong;
4079 
4080     if (type == T_INT && st_size == BytesPerInt
4081         && (st_off & BytesPerInt) == BytesPerInt) {
4082       jlong lcon = tiles[j];
4083       if (!Matcher::isSimpleConstant64(lcon) &&
4084           st->Opcode() == Op_StoreI) {
4085         // This StoreI is already optimal by itself.
4086         jint* intcon = (jint*) &tiles[j];
4087         intcon[1] = 0;  // undo the store_constant()
4088 
4089         // If the previous store is also optimal by itself, back up and
4090         // undo the action of the previous loop iteration... if we can.
4091         // But if we can't, just let the previous half take care of itself.
4092         st = nodes[j];
4093         st_off -= BytesPerInt;
4094         con = intcon[0];
4095         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
4096           assert(st_off >= header_size, "still ignoring header");
4097           assert(get_store_offset(st, phase) == st_off, "must be");
4098           assert(in(i-1) == zmem, "must be");
4099           DEBUG_ONLY(const Type* tcon = phase->type(st->in(MemNode::ValueIn)));
4100           assert(con == tcon->is_int()->get_con(), "must be");
4101           // Undo the effects of the previous loop trip, which swallowed st:
4102           intcon[0] = 0;        // undo store_constant()
4103           set_req(i-1, st);     // undo set_req(i, zmem)
4104           nodes[j] = NULL;      // undo nodes[j] = st
4105           --old_subword;        // undo ++old_subword
4106         }
4107         continue;               // This StoreI is already optimal.
4108       }
4109     }
4110 
4111     // This store is not needed.
4112     set_req(i, zmem);
4113     nodes[j] = st;              // record for the moment
4114     if (st_size < BytesPerLong) // something has changed
4115           ++old_subword;        // includes int/float, but who's counting...
4116     else  ++old_long;
4117   }
4118 
4119   if ((old_subword + old_long) == 0)
4120     return;                     // nothing more to do
4121 
4122   //// Pass B: Convert any non-zero tiles into optimal constant stores.
4123   // Be sure to insert them before overlapping non-constant stores.
4124   // (E.g., byte[] x = { 1,2,y,4 }  =>  x[int 0] = 0x01020004, x[2]=y.)
4125   for (int j = 0; j < num_tiles; j++) {
4126     jlong con  = tiles[j];
4127     jlong init = inits[j];
4128     if (con == 0)  continue;
4129     jint con0,  con1;           // split the constant, address-wise
4130     jint init0, init1;          // split the init map, address-wise
4131     { union { jlong con; jint intcon[2]; } u;
4132       u.con = con;
4133       con0  = u.intcon[0];
4134       con1  = u.intcon[1];
4135       u.con = init;
4136       init0 = u.intcon[0];
4137       init1 = u.intcon[1];
4138     }
4139 
4140     Node* old = nodes[j];
4141     assert(old != NULL, "need the prior store");
4142     intptr_t offset = (j * BytesPerLong);
4143 
4144     bool split = !Matcher::isSimpleConstant64(con);
4145 
4146     if (offset < header_size) {
4147       assert(offset + BytesPerInt >= header_size, "second int counts");
4148       assert(*(jint*)&tiles[j] == 0, "junk in header");
4149       split = true;             // only the second word counts
4150       // Example:  int a[] = { 42 ... }
4151     } else if (con0 == 0 && init0 == -1) {
4152       split = true;             // first word is covered by full inits
4153       // Example:  int a[] = { ... foo(), 42 ... }
4154     } else if (con1 == 0 && init1 == -1) {
4155       split = true;             // second word is covered by full inits
4156       // Example:  int a[] = { ... 42, foo() ... }
4157     }
4158 
4159     // Here's a case where init0 is neither 0 nor -1:
4160     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4161     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4162     // In this case the tile is not split; it is (jlong)42.
4163     // The big tile is stored down, and then the foo() value is inserted.
4164     // (If there were foo(),foo() instead of foo(),0, init0 would be -1.)
4165 
4166     Node* ctl = old->in(MemNode::Control);
4167     Node* adr = make_raw_address(offset, phase);
4168     const TypePtr* atp = TypeRawPtr::BOTTOM;
4169 
4170     // One or two coalesced stores to plop down.
4171     Node*    st[2];
4172     intptr_t off[2];
4173     int  nst = 0;
4174     if (!split) {
4175       ++new_long;
4176       off[nst] = offset;

4178                                   phase->longcon(con), T_LONG, MemNode::unordered);
4179     } else {
4180       // Omit either if it is a zero.
4181       if (con0 != 0) {
4182         ++new_int;

4184         st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4185                                     phase->intcon(con0), T_INT, MemNode::unordered);
4186       }
4187       if (con1 != 0) {
4188         ++new_int;
4189         offset += BytesPerInt;

4191         off[nst]  = offset;

4193                                     phase->intcon(con1), T_INT, MemNode::unordered);
4194       }
4195     }

4198     // Insert each one just before any overlapping non-constant stores.
4199     while (nst > 0) {
4200       Node* st1 = st[--nst];

4202       st1 = phase->transform(st1);
4203       offset = off[nst];
4204       assert(offset >= header_size, "do not smash header");

4206       guarantee(ins_idx != 0, "must re-insert constant store");

4208       if (ins_idx > InitializeNode::RawStores && in(ins_idx-1) == zmem)

4220                    old_subword, old_long, new_int, new_long);

4226 // Explore forward from in(start) to find the first fully initialized
4227 // word, and return its offset.  Skip groups of subword stores which
4228 // together initialize full words.  If in(start) is itself part of a
4229 // fully initialized word, return the offset of in(start).  If there
4230 // are no following full-word stores, or if something is fishy, return
4231 // a negative value.
4232 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4233   int       int_map = 0;
4234   intptr_t  int_map_off = 0;
4235   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4236 
4237   for (uint i = start, limit = req(); i < limit; i++) {
4238     Node* st = in(i);
4239 
4240     intptr_t st_off = get_store_offset(st, phase);
4241     if (st_off < 0)  break;  // return conservative answer
4242 
4243     int st_size = st->as_Store()->memory_size();
4244     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4245       return st_off;            // we found a complete word init
4246     }
4247 
4248     // update the map:
4249 
4250     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4251     if (this_int_off != int_map_off) {
4252       // reset the map:
4253       int_map = 0;
4254       int_map_off = this_int_off;
4255     }
4256 
4257     int subword_off = st_off - this_int_off;
4258     int_map |= right_n_bits(st_size) << subword_off;
4259     if ((int_map & FULL_MAP) == FULL_MAP) {
4260       return this_int_off;      // we found a complete word init
4261     }
4262 
4263     // Did this store hit or cross the word boundary?
4264     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4265     if (next_int_off == this_int_off + BytesPerInt) {
4266       // We passed the current int, without fully initializing it.
4267       int_map_off = next_int_off;
4268       int_map >>= BytesPerInt;
4269     } else if (next_int_off > this_int_off + BytesPerInt) {
4270       // We passed the current and next int.
4271       return this_int_off + BytesPerInt;
4272     }
4273   }
4274 
4275   return -1;
4276 }

4279 // Called when the associated AllocateNode is expanded into CFG.
4280 // At this point, we may perform additional optimizations.
4281 // Linearize the stores by ascending offset, to make memory
4282 // activity as coherent as possible.
4283 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4284                                       intptr_t header_size,
4285                                       Node* size_in_bytes,
4286                                       PhaseIterGVN* phase) {
4287   assert(!is_complete(), "not already complete");
4288   assert(stores_are_sane(phase), "");
4289   assert(allocation() != NULL, "must be present");
4290 
4291   remove_extra_zeroes();
4292 
4293   if (ReduceFieldZeroing || ReduceBulkZeroing)
4294     // reduce instruction count for common initialization patterns
4295     coalesce_subword_stores(header_size, size_in_bytes, phase);
4296 
4297   Node* zmem = zero_memory();   // initially zero memory state
4298   Node* inits = zmem;           // accumulating a linearized chain of inits
4299   #ifdef ASSERT
4300   intptr_t first_offset = allocation()->minimum_header_size();
4301   intptr_t last_init_off = first_offset;  // previous init offset
4302   intptr_t last_init_end = first_offset;  // previous init offset+size
4303   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4304   #endif
4305   intptr_t zeroes_done = header_size;
4306 
4307   bool do_zeroing = true;       // we might give up if inits are very sparse
4308   int  big_init_gaps = 0;       // how many large gaps have we seen?
4309 
4310   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4311   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4312 
4313   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4314     Node* st = in(i);
4315     intptr_t st_off = get_store_offset(st, phase);
4316     if (st_off < 0)
4317       break;                    // unknown junk in the inits
4318     if (st->in(MemNode::Memory) != zmem)
4319       break;                    // complicated store chains somehow in list
4320 
4321     int st_size = st->as_Store()->memory_size();
4322     intptr_t next_init_off = st_off + st_size;
4323 
4324     if (do_zeroing && zeroes_done < next_init_off) {
4325       // See if this store needs a zero before it or under it.
4326       intptr_t zeroes_needed = st_off;
4327 
4328       if (st_size < BytesPerInt) {
4329         // Look for subword stores which only partially initialize words.
4330         // If we find some, we must lay down some word-level zeroes first,
4331         // underneath the subword stores.
4332         //
4333         // Examples:
4334         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4335         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4336         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4337         //
4338         // Note:  coalesce_subword_stores may have already done this,
4339         // if it was prompted by constant non-zero subword initializers.
4340         // But this case can still arise with non-constant stores.
4341 
4342         intptr_t next_full_store = find_next_fullword_store(i, phase);
4343 
4344         // In the examples above:
4345         //   in(i)          p   q   r   s     x   y     z
4346         //   st_off        12  13  14  15    12  13    14
4347         //   st_size        1   1   1   1     1   1     1
4348         //   next_full_s.  12  16  16  16    16  16    16
4349         //   z's_done      12  16  16  16    12  16    12
4350         //   z's_needed    12  16  16  16    16  16    16
4351         //   zsize          0   0   0   0     4   0     4
4352         if (next_full_store < 0) {
4353           // Conservative tack:  Zero to end of current word.
4354           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4355         } else {
4356           // Zero to beginning of next fully initialized word.
4357           // Or, don't zero at all, if we are already in that word.
4358           assert(next_full_store >= zeroes_needed, "must go forward");
4359           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4360           zeroes_needed = next_full_store;
4361         }
4362       }
4363 
4364       if (zeroes_needed > zeroes_done) {
4365         intptr_t zsize = zeroes_needed - zeroes_done;
4366         // Do some incremental zeroing on rawmem, in parallel with inits.
4367         zeroes_done = align_down(zeroes_done, BytesPerInt);
4368         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4369                                               allocation()->in(AllocateNode::DefaultValue),
4370                                               allocation()->in(AllocateNode::RawDefaultValue),
4371                                               zeroes_done, zeroes_needed,
4372                                               phase);
4373         zeroes_done = zeroes_needed;
4374         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4375           do_zeroing = false;   // leave the hole, next time
4376       }
4377     }
4378 
4379     // Collect the store and move on:
4380     phase->replace_input_of(st, MemNode::Memory, inits);
4381     inits = st;                 // put it on the linearized chain
4382     set_req(i, zmem);           // unhook from previous position
4383 
4384     if (zeroes_done == st_off)
4385       zeroes_done = next_init_off;
4386 
4387     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4388 
4389     #ifdef ASSERT
4390     // Various order invariants.  Weaker than stores_are_sane because
4391     // a large constant tile can be filled in by smaller non-constant stores.
4392     assert(st_off >= last_init_off, "inits do not reverse");
4393     last_init_off = st_off;
4394     const Type* val = NULL;
4395     if (st_size >= BytesPerInt &&
4396         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4397         (int)val->basic_type() < (int)T_OBJECT) {
4398       assert(st_off >= last_tile_end, "tiles do not overlap");
4399       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4400       last_tile_end = MAX2(last_tile_end, next_init_off);
4401     } else {
4402       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4403       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4404       assert(st_off      >= last_init_end, "inits do not overlap");
4405       last_init_end = next_init_off;  // it's a non-tile
4406     }
4407     #endif //ASSERT
4408   }
4409 
4410   remove_extra_zeroes();        // clear out all the zmems left over
4411   add_req(inits);
4412 
4413   if (!(UseTLAB && ZeroTLAB)) {
4414     // If anything remains to be zeroed, zero it all now.
4415     zeroes_done = align_down(zeroes_done, BytesPerInt);
4416     // if it is the last unused 4 bytes of an instance, forget about it
4417     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4418     if (zeroes_done + BytesPerLong >= size_limit) {
4419       AllocateNode* alloc = allocation();
4420       assert(alloc != NULL, "must be present");
4421       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4422         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4423         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4424         if (zeroes_done == k->layout_helper())
4425           zeroes_done = size_limit;
4426       }
4427     }
4428     if (zeroes_done < size_limit) {
4429       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4430                                             allocation()->in(AllocateNode::DefaultValue),
4431                                             allocation()->in(AllocateNode::RawDefaultValue),
4432                                             zeroes_done, size_in_bytes, phase);
4433     }
4434   }
4435 
4436   set_complete(phase);
4437   return rawmem;
4438 }

4466 //============================MergeMemNode=====================================
4467 //
4468 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4469 // contributing store or call operations.  Each contributor provides the memory
4470 // state for a particular "alias type" (see Compile::alias_type).  For example,
4471 // if a MergeMem has an input X for alias category #6, then any memory reference
4472 // to alias category #6 may use X as its memory state input, as an exact equivalent
4473 // to using the MergeMem as a whole.
4474 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4475 //
4476 // (Here, the <N> notation gives the index of the relevant adr_type.)
4477 //
4478 // In one special case (and more cases in the future), alias categories overlap.
4479 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4480 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4481 // it is exactly equivalent to that state W:
4482 //   MergeMem(<Bot>: W) <==> W
4483 //
4484 // Usually, the merge has more than one input.  In that case, where inputs
4485 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4486 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4487 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4488 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4489 //
4490 // A merge can take a "wide" memory state as one of its narrow inputs.
4491 // This simply means that the merge observes out only the relevant parts of
4492 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4493 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4494 //
4495 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4496 // and that memory slices "leak through":
4497 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4498 //
4499 // But, in such a cascade, repeated memory slices can "block the leak":
4500 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4501 //
4502 // In the last example, Y is not part of the combined memory state of the
4503 // outermost MergeMem.  The system must, of course, prevent unschedulable
4504 // memory states from arising, so you can be sure that the state Y is somehow
4505 // a precursor to state Y'.
4506 //
4507 //
4508 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4509 // of each MergeMemNode array are exactly the numerical alias indexes, including
4510 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4511 // Compile::alias_type (and kin) produce and manage these indexes.
4512 //
4513 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4514 // (Note that this provides quick access to the top node inside MergeMem methods,
4515 // without the need to reach out via TLS to Compile::current.)
4516 //
4517 // As a consequence of what was just described, a MergeMem that represents a full
4518 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4519 // containing all alias categories.
4520 //
4521 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4522 //
4523 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4524 // a memory state for the alias type <N>, or else the top node, meaning that
4525 // there is no particular input for that alias type.  Note that the length of
4526 // a MergeMem is variable, and may be extended at any time to accommodate new
4527 // memory states at larger alias indexes.  When merges grow, they are of course
4528 // filled with "top" in the unused in() positions.
4529 //
4530 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4531 // (Top was chosen because it works smoothly with passes like GCM.)
4532 //
4533 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4534 // the type of random VM bits like TLS references.)  Since it is always the
4535 // first non-Bot memory slice, some low-level loops use it to initialize an
4536 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4537 //
4538 //
4539 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4540 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4541 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4542 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4543 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4544 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4545 //
4546 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4547 // really that different from the other memory inputs.  An abbreviation called
4548 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4549 //
4550 //
4551 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4552 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4553 // that "emerges though" the base memory will be marked as excluding the alias types
4554 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4555 //
4556 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4557 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4558 //
4559 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4560 // (It is currently unimplemented.)  As you can see, the resulting merge is
4561 // actually a disjoint union of memory states, rather than an overlay.
4562 //

4624 //------------------------------Ideal------------------------------------------
4625 // This method is invoked recursively on chains of MergeMem nodes
4626 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4627   // Remove chain'd MergeMems
4628   //
4629   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4630   // relative to the "in(Bot)".  Since we are patching both at the same time,
4631   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4632   // but rewrite each "in(i)" relative to the new "in(Bot)".
4633   Node *progress = NULL;
4634 
4635 
4636   Node* old_base = base_memory();
4637   Node* empty_mem = empty_memory();
4638   if (old_base == empty_mem)
4639     return NULL; // Dead memory path.
4640 
4641   MergeMemNode* old_mbase;
4642   if (old_base != NULL && old_base->is_MergeMem())
4643     old_mbase = old_base->as_MergeMem();
4644   else
4645     old_mbase = NULL;
4646   Node* new_base = old_base;
4647 
4648   // simplify stacked MergeMems in base memory
4649   if (old_mbase)  new_base = old_mbase->base_memory();
4650 
4651   // the base memory might contribute new slices beyond my req()
4652   if (old_mbase)  grow_to_match(old_mbase);
4653 
4654   // Look carefully at the base node if it is a phi.
4655   PhiNode* phi_base;
4656   if (new_base != NULL && new_base->is_Phi())
4657     phi_base = new_base->as_Phi();
4658   else
4659     phi_base = NULL;
4660 
4661   Node*    phi_reg = NULL;
4662   uint     phi_len = (uint)-1;
4663   if (phi_base != NULL && !phi_base->is_copy()) {
4664     // do not examine phi if degraded to a copy
4665     phi_reg = phi_base->region();
4666     phi_len = phi_base->req();
4667     // see if the phi is unfinished
4668     for (uint i = 1; i < phi_len; i++) {
4669       if (phi_base->in(i) == NULL) {
4670         // incomplete phi; do not look at it yet!
4671         phi_reg = NULL;
4672         phi_len = (uint)-1;
4673         break;
4674       }
4675     }
4676   }
4677 
4678   // Note:  We do not call verify_sparse on entry, because inputs
4679   // can normalize to the base_memory via subsume_node or similar
4680   // mechanisms.  This method repairs that damage.
4681 
4682   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4683 
4684   // Look at each slice.
4685   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4686     Node* old_in = in(i);
4687     // calculate the old memory value
4688     Node* old_mem = old_in;
4689     if (old_mem == empty_mem)  old_mem = old_base;
4690     assert(old_mem == memory_at(i), "");
4691 
4692     // maybe update (reslice) the old memory value
4693 
4694     // simplify stacked MergeMems
4695     Node* new_mem = old_mem;
4696     MergeMemNode* old_mmem;
4697     if (old_mem != NULL && old_mem->is_MergeMem())
4698       old_mmem = old_mem->as_MergeMem();
4699     else
4700       old_mmem = NULL;
4701     if (old_mmem == this) {
4702       // This can happen if loops break up and safepoints disappear.
4703       // A merge of BotPtr (default) with a RawPtr memory derived from a
4704       // safepoint can be rewritten to a merge of the same BotPtr with
4705       // the BotPtr phi coming into the loop.  If that phi disappears
4706       // also, we can end up with a self-loop of the mergemem.
4707       // In general, if loops degenerate and memory effects disappear,
4708       // a mergemem can be left looking at itself.  This simply means
4709       // that the mergemem's default should be used, since there is
4710       // no longer any apparent effect on this slice.
4711       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4712       //       from start.  Update the input to TOP.
4713       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4714     }
4715     else if (old_mmem != NULL) {
4716       new_mem = old_mmem->memory_at(i);
4717     }
4718     // else preceding memory was not a MergeMem
4719 
4720     // replace equivalent phis (unfortunately, they do not GVN together)
4721     if (new_mem != NULL && new_mem != new_base &&
4722         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4723       if (new_mem->is_Phi()) {
4724         PhiNode* phi_mem = new_mem->as_Phi();
4725         for (uint i = 1; i < phi_len; i++) {
4726           if (phi_base->in(i) != phi_mem->in(i)) {
4727             phi_mem = NULL;
4728             break;
4729           }
4730         }
4731         if (phi_mem != NULL) {
4732           // equivalent phi nodes; revert to the def
4733           new_mem = new_base;
4734         }
4735       }
4736     }
4737 
4738     // maybe store down a new value
4739     Node* new_in = new_mem;
4740     if (new_in == new_base)  new_in = empty_mem;
4741 
4742     if (new_in != old_in) {
4743       // Warning:  Do not combine this "if" with the previous "if"
4744       // A memory slice might have be be rewritten even if it is semantically
4745       // unchanged, if the base_memory value has changed.
4746       set_req(i, new_in);
4747       progress = this;          // Report progress
4748     }
4749   }
4750 
4751   if (new_base != old_base) {
4752     set_req(Compile::AliasIdxBot, new_base);
4753     // Don't use set_base_memory(new_base), because we need to update du.
4754     assert(base_memory() == new_base, "");
4755     progress = this;
4756   }
4757 
4758   if( base_memory() == this ) {
4759     // a self cycle indicates this memory path is dead
4760     set_req(Compile::AliasIdxBot, empty_mem);
4761   }
4762 
4763   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4764   // Recursion must occur after the self cycle check above
4765   if( base_memory()->is_MergeMem() ) {
4766     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4767     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4768     if( m != NULL &&
4769         (m->is_top() ||
4770          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4771       // propagate rollup of dead cycle to self
4772       set_req(Compile::AliasIdxBot, empty_mem);
4773     }
4774   }
4775 
4776   if( base_memory() == empty_mem ) {
4777     progress = this;
4778     // Cut inputs during Parse phase only.
4779     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4780     if( !can_reshape ) {
4781       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4782         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4783       }
4784     }
4785   }
4786 
4787   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4788     // Check if PhiNode::Ideal's "Split phis through memory merges"
4789     // transform should be attempted. Look for this->phi->this cycle.
4790     uint merge_width = req();
4791     if (merge_width > Compile::AliasIdxRaw) {
4792       PhiNode* phi = base_memory()->as_Phi();
4793       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4794         if (phi->in(i) == this) {
4795           phase->is_IterGVN()->_worklist.push(phi);
4796           break;
4797         }
4798       }
4799     }
4800   }
4801 
4802   assert(progress || verify_sparse(), "please, no dups of base");
4803   return progress;
4804 }

4848 // verify a narrow slice (either incoming or outgoing)
4849 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4850   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4851   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4852   if (Node::in_dump())               return;  // muzzle asserts when printing
4853   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4854   assert(n != NULL, "");
4855   // Elide intervening MergeMem's
4856   while (n->is_MergeMem()) {
4857     n = n->as_MergeMem()->memory_at(alias_idx);
4858   }
4859   Compile* C = Compile::current();
4860   const TypePtr* n_adr_type = n->adr_type();
4861   if (n == m->empty_memory()) {
4862     // Implicit copy of base_memory()
4863   } else if (n_adr_type != TypePtr::BOTTOM) {
4864     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4865     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4866   } else {
4867     // A few places like make_runtime_call "know" that VM calls are narrow,
4868     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4869     bool expected_wide_mem = false;
4870     if (n == m->base_memory()) {
4871       expected_wide_mem = true;
4872     } else if (alias_idx == Compile::AliasIdxRaw ||
4873                n == m->memory_at(Compile::AliasIdxRaw)) {
4874       expected_wide_mem = true;
4875     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4876       // memory can "leak through" calls on channels that
4877       // are write-once.  Allow this also.
4878       expected_wide_mem = true;
4879     }
4880     assert(expected_wide_mem, "expected narrow slice replacement");
4881   }
4882 }

4889 Node* MergeMemNode::memory_at(uint alias_idx) const {
4890   assert(alias_idx >= Compile::AliasIdxRaw ||
4891          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4892          "must avoid base_memory and AliasIdxTop");
4893 
4894   // Otherwise, it is a narrow slice.
4895   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4896   Compile *C = Compile::current();
4897   if (is_empty_memory(n)) {
4898     // the array is sparse; empty slots are the "top" node
4899     n = base_memory();
4900     assert(Node::in_dump()
4901            || n == NULL || n->bottom_type() == Type::TOP
4902            || n->adr_type() == NULL // address is TOP
4903            || n->adr_type() == TypePtr::BOTTOM
4904            || n->adr_type() == TypeRawPtr::BOTTOM
4905            || Compile::current()->AliasLevel() == 0,
4906            "must be a wide memory");
4907     // AliasLevel == 0 if we are organizing the memory states manually.
4908     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.
4909   } else {
4910     // make sure the stored slice is sane
4911     #ifdef ASSERT
4912     if (VMError::is_error_reported() || Node::in_dump()) {
4913     } else if (might_be_same(n, base_memory())) {
4914       // Give it a pass:  It is a mostly harmless repetition of the base.
4915       // This can arise normally from node subsumption during optimization.
4916     } else {
4917       verify_memory_slice(this, alias_idx, n);
4918     }
4919     #endif
4920   }

4925 void MergeMemNode::set_memory_at(uint alias_idx, Node *n) {
