  97 // Return the index at which m must be inserted (or already exists).
  98 // The sort order is by the address of the ciMethod, with is_virtual as minor key.
  99 class IntrinsicDescPair {
 100  private:
 101   ciMethod* _m;
 102   bool _is_virtual;
 103  public:
 104   IntrinsicDescPair(ciMethod* m, bool is_virtual) : _m(m), _is_virtual(is_virtual) {}
 105   static int compare(IntrinsicDescPair* const& key, CallGenerator* const& elt) {
 106     ciMethod* m= elt->method();
 107     ciMethod* key_m = key->_m;
 108     if (key_m < m)      return -1;
 109     else if (key_m > m) return 1;
 110     else {
 111       bool is_virtual = elt->is_virtual();
 112       bool key_virtual = key->_is_virtual;
 113       if (key_virtual < is_virtual)      return -1;
 114       else if (key_virtual > is_virtual) return 1;
 115       else                               return 0;
 116     }
 117   }
 118 };
 119 int Compile::intrinsic_insertion_index(ciMethod* m, bool is_virtual, bool& found) {
 120 #ifdef ASSERT
 121   for (int i = 1; i < _intrinsics->length(); i++) {
 122     CallGenerator* cg1 = _intrinsics->at(i-1);
 123     CallGenerator* cg2 = _intrinsics->at(i);
 124     assert(cg1->method() != cg2->method()
 125            ? cg1->method()     < cg2->method()
 126            : cg1->is_virtual() < cg2->is_virtual(),
 127            "compiler intrinsics list must stay sorted");
 128   }
 129 #endif
 130   IntrinsicDescPair pair(m, is_virtual);
 131   return _intrinsics->find_sorted<IntrinsicDescPair*, IntrinsicDescPair::compare>(&pair, found);
 132 }

 134 void Compile::register_intrinsic(CallGenerator* cg) {
 135   if (_intrinsics == NULL) {
 136     _intrinsics = new (comp_arena())GrowableArray<CallGenerator*>(comp_arena(), 60, 0, NULL);
 137   }
 138   int len = _intrinsics->length();
 139   bool found = false;
 140   int index = intrinsic_insertion_index(cg->method(), cg->is_virtual(), found);
 141   assert(!found, "registering twice");
 142   _intrinsics->insert_before(index, cg);
 143   assert(find_intrinsic(cg->method(), cg->is_virtual()) == cg, "registration worked");
 144 }
 145 
 146 CallGenerator* Compile::find_intrinsic(ciMethod* m, bool is_virtual) {
 147   assert(m->is_loaded(), "don't try this on unloaded methods");
 148   if (_intrinsics != NULL) {
 149     bool found = false;
 150     int index = intrinsic_insertion_index(m, is_virtual, found);
 151      if (found) {
 152       return _intrinsics->at(index);
 153     }
 154   }
 155   // Lazily create intrinsics for intrinsic IDs well-known in the runtime.
 156   if (m->intrinsic_id() != vmIntrinsics::_none &&
 157       m->intrinsic_id() <= vmIntrinsics::LAST_COMPILER_INLINE) {
 158     CallGenerator* cg = make_vm_intrinsic(m, is_virtual);
 159     if (cg != NULL) {
 160       // Save it for next time:
 161       register_intrinsic(cg);
 162       return cg;
 163     } else {
 164       gather_intrinsic_statistics(m->intrinsic_id(), is_virtual, _intrinsic_disabled);
 165     }
 166   }
 167   return NULL;
 168 }

 177 juint  Compile::_intrinsic_hist_count[vmIntrinsics::ID_LIMIT] = {0};
 178 jubyte Compile::_intrinsic_hist_flags[vmIntrinsics::ID_LIMIT] = {0};
 179 
 180 bool Compile::gather_intrinsic_statistics(vmIntrinsics::ID id, bool is_virtual, int flags) {
 181   assert(id > vmIntrinsics::_none && id < vmIntrinsics::ID_LIMIT, "oob");
 182   int oflags = _intrinsic_hist_flags[id];
 183   assert(flags != 0, "what happened?");
 184   if (is_virtual) {
 185     flags |= _intrinsic_virtual;
 186   }
 187   bool changed = (flags != oflags);
 188   if ((flags & _intrinsic_worked) != 0) {
 189     juint count = (_intrinsic_hist_count[id] += 1);
 190     if (count == 1) {
 191       changed = true;           // first time
 192     }
 193     // increment the overall count also:
 194     _intrinsic_hist_count[vmIntrinsics::_none] += 1;
 195   }
 196   if (changed) {
 197     if (((oflags ^ flags) & _intrinsic_virtual) != 0) {
 198       // Something changed about the intrinsic's virtuality.
 199       if ((flags & _intrinsic_virtual) != 0) {
 200         // This is the first use of this intrinsic as a virtual call.
 201         if (oflags != 0) {
 202           // We already saw it as a non-virtual, so note both cases.
 203           flags |= _intrinsic_both;
 204         }
 205       } else if ((oflags & _intrinsic_both) == 0) {
 206         // This is the first use of this intrinsic as a non-virtual
 207         flags |= _intrinsic_both;
 208       }
 209     }
 210     _intrinsic_hist_flags[id] = (jubyte) (oflags | flags);
 211   }
 212   // update the overall flags also:
 213   _intrinsic_hist_flags[vmIntrinsics::_none] |= (jubyte) flags;
 214   return changed;
 215 }

 217 static char* format_flags(int flags, char* buf) {
 218   buf[0] = 0;
 219   if ((flags & Compile::_intrinsic_worked) != 0)    strcat(buf, ",worked");
 220   if ((flags & Compile::_intrinsic_failed) != 0)    strcat(buf, ",failed");
 221   if ((flags & Compile::_intrinsic_disabled) != 0)  strcat(buf, ",disabled");
 222   if ((flags & Compile::_intrinsic_virtual) != 0)   strcat(buf, ",virtual");
 223   if ((flags & Compile::_intrinsic_both) != 0)      strcat(buf, ",nonvirtual");
 224   if (buf[0] == 0)  strcat(buf, ",");
 225   assert(buf[0] == ',', "must be");
 226   return &buf[1];
 227 }

 233   tty->print_cr("Compiler intrinsic usage:");

 236   #define PRINT_STAT_LINE(name, c, f) \
 237     tty->print_cr("  %4d (%4.1f%%) %s (%s)", (int)(c), ((c) * 100.0) / total, name, f);

 245   }
 246   PRINT_STAT_LINE("total", total, format_flags(_intrinsic_hist_flags[vmIntrinsics::_none], flagsbuf));

 268 void Compile::gvn_replace_by(Node* n, Node* nn) {
 269   for (DUIterator_Last imin, i = n->last_outs(imin); i >= imin; ) {
 270     Node* use = n->last_out(i);
 271     bool is_in_table = initial_gvn()->hash_delete(use);
 272     uint uses_found = 0;
 273     for (uint j = 0; j < use->len(); j++) {
 274       if (use->in(j) == n) {
 275         if (j < use->req())
 276           use->set_req(j, nn);
 277         else
 278           use->set_prec(j, nn);
 279         uses_found++;
 280       }
 281     }
 282     if (is_in_table) {
 283       // reinsert into table
 284       initial_gvn()->hash_find_insert(use);
 285     }
 286     record_for_igvn(use);
 287     i -= uses_found;    // we deleted 1 or more copies of this edge
 288   }
 289 }

 292 static inline bool not_a_node(const Node* n) {
 293   if (n == NULL)                   return true;
 294   if (((intptr_t)n & 1) != 0)      return true;  // uninitialized, etc.
 295   if (*(address*)n == badAddress)  return true;  // kill by Node::destruct
 296   return false;
 297 }

 299 // Identify all nodes that are reachable from below, useful.
 300 // Use breadth-first pass that records state in a Unique_Node_List,
 301 // recursive traversal is slower.
 302 void Compile::identify_useful_nodes(Unique_Node_List &useful) {
 303   int estimated_worklist_size = live_nodes();
 304   useful.map( estimated_worklist_size, NULL );  // preallocate space
 305 
 306   // Initialize worklist
 307   if (root() != NULL)     { useful.push(root()); }
 308   // If 'top' is cached, declare it useful to preserve cached node
 309   if( cached_top_node() ) { useful.push(cached_top_node()); }
 310 
 311   // Push all useful nodes onto the list, breadthfirst
 312   for( uint next = 0; next < useful.size(); ++next ) {
 313     assert( next < unique(), "Unique useful nodes < total nodes");
 314     Node *n  = useful.at(next);
 315     uint max = n->len();
 316     for( uint i = 0; i < max; ++i ) {
 317       Node *m = n->in(i);
 318       if (not_a_node(m))  continue;
 319       useful.push(m);
 320     }
 321   }
 322 }

 324 // Update dead_node_list with any missing dead nodes using useful
 325 // list. Consider all non-useful nodes to be useless i.e., dead nodes.
 326 void Compile::update_dead_node_list(Unique_Node_List &useful) {
 327   uint max_idx = unique();
 328   VectorSet& useful_node_set = useful.member_set();
 329 
 330   for (uint node_idx = 0; node_idx < max_idx; node_idx++) {
 331     // If node with index node_idx is not in useful set,
 332     // mark it as dead in dead node list.
 333     if (!useful_node_set.test(node_idx)) {
 334       record_dead_node(node_idx);
 335     }
 336   }
 337 }

 339 void Compile::remove_useless_late_inlines(GrowableArray<CallGenerator*>* inlines, Unique_Node_List &useful) {
 340   int shift = 0;
 341   for (int i = 0; i < inlines->length(); i++) {
 342     CallGenerator* cg = inlines->at(i);
 343     CallNode* call = cg->call_node();
 344     if (shift > 0) {
 345       inlines->at_put(i-shift, cg);
 346     }
 347     if (!useful.member(call)) {
 348       shift++;
 349     }
 350   }
 351   inlines->trunc_to(inlines->length()-shift);
 352 }

 354 // Disconnect all useless nodes by disconnecting those at the boundary.
 355 void Compile::remove_useless_nodes(Unique_Node_List &useful) {
 356   uint next = 0;
 357   while (next < useful.size()) {
 358     Node *n = useful.at(next++);
 359     if (n->is_SafePoint()) {
 360       // We're done with a parsing phase. Replaced nodes are not valid
 361       // beyond that point.
 362       n->as_SafePoint()->delete_replaced_nodes();
 363     }
 364     // Use raw traversal of out edges since this code removes out edges
 365     int max = n->outcnt();
 366     for (int j = 0; j < max; ++j) {
 367       Node* child = n->raw_out(j);
 368       if (! useful.member(child)) {
 369         assert(!child->is_top() || child != top(),
 370                "If top is cached in Compile object it is in useful list");
 371         // Only need to remove this out-edge to the useless node
 372         n->raw_del_out(j);
 373         --j;
 374         --max;
 375       }
 376     }
 377     if (n->outcnt() == 1 && n->has_special_unique_user()) {
 378       record_for_igvn(n->unique_out());
 379     }
 380   }
 381   // Remove useless macro and predicate opaq nodes
 382   for (int i = C->macro_count()-1; i >= 0; i--) {
 383     Node* n = C->macro_node(i);
 384     if (!useful.member(n)) {
 385       remove_macro_node(n);
 386     }
 387   }
 388   // Remove useless CastII nodes with range check dependency
 389   for (int i = range_check_cast_count() - 1; i >= 0; i--) {
 390     Node* cast = range_check_cast_node(i);
 391     if (!useful.member(cast)) {
 392       remove_range_check_cast(cast);
 393     }
 394   }
 395   // Remove useless expensive nodes
 396   for (int i = C->expensive_count()-1; i >= 0; i--) {
 397     Node* n = C->expensive_node(i);
 398     if (!useful.member(n)) {
 399       remove_expensive_node(n);
 400     }
 401   }
 402   // Remove useless Opaque4 nodes
 403   for (int i = opaque4_count() - 1; i >= 0; i--) {
 404     Node* opaq = opaque4_node(i);
 405     if (!useful.member(opaq)) {
 406       remove_opaque4_node(opaq);
 407     }
 408   }
 409   // Remove useless inline type nodes
 410   for (int i = _inline_type_nodes->length() - 1; i >= 0; i--) {
 411     Node* vt = _inline_type_nodes->at(i);
 412     if (!useful.member(vt)) {
 413       _inline_type_nodes->remove(vt);
 414     }
 415   }
 416   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 417   bs->eliminate_useless_gc_barriers(useful, this);
 418   // clean up the late inline lists
 419   remove_useless_late_inlines(&_string_late_inlines, useful);
 420   remove_useless_late_inlines(&_boxing_late_inlines, useful);
 421   remove_useless_late_inlines(&_late_inlines, useful);
 422   debug_only(verify_graph_edges(true/*check for no_dead_code*/);)
 423 }

 582   // We can always print a disassembly, either abstract (hex dump) or
 583   // with the help of a suitable hsdis library. Thus, we should not
 584   // couple print_assembly and print_opto_assembly controls.
 585   // But: always print opto and regular assembly on compile command 'print'.
 586   bool print_assembly = directive->PrintAssemblyOption;
 587   set_print_assembly(print_opto_assembly || print_assembly);
 588 #else
 589   set_print_assembly(false); // must initialize.
 590 #endif
 591 
 592 #ifndef PRODUCT
 593   set_parsed_irreducible_loop(false);
 594 
 595   if (directive->ReplayInlineOption) {
 596     _replay_inline_data = ciReplay::load_inline_data(method(), entry_bci(), ci_env->comp_level());
 597   }
 598 #endif
 599   set_print_inlining(directive->PrintInliningOption || PrintOptoInlining);
 600   set_print_intrinsics(directive->PrintIntrinsicsOption);
 601   set_has_irreducible_loop(true); // conservative until build_loop_tree() reset it
 602 
 603   if (ProfileTraps RTM_OPT_ONLY( || UseRTMLocking )) {
 604     // Make sure the method being compiled gets its own MDO,
 605     // so we can at least track the decompile_count().
 606     // Need MDO to record RTM code generation state.
 607     method()->ensure_method_data();
 608   }
 609 
 610   Init(::AliasLevel);
 611 
 612 
 613   print_compile_messages();
 614 
 615   _ilt = InlineTree::build_inline_tree_root();
 616 
 617   // Even if NO memory addresses are used, MergeMem nodes must have at least 1 slice
 618   assert(num_alias_types() >= AliasIdxRaw, "");
 619 
 620 #define MINIMUM_NODE_HASH  1023
 621   // Node list that Iterative GVN will start with
 622   Unique_Node_List for_igvn(comp_arena());
 623   set_for_igvn(&for_igvn);
 624 
 625   // GVN that will be run immediately on new nodes
 626   uint estimated_size = method()->code_size()*4+64;
 627   estimated_size = (estimated_size < MINIMUM_NODE_HASH ? MINIMUM_NODE_HASH : estimated_size);
 628   PhaseGVN gvn(node_arena(), estimated_size);
 629   set_initial_gvn(&gvn);
 630 
 631   print_inlining_init();
 632   { // Scope for timing the parser
 633     TracePhase tp("parse", &timers[_t_parser]);
 634 
 635     // Put top into the hash table ASAP.
 636     initial_gvn()->transform_no_reclaim(top());
 637 
 638     // Set up tf(), start(), and find a CallGenerator.
 639     CallGenerator* cg = NULL;
 640     if (is_osr_compilation()) {
 641       init_tf(TypeFunc::make(method(), /* is_osr_compilation = */ true));
 642       StartNode* s = new StartOSRNode(root(), tf()->domain_sig());
 643       initial_gvn()->set_type_bottom(s);
 644       init_start(s);
 645       cg = CallGenerator::for_osr(method(), entry_bci());
 646     } else {
 647       // Normal case.
 648       init_tf(TypeFunc::make(method()));
 649       StartNode* s = new StartNode(root(), tf()->domain_cc());
 650       initial_gvn()->set_type_bottom(s);
 651       init_start(s);
 652       if (method()->intrinsic_id() == vmIntrinsics::_Reference_get) {
 653         // With java.lang.ref.reference.get() we must go through the
 654         // intrinsic - even when get() is the root
 655         // method of the compile - so that, if necessary, the value in
 656         // the referent field of the reference object gets recorded by
 657         // the pre-barrier code.
 658         cg = find_intrinsic(method(), false);
 659       }
 660       if (cg == NULL) {
 661         float past_uses = method()->interpreter_invocation_count();
 662         float expected_uses = past_uses;
 663         cg = CallGenerator::for_inline(method(), expected_uses);
 664       }
 665     }
 666     if (failing())  return;
 667     if (cg == NULL) {
 668       record_method_not_compilable("cannot parse method");
 669       return;
 670     }
 671     JVMState* jvms = build_start_state(start(), tf());
 672     if ((jvms = cg->generate(jvms)) == NULL) {
 673       if (!failure_reason_is(C2Compiler::retry_class_loading_during_parsing())) {
 674         record_method_not_compilable("method parse failed");
 675       }
 676       return;
 677     }
 678     GraphKit kit(jvms);
 679 
 680     if (!kit.stopped()) {
 681       // Accept return values, and transfer control we know not where.
 682       // This is done by a special, unique ReturnNode bound to root.
 683       return_values(kit.jvms());
 684     }
 685 
 686     if (kit.has_exceptions()) {
 687       // Any exceptions that escape from this call must be rethrown
 688       // to whatever caller is dynamically above us on the stack.
 689       // This is done by a special, unique RethrowNode bound to root.
 690       rethrow_exceptions(kit.transfer_exceptions_into_jvms());
 691     }
 692 
 693     assert(IncrementalInline || (_late_inlines.length() == 0 && !has_mh_late_inlines()), "incremental inlining is off");
 694 
 695     if (_late_inlines.length() == 0 && !has_mh_late_inlines() && !failing() && has_stringbuilder()) {
 696       inline_string_calls(true);
 697     }
 698 
 699     if (failing())  return;
 700 
 701     print_method(PHASE_BEFORE_REMOVEUSELESS, 3);
 702 
 703     // Remove clutter produced by parsing.
 704     if (!failing()) {
 705       ResourceMark rm;
 706       PhaseRemoveUseless pru(initial_gvn(), &for_igvn);
 707     }
 708   }
 709 
 710   // Note:  Large methods are capped off in do_one_bytecode().
 711   if (failing())  return;
 712 
 713   // After parsing, node notes are no longer automagic.
 714   // They must be propagated by register_new_node_with_optimizer(),
 715   // clone(), or the like.
 716   set_default_node_notes(NULL);
 717 
 718   for (;;) {
 719     int successes = Inline_Warm();
 720     if (failing())  return;
 721     if (successes == 0)  break;
 722   }
 723 
 724   // Drain the list.
 725   Finish_Warm();
 726 #ifndef PRODUCT
 727   if (should_print(1)) {
 728     _printer->print_inlining();
 729   }
 730 #endif
 731 
 732   if (failing())  return;
 733   NOT_PRODUCT( verify_graph_edges(); )
 734 
 735   // Now optimize
 736   Optimize();
 737   if (failing())  return;
 738   NOT_PRODUCT( verify_graph_edges(); )
 739 
 740 #ifndef PRODUCT
 741   if (print_ideal()) {
 742     ttyLocker ttyl;  // keep the following output all in one block
 743     // This output goes directly to the tty, not the compiler log.
 744     // To enable tools to match it up with the compilation activity,
 745     // be sure to tag this tty output with the compile ID.
 746     if (xtty != NULL) {
 747       xtty->head("ideal compile_id='%d'%s", compile_id(),
 748                  is_osr_compilation()    ? " compile_kind='osr'" :
 749                  "");
 750     }
 751     root()->dump(9999);
 752     if (xtty != NULL) {
 753       xtty->tail("ideal");
 754     }
 755   }
 756 #endif
 757 
 758 #ifdef ASSERT
 759   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 760   bs->verify_gc_barriers(this, BarrierSetC2::BeforeCodeGen);
 761 #endif
 762 
 763   // Dump compilation data to replay it.
 764   if (directive->DumpReplayOption) {
 765     env()->dump_replay_data(_compile_id);
 766   }
 767   if (directive->DumpInlineOption && (ilt() != NULL)) {
 768     env()->dump_inline_data(_compile_id);
 769   }
 770 
 771   // Now that we know the size of all the monitors we can add a fixed slot
 772   // for the original deopt pc.
 773   int next_slot = fixed_slots() + (sizeof(address) / VMRegImpl::stack_slot_size);
 774   if (needs_stack_repair()) {
 775     // One extra slot for the special stack increment value
 776     next_slot += 2;
 777   }
 778   set_fixed_slots(next_slot);
 779 
 780   // Compute when to use implicit null checks. Used by matching trap based
 781   // nodes and NullCheck optimization.
 782   set_allowed_deopt_reasons();
 783 
 784   // Now generate code
 785   Code_Gen();
 786 }

1033 /**
1034  * Return the 'StartNode'. We must not have a pending failure, since the ideal graph
1035  * can be in an inconsistent state, i.e., we can get segmentation faults when traversing
1036  * the ideal graph.
1037  */
1038 StartNode* Compile::start() const {
1039   assert (!failing(), "Must not have pending failure. Reason is: %s", failure_reason());
1040   for (DUIterator_Fast imax, i = root()->fast_outs(imax); i < imax; i++) {
1041     Node* start = root()->fast_out(i);
1042     if (start->is_Start()) {
1043       return start->as_Start();
1044     }
1045   }
1046   fatal("Did not find Start node!");
1047   return NULL;
1048 }

1089 void Compile::print_missing_nodes() {
1090 
1091   // Return if CompileLog is NULL and PrintIdealNodeCount is false.
1092   if ((_log == NULL) && (! PrintIdealNodeCount)) {
1093     return;
1094   }
1095 
1096   // This is an expensive function. It is executed only when the user
1097   // specifies VerifyIdealNodeCount option or otherwise knows the
1098   // additional work that needs to be done to identify reachable nodes
1099   // by walking the flow graph and find the missing ones using
1100   // _dead_node_list.
1101 
1102   Unique_Node_List useful(comp_arena());
1103   // Get useful node list by walking the graph.
1104   identify_useful_nodes(useful);
1105 
1106   uint l_nodes = C->live_nodes();
1107   uint l_nodes_by_walk = useful.size();
1108 
1109   if (l_nodes != l_nodes_by_walk) {
1110     if (_log != NULL) {
1111       _log->begin_head("mismatched_nodes count='%d'", abs((int) (l_nodes - l_nodes_by_walk)));
1112       _log->stamp();
1113       _log->end_head();
1114     }
1115     VectorSet& useful_member_set = useful.member_set();
1116     int last_idx = l_nodes_by_walk;
1117     for (int i = 0; i < last_idx; i++) {
1118       if (useful_member_set.test(i)) {
1119         if (_dead_node_list.test(i)) {
1120           if (_log != NULL) {
1121             _log->elem("mismatched_node_info node_idx='%d' type='both live and dead'", i);
1122           }
1123           if (PrintIdealNodeCount) {
1124             // Print the log message to tty
1125               tty->print_cr("mismatched_node idx='%d' both live and dead'", i);
1126               useful.at(i)->dump();
1127           }
1128         }
1129       }
1130       else if (! _dead_node_list.test(i)) {
1131         if (_log != NULL) {
1132           _log->elem("mismatched_node_info node_idx='%d' type='neither live nor dead'", i);
1133         }
1134         if (PrintIdealNodeCount) {
1135           // Print the log message to tty
1136           tty->print_cr("mismatched_node idx='%d' type='neither live nor dead'", i);

1235 //------------------------------flatten_alias_type-----------------------------
1236 const TypePtr *Compile::flatten_alias_type( const TypePtr *tj ) const {
1237   int offset = tj->offset();
1238   TypePtr::PTR ptr = tj->ptr();
1239 
1240   // Known instance (scalarizable allocation) alias only with itself.
1241   bool is_known_inst = tj->isa_oopptr() != NULL &&
1242                        tj->is_oopptr()->is_known_instance();
1243 
1244   // Process weird unsafe references.
1245   if (offset == Type::OffsetBot && (tj->isa_instptr() /*|| tj->isa_klassptr()*/)) {
1246     bool default_value_load = EnableValhalla && tj->is_instptr()->klass() == ciEnv::current()->Class_klass();
1247     assert(InlineUnsafeOps || default_value_load, "indeterminate pointers come only from unsafe ops");
1248     assert(!is_known_inst, "scalarizable allocation should not have unsafe references");
1249     tj = TypeOopPtr::BOTTOM;
1250     ptr = tj->ptr();
1251     offset = tj->offset();
1252   }
1253 
1254   // Array pointers need some flattening
1255   const TypeAryPtr *ta = tj->isa_aryptr();
1256   if (ta && ta->is_stable()) {
1257     // Erase stability property for alias analysis.
1258     tj = ta = ta->cast_to_stable(false);
1259   }
1260   if (ta && ta->is_not_flat()) {
1261     // Erase not flat property for alias analysis.
1262     tj = ta = ta->cast_to_not_flat(false);
1263   }
1264   if (ta && ta->is_not_null_free()) {
1265     // Erase not null free property for alias analysis.
1266     tj = ta = ta->cast_to_not_null_free(false);
1267   }
1268 
1269   if( ta && is_known_inst ) {
1270     if ( offset != Type::OffsetBot &&
1271          offset > arrayOopDesc::length_offset_in_bytes() ) {
1272       offset = Type::OffsetBot; // Flatten constant access into array body only
1273       tj = ta = TypeAryPtr::make(ptr, ta->ary(), ta->klass(), true, Type::Offset(offset), ta->field_offset(), ta->instance_id());
1274     }
1275   } else if( ta && _AliasLevel >= 2 ) {
1276     // For arrays indexed by constant indices, we flatten the alias
1277     // space to include all of the array body.  Only the header, klass
1278     // and array length can be accessed un-aliased.
1279     // For flattened inline type array, each field has its own slice so
1280     // we must include the field offset.
1281     if( offset != Type::OffsetBot ) {
1282       if( ta->const_oop() ) { // MethodData* or Method*
1283         offset = Type::OffsetBot;   // Flatten constant access into array body
1284         tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1285       } else if( offset == arrayOopDesc::length_offset_in_bytes() ) {
1286         // range is OK as-is.
1287         tj = ta = TypeAryPtr::RANGE;
1288       } else if( offset == oopDesc::klass_offset_in_bytes() ) {
1289         tj = TypeInstPtr::KLASS; // all klass loads look alike
1290         ta = TypeAryPtr::RANGE; // generic ignored junk
1291         ptr = TypePtr::BotPTR;
1292       } else if( offset == oopDesc::mark_offset_in_bytes() ) {
1293         tj = TypeInstPtr::MARK;
1294         ta = TypeAryPtr::RANGE; // generic ignored junk
1295         ptr = TypePtr::BotPTR;
1296       } else {                  // Random constant offset into array body
1297         offset = Type::OffsetBot;   // Flatten constant access into array body
1298         tj = ta = TypeAryPtr::make(ptr,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1299       }
1300     }
1301     // Arrays of fixed size alias with arrays of unknown size.
1302     if (ta->size() != TypeInt::POS) {
1303       const TypeAry *tary = TypeAry::make(ta->elem(), TypeInt::POS);
1304       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,ta->klass(),false,Type::Offset(offset), ta->field_offset());
1305     }
1306     // Arrays of known objects become arrays of unknown objects.
1307     if (ta->elem()->isa_narrowoop() && ta->elem() != TypeNarrowOop::BOTTOM) {
1308       const TypeAry *tary = TypeAry::make(TypeNarrowOop::BOTTOM, ta->size());
1309       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1310     }
1311     if (ta->elem()->isa_oopptr() && ta->elem() != TypeInstPtr::BOTTOM) {
1312       const TypeAry *tary = TypeAry::make(TypeInstPtr::BOTTOM, ta->size());
1313       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1314     }
1315     // Initially all flattened array accesses share a single slice
1316     if (ta->is_flat() && ta->elem() != TypeInlineType::BOTTOM && _flattened_accesses_share_alias) {
1317       const TypeAry *tary = TypeAry::make(TypeInlineType::BOTTOM, ta->size());
1318       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), Type::Offset(Type::OffsetBot));
1319     }
1320     // Arrays of bytes and of booleans both use 'bastore' and 'baload' so
1321     // cannot be distinguished by bytecode alone.
1322     if (ta->elem() == TypeInt::BOOL) {
1323       const TypeAry *tary = TypeAry::make(TypeInt::BYTE, ta->size());
1324       ciKlass* aklass = ciTypeArrayKlass::make(T_BYTE);
1325       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,aklass,false,Type::Offset(offset), ta->field_offset());
1326     }
1327     // During the 2nd round of IterGVN, NotNull castings are removed.
1328     // Make sure the Bottom and NotNull variants alias the same.
1329     // Also, make sure exact and non-exact variants alias the same.
1330     if (ptr == TypePtr::NotNull || ta->klass_is_exact() || ta->speculative() != NULL) {
1331       tj = ta = TypeAryPtr::make(TypePtr::BotPTR,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1332     }
1333   }
1334 
1335   // Oop pointers need some flattening
1336   const TypeInstPtr *to = tj->isa_instptr();
1337   if( to && _AliasLevel >= 2 && to != TypeOopPtr::BOTTOM ) {
1338     ciInstanceKlass *k = to->klass()->as_instance_klass();
1339     if( ptr == TypePtr::Constant ) {
1340       if (to->klass() != ciEnv::current()->Class_klass() ||
1341           offset < k->size_helper() * wordSize) {
1342         // No constant oop pointers (such as Strings); they alias with
1343         // unknown strings.
1344         assert(!is_known_inst, "not scalarizable allocation");
1345         tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset));
1346       }
1347     } else if( is_known_inst ) {
1348       tj = to; // Keep NotNull and klass_is_exact for instance type
1349     } else if( ptr == TypePtr::NotNull || to->klass_is_exact() ) {
1350       // During the 2nd round of IterGVN, NotNull castings are removed.
1351       // Make sure the Bottom and NotNull variants alias the same.
1352       // Also, make sure exact and non-exact variants alias the same.
1353       tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset));
1354     }
1355     if (to->speculative() != NULL) {
1356       tj = to = TypeInstPtr::make(to->ptr(),to->klass(),to->klass_is_exact(),to->const_oop(),Type::Offset(to->offset()), to->klass()->flatten_array(), to->instance_id());
1357     }
1358     // Canonicalize the holder of this field
1359     if (offset >= 0 && offset < instanceOopDesc::base_offset_in_bytes()) {
1360       // First handle header references such as a LoadKlassNode, even if the
1361       // object's klass is unloaded at compile time (4965979).
1362       if (!is_known_inst) { // Do it only for non-instance types
1363         tj = to = TypeInstPtr::make(TypePtr::BotPTR, env()->Object_klass(), false, NULL, Type::Offset(offset));
1364       }
1365     } else if (offset < 0 || offset >= k->size_helper() * wordSize) {
1366       // Static fields are in the space above the normal instance
1367       // fields in the java.lang.Class instance.
1368       if (to->klass() != ciEnv::current()->Class_klass()) {
1369         to = NULL;
1370         tj = TypeOopPtr::BOTTOM;
1371         offset = tj->offset();
1372       }
1373     } else {
1374       ciInstanceKlass *canonical_holder = k->get_canonical_holder(offset);
1375       if (!k->equals(canonical_holder) || tj->offset() != offset) {
1376         if( is_known_inst ) {
1377           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, true, NULL, Type::Offset(offset), canonical_holder->flatten_array(), to->instance_id());
1378         } else {
1379           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, false, NULL, Type::Offset(offset));
1380         }
1381       }
1382     }
1383   }
1384 
1385   // Klass pointers to object array klasses need some flattening
1386   const TypeKlassPtr *tk = tj->isa_klassptr();
1387   if( tk ) {
1388     // If we are referencing a field within a Klass, we need
1389     // to assume the worst case of an Object.  Both exact and
1390     // inexact types must flatten to the same alias class so
1391     // use NotNull as the PTR.
1392     if ( offset == Type::OffsetBot || (offset >= 0 && (size_t)offset < sizeof(Klass)) ) {
1393 
1394       tj = tk = TypeKlassPtr::make(TypePtr::NotNull,
1395                                    TypeKlassPtr::OBJECT->klass(),
1396                                    Type::Offset(offset));

1397     }
1398 
1399     ciKlass* klass = tk->klass();
1400     if (klass != NULL && klass->is_obj_array_klass()) {
1401       ciKlass* k = TypeAryPtr::OOPS->klass();
1402       if( !k || !k->is_loaded() )                  // Only fails for some -Xcomp runs
1403         k = TypeInstPtr::BOTTOM->klass();
1404       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, k, Type::Offset(offset));
1405     }
1406 
1407     // Check for precise loads from the primary supertype array and force them
1408     // to the supertype cache alias index.  Check for generic array loads from
1409     // the primary supertype array and also force them to the supertype cache
1410     // alias index.  Since the same load can reach both, we need to merge
1411     // these 2 disparate memories into the same alias class.  Since the
1412     // primary supertype array is read-only, there's no chance of confusion
1413     // where we bypass an array load and an array store.
1414     int primary_supers_offset = in_bytes(Klass::primary_supers_offset());
1415     if (offset == Type::OffsetBot ||
1416         (offset >= primary_supers_offset &&
1417          offset < (int)(primary_supers_offset + Klass::primary_super_limit() * wordSize)) ||
1418         offset == (int)in_bytes(Klass::secondary_super_cache_offset())) {
1419       offset = in_bytes(Klass::secondary_super_cache_offset());
1420       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, tk->klass(), Type::Offset(offset));
1421     }
1422   }
1423 
1424   // Flatten all Raw pointers together.
1425   if (tj->base() == Type::RawPtr)
1426     tj = TypeRawPtr::BOTTOM;
1427 
1428   if (tj->base() == Type::AnyPtr)
1429     tj = TypePtr::BOTTOM;      // An error, which the caller must check for.
1430 
1431   // Flatten all to bottom for now
1432   switch( _AliasLevel ) {
1433   case 0:
1434     tj = TypePtr::BOTTOM;
1435     break;
1436   case 1:                       // Flatten to: oop, static, field or array
1437     switch (tj->base()) {
1438     //case Type::AryPtr: tj = TypeAryPtr::RANGE;    break;
1439     case Type::RawPtr:   tj = TypeRawPtr::BOTTOM;   break;
1440     case Type::AryPtr:   // do not distinguish arrays at all
1441     case Type::InstPtr:  tj = TypeInstPtr::BOTTOM;  break;
1442     case Type::KlassPtr: tj = TypeKlassPtr::OBJECT; break;
1443     case Type::AnyPtr:   tj = TypePtr::BOTTOM;      break;  // caller checks it
1444     default: ShouldNotReachHere();
1445     }
1446     break;
1447   case 2:                       // No collapsing at level 2; keep all splits
1448   case 3:                       // No collapsing at level 3; keep all splits
1449     break;
1450   default:
1451     Unimplemented();
1452   }
1453 
1454   offset = tj->offset();
1455   assert( offset != Type::OffsetTop, "Offset has fallen from constant" );
1456 
1457   assert( (offset != Type::OffsetBot && tj->base() != Type::AryPtr) ||
1458           (offset == Type::OffsetBot && tj->base() == Type::AryPtr) ||
1459           (offset == Type::OffsetBot && tj == TypeOopPtr::BOTTOM) ||
1460           (offset == Type::OffsetBot && tj == TypePtr::BOTTOM) ||
1461           (offset == oopDesc::mark_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1462           (offset == oopDesc::klass_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1463           (offset == arrayOopDesc::length_offset_in_bytes() && tj->base() == Type::AryPtr),
1464           "For oops, klasses, raw offset must be constant; for arrays the offset is never known" );
1465   assert( tj->ptr() != TypePtr::TopPTR &&
1466           tj->ptr() != TypePtr::AnyNull &&
1467           tj->ptr() != TypePtr::Null, "No imprecise addresses" );
1468 //    assert( tj->ptr() != TypePtr::Constant ||
1469 //            tj->base() == Type::RawPtr ||
1470 //            tj->base() == Type::KlassPtr, "No constant oop addresses" );
1471 
1472   return tj;
1473 }

1558 //--------------------------------find_alias_type------------------------------
1559 Compile::AliasType* Compile::find_alias_type(const TypePtr* adr_type, bool no_create, ciField* original_field, bool uncached) {
1560   if (_AliasLevel == 0)
1561     return alias_type(AliasIdxBot);
1562 
1563   AliasCacheEntry* ace = NULL;
1564   if (!uncached) {
1565     ace = probe_alias_cache(adr_type);
1566     if (ace->_adr_type == adr_type) {
1567       return alias_type(ace->_index);
1568     }
1569   }
1570 
1571   // Handle special cases.
1572   if (adr_type == NULL)             return alias_type(AliasIdxTop);
1573   if (adr_type == TypePtr::BOTTOM)  return alias_type(AliasIdxBot);
1574 
1575   // Do it the slow way.
1576   const TypePtr* flat = flatten_alias_type(adr_type);
1577 
1578 #ifdef ASSERT
1579   {
1580     ResourceMark rm;
1581     assert(flat == flatten_alias_type(flat), "not idempotent: adr_type = %s; flat = %s => %s",
1582            Type::str(adr_type), Type::str(flat), Type::str(flatten_alias_type(flat)));
1583     assert(flat != TypePtr::BOTTOM, "cannot alias-analyze an untyped ptr: adr_type = %s",
1584            Type::str(adr_type));
1585     if (flat->isa_oopptr() && !flat->isa_klassptr()) {
1586       const TypeOopPtr* foop = flat->is_oopptr();
1587       // Scalarizable allocations have exact klass always.
1588       bool exact = !foop->klass_is_exact() || foop->is_known_instance();
1589       const TypePtr* xoop = foop->cast_to_exactness(exact)->is_ptr();
1590       assert(foop == flatten_alias_type(xoop), "exactness must not affect alias type: foop = %s; xoop = %s",
1591              Type::str(foop), Type::str(xoop));
1592     }
1593   }
1594 #endif
1595 
1596   int idx = AliasIdxTop;
1597   for (int i = 0; i < num_alias_types(); i++) {
1598     if (alias_type(i)->adr_type() == flat) {
1599       idx = i;
1600       break;
1601     }
1602   }
1603 
1604   if (idx == AliasIdxTop) {
1605     if (no_create)  return NULL;
1606     // Grow the array if necessary.
1607     if (_num_alias_types == _max_alias_types)  grow_alias_types();
1608     // Add a new alias type.
1609     idx = _num_alias_types++;
1610     _alias_types[idx]->Init(idx, flat);
1611     if (flat == TypeInstPtr::KLASS)  alias_type(idx)->set_rewritable(false);
1612     if (flat == TypeAryPtr::RANGE)   alias_type(idx)->set_rewritable(false);
1613     if (flat->isa_instptr()) {
1614       if (flat->offset() == java_lang_Class::klass_offset()
1615           && flat->is_instptr()->klass() == env()->Class_klass())
1616         alias_type(idx)->set_rewritable(false);
1617     }
1618     ciField* field = NULL;
1619     if (flat->isa_aryptr()) {
1620 #ifdef ASSERT
1621       const int header_size_min  = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1622       // (T_BYTE has the weakest alignment and size restrictions...)
1623       assert(flat->offset() < header_size_min, "array body reference must be OffsetBot");
1624 #endif
1625       const Type* elemtype = flat->is_aryptr()->elem();
1626       if (flat->offset() == TypePtr::OffsetBot) {
1627         alias_type(idx)->set_element(elemtype);
1628       }
1629       int field_offset = flat->is_aryptr()->field_offset().get();
1630       if (elemtype->isa_inlinetype() &&
1631           elemtype->inline_klass() != NULL &&
1632           field_offset != Type::OffsetBot) {
1633         ciInlineKlass* vk = elemtype->inline_klass();
1634         field_offset += vk->first_field_offset();
1635         field = vk->get_field_by_offset(field_offset, false);
1636       }
1637     }
1638     if (flat->isa_klassptr()) {
1639       if (flat->offset() == in_bytes(Klass::super_check_offset_offset()))
1640         alias_type(idx)->set_rewritable(false);
1641       if (flat->offset() == in_bytes(Klass::modifier_flags_offset()))
1642         alias_type(idx)->set_rewritable(false);
1643       if (flat->offset() == in_bytes(Klass::access_flags_offset()))
1644         alias_type(idx)->set_rewritable(false);
1645       if (flat->offset() == in_bytes(Klass::java_mirror_offset()))
1646         alias_type(idx)->set_rewritable(false);
1647       if (flat->offset() == in_bytes(Klass::layout_helper_offset()))
1648         alias_type(idx)->set_rewritable(false);
1649       if (flat->offset() == in_bytes(Klass::secondary_super_cache_offset()))
1650         alias_type(idx)->set_rewritable(false);
1651     }
1652     // %%% (We would like to finalize JavaThread::threadObj_offset(),
1653     // but the base pointer type is not distinctive enough to identify
1654     // references into JavaThread.)
1655 
1656     // Check for final fields.
1657     const TypeInstPtr* tinst = flat->isa_instptr();
1658     if (tinst && tinst->offset() >= instanceOopDesc::base_offset_in_bytes()) {
1659       if (tinst->const_oop() != NULL &&
1660           tinst->klass() == ciEnv::current()->Class_klass() &&
1661           tinst->offset() >= (tinst->klass()->as_instance_klass()->size_helper() * wordSize)) {
1662         // static field
1663         ciInstanceKlass* k = tinst->const_oop()->as_instance()->java_lang_Class_klass()->as_instance_klass();
1664         field = k->get_field_by_offset(tinst->offset(), true);
1665       } else if (tinst->klass()->is_inlinetype()) {
1666         // Inline type field
1667         ciInlineKlass* vk = tinst->inline_klass();
1668         field = vk->get_field_by_offset(tinst->offset(), false);
1669       } else {
1670         ciInstanceKlass* k = tinst->klass()->as_instance_klass();
1671         field = k->get_field_by_offset(tinst->offset(), false);
1672       }
1673     }
1674     assert(field == NULL ||
1675            original_field == NULL ||
1676            (field->holder() == original_field->holder() &&
1677             field->offset() == original_field->offset() &&
1678             field->is_static() == original_field->is_static()), "wrong field?");
1679     // Set field() and is_rewritable() attributes.
1680     if (field != NULL) {
1681       alias_type(idx)->set_field(field);
1682       if (flat->isa_aryptr()) {
1683         // Fields of flat arrays are rewritable although they are declared final
1684         assert(flat->is_aryptr()->is_flat(), "must be a flat array");
1685         alias_type(idx)->set_rewritable(true);
1686       }
1687     }
1688   }
1689 
1690   // Fill the cache for next time.
1691   if (!uncached) {
1692     ace->_adr_type = adr_type;
1693     ace->_index    = idx;
1694     assert(alias_type(adr_type) == alias_type(idx),  "type must be installed");
1695 
1696     // Might as well try to fill the cache for the flattened version, too.
1697     AliasCacheEntry* face = probe_alias_cache(flat);
1698     if (face->_adr_type == NULL) {
1699       face->_adr_type = flat;
1700       face->_index    = idx;
1701       assert(alias_type(flat) == alias_type(idx), "flat type must work too");
1702     }
1703   }
1704 
1705   return alias_type(idx);
1706 }

1735 //-----------------------------must_alias--------------------------------------
1736 // True if all values of the given address type are in the given alias category.
1737 bool Compile::must_alias(const TypePtr* adr_type, int alias_idx) {
1738   if (alias_idx == AliasIdxBot)         return true;  // the universal category
1739   if (adr_type == NULL)                 return true;  // NULL serves as TypePtr::TOP
1740   if (alias_idx == AliasIdxTop)         return false; // the empty category
1741   if (adr_type->base() == Type::AnyPtr) return false; // TypePtr::BOTTOM or its twins
1742 
1743   // the only remaining possible overlap is identity
1744   int adr_idx = get_alias_index(adr_type);
1745   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1746   assert(adr_idx == alias_idx ||
1747          (alias_type(alias_idx)->adr_type() != TypeOopPtr::BOTTOM
1748           && adr_type                       != TypeOopPtr::BOTTOM),
1749          "should not be testing for overlap with an unsafe pointer");
1750   return adr_idx == alias_idx;
1751 }

1753 //------------------------------can_alias--------------------------------------
1754 // True if any values of the given address type are in the given alias category.
1755 bool Compile::can_alias(const TypePtr* adr_type, int alias_idx) {
1756   if (alias_idx == AliasIdxTop)         return false; // the empty category
1757   if (adr_type == NULL)                 return false; // NULL serves as TypePtr::TOP
1758   // Known instance doesn't alias with bottom memory
1759   if (alias_idx == AliasIdxBot)         return !adr_type->is_known_instance();                   // the universal category
1760   if (adr_type->base() == Type::AnyPtr) return !C->get_adr_type(alias_idx)->is_known_instance(); // TypePtr::BOTTOM or its twins
1761 
1762   // the only remaining possible overlap is identity
1763   int adr_idx = get_alias_index(adr_type);
1764   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1765   return adr_idx == alias_idx;
1766 }

1777 //----------------------------Inline_Warm--------------------------------------
1778 int Compile::Inline_Warm() {
1779   // If there is room, try to inline some more warm call sites.
1780   // %%% Do a graph index compaction pass when we think we're out of space?
1781   if (!InlineWarmCalls)  return 0;
1782 
1783   int calls_made_hot = 0;
1784   int room_to_grow   = NodeCountInliningCutoff - unique();
1785   int amount_to_grow = MIN2(room_to_grow, (int)NodeCountInliningStep);
1786   int amount_grown   = 0;
1787   WarmCallInfo* call;
1788   while (amount_to_grow > 0 && (call = pop_warm_call()) != NULL) {
1789     int est_size = (int)call->size();
1790     if (est_size > (room_to_grow - amount_grown)) {
1791       // This one won't fit anyway.  Get rid of it.
1792       call->make_cold();
1793       continue;
1794     }
1795     call->make_hot();
1796     calls_made_hot++;
1797     amount_grown   += est_size;
1798     amount_to_grow -= est_size;
1799   }
1800 
1801   if (calls_made_hot > 0)  set_major_progress();
1802   return calls_made_hot;
1803 }

1806 //----------------------------Finish_Warm--------------------------------------
1807 void Compile::Finish_Warm() {
1808   if (!InlineWarmCalls)  return;
1809   if (failing())  return;
1810   if (warm_calls() == NULL)  return;
1811 
1812   // Clean up loose ends, if we are out of space for inlining.
1813   WarmCallInfo* call;
1814   while ((call = pop_warm_call()) != NULL) {
1815     call->make_cold();
1816   }
1817 }

1878 // Does the return value keep otherwise useless inline type allocations alive?
1879 static bool return_val_keeps_allocations_alive(Node* ret_val) {
1880   ResourceMark rm;
1881   Unique_Node_List wq;
1882   wq.push(ret_val);
1883   bool some_allocations = false;
1884   for (uint i = 0; i < wq.size(); i++) {
1885     Node* n = wq.at(i);
1886     assert(!n->is_InlineType(), "chain of inline type nodes");
1887     if (n->outcnt() > 1) {
1888       // Some other use for the allocation
1889       return false;
1890     } else if (n->is_InlineTypePtr()) {
1891       wq.push(n->in(1));
1892     } else if (n->is_Phi()) {
1893       for (uint j = 1; j < n->req(); j++) {
1894         wq.push(n->in(j));
1895       }
1896     } else if (n->is_CheckCastPP() &&
1897                n->in(1)->is_Proj() &&
1898                n->in(1)->in(0)->is_Allocate()) {
1899       some_allocations = true;
1900     }
1901   }
1902   return some_allocations;

1905 void Compile::process_inline_types(PhaseIterGVN &igvn, bool post_ea) {
1906   // Make inline types scalar in safepoints
1907   for (int i = _inline_type_nodes->length()-1; i >= 0; i--) {
1908     InlineTypeBaseNode* vt = _inline_type_nodes->at(i)->as_InlineTypeBase();
1909     vt->make_scalar_in_safepoints(&igvn);
1910   }
1911   // Remove InlineTypePtr nodes only after EA to give scalar replacement a chance
1912   // to remove buffer allocations. InlineType nodes are kept until loop opts and
1913   // removed via InlineTypeNode::remove_redundant_allocations.
1914   if (post_ea) {
1915     while (_inline_type_nodes->length() > 0) {
1916       InlineTypeBaseNode* vt = _inline_type_nodes->pop()->as_InlineTypeBase();
1917       if (vt->is_InlineTypePtr()) {
1918         igvn.replace_node(vt, vt->get_oop());
1919       }
1920     }
1921   }
1922   // Make sure that the return value does not keep an unused allocation alive
1923   if (tf()->returns_inline_type_as_fields()) {
1924     Node* ret = NULL;
1925     for (uint i = 1; i < root()->req(); i++){
1926       Node* in = root()->in(i);
1927       if (in->Opcode() == Op_Return) {
1928         assert(ret == NULL, "only one return");
1929         ret = in;
1930       }
1931     }
1932     if (ret != NULL) {
1933       Node* ret_val = ret->in(TypeFunc::Parms);
1934       if (igvn.type(ret_val)->isa_oopptr() &&
1935           return_val_keeps_allocations_alive(ret_val)) {
1936         igvn.replace_input_of(ret, TypeFunc::Parms, InlineTypeNode::tagged_klass(igvn.type(ret_val)->inline_klass(), igvn));
1937         assert(ret_val->outcnt() == 0, "should be dead now");
1938         igvn.remove_dead_node(ret_val);
1939       }
1940     }
1941   }
1942   igvn.optimize();
1943 }

1945 void Compile::adjust_flattened_array_access_aliases(PhaseIterGVN& igvn) {
1946   if (!_has_flattened_accesses) {
1947     return;
1948   }
1949   // Initially, all flattened array accesses share the same slice to
1950   // keep dependencies with Object[] array accesses (that could be
1951   // to a flattened array) correct. We're done with parsing so we
1952   // now know all flattened array accesses in this compile
1953   // unit. Let's move flattened array accesses to their own slice,
1954   // one per element field. This should help memory access
1955   // optimizations.
1956   ResourceMark rm;
1957   Unique_Node_List wq;
1958   wq.push(root());
1959 
1960   Node_List mergememnodes;
1961   Node_List memnodes;
1962 
1963   // Alias index currently shared by all flattened memory accesses
1964   int index = get_alias_index(TypeAryPtr::INLINES);
1965 
1966   // Find MergeMem nodes and flattened array accesses
1967   for (uint i = 0; i < wq.size(); i++) {
1968     Node* n = wq.at(i);
1969     if (n->is_Mem()) {
1970       const TypePtr* adr_type = NULL;
1971       if (n->Opcode() == Op_StoreCM) {
1972         adr_type = get_adr_type(get_alias_index(n->in(MemNode::OopStore)->adr_type()));
1973       } else {
1974         adr_type = get_adr_type(get_alias_index(n->adr_type()));
1975       }
1976       if (adr_type == TypeAryPtr::INLINES) {
1977         memnodes.push(n);
1978       }
1979     } else if (n->is_MergeMem()) {
1980       MergeMemNode* mm = n->as_MergeMem();
1981       if (mm->memory_at(index) != mm->base_memory()) {
1982         mergememnodes.push(n);
1983       }
1984     }
1985     for (uint j = 0; j < n->req(); j++) {
1986       Node* m = n->in(j);
1987       if (m != NULL) {
1988         wq.push(m);
1989       }
1990     }
1991   }
1992 
1993   if (memnodes.size() > 0) {
1994     _flattened_accesses_share_alias = false;
1995 
1996     // We are going to change the slice for the flattened array
1997     // accesses so we need to clear the cache entries that refer to
1998     // them.
1999     for (uint i = 0; i < AliasCacheSize; i++) {
2000       AliasCacheEntry* ace = &_alias_cache[i];
2001       if (ace->_adr_type != NULL &&
2002           ace->_adr_type->isa_aryptr() &&
2003           ace->_adr_type->is_aryptr()->is_flat()) {
2004         ace->_adr_type = NULL;
2005         ace->_index = (i != 0) ? 0 : AliasIdxTop; // Make sure the NULL adr_type resolves to AliasIdxTop
2006       }
2007     }
2008 
2009     // Find what aliases we are going to add
2010     int start_alias = num_alias_types()-1;
2011     int stop_alias = 0;
2012 
2013     for (uint i = 0; i < memnodes.size(); i++) {
2014       Node* m = memnodes.at(i);
2015       const TypePtr* adr_type = NULL;
2016       if (m->Opcode() == Op_StoreCM) {
2017         adr_type = m->in(MemNode::OopStore)->adr_type();
2018         Node* clone = new StoreCMNode(m->in(MemNode::Control), m->in(MemNode::Memory), m->in(MemNode::Address),
2019                                       m->adr_type(), m->in(MemNode::ValueIn), m->in(MemNode::OopStore),
2020                                       get_alias_index(adr_type));
2021         igvn.register_new_node_with_optimizer(clone);
2022         igvn.replace_node(m, clone);
2023       } else {
2024         adr_type = m->adr_type();
2025 #ifdef ASSERT
2026         m->as_Mem()->set_adr_type(adr_type);
2027 #endif
2028       }
2029       int idx = get_alias_index(adr_type);
2030       start_alias = MIN2(start_alias, idx);
2031       stop_alias = MAX2(stop_alias, idx);
2032     }
2033 
2034     assert(stop_alias >= start_alias, "should have expanded aliases");
2035 
2036     Node_Stack stack(0);
2037 #ifdef ASSERT
2038     VectorSet seen(Thread::current()->resource_area());
2039 #endif
2040     // Now let's fix the memory graph so each flattened array access
2041     // is moved to the right slice. Start from the MergeMem nodes.
2042     uint last = unique();
2043     for (uint i = 0; i < mergememnodes.size(); i++) {
2044       MergeMemNode* current = mergememnodes.at(i)->as_MergeMem();
2045       Node* n = current->memory_at(index);
2046       MergeMemNode* mm = NULL;
2047       do {
2048         // Follow memory edges through memory accesses, phis and
2049         // narrow membars and push nodes on the stack. Once we hit
2050         // bottom memory, we pop element off the stack one at a
2051         // time, in reverse order, and move them to the right slice
2052         // by changing their memory edges.
2053         if ((n->is_Phi() && n->adr_type() != TypePtr::BOTTOM) || n->is_Mem() || n->adr_type() == TypeAryPtr::INLINES) {
2054           assert(!seen.test_set(n->_idx), "");
2055           // Uses (a load for instance) will need to be moved to the
2056           // right slice as well and will get a new memory state
2057           // that we don't know yet. The use could also be the
2058           // backedge of a loop. We put a place holder node between
2059           // the memory node and its uses. We replace that place
2060           // holder with the correct memory state once we know it,
2061           // i.e. when nodes are popped off the stack. Using the
2062           // place holder make the logic work in the presence of
2063           // loops.
2064           if (n->outcnt() > 1) {
2065             Node* place_holder = NULL;
2066             assert(!n->has_out_with(Op_Node), "");
2067             for (DUIterator k = n->outs(); n->has_out(k); k++) {
2068               Node* u = n->out(k);
2069               if (u != current && u->_idx < last) {
2070                 bool success = false;
2071                 for (uint l = 0; l < u->req(); l++) {
2072                   if (!stack.is_empty() && u == stack.node() && l == stack.index()) {
2073                     continue;
2074                   }
2075                   Node* in = u->in(l);
2076                   if (in == n) {
2077                     if (place_holder == NULL) {
2078                       place_holder = new Node(1);
2079                       place_holder->init_req(0, n);
2080                     }
2081                     igvn.replace_input_of(u, l, place_holder);
2082                     success = true;
2083                   }
2084                 }
2085                 if (success) {
2086                   --k;
2087                 }
2088               }
2089             }
2090           }
2091           if (n->is_Phi()) {
2092             stack.push(n, 1);
2093             n = n->in(1);
2094           } else if (n->is_Mem()) {
2095             stack.push(n, n->req());
2096             n = n->in(MemNode::Memory);
2097           } else {
2098             assert(n->is_Proj() && n->in(0)->Opcode() == Op_MemBarCPUOrder, "");
2099             stack.push(n, n->req());
2100             n = n->in(0)->in(TypeFunc::Memory);
2101           }
2102         } else {
2103           assert(n->adr_type() == TypePtr::BOTTOM || (n->Opcode() == Op_Node && n->_idx >= last) || (n->is_Proj() && n->in(0)->is_Initialize()), "");
2104           // Build a new MergeMem node to carry the new memory state
2105           // as we build it. IGVN should fold extraneous MergeMem
2106           // nodes.
2107           mm = MergeMemNode::make(n);
2108           igvn.register_new_node_with_optimizer(mm);
2109           while (stack.size() > 0) {
2110             Node* m = stack.node();
2111             uint idx = stack.index();
2112             if (m->is_Mem()) {
2113               // Move memory node to its new slice
2114               const TypePtr* adr_type = m->adr_type();
2115               int alias = get_alias_index(adr_type);
2116               Node* prev = mm->memory_at(alias);
2117               igvn.replace_input_of(m, MemNode::Memory, prev);
2118               mm->set_memory_at(alias, m);
2119             } else if (m->is_Phi()) {
2120               // We need as many new phis as there are new aliases
2121               igvn.replace_input_of(m, idx, mm);
2122               if (idx == m->req()-1) {
2123                 Node* r = m->in(0);
2124                 for (uint j = (uint)start_alias; j <= (uint)stop_alias; j++) {
2125                   const Type* adr_type = get_adr_type(j);
2126                   if (!adr_type->isa_aryptr() || !adr_type->is_aryptr()->is_flat()) {
2127                     continue;
2128                   }
2129                   Node* phi = new PhiNode(r, Type::MEMORY, get_adr_type(j));
2130                   igvn.register_new_node_with_optimizer(phi);
2131                   for (uint k = 1; k < m->req(); k++) {
2132                     phi->init_req(k, m->in(k)->as_MergeMem()->memory_at(j));
2133                   }
2134                   mm->set_memory_at(j, phi);
2135                 }
2136                 Node* base_phi = new PhiNode(r, Type::MEMORY, TypePtr::BOTTOM);
2137                 igvn.register_new_node_with_optimizer(base_phi);
2138                 for (uint k = 1; k < m->req(); k++) {
2139                   base_phi->init_req(k, m->in(k)->as_MergeMem()->base_memory());
2140                 }
2141                 mm->set_base_memory(base_phi);
2142               }
2143             } else {
2144               // This is a MemBarCPUOrder node from
2145               // Parse::array_load()/Parse::array_store(), in the
2146               // branch that handles flattened arrays hidden under
2147               // an Object[] array. We also need one new membar per
2148               // new alias to keep the unknown access that the

2735   } else { // not a bitwise operations
2736     if (!inputs.member(n)) {

2776 //
2777 //  A macro logic node represents a truth table. It has 4 inputs,
2778 //  First three inputs corresponds to 3 columns of a truth table
2779 //  and fourth input captures the logic function.
2780 //
2781 //  eg.  fn = (in1 AND in2) OR in3;
2782 //
2783 //      MacroNode(in1,in2,in3,fn)
2784 //
2785 //  -----------------
2786 //  in1 in2 in3  fn
2787 //  -----------------
2788 //  0    0   0    0
2789 //  0    0   1    1
2790 //  0    1   0    0
2791 //  0    1   1    1
2792 //  1    0   0    0
2793 //  1    0   1    1
2794 //  1    1   0    1
2795 //  1    1   1    1
2796 //

2798 uint Compile::eval_macro_logic_op(uint func, uint in1 , uint in2, uint in3) {
2799   int res = 0;
2800   for (int i = 0; i < 8; i++) {
2801     int bit1 = extract_bit(in1, i);
2802     int bit2 = extract_bit(in2, i);
2803     int bit3 = extract_bit(in3, i);
2804 
2805     int func_bit_pos = (bit1 << 2 | bit2 << 1 | bit3);
2806     int func_bit = extract_bit(func, func_bit_pos);
2807 
2808     res |= func_bit << i;
2809   }
2810   return res;
2811 }

2835 uint Compile::compute_truth_table(Unique_Node_List& partition, Unique_Node_List& inputs) {
2836   assert(inputs.size() <= 3, "sanity");
2837   ResourceMark rm;
2838   uint res = 0;
2839   ResourceHashtable<Node*,uint> eval_map;
2840 
2841   // Populate precomputed functions for inputs.
2842   // Each input corresponds to one column of 3 input truth-table.
2843   uint input_funcs[] = { 0xAA,   // (_, _, a) -> a
2844                          0xCC,   // (_, b, _) -> b
2845                          0xF0 }; // (c, _, _) -> c
2846   for (uint i = 0; i < inputs.size(); i++) {
2847     eval_map.put(inputs.at(i), input_funcs[i]);
2848   }
2849 
2850   for (uint i = 0; i < partition.size(); i++) {
2851     Node* n = partition.at(i);
2852 
2853     uint func1 = 0, func2 = 0, func3 = 0;
2854     eval_operands(n, func1, func2, func3, eval_map);
2855 
2856     switch (n->Opcode()) {
2857       case Op_OrV:
2858         assert(func3 == 0, "not binary");
2859         res = func1 | func2;
2860         break;
2861       case Op_AndV:
2862         assert(func3 == 0, "not binary");
2863         res = func1 & func2;
2864         break;
2865       case Op_XorV:
2866         if (VectorNode::is_vector_bitwise_not_pattern(n)) {
2867           assert(func2 == 0 && func3 == 0, "not unary");
2868           res = (~func1) & 0xFF;
2869         } else {
2870           assert(func3 == 0, "not binary");
2871           res = func1 ^ func2;
2872         }
2873         break;
2874       case Op_MacroLogicV:
2875         // Ordering of inputs may change during evaluation of sub-tree
2876         // containing MacroLogic node as a child node, thus a re-evaluation
2877         // makes sure that function is evaluated in context of current
2878         // inputs.
2879         res = eval_macro_logic_op(n->in(4)->get_int(), func1, func2, func3);
2880         break;
2881 
2882       default: assert(false, "not supported: %s", n->Name());
2883     }
2884     assert(res <= 0xFF, "invalid");
2885     eval_map.put(n, res);
2886   }
2887   return res;
2888 }

2890 bool Compile::compute_logic_cone(Node* n, Unique_Node_List& partition, Unique_Node_List& inputs) {
2891   assert(partition.size() == 0, "not empty");
2892   assert(inputs.size() == 0, "not empty");
2893   if (is_vector_ternary_bitwise_op(n)) {
2894     return false;
2895   }
2896 
2897   bool is_unary_op = is_vector_unary_bitwise_op(n);
2898   if (is_unary_op) {
2899     assert(collect_unique_inputs(n, partition, inputs) == 1, "not unary");
2900     return false; // too few inputs
2901   }
2902 
2903   assert(is_vector_binary_bitwise_op(n), "not binary");
2904   Node* in1 = n->in(1);
2905   Node* in2 = n->in(2);
2906 
2907   int in1_unique_inputs_cnt = collect_unique_inputs(in1, partition, inputs);
2908   int in2_unique_inputs_cnt = collect_unique_inputs(in2, partition, inputs);
2909   partition.push(n);
2910 
2911   // Too many inputs?
2912   if (inputs.size() > 3) {
2913     partition.clear();
2914     inputs.clear();
2915     { // Recompute in2 inputs
2916       Unique_Node_List not_used;
2917       in2_unique_inputs_cnt = collect_unique_inputs(in2, not_used, not_used);
2918     }
2919     // Pick the node with minimum number of inputs.
2920     if (in1_unique_inputs_cnt >= 3 && in2_unique_inputs_cnt >= 3) {
2921       return false; // still too many inputs
2922     }
2923     // Recompute partition & inputs.
2924     Node* child       = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in1 : in2);
2925     collect_unique_inputs(child, partition, inputs);
2926 
2927     Node* other_input = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in2 : in1);
2928     inputs.push(other_input);
2929 
2930     partition.push(n);
2931   }
2932 
2933   return (partition.size() == 2 || partition.size() == 3) &&
2934          (inputs.size()    == 2 || inputs.size()    == 3);
2935 }

2938 void Compile::process_logic_cone_root(PhaseIterGVN &igvn, Node *n, VectorSet &visited) {
2939   assert(is_vector_bitwise_op(n), "not a root");
2940 
2941   visited.set(n->_idx);
2942 
2943   // 1) Do a DFS walk over the logic cone.
2944   for (uint i = 1; i < n->req(); i++) {
2945     Node* in = n->in(i);
2946     if (!visited.test(in->_idx) && is_vector_bitwise_op(in)) {
2947       process_logic_cone_root(igvn, in, visited);
2948     }
2949   }
2950 
2951   // 2) Bottom up traversal: Merge node[s] with
2952   // the parent to form macro logic node.
2953   Unique_Node_List partition;
2954   Unique_Node_List inputs;
2955   if (compute_logic_cone(n, partition, inputs)) {
2956     const TypeVect* vt = n->bottom_type()->is_vect();
2957     Node* macro_logic = xform_to_MacroLogicV(igvn, vt, partition, inputs);
2958     igvn.replace_node(n, macro_logic);
2959   }
2960 }

2980 //------------------------------Code_Gen---------------------------------------
2981 // Given a graph, generate code for it
2982 void Compile::Code_Gen() {
2983   if (failing()) {
2984     return;
2985   }
2986 
2987   // Perform instruction selection.  You might think we could reclaim Matcher
2988   // memory PDQ, but actually the Matcher is used in generating spill code.
2989   // Internals of the Matcher (including some VectorSets) must remain live
2990   // for awhile - thus I cannot reclaim Matcher memory lest a VectorSet usage
2991   // set a bit in reclaimed memory.
2992 
2993   // In debug mode can dump m._nodes.dump() for mapping of ideal to machine
2994   // nodes.  Mapping is only valid at the root of each matched subtree.
2995   NOT_PRODUCT( verify_graph_edges(); )
2996

2998   _matcher = &matcher;
2999   {
3000     TracePhase tp("matcher", &timers[_t_matcher]);
3001     matcher.match();
3002     if (failing()) {
3003       return;
3004     }
3005   }
3006 
3007   // In debug mode can dump m._nodes.dump() for mapping of ideal to machine
3008   // nodes.  Mapping is only valid at the root of each matched subtree.
3009   NOT_PRODUCT( verify_graph_edges(); )
3010 
3011   // If you have too many nodes, or if matching has failed, bail out
3012   check_node_count(0, "out of nodes matching instructions");

3016

3018

3045       return;

3047   }

3058     } else {

3060     }
3061     cfg.fixup_flow();
3062   }
3063

3065   if( OptoPeephole ) {

3070

3072   if (Matcher::require_postalloc_expand) {
3073     TracePhase tp("postalloc_expand", &timers[_t_postalloc_expand]);

3078   {
3079     TracePhase tp("output", &timers[_t_output]);

3081     output.Output();

3084   }
3085

3087 
3088   // He's dead, Jim.
3089   _cfg     = (PhaseCFG*)((intptr_t)0xdeadbeef);
3090   _regalloc = (PhaseChaitin*)((intptr_t)0xdeadbeef);
3091 }

3131 // Eliminate trivially redundant StoreCMs and accumulate their
3132 // precedence edges.
3133 void Compile::eliminate_redundant_card_marks(Node* n) {
3134   assert(n->Opcode() == Op_StoreCM, "expected StoreCM");
3135   if (n->in(MemNode::Address)->outcnt() > 1) {
3136     // There are multiple users of the same address so it might be
3137     // possible to eliminate some of the StoreCMs
3138     Node* mem = n->in(MemNode::Memory);
3139     Node* adr = n->in(MemNode::Address);
3140     Node* val = n->in(MemNode::ValueIn);
3141     Node* prev = n;
3142     bool done = false;
3143     // Walk the chain of StoreCMs eliminating ones that match.  As
3144     // long as it's a chain of single users then the optimization is
3145     // safe.  Eliminating partially redundant StoreCMs would require
3146     // cloning copies down the other paths.
3147     while (mem->Opcode() == Op_StoreCM && mem->outcnt() == 1 && !done) {
3148       if (adr == mem->in(MemNode::Address) &&
3149           val == mem->in(MemNode::ValueIn)) {
3150         // redundant StoreCM
3151         if (mem->req() > MemNode::OopStore) {
3152           // Hasn't been processed by this code yet.
3153           n->add_prec(mem->in(MemNode::OopStore));
3154         } else {
3155           // Already converted to precedence edge
3156           for (uint i = mem->req(); i < mem->len(); i++) {
3157             // Accumulate any precedence edges
3158             if (mem->in(i) != NULL) {
3159               n->add_prec(mem->in(i));
3160             }
3161           }
3162           // Everything above this point has been processed.
3163           done = true;
3164         }
3165         // Eliminate the previous StoreCM
3166         prev->set_req(MemNode::Memory, mem->in(MemNode::Memory));
3167         assert(mem->outcnt() == 0, "should be dead");
3168         mem->disconnect_inputs(NULL, this);
3169       } else {
3170         prev = mem;
3171       }
3172       mem = prev->in(MemNode::Memory);
3173     }
3174   }
3175 }

4012 //------------------------------final_graph_reshaping--------------------------
4013 // Final Graph Reshaping.
4014 //
4015 // (1) Clone simple inputs to uncommon calls, so they can be scheduled late
4016 //     and not commoned up and forced early.  Must come after regular
4017 //     optimizations to avoid GVN undoing the cloning.  Clone constant
4018 //     inputs to Loop Phis; these will be split by the allocator anyways.
4019 //     Remove Opaque nodes.
4020 // (2) Move last-uses by commutative operations to the left input to encourage
4021 //     Intel update-in-place two-address operations and better register usage
4022 //     on RISCs.  Must come after regular optimizations to avoid GVN Ideal
4023 //     calls canonicalizing them back.
4024 // (3) Count the number of double-precision FP ops, single-precision FP ops
4025 //     and call sites.  On Intel, we can get correct rounding either by
4026 //     forcing singles to memory (requires extra stores and loads after each
4027 //     FP bytecode) or we can set a rounding mode bit (requires setting and
4028 //     clearing the mode bit around call sites).  The mode bit is only used
4029 //     if the relative frequency of single FP ops to calls is low enough.
4030 //     This is a key transform for SPEC mpeg_audio.
4031 // (4) Detect infinite loops; blobs of code reachable from above but not
4032 //     below.  Several of the Code_Gen algorithms fail on such code shapes,
4033 //     so we simply bail out.  Happens a lot in ZKM.jar, but also happens
4034 //     from time to time in other codes (such as -Xcomp finalizer loops, etc).
4035 //     Detection is by looking for IfNodes where only 1 projection is
4036 //     reachable from below or CatchNodes missing some targets.
4037 // (5) Assert for insane oop offsets in debug mode.
4038 
4039 bool Compile::final_graph_reshaping() {
4040   // an infinite loop may have been eliminated by the optimizer,
4041   // in which case the graph will be empty.
4042   if (root()->req() == 1) {
4043     record_method_not_compilable("trivial infinite loop");
4044     return true;
4045   }
4046 
4047   // Expensive nodes have their control input set to prevent the GVN
4048   // from freely commoning them. There's no GVN beyond this point so
4049   // no need to keep the control input. We want the expensive nodes to
4050   // be freely moved to the least frequent code path by gcm.
4051   assert(OptimizeExpensiveOps || expensive_count() == 0, "optimization off but list non empty?");
4052   for (int i = 0; i < expensive_count(); i++) {
4053     _expensive_nodes->at(i)->set_req(0, NULL);
4054   }
4055 
4056   Final_Reshape_Counts frc;
4057 
4058   // Visit everybody reachable!
4059   // Allocate stack of size C->live_nodes()/2 to avoid frequent realloc
4060   Node_Stack nstack(live_nodes() >> 1);
4061   final_graph_reshaping_walk(nstack, root(), frc);
4062 
4063   // Check for unreachable (from below) code (i.e., infinite loops).
4064   for( uint i = 0; i < frc._tests.size(); i++ ) {
4065     MultiBranchNode *n = frc._tests[i]->as_MultiBranch();
4066     // Get number of CFG targets.
4067     // Note that PCTables include exception targets after calls.
4068     uint required_outcnt = n->required_outcnt();
4069     if (n->outcnt() != required_outcnt) {
4070       // Check for a few special cases.  Rethrow Nodes never take the
4071       // 'fall-thru' path, so expected kids is 1 less.
4072       if (n->is_PCTable() && n->in(0) && n->in(0)->in(0)) {
4073         if (n->in(0)->in(0)->is_Call()) {
4074           CallNode *call = n->in(0)->in(0)->as_Call();
4075           if (call->entry_point() == OptoRuntime::rethrow_stub()) {
4076             required_outcnt--;      // Rethrow always has 1 less kid
4077           } else if (call->req() > TypeFunc::Parms &&
4078                      call->is_CallDynamicJava()) {
4079             // Check for null receiver. In such case, the optimizer has
4080             // detected that the virtual call will always result in a null
4081             // pointer exception. The fall-through projection of this CatchNode
4082             // will not be populated.
4083             Node *arg0 = call->in(TypeFunc::Parms);
4084             if (arg0->is_Type() &&
4085                 arg0->as_Type()->type()->higher_equal(TypePtr::NULL_PTR)) {
4086               required_outcnt--;
4087             }
4088           } else if (call->entry_point() == OptoRuntime::new_array_Java() &&
4089                      call->req() > TypeFunc::Parms+1 &&
4090                      call->is_CallStaticJava()) {
4091             // Check for negative array length. In such case, the optimizer has
4092             // detected that the allocation attempt will always result in an
4093             // exception. There is no fall-through projection of this CatchNode .
4094             Node *arg1 = call->in(TypeFunc::Parms+1);
4095             if (arg1->is_Type() &&
4096                 arg1->as_Type()->type()->join(TypeInt::POS)->empty()) {
4097               required_outcnt--;
4098             }
4099           }
4100         }
4101       }
4102       // Recheck with a better notion of 'required_outcnt'
4103       if (n->outcnt() != required_outcnt) {
4104         record_method_not_compilable("malformed control flow");
4105         return true;            // Not all targets reachable!
4106       }
4107     }
4108     // Check that I actually visited all kids.  Unreached kids
4109     // must be infinite loops.
4110     for (DUIterator_Fast jmax, j = n->fast_outs(jmax); j < jmax; j++)
4111       if (!frc._visited.test(n->fast_out(j)->_idx)) {
4112         record_method_not_compilable("infinite loop");
4113         return true;            // Found unvisited kid; must be unreach
4114       }
4115 
4116     // Here so verification code in final_graph_reshaping_walk()
4117     // always see an OuterStripMinedLoopEnd
4118     if (n->is_OuterStripMinedLoopEnd()) {
4119       IfNode* init_iff = n->as_If();
4120       Node* iff = new IfNode(init_iff->in(0), init_iff->in(1), init_iff->_prob, init_iff->_fcnt);
4121       n->subsume_by(iff, this);
4122     }
4123   }
4124 
4125 #ifdef IA32
4126   // If original bytecodes contained a mixture of floats and doubles
4127   // check if the optimizer has made it homogenous, item (3).
4128   if (UseSSE == 0 &&
4129       frc.get_float_count() > 32 &&
4130       frc.get_double_count() == 0 &&
4131       (10 * frc.get_call_count() < frc.get_float_count()) ) {
4132     set_24_bit_selection_and_mode(false, true);
4133   }
4134 #endif // IA32
4135 
4136   set_java_calls(frc.get_java_call_count());
4137   set_inner_loops(frc.get_inner_loop_count());
4138 
4139   // No infinite loops, no reason to bail out.
4140   return false;
4141 }

4191 // Report if there are too many recompiles at the current method and bci.
4192 // Consults PerBytecodeRecompilationCutoff and PerMethodRecompilationCutoff.
4193 // Is not eager to return true, since this will cause the compiler to use
4194 // Action_none for a trap point, to avoid too many recompilations.
4195 bool Compile::too_many_recompiles(ciMethod* method,
4196                                   int bci,
4197                                   Deoptimization::DeoptReason reason) {
4198   ciMethodData* md = method->method_data();
4199   if (md->is_empty()) {
4200     // Assume the trap has not occurred, or that it occurred only
4201     // because of a transient condition during start-up in the interpreter.
4202     return false;
4203   }
4204   // Pick a cutoff point well within PerBytecodeRecompilationCutoff.
4205   uint bc_cutoff = (uint) PerBytecodeRecompilationCutoff / 8;
4206   uint m_cutoff  = (uint) PerMethodRecompilationCutoff / 2 + 1;  // not zero
4207   Deoptimization::DeoptReason per_bc_reason
4208     = Deoptimization::reason_recorded_per_bytecode_if_any(reason);
4209   ciMethod* m = Deoptimization::reason_is_speculate(reason) ? this->method() : NULL;
4210   if ((per_bc_reason == Deoptimization::Reason_none
4211        || md->has_trap_at(bci, m, reason) != 0)
4212       // The trap frequency measure we care about is the recompile count:
4213       && md->trap_recompiled_at(bci, m)
4214       && md->overflow_recompile_count() >= bc_cutoff) {
4215     // Do not emit a trap here if it has already caused recompilations.
4216     // Also, if there are multiple reasons, or if there is no per-BCI record,
4217     // assume the worst.
4218     if (log())
4219       log()->elem("observe trap='%s recompiled' count='%d' recompiles2='%d'",
4220                   Deoptimization::trap_reason_name(reason),
4221                   md->trap_count(reason),
4222                   md->overflow_recompile_count());
4223     return true;
4224   } else if (trap_count(reason) != 0
4225              && decompile_count() >= m_cutoff) {
4226     // Too many recompiles globally, and we have seen this sort of trap.
4227     // Use cumulative decompile_count, not just md->decompile_count.
4228     if (log())
4229       log()->elem("observe trap='%s' count='%d' mcount='%d' decompiles='%d' mdecompiles='%d'",
4230                   Deoptimization::trap_reason_name(reason),
4231                   md->trap_count(reason), trap_count(reason),
4232                   md->decompile_count(), decompile_count());
4233     return true;
4234   } else {
4235     // The coast is clear.
4236     return false;
4237   }

4394 //----------------------------static_subtype_check-----------------------------
4395 // Shortcut important common cases when superklass is exact:
4396 // (0) superklass is java.lang.Object (can occur in reflective code)
4397 // (1) subklass is already limited to a subtype of superklass => always ok
4398 // (2) subklass does not overlap with superklass => always fail
4399 // (3) superklass has NO subtypes and we can check with a simple compare.
4400 int Compile::static_subtype_check(ciKlass* superk, ciKlass* subk) {
4401   if (StressReflectiveCode || superk == NULL || subk == NULL) {
4402     return SSC_full_test;       // Let caller generate the general case.
4403   }
4404 
4405   if (superk == env()->Object_klass()) {
4406     return SSC_always_true;     // (0) this test cannot fail
4407   }
4408 
4409   ciType* superelem = superk;
4410   if (superelem->is_array_klass()) {
4411     ciArrayKlass* ak = superelem->as_array_klass();
4412     superelem = superelem->as_array_klass()->base_element_type();
4413   }
4414 
4415   if (!subk->is_interface()) {  // cannot trust static interface types yet
4416     if (subk->is_subtype_of(superk)) {
4417       return SSC_always_true;   // (1) false path dead; no dynamic test needed
4418     }
4419     if (!(superelem->is_klass() && superelem->as_klass()->is_interface()) &&
4420         !superk->is_subtype_of(subk)) {
4421       return SSC_always_false;
4422     }
4423   }
4424 
4425   // If casting to an instance klass, it must have no subtypes
4426   if (superk->is_interface()) {
4427     // Cannot trust interfaces yet.
4428     // %%% S.B. superk->nof_implementors() == 1
4429   } else if (superelem->is_instance_klass()) {
4430     ciInstanceKlass* ik = superelem->as_instance_klass();

4466 // Convert integer value to a narrowed long type dependent on ctrl (for example, a range check)

4477     phase->C->add_range_check_cast(value);

4720 bool Compile::should_optimize_expensive_nodes(PhaseIterGVN &igvn) {
4721   if (_expensive_nodes->length() == 0) {
4722     return false;
4723   }
4724 
4725   assert(OptimizeExpensiveOps, "optimization off?");
4726 
4727   // Take this opportunity to remove dead nodes from the list
4728   int j = 0;
4729   for (int i = 0; i < _expensive_nodes->length(); i++) {
4730     Node* n = _expensive_nodes->at(i);
4731     if (!n->is_unreachable(igvn)) {
4732       assert(n->is_expensive(), "should be expensive");
4733       _expensive_nodes->at_put(j, n);
4734       j++;
4735     }
4736   }
4737   _expensive_nodes->trunc_to(j);
4738 
4739   // Then sort the list so that similar nodes are next to each other
4740   // and check for at least two nodes of identical kind with same data
4741   // inputs.
4742   sort_expensive_nodes();
4743 
4744   for (int i = 0; i < _expensive_nodes->length()-1; i++) {
4745     if (cmp_expensive_nodes(_expensive_nodes->adr_at(i), _expensive_nodes->adr_at(i+1)) == 0) {
4746       return true;
4747     }
4748   }
4749 
4750   return false;
4751 }

4753 void Compile::cleanup_expensive_nodes(PhaseIterGVN &igvn) {
4754   if (_expensive_nodes->length() == 0) {
4755     return;
4756   }
4757 
4758   assert(OptimizeExpensiveOps, "optimization off?");
4759 
4760   // Sort to bring similar nodes next to each other and clear the
4761   // control input of nodes for which there's only a single copy.
4762   sort_expensive_nodes();
4763 
4764   int j = 0;
4765   int identical = 0;
4766   int i = 0;
4767   bool modified = false;
4768   for (; i < _expensive_nodes->length()-1; i++) {
4769     assert(j <= i, "can't write beyond current index");
4770     if (_expensive_nodes->at(i)->Opcode() == _expensive_nodes->at(i+1)->Opcode()) {
4771       identical++;
4772       _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4773       continue;
4774     }
4775     if (identical > 0) {
4776       _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4777       identical = 0;
4778     } else {
4779       Node* n = _expensive_nodes->at(i);
4780       igvn.replace_input_of(n, 0, NULL);
4781       igvn.hash_insert(n);
4782       modified = true;
4783     }
4784   }
4785   if (identical > 0) {
4786     _expensive_nodes->at_put(j++, _expensive_nodes->at(i));
4787   } else if (_expensive_nodes->length() >= 1) {
4788     Node* n = _expensive_nodes->at(i);
4789     igvn.replace_input_of(n, 0, NULL);
4790     igvn.hash_insert(n);
4791     modified = true;
4792   }
4793   _expensive_nodes->trunc_to(j);
4794   if (modified) {
4795     igvn.optimize();
4796   }
4797 }

4895 // Auxiliary method to support randomized stressing/fuzzing.
4896 //
4897 // This method can be called the arbitrary number of times, with current count
4898 // as the argument. The logic allows selecting a single candidate from the
4899 // running list of candidates as follows:
4900 //    int count = 0;
4901 //    Cand* selected = null;
4902 //    while(cand = cand->next()) {
4903 //      if (randomized_select(++count)) {
4904 //        selected = cand;
4905 //      }
4906 //    }
4907 //
4908 // Including count equalizes the chances any candidate is "selected".
4909 // This is useful when we don't have the complete list of candidates to choose
4910 // from uniformly. In this case, we need to adjust the randomicity of the
4911 // selection, or else we will end up biasing the selection towards the latter
4912 // candidates.
4913 //
4914 // Quick back-envelope calculation shows that for the list of n candidates
4915 // the equal probability for the candidate to persist as "best" can be
4916 // achieved by replacing it with "next" k-th candidate with the probability
4917 // of 1/k. It can be easily shown that by the end of the run, the
4918 // probability for any candidate is converged to 1/n, thus giving the
4919 // uniform distribution among all the candidates.
4920 //
4921 // We don't care about the domain size as long as (RANDOMIZED_DOMAIN / count) is large.
4922 #define RANDOMIZED_DOMAIN_POW 29
4923 #define RANDOMIZED_DOMAIN (1 << RANDOMIZED_DOMAIN_POW)
4924 #define RANDOMIZED_DOMAIN_MASK ((1 << (RANDOMIZED_DOMAIN_POW + 1)) - 1)
4925 bool Compile::randomized_select(int count) {
4926   assert(count > 0, "only positive");
4927   return (os::random() & RANDOMIZED_DOMAIN_MASK) < (RANDOMIZED_DOMAIN / count);
4928 }
