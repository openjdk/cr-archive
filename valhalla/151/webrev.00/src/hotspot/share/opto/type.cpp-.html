  56 const Type::Offset Type::Offset::meet(const Type::Offset other) const {
  57   // Either is 'TOP' offset?  Return the other offset!
  58   int offset = other._offset;
  59   if (_offset == OffsetTop) return Offset(offset);
  60   if (offset == OffsetTop) return Offset(_offset);
  61   // If either is different, return 'BOTTOM' offset
  62   if (_offset != offset) return bottom;
  63   return Offset(_offset);
  64 }

  66 const Type::Offset Type::Offset::dual() const {
  67   if (_offset == OffsetTop) return bottom;// Map 'TOP' into 'BOTTOM'
  68   if (_offset == OffsetBot) return top;// Map 'BOTTOM' into 'TOP'
  69   return Offset(_offset);               // Map everything else into self
  70 }

  72 const Type::Offset Type::Offset::add(intptr_t offset) const {
  73   // Adding to 'TOP' offset?  Return 'TOP'!
  74   if (_offset == OffsetTop || offset == OffsetTop) return top;
  75   // Adding to 'BOTTOM' offset?  Return 'BOTTOM'!
  76   if (_offset == OffsetBot || offset == OffsetBot) return bottom;
  77   // Addition overflows or "accidentally" equals to OffsetTop? Return 'BOTTOM'!
  78   offset += (intptr_t)_offset;
  79   if (offset != (int)offset || offset == OffsetTop) return bottom;
  80 
  81   // assert( _offset >= 0 && _offset+offset >= 0, "" );
  82   // It is possible to construct a negative offset during PhaseCCP
  83 
  84   return Offset((int)offset);        // Sum valid offsets
  85 }

  87 void Type::Offset::dump2(outputStream *st) const {
  88   if (_offset == 0) {
  89     return;
  90   } else if (_offset == OffsetTop) {
  91     st->print("+top");
  92   }
  93   else if (_offset == OffsetBot) {
  94     st->print("+bot");
  95   } else if (_offset) {
  96     st->print("+%d", _offset);
  97   }

 177 //------------------------------get_const_type---------------------------
 178 const Type* Type::get_const_type(ciType* type) {
 179   if (type == NULL) {
 180     return NULL;
 181   } else if (type->is_primitive_type()) {
 182     return get_const_basic_type(type->basic_type());
 183   } else {
 184     return TypeOopPtr::make_from_klass(type->as_klass());
 185   }

 188 //---------------------------array_element_basic_type---------------------------------
 189 // Mapping to the array element's basic type.
 190 BasicType Type::array_element_basic_type() const {
 191   BasicType bt = basic_type();
 192   if (bt == T_INT) {
 193     if (this == TypeInt::INT)   return T_INT;
 194     if (this == TypeInt::CHAR)  return T_CHAR;
 195     if (this == TypeInt::BYTE)  return T_BYTE;
 196     if (this == TypeInt::BOOL)  return T_BOOLEAN;
 197     if (this == TypeInt::SHORT) return T_SHORT;
 198     return T_VOID;
 199   }
 200   return bt;
 201 }

 203 // For two instance arrays of same dimension, return the base element types.
 204 // Otherwise or if the arrays have different dimensions, return NULL.
 205 void Type::get_arrays_base_elements(const Type *a1, const Type *a2,
 206                                     const TypeInstPtr **e1, const TypeInstPtr **e2) {
 207 
 208   if (e1) *e1 = NULL;
 209   if (e2) *e2 = NULL;
 210   const TypeAryPtr* a1tap = (a1 == NULL) ? NULL : a1->isa_aryptr();
 211   const TypeAryPtr* a2tap = (a2 == NULL) ? NULL : a2->isa_aryptr();
 212 
 213   if (a1tap != NULL && a2tap != NULL) {
 214     // Handle multidimensional arrays
 215     const TypePtr* a1tp = a1tap->elem()->make_ptr();
 216     const TypePtr* a2tp = a2tap->elem()->make_ptr();
 217     while (a1tp && a1tp->isa_aryptr() && a2tp && a2tp->isa_aryptr()) {
 218       a1tap = a1tp->is_aryptr();
 219       a2tap = a2tp->is_aryptr();
 220       a1tp = a1tap->elem()->make_ptr();
 221       a2tp = a2tap->elem()->make_ptr();
 222     }
 223     if (a1tp && a1tp->isa_instptr() && a2tp && a2tp->isa_instptr()) {
 224       if (e1) *e1 = a1tp->is_instptr();
 225       if (e2) *e2 = a2tp->is_instptr();
 226     }
 227   }
 228 }

 230 //---------------------------get_typeflow_type---------------------------------
 231 // Import a type produced by ciTypeFlow.
 232 const Type* Type::get_typeflow_type(ciType* type) {
 233   switch (type->basic_type()) {
 234 
 235   case ciTypeFlow::StateVector::T_BOTTOM:
 236     assert(type == ciTypeFlow::StateVector::bottom_type(), "");
 237     return Type::BOTTOM;
 238 
 239   case ciTypeFlow::StateVector::T_TOP:
 240     assert(type == ciTypeFlow::StateVector::top_type(), "");
 241     return Type::TOP;
 242 
 243   case ciTypeFlow::StateVector::T_NULL:
 244     assert(type == ciTypeFlow::StateVector::null_type(), "");
 245     return TypePtr::NULL_PTR;
 246 
 247   case ciTypeFlow::StateVector::T_LONG2:
 248     // The ciTypeFlow pass pushes a long, then the half.
 249     // We do the same.
 250     assert(type == ciTypeFlow::StateVector::long2_type(), "");
 251     return TypeInt::TOP;
 252 
 253   case ciTypeFlow::StateVector::T_DOUBLE2:
 254     // The ciTypeFlow pass pushes double, then the half.
 255     // Our convention is the same.
 256     assert(type == ciTypeFlow::StateVector::double2_type(), "");
 257     return Type::TOP;
 258 
 259   case T_ADDRESS:
 260     assert(type->is_return_address(), "");
 261     return TypeRawPtr::make((address)(intptr_t)type->as_return_address()->bci());
 262 
 263   case T_INLINE_TYPE: {
 264     ciInlineKlass* vk = type->as_inline_klass();
 265     if (vk->is_scalarizable()) {
 266       return TypeInlineType::make(vk);
 267     } else {
 268       return TypeOopPtr::make_from_klass(vk)->join_speculative(TypePtr::NOTNULL);
 269     }
 270   }
 271 
 272   default:
 273     // make sure we did not mix up the cases:
 274     assert(type != ciTypeFlow::StateVector::bottom_type(), "");
 275     assert(type != ciTypeFlow::StateVector::top_type(), "");
 276     assert(type != ciTypeFlow::StateVector::null_type(), "");
 277     assert(type != ciTypeFlow::StateVector::long2_type(), "");
 278     assert(type != ciTypeFlow::StateVector::double2_type(), "");
 279     assert(!type->is_return_address(), "");
 280 
 281     return Type::get_const_type(type);
 282   }
 283 }

 286 //-----------------------make_from_constant------------------------------------
 287 const Type* Type::make_from_constant(ciConstant constant, bool require_constant,
 288                                      int stable_dimension, bool is_narrow_oop,
 289                                      bool is_autobox_cache) {
 290   switch (constant.basic_type()) {
 291     case T_BOOLEAN:  return TypeInt::make(constant.as_boolean());
 292     case T_CHAR:     return TypeInt::make(constant.as_char());
 293     case T_BYTE:     return TypeInt::make(constant.as_byte());
 294     case T_SHORT:    return TypeInt::make(constant.as_short());
 295     case T_INT:      return TypeInt::make(constant.as_int());
 296     case T_LONG:     return TypeLong::make(constant.as_long());
 297     case T_FLOAT:    return TypeF::make(constant.as_float());
 298     case T_DOUBLE:   return TypeD::make(constant.as_double());
 299     case T_ARRAY:
 300     case T_INLINE_TYPE:
 301     case T_OBJECT: {
 302         const Type* con_type = NULL;
 303         ciObject* oop_constant = constant.as_object();
 304         if (oop_constant->is_null_object()) {
 305           con_type = Type::get_zero_type(T_OBJECT);
 306         } else {
 307           guarantee(require_constant || oop_constant->should_be_constant(), "con_type must get computed");
 308           con_type = TypeOopPtr::make_from_constant(oop_constant, require_constant);
 309           if (Compile::current()->eliminate_boxing() && is_autobox_cache) {
 310             con_type = con_type->is_aryptr()->cast_to_autobox_cache(true);
 311           }
 312           if (stable_dimension > 0) {
 313             assert(FoldStableValues, "sanity");
 314             assert(!con_type->is_zero_type(), "default value for stable field");
 315             con_type = con_type->is_aryptr()->cast_to_stable(true, stable_dimension);
 316           }
 317         }
 318         if (is_narrow_oop) {
 319           con_type = con_type->make_narrowoop();
 320         }
 321         return con_type;
 322       }
 323     case T_ILLEGAL:
 324       // Invalid ciConstant returned due to OutOfMemoryError in the CI
 325       assert(Compile::current()->env()->failing(), "otherwise should not see this");
 326       return NULL;
 327     default:
 328       // Fall through to failure
 329       return NULL;
 330   }
 331 }

 333 static ciConstant check_mismatched_access(ciConstant con, BasicType loadbt, bool is_unsigned) {
 334   BasicType conbt = con.basic_type();
 335   switch (conbt) {
 336     case T_BOOLEAN: conbt = T_BYTE;   break;
 337     case T_ARRAY:   conbt = T_OBJECT; break;
 338     case T_INLINE_TYPE: conbt = T_OBJECT; break;
 339     default:                          break;
 340   }
 341   switch (loadbt) {
 342     case T_BOOLEAN:   loadbt = T_BYTE;   break;
 343     case T_NARROWOOP: loadbt = T_OBJECT; break;
 344     case T_ARRAY:     loadbt = T_OBJECT; break;
 345     case T_INLINE_TYPE: loadbt = T_OBJECT; break;
 346     case T_ADDRESS:   loadbt = T_OBJECT; break;
 347     default:                             break;
 348   }
 349   if (conbt == loadbt) {
 350     if (is_unsigned && conbt == T_BYTE) {
 351       // LoadB (T_BYTE) with a small mask (<=8-bit) is converted to LoadUB (T_BYTE).
 352       return ciConstant(T_INT, con.as_int() & 0xFF);
 353     } else {
 354       return con;
 355     }
 356   }
 357   if (conbt == T_SHORT && loadbt == T_CHAR) {
 358     // LoadS (T_SHORT) with a small mask (<=16-bit) is converted to LoadUS (T_CHAR).
 359     return ciConstant(T_INT, con.as_int() & 0xFFFF);
 360   }
 361   return ciConstant(); // T_ILLEGAL
 362 }

 364 // Try to constant-fold a stable array element.
 365 const Type* Type::make_constant_from_array_element(ciArray* array, int off, int stable_dimension,
 366                                                    BasicType loadbt, bool is_unsigned_load) {
 367   // Decode the results of GraphKit::array_element_address.
 368   ciConstant element_value = array->element_value_by_offset(off);
 369   if (element_value.basic_type() == T_ILLEGAL) {
 370     return NULL; // wrong offset
 371   }
 372   ciConstant con = check_mismatched_access(element_value, loadbt, is_unsigned_load);
 373 
 374   assert(con.basic_type() != T_ILLEGAL, "elembt=%s; loadbt=%s; unsigned=%d",
 375          type2name(element_value.basic_type()), type2name(loadbt), is_unsigned_load);
 376 
 377   if (con.is_valid() &&          // not a mismatched access
 378       !con.is_null_or_zero()) {  // not a default value
 379     bool is_narrow_oop = (loadbt == T_NARROWOOP);
 380     return Type::make_from_constant(con, /*require_constant=*/true, stable_dimension, is_narrow_oop, /*is_autobox_cache=*/false);
 381   }
 382   return NULL;
 383 }

 385 const Type* Type::make_constant_from_field(ciInstance* holder, int off, bool is_unsigned_load, BasicType loadbt) {
 386   ciField* field;
 387   ciType* type = holder->java_mirror_type();
 388   if (type != NULL && type->is_instance_klass() && off >= InstanceMirrorKlass::offset_of_static_fields()) {
 389     // Static field
 390     field = type->as_instance_klass()->get_field_by_offset(off, /*is_static=*/true);
 391   } else {
 392     // Instance field
 393     field = holder->klass()->as_instance_klass()->get_field_by_offset(off, /*is_static=*/false);
 394   }
 395   if (field == NULL) {
 396     return NULL; // Wrong offset
 397   }
 398   return Type::make_constant_from_field(field, holder, loadbt, is_unsigned_load);
 399 }

 401 const Type* Type::make_constant_from_field(ciField* field, ciInstance* holder,
 402                                            BasicType loadbt, bool is_unsigned_load) {
 403   if (!field->is_constant()) {
 404     return NULL; // Non-constant field
 405   }
 406   ciConstant field_value;
 407   if (field->is_static()) {
 408     // final static field
 409     field_value = field->constant_value();
 410   } else if (holder != NULL) {
 411     // final or stable non-static field
 412     // Treat final non-static fields of trusted classes (classes in
 413     // java.lang.invoke and sun.invoke packages and subpackages) as
 414     // compile time constants.
 415     field_value = field->constant_value_of(holder);
 416   }
 417   if (!field_value.is_valid()) {
 418     return NULL; // Not a constant
 419   }
 420 
 421   ciConstant con = check_mismatched_access(field_value, loadbt, is_unsigned_load);
 422 
 423   assert(con.is_valid(), "elembt=%s; loadbt=%s; unsigned=%d",
 424          type2name(field_value.basic_type()), type2name(loadbt), is_unsigned_load);
 425 
 426   bool is_stable_array = FoldStableValues && field->is_stable() && field->type()->is_array_klass();
 427   int stable_dimension = (is_stable_array ? field->type()->as_array_klass()->dimension() : 0);
 428   bool is_narrow_oop = (loadbt == T_NARROWOOP);
 429 
 430   const Type* con_type = make_from_constant(con, /*require_constant=*/ true,
 431                                             stable_dimension, is_narrow_oop,
 432                                             field->is_autobox_cache());
 433   if (con_type != NULL && field->is_call_site_target()) {
 434     ciCallSite* call_site = holder->as_call_site();
 435     if (!call_site->is_fully_initialized_constant_call_site()) {
 436       ciMethodHandle* target = con.as_object()->as_method_handle();
 437       Compile::current()->dependencies()->assert_call_site_target_value(call_site, target);
 438     }
 439   }
 440   return con_type;
 441 }

 474 //--------------------------Initialize_shared----------------------------------
 475 void Type::Initialize_shared(Compile* current) {
 476   // This method does not need to be locked because the first system
 477   // compilations (stub compilations) occur serially.  If they are
 478   // changed to proceed in parallel, then this section will need
 479   // locking.
 480 
 481   Arena* save = current->type_arena();
 482   Arena* shared_type_arena = new (mtCompiler)Arena(mtCompiler);
 483 
 484   current->set_type_arena(shared_type_arena);
 485   _shared_type_dict =
 486     new (shared_type_arena) Dict( (CmpKey)Type::cmp, (Hash)Type::uhash,
 487                                   shared_type_arena, 128 );
 488   current->set_type_dict(_shared_type_dict);
 489 
 490   // Make shared pre-built types.
 491   CONTROL = make(Control);      // Control only
 492   TOP     = make(Top);          // No values in set
 493   MEMORY  = make(Memory);       // Abstract store only
 494   ABIO    = make(Abio);         // State-of-machine only
 495   RETURN_ADDRESS=make(Return_Address);
 496   FLOAT   = make(FloatBot);     // All floats
 497   DOUBLE  = make(DoubleBot);    // All doubles

 766 //------------------------------hashcons---------------------------------------
 767 // Do the hash-cons trick.  If the Type already exists in the type table,
 768 // delete the current Type and return the existing Type.  Otherwise stick the
 769 // current Type in the Type table.
 770 const Type *Type::hashcons(void) {
 771   debug_only(base());           // Check the assertion in Type::base().
 772   // Look up the Type in the Type dictionary
 773   Dict *tdic = type_dict();
 774   Type* old = (Type*)(tdic->Insert(this, this, false));
 775   if( old ) {                   // Pre-existing Type?
 776     if( old != this )           // Yes, this guy is not the pre-existing?
 777       delete this;              // Yes, Nuke this guy
 778     assert( old->_dual, "" );
 779     return old;                 // Return pre-existing
 780   }
 781 
 782   // Every type has a dual (to make my lattice symmetric).
 783   // Since we just discovered a new Type, compute its dual right now.
 784   assert( !_dual, "" );         // No dual yet
 785   _dual = xdual();              // Compute the dual
 786   if (cmp(this, _dual) == 0) {  // Handle self-symmetric
 787     if (_dual != this) {
 788       delete _dual;
 789       _dual = this;
 790     }
 791     return this;
 792   }
 793   assert( !_dual->_dual, "" );  // No reverse dual yet
 794   assert( !(*tdic)[_dual], "" ); // Dual not in type system either
 795   // New Type, insert into Type table
 796   tdic->Insert((void*)_dual,(void*)_dual);
 797   ((Type*)_dual)->_dual = this; // Finish up being symmetric
 798 #ifdef ASSERT
 799   Type *dual_dual = (Type*)_dual->xdual();
 800   assert( eq(dual_dual), "xdual(xdual()) should be identity" );
 801   delete dual_dual;
 802 #endif
 803   return this;                  // Return new Type
 804 }

 904 //------------------------------meet-------------------------------------------
 905 // Compute the MEET of two types.  NOT virtual.  It enforces that meet is
 906 // commutative and the lattice is symmetric.
 907 const Type *Type::meet_helper(const Type *t, bool include_speculative) const {
 908   if (isa_narrowoop() && t->isa_narrowoop()) {
 909     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 910     return result->make_narrowoop();
 911   }
 912   if (isa_narrowklass() && t->isa_narrowklass()) {
 913     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 914     return result->make_narrowklass();
 915   }
 916 
 917   const Type *this_t = maybe_remove_speculative(include_speculative);
 918   t = t->maybe_remove_speculative(include_speculative);
 919 
 920   const Type *mt = this_t->xmeet(t);
 921 #ifdef ASSERT
 922   if (isa_narrowoop() || t->isa_narrowoop()) return mt;
 923   if (isa_narrowklass() || t->isa_narrowklass()) return mt;
 924   Compile* C = Compile::current();
 925   if (!C->_type_verify_symmetry) {
 926     return mt;
 927   }
 928   this_t->check_symmetrical(t, mt);
 929   // In the case of an array, computing the meet above, caused the
 930   // computation of the meet of the elements which at verification
 931   // time caused the computation of the meet of the dual of the
 932   // elements. Computing the meet of the dual of the arrays here
 933   // causes the meet of the dual of the elements to be computed which
 934   // would cause the meet of the dual of the dual of the elements,
 935   // that is the meet of the elements already computed above to be
 936   // computed. Avoid redundant computations by requesting no
 937   // verification.
 938   C->_type_verify_symmetry = false;
 939   const Type *mt_dual = this_t->_dual->xmeet(t->_dual);
 940   this_t->_dual->check_symmetrical(t->_dual, mt_dual);
 941   assert(!C->_type_verify_symmetry, "shouldn't have changed");
 942   C->_type_verify_symmetry = true;
 943 #endif
 944   return mt;
 945 }

 947 //------------------------------xmeet------------------------------------------
 948 // Compute the MEET of two types.  It returns a new Type object.
 949 const Type *Type::xmeet( const Type *t ) const {
 950   // Perform a fast test for common case; meeting the same types together.
 951   if( this == t ) return this;  // Meeting same type-rep?
 952 
 953   // Meeting TOP with anything?
 954   if( _base == Top ) return t;
 955 
 956   // Meeting BOTTOM with anything?
 957   if( _base == Bottom ) return BOTTOM;
 958 
 959   // Current "this->_base" is one of: Bad, Multi, Control, Top,
 960   // Abio, Abstore, Floatxxx, Doublexxx, Bottom, lastype.
 961   switch (t->base()) {  // Switch on original type
 962 
 963   // Cut in half the number of cases I must handle.  Only need cases for when
 964   // the given enum "t->type" is less than or equal to the local enum "type".
 965   case FloatCon:
 966   case DoubleCon:
 967   case Int:
 968   case Long:
 969     return t->xmeet(this);
 970 
 971   case OopPtr:
 972     return t->xmeet(this);
 973 
 974   case InstPtr:
 975     return t->xmeet(this);
 976 
 977   case MetadataPtr:
 978   case KlassPtr:
 979     return t->xmeet(this);
 980 
 981   case AryPtr:
 982     return t->xmeet(this);
 983 
 984   case NarrowOop:
 985     return t->xmeet(this);
 986 
 987   case NarrowKlass:
 988     return t->xmeet(this);
 989 
 990   case InlineType:
 991     return t->xmeet(this);
 992 
 993   case Bad:                     // Type check
 994   default:                      // Bogus type not in lattice
 995     typerr(t);
 996     return Type::BOTTOM;
 997 
 998   case Bottom:                  // Ye Olde Default
 999     return t;
1000 
1001   case FloatTop:
1002     if( _base == FloatTop ) return this;
1003   case FloatBot:                // Float
1004     if( _base == FloatBot || _base == FloatTop ) return FLOAT;
1005     if( _base == DoubleTop || _base == DoubleBot ) return Type::BOTTOM;
1006     typerr(t);
1007     return Type::BOTTOM;
1008 
1009   case DoubleTop:
1010     if( _base == DoubleTop ) return this;
1011   case DoubleBot:               // Double
1012     if( _base == DoubleBot || _base == DoubleTop ) return DOUBLE;
1013     if( _base == FloatTop || _base == FloatBot ) return Type::BOTTOM;
1014     typerr(t);
1015     return Type::BOTTOM;
1016 
1017   // These next few cases must match exactly or it is a compile-time error.
1018   case Control:                 // Control of code
1019   case Abio:                    // State of world outside of program
1020   case Memory:
1021     if( _base == t->_base )  return this;
1022     typerr(t);
1023     return Type::BOTTOM;
1024 
1025   case Top:                     // Top of the lattice
1026     return this;
1027   }
1028 
1029   // The type is unchanged
1030   return this;
1031 }

1495 //------------------------------widen------------------------------------------
1496 // Only happens for optimistic top-down optimizations.
1497 const Type *TypeInt::widen( const Type *old, const Type* limit ) const {
1498   // Coming from TOP or such; no widening
1499   if( old->base() != Int ) return this;
1500   const TypeInt *ot = old->is_int();
1501 
1502   // If new guy is equal to old guy, no widening
1503   if( _lo == ot->_lo && _hi == ot->_hi )
1504     return old;
1505 
1506   // If new guy contains old, then we widened
1507   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1508     // New contains old
1509     // If new guy is already wider than old, no widening
1510     if( _widen > ot->_widen ) return this;
1511     // If old guy was a constant, do not bother
1512     if (ot->_lo == ot->_hi)  return this;
1513     // Now widen new guy.
1514     // Check for widening too far
1515     if (_widen == WidenMax) {
1516       int max = max_jint;
1517       int min = min_jint;
1518       if (limit->isa_int()) {
1519         max = limit->is_int()->_hi;
1520         min = limit->is_int()->_lo;
1521       }
1522       if (min < _lo && _hi < max) {
1523         // If neither endpoint is extremal yet, push out the endpoint
1524         // which is closer to its respective limit.
1525         if (_lo >= 0 ||                 // easy common case
1526             (juint)(_lo - min) >= (juint)(max - _hi)) {
1527           // Try to widen to an unsigned range type of 31 bits:
1528           return make(_lo, max, WidenMax);
1529         } else {
1530           return make(min, _hi, WidenMax);
1531         }
1532       }
1533       return TypeInt::INT;
1534     }
1535     // Returned widened new guy
1536     return make(_lo,_hi,_widen+1);
1537   }
1538 
1539   // If old guy contains new, then we probably widened too far & dropped to
1540   // bottom.  Return the wider fellow.
1541   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1542     return old;
1543 
1544   //fatal("Integer value range is not subset");
1545   //return this;
1546   return TypeInt::INT;
1547 }

1549 //------------------------------narrow---------------------------------------
1550 // Only happens for pessimistic optimizations.
1551 const Type *TypeInt::narrow( const Type *old ) const {
1552   if (_lo >= _hi)  return this;   // already narrow enough
1553   if (old == NULL)  return this;
1554   const TypeInt* ot = old->isa_int();
1555   if (ot == NULL)  return this;
1556   jint olo = ot->_lo;
1557   jint ohi = ot->_hi;
1558 
1559   // If new guy is equal to old guy, no narrowing
1560   if (_lo == olo && _hi == ohi)  return old;
1561 
1562   // If old guy was maximum range, allow the narrowing
1563   if (olo == min_jint && ohi == max_jint)  return this;
1564 
1565   if (_lo < olo || _hi > ohi)
1566     return this;                // doesn't narrow; pretty wierd
1567 
1568   // The new type narrows the old type, so look for a "death march".
1569   // See comments on PhaseTransform::saturate.
1570   juint nrange = (juint)_hi - _lo;
1571   juint orange = (juint)ohi - olo;

1708 // with reference count equal to the number of Types pointing at it.
1709 // Caller should wrap a Types around it.
1710 const Type *TypeLong::xmeet( const Type *t ) const {
1711   // Perform a fast test for common case; meeting the same types together.
1712   if( this == t ) return this;  // Meeting same type?
1713 
1714   // Currently "this->_base" is a TypeLong
1715   switch (t->base()) {          // Switch on original type
1716   case AnyPtr:                  // Mixing with oops happens when javac
1717   case RawPtr:                  // reuses local variables
1718   case OopPtr:
1719   case InstPtr:
1720   case AryPtr:
1721   case MetadataPtr:
1722   case KlassPtr:
1723   case NarrowOop:
1724   case NarrowKlass:

1728   case FloatBot:

1735     typerr(t);

1737     return this;
1738   case Long:                    // Long vs Long?
1739     break;
1740   }

1742   // Expand covered set

1754 //------------------------------widen------------------------------------------
1755 // Only happens for optimistic top-down optimizations.
1756 const Type *TypeLong::widen( const Type *old, const Type* limit ) const {
1757   // Coming from TOP or such; no widening
1758   if( old->base() != Long ) return this;
1759   const TypeLong *ot = old->is_long();
1760 
1761   // If new guy is equal to old guy, no widening
1762   if( _lo == ot->_lo && _hi == ot->_hi )
1763     return old;
1764 
1765   // If new guy contains old, then we widened
1766   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1767     // New contains old
1768     // If new guy is already wider than old, no widening
1769     if( _widen > ot->_widen ) return this;
1770     // If old guy was a constant, do not bother
1771     if (ot->_lo == ot->_hi)  return this;
1772     // Now widen new guy.
1773     // Check for widening too far
1774     if (_widen == WidenMax) {
1775       jlong max = max_jlong;
1776       jlong min = min_jlong;
1777       if (limit->isa_long()) {
1778         max = limit->is_long()->_hi;
1779         min = limit->is_long()->_lo;
1780       }
1781       if (min < _lo && _hi < max) {
1782         // If neither endpoint is extremal yet, push out the endpoint
1783         // which is closer to its respective limit.
1784         if (_lo >= 0 ||                 // easy common case
1785             ((julong)_lo - min) >= ((julong)max - _hi)) {
1786           // Try to widen to an unsigned range type of 32/63 bits:
1787           if (max >= max_juint && _hi < max_juint)
1788             return make(_lo, max_juint, WidenMax);
1789           else
1790             return make(_lo, max, WidenMax);
1791         } else {
1792           return make(min, _hi, WidenMax);
1793         }
1794       }
1795       return TypeLong::LONG;
1796     }
1797     // Returned widened new guy
1798     return make(_lo,_hi,_widen+1);
1799   }
1800 
1801   // If old guy contains new, then we probably widened too far & dropped to
1802   // bottom.  Return the wider fellow.
1803   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1804     return old;
1805 
1806   //  fatal("Long value range is not subset");
1807   // return this;
1808   return TypeLong::LONG;
1809 }

1811 //------------------------------narrow----------------------------------------
1812 // Only happens for pessimistic optimizations.
1813 const Type *TypeLong::narrow( const Type *old ) const {
1814   if (_lo >= _hi)  return this;   // already narrow enough
1815   if (old == NULL)  return this;
1816   const TypeLong* ot = old->isa_long();
1817   if (ot == NULL)  return this;
1818   jlong olo = ot->_lo;
1819   jlong ohi = ot->_hi;
1820 
1821   // If new guy is equal to old guy, no narrowing
1822   if (_lo == olo && _hi == ohi)  return old;
1823 
1824   // If old guy was maximum range, allow the narrowing
1825   if (olo == min_jlong && ohi == max_jlong)  return this;
1826 
1827   if (_lo < olo || _hi > ohi)
1828     return this;                // doesn't narrow; pretty wierd
1829 
1830   // The new type narrows the old type, so look for a "death march".
1831   // See comments on PhaseTransform::saturate.
1832   julong nrange = _hi - _lo;
1833   julong orange = ohi - olo;
1834   if (nrange < max_julong - 1 && nrange > (orange >> 1) + (SMALLINT*2)) {
1835     // Use the new type only if the range shrinks a lot.
1836     // We do not want the optimizer computing 2^31 point by point.
1837     return old;
1838   }
1839 
1840   return this;
1841 }

3121 const Type *TypeRawPtr::xmeet( const Type *t ) const {
3122   // Perform a fast test for common case; meeting the same types together.
3123   if( this == t ) return this;  // Meeting same type-rep?
3124

3126   switch( t->base() ) {         // switch on original type
3127   case Bottom:                  // Ye Olde Default
3128     return t;
3129   case Top:
3130     return this;
3131   case AnyPtr:                  // Meeting to AnyPtrs
3132     break;
3133   case RawPtr: {                // might be top, bot, any/not or constant
3134     enum PTR tptr = t->is_ptr()->ptr();

3137       if( tptr == Constant && _ptr != Constant)  return t;

3435 //--------------------------make_from_klass_common-----------------------------
3436 // Computes the element-type given a klass.
3437 const TypeOopPtr* TypeOopPtr::make_from_klass_common(ciKlass *klass, bool klass_change, bool try_for_exact) {
3438   if (klass->is_instance_klass() || klass->is_inlinetype()) {
3439     Compile* C = Compile::current();
3440     Dependencies* deps = C->dependencies();
3441     assert((deps != NULL) == (C->method() != NULL && C->method()->code_size() > 0), "sanity");
3442     // Element is an instance
3443     bool klass_is_exact = false;
3444     if (klass->is_loaded()) {
3445       // Try to set klass_is_exact.
3446       ciInstanceKlass* ik = klass->as_instance_klass();
3447       klass_is_exact = ik->is_final();
3448       if (!klass_is_exact && klass_change
3449           && deps != NULL && UseUniqueSubclasses) {
3450         ciInstanceKlass* sub = ik->unique_concrete_subklass();
3451         if (sub != NULL) {
3452           deps->assert_abstract_with_unique_concrete_subtype(ik, sub);
3453           klass = ik = sub;
3454           klass_is_exact = sub->is_final();
3455         }
3456       }
3457       if (!klass_is_exact && try_for_exact && deps != NULL &&
3458           !ik->is_interface() && !ik->has_subklass()) {
3459         // Add a dependence; if concrete subclass added we need to recompile
3460         deps->assert_leaf_type(ik);
3461         klass_is_exact = true;
3462       }
3463     }
3464     return TypeInstPtr::make(TypePtr::BotPTR, klass, klass_is_exact, NULL, Offset(0), klass->flatten_array());
3465   } else if (klass->is_obj_array_klass()) {
3466     // Element is an object or inline type array. Recursively call ourself.
3467     const TypeOopPtr* etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ false, try_for_exact);
3468     if (etype->is_inlinetypeptr()) {
3469       etype = etype->join_speculative(TypePtr::NOTNULL)->is_oopptr();
3470     }
3471     // Determine null-free/flattened properties
3472     const TypeOopPtr* exact_etype = etype;
3473     if (etype->can_be_inline_type()) {
3474       // Use exact type if element can be an inline type
3475       exact_etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ true, /* try_for_exact= */ true);
3476     }
3477     bool not_null_free = !exact_etype->can_be_inline_type();
3478     bool not_flat = !UseFlatArray || not_null_free || (exact_etype->is_inlinetypeptr() && !exact_etype->inline_klass()->flatten_array());
3479 
3480     bool xk = etype->klass_is_exact();
3481     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS, false, not_flat, not_null_free);
3482     // We used to pass NotNull in here, asserting that the sub-arrays
3483     // are all not-null.  This is not true in generally, as code can
3484     // slam NULLs down in the subarrays.
3485     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, xk, Offset(0));
3486     return arr;
3487   } else if (klass->is_type_array_klass()) {
3488     // Element is an typeArray
3489     const Type* etype = get_const_basic_type(klass->as_type_array_klass()->element_type());
3490     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS,
3491                                         /* stable= */ false, /* not_flat= */ true, /* not_null_free= */ true);
3492     // We used to pass NotNull in here, asserting that the array pointer
3493     // is not-null. That was not true in general.
3494     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3495     return arr;
3496   } else if (klass->is_flat_array_klass()) {
3497     ciInlineKlass* vk = klass->as_array_klass()->element_klass()->as_inline_klass();
3498     const TypeAry* arr0 = TypeAry::make(TypeInlineType::make(vk), TypeInt::POS);
3499     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3500     return arr;
3501   } else {
3502     ShouldNotReachHere();
3503     return NULL;
3504   }
3505 }

3590 //-----------------------------filter------------------------------------------
3591 // Do not allow interface-vs.-noninterface joins to collapse to top.
3592 const Type *TypeOopPtr::filter_helper(const Type *kills, bool include_speculative) const {
3593 
3594   const Type* ft = join_helper(kills, include_speculative);
3595   const TypeInstPtr* ftip = ft->isa_instptr();
3596   const TypeInstPtr* ktip = kills->isa_instptr();
3597 
3598   if (ft->empty()) {
3599     // Check for evil case of 'this' being a class and 'kills' expecting an
3600     // interface.  This can happen because the bytecodes do not contain
3601     // enough type info to distinguish a Java-level interface variable
3602     // from a Java-level object variable.  If we meet 2 classes which
3603     // both implement interface I, but their meet is at 'j/l/O' which
3604     // doesn't implement I, we have no way to tell if the result should
3605     // be 'I' or 'j/l/O'.  Thus we'll pick 'j/l/O'.  If this then flows
3606     // into a Phi which "knows" it's an Interface type we'll have to
3607     // uplift the type.
3608     if (!empty()) {
3609       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3610         return kills;           // Uplift to interface
3611       }
3612       // Also check for evil cases of 'this' being a class array
3613       // and 'kills' expecting an array of interfaces.
3614       Type::get_arrays_base_elements(ft, kills, NULL, &ktip);
3615       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3616         return kills;           // Uplift to array of interface
3617       }
3618     }
3619 
3620     return Type::TOP;           // Canonical empty value
3621   }
3622 
3623   // If we have an interface-typed Phi or cast and we narrow to a class type,
3624   // the join should report back the class.  However, if we have a J/L/Object
3625   // class-typed Phi and an interface flows in, it's possible that the meet &
3626   // join report an interface back out.  This isn't possible but happens
3627   // because the type system doesn't interact well with interfaces.
3628   if (ftip != NULL && ktip != NULL &&
3629       ftip->is_loaded() &&  ftip->klass()->is_interface() &&
3630       ktip->is_loaded() && !ktip->klass()->is_interface()) {
3631     assert(!ftip->klass_is_exact(), "interface could not be exact");
3632     return ktip->cast_to_ptr_type(ftip->ptr());
3633   }
3634 
3635   return ft;
3636 }

3870 //------------------------------xmeet_unloaded---------------------------------
3871 // Compute the MEET of two InstPtrs when at least one is unloaded.
3872 // Assume classes are different since called after check for same name/class-loader
3873 const TypeInstPtr *TypeInstPtr::xmeet_unloaded(const TypeInstPtr *tinst) const {
3874     Offset off = meet_offset(tinst->offset());
3875     PTR ptr = meet_ptr(tinst->ptr());
3876     int instance_id = meet_instance_id(tinst->instance_id());
3877     const TypePtr* speculative = xmeet_speculative(tinst);
3878     int depth = meet_inline_depth(tinst->inline_depth());
3879 
3880     const TypeInstPtr *loaded    = is_loaded() ? this  : tinst;
3881     const TypeInstPtr *unloaded  = is_loaded() ? tinst : this;
3882     if( loaded->klass()->equals(ciEnv::current()->Object_klass()) ) {
3883       //
3884       // Meet unloaded class with java/lang/Object
3885       //
3886       // Meet
3887       //          |                     Unloaded Class
3888       //  Object  |   TOP    |   AnyNull | Constant |   NotNull |  BOTTOM   |
3889       //  ===================================================================
3890       //   TOP    | ..........................Unloaded......................|
3891       //  AnyNull |  U-AN    |................Unloaded......................|
3892       // Constant | ... O-NN .................................. |   O-BOT   |
3893       //  NotNull | ... O-NN .................................. |   O-BOT   |
3894       //  BOTTOM  | ........................Object-BOTTOM ..................|
3895       //

3923   if( this == t ) return this;  // Meeting same type-rep?

3927 
3928   case Int:                     // Mixing ints & oops happens when javac
3929   case Long:                    // reuses local variables
3930   case FloatTop:
3931   case FloatCon:
3932   case FloatBot:
3933   case DoubleTop:
3934   case DoubleCon:

3936   case NarrowOop:

3971     case Constant:
3972     case NotNull:

4218       // NotNull if they do (neither constant is NULL; that is a special case

4229         else if (above_centerline(tinst ->_ptr))

4231         else

4238     // class hierarchy - which means we have to fall to at least NotNull.

4247   } // End of case InstPtr
4248

4254       } else {

4466 //-----------------------------narrow_size_type-------------------------------
4467 // Narrow the given size type to the index range for the given array base type.
4468 // Return NULL if the resulting int type becomes empty.
4469 const TypeInt* TypeAryPtr::narrow_size_type(const TypeInt* size) const {
4470   jint hi = size->_hi;
4471   jint lo = size->_lo;
4472   jint min_lo = 0;
4473   jint max_hi = max_array_length(elem()->basic_type());
4474   //if (index_not_size)  --max_hi;     // type of a valid array index, FTR
4475   bool chg = false;
4476   if (lo < min_lo) {
4477     lo = min_lo;
4478     if (size->is_con()) {
4479       hi = lo;
4480     }
4481     chg = true;
4482   }
4483   if (hi > max_hi) {
4484     hi = max_hi;
4485     if (size->is_con()) {
4486       lo = hi;
4487     }
4488     chg = true;
4489   }
4490   // Negative length arrays will produce weird intermediate dead fast-path code
4491   if (lo > hi)
4492     return TypeInt::ZERO;
4493   if (!chg)
4494     return size;
4495   return TypeInt::make(lo, hi, Type::WidenMin);
4496 }

4582 //------------------------------meet-------------------------------------------
4583 // Compute the MEET of two types.  It returns a new Type object.
4584 const Type *TypeAryPtr::xmeet_helper(const Type *t) const {
4585   // Perform a fast test for common case; meeting the same types together.
4586   if( this == t ) return this;  // Meeting same type-rep?
4587   // Current "this->_base" is Pointer
4588   switch (t->base()) {          // switch on original type
4589 
4590   // Mixing ints & oops happens when javac reuses local variables
4591   case Int:
4592   case Long:
4593   case FloatTop:
4594   case FloatCon:
4595   case FloatBot:
4596   case DoubleTop:
4597   case DoubleCon:
4598   case DoubleBot:
4599   case NarrowOop:
4600   case NarrowKlass:
4601   case Bottom:                  // Ye Olde Default
4602     return Type::BOTTOM;
4603   case Top:
4604     return this;
4605 
4606   default:                      // All else is a mistake
4607     typerr(t);
4608 
4609   case OopPtr: {                // Meeting to OopPtrs
4610     // Found a OopPtr type vs self-AryPtr type
4611     const TypeOopPtr *tp = t->is_oopptr();
4612     Offset offset = meet_offset(tp->offset());
4613     PTR ptr = meet_ptr(tp->ptr());
4614     int depth = meet_inline_depth(tp->inline_depth());
4615     const TypePtr* speculative = xmeet_speculative(tp);
4616     switch (tp->ptr()) {
4617     case TopPTR:
4618     case AnyNull: {
4619       int instance_id = meet_instance_id(InstanceTop);
4620       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4621                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4622     }
4623     case BotPTR:
4624     case NotNull: {
4625       int instance_id = meet_instance_id(tp->instance_id());
4626       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4627     }
4628     default: ShouldNotReachHere();
4629     }
4630   }
4631 
4632   case AnyPtr: {                // Meeting two AnyPtrs
4633     // Found an AnyPtr type vs self-AryPtr type
4634     const TypePtr *tp = t->is_ptr();
4635     Offset offset = meet_offset(tp->offset());
4636     PTR ptr = meet_ptr(tp->ptr());
4637     const TypePtr* speculative = xmeet_speculative(tp);
4638     int depth = meet_inline_depth(tp->inline_depth());
4639     switch (tp->ptr()) {
4640     case TopPTR:
4641       return this;
4642     case BotPTR:
4643     case NotNull:
4644       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4645     case Null:
4646       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4647       // else fall through to AnyNull
4648     case AnyNull: {
4649       int instance_id = meet_instance_id(InstanceTop);
4650       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4651                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4652     }
4653     default: ShouldNotReachHere();
4654     }
4655   }
4656 
4657   case MetadataPtr:
4658   case KlassPtr:
4659   case RawPtr: return TypePtr::BOTTOM;
4660 
4661   case AryPtr: {                // Meeting 2 references?
4662     const TypeAryPtr *tap = t->is_aryptr();
4663     Offset off = meet_offset(tap->offset());
4664     Offset field_off = meet_field_offset(tap->field_offset());
4665     const TypeAry *tary = _ary->meet_speculative(tap->_ary)->is_ary();
4666     PTR ptr = meet_ptr(tap->ptr());
4667     int instance_id = meet_instance_id(tap->instance_id());
4668     const TypePtr* speculative = xmeet_speculative(tap);
4669     int depth = meet_inline_depth(tap->inline_depth());
4670     ciKlass* lazy_klass = NULL;
4671     if (tary->_elem->isa_int()) {
4672       // Integral array element types have irrelevant lattice relations.
4673       // It is the klass that determines array layout, not the element type.
4674       if (_klass == NULL)
4675         lazy_klass = tap->_klass;
4676       else if (tap->_klass == NULL || tap->_klass == _klass) {
4677         lazy_klass = _klass;
4678       } else {
4679         // Something like byte[int+] meets char[int+].
4680         // This must fall to bottom, not (int[-128..65535])[int+].
4681         instance_id = InstanceBot;
4682         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4683       }
4684     } else // Non integral arrays.
4685       // Must fall to bottom if exact klasses in upper lattice
4686       // are not equal or super klass is exact.
4687       if ((above_centerline(ptr) || ptr == Constant) && klass() != tap->klass() &&
4688           // meet with top[] and bottom[] are processed further down:
4689           tap->_klass != NULL && this->_klass != NULL &&
4690           // both are exact and not equal:
4691           ((tap->_klass_is_exact && this->_klass_is_exact) ||
4692            // 'tap' is exact and super or unrelated:
4693            (tap->_klass_is_exact && !tap->klass()->is_subtype_of(klass())) ||
4694            // 'this' is exact and super or unrelated:
4695            (this->_klass_is_exact && !klass()->is_subtype_of(tap->klass())))) {
4696       if (above_centerline(ptr) || (tary->_elem->make_ptr() && above_centerline(tary->_elem->make_ptr()->_ptr))) {
4697         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4698       }
4699       return make(NotNull, NULL, tary, lazy_klass, false, off, field_off, InstanceBot, speculative, depth);
4700     } else if (klass() != NULL && tap->klass() != NULL && klass()->is_flat_array_klass() != tap->klass()->is_flat_array_klass()) {
4701       // Meeting flattened inline type array with non-flattened array. Adjust (field) offset accordingly.
4702       if (tary->_elem->isa_inlinetype()) {
4703         // Result is flattened
4704         off = Offset(elem()->isa_inlinetype() ? offset() : tap->offset());
4705         field_off = elem()->isa_inlinetype() ? field_offset() : tap->field_offset();
4706       } else if (tary->_elem->make_oopptr() != NULL && tary->_elem->make_oopptr()->isa_instptr() && below_centerline(ptr)) {
4707         // Result is non-flattened
4708         off = Offset(flattened_offset()).meet(Offset(tap->flattened_offset()));
4709         field_off = Offset::bottom;
4710       }
4711     }
4712 
4713     bool xk = false;
4714     switch (tap->ptr()) {
4715     case AnyNull:
4716     case TopPTR:
4717       // Compute new klass on demand, do not use tap->_klass
4718       if (below_centerline(this->_ptr)) {
4719         xk = this->_klass_is_exact;
4720       } else {
4721         xk = (tap->_klass_is_exact || this->_klass_is_exact);
4722       }
4723       return make(ptr, const_oop(), tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4724     case Constant: {
4725       ciObject* o = const_oop();
4726       if( _ptr == Constant ) {
4727         if( tap->const_oop() != NULL && !o->equals(tap->const_oop()) ) {
4728           xk = (klass() == tap->klass());
4729           ptr = NotNull;
4730           o = NULL;
4731           instance_id = InstanceBot;
4732         } else {
4733           xk = true;
4734         }
4735       } else if(above_centerline(_ptr)) {
4736         o = tap->const_oop();
4737         xk = true;
4738       } else {
4739         // Only precise for identical arrays
4740         xk = this->_klass_is_exact && (klass() == tap->klass());
4741       }
4742       return TypeAryPtr::make(ptr, o, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4743     }
4744     case NotNull:
4745     case BotPTR:
4746       // Compute new klass on demand, do not use tap->_klass
4747       if (above_centerline(this->_ptr))
4748             xk = tap->_klass_is_exact;
4749       else  xk = (tap->_klass_is_exact & this->_klass_is_exact) &&
4750               (klass() == tap->klass()); // Only precise for identical arrays
4751       return TypeAryPtr::make(ptr, NULL, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4752     default: ShouldNotReachHere();
4753     }
4754   }
4755 
4756   // All arrays inherit from Object class
4757   case InstPtr: {
4758     const TypeInstPtr *tp = t->is_instptr();
4759     Offset offset = meet_offset(tp->offset());
4760     PTR ptr = meet_ptr(tp->ptr());
4761     int instance_id = meet_instance_id(tp->instance_id());
4762     const TypePtr* speculative = xmeet_speculative(tp);
4763     int depth = meet_inline_depth(tp->inline_depth());
4764     switch (ptr) {
4765     case TopPTR:
4766     case AnyNull:                // Fall 'down' to dual of object klass
4767       // For instances when a subclass meets a superclass we fall
4768       // below the centerline when the superclass is exact. We need to
4769       // do the same here.
4770       if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact() && !tp->flat_array()) {
4771         return TypeAryPtr::make(ptr, _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4772       } else {
4773         // cannot subclass, so the meet has to fall badly below the centerline
4774         ptr = NotNull;
4775         instance_id = InstanceBot;
4776         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4777       }
4778     case Constant:
4779     case NotNull:
4780     case BotPTR:                // Fall down to object klass
4781       // LCA is object_klass, but if we subclass from the top we can do better
4782       if (above_centerline(tp->ptr())) {
4783         // If 'tp'  is above the centerline and it is Object class
4784         // then we can subclass in the Java class hierarchy.
4785         // For instances when a subclass meets a superclass we fall
4786         // below the centerline when the superclass is exact. We need
4787         // to do the same here.
4788         if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact() && !tp->flat_array()) {
4789           // that is, my array type is a subtype of 'tp' klass
4790           return make(ptr, (ptr == Constant ? const_oop() : NULL),
4791                       _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4792         }
4793       }
4794       // The other case cannot happen, since t cannot be a subtype of an array.
4795       // The meet falls down to Object class below centerline.
4796       if( ptr == Constant )
4797          ptr = NotNull;
4798       instance_id = InstanceBot;
4799       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4800     default: typerr(t);
4801     }
4802   }
4803 
4804   case InlineType: {
4805     const TypeInlineType* tv = t->is_inlinetype();
4806     if (above_centerline(ptr())) {
4807       return TypeInstPtr::NOTNULL;
4808     } else {
4809       PTR ptr = this->_ptr;
4810       if (ptr == Constant) {
4811         ptr = NotNull;
4812       }
4813       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4814     }
4815   }
4816   }
4817   return this;                  // Lint noise
4818 }

4938 const TypePtr* TypeAryPtr::add_field_offset_and_offset(intptr_t offset) const {
4939   int adj = 0;
4940   if (offset != Type::OffsetBot && offset != Type::OffsetTop) {
4941     const Type* elemtype = elem();
4942     if (elemtype->isa_inlinetype()) {
4943       if (_offset.get() != OffsetBot && _offset.get() != OffsetTop) {
4944         adj = _offset.get();
4945         offset += _offset.get();
4946       }
4947       uint header = arrayOopDesc::base_offset_in_bytes(T_OBJECT);
4948       if (_field_offset.get() != OffsetBot && _field_offset.get() != OffsetTop) {
4949         offset += _field_offset.get();
4950         if (_offset.get() == OffsetBot || _offset.get() == OffsetTop) {
4951           offset += header;
4952         }
4953       }
4954       if (offset >= (intptr_t)header || offset < 0) {
4955         // Try to get the field of the inline type array element we are pointing to
4956         ciKlass* arytype_klass = klass();
4957         ciFlatArrayKlass* vak = arytype_klass->as_flat_array_klass();
4958         ciInlineKlass* vk = vak->element_klass()->as_inline_klass();
4959         int shift = vak->log2_element_size();
4960         int mask = (1 << shift) - 1;
4961         intptr_t field_offset = ((offset - header) & mask);
4962         ciField* field = vk->get_field_by_offset(field_offset + vk->first_field_offset(), false);
4963         if (field == NULL) {
4964           // This may happen with nested AddP(base, AddP(base, base, offset), longcon(16))
4965           return add_offset(offset);
4966         } else {
4967           return with_field_offset(field_offset)->add_offset(offset - field_offset - adj);
4968         }
4969       }
4970     }
4971   }
4972   return add_offset(offset - adj);
4973 }

5214   if( this == t ) return this;  // Meeting same type-rep?
5215 
5216   // Current "this->_base" is OopPtr

5402 //----------------------compute_klass------------------------------------------
5403 // Compute the defining klass for this class
5404 ciKlass* TypeAryPtr::compute_klass(DEBUG_ONLY(bool verify)) const {
5405   // Compute _klass based on element type.
5406   ciKlass* k_ary = NULL;
5407   const TypeAryPtr *tary;
5408   const Type* el = elem();
5409   if (el->isa_narrowoop()) {
5410     el = el->make_ptr();
5411   }
5412 
5413   // Get element klass
5414   if (el->isa_instptr()) {
5415     // Compute object array klass from element klass
5416     k_ary = ciArrayKlass::make(el->is_oopptr()->klass());
5417   } else if (el->isa_inlinetype()) {
5418     // If element type is TypeInlineType::BOTTOM, inline_klass() will be null.
5419     if (el->inline_klass() != NULL) {
5420       k_ary = ciArrayKlass::make(el->inline_klass());
5421     }
5422   } else if ((tary = el->isa_aryptr()) != NULL) {
5423     // Compute array klass from element klass
5424     ciKlass* k_elem = tary->klass();
5425     // If element type is something like bottom[], k_elem will be null.
5426     if (k_elem != NULL)
5427       k_ary = ciObjArrayKlass::make(k_elem);
5428   } else if ((el->base() == Type::Top) ||
5429              (el->base() == Type::Bottom)) {
5430     // element type of Bottom occurs from meet of basic type
5431     // and object; Top occurs when doing join on Bottom.
5432     // Leave k_ary at NULL.
5433   } else {
5434     // Cannot compute array klass directly from basic type,
5435     // since subtypes of TypeInt all have basic type T_INT.
5436 #ifdef ASSERT
5437     if (verify && el->isa_int()) {
5438       // Check simple cases when verifying klass.
5439       BasicType bt = T_ILLEGAL;
5440       if (el == TypeInt::BYTE) {
5441         bt = T_BYTE;
5442       } else if (el == TypeInt::SHORT) {
5443         bt = T_SHORT;
5444       } else if (el == TypeInt::CHAR) {
5445         bt = T_CHAR;
5446       } else if (el == TypeInt::INT) {
5447         bt = T_INT;
5448       } else {
5449         return _klass; // just return specified klass
5450       }
5451       return ciTypeArrayKlass::make(bt);
5452     }
5453 #endif
5454     assert(!el->isa_int(),
5455            "integral arrays must be pre-equipped with a class");
5456     // Compute array klass directly from basic type
5457     k_ary = ciTypeArrayKlass::make(el->basic_type());
5458   }
5459   return k_ary;

5533 //------------------------------xmeet------------------------------------------
5534 // Compute the MEET of two types, return a new Type object.
5535 const Type    *TypeKlassPtr::xmeet( const Type *t ) const {
5536   // Perform a fast test for common case; meeting the same types together.
5537   if( this == t ) return this;  // Meeting same type-rep?
5538 
5539   // Current "this->_base" is Pointer
5540   switch (t->base()) {          // switch on original type
5541 
5542   case Int:                     // Mixing ints & oops happens when javac
5543   case Long:                    // reuses local variables
5544   case FloatTop:
5545   case FloatCon:
5546   case FloatBot:
5547   case DoubleTop:
5548   case DoubleCon:
5549   case DoubleBot:
5550   case NarrowOop:
5551   case NarrowKlass:
5552   case Bottom:                  // Ye Olde Default
5553     return Type::BOTTOM;
5554   case Top:
5555     return this;
5556 
5557   default:                      // All else is a mistake
5558     typerr(t);
5559 
5560   case AnyPtr: {                // Meeting to AnyPtrs
5561     // Found an AnyPtr type vs self-KlassPtr type
5562     const TypePtr *tp = t->is_ptr();
5563     Offset offset = meet_offset(tp->offset());
5564     PTR ptr = meet_ptr(tp->ptr());
5565     switch (tp->ptr()) {
5566     case TopPTR:
5567       return this;
5568     case Null:
5569       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5570     case AnyNull:
5571       return make(ptr, klass(), offset, flat_array());
5572     case BotPTR:
5573     case NotNull:
5574       return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5575     default: typerr(t);
5576     }
5577   }
5578 
5579   case RawPtr:
5580   case MetadataPtr:
5581   case OopPtr:
5582   case AryPtr:                  // Meet with AryPtr
5583   case InstPtr:                 // Meet with InstPtr
5584     return TypePtr::BOTTOM;
5585 
5586   //
5587   //             A-top         }
5588   //           /   |   \       }  Tops
5589   //       B-top A-any C-top   }
5590   //          | /  |  \ |      }  Any-nulls
5591   //       B-any   |   C-any   }
5592   //          |    |    |
5593   //       B-con A-con C-con   } constants; not comparable across classes
5594   //          |    |    |
5595   //       B-not   |   C-not   }
5596   //          | \  |  / |      }  not-nulls
5597   //       B-bot A-not C-bot   }
5598   //           \   |   /       }  Bottoms
5599   //             A-bot         }
5600   //
5601 
5602   case KlassPtr: {  // Meet two KlassPtr types
5603     const TypeKlassPtr *tkls = t->is_klassptr();
5604     Offset  off  = meet_offset(tkls->offset());
5605     PTR  ptr     = meet_ptr(tkls->ptr());
5606 
5607     if (klass() == NULL || tkls->klass() == NULL) {
5608       ciKlass* k = NULL;
5609       if (ptr == Constant) {
5610         k = (klass() == NULL) ? tkls->klass() : klass();
5611       }
5612       return make(ptr, k, off, false);
5613     }
5614 
5615     // Check for easy case; klasses are equal (and perhaps not loaded!)
5616     // If we have constants, then we created oops so classes are loaded
5617     // and we can handle the constants further down.  This case handles
5618     // not-loaded classes
5619     if (ptr != Constant && tkls->klass()->equals(klass()) && flat_array() == tkls->flat_array()) {
5620       return make(ptr, klass(), off, flat_array());
5621     }
5622 
5623     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
5624     ciKlass* tkls_klass = tkls->klass();
5625     ciKlass* this_klass = this->klass();
5626     assert( tkls_klass->is_loaded(), "This class should have been loaded.");
5627     assert( this_klass->is_loaded(), "This class should have been loaded.");
5628     bool tkls_flat_array = tkls->flat_array();
5629     bool this_flat_array  = this->flat_array();
5630     bool flat_array = below_centerline(ptr) ? (this_flat_array && tkls_flat_array) : (this_flat_array || tkls_flat_array);
5631 
5632     // If 'this' type is above the centerline and is a superclass of the
5633     // other, we can treat 'this' as having the same type as the other.
5634     if ((above_centerline(this->ptr())) &&
5635         tkls_klass->is_subtype_of(this_klass)) {
5636       this_klass = tkls_klass;
5637     }
5638     // If 'tinst' type is above the centerline and is a superclass of the
5639     // other, we can treat 'tinst' as having the same type as the other.
5640     if ((above_centerline(tkls->ptr())) &&
5641         this_klass->is_subtype_of(tkls_klass)) {
5642       tkls_klass = this_klass;
5643     }
5644 
5645     // Check for classes now being equal
5646     if (tkls_klass->equals(this_klass)) {
5647       // If the klasses are equal, the constants may still differ.  Fall to
5648       // NotNull if they do (neither constant is NULL; that is a special case
5649       // handled elsewhere).
5650       if( ptr == Constant ) {
5651         if (this->_ptr == Constant && tkls->_ptr == Constant &&
5652             this->klass()->equals(tkls->klass()));
5653         else if (above_centerline(this->ptr()));
5654         else if (above_centerline(tkls->ptr()));
5655         else
5656           ptr = NotNull;
5657       }
5658       return make(ptr, this_klass, off, flat_array);
5659     } // Else classes are not equal
5660 
5661     // Since klasses are different, we require the LCA in the Java
5662     // class hierarchy - which means we have to fall to at least NotNull.
5663     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
5664       ptr = NotNull;
5665     // Now we find the LCA of Java classes
5666     ciKlass* k = this_klass->least_common_ancestor(tkls_klass);
5667     return   make(ptr, k, off, k->is_inlinetype() && k->flatten_array());
5668   } // End of case KlassPtr
5669 
5670   } // End of switch
5671   return this;                  // Return the double constant
5672 }

5682   assert( _ptr == Null || _ptr == Constant, "" );
5683   assert(offset() >= 0, "");
5684 
5685   if (offset() != 0) {
5686     // After being ported to the compiler interface, the compiler no longer
5687     // directly manipulates the addresses of oops.  Rather, it only has a pointer
5688     // to a handle at compile time.  This handle is embedded in the generated
5689     // code and dereferenced at the time the nmethod is made.  Until that time,

5754 //------------------------------make-------------------------------------------
5755 const TypeFunc* TypeFunc::make(ciMethod* method, bool is_osr_compilation) {
5756   Compile* C = Compile::current();
5757   const TypeFunc* tf = NULL;
5758   if (!is_osr_compilation) {
5759     tf = C->last_tf(method); // check cache
5760     if (tf != NULL)  return tf;  // The hit rate here is almost 50%.
5761   }
5762   // Inline types are not passed/returned by reference, instead each field of
5763   // the inline type is passed/returned as an argument. We maintain two views of
5764   // the argument/return list here: one based on the signature (with an inline
5765   // type argument/return as a single slot), one based on the actual calling
5766   // convention (with an inline type argument/return as a list of its fields).
5767   bool has_scalar_args = method->has_scalarized_args() && !is_osr_compilation;
5768   const TypeTuple* domain_sig = is_osr_compilation ? osr_domain() : TypeTuple::make_domain(method, false);
5769   const TypeTuple* domain_cc = has_scalar_args ? TypeTuple::make_domain(method, true) : domain_sig;
5770   ciSignature* sig = method->signature();
5771   bool has_scalar_ret = sig->return_type()->is_inlinetype() && sig->return_type()->as_inline_klass()->can_be_returned_as_fields();
5772   const TypeTuple* range_sig = TypeTuple::make_range(sig, false);
5773   const TypeTuple* range_cc = has_scalar_ret ? TypeTuple::make_range(sig, true) : range_sig;
5774   tf = TypeFunc::make(domain_sig, domain_cc, range_sig, range_cc);
5775   if (!is_osr_compilation) {
5776     C->set_last_tf(method, tf);  // fill cache
5777   }
5778   return tf;
5779 }
