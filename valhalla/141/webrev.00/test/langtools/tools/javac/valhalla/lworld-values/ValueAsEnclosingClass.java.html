31 public inline class ValueAsEnclosingClass {
32 
33     static inline class V {
34         int y = 52;
35 
36         class Bar { }
37         static class Baz { }
38     }
39 
40     class Inner { }
41 
42     static ValueAsEnclosingClass.Inner xi = new ValueAsEnclosingClass().new Inner();
43     ValueAsEnclosingClass.ref.Inner xri = xi;
44 
45     public static void main(String[] args) {
46         new V().new Bar();
47         V.Baz baz1 = new V.Baz();
48         V.ref.Baz baz2 = baz1;
49         new ValueAsEnclosingClass();
50     }
51 }
