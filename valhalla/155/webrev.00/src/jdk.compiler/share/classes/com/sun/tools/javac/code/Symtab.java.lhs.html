 75 /** A class that defines all predefined constants and operators
 76  *  as well as special classes such as java.lang.Object, which need
 77  *  to be known to the compiler. All symbols are held in instance
 78  *  fields. This makes it possible to work in multiple concurrent
 79  *  projects, which might use different class files for library classes.
 80  *
 81  *  <p><b>This is NOT part of any supported API.
 82  *  If you write code that depends on this, you do so at your own risk.
 83  *  This code and its internal interfaces are subject to change or
 84  *  deletion without notice.</b>
 85  */

 90     /** Get the symbol table instance. */
 91     public static Symtab instance(Context context) {
 92         Symtab instance = context.get(symtabKey);
 93         if (instance == null)
 94             instance = new Symtab(context);
 95         return instance;
 96     }

266     public VarSymbol getClassField(Type type, Types types) {
267         return classFields.computeIfAbsent(
268             new UniqueType(type, types), k -> {
269                 Type arg = null;
270                 if (type.getTag() == ARRAY || type.getTag() == CLASS) {
271                     /* Temporary treatment for inline class: Given an inline class V that implements
272                        I1, I2, ... In, V.class is typed to be Class<? extends Object & I1 & I2 .. & In>
273                     */
274                     if (type.isValue()) {
275                         List<Type> bounds = List.of(objectType).appendList(((ClassSymbol) type.tsym).getInterfaces());
276                         arg = new WildcardType(bounds.size() > 1 ? types.makeIntersectionType(bounds) : objectType, BoundKind.EXTENDS, boundClass);
277                     } else {
278                         arg = types.erasure(type);
279                     }
280                 }
281                 else if (type.isPrimitiveOrVoid())
282                     arg = types.boxedClass(type).type;
283                 else
284                     throw new AssertionError(type);
285 
286                 Type t = new ClassType(
287                     classType.getEnclosingType(), List.of(arg), classType.tsym);
288                 return new VarSymbol(
289                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
290             });
291     }

321     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
322         final Completer completer = type.tsym.completer;
323         type.tsym.completer = new Completer() {
324             @Override
325             public void complete(Symbol sym) throws CompletionFailure {
326                 try {
327                     completer.complete(sym);
328                 } catch (CompletionFailure e) {
329                     sym.flags_field |= (PUBLIC | INTERFACE);
330                     ((ClassType) sym.type).supertype_field = objectType;
331                 }
332             }
333 
334             @Override
335             public boolean isTerminal() {
336                 return completer.isTerminal();
337             }
338         };
339     }

341     public void synthesizeBoxTypeIfMissing(final Type type) {
342         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
343         final Completer completer = sym.completer;
344         sym.completer = new Completer() {
345             @Override
346             public void complete(Symbol sym) throws CompletionFailure {
347                 try {
348                     completer.complete(sym);
349                 } catch (CompletionFailure e) {
350                     sym.flags_field |= PUBLIC;
351                     ((ClassType) sym.type).supertype_field = objectType;
352                     MethodSymbol boxMethod =
353                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
354                                          new MethodType(List.of(type), sym.type,
355                                 List.nil(), methodClass),
356                             sym);
357                     sym.members().enter(boxMethod);
358                     MethodSymbol unboxMethod =
359                         new MethodSymbol(PUBLIC,
360                             type.tsym.name.append(names.Value), // x.intValue()
361                             new MethodType(List.nil(), type,
362                                 List.nil(), methodClass),
363                             sym);
364                     sym.members().enter(unboxMethod);
365                 }
366             }
367 
368             @Override
369             public boolean isTerminal() {
370                 return completer.isTerminal();
371             }
372         };
373     }

375     // Enter a synthetic class that is used to mark classes in ct.sym.
376     // This class does not have a class file.
377     private Type enterSyntheticAnnotation(String name) {
378         // for now, leave the module null, to prevent problems from synthesizing the
379         // existence of a class in any specific module, including noModule
380         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
381         ClassSymbol sym = (ClassSymbol)type.tsym;
382         sym.completer = Completer.NULL_COMPLETER;
383         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
384         sym.erasure_field = type;
385         sym.members_field = WriteableScope.create(sym);
386         type.typarams_field = List.nil();
387         type.allparams_field = List.nil();
388         type.supertype_field = annotationType;
389         type.interfaces_field = List.nil();
390         return type;
391     }

397         context.put(symtabKey, this);
398 
399         names = Names.instance(context);
400 
401         // Create the unknown type
402         unknownType = new UnknownType();
403 
404         messages = JavacMessages.instance(context);
405 
406         MissingInfoHandler missingInfoHandler = MissingInfoHandler.instance(context);
407 
408         Target target = Target.instance(context);
409         rootPackage = new RootPackageSymbol(names.empty, null,
410                                             missingInfoHandler,
411                                             target.runtimeUseNestAccess());
412 
413         // create the basic builtin symbols
414         unnamedModule = new ModuleSymbol(names.empty, null) {
415                 {
416                     directives = List.nil();
417                     exports = List.nil();
418                     provides = List.nil();
419                     uses = List.nil();
420                     ModuleSymbol java_base = enterModule(names.java_base);
421                     com.sun.tools.javac.code.Directive.RequiresDirective d =
422                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
423                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
424                     requires = List.of(d);
425                 }
426                 @Override
427                 public String toString() {
428                     return messages.getLocalizedString("compiler.misc.unnamed.module");
429                 }
430             };
431         addRootPackageFor(unnamedModule);
432         unnamedModule.enclosedPackages = unnamedModule.enclosedPackages.prepend(unnamedModule.unnamedPackage);
433 
434         errModule = new ModuleSymbol(names.empty, null) {
435                 {
436                     directives = List.nil();
437                     exports = List.nil();
438                     provides = List.nil();
439                     uses = List.nil();
440                     ModuleSymbol java_base = enterModule(names.java_base);
441                     com.sun.tools.javac.code.Directive.RequiresDirective d =
442                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
443                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
444                     requires = List.of(d);
445                 }
446             };
447         addRootPackageFor(errModule);
448 
449         noModule = new ModuleSymbol(names.empty, null) {
450             @Override public boolean isNoModule() {

452             }
453         };
454         addRootPackageFor(noModule);
455 
456         noSymbol = new TypeSymbol(NIL, 0, names.empty, Type.noType, rootPackage) {
457             @Override @DefinedBy(Api.LANGUAGE_MODEL)
458             public <R, P> R accept(ElementVisitor<R, P> v, P p) {
459                 return v.visitUnknown(this, p);
460             }
461         };
462 
463         // create the error symbols
464         errSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.any, null, rootPackage);
465         errType = new ErrorType(errSymbol, Type.noType);
466 
467         unknownSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.fromString("<any?>"), null, rootPackage);
468         unknownSymbol.members_field = new Scope.ErrorScope(unknownSymbol);
469         unknownSymbol.type = unknownType;
470 
471         // initialize builtin types
472         initType(byteType, "byte", "Byte");
473         initType(shortType, "short", "Short");
474         initType(charType, "char", "Character");
475         initType(intType, "int", "Integer");
476         initType(longType, "long", "Long");
477         initType(floatType, "float", "Float");
478         initType(doubleType, "double", "Double");
479         initType(booleanType, "boolean", "Boolean");
480         initType(voidType, "void", "Void");
481         initType(botType, "<nulltype>");
482         initType(errType, errSymbol);
483         initType(unknownType, unknownSymbol);
484 
485         // the builtin class of all arrays
486         arrayClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Array, noSymbol);
487 
488         // VGJ
489         boundClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Bound, noSymbol);
490         boundClass.members_field = new Scope.ErrorScope(boundClass);
491 
492         // the builtin class of all methods
493         methodClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Method, noSymbol);
494         methodClass.members_field = new Scope.ErrorScope(boundClass);
495 
496         // Create class to hold all predefined constants and operations.
497         predefClass = new ClassSymbol(PUBLIC|ACYCLIC, names.empty, rootPackage);
498         WriteableScope scope = WriteableScope.create(predefClass);
499         predefClass.members_field = scope;

502         initialCompleter = ClassFinder.instance(context).getCompleter();

504 
505         // Enter symbols for basic types.
506         scope.enter(byteType.tsym);
507         scope.enter(shortType.tsym);
508         scope.enter(charType.tsym);
509         scope.enter(intType.tsym);
510         scope.enter(longType.tsym);
511         scope.enter(floatType.tsym);
512         scope.enter(doubleType.tsym);
513         scope.enter(booleanType.tsym);
514         scope.enter(errType.tsym);

559         annotationType = enterClass("java.lang.annotation.Annotation");
560         classLoaderType = enterClass("java.lang.ClassLoader");
561         enumSym = enterClass(java_base, names.java_lang_Enum);

563             new MethodSymbol(PROTECTED|FINAL|HYPOTHETICAL,
564                              names.finalize,
565                              new MethodType(List.nil(), voidType,
566                                             List.nil(), methodClass),

622         // classes in ct.sym.  This class does not have a class file.
623         profileType = enterSyntheticAnnotation("jdk.Profile+Annotation");
624         MethodSymbol m = new MethodSymbol(PUBLIC | ABSTRACT, names.value, intType, profileType.tsym);
625         profileType.tsym.members().enter(m);

627         // Enter a class for arrays.
628         // The class implements java.lang.Cloneable and java.io.Serializable.
629         // It has a final length field and a clone method.
630         ClassType arrayClassType = (ClassType)arrayClass.type;

664      *  and owner and enter in `classes' unless already there.
665      */
666     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
667         Assert.checkNonNull(msym);
668         Name flatname = TypeSymbol.formFlatName(name, owner);
669         ClassSymbol c = getClass(msym, flatname);
670         if (c == null) {
671             c = defineClass(name, owner);
672             doEnterClass(msym, c);
673         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
674             // reassign fields of classes that might have been loaded with
675             // their flat names.
676             c.owner.members().remove(c);
677             c.name = name;
678             c.owner = owner;
679             c.fullname = ClassSymbol.formFullName(name, owner);
680         }
681         return c;
682     }

689     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
690         Assert.checkNonNull(msym);
691 
692         if (flatName.isEmpty()) {
693             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
694             return msym.unnamedPackage;
695         }
696 
697         if (msym == noModule) {
698             return enterPackage(msym, flatName);
699         }
700 
701         msym.complete();
702 
703         PackageSymbol pack;
704 
705         pack = msym.visiblePackages.get(flatName);
706 
707         if (pack != null)
708             return pack;
709 
710         pack = getPackage(msym, flatName);
711 
712         if (pack != null && pack.exists())
713             return pack;
714 
715         boolean dependsOnUnnamed = msym.requires != null &&
716                                    msym.requires.stream()
717                                                 .map(rd -> rd.module)
718                                                 .anyMatch(mod -> mod == unnamedModule);
719 
720         if (dependsOnUnnamed) {
721             //msyms depends on the unnamed module, for which we generally don't know
722             //the list of packages it "exports" ahead of time. So try to lookup the package in the
723             //current module, and in the unnamed module and see if it exists in one of them
724             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
725 
726             if (unnamedPack != null && unnamedPack.exists()) {
727                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
728                 return unnamedPack;
729             }
730 
731             pack = enterPackage(msym, flatName);
732             pack.complete();
733             if (pack.exists())
734                 return pack;
735 
736             unnamedPack = enterPackage(unnamedModule, flatName);
737             unnamedPack.complete();
738             if (unnamedPack.exists()) {
739                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
740                 return unnamedPack;
741             }
742 
743             return pack;
744         }
745 
746         return enterPackage(msym, flatName);
747     }

788      */
789     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
790         Assert.checkNonNull(currModule);
791         PackageSymbol p = getPackage(currModule, fullname);
792         if (p == null) {
793             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
794             p = new PackageSymbol(
795                     Convert.shortName(fullname),
796                     enterPackage(currModule, Convert.packagePart(fullname)));
797             p.completer = initialCompleter;
798             p.modle = currModule;
799             doEnterPackage(currModule, p);
800         }
801         return p;
802     }
