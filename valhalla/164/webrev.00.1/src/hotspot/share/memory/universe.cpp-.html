 279 void initialize_basic_type_klass(Klass* k, TRAPS) {
 280   Klass* ok = SystemDictionary::Object_klass();
 281 #if INCLUDE_CDS
 282   if (UseSharedSpaces) {
 283     ClassLoaderData* loader_data = ClassLoaderData::the_null_class_loader_data();
 284     assert(k->super() == ok, "u3");
 285     if (k->is_instance_klass()) {
 286       InstanceKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), NULL, CHECK);
 287     } else {
 288       ArrayKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), CHECK);
 289     }
 290   } else
 291 #endif
 292   {
 293     k->initialize_supers(ok, NULL, CHECK);
 294   }
 295   k->append_to_sibling_list();
 296 }

 298 void Universe::genesis(TRAPS) {
 299   ResourceMark rm(THREAD);
 300 
 301   { AutoModifyRestore<bool> temporarily(_bootstrapping, true);
 302 
 303     { MutexLocker mc(THREAD, Compile_lock);
 304 
 305       java_lang_Class::allocate_fixup_lists();
 306 
 307       // determine base vtable size; without that we cannot create the array klasses
 308       compute_base_vtable_size();
 309 
 310       if (!UseSharedSpaces) {
 311         for (int i = T_BOOLEAN; i < T_LONG+1; i++) {
 312           _typeArrayKlassObjs[i] = TypeArrayKlass::create_klass((BasicType)i, CHECK);
 313         }
 314 
 315         ClassLoaderData* null_cld = ClassLoaderData::the_null_class_loader_data();
 316 
 317         _the_array_interfaces_array     = MetadataFactory::new_array<Klass*>(null_cld, 2, NULL, CHECK);
 318         _the_empty_int_array            = MetadataFactory::new_array<int>(null_cld, 0, CHECK);
 319         _the_empty_short_array          = MetadataFactory::new_array<u2>(null_cld, 0, CHECK);
 320         _the_empty_method_array         = MetadataFactory::new_array<Method*>(null_cld, 0, CHECK);
 321         _the_empty_klass_array          = MetadataFactory::new_array<Klass*>(null_cld, 0, CHECK);
 322         _the_empty_instance_klass_array = MetadataFactory::new_array<InstanceKlass*>(null_cld, 0, CHECK);
 323       }
 324     }
 325 
 326     vmSymbols::initialize(CHECK);
 327 
 328     SystemDictionary::initialize(CHECK);
 329 
 330     Klass* ok = SystemDictionary::Object_klass();
 331 
 332     _the_null_string            = StringTable::intern("null", CHECK);
 333     _the_min_jint_string       = StringTable::intern("-2147483648", CHECK);
 334 
 335 #if INCLUDE_CDS
 336     if (UseSharedSpaces) {
 337       // Verify shared interfaces array.
 338       assert(_the_array_interfaces_array->at(0) ==
 339              SystemDictionary::Cloneable_klass(), "u3");
 340       assert(_the_array_interfaces_array->at(1) ==
 341              SystemDictionary::Serializable_klass(), "u3");
 342     } else
 343 #endif
 344     {
 345       // Set up shared interfaces array.  (Do this before supers are set up.)
 346       _the_array_interfaces_array->at_put(0, SystemDictionary::Cloneable_klass());
 347       _the_array_interfaces_array->at_put(1, SystemDictionary::Serializable_klass());
 348     }
 349 
 350     initialize_basic_type_klass(boolArrayKlassObj(), CHECK);
 351     initialize_basic_type_klass(charArrayKlassObj(), CHECK);
 352     initialize_basic_type_klass(floatArrayKlassObj(), CHECK);
 353     initialize_basic_type_klass(doubleArrayKlassObj(), CHECK);
 354     initialize_basic_type_klass(byteArrayKlassObj(), CHECK);
 355     initialize_basic_type_klass(shortArrayKlassObj(), CHECK);
 356     initialize_basic_type_klass(intArrayKlassObj(), CHECK);
 357     initialize_basic_type_klass(longArrayKlassObj(), CHECK);
 358   } // end of core bootstrapping
 359 
 360   {
 361     Handle tns = java_lang_String::create_from_str("<null_sentinel>", CHECK);
 362     _the_null_sentinel = tns();
 363   }
 364 
 365   // Maybe this could be lifted up now that object array can be initialized
 366   // during the bootstrapping.
 367 
 368   // OLD
 369   // Initialize _objectArrayKlass after core bootstraping to make
 370   // sure the super class is set up properly for _objectArrayKlass.
 371   // ---
 372   // NEW
 373   // Since some of the old system object arrays have been converted to
 374   // ordinary object arrays, _objectArrayKlass will be loaded when
 375   // SystemDictionary::initialize(CHECK); is run. See the extra check
 376   // for Object_klass_loaded in objArrayKlassKlass::allocate_objArray_klass_impl.
 377   _objectArrayKlassObj = InstanceKlass::
 378     cast(SystemDictionary::Object_klass())->array_klass(1, CHECK);
 379   // OLD
 380   // Add the class to the class hierarchy manually to make sure that
 381   // its vtable is initialized after core bootstrapping is completed.
 382   // ---
 383   // New
 384   // Have already been initialized.
 385   _objectArrayKlassObj->append_to_sibling_list();
 386 
 387   #ifdef ASSERT
 388   if (FullGCALot) {
 389     // Allocate an array of dummy objects.
 390     // We'd like these to be at the bottom of the old generation,
 391     // so that when we free one and then collect,
 392     // (almost) the whole heap moves
 393     // and we find out if we actually update all the oops correctly.
 394     // But we can't allocate directly in the old generation,
 395     // so we allocate wherever, and hope that the first collection
 396     // moves these objects to the bottom of the old generation.
 397     // We can allocate directly in the permanent generation, so we do.
 398     int size = FullGCALotDummies * 2;
 399 
 400     objArrayOop    naked_array = oopFactory::new_objArray(SystemDictionary::Object_klass(), size, CHECK);
 401     objArrayHandle dummy_array(THREAD, naked_array);
 402     int i = 0;
 403     while (i < size) {
 404         // Allocate dummy in old generation
 405       oop dummy = SystemDictionary::Object_klass()->allocate_instance(CHECK);
 406       dummy_array->obj_at_put(i++, dummy);
 407     }
 408     {
 409       // Only modify the global variable inside the mutex.
 410       // If we had a race to here, the other dummy_array instances
 411       // and their elements just get dropped on the floor, which is fine.
 412       MutexLocker ml(THREAD, FullGCALot_lock);
 413       if (_fullgc_alot_dummy_array == NULL) {
 414         _fullgc_alot_dummy_array = dummy_array();
 415       }
 416     }
 417     assert(i == _fullgc_alot_dummy_array->length(), "just checking");
 418   }
 419   #endif
 420 }

 469 void Universe::fixup_mirrors(TRAPS) {
 470   // Bootstrap problem: all classes gets a mirror (java.lang.Class instance) assigned eagerly,
 471   // but we cannot do that for classes created before java.lang.Class is loaded. Here we simply
 472   // walk over permanent objects created so far (mostly classes) and fixup their mirrors. Note
 473   // that the number of objects allocated at this point is very small.
 474   assert(SystemDictionary::Class_klass_loaded(), "java.lang.Class should be loaded");
 475   HandleMark hm(THREAD);
 476 
 477   if (!UseSharedSpaces) {
 478     // Cache the start of the static fields
 479     InstanceMirrorKlass::init_offset_of_static_fields();
 480   }
 481 
 482   GrowableArray <Klass*>* list = java_lang_Class::fixup_mirror_list();
 483   int list_length = list->length();
 484   for (int i = 0; i < list_length; i++) {
 485     Klass* k = list->at(i);
 486     assert(k->is_klass(), "List should only hold classes");
 487     EXCEPTION_MARK;
 488     java_lang_Class::fixup_mirror(k, CATCH);
 489   }
 490   delete java_lang_Class::fixup_mirror_list();
 491   java_lang_Class::set_fixup_mirror_list(NULL);
 492 }

 526 void Universe::reinitialize_vtable_of(Klass* ko, TRAPS) {

 562 
 563 bool Universe::should_fill_in_stack_trace(Handle throwable) {
 564   // never attempt to fill in the stack trace of preallocated errors that do not have
 565   // backtrace. These errors are kept alive forever and may be "re-used" when all
 566   // preallocated errors with backtrace have been consumed. Also need to avoid
 567   // a potential loop which could happen if an out of memory occurs when attempting
 568   // to allocate the backtrace.
 569   return ((throwable() != Universe::_out_of_memory_error_java_heap) &&
 570           (throwable() != Universe::_out_of_memory_error_metaspace)  &&
 571           (throwable() != Universe::_out_of_memory_error_class_metaspace)  &&
 572           (throwable() != Universe::_out_of_memory_error_array_size) &&
 573           (throwable() != Universe::_out_of_memory_error_gc_overhead_limit) &&
 574           (throwable() != Universe::_out_of_memory_error_realloc_objects) &&
 575           (throwable() != Universe::_out_of_memory_error_retry));
 576 }

 578 
 579 oop Universe::gen_out_of_memory_error(oop default_err) {
 580   // generate an out of memory error:
 581   // - if there is a preallocated error and stack traces are available
 582   //   (j.l.Throwable is initialized), then return the preallocated
 583   //   error with a filled in stack trace, and with the message
 584   //   provided by the default error.
 585   // - otherwise, return the default error, without a stack trace.
 586   int next;
 587   if ((_preallocated_out_of_memory_error_avail_count > 0) &&
 588       SystemDictionary::Throwable_klass()->is_initialized()) {
 589     next = (int)Atomic::add(&_preallocated_out_of_memory_error_avail_count, -1);
 590     assert(next < (int)PreallocatedOutOfMemoryErrorCount, "avail count is corrupt");
 591   } else {
 592     next = -1;
 593   }
 594   if (next < 0) {
 595     // all preallocated errors have been used.
 596     // return default
 597     return default_err;
 598   } else {
 599     Thread* THREAD = Thread::current();
 600     Handle default_err_h(THREAD, default_err);
 601     // get the error object at the slot and set set it to NULL so that the
 602     // array isn't keeping it alive anymore.
 603     Handle exc(THREAD, preallocated_out_of_memory_errors()->obj_at(next));
 604     assert(exc() != NULL, "slot has been used already");
 605     preallocated_out_of_memory_errors()->obj_at_put(next, NULL);
 606 
 607     // use the message from the default error
 608     oop msg = java_lang_Throwable::message(default_err_h());
 609     assert(msg != NULL, "no message");
 610     java_lang_Throwable::set_message(exc(), msg);
 611 
 612     // populate the stack trace and return it.
 613     java_lang_Throwable::fill_in_stack_trace_of_preallocated_backtrace(exc);
 614     return exc();
 615   }
 616 }

 618 intptr_t Universe::_non_oop_bits = 0;
 619 
 620 void* Universe::non_oop_word() {
 621   // Neither the high bits nor the low bits of this value is allowed
 622   // to look like (respectively) the high or low bits of a real oop.
 623   //
 624   // High and low are CPU-specific notions, but low always includes
 625   // the low-order bit.  Since oops are always aligned at least mod 4,
 626   // setting the low-order bit will ensure that the low half of the
 627   // word will never look like that of a real oop.
 628   //
 629   // Using the OS-supplied non-memory-address word (usually 0 or -1)
 630   // will take care of the high bits, however many there are.
 631 
 632   if (_non_oop_bits == 0) {
 633     _non_oop_bits = (intptr_t)os::non_memory_address_word() | 1;
 634   }
 635 
 636   return (void*)_non_oop_bits;
 637 }

 739 ReservedHeapSpace Universe::reserve_heap(size_t heap_size, size_t alignment) {
 740 
 741   assert(alignment <= Arguments::conservative_max_heap_alignment(),
 742          "actual alignment " SIZE_FORMAT " must be within maximum heap alignment " SIZE_FORMAT,
 743          alignment, Arguments::conservative_max_heap_alignment());
 744 
 745   size_t total_reserved = align_up(heap_size, alignment);
 746   assert(!UseCompressedOops || (total_reserved <= (OopEncodingHeapMax - os::vm_page_size())),
 747       "heap size is too big for compressed oops");
 748 
 749   bool use_large_pages = UseLargePages && is_aligned(alignment, os::large_page_size());
 750   assert(!UseLargePages
 751       || UseParallelGC
 752       || use_large_pages, "Wrong alignment to use large pages");
 753 
 754   // Now create the space.
 755   ReservedHeapSpace total_rs(total_reserved, alignment, use_large_pages, AllocateHeapAt);
 756 
 757   if (total_rs.is_reserved()) {
 758     assert((total_reserved == total_rs.size()) && ((uintptr_t)total_rs.base() % alignment == 0),
 759            "must be exactly of required size and alignment");
 760     // We are good.
 761 
 762     if (AllocateHeapAt != NULL) {
 763       log_info(gc,heap)("Successfully allocated Java heap at location %s", AllocateHeapAt);
 764     }
 765 
 766     if (UseCompressedOops) {
 767       CompressedOops::initialize(total_rs);
 768     }
 769 
 770     Universe::calculate_verify_data((HeapWord*)total_rs.base(), (HeapWord*)total_rs.end());
 771 
 772     return total_rs;
 773   }
 774 
 775   vm_exit_during_initialization(
 776     err_msg("Could not reserve enough space for " SIZE_FORMAT "KB object heap",
 777             total_reserved/K));
 778 
 779   // satisfy compiler
 780   ShouldNotReachHere();
 781   return ReservedHeapSpace(0, 0, false);
 782 }

 853 bool universe_post_init() {
 854   assert(!is_init_completed(), "Error: initialization not yet completed!");
 855   Universe::_fully_initialized = true;
 856   EXCEPTION_MARK;
 857   if (!UseSharedSpaces) {
 858     ResourceMark rm;
 859     Universe::reinitialize_vtables(CHECK_false);
 860     Universe::reinitialize_itables(CHECK_false);
 861   }
 862 
 863   HandleMark hm(THREAD);
 864   // Setup preallocated empty java.lang.Class array
 865   Universe::_the_empty_class_klass_array = oopFactory::new_objArray(SystemDictionary::Class_klass(), 0, CHECK_false);
 866 
 867   // Setup preallocated OutOfMemoryError errors
 868   Klass* k = SystemDictionary::OutOfMemoryError_klass();
 869   InstanceKlass* ik = InstanceKlass::cast(k);
 870   Universe::_out_of_memory_error_java_heap = ik->allocate_instance(CHECK_false);
 871   Universe::_out_of_memory_error_metaspace = ik->allocate_instance(CHECK_false);
 872   Universe::_out_of_memory_error_class_metaspace = ik->allocate_instance(CHECK_false);
 873   Universe::_out_of_memory_error_array_size = ik->allocate_instance(CHECK_false);
 874   Universe::_out_of_memory_error_gc_overhead_limit =
 875     ik->allocate_instance(CHECK_false);
 876   Universe::_out_of_memory_error_realloc_objects = ik->allocate_instance(CHECK_false);
 877   Universe::_out_of_memory_error_retry = ik->allocate_instance(CHECK_false);
 878 
 879   // Setup preallocated cause message for delayed StackOverflowError
 880   if (StackReservedPages > 0) {
 881     Universe::_delayed_stack_overflow_error_message =
 882       java_lang_String::create_oop_from_str("Delayed StackOverflowError due to ReservedStackAccess annotated method", CHECK_false);
 883   }
 884 
 885   // Setup preallocated NullPointerException
 886   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 887   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_NullPointerException(), true, CHECK_false);
 888   Universe::_null_ptr_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 889   // Setup preallocated ArithmeticException
 890   // (this is currently used for a cheap & dirty solution in compiler exception handling)
 891   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_ArithmeticException(), true, CHECK_false);
 892   Universe::_arithmetic_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 893   // Virtual Machine Error for when we get into a situation we can't resolve
 894   k = SystemDictionary::VirtualMachineError_klass();
 895   bool linked = InstanceKlass::cast(k)->link_class_or_fail(CHECK_false);
 896   if (!linked) {
 897      tty->print_cr("Unable to link/verify VirtualMachineError class");
 898      return false; // initialization failed
 899   }
 900   Universe::_virtual_machine_error_instance =
 901     InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 902 
 903   Universe::_vm_exception = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 904 
 905   Handle msg = java_lang_String::create_from_str("Java heap space", CHECK_false);
 906   java_lang_Throwable::set_message(Universe::_out_of_memory_error_java_heap, msg());
 907 
 908   msg = java_lang_String::create_from_str("Metaspace", CHECK_false);
 909   java_lang_Throwable::set_message(Universe::_out_of_memory_error_metaspace, msg());
 910   msg = java_lang_String::create_from_str("Compressed class space", CHECK_false);
 911   java_lang_Throwable::set_message(Universe::_out_of_memory_error_class_metaspace, msg());
 912 
 913   msg = java_lang_String::create_from_str("Requested array size exceeds VM limit", CHECK_false);
 914   java_lang_Throwable::set_message(Universe::_out_of_memory_error_array_size, msg());
 915 
 916   msg = java_lang_String::create_from_str("GC overhead limit exceeded", CHECK_false);
 917   java_lang_Throwable::set_message(Universe::_out_of_memory_error_gc_overhead_limit, msg());
 918 
 919   msg = java_lang_String::create_from_str("Java heap space: failed reallocation of scalar replaced objects", CHECK_false);
 920   java_lang_Throwable::set_message(Universe::_out_of_memory_error_realloc_objects, msg());
 921 
 922   msg = java_lang_String::create_from_str("Java heap space: failed retryable allocation", CHECK_false);
 923   java_lang_Throwable::set_message(Universe::_out_of_memory_error_retry, msg());
 924 
 925   msg = java_lang_String::create_from_str("/ by zero", CHECK_false);
 926   java_lang_Throwable::set_message(Universe::_arithmetic_exception_instance, msg());
 927 
 928   // Setup the array of errors that have preallocated backtrace
 929   k = Universe::_out_of_memory_error_java_heap->klass();
 930   assert(k->name() == vmSymbols::java_lang_OutOfMemoryError(), "should be out of memory error");
 931   ik = InstanceKlass::cast(k);
 932 
 933   int len = (StackTraceInThrowable) ? (int)PreallocatedOutOfMemoryErrorCount : 0;
 934   Universe::_preallocated_out_of_memory_error_array = oopFactory::new_objArray(ik, len, CHECK_false);
 935   for (int i=0; i<len; i++) {
 936     oop err = ik->allocate_instance(CHECK_false);
 937     Handle err_h = Handle(THREAD, err);
 938     java_lang_Throwable::allocate_backtrace(err_h, CHECK_false);
 939     Universe::preallocated_out_of_memory_errors()->obj_at_put(i, err_h());
 940   }
 941   Universe::_preallocated_out_of_memory_error_avail_count = (jint)len;
 942 
 943   Universe::initialize_known_methods(CHECK_false);
 944 
 945   // This needs to be done before the first scavenge/gc, since
 946   // it's an input to soft ref clearing policy.
 947   {
 948     MutexLocker x(THREAD, Heap_lock);
 949     Universe::update_heap_info_at_gc();
 950   }
 951 
 952   // ("weak") refs processing infrastructure initialization
 953   Universe::heap()->post_initialize();
 954 
 955   MemoryService::add_metaspace_memory_pools();
 956 
 957   MemoryService::set_universe_heap(Universe::heap());
 958 #if INCLUDE_CDS
 959   MetaspaceShared::post_initialize(CHECK_false);
 960 #endif
 961   return true;
 962 }

1003 void Universe::initialize_verify_flags() {
1004   verify_flags = 0;
1005   const char delimiter[] = " ,";
1006 
1007   size_t length = strlen(VerifySubSet);
1008   char* subset_list = NEW_C_HEAP_ARRAY(char, length + 1, mtInternal);
1009   strncpy(subset_list, VerifySubSet, length + 1);
1010   char* save_ptr;
1011 
1012   char* token = strtok_r(subset_list, delimiter, &save_ptr);
1013   while (token != NULL) {
1014     if (strcmp(token, "threads") == 0) {
1015       verify_flags |= Verify_Threads;
1016     } else if (strcmp(token, "heap") == 0) {
1017       verify_flags |= Verify_Heap;
1018     } else if (strcmp(token, "symbol_table") == 0) {
1019       verify_flags |= Verify_SymbolTable;
1020     } else if (strcmp(token, "string_table") == 0) {
1021       verify_flags |= Verify_StringTable;
1022     } else if (strcmp(token, "codecache") == 0) {
1023       verify_flags |= Verify_CodeCache;
1024     } else if (strcmp(token, "dictionary") == 0) {
1025       verify_flags |= Verify_SystemDictionary;
1026     } else if (strcmp(token, "classloader_data_graph") == 0) {
1027       verify_flags |= Verify_ClassLoaderDataGraph;
1028     } else if (strcmp(token, "metaspace") == 0) {
1029       verify_flags |= Verify_MetaspaceUtils;
1030     } else if (strcmp(token, "jni_handles") == 0) {
1031       verify_flags |= Verify_JNIHandles;
1032     } else if (strcmp(token, "codecache_oops") == 0) {
1033       verify_flags |= Verify_CodeCacheOops;
1034     } else if (strcmp(token, "resolved_method_table") == 0) {
1035       verify_flags |= Verify_ResolvedMethodTable;
1036     } else {
1037       vm_exit_during_initialization(err_msg("VerifySubSet: \'%s\' memory sub-system is unknown, please correct it", token));
1038     }
1039     token = strtok_r(NULL, delimiter, &save_ptr);
1040   }
1041   FREE_C_HEAP_ARRAY(char, subset_list);
1042 }

1051 void Universe::verify(VerifyOption option, const char* prefix) {
1052   // The use of _verify_in_progress is a temporary work around for
1053   // 6320749.  Don't bother with a creating a class to set and clear
1054   // it since it is only used in this method and the control flow is
1055   // straight forward.
1056   _verify_in_progress = true;
1057 
1058   COMPILER2_PRESENT(
1059     assert(!DerivedPointerTable::is_active(),
1060          "DPT should not be active during verification "
1061          "(of thread stacks below)");
1062   )
1063 
1064   ResourceMark rm;
1065   HandleMark hm;  // Handles created during verification can be zapped
1066   _verify_count++;
1067 
1068   FormatBuffer<> title("Verifying %s", prefix);
1069   GCTraceTime(Info, gc, verify) tm(title.buffer());
1070   if (should_verify_subset(Verify_Threads)) {
1071     log_debug(gc, verify)("Threads");
1072     Threads::verify();
1073   }
1074   if (should_verify_subset(Verify_Heap)) {
1075     log_debug(gc, verify)("Heap");
1076     heap()->verify(option);
1077   }
1078   if (should_verify_subset(Verify_SymbolTable)) {
1079     log_debug(gc, verify)("SymbolTable");
1080     SymbolTable::verify();
1081   }
1082   if (should_verify_subset(Verify_StringTable)) {
1083     log_debug(gc, verify)("StringTable");
1084     StringTable::verify();
1085   }

1123 void Universe::calculate_verify_data(HeapWord* low_boundary, HeapWord* high_boundary) {
1124   assert(low_boundary < high_boundary, "bad interval");
1125 
1126   // decide which low-order bits we require to be clear:
1127   size_t alignSize = MinObjAlignmentInBytes;
1128   size_t min_object_size = CollectedHeap::min_fill_size();
1129 
1130   // make an inclusive limit:
1131   uintptr_t max = (uintptr_t)high_boundary - min_object_size*wordSize;
1132   uintptr_t min = (uintptr_t)low_boundary;
1133   assert(min < max, "bad interval");
1134   uintptr_t diff = max ^ min;
1135 
1136   // throw away enough low-order bits to make the diff vanish
1137   uintptr_t mask = (uintptr_t)(-1);
1138   while ((mask & diff) != 0)
1139     mask <<= 1;
1140   uintptr_t bits = (min & mask);
1141   assert(bits == (max & mask), "correct mask");
1142   // check an intermediate value between min and max, just to make sure:
1143   assert(bits == ((min + (max-min)/2) & mask), "correct mask");
1144 
1145   // require address alignment, too:
1146   mask |= (alignSize - 1);
1147 
1148   if (!(_verify_oop_mask == 0 && _verify_oop_bits == (uintptr_t)-1)) {
1149     assert(_verify_oop_mask == mask && _verify_oop_bits == bits, "mask stability");
1150   }
1151   _verify_oop_mask = mask;
1152   _verify_oop_bits = bits;
1153 }
