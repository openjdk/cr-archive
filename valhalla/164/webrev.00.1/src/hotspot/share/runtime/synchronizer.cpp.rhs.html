 178 // =====================> Spin-lock functions
 179 
 180 // ObjectMonitors are not lockable outside of this file. We use spin-locks
 181 // implemented using a bit in the _next_om field instead of the heavier
 182 // weight locking mechanisms for faster list management.
 183 
 184 #define OM_LOCK_BIT 0x1
 185 
 186 // Return true if the ObjectMonitor is locked.
 187 // Otherwise returns false.
 188 static bool is_locked(ObjectMonitor* om) {
 189   return ((intptr_t)om->next_om() & OM_LOCK_BIT) == OM_LOCK_BIT;
 190 }
 191 
 192 // Mark an ObjectMonitor* with OM_LOCK_BIT and return it.
 193 static ObjectMonitor* mark_om_ptr(ObjectMonitor* om) {
 194   return (ObjectMonitor*)((intptr_t)om | OM_LOCK_BIT);
 195 }
 196 
 197 // Return the unmarked next field in an ObjectMonitor. Note: the next
 198 // field may or may not have been marked with OM_LOCK_BIT originally.
 199 static ObjectMonitor* unmarked_next(ObjectMonitor* om) {
 200   return (ObjectMonitor*)((intptr_t)om->next_om() & ~OM_LOCK_BIT);
 201 }
 202 
 203 // Try to lock an ObjectMonitor. Returns true if locking was successful.
 204 // Otherwise returns false.
 205 static bool try_om_lock(ObjectMonitor* om) {
 206   // Get current next field without any OM_LOCK_BIT value.
 207   ObjectMonitor* next = unmarked_next(om);
 208   if (om->try_set_next_om(next, mark_om_ptr(next)) != next) {
 209     return false;  // Cannot lock the ObjectMonitor.
 210   }
 211   return true;
 212 }
 213 
 214 // Lock an ObjectMonitor.
 215 static void om_lock(ObjectMonitor* om) {
 216   while (true) {
 217     if (try_om_lock(om)) {
 218       return;
 219     }
 220   }
 221 }
 222 
 223 // Unlock an ObjectMonitor.
 224 static void om_unlock(ObjectMonitor* om) {
 225   ObjectMonitor* next = om->next_om();
 226   guarantee(((intptr_t)next & OM_LOCK_BIT) == OM_LOCK_BIT, "next=" INTPTR_FORMAT
 227             " must have OM_LOCK_BIT=%x set.", p2i(next), OM_LOCK_BIT);
 228 
 229   next = (ObjectMonitor*)((intptr_t)next & ~OM_LOCK_BIT);  // Clear OM_LOCK_BIT.
 230   om->set_next_om(next);
 231 }
 232 
 233 // Get the list head after locking it. Returns the list head or NULL
 234 // if the list is empty.
 235 static ObjectMonitor* get_list_head_locked(ObjectMonitor** list_p) {
 236   while (true) {
 237     ObjectMonitor* mid = Atomic::load(list_p);
 238     if (mid == NULL) {
 239       return NULL;  // The list is empty.
 240     }
 241     if (try_om_lock(mid)) {
 242       if (Atomic::load(list_p) != mid) {
 243         // The list head changed before we could lock it so we have to retry.
 244         om_unlock(mid);
 245         continue;
 246       }
 247       return mid;
 248     }
 249   }
 250 }

 257 // Prepend a list of ObjectMonitors to the specified *list_p. 'tail' is
 258 // the last ObjectMonitor in the list and there are 'count' on the list.
 259 // Also updates the specified *count_p.
 260 static void prepend_list_to_common(ObjectMonitor* list, ObjectMonitor* tail,
 261                                    int count, ObjectMonitor** list_p,
 262                                    int* count_p) {
 263   while (true) {
 264     ObjectMonitor* cur = Atomic::load(list_p);
 265     // Prepend list to *list_p.
 266     if (!try_om_lock(tail)) {
 267       // Failed to lock tail due to a list walker so try it all again.
 268       continue;
 269     }
 270     tail->set_next_om(cur);  // tail now points to cur (and unlocks tail)
 271     if (cur == NULL) {
 272       // No potential race with takers or other prependers since
 273       // *list_p is empty.
 274       if (Atomic::cmpxchg(list_p, cur, list) == cur) {
 275         // Successfully switched *list_p to the list value.
 276         Atomic::add(count_p, count);
 277         break;
 278       }
 279       // Implied else: try it all again
 280     } else {
 281       if (!try_om_lock(cur)) {
 282         continue;  // failed to lock cur so try it all again
 283       }
 284       // We locked cur so try to switch *list_p to the list value.
 285       if (Atomic::cmpxchg(list_p, cur, list) != cur) {
 286         // The list head has changed so unlock cur and try again:
 287         om_unlock(cur);
 288         continue;
 289       }
 290       Atomic::add(count_p, count);
 291       om_unlock(cur);
 292       break;
 293     }
 294   }
 295 }

 297 // Prepend a newly allocated block of ObjectMonitors to g_block_list and
 298 // om_list_globals._free_list. Also updates om_list_globals._population
 299 // and om_list_globals._free_count.
 300 void ObjectSynchronizer::prepend_block_to_lists(PaddedObjectMonitor* new_blk) {
 301   // First we handle g_block_list:
 302   while (true) {
 303     PaddedObjectMonitor* cur = Atomic::load(&g_block_list);
 304     // Prepend new_blk to g_block_list. The first ObjectMonitor in
 305     // a block is reserved for use as linkage to the next block.
 306     new_blk[0].set_next_om(cur);
 307     if (Atomic::cmpxchg(&g_block_list, cur, new_blk) == cur) {
 308       // Successfully switched g_block_list to the new_blk value.
 309       Atomic::add(&om_list_globals._population, _BLOCKSIZE - 1);
 310       break;
 311     }
 312     // Implied else: try it all again
 313   }
 314 
 315   // Second we handle om_list_globals._free_list:
 316   prepend_list_to_common(new_blk + 1, &new_blk[_BLOCKSIZE - 1], _BLOCKSIZE - 1,
 317                          &om_list_globals._free_list, &om_list_globals._free_count);
 318 }

 347 // Prepend an ObjectMonitor to the specified list. Also updates
 348 // the specified counter.
 349 static void prepend_to_common(ObjectMonitor* m, ObjectMonitor** list_p,
 350                               int* count_p) {
 351   while (true) {
 352     om_lock(m);  // Lock m so we can safely update its next field.
 353     ObjectMonitor* cur = NULL;
 354     // Lock the list head to guard against races with a list walker
 355     // or async deflater thread (which only races in om_in_use_list):
 356     if ((cur = get_list_head_locked(list_p)) != NULL) {
 357       // List head is now locked so we can safely switch it.
 358       m->set_next_om(cur);  // m now points to cur (and unlocks m)
 359       Atomic::store(list_p, m);  // Switch list head to unlocked m.
 360       om_unlock(cur);
 361       break;
 362     }
 363     // The list is empty so try to set the list head.
 364     assert(cur == NULL, "cur must be NULL: cur=" INTPTR_FORMAT, p2i(cur));
 365     m->set_next_om(cur);  // m now points to NULL (and unlocks m)
 366     if (Atomic::cmpxchg(list_p, cur, m) == cur) {
 367       // List head is now unlocked m.
 368       break;
 369     }
 370     // Implied else: try it all again
 371   }
 372   Atomic::inc(count_p);
 373 }

 387 // Take an ObjectMonitor from the start of the specified list. Also
 388 // decrements the specified counter. Returns NULL if none are available.
 389 static ObjectMonitor* take_from_start_of_common(ObjectMonitor** list_p,
 390                                                 int* count_p) {
 391   ObjectMonitor* take = NULL;
 392   // Lock the list head to guard against races with a list walker
 393   // or async deflater thread (which only races in om_list_globals._free_list):
 394   if ((take = get_list_head_locked(list_p)) == NULL) {
 395     return NULL;  // None are available.
 396   }
 397   ObjectMonitor* next = unmarked_next(take);
 398   // Switch locked list head to next (which unlocks the list head, but
 399   // leaves take locked):
 400   Atomic::store(list_p, next);
 401   Atomic::dec(count_p);
 402   // Unlock take, but leave the next value for any lagging list
 403   // walkers. It will get cleaned up when take is prepended to
 404   // the in-use list:
 405   om_unlock(take);
 406   return take;
 407 }

 424 // =====================> Quick functions
 425 
 426 // The quick_* forms are special fast-path variants used to improve
 427 // performance.  In the simplest case, a "quick_*" implementation could
 428 // simply return false, in which case the caller will perform the necessary
 429 // state transitions and call the slow-path form.
 430 // The fast-path is designed to handle frequently arising cases in an efficient
 431 // manner and is just a degenerate "optimistic" variant of the slow-path.
 432 // returns true  -- to indicate the call was satisfied.
 433 // returns false -- to indicate the call needs the services of the slow-path.
 434 // A no-loitering ordinance is in effect for code in the quick_* family
 435 // operators: safepoints or indefinite blocking (blocking that might span a
 436 // safepoint) are forbidden. Generally the thread_state() is _in_Java upon
 437 // entry.
 438 //
 439 // Consider: An interesting optimization is to have the JIT recognize the
 440 // following common idiom:
 441 //   synchronized (someobj) { .... ; notify(); }
 442 // That is, we find a notify() or notifyAll() call that immediately precedes
 443 // the monitorexit operation.  In that case the JIT could fuse the operations
 444 // into a single notifyAndExit() runtime primitive.
 445 
 446 bool ObjectSynchronizer::quick_notify(oopDesc* obj, Thread* self, bool all) {
 447   assert(!SafepointSynchronize::is_at_safepoint(), "invariant");
 448   assert(self->is_Java_thread(), "invariant");
 449   assert(((JavaThread *) self)->thread_state() == _thread_in_Java, "invariant");
 450   NoSafepointVerifier nsv;
 451   if (obj == NULL) return false;  // slow-path for invalid obj
 452   assert(!EnableValhalla || !obj->klass()->is_inline_klass(), "monitor op on inline type");
 453   const markWord mark = obj->mark();
 454 
 455   if (mark.has_locker() && self->is_lock_owned((address)mark.locker())) {
 456     // Degenerate notify
 457     // stack-locked by caller so by definition the implied waitset is empty.
 458     return true;
 459   }
 460 
 461   if (mark.has_monitor()) {
 462     ObjectMonitor* const mon = mark.monitor();
 463     assert(mon->object() == obj, "invariant");
 464     if (mon->owner() != self) return false;  // slow-path for IMS exception
 465 
 466     if (mon->first_waiter() != NULL) {
 467       // We have one or more waiters. Since this is an inflated monitor
 468       // that we own, we can transfer one or more threads from the waitset
 469       // to the entrylist here and now, avoiding the slow-path.
 470       if (all) {
 471         DTRACE_MONITOR_PROBE(notifyAll, mon, obj, self);
 472       } else {
 473         DTRACE_MONITOR_PROBE(notify, mon, obj, self);
 474       }
 475       int free_count = 0;
 476       do {
 477         mon->INotify(self);
 478         ++free_count;
 479       } while (mon->first_waiter() != NULL && all);
 480       OM_PERFDATA_OP(Notifications, inc(free_count));
 481     }
 482     return true;
 483   }
 484 
 485   // biased locking and any other IMS exception states take the slow-path
 486   return false;
 487 }

 490 // The LockNode emitted directly at the synchronization site would have
 491 // been too big if it were to have included support for the cases of inflated
 492 // recursive enter and exit, so they go here instead.
 493 // Note that we can't safely call AsyncPrintJavaStack() from within
 494 // quick_enter() as our thread state remains _in_Java.
 495 
 496 bool ObjectSynchronizer::quick_enter(oop obj, Thread* self,
 497                                      BasicLock * lock) {
 498   assert(!SafepointSynchronize::is_at_safepoint(), "invariant");
 499   assert(self->is_Java_thread(), "invariant");
 500   assert(((JavaThread *) self)->thread_state() == _thread_in_Java, "invariant");
 501   NoSafepointVerifier nsv;
 502   if (obj == NULL) return false;       // Need to throw NPE
 503   assert(!EnableValhalla || !obj->klass()->is_inline_klass(), "monitor op on inline type");
 504   const markWord mark = obj->mark();
 505 
 506   if (mark.has_monitor()) {
 507     ObjectMonitor* const m = mark.monitor();
 508     // An async deflation can race us before we manage to make the
 509     // ObjectMonitor busy by setting the owner below. If we detect
 510     // that race we just bail out to the slow-path here.
 511     if (m->object() == NULL) {
 512       return false;
 513     }
 514     Thread* const owner = (Thread *) m->_owner;
 515 
 516     // Lock contention and Transactional Lock Elision (TLE) diagnostics
 517     // and observability
 518     // Case: light contention possibly amenable to TLE
 519     // Case: TLE inimical operations such as nested/recursive synchronization
 520 
 521     if (owner == self) {
 522       m->_recursions++;
 523       return true;
 524     }
 525 
 526     // This Java Monitor is inflated so obj's header will never be
 527     // displaced to this thread's BasicLock. Make the displaced header
 528     // non-NULL so this BasicLock is not seen as recursive nor as
 529     // being locked. We do this unconditionally so that this thread's
 530     // BasicLock cannot be mis-interpreted by any stack walkers. For
 531     // performance reasons, stack walkers generally first check for
 532     // Biased Locking in the object's header, the second check is for
 533     // stack-locking in the object's header, the third check is for
 534     // recursive stack-locking in the displaced header in the BasicLock,
 535     // and last are the inflated Java Monitor (ObjectMonitor) checks.
 536     lock->set_displaced_header(markWord::unused_mark());
 537 
 538     if (owner == NULL && m->try_set_owner_from(NULL, self) == NULL) {
 539       assert(m->_recursions == 0, "invariant");
 540       return true;
 541     }
 542   }
 543 
 544   // Note that we could inflate in quick_enter.
 545   // This is likely a useful optimization
 546   // Critically, in quick_enter() we must not:
 547   // -- perform bias revocation, or
 548   // -- block indefinitely, or
 549   // -- reach a safepoint
 550 
 551   return false;        // revert to slow-path
 552 }

 554 // -----------------------------------------------------------------------------
 555 // Monitor Enter/Exit
 556 // The interpreter and compiler assembly code tries to lock using the fast path
 557 // of this algorithm. Make sure to update that code if the following function is
 558 // changed. The implementation is extremely sensitive to race condition. Be careful.
 559 
 560 void ObjectSynchronizer::enter(Handle obj, BasicLock* lock, TRAPS) {
 561   CHECK_THROW_NOSYNC_IMSE(obj);
 562   if (UseBiasedLocking) {
 563     if (!SafepointSynchronize::is_at_safepoint()) {
 564       BiasedLocking::revoke(obj, THREAD);
 565     } else {
 566       BiasedLocking::revoke_at_safepoint(obj);
 567     }
 568   }
 569 
 570   markWord mark = obj->mark();
 571   assert(!mark.has_bias_pattern(), "should not see bias pattern here");
 572 
 573   if (mark.is_neutral()) {
 574     // Anticipate successful CAS -- the ST of the displaced mark must
 575     // be visible <= the ST performed by the CAS.
 576     lock->set_displaced_header(mark);
 577     if (mark == obj()->cas_set_mark(markWord::from_pointer(lock), mark)) {
 578       return;
 579     }
 580     // Fall through to inflate() ...
 581   } else if (mark.has_locker() &&
 582              THREAD->is_lock_owned((address)mark.locker())) {
 583     assert(lock != mark.locker(), "must not re-lock the same lock");
 584     assert(lock != (BasicLock*)obj->mark().value(), "don't relock with same BasicLock");
 585     lock->set_displaced_header(markWord::from_pointer(NULL));
 586     return;
 587   }
 588 
 589   // The object header will never be displaced to this lock,
 590   // so it does not matter what the value is, except that it
 591   // must be non-zero to avoid looking like a re-entrant lock,
 592   // and must not look locked either.
 593   lock->set_displaced_header(markWord::unused_mark());
 594   // An async deflation can race after the inflate() call and before
 595   // enter() can make the ObjectMonitor busy. enter() returns false if
 596   // we have lost the race to async deflation and we simply try again.
 597   while (true) {
 598     ObjectMonitor* monitor = inflate(THREAD, obj(), inflate_cause_monitor_enter);
 599     if (monitor->enter(THREAD)) {
 600       return;
 601     }
 602   }
 603 }

 605 void ObjectSynchronizer::exit(oop object, BasicLock* lock, TRAPS) {
 606   markWord mark = object->mark();
 607   if (EnableValhalla && mark.is_always_locked()) {
 608     return;
 609   }
 610   assert(!EnableValhalla || !object->klass()->is_inline_klass(), "monitor op on inline type");
 611   // We cannot check for Biased Locking if we are racing an inflation.
 612   assert(mark == markWord::INFLATING() ||
 613          !mark.has_bias_pattern(), "should not see bias pattern here");
 614 
 615   markWord dhw = lock->displaced_header();
 616   if (dhw.value() == 0) {
 617     // If the displaced header is NULL, then this exit matches up with
 618     // a recursive enter. No real work to do here except for diagnostics.
 619 #ifndef PRODUCT
 620     if (mark != markWord::INFLATING()) {
 621       // Only do diagnostics if we are not racing an inflation. Simply
 622       // exiting a recursive enter of a Java Monitor that is being
 623       // inflated is safe; see the has_monitor() comment below.
 624       assert(!mark.is_neutral(), "invariant");
 625       assert(!mark.has_locker() ||
 626              THREAD->is_lock_owned((address)mark.locker()), "invariant");
 627       if (mark.has_monitor()) {
 628         // The BasicLock's displaced_header is marked as a recursive
 629         // enter and we have an inflated Java Monitor (ObjectMonitor).
 630         // This is a special case where the Java Monitor was inflated
 631         // after this thread entered the stack-lock recursively. When a
 632         // Java Monitor is inflated, we cannot safely walk the Java
 633         // Monitor owner's stack and update the BasicLocks because a
 634         // Java Monitor can be asynchronously inflated by a thread that
 635         // does not own the Java Monitor.
 636         ObjectMonitor* m = mark.monitor();
 637         assert(((oop)(m->object()))->mark() == mark, "invariant");
 638         assert(m->is_entered(THREAD), "invariant");
 639       }
 640     }
 641 #endif
 642     return;
 643   }
 644 
 645   if (mark == markWord::from_pointer(lock)) {
 646     // If the object is stack-locked by the current thread, try to
 647     // swing the displaced header from the BasicLock back to the mark.
 648     assert(dhw.is_neutral(), "invariant");
 649     if (object->cas_set_mark(dhw, mark) == mark) {
 650       return;
 651     }
 652   }
 653 
 654   // We have to take the slow-path of possible inflation and then exit.
 655   // The ObjectMonitor* can't be async deflated until ownership is
 656   // dropped inside exit() and the ObjectMonitor* must be !is_busy().
 657   ObjectMonitor* monitor = inflate(THREAD, object, inflate_cause_vm_internal);
 658   monitor->exit(true, THREAD);
 659 }

 851 // -----------------------------------------------------------------------------
 852 // Hash Code handling
 853 //
 854 // Performance concern:
 855 // OrderAccess::storestore() calls release() which at one time stored 0
 856 // into the global volatile OrderAccess::dummy variable. This store was
 857 // unnecessary for correctness. Many threads storing into a common location
 858 // causes considerable cache migration or "sloshing" on large SMP systems.
 859 // As such, I avoided using OrderAccess::storestore(). In some cases
 860 // OrderAccess::fence() -- which incurs local latency on the executing
 861 // processor -- is a better choice as it scales on SMP systems.
 862 //
 863 // See http://blogs.oracle.com/dave/entry/biased_locking_in_hotspot for
 864 // a discussion of coherency costs. Note that all our current reference
 865 // platforms provide strong ST-ST order, so the issue is moot on IA32,
 866 // x64, and SPARC.
 867 //
 868 // As a general policy we use "volatile" to control compiler-based reordering
 869 // and explicit fences (barriers) to control for architectural reordering
 870 // performed by the CPU(s) or platform.
 871 
 872 struct SharedGlobals {
 873   char         _pad_prefix[OM_CACHE_LINE_SIZE];
 874   // These are highly shared mostly-read variables.
 875   // To avoid false-sharing they need to be the sole occupants of a cache line.
 876   volatile int stw_random;
 877   volatile int stw_cycle;
 878   DEFINE_PAD_MINUS_SIZE(1, OM_CACHE_LINE_SIZE, sizeof(volatile int) * 2);
 879   // Hot RW variable -- Sequester to avoid false-sharing
 880   volatile int hc_sequence;
 881   DEFINE_PAD_MINUS_SIZE(2, OM_CACHE_LINE_SIZE, sizeof(volatile int));
 882 };
 883 
 884 static SharedGlobals GVars;

 886 static markWord read_stable_mark(oop obj) {
 887   markWord mark = obj->mark();
 888   if (!mark.is_being_inflated()) {
 889     return mark;       // normal fast-path return
 890   }
 891 
 892   int its = 0;
 893   for (;;) {
 894     markWord mark = obj->mark();
 895     if (!mark.is_being_inflated()) {
 896       return mark;    // normal fast-path return
 897     }
 898 
 899     // The object is being inflated by some other thread.
 900     // The caller of read_stable_mark() must wait for inflation to complete.
 901     // Avoid live-lock
 902     // TODO: consider calling SafepointSynchronize::do_call_back() while
 903     // spinning to see if there's a safepoint pending.  If so, immediately
 904     // yielding or blocking would be appropriate.  Avoid spinning while
 905     // there is a safepoint pending.
 906     // TODO: add inflation contention performance counters.
 907     // TODO: restrict the aggregate number of spinners.
 908 
 909     ++its;
 910     if (its > 10000 || !os::is_MP()) {
 911       if (its & 1) {
 912         os::naked_yield();
 913       } else {
 914         // Note that the following code attenuates the livelock problem but is not
 915         // a complete remedy.  A more complete solution would require that the inflating
 916         // thread hold the associated inflation lock.  The following code simply restricts
 917         // the number of spinners to at most one.  We'll have N-2 threads blocked
 918         // on the inflationlock, 1 thread holding the inflation lock and using
 919         // a yield/park strategy, and 1 thread in the midst of inflation.
 920         // A more refined approach would be to change the encoding of INFLATING
 921         // to allow encapsulation of a native thread pointer.  Threads waiting for
 922         // inflation to complete would use CAS to push themselves onto a singly linked
 923         // list rooted at the markword.  Once enqueued, they'd loop, checking a per-thread flag
 924         // and calling park().  When inflation was complete the thread that accomplished inflation
 925         // would detach the list and set the markword to inflated with a single CAS and
 926         // then for each thread on the list, set the flag and unpark() the thread.
 927         // This is conceptually similar to muxAcquire-muxRelease, except that muxRelease
 928         // wakes at most one thread whereas we need to wake the entire list.
 929         int ix = (cast_from_oop<intptr_t>(obj) >> 5) & (NINFLATIONLOCKS-1);
 930         int YieldThenBlock = 0;
 931         assert(ix >= 0 && ix < NINFLATIONLOCKS, "invariant");
 932         assert((NINFLATIONLOCKS & (NINFLATIONLOCKS-1)) == 0, "invariant");
 933         Thread::muxAcquire(gInflationLocks + ix, "gInflationLock");
 934         while (obj->mark() == markWord::INFLATING()) {
 935           // Beware: NakedYield() is advisory and has almost no effect on some platforms
 936           // so we periodically call self->_ParkEvent->park(1).
 937           // We use a mixed spin/yield/block mechanism.
 938           if ((YieldThenBlock++) >= 16) {
 939             Thread::current()->_ParkEvent->park(1);
 940           } else {
 941             os::naked_yield();
 942           }
 943         }
 944         Thread::muxRelease(gInflationLocks + ix);
 945       }
 946     } else {
 947       SpinPause();       // SMP-polite spinning
 948     }
 949   }
 950 }

 952 // hashCode() generation :
 953 //
 954 // Possibilities:
 955 // * MD5Digest of {obj,stw_random}
 956 // * CRC32 of {obj,stw_random} or any linear-feedback shift register function.
 957 // * A DES- or AES-style SBox[] mechanism
 958 // * One of the Phi-based schemes, such as:
 959 //   2654435761 = 2^32 * Phi (golden ratio)
 960 //   HashCodeValue = ((uintptr_t(obj) >> 3) * 2654435761) ^ GVars.stw_random ;
 961 // * A variation of Marsaglia's shift-xor RNG scheme.
 962 // * (obj ^ stw_random) is appealing, but can result
 963 //   in undesirable regularity in the hashCode values of adjacent objects
 964 //   (objects allocated back-to-back, in particular).  This could potentially
 965 //   result in hashtable collisions and reduced hashtable efficiency.
 966 //   There are simple ways to "diffuse" the middle address bits over the
 967 //   generated hashCode values:
 968 
 969 static inline intptr_t get_next_hash(Thread* self, oop obj) {
 970   intptr_t value = 0;
 971   if (hashCode == 0) {
 972     // This form uses global Park-Miller RNG.
 973     // On MP system we'll have lots of RW access to a global, so the
 974     // mechanism induces lots of coherency traffic.
 975     value = os::random();
 976   } else if (hashCode == 1) {
 977     // This variation has the property of being stable (idempotent)
 978     // between STW operations.  This can be useful in some of the 1-0
 979     // synchronization schemes.
 980     intptr_t addr_bits = cast_from_oop<intptr_t>(obj) >> 3;
 981     value = addr_bits ^ (addr_bits >> 5) ^ GVars.stw_random;
 982   } else if (hashCode == 2) {
 983     value = 1;            // for sensitivity testing
 984   } else if (hashCode == 3) {
 985     value = ++GVars.hc_sequence;
 986   } else if (hashCode == 4) {
 987     value = cast_from_oop<intptr_t>(obj);
 988   } else {
 989     // Marsaglia's xor-shift scheme with thread-specific state
 990     // This is probably the best overall implementation -- we'll
 991     // likely make this the default in future releases.
 992     unsigned t = self->_hashStateX;
 993     t ^= (t << 11);
 994     self->_hashStateX = self->_hashStateY;
 995     self->_hashStateY = self->_hashStateZ;
 996     self->_hashStateZ = self->_hashStateW;
 997     unsigned v = self->_hashStateW;
 998     v = (v ^ (v >> 19)) ^ (t ^ (t >> 8));
 999     self->_hashStateW = v;
1000     value = v;
1001   }
1002 
1003   value &= markWord::hash_mask;
1004   if (value == 0) value = 0xBAD;
1005   assert(value != markWord::no_hash, "invariant");
1006   return value;
1007 }

1009 intptr_t ObjectSynchronizer::FastHashCode(Thread* self, oop obj) {
1010   if (EnableValhalla && obj->klass()->is_inline_klass()) {
1011     // VM should be calling bootstrap method
1012     ShouldNotReachHere();
1013   }
1014   if (UseBiasedLocking) {
1015     // NOTE: many places throughout the JVM do not expect a safepoint
1016     // to be taken here. However, we only ever bias Java instances and all
1017     // of the call sites of identity_hash that might revoke biases have
1018     // been checked to make sure they can handle a safepoint. The
1019     // added check of the bias pattern is to avoid useless calls to
1020     // thread-local storage.
1021     if (obj->mark().has_bias_pattern()) {
1022       // Handle for oop obj in case of STW safepoint
1023       Handle hobj(self, obj);
1024       // Relaxing assertion for bug 6320749.
1025       assert(Universe::verify_in_progress() ||
1026              !SafepointSynchronize::is_at_safepoint(),
1027              "biases should not be seen by VM thread here");
1028       BiasedLocking::revoke(hobj, JavaThread::current());
1029       obj = hobj();
1030       assert(!obj->mark().has_bias_pattern(), "biases should be revoked by now");
1031     }
1032   }
1033 
1034   // hashCode() is a heap mutator ...
1035   // Relaxing assertion for bug 6320749.
1036   assert(Universe::verify_in_progress() || DumpSharedSpaces ||
1037          !SafepointSynchronize::is_at_safepoint(), "invariant");
1038   assert(Universe::verify_in_progress() || DumpSharedSpaces ||
1039          self->is_Java_thread() , "invariant");
1040   assert(Universe::verify_in_progress() || DumpSharedSpaces ||
1041          ((JavaThread *)self)->thread_state() != _thread_blocked, "invariant");
1042 
1043   while (true) {
1044     ObjectMonitor* monitor = NULL;
1045     markWord temp, test;
1046     intptr_t hash;
1047     markWord mark = read_stable_mark(obj);
1048 
1049     // object should remain ineligible for biased locking
1050     assert(!mark.has_bias_pattern(), "invariant");
1051 
1052     if (mark.is_neutral()) {            // if this is a normal header
1053       hash = mark.hash();
1054       if (hash != 0) {                  // if it has a hash, just return it
1055         return hash;
1056       }
1057       hash = get_next_hash(self, obj);  // get a new hash
1058       temp = mark.copy_set_hash(hash);  // merge the hash into header
1059                                         // try to install the hash
1060       test = obj->cas_set_mark(temp, mark);
1061       if (test == mark) {               // if the hash was installed, return it
1062         return hash;
1063       }
1064       // Failed to install the hash. It could be that another thread
1065       // installed the hash just before our attempt or inflation has
1066       // occurred or... so we fall thru to inflate the monitor for
1067       // stability and then install the hash.
1068     } else if (mark.has_monitor()) {
1069       monitor = mark.monitor();
1070       temp = monitor->header();
1071       assert(temp.is_neutral(), "invariant: header=" INTPTR_FORMAT, temp.value());
1072       hash = temp.hash();
1073       if (hash != 0) {
1074         // It has a hash.
1075 
1076         // Separate load of dmw/header above from the loads in
1077         // is_being_async_deflated().
1078         if (support_IRIW_for_not_multiple_copy_atomic_cpu) {
1079           // A non-multiple copy atomic (nMCA) machine needs a bigger
1080           // hammer to separate the load above and the loads below.
1081           OrderAccess::fence();
1082         } else {
1083           OrderAccess::loadload();
1084         }
1085         if (monitor->is_being_async_deflated()) {
1086           // But we can't safely use the hash if we detect that async
1087           // deflation has occurred. So we attempt to restore the
1088           // header/dmw to the object's header so that we only retry
1089           // once if the deflater thread happens to be slow.
1090           monitor->install_displaced_markword_in_object(obj);
1091           continue;
1092         }
1093         return hash;
1094       }
1095       // Fall thru so we only have one place that installs the hash in
1096       // the ObjectMonitor.
1097     } else if (self->is_lock_owned((address)mark.locker())) {
1098       // This is a stack lock owned by the calling thread so fetch the
1099       // displaced markWord from the BasicLock on the stack.
1100       temp = mark.displaced_mark_helper();
1101       assert(temp.is_neutral(), "invariant: header=" INTPTR_FORMAT, temp.value());
1102       hash = temp.hash();
1103       if (hash != 0) {                  // if it has a hash, just return it
1104         return hash;
1105       }
1106       // WARNING:
1107       // The displaced header in the BasicLock on a thread's stack
1108       // is strictly immutable. It CANNOT be changed in ANY cases.
1109       // So we have to inflate the stack lock into an ObjectMonitor
1110       // even if the current thread owns the lock. The BasicLock on
1111       // a thread's stack can be asynchronously read by other threads
1112       // during an inflate() call so any change to that stack memory
1113       // may not propagate to other threads correctly.
1114     }
1115 
1116     // Inflate the monitor to set the hash.
1117 
1118     // An async deflation can race after the inflate() call and before we
1119     // can update the ObjectMonitor's header with the hash value below.
1120     monitor = inflate(self, obj, inflate_cause_hash_code);
1121     // Load ObjectMonitor's header/dmw field and see if it has a hash.
1122     mark = monitor->header();
1123     assert(mark.is_neutral(), "invariant: header=" INTPTR_FORMAT, mark.value());
1124     hash = mark.hash();
1125     if (hash == 0) {                    // if it does not have a hash
1126       hash = get_next_hash(self, obj);  // get a new hash
1127       temp = mark.copy_set_hash(hash);  // merge the hash into header
1128       assert(temp.is_neutral(), "invariant: header=" INTPTR_FORMAT, temp.value());
1129       uintptr_t v = Atomic::cmpxchg((volatile uintptr_t*)monitor->header_addr(), mark.value(), temp.value());
1130       test = markWord(v);
1131       if (test != mark) {
1132         // The attempt to update the ObjectMonitor's header/dmw field
1133         // did not work. This can happen if another thread managed to
1134         // merge in the hash just before our cmpxchg().
1135         // If we add any new usages of the header/dmw field, this code
1136         // will need to be updated.
1137         hash = test.hash();
1138         assert(test.is_neutral(), "invariant: header=" INTPTR_FORMAT, test.value());
1139         assert(hash != 0, "should only have lost the race to a thread that set a non-zero hash");
1140       }
1141       if (monitor->is_being_async_deflated()) {
1142         // If we detect that async deflation has occurred, then we
1143         // attempt to restore the header/dmw to the object's header
1144         // so that we only retry once if the deflater thread happens
1145         // to be slow.
1146         monitor->install_displaced_markword_in_object(obj);
1147         continue;
1148       }
1149     }
1150     // We finally get the hash.
1151     return hash;
1152   }
1153 }

1398 // -----------------------------------------------------------------------------
1399 // ObjectMonitor Lifecycle
1400 // -----------------------
1401 // Inflation unlinks monitors from om_list_globals._free_list or a per-thread
1402 // free list and associates them with objects. Async deflation disassociates
1403 // idle monitors from objects. Such scavenged monitors are returned to the
1404 // om_list_globals._free_list.
1405 //
1406 // ObjectMonitors reside in type-stable memory (TSM) and are immortal.
1407 //
1408 // Lifecycle:
1409 // --   unassigned and on the om_list_globals._free_list
1410 // --   unassigned and on a per-thread free list
1411 // --   assigned to an object.  The object is inflated and the mark refers
1412 //      to the ObjectMonitor.
1413 
1414 ObjectMonitor* ObjectSynchronizer::om_alloc(Thread* self) {
1415   // A large MAXPRIVATE value reduces both list lock contention
1416   // and list coherency traffic, but also tends to increase the
1417   // number of ObjectMonitors in circulation as well as the
1418   // scavenge costs.  As usual, we lean toward time in space-time
1419   // tradeoffs.
1420   const int MAXPRIVATE = 1024;
1421   NoSafepointVerifier nsv;
1422 
1423   for (;;) {
1424     ObjectMonitor* m;
1425 
1426     // 1: try to allocate from the thread's local om_free_list.
1427     // Threads will attempt to allocate first from their local list, then
1428     // from the global list, and only after those attempts fail will the
1429     // thread attempt to instantiate new monitors. Thread-local free lists
1430     // improve allocation latency, as well as reducing coherency traffic
1431     // on the shared global list.
1432     m = take_from_start_of_om_free_list(self);
1433     if (m != NULL) {
1434       guarantee(m->object() == NULL, "invariant");
1435       m->set_allocation_state(ObjectMonitor::New);
1436       prepend_to_om_in_use_list(self, m);
1437       return m;
1438     }
1439 
1440     // 2: try to allocate from the global om_list_globals._free_list
1441     // If we're using thread-local free lists then try
1442     // to reprovision the caller's free list.
1443     if (Atomic::load(&om_list_globals._free_list) != NULL) {
1444       // Reprovision the thread's om_free_list.
1445       // Use bulk transfers to reduce the allocation rate and heat
1446       // on various locks.
1447       for (int i = self->om_free_provision; --i >= 0;) {
1448         ObjectMonitor* take = take_from_start_of_global_free_list();
1449         if (take == NULL) {
1450           break;  // No more are available.
1451         }
1452         guarantee(take->object() == NULL, "invariant");
1453         // We allowed 3 field values to linger during async deflation.
1454         // Clear or restore them as appropriate.
1455         take->set_header(markWord::zero());
1456         // DEFLATER_MARKER is the only non-NULL value we should see here.
1457         take->try_set_owner_from(DEFLATER_MARKER, NULL);
1458         if (take->contentions() < 0) {
1459           // Add back max_jint to restore the contentions field to its
1460           // proper value.
1461           take->add_to_contentions(max_jint);
1462 
1463 #ifdef ASSERT
1464           jint l_contentions = take->contentions();
1465           assert(l_contentions >= 0, "must not be negative: l_contentions=%d, contentions=%d",
1466                  l_contentions, take->contentions());
1467 #endif
1468         }
1469         take->Recycle();
1470         // Since we're taking from the global free-list, take must be Free.
1471         // om_release() also sets the allocation state to Free because it
1472         // is called from other code paths.
1473         assert(take->is_free(), "invariant");
1474         om_release(self, take, false);
1475       }
1476       self->om_free_provision += 1 + (self->om_free_provision / 2);
1477       if (self->om_free_provision > MAXPRIVATE) self->om_free_provision = MAXPRIVATE;
1478       continue;
1479     }
1480 
1481     // 3: allocate a block of new ObjectMonitors
1482     // Both the local and global free lists are empty -- resort to malloc().
1483     // In the current implementation ObjectMonitors are TSM - immortal.
1484     // Ideally, we'd write "new ObjectMonitor[_BLOCKSIZE], but we want
1485     // each ObjectMonitor to start at the beginning of a cache line,
1486     // so we use align_up().
1487     // A better solution would be to use C++ placement-new.
1488     // BEWARE: As it stands currently, we don't run the ctors!
1489     assert(_BLOCKSIZE > 1, "invariant");
1490     size_t neededsize = sizeof(PaddedObjectMonitor) * _BLOCKSIZE;
1491     PaddedObjectMonitor* temp;
1492     size_t aligned_size = neededsize + (OM_CACHE_LINE_SIZE - 1);
1493     void* real_malloc_addr = NEW_C_HEAP_ARRAY(char, aligned_size, mtInternal);
1494     temp = (PaddedObjectMonitor*)align_up(real_malloc_addr, OM_CACHE_LINE_SIZE);
1495     (void)memset((void *) temp, 0, neededsize);
1496 
1497     // Format the block.
1498     // initialize the linked list, each monitor points to its next
1499     // forming the single linked free list, the very first monitor
1500     // will points to next block, which forms the block list.
1501     // The trick of using the 1st element in the block as g_block_list
1502     // linkage should be reconsidered.  A better implementation would
1503     // look like: class Block { Block * next; int N; ObjectMonitor Body [N] ; }
1504 
1505     for (int i = 1; i < _BLOCKSIZE; i++) {
1506       temp[i].set_next_om((ObjectMonitor*)&temp[i + 1]);
1507       assert(temp[i].is_free(), "invariant");
1508     }
1509 
1510     // terminate the last monitor as the end of list
1511     temp[_BLOCKSIZE - 1].set_next_om((ObjectMonitor*)NULL);
1512 
1513     // Element [0] is reserved for global list linkage
1514     temp[0].set_object(CHAINMARKER);
1515 
1516     // Consider carving out this thread's current request from the
1517     // block in hand.  This avoids some lock traffic and redundant
1518     // list activity.
1519 
1520     prepend_block_to_lists(temp);
1521   }
1522 }

1524 // Place "m" on the caller's private per-thread om_free_list.
1525 // In practice there's no need to clamp or limit the number of
1526 // monitors on a thread's om_free_list as the only non-allocation time
1527 // we'll call om_release() is to return a monitor to the free list after
1528 // a CAS attempt failed. This doesn't allow unbounded #s of monitors to
1529 // accumulate on a thread's free list.
1530 //
1531 // Key constraint: all ObjectMonitors on a thread's free list and the global
1532 // free list must have their object field set to null. This prevents the
1533 // scavenger -- deflate_monitor_list_using_JT() -- from reclaiming them
1534 // while we are trying to release them.
1535 
1536 void ObjectSynchronizer::om_release(Thread* self, ObjectMonitor* m,
1537                                     bool from_per_thread_alloc) {
1538   guarantee(m->header().value() == 0, "invariant");
1539   guarantee(m->object() == NULL, "invariant");
1540   NoSafepointVerifier nsv;
1541 
1542   if ((m->is_busy() | m->_recursions) != 0) {
1543     stringStream ss;
1544     fatal("freeing in-use monitor: %s, recursions=" INTX_FORMAT,
1545           m->is_busy_to_string(&ss), m->_recursions);
1546   }
1547   m->set_allocation_state(ObjectMonitor::Free);
1548   // _next_om is used for both per-thread in-use and free lists so
1549   // we have to remove 'm' from the in-use list first (as needed).
1550   if (from_per_thread_alloc) {
1551     // Need to remove 'm' from om_in_use_list.
1552     ObjectMonitor* mid = NULL;
1553     ObjectMonitor* next = NULL;
1554 
1555     // This list walk can race with another list walker or with async
1556     // deflation so we have to worry about an ObjectMonitor being
1557     // removed from this list while we are walking it.
1558 
1559     // Lock the list head to avoid racing with another list walker
1560     // or with async deflation.
1561     if ((mid = get_list_head_locked(&self->om_in_use_list)) == NULL) {
1562       fatal("thread=" INTPTR_FORMAT " in-use list must not be empty.", p2i(self));
1563     }
1564     next = unmarked_next(mid);
1565     if (m == mid) {
1566       // First special case:
1567       // 'm' matches mid, is the list head and is locked. Switch the list
1568       // head to next which unlocks the list head, but leaves the extracted
1569       // mid locked:
1570       Atomic::store(&self->om_in_use_list, next);
1571     } else if (m == next) {
1572       // Second special case:
1573       // 'm' matches next after the list head and we already have the list
1574       // head locked so set mid to what we are extracting:
1575       mid = next;
1576       // Lock mid to prevent races with a list walker or an async
1577       // deflater thread that's ahead of us. The locked list head
1578       // prevents races from behind us.
1579       om_lock(mid);
1580       // Update next to what follows mid (if anything):
1581       next = unmarked_next(mid);
1582       // Switch next after the list head to new next which unlocks the
1583       // list head, but leaves the extracted mid locked:
1584       self->om_in_use_list->set_next_om(next);
1585     } else {
1586       // We have to search the list to find 'm'.
1587       guarantee(next != NULL, "thread=" INTPTR_FORMAT ": om_in_use_list=" INTPTR_FORMAT
1588                 " is too short.", p2i(self), p2i(self->om_in_use_list));
1589       // Our starting anchor is next after the list head which is the
1590       // last ObjectMonitor we checked:
1591       ObjectMonitor* anchor = next;
1592       // Lock anchor to prevent races with a list walker or an async
1593       // deflater thread that's ahead of us. The locked list head
1594       // prevents races from behind us.
1595       om_lock(anchor);
1596       om_unlock(mid);  // Unlock the list head now that anchor is locked.
1597       while ((mid = unmarked_next(anchor)) != NULL) {
1598         if (m == mid) {
1599           // We found 'm' on the per-thread in-use list so extract it.
1600           // Update next to what follows mid (if anything):
1601           next = unmarked_next(mid);
1602           // Switch next after the anchor to new next which unlocks the
1603           // anchor, but leaves the extracted mid locked:
1604           anchor->set_next_om(next);
1605           break;
1606         } else {
1607           // Lock the next anchor to prevent races with a list walker
1608           // or an async deflater thread that's ahead of us. The locked
1609           // current anchor prevents races from behind us.
1610           om_lock(mid);
1611           // Unlock current anchor now that next anchor is locked:
1612           om_unlock(anchor);
1613           anchor = mid;  // Advance to new anchor and try again.
1614         }
1615       }
1616     }
1617 
1618     if (mid == NULL) {
1619       // Reached end of the list and didn't find 'm' so:
1620       fatal("thread=" INTPTR_FORMAT " must find m=" INTPTR_FORMAT "on om_in_use_list="
1621             INTPTR_FORMAT, p2i(self), p2i(m), p2i(self->om_in_use_list));
1622     }
1623 
1624     // At this point mid is disconnected from the in-use list so
1625     // its lock no longer has any effects on the in-use list.
1626     Atomic::dec(&self->om_in_use_count);
1627     // Unlock mid, but leave the next value for any lagging list
1628     // walkers. It will get cleaned up when mid is prepended to
1629     // the thread's free list:
1630     om_unlock(mid);
1631   }
1632 
1633   prepend_to_om_free_list(self, m);
1634   guarantee(m->is_free(), "invariant");
1635 }

1637 // Return ObjectMonitors on a moribund thread's free and in-use
1638 // lists to the appropriate global lists. The ObjectMonitors on the
1639 // per-thread in-use list may still be in use by other threads.
1640 //
1641 // We currently call om_flush() from Threads::remove() before the
1642 // thread has been excised from the thread list and is no longer a
1643 // mutator. In particular, this ensures that the thread's in-use
1644 // monitors are scanned by a GC safepoint, either via Thread::oops_do()
1645 // (before om_flush() is called) or via ObjectSynchronizer::oops_do()
1646 // (after om_flush() is called).
1647 //
1648 // deflate_global_idle_monitors_using_JT() and
1649 // deflate_per_thread_idle_monitors_using_JT() (in another thread) can
1650 // run at the same time as om_flush() so we have to follow a careful
1651 // protocol to prevent list corruption.
1652 
1653 void ObjectSynchronizer::om_flush(Thread* self) {
1654   // Process the per-thread in-use list first to be consistent.
1655   int in_use_count = 0;
1656   ObjectMonitor* in_use_list = NULL;
1657   ObjectMonitor* in_use_tail = NULL;
1658   NoSafepointVerifier nsv;
1659 
1660   // This function can race with a list walker or with an async
1661   // deflater thread so we lock the list head to prevent confusion.
1662   // An async deflater thread checks to see if the target thread
1663   // is exiting, but if it has made it past that check before we
1664   // started exiting, then it is racing to get to the in-use list.
1665   if ((in_use_list = get_list_head_locked(&self->om_in_use_list)) != NULL) {
1666     // At this point, we have locked the in-use list head so a racing
1667     // thread cannot come in after us. However, a racing thread could
1668     // be ahead of us; we'll detect that and delay to let it finish.
1669     //
1670     // The thread is going away, however the ObjectMonitors on the
1671     // om_in_use_list may still be in-use by other threads. Link
1672     // them to in_use_tail, which will be linked into the global
1673     // in-use list (om_list_globals._in_use_list) below.
1674     //
1675     // Account for the in-use list head before the loop since it is
1676     // already locked (by this thread):
1677     in_use_tail = in_use_list;
1678     in_use_count++;
1679     for (ObjectMonitor* cur_om = unmarked_next(in_use_list); cur_om != NULL;) {
1680       if (is_locked(cur_om)) {
1681         // cur_om is locked so there must be a racing walker or async
1682         // deflater thread ahead of us so we'll give it a chance to finish.
1683         while (is_locked(cur_om)) {
1684           os::naked_short_sleep(1);
1685         }
1686         // Refetch the possibly changed next field and try again.
1687         cur_om = unmarked_next(in_use_tail);
1688         continue;
1689       }
1690       if (cur_om->object() == NULL) {
1691         // cur_om was deflated and the object ref was cleared while it
1692         // was locked. We happened to see it just after it was unlocked
1693         // (and added to the free list). Refetch the possibly changed
1694         // next field and try again.
1695         cur_om = unmarked_next(in_use_tail);
1696         continue;
1697       }
1698       in_use_tail = cur_om;
1699       in_use_count++;
1700       cur_om = unmarked_next(cur_om);
1701     }
1702     guarantee(in_use_tail != NULL, "invariant");
1703 #ifdef ASSERT
1704     int l_om_in_use_count = Atomic::load(&self->om_in_use_count);
1705     assert(l_om_in_use_count == in_use_count, "in-use counts don't match: "
1706            "l_om_in_use_count=%d, in_use_count=%d", l_om_in_use_count, in_use_count);
1707 #endif
1708     Atomic::store(&self->om_in_use_count, 0);
1709     // Clear the in-use list head (which also unlocks it):
1710     Atomic::store(&self->om_in_use_list, (ObjectMonitor*)NULL);
1711     om_unlock(in_use_list);
1712   }
1713 
1714   int free_count = 0;
1715   ObjectMonitor* free_list = NULL;
1716   ObjectMonitor* free_tail = NULL;
1717   // This function can race with a list walker thread so we lock the
1718   // list head to prevent confusion.
1719   if ((free_list = get_list_head_locked(&self->om_free_list)) != NULL) {
1720     // At this point, we have locked the free list head so a racing
1721     // thread cannot come in after us. However, a racing thread could
1722     // be ahead of us; we'll detect that and delay to let it finish.
1723     //
1724     // The thread is going away. Set 'free_tail' to the last per-thread free
1725     // monitor which will be linked to om_list_globals._free_list below.
1726     //
1727     // Account for the free list head before the loop since it is
1728     // already locked (by this thread):
1729     free_tail = free_list;
1730     free_count++;
1731     for (ObjectMonitor* s = unmarked_next(free_list); s != NULL; s = unmarked_next(s)) {
1732       if (is_locked(s)) {
1733         // s is locked so there must be a racing walker thread ahead
1734         // of us so we'll give it a chance to finish.
1735         while (is_locked(s)) {
1736           os::naked_short_sleep(1);
1737         }
1738       }
1739       free_tail = s;
1740       free_count++;
1741       guarantee(s->object() == NULL, "invariant");
1742       if (s->is_busy()) {
1743         stringStream ss;
1744         fatal("must be !is_busy: %s", s->is_busy_to_string(&ss));
1745       }
1746     }
1747     guarantee(free_tail != NULL, "invariant");
1748 #ifdef ASSERT
1749     int l_om_free_count = Atomic::load(&self->om_free_count);
1750     assert(l_om_free_count == free_count, "free counts don't match: "
1751            "l_om_free_count=%d, free_count=%d", l_om_free_count, free_count);
1752 #endif
1753     Atomic::store(&self->om_free_count, 0);
1754     Atomic::store(&self->om_free_list, (ObjectMonitor*)NULL);
1755     om_unlock(free_list);
1756   }
1757 
1758   if (free_tail != NULL) {
1759     prepend_list_to_global_free_list(free_list, free_tail, free_count);
1760   }
1761 
1762   if (in_use_tail != NULL) {
1763     prepend_list_to_global_in_use_list(in_use_list, in_use_tail, in_use_count);
1764   }
1765 
1766   LogStreamHandle(Debug, monitorinflation) lsh_debug;
1767   LogStreamHandle(Info, monitorinflation) lsh_info;
1768   LogStream* ls = NULL;
1769   if (log_is_enabled(Debug, monitorinflation)) {
1770     ls = &lsh_debug;
1771   } else if ((free_count != 0 || in_use_count != 0) &&
1772              log_is_enabled(Info, monitorinflation)) {
1773     ls = &lsh_info;
1774   }
1775   if (ls != NULL) {
1776     ls->print_cr("om_flush: jt=" INTPTR_FORMAT ", free_count=%d"
1777                  ", in_use_count=%d" ", om_free_provision=%d",
1778                  p2i(self), free_count, in_use_count, self->om_free_provision);
1779   }

1806 ObjectMonitor* ObjectSynchronizer::inflate(Thread* self, oop object,
1807                                            const InflateCause cause) {
1808   // Inflate mutates the heap ...
1809   // Relaxing assertion for bug 6320749.
1810   assert(Universe::verify_in_progress() ||
1811          !SafepointSynchronize::is_at_safepoint(), "invariant");
1812 
1813   if (EnableValhalla) {
1814     guarantee(!object->klass()->is_inline_klass(), "Attempt to inflate inline type");
1815   }
1816 
1817   EventJavaMonitorInflate event;
1818 
1819   for (;;) {
1820     const markWord mark = object->mark();
1821     assert(!mark.has_bias_pattern(), "invariant");
1822 
1823     // The mark can be in one of the following states:
1824     // *  Inflated     - just return
1825     // *  Stack-locked - coerce it to inflated
1826     // *  INFLATING    - busy wait for conversion to complete
1827     // *  Neutral      - aggressively inflate the object.
1828     // *  BIASED       - Illegal.  We should never see this
1829 
1830     // CASE: inflated
1831     if (mark.has_monitor()) {
1832       ObjectMonitor* inf = mark.monitor();
1833       markWord dmw = inf->header();
1834       assert(dmw.is_neutral(), "invariant: header=" INTPTR_FORMAT, dmw.value());
1835       assert(ObjectSynchronizer::verify_objmon_isinpool(inf), "monitor is invalid");
1836       return inf;
1837     }
1838 
1839     // CASE: inflation in progress - inflating over a stack-lock.
1840     // Some other thread is converting from stack-locked to inflated.
1841     // Only that thread can complete inflation -- other threads must wait.
1842     // The INFLATING value is transient.
1843     // Currently, we spin/yield/park and poll the markword, waiting for inflation to finish.
1844     // We could always eliminate polling by parking the thread on some auxiliary list.
1845     if (mark == markWord::INFLATING()) {
1846       read_stable_mark(object);
1847       continue;
1848     }
1849 
1850     // CASE: stack-locked
1851     // Could be stack-locked either by this thread or by some other thread.
1852     //
1853     // Note that we allocate the objectmonitor speculatively, _before_ attempting
1854     // to install INFLATING into the mark word.  We originally installed INFLATING,
1855     // allocated the objectmonitor, and then finally STed the address of the
1856     // objectmonitor into the mark.  This was correct, but artificially lengthened
1857     // the interval in which INFLATED appeared in the mark, thus increasing
1858     // the odds of inflation contention.
1859     //
1860     // We now use per-thread private objectmonitor free lists.
1861     // These list are reprovisioned from the global free list outside the
1862     // critical INFLATING...ST interval.  A thread can transfer
1863     // multiple objectmonitors en-mass from the global free list to its local free list.
1864     // This reduces coherency traffic and lock contention on the global free list.
1865     // Using such local free lists, it doesn't matter if the om_alloc() call appears
1866     // before or after the CAS(INFLATING) operation.
1867     // See the comments in om_alloc().
1868 
1869     LogStreamHandle(Trace, monitorinflation) lsh;
1870 
1871     if (mark.has_locker()) {
1872       ObjectMonitor* m = om_alloc(self);
1873       // Optimistically prepare the objectmonitor - anticipate successful CAS
1874       // We do this before the CAS in order to minimize the length of time
1875       // in which INFLATING appears in the mark.
1876       m->Recycle();
1877       m->_Responsible  = NULL;
1878       m->_SpinDuration = ObjectMonitor::Knob_SpinLimit;   // Consider: maintain by type/class
1879 
1880       markWord cmp = object->cas_set_mark(markWord::INFLATING(), mark);
1881       if (cmp != mark) {
1882         // om_release() will reset the allocation state from New to Free.
1883         om_release(self, m, true);
1884         continue;       // Interference -- just retry
1885       }
1886 
1887       // We've successfully installed INFLATING (0) into the mark-word.
1888       // This is the only case where 0 will appear in a mark-word.
1889       // Only the singular thread that successfully swings the mark-word
1890       // to 0 can perform (or more precisely, complete) inflation.
1891       //
1892       // Why do we CAS a 0 into the mark-word instead of just CASing the
1893       // mark-word from the stack-locked value directly to the new inflated state?
1894       // Consider what happens when a thread unlocks a stack-locked object.
1895       // It attempts to use CAS to swing the displaced header value from the
1896       // on-stack BasicLock back into the object header.  Recall also that the
1897       // header value (hash code, etc) can reside in (a) the object header, or
1898       // (b) a displaced header associated with the stack-lock, or (c) a displaced
1899       // header in an ObjectMonitor.  The inflate() routine must copy the header
1900       // value from the BasicLock on the owner's stack to the ObjectMonitor, all
1901       // the while preserving the hashCode stability invariants.  If the owner
1902       // decides to release the lock while the value is 0, the unlock will fail
1903       // and control will eventually pass from slow_exit() to inflate.  The owner
1904       // will then spin, waiting for the 0 value to disappear.   Put another way,
1905       // the 0 causes the owner to stall if the owner happens to try to
1906       // drop the lock (restoring the header from the BasicLock to the object)
1907       // while inflation is in-progress.  This protocol avoids races that might
1908       // would otherwise permit hashCode values to change or "flicker" for an object.
1909       // Critically, while object->mark is 0 mark.displaced_mark_helper() is stable.
1910       // 0 serves as a "BUSY" inflate-in-progress indicator.
1911 
1912 
1913       // fetch the displaced mark from the owner's stack.
1914       // The owner can't die or unwind past the lock while our INFLATING
1915       // object is in the mark.  Furthermore the owner can't complete
1916       // an unlock on the object, either.
1917       markWord dmw = mark.displaced_mark_helper();
1918       // Catch if the object's header is not neutral (not locked and
1919       // not marked is what we care about here).
1920       assert(dmw.is_neutral(), "invariant: header=" INTPTR_FORMAT, dmw.value());
1921 
1922       // Setup monitor fields to proper values -- prepare the monitor
1923       m->set_header(dmw);
1924 
1925       // Optimization: if the mark.locker stack address is associated
1926       // with this thread we could simply set m->_owner = self.
1927       // Note that a thread can inflate an object
1928       // that it has stack-locked -- as might happen in wait() -- directly
1929       // with CAS.  That is, we can avoid the xchg-NULL .... ST idiom.
1930       m->set_owner_from(NULL, DEFLATER_MARKER, mark.locker());
1931       m->set_object(object);
1932       // TODO-FIXME: assert BasicLock->dhw != 0.
1933 
1934       // Must preserve store ordering. The monitor state must
1935       // be stable at the time of publishing the monitor address.
1936       guarantee(object->mark() == markWord::INFLATING(), "invariant");
1937       object->release_set_mark(markWord::encode(m));
1938 
1939       // Once ObjectMonitor is configured and the object is associated
1940       // with the ObjectMonitor, it is safe to allow async deflation:
1941       assert(m->is_new(), "freshly allocated monitor must be new");
1942       m->set_allocation_state(ObjectMonitor::Old);
1943 
1944       // Hopefully the performance counters are allocated on distinct cache lines
1945       // to avoid false sharing on MP systems ...
1946       OM_PERFDATA_OP(Inflations, inc());
1947       if (log_is_enabled(Trace, monitorinflation)) {
1948         ResourceMark rm(self);
1949         lsh.print_cr("inflate(has_locker): object=" INTPTR_FORMAT ", mark="
1950                      INTPTR_FORMAT ", type='%s'", p2i(object),
1951                      object->mark().value(), object->klass()->external_name());
1952       }
1953       if (event.should_commit()) {
1954         post_monitor_inflate_event(&event, object, cause);
1955       }
1956       return m;
1957     }
1958 
1959     // CASE: neutral
1960     // TODO-FIXME: for entry we currently inflate and then try to CAS _owner.
1961     // If we know we're inflating for entry it's better to inflate by swinging a
1962     // pre-locked ObjectMonitor pointer into the object header.   A successful
1963     // CAS inflates the object *and* confers ownership to the inflating thread.
1964     // In the current implementation we use a 2-step mechanism where we CAS()
1965     // to inflate and then CAS() again to try to swing _owner from NULL to self.
1966     // An inflateTry() method that we could call from enter() would be useful.
1967 
1968     // Catch if the object's header is not neutral (not locked and
1969     // not marked is what we care about here).
1970     assert(mark.is_neutral(), "invariant: header=" INTPTR_FORMAT, mark.value());
1971     ObjectMonitor* m = om_alloc(self);
1972     // prepare m for installation - set monitor to initial state
1973     m->Recycle();
1974     m->set_header(mark);
1975     // DEFLATER_MARKER is the only non-NULL value we should see here.
1976     m->try_set_owner_from(DEFLATER_MARKER, NULL);
1977     m->set_object(object);
1978     m->_Responsible  = NULL;
1979     m->_SpinDuration = ObjectMonitor::Knob_SpinLimit;       // consider: keep metastats by type/class
1980 
1981     if (object->cas_set_mark(markWord::encode(m), mark) != mark) {
1982       m->set_header(markWord::zero());
1983       m->set_object(NULL);
1984       m->Recycle();
1985       // om_release() will reset the allocation state from New to Free.
1986       om_release(self, m, true);
1987       m = NULL;
1988       continue;
1989       // interference - the markword changed - just retry.
1990       // The state-transitions are one-way, so there's no chance of
1991       // live-lock -- "Inflated" is an absorbing state.
1992     }
1993 
1994     // Once the ObjectMonitor is configured and object is associated
1995     // with the ObjectMonitor, it is safe to allow async deflation:
1996     assert(m->is_new(), "freshly allocated monitor must be new");
1997     m->set_allocation_state(ObjectMonitor::Old);
1998 
1999     // Hopefully the performance counters are allocated on distinct
2000     // cache lines to avoid false sharing on MP systems ...
2001     OM_PERFDATA_OP(Inflations, inc());
2002     if (log_is_enabled(Trace, monitorinflation)) {
2003       ResourceMark rm(self);
2004       lsh.print_cr("inflate(neutral): object=" INTPTR_FORMAT ", mark="
2005                    INTPTR_FORMAT ", type='%s'", p2i(object),
2006                    object->mark().value(), object->klass()->external_name());
2007     }
2008     if (event.should_commit()) {
2009       post_monitor_inflate_event(&event, object, cause);
2010     }
2011     return m;
2012   }

2034 // Deflate the specified ObjectMonitor if not in-use using a JavaThread.
2035 // Returns true if it was deflated and false otherwise.
2036 //
2037 // The async deflation protocol sets owner to DEFLATER_MARKER and
2038 // makes contentions negative as signals to contending threads that
2039 // an async deflation is in progress. There are a number of checks
2040 // as part of the protocol to make sure that the calling thread has
2041 // not lost the race to a contending thread.
2042 //
2043 // The ObjectMonitor has been successfully async deflated when:
2044 //   (contentions < 0)
2045 // Contending threads that see that condition know to retry their operation.
2046 //
2047 bool ObjectSynchronizer::deflate_monitor_using_JT(ObjectMonitor* mid,
2048                                                   ObjectMonitor** free_head_p,
2049                                                   ObjectMonitor** free_tail_p) {
2050   assert(Thread::current()->is_Java_thread(), "precondition");
2051   // A newly allocated ObjectMonitor should not be seen here so we
2052   // avoid an endless inflate/deflate cycle.
2053   assert(mid->is_old(), "must be old: allocation_state=%d",
2054          (int) mid->allocation_state());
2055 
2056   if (mid->is_busy()) {
2057     // Easy checks are first - the ObjectMonitor is busy so no deflation.
2058     return false;
2059   }
2060 
2061   // Set a NULL owner to DEFLATER_MARKER to force any contending thread
2062   // through the slow path. This is just the first part of the async
2063   // deflation dance.
2064   if (mid->try_set_owner_from(NULL, DEFLATER_MARKER) != NULL) {
2065     // The owner field is no longer NULL so we lost the race since the
2066     // ObjectMonitor is now busy.
2067     return false;
2068   }
2069 
2070   if (mid->contentions() > 0 || mid->_waiters != 0) {
2071     // Another thread has raced to enter the ObjectMonitor after
2072     // mid->is_busy() above or has already entered and waited on
2073     // it which makes it busy so no deflation. Restore owner to
2074     // NULL if it is still DEFLATER_MARKER.
2075     if (mid->try_set_owner_from(DEFLATER_MARKER, NULL) != DEFLATER_MARKER) {
2076       // Deferred decrement for the JT EnterI() that cancelled the async deflation.
2077       mid->add_to_contentions(-1);
2078     }
2079     return false;
2080   }
2081 
2082   // Make a zero contentions field negative to force any contending threads
2083   // to retry. This is the second part of the async deflation dance.
2084   if (Atomic::cmpxchg(&mid->_contentions, (jint)0, -max_jint) != 0) {
2085     // Contentions was no longer 0 so we lost the race since the
2086     // ObjectMonitor is now busy. Restore owner to NULL if it is
2087     // still DEFLATER_MARKER:
2088     if (mid->try_set_owner_from(DEFLATER_MARKER, NULL) != DEFLATER_MARKER) {
2089       // Deferred decrement for the JT EnterI() that cancelled the async deflation.
2090       mid->add_to_contentions(-1);
2091     }
2092     return false;
2093   }
2094 
2095   // Sanity checks for the races:
2096   guarantee(mid->owner_is_DEFLATER_MARKER(), "must be deflater marker");
2097   guarantee(mid->contentions() < 0, "must be negative: contentions=%d",
2098             mid->contentions());
2099   guarantee(mid->_waiters == 0, "must be 0: waiters=%d", mid->_waiters);
2100   guarantee(mid->_cxq == NULL, "must be no contending threads: cxq="
2101             INTPTR_FORMAT, p2i(mid->_cxq));
2102   guarantee(mid->_EntryList == NULL,
2103             "must be no entering threads: EntryList=" INTPTR_FORMAT,
2104             p2i(mid->_EntryList));
2105 
2106   const oop obj = (oop) mid->object();
2107   if (log_is_enabled(Trace, monitorinflation)) {
2108     ResourceMark rm;
2109     log_trace(monitorinflation)("deflate_monitor_using_JT: "
2110                                 "object=" INTPTR_FORMAT ", mark="
2111                                 INTPTR_FORMAT ", type='%s'",
2112                                 p2i(obj), obj->mark().value(),
2113                                 obj->klass()->external_name());
2114   }
2115 
2116   // Install the old mark word if nobody else has already done it.
2117   mid->install_displaced_markword_in_object(obj);
2118   mid->clear_common();
2119 
2120   assert(mid->object() == NULL, "must be NULL: object=" INTPTR_FORMAT,
2121          p2i(mid->object()));
2122   assert(mid->is_free(), "must be free: allocation_state=%d",
2123          (int)mid->allocation_state());
2124 
2125   // Move the deflated ObjectMonitor to the working free list
2126   // defined by free_head_p and free_tail_p.
2127   if (*free_head_p == NULL) {
2128     // First one on the list.
2129     *free_head_p = mid;
2130   }
2131   if (*free_tail_p != NULL) {
2132     // We append to the list so the caller can use mid->_next_om
2133     // to fix the linkages in its context.
2134     ObjectMonitor* prevtail = *free_tail_p;
2135     // prevtail should have been cleaned up by the caller:
2136 #ifdef ASSERT
2137     ObjectMonitor* l_next_om = unmarked_next(prevtail);
2138     assert(l_next_om == NULL, "must be NULL: _next_om=" INTPTR_FORMAT, p2i(l_next_om));
2139 #endif
2140     om_lock(prevtail);
2141     prevtail->set_next_om(mid);  // prevtail now points to mid (and is unlocked)
2142   }
2143   *free_tail_p = mid;
2144 
2145   // At this point, mid->_next_om still refers to its current
2146   // value and another ObjectMonitor's _next_om field still
2147   // refers to this ObjectMonitor. Those linkages have to be
2148   // cleaned up by the caller who has the complete context.
2149 
2150   // We leave owner == DEFLATER_MARKER and contentions < 0
2151   // to force any racing threads to retry.
2152   return true;  // Success, ObjectMonitor has been deflated.
2153 }

2158 // If a safepoint has started, then we save state via saved_mid_in_use_p

2161 int ObjectSynchronizer::deflate_monitor_list_using_JT(ObjectMonitor** list_p,
2162                                                       int* count_p,

2167 
2168   ObjectMonitor* cur_mid_in_use = NULL;
2169   ObjectMonitor* mid = NULL;

2175   // We use the more complicated lock-cur_mid_in_use-and-mid-as-we-go
2176   // protocol because om_release() can do list deletions in parallel;
2177   // this also prevents races with a list walker thread. We also
2178   // lock-next-next-as-we-go to prevent an om_flush() that is behind
2179   // this thread from passing us.
2180   if (*saved_mid_in_use_p == NULL) {
2181     // No saved state so start at the beginning.
2182     // Lock the list head so we can possibly deflate it:
2183     if ((mid = get_list_head_locked(list_p)) == NULL) {
2184       return 0;  // The list is empty so nothing to deflate.
2185     }
2186     next = unmarked_next(mid);
2187   } else {
2188     // We're restarting after a safepoint so restore the necessary state
2189     // before we resume.
2190     cur_mid_in_use = *saved_mid_in_use_p;
2191     // Lock cur_mid_in_use so we can possibly update its
2192     // next field to extract a deflated ObjectMonitor.
2193     om_lock(cur_mid_in_use);
2194     mid = unmarked_next(cur_mid_in_use);
2195     if (mid == NULL) {
2196       om_unlock(cur_mid_in_use);
2197       *saved_mid_in_use_p = NULL;
2198       return 0;  // The remainder is empty so nothing more to deflate.
2199     }
2200     // Lock mid so we can possibly deflate it:
2201     om_lock(mid);
2202     next = unmarked_next(mid);
2203   }
2204 
2205   while (true) {
2206     // The current mid is locked at this point. If we have a
2207     // cur_mid_in_use, then it is also locked at this point.
2208 
2209     if (next != NULL) {
2210       // We lock next so that an om_flush() thread that is behind us
2211       // cannot pass us when we unlock the current mid.
2212       om_lock(next);
2213       next_next = unmarked_next(next);
2214     }
2215 
2216     // Only try to deflate if there is an associated Java object and if
2217     // mid is old (is not newly allocated and is not newly freed).
2218     if (mid->object() != NULL && mid->is_old() &&
2219         deflate_monitor_using_JT(mid, free_head_p, free_tail_p)) {
2220       // Deflation succeeded and already updated free_head_p and
2221       // free_tail_p as needed. Finish the move to the local free list
2222       // by unlinking mid from the global or per-thread in-use list.
2223       if (cur_mid_in_use == NULL) {
2224         // mid is the list head and it is locked. Switch the list head
2225         // to next which is also locked (if not NULL) and also leave
2226         // mid locked:
2227         Atomic::store(list_p, next);
2228       } else {
2229         ObjectMonitor* locked_next = mark_om_ptr(next);
2230         // mid and cur_mid_in_use are locked. Switch cur_mid_in_use's
2231         // next field to locked_next and also leave mid locked:
2232         cur_mid_in_use->set_next_om(locked_next);
2233       }
2234       // At this point mid is disconnected from the in-use list so
2235       // its lock longer has any effects on in-use list.
2236       deflated_count++;
2237       Atomic::dec(count_p);
2238       // mid is current tail in the free_head_p list so NULL terminate it
2239       // (which also unlocks it):
2240       mid->set_next_om(NULL);
2241 
2242       // All the list management is done so move on to the next one:
2243       mid = next;  // mid keeps non-NULL next's locked state
2244       next = next_next;
2245     } else {
2246       // mid is considered in-use if it does not have an associated
2247       // Java object or mid is not old or deflation did not succeed.
2248       // A mid->is_new() node can be seen here when it is freshly
2249       // returned by om_alloc() (and skips the deflation code path).
2250       // A mid->is_old() node can be seen here when deflation failed.
2251       // A mid->is_free() node can be seen here when a fresh node from
2252       // om_alloc() is released by om_release() due to losing the race
2253       // in inflate().
2254 
2255       // All the list management is done so move on to the next one:
2256       if (cur_mid_in_use != NULL) {
2257         om_unlock(cur_mid_in_use);
2258       }
2259       // The next cur_mid_in_use keeps mid's lock state so
2260       // that it is stable for a possible next field change. It
2261       // cannot be modified by om_release() while it is locked.
2262       cur_mid_in_use = mid;
2263       mid = next;  // mid keeps non-NULL next's locked state
2264       next = next_next;
2265 
2266       if (SafepointMechanism::should_block(self) &&
2267           cur_mid_in_use != Atomic::load(list_p) && cur_mid_in_use->is_old()) {
2268         // If a safepoint has started and cur_mid_in_use is not the list
2269         // head and is old, then it is safe to use as saved state. Return
2270         // to the caller before blocking.
2271         *saved_mid_in_use_p = cur_mid_in_use;
2272         om_unlock(cur_mid_in_use);
2273         if (mid != NULL) {
2274           om_unlock(mid);
2275         }
2276         return deflated_count;
2277       }
2278     }
2279     if (mid == NULL) {
2280       if (cur_mid_in_use != NULL) {
2281         om_unlock(cur_mid_in_use);
2282       }
2283       break;  // Reached end of the list so nothing more to deflate.
2284     }
2285 
2286     // The current mid's next field is locked at this point. If we have
2287     // a cur_mid_in_use, then it is also locked at this point.
2288   }
2289   // We finished the list without a safepoint starting so there's
2290   // no need to save state.
2291   *saved_mid_in_use_p = NULL;
2292   return deflated_count;
2293 }

2394   ObjectMonitor* free_tail_p = NULL;

2397

2399     timer.start();
2400   }

2403     OM_PERFDATA_OP(MonExtant, set_value(Atomic::load(&om_list_globals._in_use_count)));
2404   } else {

2408   do {
2409     int local_deflated_count;
2410     if (is_global) {

2428       // is trying to prepend an ObjectMonitor on its in-use list

2443     }

2447       timer.stop();

2451         } else {
2452           log_debug(monitorinflation)("jt=" INTPTR_FORMAT ": pausing deflation of per-thread idle monitors for a safepoint.", p2i(target));
2453         }
2454         assert(SafepointMechanism::should_block(self), "sanity check");
2455         ThreadBlockInVM blocker(self);
2456       }
2457       // Prepare for another loop after the safepoint.
2458       free_head_p = NULL;
2459       free_tail_p = NULL;
2460       if (log_is_enabled(Info, monitorinflation)) {
2461         timer.start();
2462       }
2463     }
2464   } while (saved_mid_in_use_p != NULL);
2465   timer.stop();
2466 
2467   LogStreamHandle(Debug, monitorinflation) lsh_debug;
2468   LogStreamHandle(Info, monitorinflation) lsh_info;
2469   LogStream* ls = NULL;
2470   if (log_is_enabled(Debug, monitorinflation)) {
2471     ls = &lsh_debug;
2472   } else if (deflated_count != 0 && log_is_enabled(Info, monitorinflation)) {
2473     ls = &lsh_info;
2474   }
2475   if (ls != NULL) {
2476     if (is_global) {
2477       ls->print_cr("async-deflating global idle monitors, %3.7f secs, %d monitors", timer.seconds(), deflated_count);
2478     } else {
2479       ls->print_cr("jt=" INTPTR_FORMAT ": async-deflating per-thread idle monitors, %3.7f secs, %d monitors", p2i(target), timer.seconds(), deflated_count);
2480     }
2481   }

2683 // Lock the next ObjectMonitor for traversal and unlock the current
2684 // ObjectMonitor. Returns the next ObjectMonitor if there is one.
2685 // Otherwise returns NULL (after unlocking the current ObjectMonitor).
2686 // This function is used by the various list walker functions to
2687 // safely walk a list without allowing an ObjectMonitor to be moved
2688 // to another list in the middle of a walk.
2689 static ObjectMonitor* lock_next_for_traversal(ObjectMonitor* cur) {
2690   assert(is_locked(cur), "cur=" INTPTR_FORMAT " must be locked", p2i(cur));
2691   ObjectMonitor* next = unmarked_next(cur);
2692   if (next == NULL) {  // Reached the end of the list.
2693     om_unlock(cur);
2694     return NULL;
2695   }
2696   om_lock(next);   // Lock next before unlocking current to keep

2793 // Check an in-use monitor entry; log any errors.
2794 void ObjectSynchronizer::chk_in_use_entry(JavaThread* jt, ObjectMonitor* n,
2795                                           outputStream * out, int *error_cnt_p) {
2796   if (n->header().value() == 0) {
2797     if (jt != NULL) {
2798       out->print_cr("ERROR: jt=" INTPTR_FORMAT ", monitor=" INTPTR_FORMAT
2799                     ": in-use per-thread monitor must have non-NULL _header "
2800                     "field.", p2i(jt), p2i(n));
2801     } else {
2802       out->print_cr("ERROR: monitor=" INTPTR_FORMAT ": in-use global monitor "
2803                     "must have non-NULL _header field.", p2i(n));
2804     }
2805     *error_cnt_p = *error_cnt_p + 1;
2806   }
2807   if (n->object() == NULL) {
2808     if (jt != NULL) {
2809       out->print_cr("ERROR: jt=" INTPTR_FORMAT ", monitor=" INTPTR_FORMAT
2810                     ": in-use per-thread monitor must have non-NULL _object "
2811                     "field.", p2i(jt), p2i(n));
2812     } else {
2813       out->print_cr("ERROR: monitor=" INTPTR_FORMAT ": in-use global monitor "
2814                     "must have non-NULL _object field.", p2i(n));
2815     }
2816     *error_cnt_p = *error_cnt_p + 1;
2817   }
2818   const oop obj = (oop)n->object();
2819   const markWord mark = obj->mark();
2820   if (!mark.has_monitor()) {
2821     if (jt != NULL) {
2822       out->print_cr("ERROR: jt=" INTPTR_FORMAT ", monitor=" INTPTR_FORMAT
2823                     ": in-use per-thread monitor's object does not think "
2824                     "it has a monitor: obj=" INTPTR_FORMAT ", mark="
2825                     INTPTR_FORMAT,  p2i(jt), p2i(n), p2i(obj), mark.value());
2826     } else {
2827       out->print_cr("ERROR: monitor=" INTPTR_FORMAT ": in-use global "
2828                     "monitor's object does not think it has a monitor: obj="
2829                     INTPTR_FORMAT ", mark=" INTPTR_FORMAT, p2i(n),
2830                     p2i(obj), mark.value());
2831     }
2832     *error_cnt_p = *error_cnt_p + 1;
2833   }
2834   ObjectMonitor* const obj_mon = mark.monitor();
2835   if (n != obj_mon) {
2836     if (jt != NULL) {
2837       out->print_cr("ERROR: jt=" INTPTR_FORMAT ", monitor=" INTPTR_FORMAT
2838                     ": in-use per-thread monitor's object does not refer "
2839                     "to the same monitor: obj=" INTPTR_FORMAT ", mark="
2840                     INTPTR_FORMAT ", obj_mon=" INTPTR_FORMAT, p2i(jt),
2841                     p2i(n), p2i(obj), mark.value(), p2i(obj_mon));
2842     } else {
2843       out->print_cr("ERROR: monitor=" INTPTR_FORMAT ": in-use global "
2844                     "monitor's object does not refer to the same monitor: obj="
2845                     INTPTR_FORMAT ", mark=" INTPTR_FORMAT ", obj_mon="
2846                     INTPTR_FORMAT, p2i(n), p2i(obj), mark.value(), p2i(obj_mon));
2847     }
2848     *error_cnt_p = *error_cnt_p + 1;
2849   }
2850 }
