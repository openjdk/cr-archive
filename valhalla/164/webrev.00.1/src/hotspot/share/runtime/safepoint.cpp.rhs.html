  73 static void post_safepoint_begin_event(EventSafepointBegin& event,
  74                                        uint64_t safepoint_id,
  75                                        int thread_count,
  76                                        int critical_thread_count) {
  77   if (event.should_commit()) {
  78     event.set_safepointId(safepoint_id);
  79     event.set_totalThreadCount(thread_count);
  80     event.set_jniCriticalThreadCount(critical_thread_count);
  81     event.commit();
  82   }
  83 }
  84 
  85 static void post_safepoint_cleanup_event(EventSafepointCleanup& event, uint64_t safepoint_id) {
  86   if (event.should_commit()) {
  87     event.set_safepointId(safepoint_id);
  88     event.commit();
  89   }
  90 }
  91 
  92 static void post_safepoint_synchronize_event(EventSafepointStateSynchronization& event,
  93                                              uint64_t safepoint_id,
  94                                              int initial_number_of_threads,
  95                                              int threads_waiting_to_block,
  96                                              uint64_t iterations) {
  97   if (event.should_commit()) {
  98     event.set_safepointId(safepoint_id);
  99     event.set_initialThreadCount(initial_number_of_threads);
 100     event.set_runningThreadCount(threads_waiting_to_block);
 101     event.set_iterations(iterations);
 102     event.commit();
 103   }
 104 }
 105 
 106 static void post_safepoint_cleanup_task_event(EventSafepointCleanupTask& event,
 107                                               uint64_t safepoint_id,
 108                                               const char* name) {
 109   if (event.should_commit()) {
 110     event.set_safepointId(safepoint_id);
 111     event.set_name(name);
 112     event.commit();
 113   }
 114 }
 115 
 116 static void post_safepoint_end_event(EventSafepointEnd& event, uint64_t safepoint_id) {
 117   if (event.should_commit()) {
 118     event.set_safepointId(safepoint_id);
 119     event.commit();
 120   }
 121 }

 168 bool SafepointSynchronize::thread_not_running(ThreadSafepointState *cur_state) {
 169   if (!cur_state->is_running()) {
 170     return true;
 171   }
 172   cur_state->examine_state_of_thread(SafepointSynchronize::safepoint_counter());
 173   if (!cur_state->is_running()) {
 174     return true;
 175   }
 176   LogTarget(Trace, safepoint) lt;
 177   if (lt.is_enabled()) {
 178     ResourceMark rm;
 179     LogStream ls(lt);
 180     cur_state->print_on(&ls);
 181   }
 182   return false;
 183 }

 185 #ifdef ASSERT
 186 static void assert_list_is_valid(const ThreadSafepointState* tss_head, int still_running) {
 187   int a = 0;
 188   const ThreadSafepointState *tmp_tss = tss_head;
 189   while (tmp_tss != NULL) {
 190     ++a;
 191     assert(tmp_tss->is_running(), "Illegal initial state");
 192     tmp_tss = tmp_tss->get_next();
 193   }
 194   assert(a == still_running, "Must be the same");
 195 }
 196 #endif // ASSERT

 198 static void back_off(int64_t start_time) {
 199   // We start with fine-grained nanosleeping until a millisecond has
 200   // passed, at which point we resort to plain naked_short_sleep.
 201   if (os::javaTimeNanos() - start_time < NANOSECS_PER_MILLISEC) {
 202     os::naked_short_nanosleep(10 * (NANOUNITS / MICROUNITS));
 203   } else {
 204     os::naked_short_sleep(1);
 205   }
 206 }

 208 int SafepointSynchronize::synchronize_threads(jlong safepoint_limit_time, int nof_threads, int* initial_running)
 209 {
 210   JavaThreadIteratorWithHandle jtiwh;
 211 
 212 #ifdef ASSERT
 213   for (; JavaThread *cur = jtiwh.next(); ) {
 214     assert(cur->safepoint_state()->is_running(), "Illegal initial state");
 215   }
 216   jtiwh.rewind();
 217 #endif // ASSERT
 218 
 219   // Iterate through all threads until it has been determined how to stop them all at a safepoint.
 220   int still_running = nof_threads;
 221   ThreadSafepointState *tss_head = NULL;
 222   ThreadSafepointState **p_prev = &tss_head;
 223   for (; JavaThread *cur = jtiwh.next(); ) {
 224     ThreadSafepointState *cur_tss = cur->safepoint_state();
 225     assert(cur_tss->get_next() == NULL, "Must be NULL");
 226     if (thread_not_running(cur_tss)) {
 227       --still_running;
 228     } else {
 229       *p_prev = cur_tss;
 230       p_prev = cur_tss->next_ptr();
 231     }
 232   }
 233   *p_prev = NULL;
 234 
 235   DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 236 
 237   *initial_running = still_running;
 238 
 239   // If there is no thread still running, we are already done.
 240   if (still_running <= 0) {
 241     assert(tss_head == NULL, "Must be empty");
 242     return 1;
 243   }
 244 
 245   int iterations = 1; // The first iteration is above.
 246   int64_t start_time = os::javaTimeNanos();
 247 
 248   do {
 249     // Check if this has taken too long:
 250     if (SafepointTimeout && safepoint_limit_time < os::javaTimeNanos()) {
 251       print_safepoint_timeout();
 252     }
 253 
 254     p_prev = &tss_head;
 255     ThreadSafepointState *cur_tss = tss_head;
 256     while (cur_tss != NULL) {
 257       assert(cur_tss->is_running(), "Illegal initial state");
 258       if (thread_not_running(cur_tss)) {
 259         --still_running;
 260         *p_prev = NULL;
 261         ThreadSafepointState *tmp = cur_tss;
 262         cur_tss = cur_tss->get_next();
 263         tmp->set_next(NULL);
 264       } else {
 265         *p_prev = cur_tss;
 266         p_prev = cur_tss->next_ptr();
 267         cur_tss = cur_tss->get_next();
 268       }
 269     }
 270 
 271     DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 272 
 273     if (still_running > 0) {
 274       back_off(start_time);
 275     }
 276 
 277     iterations++;
 278   } while (still_running > 0);
 279 
 280   assert(tss_head == NULL, "Must be empty");
 281 
 282   return iterations;
 283 }

 285 void SafepointSynchronize::arm_safepoint() {
 286   // Begin the process of bringing the system to a safepoint.
 287   // Java threads can be in several different states and are
 288   // stopped by different mechanisms:
 289   //
 290   //  1. Running interpreted
 291   //     When executing branching/returning byte codes interpreter
 292   //     checks if the poll is armed, if so blocks in SS::block().
 293   //  2. Running in native code
 294   //     When returning from the native code, a Java thread must check
 295   //     the safepoint _state to see if we must block.  If the
 296   //     VM thread sees a Java thread in native, it does
 297   //     not wait for this thread to block.  The order of the memory
 298   //     writes and reads of both the safepoint state and the Java
 299   //     threads state is critical.  In order to guarantee that the
 300   //     memory writes are serialized with respect to each other,
 301   //     the VM thread issues a memory barrier instruction.
 302   //  3. Running compiled Code
 303   //     Compiled code reads the local polling page that
 304   //     is set to fault if we are trying to get to a safepoint.
 305   //  4. Blocked
 306   //     A thread which is blocked will not be allowed to return from the
 307   //     block condition until the safepoint operation is complete.
 308   //  5. In VM or Transitioning between states
 309   //     If a Java thread is currently running in the VM or transitioning
 310   //     between states, the safepointing code will poll the thread state
 311   //     until the thread blocks itself when it attempts transitions to a
 312   //     new state or locking a safepoint checked monitor.
 313 
 314   // We must never miss a thread with correct safepoint id, so we must make sure we arm
 315   // the wait barrier for the next safepoint id/counter.
 316   // Arming must be done after resetting _current_jni_active_count, _waiting_to_block.
 317   _wait_barrier->arm(static_cast<int>(_safepoint_counter + 1));
 318 
 319   assert((_safepoint_counter & 0x1) == 0, "must be even");
 320   // The store to _safepoint_counter must happen after any stores in arming.
 321   Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 322 
 323   // We are synchronizing
 324   OrderAccess::storestore(); // Ordered with _safepoint_counter
 325   _state = _synchronizing;
 326 
 327   // Arming the per thread poll while having _state != _not_synchronized means safepointing
 328   log_trace(safepoint)("Setting thread local yield flag for threads");
 329   OrderAccess::storestore(); // storestore, global state -> local state
 330   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 331     // Make sure the threads start polling, it is time to yield.
 332     SafepointMechanism::arm_local_poll(cur);
 333   }
 334 
 335   OrderAccess::fence(); // storestore|storeload, global state -> local state
 336 }

 338 // Roll all threads forward to a safepoint and suspend them all
 339 void SafepointSynchronize::begin() {
 340   assert(Thread::current()->is_VM_thread(), "Only VM thread may execute a safepoint");
 341 
 342   EventSafepointBegin begin_event;
 343   SafepointTracing::begin(VMThread::vm_op_type());
 344 
 345   Universe::heap()->safepoint_synchronize_begin();
 346 
 347   // By getting the Threads_lock, we assure that no threads are about to start or
 348   // exit. It is released again in SafepointSynchronize::end().
 349   Threads_lock->lock();
 350 
 351   assert( _state == _not_synchronized, "trying to safepoint synchronize with wrong state");
 352 
 353   int nof_threads = Threads::number_of_threads();
 354 
 355   _nof_threads_hit_polling_page = 0;
 356 
 357   log_debug(safepoint)("Safepoint synchronization initiated using %s wait barrier. (%d threads)", _wait_barrier->description(), nof_threads);
 358 
 359   // Reset the count of active JNI critical threads
 360   _current_jni_active_count = 0;
 361 
 362   // Set number of threads to wait for
 363   _waiting_to_block = nof_threads;
 364 
 365   jlong safepoint_limit_time = 0;
 366   if (SafepointTimeout) {
 367     // Set the limit time, so that it can be compared to see if this has taken
 368     // too long to complete.
 369     safepoint_limit_time = SafepointTracing::start_of_safepoint() + (jlong)SafepointTimeoutDelay * (NANOUNITS / MILLIUNITS);
 370     timeout_error_printed = false;
 371   }
 372 
 373   EventSafepointStateSynchronization sync_event;
 374   int initial_running = 0;
 375 
 376   // Arms the safepoint, _current_jni_active_count and _waiting_to_block must be set before.
 377   arm_safepoint();
 378 
 379   // Will spin until all threads are safe.
 380   int iterations = synchronize_threads(safepoint_limit_time, nof_threads, &initial_running);
 381   assert(_waiting_to_block == 0, "No thread should be running");
 382 
 383 #ifndef PRODUCT
 384   if (safepoint_limit_time != 0) {
 385     jlong current_time = os::javaTimeNanos();
 386     if (safepoint_limit_time < current_time) {
 387       log_warning(safepoint)("# SafepointSynchronize: Finished after "
 388                     INT64_FORMAT_W(6) " ms",
 389                     (int64_t)(current_time - SafepointTracing::start_of_safepoint()) / (NANOUNITS / MILLIUNITS));
 390     }
 391   }
 392 #endif
 393 
 394   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 395 
 396   // Record state
 397   _state = _synchronized;
 398 
 399   OrderAccess::fence();
 400 
 401   // Set the new id
 402   ++_safepoint_id;
 403 
 404 #ifdef ASSERT
 405   // Make sure all the threads were visited.
 406   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 407     assert(cur->was_visited_for_critical_count(_safepoint_counter), "missed a thread");
 408   }
 409 #endif // ASSERT
 410 
 411   // Update the count of active JNI critical regions
 412   GCLocker::set_jni_lock_count(_current_jni_active_count);
 413 
 414   post_safepoint_synchronize_event(sync_event,
 415                                    _safepoint_id,
 416                                    initial_running,
 417                                    _waiting_to_block, iterations);
 418 
 419   SafepointTracing::synchronized(nof_threads, initial_running, _nof_threads_hit_polling_page);
 420 
 421   // We do the safepoint cleanup first since a GC related safepoint
 422   // needs cleanup to be completed before running the GC op.
 423   EventSafepointCleanup cleanup_event;
 424   do_cleanup_tasks();
 425   post_safepoint_cleanup_event(cleanup_event, _safepoint_id);
 426 
 427   post_safepoint_begin_event(begin_event, _safepoint_id, nof_threads, _current_jni_active_count);
 428   SafepointTracing::cleanup();
 429 }

 431 void SafepointSynchronize::disarm_safepoint() {
 432   uint64_t active_safepoint_counter = _safepoint_counter;
 433   {
 434     JavaThreadIteratorWithHandle jtiwh;
 435 #ifdef ASSERT
 436     // A pending_exception cannot be installed during a safepoint.  The threads
 437     // may install an async exception after they come back from a safepoint into
 438     // pending_exception after they unblock.  But that should happen later.
 439     for (; JavaThread *cur = jtiwh.next(); ) {
 440       assert (!(cur->has_pending_exception() &&
 441                 cur->safepoint_state()->is_at_poll_safepoint()),
 442               "safepoint installed a pending exception");
 443     }
 444 #endif // ASSERT
 445 
 446     OrderAccess::fence(); // keep read and write of _state from floating up
 447     assert(_state == _synchronized, "must be synchronized before ending safepoint synchronization");
 448 
 449     // Change state first to _not_synchronized.
 450     // No threads should see _synchronized when running.
 451     _state = _not_synchronized;
 452 
 453     // Set the next dormant (even) safepoint id.
 454     assert((_safepoint_counter & 0x1) == 1, "must be odd");
 455     Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 456 
 457     OrderAccess::fence(); // Keep the local state from floating up.
 458 
 459     jtiwh.rewind();
 460     for (; JavaThread *current = jtiwh.next(); ) {
 461       // Clear the visited flag to ensure that the critical counts are collected properly.
 462       DEBUG_ONLY(current->reset_visited_for_critical_count(active_safepoint_counter);)
 463       ThreadSafepointState* cur_state = current->safepoint_state();
 464       assert(!cur_state->is_running(), "Thread not suspended at safepoint");
 465       cur_state->restart(); // TSS _running
 466       assert(cur_state->is_running(), "safepoint state has not been reset");
 467     }
 468   } // ~JavaThreadIteratorWithHandle
 469 
 470   // Release threads lock, so threads can be created/destroyed again.
 471   Threads_lock->unlock();
 472 
 473   // Wake threads after local state is correctly set.
 474   _wait_barrier->disarm();
 475 }

 477 // Wake up all threads, so they are ready to resume execution after the safepoint
 478 // operation has been carried out
 479 void SafepointSynchronize::end() {
 480   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 481   EventSafepointEnd event;
 482   assert(Thread::current()->is_VM_thread(), "Only VM thread can execute a safepoint");
 483 
 484   disarm_safepoint();
 485 
 486   Universe::heap()->safepoint_synchronize_end();
 487 
 488   SafepointTracing::end();
 489 
 490   post_safepoint_end_event(event, safepoint_id());
 491 }

 493 bool SafepointSynchronize::is_cleanup_needed() {
 494   // Need a safepoint if some inline cache buffers is non-empty
 495   if (!InlineCacheBuffer::is_empty()) return true;
 496   if (StringTable::needs_rehashing()) return true;
 497   if (SymbolTable::needs_rehashing()) return true;
 498   return false;
 499 }

 501 class ParallelSPCleanupTask : public AbstractGangTask {
 502 private:
 503   SubTasksDone _subtasks;
 504   uint _num_workers;
 505 public:
 506   ParallelSPCleanupTask(uint num_workers) :
 507     AbstractGangTask("Parallel Safepoint Cleanup"),
 508     _subtasks(SafepointSynchronize::SAFEPOINT_CLEANUP_NUM_TASKS),
 509     _num_workers(num_workers) {}
 510 
 511   void work(uint worker_id) {
 512     uint64_t safepoint_id = SafepointSynchronize::safepoint_id();
 513 
 514     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_DEFLATE_MONITORS)) {
 515       const char* name = "deflating idle monitors";
 516       EventSafepointCleanupTask event;
 517       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 518       ObjectSynchronizer::do_safepoint_work();
 519 
 520       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 521     }
 522 
 523     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_UPDATE_INLINE_CACHES)) {
 524       const char* name = "updating inline caches";
 525       EventSafepointCleanupTask event;
 526       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 527       InlineCacheBuffer::update_inline_caches();
 528 
 529       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 530     }
 531 
 532     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_COMPILATION_POLICY)) {
 533       const char* name = "compilation policy safepoint handler";
 534       EventSafepointCleanupTask event;
 535       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 536       CompilationPolicy::policy()->do_safepoint_work();
 537 
 538       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 539     }
 540 
 541     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYMBOL_TABLE_REHASH)) {
 542       if (SymbolTable::needs_rehashing()) {
 543         const char* name = "rehashing symbol table";
 544         EventSafepointCleanupTask event;
 545         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 546         SymbolTable::rehash_table();
 547 
 548         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 549       }
 550     }
 551 
 552     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_STRING_TABLE_REHASH)) {
 553       if (StringTable::needs_rehashing()) {
 554         const char* name = "rehashing string table";
 555         EventSafepointCleanupTask event;
 556         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 557         StringTable::rehash_table();
 558 
 559         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 560       }
 561     }
 562 
 563     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYSTEM_DICTIONARY_RESIZE)) {
 564       if (Dictionary::does_any_dictionary_needs_resizing()) {
 565         const char* name = "resizing system dictionaries";
 566         EventSafepointCleanupTask event;
 567         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 568         ClassLoaderDataGraph::resize_dictionaries();
 569 
 570         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 571       }
 572     }
 573 
 574     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_REQUEST_OOPSTORAGE_CLEANUP)) {
 575       // Don't bother reporting event or time for this very short operation.
 576       // To have any utility we'd also want to report whether needed.
 577       OopStorage::trigger_cleanup_if_needed();
 578     }
 579 
 580     _subtasks.all_tasks_completed(_num_workers);
 581   }
 582 };

 584 // Various cleaning tasks that should be done periodically at safepoints.
 585 void SafepointSynchronize::do_cleanup_tasks() {
 586 
 587   TraceTime timer("safepoint cleanup tasks", TRACETIME_LOG(Info, safepoint, cleanup));
 588 
 589   CollectedHeap* heap = Universe::heap();
 590   assert(heap != NULL, "heap not initialized yet?");
 591   WorkGang* cleanup_workers = heap->get_safepoint_workers();
 592   if (cleanup_workers != NULL) {
 593     // Parallel cleanup using GC provided thread pool.
 594     uint num_cleanup_workers = cleanup_workers->active_workers();
 595     ParallelSPCleanupTask cleanup(num_cleanup_workers);
 596     cleanup_workers->run_task(&cleanup);
 597   } else {
 598     // Serial cleanup using VMThread.
 599     ParallelSPCleanupTask cleanup(1);
 600     cleanup.work(0);
 601   }
 602 
 603   // Needs to be done single threaded by the VMThread.  This walks
 604   // the thread stacks looking for references to metadata before
 605   // deciding to remove it from the metaspaces.
 606   if (ClassLoaderDataGraph::should_clean_metaspaces_and_reset()) {
 607     const char* name = "cleanup live ClassLoaderData metaspaces";
 608     TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 609     ClassLoaderDataGraph::walk_metadata_and_clean_metaspaces();
 610   }
 611 
 612   assert(InlineCacheBuffer::is_empty(), "should have cleaned up ICBuffer");
 613 }

 615 // Methods for determining if a JavaThread is safepoint safe.
 616 
 617 // False means unsafe with undetermined state.
 618 // True means a determined state, but it may be an unsafe state.
 619 // If called from a non-safepoint context safepoint_count MUST be InactiveSafepointCounter.
 620 bool SafepointSynchronize::try_stable_load_state(JavaThreadState *state, JavaThread *thread, uint64_t safepoint_count) {
 621   assert((safepoint_count != InactiveSafepointCounter &&
 622           Thread::current() == (Thread*)VMThread::vm_thread() &&
 623           SafepointSynchronize::_state != _not_synchronized)
 624          || safepoint_count == InactiveSafepointCounter, "Invalid check");
 625 
 626   // To handle the thread_blocked state on the backedge of the WaitBarrier from
 627   // previous safepoint and reading the reset value (0/InactiveSafepointCounter) we
 628   // re-read state after we read thread safepoint id. The JavaThread changes its
 629   // thread state from thread_blocked before resetting safepoint id to 0.
 630   // This guarantees the second read will be from an updated thread state. It can
 631   // either be different state making this an unsafe state or it can see blocked
 632   // again. When we see blocked twice with a 0 safepoint id, either:
 633   // - It is normally blocked, e.g. on Mutex, TBIVM.
 634   // - It was in SS:block(), looped around to SS:block() and is blocked on the WaitBarrier.
 635   // - It was in SS:block() but now on a Mutex.
 636   // All of these cases are safe.
 637 
 638   *state = thread->thread_state();
 639   OrderAccess::loadload();
 640   uint64_t sid = thread->safepoint_state()->get_safepoint_id();  // Load acquire
 641   if (sid != InactiveSafepointCounter && sid != safepoint_count) {
 642     // In an old safepoint, state not relevant.
 643     return false;
 644   }
 645   return *state == thread->thread_state();
 646 }
 647 
 648 static bool safepoint_safe_with(JavaThread *thread, JavaThreadState state) {
 649   switch(state) {
 650   case _thread_in_native:
 651     // native threads are safe if they have no java stack or have walkable stack
 652     return !thread->has_last_Java_frame() || thread->frame_anchor()->walkable();
 653 
 654   case _thread_blocked:
 655     // On wait_barrier or blocked.
 656     // Blocked threads should already have walkable stack.
 657     assert(!thread->has_last_Java_frame() || thread->frame_anchor()->walkable(), "blocked and not walkable");
 658     return true;
 659 
 660   default:
 661     return false;
 662   }
 663 }
 664 
 665 bool SafepointSynchronize::handshake_safe(JavaThread *thread) {
 666   if (thread->is_ext_suspended() || thread->is_terminated()) {
 667     return true;
 668   }
 669   JavaThreadState stable_state;
 670   if (try_stable_load_state(&stable_state, thread, InactiveSafepointCounter)) {
 671     return safepoint_safe_with(thread, stable_state);
 672   }
 673   return false;
 674 }

 676 // See if the thread is running inside a lazy critical native and
 677 // update the thread critical count if so.  Also set a suspend flag to
 678 // cause the native wrapper to return into the JVM to do the unlock
 679 // once the native finishes.
 680 static void check_for_lazy_critical_native(JavaThread *thread, JavaThreadState state) {
 681   if (state == _thread_in_native &&
 682       thread->has_last_Java_frame() &&
 683       thread->frame_anchor()->walkable()) {
 684     // This thread might be in a critical native nmethod so look at
 685     // the top of the stack and increment the critical count if it
 686     // is.
 687     frame wrapper_frame = thread->last_frame();
 688     CodeBlob* stub_cb = wrapper_frame.cb();
 689     if (stub_cb != NULL &&
 690         stub_cb->is_nmethod() &&
 691         stub_cb->as_nmethod_or_null()->is_lazy_critical_native()) {
 692       // A thread could potentially be in a critical native across
 693       // more than one safepoint, so only update the critical state on
 694       // the first one.  When it returns it will perform the unlock.
 695       if (!thread->do_critical_native_unlock()) {
 696 #ifdef ASSERT
 697         if (!thread->in_critical()) {
 698           GCLocker::increment_debug_jni_lock_count();
 699         }
 700 #endif
 701         thread->enter_critical();
 702         // Make sure the native wrapper calls back on return to
 703         // perform the needed critical unlock.
 704         thread->set_critical_native_unlock();
 705       }
 706     }
 707   }
 708 }

 710 // -------------------------------------------------------------------------------------------------------
 711 // Implementation of Safepoint blocking point
 712 
 713 void SafepointSynchronize::block(JavaThread *thread) {
 714   assert(thread != NULL, "thread must be set");
 715   assert(thread->is_Java_thread(), "not a Java thread");
 716 
 717   // Threads shouldn't block if they are in the middle of printing, but...
 718   ttyLocker::break_tty_lock_for_safepoint(os::current_thread_id());
 719 
 720   // Only bail from the block() call if the thread is gone from the
 721   // thread list; starting to exit should still block.
 722   if (thread->is_terminated()) {
 723      // block current thread if we come here from native code when VM is gone
 724      thread->block_if_vm_exited();
 725 
 726      // otherwise do nothing
 727      return;
 728   }
 729 
 730   JavaThreadState state = thread->thread_state();
 731   thread->frame_anchor()->make_walkable(thread);
 732 
 733   uint64_t safepoint_id = SafepointSynchronize::safepoint_counter();
 734   // Check that we have a valid thread_state at this point
 735   switch(state) {
 736     case _thread_in_vm_trans:
 737     case _thread_in_Java:        // From compiled code
 738     case _thread_in_native_trans:
 739     case _thread_blocked_trans:
 740     case _thread_new_trans:
 741 
 742       // We have no idea where the VMThread is, it might even be at next safepoint.
 743       // So we can miss this poll, but stop at next.
 744 
 745       // Load dependent store, it must not pass loading of safepoint_id.
 746       thread->safepoint_state()->set_safepoint_id(safepoint_id); // Release store
 747 
 748       // This part we can skip if we notice we miss or are in a future safepoint.
 749       OrderAccess::storestore();
 750       // Load in wait barrier should not float up
 751       thread->set_thread_state_fence(_thread_blocked);
 752 
 753       _wait_barrier->wait(static_cast<int>(safepoint_id));
 754       assert(_state != _synchronized, "Can't be");
 755 
 756       // If barrier is disarmed stop store from floating above loads in barrier.
 757       OrderAccess::loadstore();
 758       thread->set_thread_state(state);
 759 
 760       // Then we reset the safepoint id to inactive.
 761       thread->safepoint_state()->reset_safepoint_id(); // Release store
 762 
 763       OrderAccess::fence();
 764 
 765       break;
 766 
 767     default:
 768      fatal("Illegal threadstate encountered: %d", state);
 769   }
 770   guarantee(thread->safepoint_state()->get_safepoint_id() == InactiveSafepointCounter,
 771             "The safepoint id should be set only in block path");
 772 
 773   // Check for pending. async. exceptions or suspends - except if the
 774   // thread was blocked inside the VM. has_special_runtime_exit_condition()
 775   // is called last since it grabs a lock and we only want to do that when
 776   // we must.
 777   //
 778   // Note: we never deliver an async exception at a polling point as the
 779   // compiler may not have an exception handler for it. The polling
 780   // code will notice the async and deoptimize and the exception will
 781   // be delivered. (Polling at a return point is ok though). Sure is
 782   // a lot of bother for a deprecated feature...
 783   //
 784   // We don't deliver an async exception if the thread state is
 785   // _thread_in_native_trans so JNI functions won't be called with
 786   // a surprising pending exception. If the thread state is going back to java,
 787   // async exception is checked in check_special_condition_for_native_trans().
 788 
 789   if (state != _thread_blocked_trans &&
 790       state != _thread_in_vm_trans &&
 791       thread->has_special_runtime_exit_condition()) {
 792     thread->handle_special_runtime_exit_condition(
 793       !thread->is_at_poll_safepoint() && (state != _thread_in_native_trans));
 794   }
 795 
 796   // cross_modify_fence is done by SafepointMechanism::block_if_requested_slow
 797   // which is the only caller here.
 798 }

 893 void ThreadSafepointState::examine_state_of_thread(uint64_t safepoint_count) {
 894   assert(is_running(), "better be running or just have hit safepoint poll");
 895 
 896   JavaThreadState stable_state;
 897   if (!SafepointSynchronize::try_stable_load_state(&stable_state, _thread, safepoint_count)) {
 898     // We could not get stable state of the JavaThread.
 899     // Consider it running and just return.
 900     return;
 901   }
 902 
 903   // Check for a thread that is suspended. Note that thread resume tries
 904   // to grab the Threads_lock which we own here, so a thread cannot be
 905   // resumed during safepoint synchronization.
 906 
 907   // We check to see if this thread is suspended without locking to
 908   // avoid deadlocking with a third thread that is waiting for this
 909   // thread to be suspended. The third thread can notice the safepoint
 910   // that we're trying to start at the beginning of its SR_lock->wait()
 911   // call. If that happens, then the third thread will block on the
 912   // safepoint while still holding the underlying SR_lock. We won't be
 913   // able to get the SR_lock and we'll deadlock.
 914   //
 915   // We don't need to grab the SR_lock here for two reasons:
 916   // 1) The suspend flags are both volatile and are set with an
 917   //    Atomic::cmpxchg() call so we should see the suspended
 918   //    state right away.
 919   // 2) We're being called from the safepoint polling loop; if
 920   //    we don't see the suspended state on this iteration, then
 921   //    we'll come around again.
 922   //
 923   bool is_suspended = _thread->is_ext_suspended();
 924   if (is_suspended) {
 925     account_safe_thread();
 926     return;
 927   }
 928 
 929   if (safepoint_safe_with(_thread, stable_state)) {
 930     check_for_lazy_critical_native(_thread, stable_state);
 931     account_safe_thread();
 932     return;
 933   }
 934 
 935   // All other thread states will continue to run until they
 936   // transition and self-block in state _blocked
 937   // Safepoint polling in compiled code causes the Java threads to do the same.
 938   // Note: new threads may require a malloc so they must be allowed to finish
 939 
 940   assert(is_running(), "examine_state_of_thread on non-running thread");
 941   return;
 942 }

 944 void ThreadSafepointState::account_safe_thread() {
 945   SafepointSynchronize::decrement_waiting_to_block();
 946   if (_thread->in_critical()) {
 947     // Notice that this thread is in a critical section
 948     SafepointSynchronize::increment_jni_active_count();
 949   }
 950   DEBUG_ONLY(_thread->set_visited_for_critical_count(SafepointSynchronize::safepoint_counter());)
 951   assert(!_safepoint_safe, "Must be unsafe before safe");
 952   _safepoint_safe = true;
 953 }

 955 void ThreadSafepointState::restart() {
 956   assert(_safepoint_safe, "Must be safe before unsafe");
 957   _safepoint_safe = false;
 958 }

 972 // Block the thread at poll or poll return for safepoint/handshake.
 973 void ThreadSafepointState::handle_polling_page_exception() {
 974 
 975   // Step 1: Find the nmethod from the return address
 976   address real_return_addr = thread()->saved_exception_pc();
 977 
 978   CodeBlob *cb = CodeCache::find_blob(real_return_addr);
 979   assert(cb != NULL && cb->is_compiled(), "return address should be in nmethod");
 980   CompiledMethod* nm = (CompiledMethod*)cb;
 981 
 982   // Find frame of caller
 983   frame stub_fr = thread()->last_frame();
 984   CodeBlob* stub_cb = stub_fr.cb();
 985   assert(stub_cb->is_safepoint_stub(), "must be a safepoint stub");
 986   RegisterMap map(thread(), true);
 987   frame caller_fr = stub_fr.sender(&map);
 988 
 989   // Should only be poll_return or poll
 990   assert( nm->is_at_poll_or_poll_return(real_return_addr), "should not be at call" );
 991 
 992   // This is a poll immediately before a return. The exception handling code
 993   // has already had the effect of causing the return to occur, so the execution
 994   // will continue immediately after the call. In addition, the oopmap at the
 995   // return point does not mark the return value as an oop (if it is), so
 996   // it needs a handle here to be updated.
 997   if( nm->is_at_poll_return(real_return_addr) ) {
 998     ResourceMark rm;
 999     // See if return type is an oop.
1000     Method* method = nm->method();
1001     bool return_oop = method->is_returning_oop();
1002 
1003     GrowableArray<Handle> return_values;
1004     InlineKlass* vk = NULL;
1005 
1006     if (return_oop && InlineTypeReturnedAsFields) {
1007       SignatureStream ss(method->signature());
1008       while (!ss.at_return_type()) {
1009         ss.next();
1010       }
1011       if (ss.type() == T_INLINE_TYPE) {
1012         // Check if inline type is returned as fields
1013         vk = InlineKlass::returned_inline_klass(map);
1014         if (vk != NULL) {
1015           // We're at a safepoint at the return of a method that returns
1016           // multiple values. We must make sure we preserve the oop values
1017           // across the safepoint.
1018           assert(vk == method->returned_inline_type(thread()), "bad inline klass");
1019           vk->save_oop_fields(map, return_values);
1020           return_oop = false;
1021         }
1022       }
1023     }
1024 
1025     if (return_oop) {
1026       // The oop result has been saved on the stack together with all
1027       // the other registers. In order to preserve it over GCs we need
1028       // to keep it in a handle.
1029       oop result = caller_fr.saved_oop_result(&map);
1030       assert(oopDesc::is_oop_or_null(result), "must be oop");
1031       return_values.push(Handle(thread(), result));
1032       assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
1033     }
1034 
1035     // Block the thread
1036     SafepointMechanism::block_if_requested(thread());
1037 
1038     // restore oop result, if any
1039     if (return_oop) {
1040       assert(return_values.length() == 1, "only one return value");
1041       caller_fr.set_saved_oop_result(&map, return_values.pop()());
1042     } else if (vk != NULL) {
1043       vk->restore_oop_results(map, return_values);
1044     }
1045   }
1046 
1047   // This is a safepoint poll. Verify the return address and block.
1048   else {
1049     set_at_poll_safepoint(true);
1050 
1051     // verify the blob built the "return address" correctly
1052     assert(real_return_addr == caller_fr.pc(), "must match");
1053 
1054     // Block the thread
1055     SafepointMechanism::block_if_requested(thread());
1056     set_at_poll_safepoint(false);
1057 
1058     // If we have a pending async exception deoptimize the frame
1059     // as otherwise we may never deliver it.
1060     if (thread()->has_async_condition()) {
1061       ThreadInVMfromJavaNoAsyncException __tiv(thread());
1062       Deoptimization::deoptimize_frame(thread(), caller_fr.id());
1063     }
1064 
1065     // If an exception has been installed we must check for a pending deoptimization
1066     // Deoptimize frame if exception has been thrown.
1067 
1068     if (thread()->has_pending_exception() ) {
1069       RegisterMap map(thread(), true);
1070       frame caller_fr = stub_fr.sender(&map);
1071       if (caller_fr.is_deoptimized_frame()) {
1072         // The exception patch will destroy registers that are still
1073         // live and will be needed during deoptimization. Defer the
1074         // Async exception should have deferred the exception until the
1075         // next safepoint which will be detected when we get into
1076         // the interpreter so if we have an exception now things
1077         // are messed up.
1078 
1079         fatal("Exception installed and deoptimization is pending");
1080       }
1081     }
1082   }
1083 }

1170                               (int64_t)(_max_vmop_time));

1173 void SafepointTracing::begin(VM_Operation::VMOp_Type type) {
1174   _op_count[type]++;
1175   _current_type = type;
1176 
1177   // update the time stamp to begin recording safepoint time
1178   _last_safepoint_begin_time_ns = os::javaTimeNanos();
1179   _last_safepoint_sync_time_ns = 0;
1180   _last_safepoint_cleanup_time_ns = 0;

1182   _last_app_time_ns = _last_safepoint_begin_time_ns - _last_safepoint_end_time_ns;
1183   _last_safepoint_end_time_ns = 0;

1186 }

1200 void SafepointTracing::end() {
1201   _last_safepoint_end_time_ns = os::javaTimeNanos();
1202 
1203   if (_max_sync_time < (_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns)) {
1204     _max_sync_time = _last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns;
1205   }
1206   if (_max_vmop_time < (_last_safepoint_end_time_ns - _last_safepoint_sync_time_ns)) {
1207     _max_vmop_time = _last_safepoint_end_time_ns - _last_safepoint_sync_time_ns;
1208   }
1209   if (log_is_enabled(Info, safepoint, stats)) {
1210     statistics_log();
1211   }
1212 
1213   log_info(safepoint)(
1214      "Safepoint \"%s\", "
1215      "Time since last: " JLONG_FORMAT " ns, "
1216      "Reaching safepoint: " JLONG_FORMAT " ns, "
1217      "At safepoint: " JLONG_FORMAT " ns, "
1218      "Total: " JLONG_FORMAT " ns",
1219       VM_Operation::name(_current_type),
1220       _last_app_time_ns,
1221       _last_safepoint_cleanup_time_ns - _last_safepoint_begin_time_ns,
1222       _last_safepoint_end_time_ns     - _last_safepoint_cleanup_time_ns,
1223       _last_safepoint_end_time_ns     - _last_safepoint_begin_time_ns
1224      );
1225 
1226   RuntimeService::record_safepoint_end(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns);
1227 }
