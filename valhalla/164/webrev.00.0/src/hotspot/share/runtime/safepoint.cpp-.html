  73 static void post_safepoint_begin_event(EventSafepointBegin& event,
  74                                        uint64_t safepoint_id,
  75                                        int thread_count,
  76                                        int critical_thread_count) {
  77   if (event.should_commit()) {
  78     event.set_safepointId(safepoint_id);
  79     event.set_totalThreadCount(thread_count);
  80     event.set_jniCriticalThreadCount(critical_thread_count);
  81     event.commit();
  82   }
  83 }
  84 
  85 static void post_safepoint_cleanup_event(EventSafepointCleanup& event, uint64_t safepoint_id) {
  86   if (event.should_commit()) {
  87     event.set_safepointId(safepoint_id);
  88     event.commit();
  89   }
  90 }
  91 
  92 static void post_safepoint_synchronize_event(EventSafepointStateSynchronization& event,
  93                                              uint64_t safepoint_id,
  94                                              int initial_number_of_threads,
  95                                              int threads_waiting_to_block,
  96                                              uint64_t iterations) {
  97   if (event.should_commit()) {
  98     event.set_safepointId(safepoint_id);
  99     event.set_initialThreadCount(initial_number_of_threads);
 100     event.set_runningThreadCount(threads_waiting_to_block);
 101     event.set_iterations(iterations);
 102     event.commit();
 103   }
 104 }

 106 static void post_safepoint_cleanup_task_event(EventSafepointCleanupTask& event,
 107                                               uint64_t safepoint_id,
 108                                               const char* name) {
 109   if (event.should_commit()) {
 110     event.set_safepointId(safepoint_id);
 111     event.set_name(name);
 112     event.commit();
 113   }
 114 }

 116 static void post_safepoint_end_event(EventSafepointEnd& event, uint64_t safepoint_id) {
 117   if (event.should_commit()) {
 118     event.set_safepointId(safepoint_id);
 119     event.commit();
 120   }
 121 }

 168 bool SafepointSynchronize::thread_not_running(ThreadSafepointState *cur_state) {
 169   if (!cur_state->is_running()) {
 170     return true;
 171   }
 172   cur_state->examine_state_of_thread(SafepointSynchronize::safepoint_counter());
 173   if (!cur_state->is_running()) {
 174     return true;
 175   }
 176   LogTarget(Trace, safepoint) lt;
 177   if (lt.is_enabled()) {
 178     ResourceMark rm;
 179     LogStream ls(lt);
 180     cur_state->print_on(&ls);
 181   }
 182   return false;
 183 }

 185 #ifdef ASSERT
 186 static void assert_list_is_valid(const ThreadSafepointState* tss_head, int still_running) {
 187   int a = 0;
 188   const ThreadSafepointState *tmp_tss = tss_head;
 189   while (tmp_tss != NULL) {
 190     ++a;
 191     assert(tmp_tss->is_running(), "Illegal initial state");
 192     tmp_tss = tmp_tss->get_next();
 193   }
 194   assert(a == still_running, "Must be the same");
 195 }
 196 #endif // ASSERT

 198 static void back_off(int64_t start_time) {
 199   // We start with fine-grained nanosleeping until a millisecond has
 200   // passed, at which point we resort to plain naked_short_sleep.
 201   if (os::javaTimeNanos() - start_time < NANOSECS_PER_MILLISEC) {
 202     os::naked_short_nanosleep(10 * (NANOUNITS / MICROUNITS));
 203   } else {
 204     os::naked_short_sleep(1);
 205   }
 206 }

 208 int SafepointSynchronize::synchronize_threads(jlong safepoint_limit_time, int nof_threads, int* initial_running)
 209 {
 210   JavaThreadIteratorWithHandle jtiwh;
 211 
 212 #ifdef ASSERT
 213   for (; JavaThread *cur = jtiwh.next(); ) {
 214     assert(cur->safepoint_state()->is_running(), "Illegal initial state");
 215   }
 216   jtiwh.rewind();
 217 #endif // ASSERT
 218 
 219   // Iterate through all threads until it has been determined how to stop them all at a safepoint.
 220   int still_running = nof_threads;
 221   ThreadSafepointState *tss_head = NULL;
 222   ThreadSafepointState **p_prev = &tss_head;
 223   for (; JavaThread *cur = jtiwh.next(); ) {
 224     ThreadSafepointState *cur_tss = cur->safepoint_state();
 225     assert(cur_tss->get_next() == NULL, "Must be NULL");
 226     if (thread_not_running(cur_tss)) {
 227       --still_running;
 228     } else {
 229       *p_prev = cur_tss;
 230       p_prev = cur_tss->next_ptr();
 231     }
 232   }
 233   *p_prev = NULL;
 234 
 235   DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 236 
 237   *initial_running = still_running;
 238 
 239   // If there is no thread still running, we are already done.
 240   if (still_running <= 0) {
 241     assert(tss_head == NULL, "Must be empty");
 242     return 1;
 243   }
 244 
 245   int iterations = 1; // The first iteration is above.
 246   int64_t start_time = os::javaTimeNanos();
 247 
 248   do {
 249     // Check if this has taken too long:
 250     if (SafepointTimeout && safepoint_limit_time < os::javaTimeNanos()) {
 251       print_safepoint_timeout();
 252     }
 253 
 254     p_prev = &tss_head;
 255     ThreadSafepointState *cur_tss = tss_head;
 256     while (cur_tss != NULL) {
 257       assert(cur_tss->is_running(), "Illegal initial state");
 258       if (thread_not_running(cur_tss)) {
 259         --still_running;
 260         *p_prev = NULL;
 261         ThreadSafepointState *tmp = cur_tss;
 262         cur_tss = cur_tss->get_next();
 263         tmp->set_next(NULL);
 264       } else {
 265         *p_prev = cur_tss;
 266         p_prev = cur_tss->next_ptr();
 267         cur_tss = cur_tss->get_next();
 268       }
 269     }
 270 
 271     DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 272 
 273     if (still_running > 0) {
 274       back_off(start_time);
 275     }
 276 
 277     iterations++;
 278   } while (still_running > 0);
 279 
 280   assert(tss_head == NULL, "Must be empty");
 281 
 282   return iterations;
 283 }

 285 void SafepointSynchronize::arm_safepoint() {
 286   // Begin the process of bringing the system to a safepoint.
 287   // Java threads can be in several different states and are
 288   // stopped by different mechanisms:
 289   //
 290   //  1. Running interpreted
 291   //     When executing branching/returning byte codes interpreter
 292   //     checks if the poll is armed, if so blocks in SS::block().
 293   //  2. Running in native code
 294   //     When returning from the native code, a Java thread must check
 295   //     the safepoint _state to see if we must block.  If the
 296   //     VM thread sees a Java thread in native, it does
 297   //     not wait for this thread to block.  The order of the memory
 298   //     writes and reads of both the safepoint state and the Java
 299   //     threads state is critical.  In order to guarantee that the
 300   //     memory writes are serialized with respect to each other,
 301   //     the VM thread issues a memory barrier instruction.
 302   //  3. Running compiled Code
 303   //     Compiled code reads the local polling page that
 304   //     is set to fault if we are trying to get to a safepoint.
 305   //  4. Blocked
 306   //     A thread which is blocked will not be allowed to return from the
 307   //     block condition until the safepoint operation is complete.
 308   //  5. In VM or Transitioning between states
 309   //     If a Java thread is currently running in the VM or transitioning
 310   //     between states, the safepointing code will poll the thread state
 311   //     until the thread blocks itself when it attempts transitions to a
 312   //     new state or locking a safepoint checked monitor.
 313 
 314   // We must never miss a thread with correct safepoint id, so we must make sure we arm
 315   // the wait barrier for the next safepoint id/counter.
 316   // Arming must be done after resetting _current_jni_active_count, _waiting_to_block.
 317   _wait_barrier->arm(static_cast<int>(_safepoint_counter + 1));
 318 
 319   assert((_safepoint_counter & 0x1) == 0, "must be even");
 320   // The store to _safepoint_counter must happen after any stores in arming.
 321   Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 322 
 323   // We are synchronizing
 324   OrderAccess::storestore(); // Ordered with _safepoint_counter
 325   _state = _synchronizing;
 326 
 327   // Arming the per thread poll while having _state != _not_synchronized means safepointing
 328   log_trace(safepoint)("Setting thread local yield flag for threads");
 329   OrderAccess::storestore(); // storestore, global state -> local state
 330   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 331     // Make sure the threads start polling, it is time to yield.
 332     SafepointMechanism::arm_local_poll(cur);
 333   }
 334 
 335   OrderAccess::fence(); // storestore|storeload, global state -> local state
 336 }

 338 // Roll all threads forward to a safepoint and suspend them all
 339 void SafepointSynchronize::begin() {
 340   assert(Thread::current()->is_VM_thread(), "Only VM thread may execute a safepoint");
 341 
 342   EventSafepointBegin begin_event;
 343   SafepointTracing::begin(VMThread::vm_op_type());
 344 
 345   Universe::heap()->safepoint_synchronize_begin();
 346 
 347   // By getting the Threads_lock, we assure that no threads are about to start or
 348   // exit. It is released again in SafepointSynchronize::end().
 349   Threads_lock->lock();
 350 
 351   assert( _state == _not_synchronized, "trying to safepoint synchronize with wrong state");
 352 
 353   int nof_threads = Threads::number_of_threads();
 354 
 355   _nof_threads_hit_polling_page = 0;
 356 
 357   log_debug(safepoint)("Safepoint synchronization initiated using %s wait barrier. (%d threads)", _wait_barrier->description(), nof_threads);
 358 
 359   // Reset the count of active JNI critical threads
 360   _current_jni_active_count = 0;
 361 
 362   // Set number of threads to wait for
 363   _waiting_to_block = nof_threads;
 364 
 365   jlong safepoint_limit_time = 0;
 366   if (SafepointTimeout) {
 367     // Set the limit time, so that it can be compared to see if this has taken
 368     // too long to complete.
 369     safepoint_limit_time = SafepointTracing::start_of_safepoint() + (jlong)SafepointTimeoutDelay * (NANOUNITS / MILLIUNITS);
 370     timeout_error_printed = false;
 371   }
 372 
 373   EventSafepointStateSynchronization sync_event;
 374   int initial_running = 0;
 375 
 376   // Arms the safepoint, _current_jni_active_count and _waiting_to_block must be set before.
 377   arm_safepoint();
 378 
 379   // Will spin until all threads are safe.
 380   int iterations = synchronize_threads(safepoint_limit_time, nof_threads, &initial_running);
 381   assert(_waiting_to_block == 0, "No thread should be running");
 382 
 383 #ifndef PRODUCT
 384   if (safepoint_limit_time != 0) {
 385     jlong current_time = os::javaTimeNanos();
 386     if (safepoint_limit_time < current_time) {
 387       log_warning(safepoint)("# SafepointSynchronize: Finished after "
 388                     INT64_FORMAT_W(6) " ms",
 389                     (int64_t)(current_time - SafepointTracing::start_of_safepoint()) / (NANOUNITS / MILLIUNITS));
 390     }
 391   }
 392 #endif
 393 
 394   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 395 
 396   // Record state
 397   _state = _synchronized;
 398 
 399   OrderAccess::fence();
 400 
 401   // Set the new id
 402   ++_safepoint_id;
 403 
 404 #ifdef ASSERT
 405   // Make sure all the threads were visited.
 406   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 407     assert(cur->was_visited_for_critical_count(_safepoint_counter), "missed a thread");
 408   }
 409 #endif // ASSERT
 410 
 411   // Update the count of active JNI critical regions
 412   GCLocker::set_jni_lock_count(_current_jni_active_count);
 413 
 414   post_safepoint_synchronize_event(sync_event,
 415                                    _safepoint_id,
 416                                    initial_running,
 417                                    _waiting_to_block, iterations);
 418 
 419   SafepointTracing::synchronized(nof_threads, initial_running, _nof_threads_hit_polling_page);
 420 
 421   // We do the safepoint cleanup first since a GC related safepoint
 422   // needs cleanup to be completed before running the GC op.
 423   EventSafepointCleanup cleanup_event;
 424   do_cleanup_tasks();
 425   post_safepoint_cleanup_event(cleanup_event, _safepoint_id);
 426 
 427   post_safepoint_begin_event(begin_event, _safepoint_id, nof_threads, _current_jni_active_count);
 428   SafepointTracing::cleanup();
 429 }

 431 void SafepointSynchronize::disarm_safepoint() {
 432   uint64_t active_safepoint_counter = _safepoint_counter;
 433   {
 434     JavaThreadIteratorWithHandle jtiwh;
 435 #ifdef ASSERT
 436     // A pending_exception cannot be installed during a safepoint.  The threads
 437     // may install an async exception after they come back from a safepoint into
 438     // pending_exception after they unblock.  But that should happen later.
 439     for (; JavaThread *cur = jtiwh.next(); ) {
 440       assert (!(cur->has_pending_exception() &&
 441                 cur->safepoint_state()->is_at_poll_safepoint()),
 442               "safepoint installed a pending exception");
 443     }
 444 #endif // ASSERT
 445 
 446     OrderAccess::fence(); // keep read and write of _state from floating up
 447     assert(_state == _synchronized, "must be synchronized before ending safepoint synchronization");
 448 
 449     // Change state first to _not_synchronized.
 450     // No threads should see _synchronized when running.
 451     _state = _not_synchronized;
 452 
 453     // Set the next dormant (even) safepoint id.
 454     assert((_safepoint_counter & 0x1) == 1, "must be odd");
 455     Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 456 
 457     OrderAccess::fence(); // Keep the local state from floating up.
 458 
 459     jtiwh.rewind();
 460     for (; JavaThread *current = jtiwh.next(); ) {
 461       // Clear the visited flag to ensure that the critical counts are collected properly.
 462       DEBUG_ONLY(current->reset_visited_for_critical_count(active_safepoint_counter);)
 463       ThreadSafepointState* cur_state = current->safepoint_state();
 464       assert(!cur_state->is_running(), "Thread not suspended at safepoint");
 465       cur_state->restart(); // TSS _running
 466       assert(cur_state->is_running(), "safepoint state has not been reset");
 467     }
 468   } // ~JavaThreadIteratorWithHandle
 469 
 470   // Release threads lock, so threads can be created/destroyed again.
 471   Threads_lock->unlock();
 472 
 473   // Wake threads after local state is correctly set.
 474   _wait_barrier->disarm();
 475 }

 477 // Wake up all threads, so they are ready to resume execution after the safepoint
 478 // operation has been carried out
 479 void SafepointSynchronize::end() {
 480   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 481   EventSafepointEnd event;
 482   assert(Thread::current()->is_VM_thread(), "Only VM thread can execute a safepoint");
 483 
 484   disarm_safepoint();
 485 
 486   Universe::heap()->safepoint_synchronize_end();
 487 
 488   SafepointTracing::end();
 489 
 490   post_safepoint_end_event(event, safepoint_id());
 491 }

 493 bool SafepointSynchronize::is_cleanup_needed() {
 494   // Need a cleanup safepoint if there are too many monitors in use
 495   // and the monitor deflation needs to be done at a safepoint.
 496   if (ObjectSynchronizer::is_safepoint_deflation_needed()) return true;
 497   // Need a safepoint if some inline cache buffers is non-empty
 498   if (!InlineCacheBuffer::is_empty()) return true;
 499   if (StringTable::needs_rehashing()) return true;
 500   if (SymbolTable::needs_rehashing()) return true;
 501   return false;
 502 }

 521 class ParallelSPCleanupTask : public AbstractGangTask {
 522 private:
 523   SubTasksDone _subtasks;
 524   ParallelSPCleanupThreadClosure _cleanup_threads_cl;
 525   uint _num_workers;
 526   DeflateMonitorCounters* _counters;
 527 public:
 528   ParallelSPCleanupTask(uint num_workers, DeflateMonitorCounters* counters) :
 529     AbstractGangTask("Parallel Safepoint Cleanup"),
 530     _subtasks(SafepointSynchronize::SAFEPOINT_CLEANUP_NUM_TASKS),
 531     _cleanup_threads_cl(ParallelSPCleanupThreadClosure(counters)),
 532     _num_workers(num_workers),
 533     _counters(counters) {}
 534 
 535   void work(uint worker_id) {
 536     uint64_t safepoint_id = SafepointSynchronize::safepoint_id();
 537     // All threads deflate monitors and mark nmethods (if necessary).
 538     Threads::possibly_parallel_threads_do(true, &_cleanup_threads_cl);
 539 
 540     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_DEFLATE_MONITORS)) {
 541       const char* name = "deflating global idle monitors";
 542       EventSafepointCleanupTask event;
 543       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 544       // AsyncDeflateIdleMonitors only uses DeflateMonitorCounters
 545       // when a special cleanup has been requested.
 546       // Note: This logging output will include global idle monitor
 547       // elapsed times, but not global idle monitor deflation count.
 548       ObjectSynchronizer::do_safepoint_work(_counters);
 549 
 550       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 551     }
 552 
 553     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_UPDATE_INLINE_CACHES)) {
 554       const char* name = "updating inline caches";
 555       EventSafepointCleanupTask event;
 556       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 557       InlineCacheBuffer::update_inline_caches();
 558 
 559       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 560     }
 561 
 562     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_COMPILATION_POLICY)) {
 563       const char* name = "compilation policy safepoint handler";
 564       EventSafepointCleanupTask event;
 565       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 566       CompilationPolicy::policy()->do_safepoint_work();
 567 
 568       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 569     }
 570 
 571     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYMBOL_TABLE_REHASH)) {
 572       if (SymbolTable::needs_rehashing()) {
 573         const char* name = "rehashing symbol table";
 574         EventSafepointCleanupTask event;
 575         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 576         SymbolTable::rehash_table();
 577 
 578         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 579       }
 580     }
 581 
 582     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_STRING_TABLE_REHASH)) {
 583       if (StringTable::needs_rehashing()) {
 584         const char* name = "rehashing string table";
 585         EventSafepointCleanupTask event;
 586         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 587         StringTable::rehash_table();
 588 
 589         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 590       }
 591     }
 592 
 593     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYSTEM_DICTIONARY_RESIZE)) {
 594       if (Dictionary::does_any_dictionary_needs_resizing()) {
 595         const char* name = "resizing system dictionaries";
 596         EventSafepointCleanupTask event;
 597         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 598         ClassLoaderDataGraph::resize_dictionaries();
 599 
 600         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 601       }
 602     }
 603 
 604     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_REQUEST_OOPSTORAGE_CLEANUP)) {
 605       // Don't bother reporting event or time for this very short operation.
 606       // To have any utility we'd also want to report whether needed.
 607       OopStorage::trigger_cleanup_if_needed();
 608     }
 609 
 610     _subtasks.all_tasks_completed(_num_workers);
 611   }
 612 };

 614 // Various cleaning tasks that should be done periodically at safepoints.
 615 void SafepointSynchronize::do_cleanup_tasks() {
 616 
 617   TraceTime timer("safepoint cleanup tasks", TRACETIME_LOG(Info, safepoint, cleanup));
 618 
 619   // Prepare for monitor deflation.
 620   DeflateMonitorCounters deflate_counters;
 621   ObjectSynchronizer::prepare_deflate_idle_monitors(&deflate_counters);
 622 
 623   CollectedHeap* heap = Universe::heap();
 624   assert(heap != NULL, "heap not initialized yet?");
 625   WorkGang* cleanup_workers = heap->get_safepoint_workers();
 626   if (cleanup_workers != NULL) {
 627     // Parallel cleanup using GC provided thread pool.
 628     uint num_cleanup_workers = cleanup_workers->active_workers();
 629     ParallelSPCleanupTask cleanup(num_cleanup_workers, &deflate_counters);
 630     StrongRootsScope srs(num_cleanup_workers);
 631     cleanup_workers->run_task(&cleanup);
 632   } else {
 633     // Serial cleanup using VMThread.
 634     ParallelSPCleanupTask cleanup(1, &deflate_counters);
 635     StrongRootsScope srs(1);
 636     cleanup.work(0);
 637   }
 638 
 639   // Needs to be done single threaded by the VMThread.  This walks
 640   // the thread stacks looking for references to metadata before
 641   // deciding to remove it from the metaspaces.
 642   if (ClassLoaderDataGraph::should_clean_metaspaces_and_reset()) {
 643     const char* name = "cleanup live ClassLoaderData metaspaces";
 644     TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 645     ClassLoaderDataGraph::walk_metadata_and_clean_metaspaces();
 646   }
 647 
 648   // Finish monitor deflation.
 649   ObjectSynchronizer::finish_deflate_idle_monitors(&deflate_counters);
 650 
 651   assert(InlineCacheBuffer::is_empty(), "should have cleaned up ICBuffer");
 652 }

 654 // Methods for determining if a JavaThread is safepoint safe.
 655 
 656 // False means unsafe with undetermined state.
 657 // True means a determined state, but it may be an unsafe state.
 658 // If called from a non-safepoint context safepoint_count MUST be InactiveSafepointCounter.
 659 bool SafepointSynchronize::try_stable_load_state(JavaThreadState *state, JavaThread *thread, uint64_t safepoint_count) {
 660   assert((safepoint_count != InactiveSafepointCounter &&
 661           Thread::current() == (Thread*)VMThread::vm_thread() &&
 662           SafepointSynchronize::_state != _not_synchronized)
 663          || safepoint_count == InactiveSafepointCounter, "Invalid check");
 664 
 665   // To handle the thread_blocked state on the backedge of the WaitBarrier from
 666   // previous safepoint and reading the reset value (0/InactiveSafepointCounter) we
 667   // re-read state after we read thread safepoint id. The JavaThread changes its
 668   // thread state from thread_blocked before resetting safepoint id to 0.
 669   // This guarantees the second read will be from an updated thread state. It can
 670   // either be different state making this an unsafe state or it can see blocked
 671   // again. When we see blocked twice with a 0 safepoint id, either:
 672   // - It is normally blocked, e.g. on Mutex, TBIVM.
 673   // - It was in SS:block(), looped around to SS:block() and is blocked on the WaitBarrier.
 674   // - It was in SS:block() but now on a Mutex.
 675   // All of these cases are safe.
 676 
 677   *state = thread->thread_state();
 678   OrderAccess::loadload();
 679   uint64_t sid = thread->safepoint_state()->get_safepoint_id();  // Load acquire
 680   if (sid != InactiveSafepointCounter && sid != safepoint_count) {
 681     // In an old safepoint, state not relevant.
 682     return false;
 683   }
 684   return *state == thread->thread_state();
 685 }
 686 
 687 static bool safepoint_safe_with(JavaThread *thread, JavaThreadState state) {
 688   switch(state) {
 689   case _thread_in_native:
 690     // native threads are safe if they have no java stack or have walkable stack
 691     return !thread->has_last_Java_frame() || thread->frame_anchor()->walkable();
 692 
 693   case _thread_blocked:
 694     // On wait_barrier or blocked.
 695     // Blocked threads should already have walkable stack.
 696     assert(!thread->has_last_Java_frame() || thread->frame_anchor()->walkable(), "blocked and not walkable");
 697     return true;
 698 
 699   default:
 700     return false;
 701   }
 702 }
 703 
 704 bool SafepointSynchronize::handshake_safe(JavaThread *thread) {
 705   if (thread->is_ext_suspended() || thread->is_terminated()) {
 706     return true;
 707   }
 708   JavaThreadState stable_state;
 709   if (try_stable_load_state(&stable_state, thread, InactiveSafepointCounter)) {
 710     return safepoint_safe_with(thread, stable_state);
 711   }
 712   return false;
 713 }
 714 
 715 // See if the thread is running inside a lazy critical native and
 716 // update the thread critical count if so.  Also set a suspend flag to
 717 // cause the native wrapper to return into the JVM to do the unlock
 718 // once the native finishes.
 719 static void check_for_lazy_critical_native(JavaThread *thread, JavaThreadState state) {
 720   if (state == _thread_in_native &&
 721       thread->has_last_Java_frame() &&
 722       thread->frame_anchor()->walkable()) {
 723     // This thread might be in a critical native nmethod so look at
 724     // the top of the stack and increment the critical count if it
 725     // is.
 726     frame wrapper_frame = thread->last_frame();
 727     CodeBlob* stub_cb = wrapper_frame.cb();
 728     if (stub_cb != NULL &&
 729         stub_cb->is_nmethod() &&
 730         stub_cb->as_nmethod_or_null()->is_lazy_critical_native()) {
 731       // A thread could potentially be in a critical native across
 732       // more than one safepoint, so only update the critical state on
 733       // the first one.  When it returns it will perform the unlock.
 734       if (!thread->do_critical_native_unlock()) {
 735 #ifdef ASSERT
 736         if (!thread->in_critical()) {
 737           GCLocker::increment_debug_jni_lock_count();
 738         }
 739 #endif
 740         thread->enter_critical();
 741         // Make sure the native wrapper calls back on return to
 742         // perform the needed critical unlock.
 743         thread->set_critical_native_unlock();
 744       }
 745     }
 746   }
 747 }

 749 // -------------------------------------------------------------------------------------------------------
 750 // Implementation of Safepoint blocking point
 751 
 752 void SafepointSynchronize::block(JavaThread *thread) {
 753   assert(thread != NULL, "thread must be set");
 754   assert(thread->is_Java_thread(), "not a Java thread");
 755 
 756   // Threads shouldn't block if they are in the middle of printing, but...
 757   ttyLocker::break_tty_lock_for_safepoint(os::current_thread_id());
 758 
 759   // Only bail from the block() call if the thread is gone from the
 760   // thread list; starting to exit should still block.
 761   if (thread->is_terminated()) {
 762      // block current thread if we come here from native code when VM is gone
 763      thread->block_if_vm_exited();
 764 
 765      // otherwise do nothing
 766      return;
 767   }
 768 
 769   JavaThreadState state = thread->thread_state();
 770   thread->frame_anchor()->make_walkable(thread);
 771 
 772   uint64_t safepoint_id = SafepointSynchronize::safepoint_counter();
 773   // Check that we have a valid thread_state at this point
 774   switch(state) {
 775     case _thread_in_vm_trans:
 776     case _thread_in_Java:        // From compiled code
 777     case _thread_in_native_trans:
 778     case _thread_blocked_trans:
 779     case _thread_new_trans:
 780 
 781       // We have no idea where the VMThread is, it might even be at next safepoint.
 782       // So we can miss this poll, but stop at next.
 783 
 784       // Load dependent store, it must not pass loading of safepoint_id.
 785       thread->safepoint_state()->set_safepoint_id(safepoint_id); // Release store
 786 
 787       // This part we can skip if we notice we miss or are in a future safepoint.
 788       OrderAccess::storestore();
 789       // Load in wait barrier should not float up
 790       thread->set_thread_state_fence(_thread_blocked);
 791 
 792       _wait_barrier->wait(static_cast<int>(safepoint_id));
 793       assert(_state != _synchronized, "Can't be");
 794 
 795       // If barrier is disarmed stop store from floating above loads in barrier.
 796       OrderAccess::loadstore();
 797       thread->set_thread_state(state);
 798 
 799       // Then we reset the safepoint id to inactive.
 800       thread->safepoint_state()->reset_safepoint_id(); // Release store
 801 
 802       OrderAccess::fence();
 803 
 804       break;
 805 
 806     default:
 807      fatal("Illegal threadstate encountered: %d", state);
 808   }
 809   guarantee(thread->safepoint_state()->get_safepoint_id() == InactiveSafepointCounter,
 810             "The safepoint id should be set only in block path");
 811 
 812   // Check for pending. async. exceptions or suspends - except if the
 813   // thread was blocked inside the VM. has_special_runtime_exit_condition()
 814   // is called last since it grabs a lock and we only want to do that when
 815   // we must.
 816   //
 817   // Note: we never deliver an async exception at a polling point as the
 818   // compiler may not have an exception handler for it. The polling
 819   // code will notice the async and deoptimize and the exception will
 820   // be delivered. (Polling at a return point is ok though). Sure is
 821   // a lot of bother for a deprecated feature...
 822   //
 823   // We don't deliver an async exception if the thread state is
 824   // _thread_in_native_trans so JNI functions won't be called with
 825   // a surprising pending exception. If the thread state is going back to java,
 826   // async exception is checked in check_special_condition_for_native_trans().
 827 
 828   if (state != _thread_blocked_trans &&
 829       state != _thread_in_vm_trans &&
 830       thread->has_special_runtime_exit_condition()) {
 831     thread->handle_special_runtime_exit_condition(
 832       !thread->is_at_poll_safepoint() && (state != _thread_in_native_trans));
 833   }
 834 
 835   // cross_modify_fence is done by SafepointMechanism::block_if_requested_slow
 836   // which is the only caller here.
 837 }

 932 void ThreadSafepointState::examine_state_of_thread(uint64_t safepoint_count) {
 933   assert(is_running(), "better be running or just have hit safepoint poll");
 934 
 935   JavaThreadState stable_state;
 936   if (!SafepointSynchronize::try_stable_load_state(&stable_state, _thread, safepoint_count)) {
 937     // We could not get stable state of the JavaThread.
 938     // Consider it running and just return.
 939     return;
 940   }
 941 
 942   // Check for a thread that is suspended. Note that thread resume tries
 943   // to grab the Threads_lock which we own here, so a thread cannot be
 944   // resumed during safepoint synchronization.
 945 
 946   // We check to see if this thread is suspended without locking to
 947   // avoid deadlocking with a third thread that is waiting for this
 948   // thread to be suspended. The third thread can notice the safepoint
 949   // that we're trying to start at the beginning of its SR_lock->wait()
 950   // call. If that happens, then the third thread will block on the
 951   // safepoint while still holding the underlying SR_lock. We won't be
 952   // able to get the SR_lock and we'll deadlock.
 953   //
 954   // We don't need to grab the SR_lock here for two reasons:
 955   // 1) The suspend flags are both volatile and are set with an
 956   //    Atomic::cmpxchg() call so we should see the suspended
 957   //    state right away.
 958   // 2) We're being called from the safepoint polling loop; if
 959   //    we don't see the suspended state on this iteration, then
 960   //    we'll come around again.
 961   //
 962   bool is_suspended = _thread->is_ext_suspended();
 963   if (is_suspended) {
 964     account_safe_thread();
 965     return;
 966   }
 967 
 968   if (safepoint_safe_with(_thread, stable_state)) {
 969     check_for_lazy_critical_native(_thread, stable_state);
 970     account_safe_thread();
 971     return;
 972   }
 973 
 974   // All other thread states will continue to run until they
 975   // transition and self-block in state _blocked
 976   // Safepoint polling in compiled code causes the Java threads to do the same.
 977   // Note: new threads may require a malloc so they must be allowed to finish
 978 
 979   assert(is_running(), "examine_state_of_thread on non-running thread");
 980   return;
 981 }

 983 void ThreadSafepointState::account_safe_thread() {
 984   SafepointSynchronize::decrement_waiting_to_block();
 985   if (_thread->in_critical()) {
 986     // Notice that this thread is in a critical section
 987     SafepointSynchronize::increment_jni_active_count();
 988   }
 989   DEBUG_ONLY(_thread->set_visited_for_critical_count(SafepointSynchronize::safepoint_counter());)
 990   assert(!_safepoint_safe, "Must be unsafe before safe");
 991   _safepoint_safe = true;
 992 }

 994 void ThreadSafepointState::restart() {
 995   assert(_safepoint_safe, "Must be safe before unsafe");
 996   _safepoint_safe = false;
 997 }

1011 // Block the thread at poll or poll return for safepoint/handshake.
1012 void ThreadSafepointState::handle_polling_page_exception() {
1013 
1014   // Step 1: Find the nmethod from the return address
1015   address real_return_addr = thread()->saved_exception_pc();
1016 
1017   CodeBlob *cb = CodeCache::find_blob(real_return_addr);
1018   assert(cb != NULL && cb->is_compiled(), "return address should be in nmethod");
1019   CompiledMethod* nm = (CompiledMethod*)cb;
1020 
1021   // Find frame of caller
1022   frame stub_fr = thread()->last_frame();
1023   CodeBlob* stub_cb = stub_fr.cb();
1024   assert(stub_cb->is_safepoint_stub(), "must be a safepoint stub");
1025   RegisterMap map(thread(), true);
1026   frame caller_fr = stub_fr.sender(&map);
1027 
1028   // Should only be poll_return or poll
1029   assert( nm->is_at_poll_or_poll_return(real_return_addr), "should not be at call" );
1030 
1031   // This is a poll immediately before a return. The exception handling code
1032   // has already had the effect of causing the return to occur, so the execution
1033   // will continue immediately after the call. In addition, the oopmap at the
1034   // return point does not mark the return value as an oop (if it is), so
1035   // it needs a handle here to be updated.
1036   if( nm->is_at_poll_return(real_return_addr) ) {
1037     ResourceMark rm;
1038     // See if return type is an oop.
1039     Method* method = nm->method();
1040     bool return_oop = method->is_returning_oop();
1041 
1042     GrowableArray<Handle> return_values;
1043     InlineKlass* vk = NULL;
1044 
1045     if (return_oop && InlineTypeReturnedAsFields) {
1046       SignatureStream ss(method->signature());
1047       while (!ss.at_return_type()) {
1048         ss.next();
1049       }
1050       if (ss.type() == T_INLINE_TYPE) {
1051         // Check if inline type is returned as fields
1052         vk = InlineKlass::returned_inline_klass(map);
1053         if (vk != NULL) {
1054           // We're at a safepoint at the return of a method that returns
1055           // multiple values. We must make sure we preserve the oop values
1056           // across the safepoint.
1057           assert(vk == method->returned_inline_type(thread()), "bad inline klass");
1058           vk->save_oop_fields(map, return_values);
1059           return_oop = false;
1060         }
1061       }
1062     }
1063 
1064     if (return_oop) {
1065       // The oop result has been saved on the stack together with all
1066       // the other registers. In order to preserve it over GCs we need
1067       // to keep it in a handle.
1068       oop result = caller_fr.saved_oop_result(&map);
1069       assert(oopDesc::is_oop_or_null(result), "must be oop");
1070       return_values.push(Handle(thread(), result));
1071       assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
1072     }
1073 
1074     // Block the thread
1075     SafepointMechanism::block_if_requested(thread());
1076 
1077     // restore oop result, if any
1078     if (return_oop) {
1079       assert(return_values.length() == 1, "only one return value");
1080       caller_fr.set_saved_oop_result(&map, return_values.pop()());
1081     } else if (vk != NULL) {
1082       vk->restore_oop_results(map, return_values);
1083     }
1084   }
1085 
1086   // This is a safepoint poll. Verify the return address and block.
1087   else {
1088     set_at_poll_safepoint(true);
1089 
1090     // verify the blob built the "return address" correctly
1091     assert(real_return_addr == caller_fr.pc(), "must match");
1092 
1093     // Block the thread
1094     SafepointMechanism::block_if_requested(thread());
1095     set_at_poll_safepoint(false);
1096 
1097     // If we have a pending async exception deoptimize the frame
1098     // as otherwise we may never deliver it.
1099     if (thread()->has_async_condition()) {
1100       ThreadInVMfromJavaNoAsyncException __tiv(thread());
1101       Deoptimization::deoptimize_frame(thread(), caller_fr.id());
1102     }
1103 
1104     // If an exception has been installed we must check for a pending deoptimization
1105     // Deoptimize frame if exception has been thrown.
1106 
1107     if (thread()->has_pending_exception() ) {
1108       RegisterMap map(thread(), true);
1109       frame caller_fr = stub_fr.sender(&map);
1110       if (caller_fr.is_deoptimized_frame()) {
1111         // The exception patch will destroy registers that are still
1112         // live and will be needed during deoptimization. Defer the
1113         // Async exception should have deferred the exception until the
1114         // next safepoint which will be detected when we get into
1115         // the interpreter so if we have an exception now things
1116         // are messed up.
1117 
1118         fatal("Exception installed and deoptimization is pending");
1119       }
1120     }
1121   }
1122 }

1168   if ((_cur_stat_index % 30) == 0) {
1169     print_header(&ls);
1170     _cur_stat_index = 1;  // wrap
1171   } else {
1172     _cur_stat_index++;
1173   }
1174 
1175   ls.print("%-28s [       "
1176            INT32_FORMAT_W(8) "        " INT32_FORMAT_W(8) " "
1177            "]",
1178            VM_Operation::name(_current_type),
1179            _nof_threads,
1180            _nof_running);
1181   ls.print("[       "
1182            INT64_FORMAT_W(10) " " INT64_FORMAT_W(10) " "
1183            INT64_FORMAT_W(10) " " INT64_FORMAT_W(10) " ]",
1184            (int64_t)(_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns),
1185            (int64_t)(_last_safepoint_cleanup_time_ns - _last_safepoint_sync_time_ns),
1186            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns),
1187            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_begin_time_ns));
1188 
1189   ls.print_cr(INT32_FORMAT_W(16), _page_trap);
1190 }

1192 // This method will be called when VM exits. This tries to summarize the sampling.
1193 // Current thread may already be deleted, so don't use ResourceMark.
1194 void SafepointTracing::statistics_exit_log() {
1195   if (!log_is_enabled(Info, safepoint, stats)) {
1196     return;
1197   }
1198   for (int index = 0; index < VM_Operation::VMOp_Terminating; index++) {
1199     if (_op_count[index] != 0) {
1200       log_info(safepoint, stats)("%-28s" UINT64_FORMAT_W(10), VM_Operation::name(index),
1201                _op_count[index]);
1202     }
1203   }
1204 
1205   log_info(safepoint, stats)("Maximum sync time  " INT64_FORMAT" ns",
1206                               (int64_t)(_max_sync_time));
1207   log_info(safepoint, stats)("Maximum vm operation time (except for Exit VM operation)  "
1208                               INT64_FORMAT " ns",
1209                               (int64_t)(_max_vmop_time));
1210 }

1212 void SafepointTracing::begin(VM_Operation::VMOp_Type type) {
1213   _op_count[type]++;
1214   _current_type = type;
1215 
1216   // update the time stamp to begin recording safepoint time
1217   _last_safepoint_begin_time_ns = os::javaTimeNanos();
1218   _last_safepoint_sync_time_ns = 0;
1219   _last_safepoint_cleanup_time_ns = 0;
1220 
1221   _last_app_time_ns = _last_safepoint_begin_time_ns - _last_safepoint_end_time_ns;
1222   _last_safepoint_end_time_ns = 0;
1223 
1224   RuntimeService::record_safepoint_begin(_last_app_time_ns);
1225 }

1227 void SafepointTracing::synchronized(int nof_threads, int nof_running, int traps) {
1228   _last_safepoint_sync_time_ns = os::javaTimeNanos();
1229   _nof_threads = nof_threads;
1230   _nof_running = nof_running;
1231   _page_trap   = traps;
1232   RuntimeService::record_safepoint_synchronized(_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns);
1233 }

1235 void SafepointTracing::cleanup() {
1236   _last_safepoint_cleanup_time_ns = os::javaTimeNanos();
1237 }
1238 
1239 void SafepointTracing::end() {
1240   _last_safepoint_end_time_ns = os::javaTimeNanos();
1241 
1242   if (_max_sync_time < (_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns)) {
1243     _max_sync_time = _last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns;
1244   }
1245   if (_max_vmop_time < (_last_safepoint_end_time_ns - _last_safepoint_sync_time_ns)) {
1246     _max_vmop_time = _last_safepoint_end_time_ns - _last_safepoint_sync_time_ns;
1247   }
1248   if (log_is_enabled(Info, safepoint, stats)) {
1249     statistics_log();
1250   }
1251 
1252   log_info(safepoint)(
1253      "Safepoint \"%s\", "
1254      "Time since last: " JLONG_FORMAT " ns, "
1255      "Reaching safepoint: " JLONG_FORMAT " ns, "
1256      "At safepoint: " JLONG_FORMAT " ns, "
1257      "Total: " JLONG_FORMAT " ns",
1258       VM_Operation::name(_current_type),
1259       _last_app_time_ns,
1260       _last_safepoint_cleanup_time_ns - _last_safepoint_begin_time_ns,
1261       _last_safepoint_end_time_ns     - _last_safepoint_cleanup_time_ns,
1262       _last_safepoint_end_time_ns     - _last_safepoint_begin_time_ns
1263      );
1264 
1265   RuntimeService::record_safepoint_end(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns);
1266 }
