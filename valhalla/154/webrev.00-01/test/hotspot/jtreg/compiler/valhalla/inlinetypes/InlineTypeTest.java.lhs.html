230     public static String[] concat(String prefix[], String... extra) {
231         ArrayList<String> list = new ArrayList<String>();
232         if (prefix != null) {
233             for (String s : prefix) {
234                 list.add(s);
235             }
236         }
237         if (extra != null) {
238             for (String s : extra) {
239                 list.add(s);
240             }
241         }
242 
243         return list.toArray(new String[list.size()]);
244     }

330     public static void main(String[] args) throws Throwable {
331         if (args.length != 1) {
332             throw new RuntimeException("Usage: @run main/othervm/timeout=120 -Xbootclasspath/a:." +
333                                        " -XX:+IgnoreUnrecognizedVMOptions -XX:+UnlockDiagnosticVMOptions" +
334                                        " -XX:+UnlockExperimentalVMOptions -XX:+WhiteBoxAPI" +
335                                        " compiler.valhalla.inlinetypes.InlineTypeTest <YourTestMainClass>");
336         }
337         String testMainClassName = args[0];
338         Class testMainClass = Class.forName(testMainClassName);
339         InlineTypeTest test = (InlineTypeTest)testMainClass.newInstance();
340         List<String> scenarios = null;
341         if (!SCENARIOS.isEmpty()) {
342            scenarios = Arrays.asList(SCENARIOS.split(","));
343         }
344         for (int i=0; i<test.getNumScenarios(); i++) {
345             String reason;
346             if ((reason = test.isScenarioIgnored(i)) != null) {
347                 System.out.println("Scenario #" + i + " is ignored: " + reason);
348             } else if (scenarios != null && !scenarios.contains(Integer.toString(i))) {
349                 System.out.println("Scenario #" + i + " is skipped due to -Dscenarios=" + SCENARIOS);
350             } else {
351                 System.out.println("Scenario #" + i + " -------- ");
352                 String[] cmds = InputArguments.getVmInputArgs();
353                 if (RUN_SCENARIOS_WITH_XCOMP) {
354                     cmds = concat(cmds, "-Xcomp");
355                 }
356                 cmds = concat(cmds, test.getVMParameters(i));
357                 cmds = concat(cmds, test.getExtraVMParameters(i));
358                 cmds = concat(cmds, testMainClassName);
359 
360                 OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
361                 String output = oa.getOutput();
362                 oa.shouldHaveExitValue(0);
363                 System.out.println(output);
364             }
365         }
366     }

368     // To exclude test cases, use -DExclude=<case1>,<case2>,...
369     // Each case can be just the method name, or can be <class>.<method>. The latter form is useful
370     // when you are running several tests at the same time.
371     //
372     // jtreg -DExclude=test12 TestArrays.java
373     // jtreg -DExclude=test34 TestLWorld.java
374     // -- or --
375     // jtreg -DExclude=TestArrays.test12,TestLWorld.test34 TestArrays.java TestLWorld.java
376     //
377     private List<String> buildExcludeList() {
378         List<String> exclude = null;
379         String classPrefix = getClass().getSimpleName() + ".";
380         if (!EXCLUDELIST.isEmpty()) {
381             exclude = new ArrayList(Arrays.asList(EXCLUDELIST.split(",")));
382             for (int i = exclude.size() - 1; i >= 0; i--) {
383                 String ex = exclude.get(i);
384                 if (ex.indexOf(".") > 0) {
385                     if (ex.startsWith(classPrefix)) {
386                         ex = ex.substring(classPrefix.length());
387                         exclude.set(i, ex);
388                     } else {
389                         exclude.remove(i);
390                     }
391                 }
392             }
393         }
394         return exclude;
395     }

397     protected InlineTypeTest() {
398         List<String> list = null;
399         if (!TESTLIST.isEmpty()) {
400            list = Arrays.asList(TESTLIST.split(","));
401         }
402         List<String> exclude = buildExcludeList();
403 
404         // Gather all test methods and put them in Hashtable
405         for (Method m : getClass().getDeclaredMethods()) {
406             Test[] annos = m.getAnnotationsByType(Test.class);
407             if (annos.length != 0 &&
408                 ((list == null || list.contains(m.getName())) && (exclude == null || !exclude.contains(m.getName())))) {
409                 tests.put(getClass().getSimpleName() + "::" + m.getName(), m);







410             }
411         }
412     }

414     protected void run(String[] args, Class<?>... classes) throws Throwable {
415         if (args.length == 0) {
416             // Spawn a new VM instance
417             execute_vm();
418         } else {
419             // Execute tests in the VM spawned by the above code.
420             Asserts.assertTrue(args.length == 1 && args[0].equals("run"), "must be");
421             run(classes);
422         }
423     }

425     private void execute_vm() throws Throwable {
426         Asserts.assertFalse(tests.isEmpty(), "no tests to execute");
427         String[] vmInputArgs = InputArguments.getVmInputArgs();
428         for (String arg : vmInputArgs) {
429             if (arg.startsWith("-XX:CompileThreshold")) {
430                 // Disable IR verification if non-default CompileThreshold is set
431                 VERIFY_IR = false;
432             }
433         }
434         // Each VM is launched with flags in this order, so the later ones can override the earlier one:
435         //     VERIFY_IR/VERIFY_VM flags specified below
436         //     vmInputArgs, which consists of:
437         //        @run options
438         //        getVMParameters()
439         //        getExtraVMParameters()
440         //     defaultFlags
441         //     compileCommandFlags
442         String cmds[] = null;
443         if (VERIFY_IR) {
444             // Add print flags for IR verification
445             cmds = concat(cmds, printFlags);
446             // Always trap for exception throwing to not confuse IR verification
447             cmds = concat(cmds, "-XX:-OmitStackTraceInFastThrow");
448         }
449         if (VERIFY_VM) {
450             cmds = concat(cmds, verifyFlags);
451         }
452         cmds = concat(cmds, vmInputArgs);
453         cmds = concat(cmds, defaultFlags);
454         if (COMPILE_COMMANDS) {
455           cmds = concat(cmds, compileCommandFlags);
456         }
457 
458         // Run tests in own process and verify output
459         cmds = concat(cmds, getClass().getName(), "run");
460         OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
461         // If ideal graph printing is enabled/supported, verify output
462         String output = oa.getOutput();
463         oa.shouldHaveExitValue(0);
464         if (VERIFY_IR) {
465             if (output.contains("PrintIdeal enabled")) {
466                 parseOutput(output);
467             } else {
468                 System.out.println(output);
469                 System.out.println("WARNING: IR verification failed! Running with -Xint, -Xcomp or release build?");
470             }
471         }
472     }

474     static final class TestAnnotation {
475         private final int flag;
476         private final BooleanSupplier predicate;
477 
478         private static final TestAnnotation testAnnotations[] = {
479             new TestAnnotation(InlineTypePassFieldsAsArgsOn, () -> InlineTypePassFieldsAsArgs),
480             new TestAnnotation(InlineTypePassFieldsAsArgsOff, () -> !InlineTypePassFieldsAsArgs),
481             new TestAnnotation(InlineTypeArrayFlattenOn, () -> InlineTypeArrayFlatten),
482             new TestAnnotation(InlineTypeArrayFlattenOff, () -> !InlineTypeArrayFlatten),
483             new TestAnnotation(InlineTypeReturnedAsFieldsOn, () -> InlineTypeReturnedAsFields),
484             new TestAnnotation(InlineTypeReturnedAsFieldsOff, () -> !InlineTypeReturnedAsFields),
485             new TestAnnotation(AlwaysIncrementalInlineOn, () -> AlwaysIncrementalInline),
486             new TestAnnotation(AlwaysIncrementalInlineOff, () -> !AlwaysIncrementalInline),
487             new TestAnnotation(G1GCOn, () -> G1GC),
488             new TestAnnotation(G1GCOff, () -> !G1GC),
489             new TestAnnotation(ZGCOn, () -> ZGC),
490             new TestAnnotation(ZGCOff, () -> !ZGC),
491             new TestAnnotation(ArrayLoadStoreProfileOn, () -> UseArrayLoadStoreProfile),
492             new TestAnnotation(ArrayLoadStoreProfileOff, () -> !UseArrayLoadStoreProfile),
493             new TestAnnotation(TypeProfileOn, () -> TypeProfileLevel == 222),
494             new TestAnnotation(TypeProfileOff, () -> TypeProfileLevel == 0),
495         };
496 
497         private TestAnnotation(int flag, BooleanSupplier predicate) {
498             this.flag = flag;
499             this.predicate = predicate;
500         }
501 
502         private boolean match(Test a) {
503             return (a.valid() & flag) != 0 && predicate.getAsBoolean();
504         }
505 
506         static boolean find(Test a) {
507             Stream<TestAnnotation> s = Arrays.stream(testAnnotations).filter(t -> t.match(a));
508             long c = s.count();
509             if (c > 1) {
510                 throw new RuntimeException("At most one Test annotation should match");
511             }
512             return c > 0;
513         }
514     }

516     private void parseOutput(String output) throws Exception {
517         Pattern comp_re = Pattern.compile("\\n\\s+\\d+\\s+\\d+\\s+(%| )(s| )(!| )b(n| )\\s+\\d?\\s+\\S+\\.(?<name>[^.]+::\\S+)\\s+(?<osr>@ \\d+\\s+)?[(]\\d+ bytes[)]");
518         Matcher m = comp_re.matcher(output);
519         Map<String,String> compilations = new LinkedHashMap<>();
520         int prev = 0;
521         String methodName = null;
522         while (m.find()) {
523             if (prev == 0) {
524                 // Print header
525                 System.out.print(output.substring(0, m.start()+1));
526             } else if (methodName != null) {
527                 compilations.put(methodName, output.substring(prev, m.start()+1));
528             }
529             if (m.group("osr") != null) {
530                 methodName = null;
531             } else {
532                 methodName = m.group("name");
533             }
534             prev = m.end();
535         }
536         if (prev == 0) {
537             // Print header
538             System.out.print(output);
539         } else if (methodName != null) {
540             compilations.put(methodName, output.substring(prev));
541         }
542         // Iterate over compilation output
543         for (String testName : compilations.keySet()) {
544             Method test = tests.get(testName);
545             if (test == null) {
546                 // Skip helper methods
547                 continue;
548             }
549             String graph = compilations.get(testName);
550             if (PRINT_GRAPH) {
551                 System.out.println("\nGraph for " + testName + "\n" + graph);
552             }
553             // Parse graph using regular expressions to determine if it contains forbidden nodes
554             Test[] annos = test.getAnnotationsByType(Test.class);
555             Test anno = Arrays.stream(annos).filter(TestAnnotation::find).findFirst().orElse(null);
556             if (anno == null) {
557                 Object[] res = Arrays.stream(annos).filter(a -> a.valid() == 0).toArray();
558                 if (res.length != 1) {
559                     throw new RuntimeException("Only one Test annotation should match");
560                 }
561                 anno = (Test)res[0];
562             }
563             String regexFail = anno.failOn();
564             if (!regexFail.isEmpty()) {
565                 Pattern pattern = Pattern.compile(regexFail.substring(0, regexFail.length()-1));
566                 Matcher matcher = pattern.matcher(graph);
567                 boolean found = matcher.find();
568                 Asserts.assertFalse(found, "Graph for '" + testName + "' contains forbidden node:\n" + (found ? matcher.group() : ""));
569             }
570             String[] regexMatch = anno.match();
571             int[] matchCount = anno.matchCount();
572             for (int i = 0; i < regexMatch.length; ++i) {
573                 Pattern pattern = Pattern.compile(regexMatch[i].substring(0, regexMatch[i].length()-1));
574                 Matcher matcher = pattern.matcher(graph);
575                 int count = 0;
576                 String nodes = "";
577                 while (matcher.find()) {
578                     count++;
579                     nodes += matcher.group() + "\n";
580                 }
581                 if (matchCount[i] < 0) {
582                     Asserts.assertLTE(Math.abs(matchCount[i]), count, "Graph for '" + testName + "' contains different number of match nodes (expected <= " + matchCount[i] + " but got " + count + "):\n" + nodes);
583                 } else {
584                     Asserts.assertEQ(matchCount[i], count, "Graph for '" + testName + "' contains different number of match nodes (expected " + matchCount[i] + " but got " + count + "):\n" + nodes);
585                 }
586             }
587             tests.remove(testName);
588             System.out.println(testName + " passed");
589         }
590         // Check if all tests were compiled
591         if (tests.size() != 0) {
592             for (String name : tests.keySet()) {
593                 System.out.println("Test '" + name + "' not compiled!");
594             }
595             throw new RuntimeException("Not all tests were compiled");
596         }
597     }

599     private void setup(Class<?> clazz) {
600         if (XCOMP) {
601             // Don't control compilation if -Xcomp is enabled
602             return;
603         }
604         if (DUMP_REPLAY) {
605             // Generate replay compilation files
606             String directive = "[{ match: \"*.*\", DumpReplay: true }]";
607             if (WHITE_BOX.addCompilerDirective(directive) != 1) {
608                 throw new RuntimeException("Failed to add compiler directive");
609             }
610         }
611 
612         Method[] methods = clazz.getDeclaredMethods();
613         for (Method m : methods) {
614             if (m.isAnnotationPresent(Test.class)) {
615                 // Don't inline tests
616                 WHITE_BOX.testSetDontInlineMethod(m, true);
617             }
618             if (m.isAnnotationPresent(DontCompile.class)) {
619                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, true);
620                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, false);
621                 WHITE_BOX.testSetDontInlineMethod(m, true);
622             }
623             if (m.isAnnotationPresent(ForceInline.class)) {
624                 Asserts.assertFalse(m.isAnnotationPresent(DontInline.class), "Method " + m.getName() + " has contradicting DontInline annotation");
625                 WHITE_BOX.testSetForceInlineMethod(m, true);
626             }
627             if (m.isAnnotationPresent(DontInline.class)) {
628                 Asserts.assertFalse(m.isAnnotationPresent(ForceInline.class), "Method " + m.getName() + " has contradicting ForceInline annotation");
629                 WHITE_BOX.testSetDontInlineMethod(m, true);
630             }
631             if (STRESS_CC) {
632                 // Exclude some methods from compilation with C2 to stress test the calling convention
633                 if (Utils.getRandomInstance().nextBoolean()) {
634                     System.out.println("Excluding from C2 compilation: " + m);
635                     WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_FULL_OPTIMIZATION, false);
636                 }
637             }
638         }
639         // Only force compilation now because above annotations affect inlining
640         for (Method m : methods) {
641             if (m.isAnnotationPresent(ForceCompile.class)) {
642                 Asserts.assertFalse(m.isAnnotationPresent(DontCompile.class), "Method " + m.getName() + " has contradicting DontCompile annotation");
643                 int compLevel = getCompLevel(m.getAnnotation(ForceCompile.class));
644                 enqueueMethodForCompilation(m, compLevel);
645             }
646         }
647         // Compile class initializers
648         int compLevel = getCompLevel(null);
649         WHITE_BOX.enqueueInitializerForCompilation(clazz, compLevel);
650     }

652     private void run(Class<?>... classes) throws Exception {
653         if (USE_COMPILER && PRINT_IDEAL && !XCOMP && !STRESS_CC) {
654             System.out.println("PrintIdeal enabled");
655         }
656         System.out.format("rI = %d, rL = %d\n", rI, rL);
657 
658         setup(getClass());
659         for (Class<?> clazz : classes) {
660             setup(clazz);
661         }
662 
663         // Execute tests in random order (execution sequence affects profiling)
664         TreeMap<Long, String> durations = (PRINT_TIMES || VERBOSE) ? new TreeMap<Long, String>() : null;
665         List<Method> testList = new ArrayList<Method>(tests.values());
666         Collections.shuffle(testList, Utils.getRandomInstance());



667         for (Method test : testList) {
668             if (VERBOSE) {
669                 System.out.println("Starting " + test.getName());
670             }
671             TempSkipForC1 c1skip = test.getAnnotation(TempSkipForC1.class);
672             if (TEST_C1 && c1skip != null) {
673                 System.out.println("Skipped " + test.getName() + " for C1 testing: " + c1skip.reason());
674                 continue;
675             }
676             long startTime = System.nanoTime();
677             Method verifier = getClass().getMethod(test.getName() + "_verifier", boolean.class);
678             // Warmup using verifier method
679             Warmup anno = test.getAnnotation(Warmup.class);
680             int warmup = anno == null ? WARMUP : anno.value();
681             for (int i = 0; i < warmup; ++i) {
682                 verifier.invoke(this, true);
683             }
684             boolean osrOnly = (test.getAnnotation(OSRCompileOnly.class) != null);
685             int compLevel = getCompLevel(test.getAnnotation(Test.class));
686 
687             // C1 generates a lot of code when VerifyOops is enabled and may run out of space (for a small
688             // number of test cases).
689             boolean maybeCodeBufferOverflow = (TEST_C1 && VerifyOops);
690 
691             if (osrOnly) {
692                 long started = System.currentTimeMillis();
693                 boolean stateCleared = false;
694                 for (;;) {
695                     long elapsed = System.currentTimeMillis() - started;
696                     int level = WHITE_BOX.getMethodCompilationLevel(test);
697                     if (maybeCodeBufferOverflow && elapsed > 5000 && (!WHITE_BOX.isMethodCompiled(test, false) || level != compLevel)) {
698                         System.out.println("Temporarily disabling VerifyOops");
699                         try {
700                             WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
701                             if (!stateCleared) {
702                                 WHITE_BOX.clearMethodState(test);
703                                 stateCleared = true;
704                             }
705                             verifier.invoke(this, false);
706                         } finally {
707                             WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
708                             System.out.println("Re-enabled VerifyOops");
709                         }
710                     } else {
711                         verifier.invoke(this, false);
712                     }
713 
714                     boolean b = WHITE_BOX.isMethodCompiled(test, false);
715                     if (VERBOSE) {
716                         System.out.println("Is " + test.getName() + " compiled? " + b);
717                     }
718                     if (b || XCOMP || STRESS_CC || !USE_COMPILER) {
719                         // Don't control compilation if -Xcomp is enabled, or if compiler is disabled
720                         break;
721                     }
722                     Asserts.assertTrue(OSR_TEST_TIMEOUT < 0 || elapsed < OSR_TEST_TIMEOUT, test + " not compiled after " + OSR_TEST_TIMEOUT + " ms");
723                 }
724             } else {
725                 // Trigger compilation
726                 enqueueMethodForCompilation(test, compLevel);
727                 if (maybeCodeBufferOverflow && !WHITE_BOX.isMethodCompiled(test, false)) {
728                     // Let's disable VerifyOops temporarily and retry.
729                     WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
730                     WHITE_BOX.clearMethodState(test);
731                     enqueueMethodForCompilation(test, compLevel);
732                     WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
733                 }
734                 if (!STRESS_CC && USE_COMPILER) {
735                     Asserts.assertTrue(WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");
736                     int level = WHITE_BOX.getMethodCompilationLevel(test);
737                     Asserts.assertEQ(level, compLevel, "Unexpected compilation level for " + test);
738                 }
739                 // Check result
740                 verifier.invoke(this, false);
741             }
742             if (PRINT_TIMES || VERBOSE) {
743                 long endTime = System.nanoTime();
744                 long duration = (endTime - startTime);
745                 durations.put(duration, test.getName());
746                 if (VERBOSE) {
747                     System.out.println("Done " + test.getName() + ": " + duration + " ns = " + (duration / 1000000) + " ms");
748                 }
749             }
750             if (GC_AFTER) {
751                 System.out.println("doing GC");
752                 System.gc();
753             }
754         }
755 
756         // Print execution times
757         if (PRINT_TIMES) {
758           System.out.println("\n\nTest execution times:");
759           for (Map.Entry<Long, String> entry : durations.entrySet()) {
760               System.out.format("%-10s%15d ns\n", entry.getValue() + ":", entry.getKey());
761           }
762         }
763     }

765     // Get the appropriate compilation level for a method, according to the
766     // given annotation, as well as the current test scenario and VM options.
767     //
768     private int getCompLevel(Object annotation) {
769         int compLevel;
770         if (annotation == null) {
771             compLevel = COMP_LEVEL_ANY;
772         } else if (annotation instanceof Test) {
773             compLevel = ((Test)annotation).compLevel();
774         } else {
775             compLevel = ((ForceCompile)annotation).compLevel();
776         }
777 
778         return restrictCompLevel(compLevel);
779     }
780 
781     // Get the appropriate level as permitted by the test scenario and VM options.
782     private static int restrictCompLevel(int compLevel) {
783         if (compLevel == COMP_LEVEL_ANY) {
784             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
785         }
786         if (FLIP_C1_C2) {
787             // Effectively treat all (compLevel = C1) as (compLevel = C2), and
788             //                       (compLevel = C2) as (compLevel = C1).
789             if (compLevel == COMP_LEVEL_SIMPLE) {
790                 compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
791             } else if (compLevel == COMP_LEVEL_FULL_OPTIMIZATION) {
792                 compLevel = COMP_LEVEL_SIMPLE;
793             }
794         }
795         if (!TEST_C1 && compLevel < COMP_LEVEL_FULL_OPTIMIZATION) {
796             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
797         }
798         if (TieredCompilation && compLevel > (int)TieredStopAtLevel) {
799             compLevel = (int)TieredStopAtLevel;
800         }
801         return compLevel;
802     }

804     public static void enqueueMethodForCompilation(Method m, int level) {
805         level = restrictCompLevel(level);
806         if (VERBOSE) {
807             System.out.println("enqueueMethodForCompilation " + m + ", level = " + level);
808         }
809         WHITE_BOX.enqueueMethodForCompilation(m, level);
810     }

812     // Unlike C2, C1 intrinsics never deoptimize System.arraycopy. Instead, we fall back to
813     // a normal method invocation when encountering flattened arrays.
814     static boolean isCompiledByC2(Method m) {
815         return USE_COMPILER && !XCOMP && WHITE_BOX.isMethodCompiled(m, false) &&
816             WHITE_BOX.getMethodCompilationLevel(m, false) >= COMP_LEVEL_FULL_OPTIMIZATION;
