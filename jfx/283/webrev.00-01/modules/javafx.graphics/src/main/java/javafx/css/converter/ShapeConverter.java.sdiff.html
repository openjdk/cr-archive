52     @Override public Shape convert(ParsedValue<String, Shape> value, Font font) {
53 
54         Shape shape = super.getCachedValue(value);
55         if (shape != null) return shape;
56 
57         String svg = value.getValue();

59         SVGPath path = new SVGPath();

63 
64         return path;
65     }

68
