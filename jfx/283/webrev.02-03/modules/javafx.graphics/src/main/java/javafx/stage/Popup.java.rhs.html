26 package javafx.stage;
27 
28 import javafx.collections.ObservableList;
29 import javafx.scene.Node;
30 
31 /**
32  * A Popup is a special window-like container for a scene graph. It is typically
33  * used for tooltip like notification, drop down boxes, menus, and so forth.
34  * The popup has no decorations, and essentially acts as a specialized
35  * scene/window which has no decorations, is transparent, and with a null fill.
36  *
37  * @since JavaFX 2.0
38  */
39 public class Popup extends PopupWindow {
40 
41     /**
42      * Creates a {@code Popup}.
43      */
44     public Popup() {
45     }
46 
47     /**
48      * The ObservableList of {@code Node}s to be rendered on this
49      * {@code Popup}. The content forms the complete visual representation of
50      * the Popup. Popups have no intrinsic visuals.
51      */
52     @Override public final ObservableList<Node> getContent() {
53         return super.getContent();
54     }
55 }
