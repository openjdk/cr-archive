 26 package javafx.scene.input;
 27 
 28 import java.io.File;
 29 import java.util.ArrayList;
 30 import java.util.HashMap;
 31 import java.util.List;
 32 
 33 import javafx.scene.image.Image;
 34 
 35 /**
 36  * Data container for {@link Clipboard} data. It can hold multiple data in
 37  * several data formats.
 38  * @since JavaFX 2.0
 39  */
 40 public class ClipboardContent extends HashMap<DataFormat, Object> {
 41 
 42     /**
 43      * Creates a {@code ClipboardContent}.
 44      */
 45     public ClipboardContent() {
 46     }
 47 
 48     /**
 49      * Gets whether a plain text String ({@code DataFormat.PLAIN_TEXT})
 50      * has been put to this {@code ClipboardContent}.
 51      * @return true if {@code containsKey(DataFormat.PLAIN_TEXT)}
 52      * returns true, false otherwise
 53      */
 54     public final boolean hasString() {
 55         return containsKey(DataFormat.PLAIN_TEXT);
 56     }
 57 
 58     /**
 59      * Puts a plain text String into the {@code ClipboardContent}. This is
 60      * equivalent to invoking {@code put(DataFormat.PLAIN_TEXT, s)}.
 61      * Setting this value to null effectively clears it
 62      * from the {@code ClipboardContent}.
 63      * @param s The string to place. This may be null.
 64      * @return always true (the string is always successfully put)
 65      */
 66     public final boolean putString(String s) {
 67         if (s == null) {
 68             remove(DataFormat.PLAIN_TEXT);
 69         } else {
 70             put(DataFormat.PLAIN_TEXT, s);
 71         }
 72         return true;
 73     }
 74 
 75     /**
 76      * Gets the plain text String from the {@code ClipboardContent}
 77      * which had previously been put. This is equivalent to invoking
 78      * {@code get(DataFormat.PLAIN_TEXT)}. If no such entry exists,
 79      * null is returned.
 80      * @return The String in the {@code ClipboardContent} associated
 81      * with {@code DataFormat.PLAIN_TEXT}, or null if there is not one.
 82      */
 83     public final String getString() {
 84         return (String) get(DataFormat.PLAIN_TEXT);
 85     }
 86 
 87     /**
 88      * Gets whether a URL String ({@code DataFormat.URL})
 89      * has been put to this {@code ClipboardContent}.
 90      * @return true if {@code containsKey(DataFormat.URL)}
 91      * returns true, false otherwise
 92      */
 93     public final boolean hasUrl() {
 94         return containsKey(DataFormat.URL);
 95     }
 96 
 97     /**
 98      * Puts a URL String into the {@code ClipboardContent}. This is
 99      * equivalent to invoking {@code put(DataFormat.URL, url)}.
100      * Setting this value to null effectively clears it
101      * from the {@code ClipboardContent}.
102      * @param url The string to place. This may be null.
103      * @return always true (the URL is always successfully put)
104      */
105     public final boolean putUrl(String url) {
106         if (url == null) {
107             remove(DataFormat.URL);
108         } else {
109             put(DataFormat.URL, url);
110         }
111         return true;
112     }
113 
114     /**
115      * Gets the URL String from the {@code ClipboardContent}
116      * which had previously been put. This is equivalent to invoking
117      * {@code get(DataFormat.URL)}. If no such entry exists,
118      * null is returned.
119      * @return The String in the {@code ClipboardContent} associated
120      * with {@code DataFormat.URL}, or null if there is not one.
121      */
122     public final String getUrl() {
123         return (String) get(DataFormat.URL);
124     }
125 
126     /**
127      * Gets whether an HTML String ({@code DataFormat.HTML})
128      * has been put to this {@code ClipboardContent}.
129      * @return true if {@code containsKey(DataFormat.HTML)}

131      */
132     public final boolean hasHtml() {
133         return containsKey(DataFormat.HTML);
134     }
135

137      * Puts an HTML String into the {@code ClipboardContent}. This is
138      * equivalent to invoking {@code put(DataFormat.HTML, html)}.
139      * Setting this value to null effectively clears it
140      * from the {@code ClipboardContent}.
141      * @param html The string to place. This may be null.
142      * @return always true (the HTML is always successfully put)
143      */
144     public final boolean putHtml(String html) {
145         if (html == null) {
146             remove(DataFormat.HTML);
147         } else {
148             put(DataFormat.HTML, html);
149         }
150         return true;
151     }
152

154      * Gets the HTML String from the {@code ClipboardContent}
155      * which had previously been put. This is equivalent to invoking
156      * {@code get(DataFormat.HTML)}. If no such entry exists,
157      * null is returned.
158      * @return The String in the {@code ClipboardContent} associated
159      * with {@code DataFormat.HTML}, or null if there is not one.
160      */
161     public final String getHtml() {
162         return (String) get(DataFormat.HTML);
163     }
164 
165     /**
166      * Gets whether a RTF String ({@code DataFormat.RTF})
167      * has been put to this {@code ClipboardContent}.
168      * @return true if {@code containsKey(DataFormat.RTF)}
169      * returns true, false otherwise
170      */
171     public final boolean hasRtf() {
172         return containsKey(DataFormat.RTF);
173     }
174 
175     /**
176      * Puts a RTF String into the {@code ClipboardContent}. This is

178      * Setting this value to null effectively clears it

181      * @return always true (the RTF is always successfully put)
182      */
183     public final boolean putRtf(String rtf) {
184         if (rtf == null) {
185             remove(DataFormat.RTF);
186         } else {
187             put(DataFormat.RTF, rtf);
188         }
189         return true;
190     }
191 
192     /**
193      * Gets the RTF String from the {@code ClipboardContent}
194      * which had previously been put. This is equivalent to invoking

196      * null is returned.
197      * @return The String in the {@code ClipboardContent} associated
198      * with {@code DataFormat.RTF}, or null if there is not one.

200     public final String getRtf() {
201         return (String) get(DataFormat.RTF);
202     }
203 
204     /**
205      * Gets whether an Image ({@code DataFormat.IMAGE})
206      * has been put to this {@code ClipboardContent}.
207      * @return true if {@code containsKey(DataFormat.IMAGE)}
208      * returns true, false otherwise
209      */
210     public final boolean hasImage() {
211         return containsKey(DataFormat.IMAGE);
212     };
213 
214     /**
215      * Puts an Image into the {@code ClipboardContent}. This is
216      * equivalent to invoking {@code put(DataFormat.IMAGE, i)}.
217      * Setting this value to null effectively clears it
218      * from the {@code ClipboardContent}. When an image is placed
219      * on the clipboard in this manner, an operating system dependent image
220      * is loaded onto the clipboard (such as TIFF on Mac or DIB on Windows).
221      *
222      * @param i The image to place. This may be null.
223      * @return always true (the image is always successfully put)
224      */
225     public final boolean putImage(Image i) {
226         if (i == null) {
227             remove(DataFormat.IMAGE);
228         } else {
229             put(DataFormat.IMAGE, i);
230         }
231         return true;
232     }
233 
234     /**
235      * Gets the Image from the {@code ClipboardContent}
236      * which had previously been put. This is equivalent to invoking
237      * {@code get(DataFormat.IMAGE)}. If no such entry exists,
238      * null is returned.
239      * @return The Image in the {@code ClipboardContent} associated
240      * with {@code DataFormat.IMAGE}, or null if there is not one.
241      */
242     public final Image getImage() {
243         return (Image) get(DataFormat.IMAGE);
244     }
245 
246     /**
247      * Gets whether a List of Files ({@code DataFormat.FILES})
248      * has been put to this {@code ClipboardContent}.
249      * @return true if {@code containsKey(DataFormat.FILES)}
250      * returns true, false otherwise
251      */
252     public final boolean hasFiles() {
253         return containsKey(DataFormat.FILES);
254     }
255 
256     /**
257      * Puts a List of Files into the {@code ClipboardContent}. This is
258      * equivalent to invoking {@code put(DataFormat.FILES, files)}.
259      * Setting this value to null effectively clears it
260      * from the {@code ClipboardContent}.
261      *
262      * @param files The files to place. This may be null.
263      * @return always true (the files are always successfully put)
264      */
265     public final boolean putFiles(List<File> files) {
266         if (files == null) {
267             remove(DataFormat.FILES);
268         } else {
269             put(DataFormat.FILES, files);
270         }
271         return true;
272     }
273 
274     /**
275      * Puts a List of Files into the {@code ClipboardContent}, based
276      * on the file path. This is simply a convenience method which constructs
277      * a List of Files and invokes the {@link #putFiles} method.
278      *
279      * @param filePaths The files to place. This may be null.
280      * @return always true (the files are always successfully put)
281      */
282     public final boolean putFilesByPath(List<String> filePaths) {
283         final List<File> files = new ArrayList<File>(filePaths.size());
284         for (String path : filePaths) {
285             files.add(new File(path));
286         }
287         return putFiles(files);
288     }
289 
290     /**
291      * Gets the List of Files from the {@code ClipboardContent}
292      * which had previously been put. This is equivalent to invoking
293      * {@code get(DataFormat.FILES)}. If no such entry exists,
294      * null is returned.
295      * @return The List of Files in the {@code ClipboardContent} associated
296      * with {@code DataFormat.FILES}, or null if there is not one.
297      */
298     public final List<File> getFiles() {
299         return (List<File>) get(DataFormat.FILES);
300     }
