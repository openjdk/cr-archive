 26 package javafx.scene.control;
 27 
 28 import javafx.beans.property.BooleanProperty;
 29 import javafx.beans.property.SimpleBooleanProperty;
 30 
 31 /**
 32  * The abstract base class for MultipleSelectionModel implementations that are used within
 33  * table-like controls (most notably {@link TableView} and {@link TreeTableView}.
 34  *
 35  * @param <T> The type of the underlying data model for the UI control.
 36  * @since JavaFX 8.0
 37  */
 38 public abstract class TableSelectionModel<T> extends MultipleSelectionModelBase<T> {
 39 
 40     /**
 41      * Constructor for subclasses to call.
 42      */
 43     public TableSelectionModel() {
 44     }
 45 
 46     /**
 47      * Convenience function which tests whether the given row and column index
 48      * is currently selected in this table instance. If the table control is in its
 49      * 'cell selection' mode (where individual cells can be selected, rather than
 50      * entire rows), and if the column argument is null, this method should return
 51      * true only if all cells in the given row are selected.
 52      * @param row the row
 53      * @param column the column
 54      * @return true if the given row and column index is currently selected in
 55      * this table instance
 56      */
 57     public abstract boolean isSelected(int row, TableColumnBase<T,?> column);
 58 
 59     /**
 60      * Selects the cell at the given row/column intersection. If the table control is in its
 61      * 'cell selection' mode (where individual cells can be selected, rather than
 62      * entire rows), and if the column argument is null, this method should select
 63      * all cells in the given row.
 64      * @param row the row
 65      * @param column the column
 66      */
 67     public abstract void select(int row, TableColumnBase<T,?> column);
 68 
 69     /**
 70      * Clears all selection, and then selects the cell at the given row/column
 71      * intersection. If the table control is in its
 72      * 'cell selection' mode (where individual cells can be selected, rather than
 73      * entire rows), and if the column argument is null, this method should select
 74      * all cells in the given row.
 75      * @param row the row
 76      * @param column the column
 77      */
 78     public abstract void clearAndSelect(int row, TableColumnBase<T,?> column);
 79 
 80     /**
 81      * Removes selection from the specified row/column position (in view indexes).
 82      * If this particular cell (or row if the column value is -1) is not selected,
 83      * nothing happens. If the table control is in its
 84      * 'cell selection' mode (where individual cells can be selected, rather than
 85      * entire rows), and if the column argument is null, this method should deselect
 86      * all cells in the given row.
 87      * @param row the row
 88      * @param column the column
 89      */
 90     public abstract void clearSelection(int row, TableColumnBase<T,?> column);
 91 
 92     /**
 93      * Selects the cell to the left of the currently selected cell.
 94      */
 95     public abstract void selectLeftCell();
 96 
 97     /**
 98      * Selects the cell to the right of the currently selected cell.
 99      */
100     public abstract void selectRightCell();
101 
102     /**
103      * Selects the cell directly above the currently selected cell.
104      */
105     public abstract void selectAboveCell();
106 
107     /**
108      * Selects the cell directly below the currently selected cell.
109      */
110     public abstract void selectBelowCell();
111 
112     /**
113      * Selects the cells in the range (minRow, minColumn) to (maxRow, maxColumn),
114      * inclusive.
115      * @param minRow the minRow
116      * @param minColumn the minColumn
117      * @param maxRow the maxRow
118      * @param maxColumn the maxColumn
119      */
120     public abstract void selectRange(int minRow, TableColumnBase<T,?> minColumn,
121                                      int maxRow, TableColumnBase<T,?> maxColumn);
122 
123     /**
124      * A boolean property used to represent whether the table is in
125      * row or cell selection modes. By default a table is in row selection
126      * mode which means that individual cells can not be selected. Setting
127      * <code>cellSelectionEnabled</code> to be true results in cells being
128      * able to be selected (but not rows).
129      */
130     private BooleanProperty cellSelectionEnabled =
131            new SimpleBooleanProperty(this, "cellSelectionEnabled");
132     public final BooleanProperty cellSelectionEnabledProperty() {
133        return cellSelectionEnabled;
134     }
135     public final void setCellSelectionEnabled(boolean value) {
136        cellSelectionEnabledProperty().set(value);
137     }
138     public final boolean isCellSelectionEnabled() {
139        return cellSelectionEnabled == null ? false : cellSelectionEnabled.get();
140     }
141 }
