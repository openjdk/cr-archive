26 package javafx.css;
27 
28 import com.sun.javafx.css.PseudoClassState;
29 
30 /**
31  * PseudoClass represents one unique pseudo-class state. Introducing a
32  * pseudo-class into a JavaFX class only requires that the method
33  * {@link javafx.scene.Node#pseudoClassStateChanged(javafx.css.PseudoClass, boolean)}
34  * be called when the pseudo-class state changes. Typically, the
35  * {@code pseudoClassStateChanged} method is called from the
36  * {@code protected void invalidated()} method of one of the property base
37  * classes in the {@code javafx.beans.property} package.
38  * <p>
39  * Note that if a node has a default pseudo-class state, a horizontal orientation
40  * for example, {@code pseudoClassStateChanged} should be called from the
41  * constructor to set the initial state.
42  * <p>
43  * The following example would allow &quot;xyzzy&quot; to be used as a
44  *  pseudo-class in a CSS selector.
45  * <pre><code>
46  *  public boolean isMagic() {
47  *       return magic.get();
48  *   }
49  *
50  *   public BooleanProperty magicProperty() {
51  *       return magic;
52  *   }
53  *
54  *   public BooleanProperty magic =
55  *       new BooleanPropertyBase(false) {
56  *
57  *       {@literal @}Override protected void invalidated() {
58  *           pseudoClassStateChanged(MAGIC_PSEUDO_CLASS. get());
59  *       }
60  *
61  *       {@literal @}Override public Object getBean() {
62  *           return MyControl.this;
63  *       }
64  *
65  *       {@literal @}Override public String getName() {
66  *           return "magic";
67  *       }
68  *   }
69  *
70  *   private static final PseudoClass
71  *       MAGIC_PSEUDO_CLASS = PseudoClass.getPseudoClass("xyzzy");
72  * </code></pre>
73  * @since JavaFX 8.0
74  */
75 public abstract class PseudoClass {
76 






77     /**
78      * There is only one PseudoClass instance for a given pseudoClass.
79      * @param pseudoClass the pseudo-class
80      * @return The PseudoClass for the given pseudoClass. Will not return null.
81      * @throws IllegalArgumentException if pseudoClass parameter is null or an empty String
82      */
83     public static PseudoClass getPseudoClass(String pseudoClass) {
84 
85         return PseudoClassState.getPseudoClass(pseudoClass);
86 
87     }
88 
89     /** @return the pseudo-class state */
90     abstract public String getPseudoClassName();
91 
92 }
