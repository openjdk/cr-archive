26 package javafx.css.converter;
27 
28 import javafx.css.ParsedValue;
29 import javafx.css.StyleConverter;
30 import javafx.scene.shape.SVGPath;
31 import javafx.scene.shape.Shape;
32 import javafx.scene.text.Font;
33 
34 import java.util.Map;
35 
36 /**
37  * Converts an SVG shape string into a Shape object.
38  *
39  * @since 9
40  */
41 public class ShapeConverter extends StyleConverter<String, Shape> {
42     private static final ShapeConverter INSTANCE = new ShapeConverter();
43 
44     public static StyleConverter<String, Shape> getInstance() { return INSTANCE; }
45 
46     /**
47      * Creates a default {@code ShapeConverter}.
48      */
49     public ShapeConverter() {
50     }
51 
52     @Override public Shape convert(ParsedValue<String, Shape> value, Font font) {
53 
54         Shape shape = super.getCachedValue(value);
55         if (shape != null) return shape;
56 
57         String svg = value.getValue();
58         if (svg == null || svg.isEmpty()) return null;
59         SVGPath path = new SVGPath();
60         path.setContent(svg);
61 
62         super.cacheValue(value, path);
63 
64         return path;
65     }
66 
67     private static Map<ParsedValue<String, Shape>, Shape> cache;
68 
69     public static void clearCache() { if (cache != null) cache.clear(); }
70 
71 }
