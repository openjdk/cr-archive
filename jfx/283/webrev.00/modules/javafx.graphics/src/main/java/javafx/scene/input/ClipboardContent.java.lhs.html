 26 package javafx.scene.input;
 27 
 28 import java.io.File;
 29 import java.util.ArrayList;
 30 import java.util.HashMap;
 31 import java.util.List;
 32 
 33 import javafx.scene.image.Image;
 34 
 35 /**
 36  * Data container for {@link Clipboard} data. It can hold multiple data in
 37  * several data formats.
 38  * @since JavaFX 2.0
 39  */
 40 public class ClipboardContent extends HashMap<DataFormat, Object> {







 41     /**
 42      * Gets whether a plain text String ({@code DataFormat.PLAIN_TEXT})
 43      * has been put to this {@code ClipboardContent}.
 44      * @return true if {@code containsKey(DataFormat.PLAIN_TEXT)}
 45      * returns true, false otherwise
 46      */
 47     public final boolean hasString() {
 48         return containsKey(DataFormat.PLAIN_TEXT);
 49     }
 50 
 51     /**
 52      * Puts a plain text String into the {@code ClipboardContent}. This is
 53      * equivalent to invoking {@code put(DataFormat.PLAIN_TEXT, s)}.
 54      * Setting this value to null effectively clears it
 55      * from the {@code ClipboardContent}.
 56      * @param s The string to place. This may be null.
 57      * @return always true (the string is always successfully put)
 58      */
 59     public final boolean putString(String s) {
 60         if (s == null) {
 61             remove(DataFormat.PLAIN_TEXT);
 62         } else {
 63             put(DataFormat.PLAIN_TEXT, s);
 64         }
 65         return true;
 66     }
 67 
 68     /**
 69      * Gets the plain text String from the {@code ClipboardContent}
 70      * which had previously been put. This is equivalent to invoking
 71      * {@code get(DataFormat.PLAIN_TEXT)}. If no such entry exists,
 72      * null is returned.
 73      * @return The String in the {@code ClipboardContent} associated
 74      * with {@code DataFormat.PLAIN_TEXT}, or null if there is not one.
 75      */
 76     public final String getString() {
 77         return (String) get(DataFormat.PLAIN_TEXT);
 78     }
 79 
 80     /**
 81      * Gets whether a URL String ({@code DataFormat.URL})
 82      * has been put to this {@code ClipboardContent}.
 83      * @return true if {@code containsKey(DataFormat.URL)}
 84      * returns true, false otherwise
 85      */
 86     public final boolean hasUrl() {
 87         return containsKey(DataFormat.URL);
 88     }
 89 
 90     /**
 91      * Puts a URL String into the {@code ClipboardContent}. This is
 92      * equivalent to invoking {@code put(DataFormat.URL, url)}.
 93      * Setting this value to null effectively clears it
 94      * from the {@code ClipboardContent}.
 95      * @param url The string to place. This may be null.
 96      * @return always true (the URL is always successfully put)
 97      */
 98     public final boolean putUrl(String url) {
 99         if (url == null) {
100             remove(DataFormat.URL);
101         } else {
102             put(DataFormat.URL, url);
103         }
104         return true;
105     }
106 
107     /**
108      * Gets the URL String from the {@code ClipboardContent}
109      * which had previously been put. This is equivalent to invoking
110      * {@code get(DataFormat.URL)}. If no such entry exists,
111      * null is returned.
112      * @return The String in the {@code ClipboardContent} associated
113      * with {@code DataFormat.URL}, or null if there is not one.
114      */
115     public final String getUrl() {
116         return (String) get(DataFormat.URL);
117     }
118 
119     /**
120      * Gets whether an HTML String ({@code DataFormat.HTML})
121      * has been put to this {@code ClipboardContent}.
122      * @return true if {@code containsKey(DataFormat.HTML)}
123      * returns true, false otherwise
124      */
125     public final boolean hasHtml() {
126         return containsKey(DataFormat.HTML);
127     }
128 
129     /**
130      * Puts an HTML String into the {@code ClipboardContent}. This is
131      * equivalent to invoking {@code put(DataFormat.HTML, html)}.
132      * Setting this value to null effectively clears it
133      * from the {@code ClipboardContent}.
134      * @param html The string to place. This may be null.
135      * @return always true (the HTML is always successfully put)
136      */
137     public final boolean putHtml(String html) {
138         if (html == null) {
139             remove(DataFormat.HTML);
140         } else {
141             put(DataFormat.HTML, html);
142         }
143         return true;
144     }
145 
146     /**
147      * Gets the HTML String from the {@code ClipboardContent}
148      * which had previously been put. This is equivalent to invoking
149      * {@code get(DataFormat.HTML)}. If no such entry exists,
150      * null is returned.
151      * @return The String in the {@code ClipboardContent} associated
152      * with {@code DataFormat.HTML}, or null if there is not one.
153      */
154     public final String getHtml() {
155         return (String) get(DataFormat.HTML);
156     }
157 
158     /**
159      * Gets whether a RTF String ({@code DataFormat.RTF})
160      * has been put to this {@code ClipboardContent}.
161      * @return true if {@code containsKey(DataFormat.RTF)}
162      * returns true, false otherwise
163      */
164     public final boolean hasRtf() {
165         return containsKey(DataFormat.RTF);
166     }
167 
168     /**
169      * Puts a RTF String into the {@code ClipboardContent}. This is
170      * equivalent to invoking {@code put(DataFormat.RTF, rtf)}.
171      * Setting this value to null effectively clears it
172      * from the {@code ClipboardContent}.
173      * @param rtf The string to place. This may be null.
174      * @return always true (the RTF is always successfully put)
175      */
176     public final boolean putRtf(String rtf) {
177         if (rtf == null) {
178             remove(DataFormat.RTF);
179         } else {
180             put(DataFormat.RTF, rtf);
181         }
182         return true;
183     }
184 
185     /**
186      * Gets the RTF String from the {@code ClipboardContent}
187      * which had previously been put. This is equivalent to invoking
188      * {@code get(DataFormat.RTF)}. If no such entry exists,
189      * null is returned.
190      * @return The String in the {@code ClipboardContent} associated
191      * with {@code DataFormat.RTF}, or null if there is not one.
192      */
193     public final String getRtf() {
194         return (String) get(DataFormat.RTF);
195     }
196 
197     /**
198      * Gets whether an Image ({@code DataFormat.IMAGE})
199      * has been put to this {@code ClipboardContent}.
200      * @return true if {@code containsKey(DataFormat.IMAGE)}
201      * returns true, false otherwise
202      */
203     public final boolean hasImage() {
204         return containsKey(DataFormat.IMAGE);
205     };
206 
207     /**
208      * Puts an Image into the {@code ClipboardContent}. This is
209      * equivalent to invoking {@code put(DataFormat.IMAGE, i)}.
210      * Setting this value to null effectively clears it
211      * from the {@code ClipboardContent}. When an image is placed
212      * on the clipboard in this manner, an operating system dependent image
213      * is loaded onto the clipboard (such as TIFF on Mac or DIB on Windows).
214      *
215      * @param i The image to place. This may be null.
216      * @return always true (the image is always successfully put)
217      */
218     public final boolean putImage(Image i) {
219         if (i == null) {
220             remove(DataFormat.IMAGE);
221         } else {
222             put(DataFormat.IMAGE, i);
223         }
224         return true;
225     }
226 
227     /**
228      * Gets the Image from the {@code ClipboardContent}
229      * which had previously been put. This is equivalent to invoking
230      * {@code get(DataFormat.IMAGE)}. If no such entry exists,
231      * null is returned.
232      * @return The Image in the {@code ClipboardContent} associated
233      * with {@code DataFormat.IMAGE}, or null if there is not one.
234      */
235     public final Image getImage() {
236         return (Image) get(DataFormat.IMAGE);
237     }
238 
239     /**
240      * Gets whether a List of Files ({@code DataFormat.FILES})
241      * has been put to this {@code ClipboardContent}.
242      * @return true if {@code containsKey(DataFormat.FILES)}
243      * returns true, false otherwise
244      */
245     public final boolean hasFiles() {
246         return containsKey(DataFormat.FILES);
247     }
248 
249     /**
250      * Puts a List of Files into the {@code ClipboardContent}. This is
251      * equivalent to invoking {@code put(DataFormat.FILES, files)}.
252      * Setting this value to null effectively clears it
253      * from the {@code ClipboardContent}.
254      *
255      * @param files The files to place. This may be null.
256      * @return always true (the files are always successfully put)
257      */
258     public final boolean putFiles(List<File> files) {
259         if (files == null) {
260             remove(DataFormat.FILES);
261         } else {
262             put(DataFormat.FILES, files);
263         }
264         return true;
265     }
266 
267     /**
268      * Puts a List of Files into the {@code ClipboardContent}, based
269      * on the file path. This is simply a convenience method which constructs
270      * a List of Files and invokes the {@link #putFiles} method.
271      *
272      * @param filePaths The files to place. This may be null.
273      * @return always true (the files are always successfully put)
274      */
275     public final boolean putFilesByPath(List<String> filePaths) {
276         final List<File> files = new ArrayList<File>(filePaths.size());
277         for (String path : filePaths) {
278             files.add(new File(path));
279         }
280         return putFiles(files);
281     }
282 
283     /**
284      * Gets the List of Files from the {@code ClipboardContent}
285      * which had previously been put. This is equivalent to invoking
286      * {@code get(DataFormat.FILES)}. If no such entry exists,
287      * null is returned.
288      * @return The List of Files in the {@code ClipboardContent} associated
289      * with {@code DataFormat.FILES}, or null if there is not one.
290      */
291     public final List<File> getFiles() {
292         return (List<File>) get(DataFormat.FILES);
293     }
294 }
