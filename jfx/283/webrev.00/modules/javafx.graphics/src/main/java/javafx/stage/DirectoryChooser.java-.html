 26 package javafx.stage;
 27 
 28 import com.sun.javafx.tk.Toolkit;
 29 import java.io.File;
 30 import javafx.beans.property.ObjectProperty;
 31 import javafx.beans.property.SimpleObjectProperty;
 32 import javafx.beans.property.SimpleStringProperty;
 33 import javafx.beans.property.StringProperty;
 34 
 35 // PENDING_DOC_REVIEW
 36 /**
 37  * Provides support for standard directory chooser dialogs. These dialogs have
 38  * look and feel of the platform UI components which is independent of JavaFX.
 39  *
 40  * On some platforms where file access may be restricted or not part of the user
 41  * model (for example, on some mobile or embedded devices), opening a directory
 42  * dialog may always result in a no-op (that is, null file being returned).
 43  *
 44  * @since JavaFX 2.1
 45  */
 46 public final class DirectoryChooser {
 47     /**
 48      * The title of the displayed dialog.
 49      */
 50     private StringProperty title;
 51 
 52     public final void setTitle(final String value) {
 53         titleProperty().set(value);
 54     }
 55 
 56     public final String getTitle() {
 57         return (title != null) ? title.get() : null;
 58     }
 59 
 60     public final StringProperty titleProperty() {
 61         if (title == null) {
 62             title = new SimpleStringProperty(this, "title");
 63         }
 64 
 65         return title;
 66     }
 67 
 68     /**
 69      * The initial directory for the displayed dialog.
 70      */
 71     private ObjectProperty<File> initialDirectory;
 72 
 73     public final void setInitialDirectory(final File value) {
 74         initialDirectoryProperty().set(value);
 75     }
 76 
 77     public final File getInitialDirectory() {
 78         return (initialDirectory != null) ? initialDirectory.get() : null;
 79     }
 80 
 81     public final ObjectProperty<File> initialDirectoryProperty() {
 82         if (initialDirectory == null) {
 83             initialDirectory =
 84                     new SimpleObjectProperty<File>(this, "initialDirectory");
 85         }
 86 
 87         return initialDirectory;
 88     }
 89 
 90     /**
 91      * Shows a new directory selection dialog. The method doesn't return until
 92      * the displayed dialog is dismissed. The return value specifies the
 93      * directory chosen by the user or {@code null} if no selection has been
 94      * made. If the owner window for the directory selection dialog is set,
 95      * input to all windows in the dialog's owner chain is blocked while the
 96      * dialog is being shown.
 97      *
 98      * @param ownerWindow the owner window of the displayed dialog
 99      * @return the selected directory or {@code null} if no directory has been
100      *      selected
101      */
102     public File showDialog(final Window ownerWindow) {
103         return Toolkit.getToolkit().showDirectoryChooser(
104                 (ownerWindow != null) ? ownerWindow.getPeer() : null,
105                 getTitle(),
106                 getInitialDirectory());
107     }
108 }
