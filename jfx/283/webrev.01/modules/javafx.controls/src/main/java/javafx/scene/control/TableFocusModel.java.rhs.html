26 package javafx.scene.control;
27 
28 /**
29  * The abstract base class for FocusModel implementations that are used within
30  * table-like controls (most notably {@link TableView} and {@link TreeTableView}.
31  *
32  * @param <T> The type of the underlying data model for the UI control.
33  * @param <TC> The concrete subclass of {@link TableColumnBase} that is used by the
34  *      underlying UI control (e.g. {@link TableColumn} or {@link TreeTableColumn}.
35  * @since JavaFX 8.0
36  */
37 public abstract class TableFocusModel<T, TC extends TableColumnBase<T,?>> extends FocusModel<T> {
38 
39     /***********************************************************************
40      *                                                                     *
41      * Public API                                                          *
42      *                                                                     *
43      **********************************************************************/
44 
45     /**
46      * Creates a default {@code TableFocusModel}.
47      */
48     public TableFocusModel() {
49     }
50 
51     /**
52      * Causes the item at the given index to receive the focus.
53      *
54      * @param row The row index of the item to give focus to.
55      * @param column The column of the item to give focus to. Can be null.
56      */
57     public abstract void focus(int row, TC column);
58 
59     /**
60      * Tests whether the row / cell at the given location currently has the
61      * focus within the UI control.
62      * @param row the row
63      * @param column the column
64      * @return true if the row / cell at the given location currently has the
65      * focus within the UI control
66      */
67     public abstract boolean isFocused(int row, TC column);
68 
69     /**
70      * Attempts to move focus to the cell above the currently focused cell.
71      */
72     public abstract void focusAboveCell();
73 
74     /**
75      * Attempts to move focus to the cell below the currently focused cell.
76      */
77     public abstract void focusBelowCell();
78 
79     /**
80      * Attempts to move focus to the cell to the left of the currently focused cell.
81      */
82     public abstract void focusLeftCell();
83 
84     /**
85      * Attempts to move focus to the cell to the right of the the currently focused cell.
86      */
87     public abstract void focusRightCell();
88 
89 
90 
91      /***********************************************************************
92      *                                                                     *
93      * Private Implementation                                              *
94      *                                                                     *
95      **********************************************************************/
96 
97 }
