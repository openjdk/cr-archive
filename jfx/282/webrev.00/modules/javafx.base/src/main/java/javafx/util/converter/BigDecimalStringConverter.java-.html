26 package javafx.util.converter;
27 
28 import java.math.BigDecimal;
29 import javafx.util.StringConverter;
30 
31 /**
32  * <p>{@link StringConverter} implementation for {@link BigDecimal} values.</p>
33  * @since JavaFX 2.1
34  */
35 public class BigDecimalStringConverter extends StringConverter<BigDecimal> {
36 
37     /** {@inheritDoc} */
38     @Override public BigDecimal fromString(String value) {
39         // If the specified value is null or zero-length, return null
40         if (value == null) {
41             return null;
42         }
43 
44         value = value.trim();
45 
46         if (value.length() < 1) {
47             return null;
48         }
49 
50         return new BigDecimal(value);
51     }
52 
53     /** {@inheritDoc} */
54     @Override public String toString(BigDecimal value) {
55         // If the specified value is null, return a zero-length String
56         if (value == null) {
57             return "";
58         }
59 
60         return ((BigDecimal)value).toString();
61     }
62 }
