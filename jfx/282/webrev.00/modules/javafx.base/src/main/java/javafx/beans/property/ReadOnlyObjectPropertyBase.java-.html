26 package javafx.beans.property;
27 
28 import javafx.beans.InvalidationListener;
29 import javafx.beans.value.ChangeListener;
30 
31 import com.sun.javafx.binding.ExpressionHelper;
32 
33 /**
34  * Base class for all readonly properties wrapping an arbitrary {@code Object}. This class provides a default
35  * implementation to attach listener.
36  *
37  * @see ReadOnlyObjectProperty
38  *
39  * @param <T> the type of the wrapped {@code Object}
40  * @since JavaFX 2.0
41  */
42 public abstract class ReadOnlyObjectPropertyBase<T> extends ReadOnlyObjectProperty<T> {
43 
44     ExpressionHelper<T> helper;
45 
46     @Override
47     public void addListener(InvalidationListener listener) {
48         helper = ExpressionHelper.addListener(helper, this, listener);
49     }
50 
51     @Override
52     public void removeListener(InvalidationListener listener) {
53         helper = ExpressionHelper.removeListener(helper, listener);
54     }
55 
56     @Override
57     public void addListener(ChangeListener<? super T> listener) {
58         helper = ExpressionHelper.addListener(helper, this, listener);
59     }
60 
61     @Override
62     public void removeListener(ChangeListener<? super T> listener) {
63         helper = ExpressionHelper.removeListener(helper, listener);
64     }
65 
66     /**
67      * Sends notifications to all attached
68      * {@link javafx.beans.InvalidationListener InvalidationListeners} and
69      * {@link javafx.beans.value.ChangeListener ChangeListeners}.
70      *
71      * This method needs to be called, if the value of this property changes.
72      */
73     protected void fireValueChangedEvent() {
74         ExpressionHelper.fireValueChangedEvent(helper);
75     }
76 
77 }
