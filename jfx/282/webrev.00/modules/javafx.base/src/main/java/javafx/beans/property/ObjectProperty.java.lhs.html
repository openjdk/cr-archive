 26 package javafx.beans.property;
 27 
 28 import javafx.beans.binding.Bindings;
 29 import javafx.beans.value.ObservableValue;
 30 import javafx.beans.value.WritableObjectValue;
 31 import javafx.collections.ObservableList;
 32 import javafx.collections.ObservableMap;
 33 import javafx.collections.ObservableSet;
 34 
 35 /**
 36  * This class provides a full implementation of a {@link Property} wrapping an
 37  * arbitrary {@code Object}.
 38  *
 39  * The value of a {@code ObjectProperty} can be get and set with {@link #get()},
 40  * {@link #getValue()}, {@link #set(Object)}, and {@link #setValue(Object)}.
 41  *
 42  * A property can be bound and unbound unidirectional with
 43  * {@link #bind(ObservableValue)} and {@link #unbind()}. Bidirectional bindings
 44  * can be created and removed with {@link #bindBidirectional(Property)} and
 45  * {@link #unbindBidirectional(Property)}.
 46  *
 47  * The context of a {@code ObjectProperty} can be read with {@link #getBean()}
 48  * and {@link #getName()}.
 49  *
 50  * For specialized implementations for {@link ObservableList}, {@link ObservableSet} and
 51  * {@link ObservableMap} that also report changes inside the collections, see
 52  * {@link ListProperty}, {@link SetProperty} and {@link MapProperty}, respectively.
 53  *
 54  * @see javafx.beans.value.ObservableObjectValue
 55  * @see javafx.beans.value.WritableObjectValue
 56  * @see ReadOnlyObjectProperty
 57  * @see Property
 58  *
 59  *
 60  * @param <T>
 61  *            the type of the wrapped {@code Object}
 62  * @since JavaFX 2.0
 63  */
 64 public abstract class ObjectProperty<T> extends ReadOnlyObjectProperty<T>
 65         implements Property<T>, WritableObjectValue<T> {
 66 






 67     /**
 68      * {@inheritDoc}
 69      */
 70     @Override
 71     public void setValue(T v) {
 72         set(v);
 73     }
 74 
 75     /**
 76      * {@inheritDoc}
 77      */
 78     @Override
 79     public void bindBidirectional(Property<T> other) {
 80         Bindings.bindBidirectional(this, other);
 81     }
 82 
 83     /**
 84      * {@inheritDoc}
 85      */
 86     @Override
 87     public void unbindBidirectional(Property<T> other) {
 88         Bindings.unbindBidirectional(this, other);
 89     }
 90 
 91     /**
 92      * Returns a string representation of this {@code ObjectProperty} object.
 93      * @return a string representation of this {@code ObjectProperty} object.
 94      */
 95     @Override
 96     public String toString() {
 97         final Object bean = getBean();
 98         final String name = getName();
 99         final StringBuilder result = new StringBuilder(
100                 "ObjectProperty [");
101         if (bean != null) {
102             result.append("bean: ").append(bean).append(", ");
103         }
104         if ((name != null) && (!name.equals(""))) {
105             result.append("name: ").append(name).append(", ");
106         }
107         result.append("value: ").append(get()).append("]");
108         return result.toString();
109     }
110 }
