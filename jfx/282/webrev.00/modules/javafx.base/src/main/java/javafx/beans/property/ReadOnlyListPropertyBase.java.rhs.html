 26 package javafx.beans.property;
 27 
 28 import com.sun.javafx.binding.ListExpressionHelper;
 29 import javafx.beans.InvalidationListener;
 30 import javafx.beans.value.ChangeListener;
 31 import javafx.collections.ListChangeListener;
 32 import javafx.collections.ObservableList;
 33 
 34 /**
 35  * Base class for all readonly properties wrapping an
 36  * {@link javafx.collections.ObservableList}.
 37  * This class provides a default implementation to attach listener.
 38  *
 39  * @see ReadOnlyListProperty
 40  * @since JavaFX 2.1
 41  */
 42 public abstract class ReadOnlyListPropertyBase<E> extends ReadOnlyListProperty<E> {
 43 
 44     private ListExpressionHelper<E> helper;
 45 
 46     /**
 47      * Creates a default {@code ReadOnlyListPropertyBase}.
 48      */
 49     public ReadOnlyListPropertyBase() {
 50     }
 51 
 52     @Override
 53     public void addListener(InvalidationListener listener) {
 54         helper = ListExpressionHelper.addListener(helper, this, listener);
 55     }
 56 
 57     @Override
 58     public void removeListener(InvalidationListener listener) {
 59         helper = ListExpressionHelper.removeListener(helper, listener);
 60     }
 61 
 62     @Override
 63     public void addListener(ChangeListener<? super ObservableList<E>> listener) {
 64         helper = ListExpressionHelper.addListener(helper, this, listener);
 65     }
 66 
 67     @Override
 68     public void removeListener(ChangeListener<? super ObservableList<E>> listener) {
 69         helper = ListExpressionHelper.removeListener(helper, listener);
 70     }
 71 
 72     @Override
 73     public void addListener(ListChangeListener<? super E> listener) {
 74         helper = ListExpressionHelper.addListener(helper, this, listener);
 75     }
 76 
 77     @Override
 78     public void removeListener(ListChangeListener<? super E> listener) {
 79         helper = ListExpressionHelper.removeListener(helper, listener);
 80     }
 81 
 82     /**
 83      * This method needs to be called if the reference to the
 84      * {@link javafx.collections.ObservableList} changes.
 85      *
 86      * It sends notifications to all attached
 87      * {@link javafx.beans.InvalidationListener InvalidationListeners},
 88      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
 89      * {@link javafx.collections.ListChangeListener}.
 90      *
 91      * This method needs to be called, if the value of this property changes.
 92      */
 93     protected void fireValueChangedEvent() {
 94         ListExpressionHelper.fireValueChangedEvent(helper);
 95     }
 96 
 97     /**
 98      * This method needs to be called if the content of the referenced
 99      * {@link javafx.collections.ObservableList} changes.
100      *
101      * Sends notifications to all attached
102      * {@link javafx.beans.InvalidationListener InvalidationListeners},
103      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
104      * {@link javafx.collections.ListChangeListener}.
105      *
106      * This method is called when the content of the list changes.
107      *
108      * @param change the change that needs to be propagated
109      */
110     protected void fireValueChangedEvent(ListChangeListener.Change<? extends E> change) {
111         ListExpressionHelper.fireValueChangedEvent(helper, change);
112     }
113 
114 
115 
116 }
