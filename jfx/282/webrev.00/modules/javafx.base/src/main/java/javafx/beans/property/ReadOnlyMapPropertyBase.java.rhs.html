 26 package javafx.beans.property;
 27 
 28 import com.sun.javafx.binding.MapExpressionHelper;
 29 import javafx.beans.InvalidationListener;
 30 import javafx.beans.value.ChangeListener;
 31 import javafx.collections.MapChangeListener;
 32 import javafx.collections.ObservableMap;
 33 
 34 /**
 35  * Base class for all readonly properties wrapping an {@link javafx.collections.ObservableMap}.
 36  * This class provides a default implementation to attach listener.
 37  *
 38  * @see ReadOnlyMapProperty
 39  * @since JavaFX 2.1
 40  */
 41 public abstract class ReadOnlyMapPropertyBase<K, V> extends ReadOnlyMapProperty<K, V> {
 42 
 43     private MapExpressionHelper<K, V> helper;
 44 
 45     /**
 46      * Creates a default {@code ReadOnlyMapPropertyBase}.
 47      */
 48     public ReadOnlyMapPropertyBase() {
 49     }
 50 
 51     @Override
 52     public void addListener(InvalidationListener listener) {
 53         helper = MapExpressionHelper.addListener(helper, this, listener);
 54     }
 55 
 56     @Override
 57     public void removeListener(InvalidationListener listener) {
 58         helper = MapExpressionHelper.removeListener(helper, listener);
 59     }
 60 
 61     @Override
 62     public void addListener(ChangeListener<? super ObservableMap<K, V>> listener) {
 63         helper = MapExpressionHelper.addListener(helper, this, listener);
 64     }
 65 
 66     @Override
 67     public void removeListener(ChangeListener<? super ObservableMap<K, V>> listener) {
 68         helper = MapExpressionHelper.removeListener(helper, listener);
 69     }
 70 
 71     @Override
 72     public void addListener(MapChangeListener<? super K, ? super V> listener) {
 73         helper = MapExpressionHelper.addListener(helper, this, listener);
 74     }
 75 
 76     @Override
 77     public void removeListener(MapChangeListener<? super K, ? super V> listener) {
 78         helper = MapExpressionHelper.removeListener(helper, listener);
 79     }
 80 
 81     /**
 82      * This method needs to be called if the reference to the
 83      * {@link javafx.collections.ObservableList} changes.
 84      *
 85      * It sends notifications to all attached
 86      * {@link javafx.beans.InvalidationListener InvalidationListeners},
 87      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
 88      * {@link javafx.collections.ListChangeListener}.
 89      *
 90      * This method needs to be called, if the value of this property changes.
 91      */
 92     protected void fireValueChangedEvent() {
 93         MapExpressionHelper.fireValueChangedEvent(helper);
 94     }
 95 
 96     /**
 97      * This method needs to be called if the content of the referenced
 98      * {@link javafx.collections.ObservableList} changes.
 99      *
100      * Sends notifications to all attached
101      * {@link javafx.beans.InvalidationListener InvalidationListeners},
102      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
103      * {@link javafx.collections.ListChangeListener}.
104      *
105      * This method is called when the content of the list changes.
106      *
107      * @param change the change that needs to be propagated
108      */
109     protected void fireValueChangedEvent(MapChangeListener.Change<? extends K, ? extends V> change) {
110         MapExpressionHelper.fireValueChangedEvent(helper, change);
111     }
112 
113 
114 
115 }
