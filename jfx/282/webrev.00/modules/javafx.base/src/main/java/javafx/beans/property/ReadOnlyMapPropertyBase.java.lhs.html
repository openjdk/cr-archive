 26 package javafx.beans.property;
 27 
 28 import com.sun.javafx.binding.MapExpressionHelper;
 29 import javafx.beans.InvalidationListener;
 30 import javafx.beans.value.ChangeListener;
 31 import javafx.collections.MapChangeListener;
 32 import javafx.collections.ObservableMap;
 33 
 34 /**
 35  * Base class for all readonly properties wrapping an {@link javafx.collections.ObservableMap}.
 36  * This class provides a default implementation to attach listener.
 37  *
 38  * @see ReadOnlyMapProperty
 39  * @since JavaFX 2.1
 40  */
 41 public abstract class ReadOnlyMapPropertyBase<K, V> extends ReadOnlyMapProperty<K, V> {
 42 
 43     private MapExpressionHelper<K, V> helper;
 44 






 45     @Override
 46     public void addListener(InvalidationListener listener) {
 47         helper = MapExpressionHelper.addListener(helper, this, listener);
 48     }
 49 
 50     @Override
 51     public void removeListener(InvalidationListener listener) {
 52         helper = MapExpressionHelper.removeListener(helper, listener);
 53     }
 54 
 55     @Override
 56     public void addListener(ChangeListener<? super ObservableMap<K, V>> listener) {
 57         helper = MapExpressionHelper.addListener(helper, this, listener);
 58     }
 59 
 60     @Override
 61     public void removeListener(ChangeListener<? super ObservableMap<K, V>> listener) {
 62         helper = MapExpressionHelper.removeListener(helper, listener);
 63     }
 64 
 65     @Override
 66     public void addListener(MapChangeListener<? super K, ? super V> listener) {
 67         helper = MapExpressionHelper.addListener(helper, this, listener);
 68     }
 69 
 70     @Override
 71     public void removeListener(MapChangeListener<? super K, ? super V> listener) {
 72         helper = MapExpressionHelper.removeListener(helper, listener);
 73     }
 74 
 75     /**
 76      * This method needs to be called if the reference to the
 77      * {@link javafx.collections.ObservableList} changes.
 78      *
 79      * It sends notifications to all attached
 80      * {@link javafx.beans.InvalidationListener InvalidationListeners},
 81      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
 82      * {@link javafx.collections.ListChangeListener}.
 83      *
 84      * This method needs to be called, if the value of this property changes.
 85      */
 86     protected void fireValueChangedEvent() {
 87         MapExpressionHelper.fireValueChangedEvent(helper);
 88     }
 89 
 90     /**
 91      * This method needs to be called if the content of the referenced
 92      * {@link javafx.collections.ObservableList} changes.
 93      *
 94      * Sends notifications to all attached
 95      * {@link javafx.beans.InvalidationListener InvalidationListeners},
 96      * {@link javafx.beans.value.ChangeListener ChangeListeners}, and
 97      * {@link javafx.collections.ListChangeListener}.
 98      *
 99      * This method is called when the content of the list changes.
100      *
101      * @param change the change that needs to be propagated
102      */
103     protected void fireValueChangedEvent(MapChangeListener.Change<? extends K, ? extends V> change) {
104         MapExpressionHelper.fireValueChangedEvent(helper, change);
105     }
106 
107 
108 
109 }
