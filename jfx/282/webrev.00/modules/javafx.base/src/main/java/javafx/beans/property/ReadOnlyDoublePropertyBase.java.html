26 package javafx.beans.property;
27 
28 import javafx.beans.InvalidationListener;
29 import javafx.beans.value.ChangeListener;
30 
31 import com.sun.javafx.binding.ExpressionHelper;
32 
33 /**
34  * Base class for all readonly properties wrapping a {@code double}. This class provides a default
35  * implementation to attach listener.
36  *
37  * @see ReadOnlyDoubleProperty
38  * @since JavaFX 2.0
39  */
40 public abstract class ReadOnlyDoublePropertyBase extends ReadOnlyDoubleProperty {
41 
42     ExpressionHelper<Number> helper;
43 
44     /**
45      * Creates a default {@code ReadOnlyDoublePropertyBase}.
46      */
47     public ReadOnlyDoublePropertyBase() {
48     }
49 
50     @Override
51     public void addListener(InvalidationListener listener) {
52         helper = ExpressionHelper.addListener(helper, this, listener);
53     }
54 
55     @Override
56     public void removeListener(InvalidationListener listener) {
57         helper = ExpressionHelper.removeListener(helper, listener);
58     }
59 
60     @Override
61     public void addListener(ChangeListener<? super Number> listener) {
62         helper = ExpressionHelper.addListener(helper, this, listener);
63     }
64 
65     @Override
66     public void removeListener(ChangeListener<? super Number> listener) {
67         helper = ExpressionHelper.removeListener(helper, listener);
68     }
69 
70     /**
71      * Sends notifications to all attached
72      * {@link javafx.beans.InvalidationListener InvalidationListeners} and
73      * {@link javafx.beans.value.ChangeListener ChangeListeners}.
74      *
75      * This method needs to be called, if the value of this property changes.
76      */
77     protected void fireValueChangedEvent() {
78         ExpressionHelper.fireValueChangedEvent(helper);
79     }
80 
81 }
