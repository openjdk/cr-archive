 26 package javafx.beans.property;
 27 
 28 import java.util.Objects;
 29 
 30 import com.sun.javafx.binding.BidirectionalBinding;
 31 import javafx.beans.binding.Bindings;
 32 import javafx.beans.value.ObservableValue;
 33 import javafx.beans.value.WritableFloatValue;
 34 import com.sun.javafx.binding.Logging;
 35 
 36 /**
 37  * This class defines a {@link Property} wrapping a {@code float} value.
 38  * <p>
 39  * The value of a {@code FloatProperty} can be get and set with {@link #get()},
 40  * {@link #getValue()}, {@link #set(float)}, and {@link #setValue(Number)}.
 41  * <p>
 42  * A property can be bound and unbound unidirectional with
 43  * {@link #bind(ObservableValue)} and {@link #unbind()}. Bidirectional bindings
 44  * can be created and removed with {@link #bindBidirectional(Property)} and
 45  * {@link #unbindBidirectional(Property)}.
 46  * <p>
 47  * The context of a {@code FloatProperty} can be read with {@link #getBean()}
 48  * and {@link #getName()}.
 49  * <p>
 50  * Note: setting or binding this property to a null value will set the property to "0.0". See {@link #setValue(java.lang.Number) }.
 51  *
 52  * @see javafx.beans.value.ObservableFloatValue
 53  * @see javafx.beans.value.WritableFloatValue
 54  * @see ReadOnlyFloatProperty
 55  * @see Property
 56  *
 57  * @since JavaFX 2.0
 58  */
 59 public abstract class FloatProperty extends ReadOnlyFloatProperty implements
 60         Property<Number>, WritableFloatValue {
 61 






 62     /**
 63      * {@inheritDoc}
 64      */
 65     @Override
 66     public void setValue(Number v) {
 67         if (v == null) {
 68             Logging.getLogger().fine("Attempt to set float property to null, using default value instead.", new NullPointerException());
 69             set(0.0f);
 70         } else {
 71             set(v.floatValue());
 72         }
 73     }
 74 
 75     /**
 76      * {@inheritDoc}
 77      */
 78     @Override
 79     public void bindBidirectional(Property<Number> other) {
 80         Bindings.bindBidirectional(this, other);
 81     }
 82 
 83     /**
 84      * {@inheritDoc}
 85      */
 86     @Override
 87     public void unbindBidirectional(Property<Number> other) {
 88         Bindings.unbindBidirectional(this, other);
 89     }
 90 
 91     /**
 92      * Returns a string representation of this {@code FloatProperty} object.
 93      * @return a string representation of this {@code FloatProperty} object.
 94      */
 95     @Override
 96     public String toString() {
 97         final Object bean = getBean();
 98         final String name = getName();
 99         final StringBuilder result = new StringBuilder(
100                 "FloatProperty [");
101         if (bean != null) {
102             result.append("bean: ").append(bean).append(", ");
103         }
104         if ((name != null) && (!name.equals(""))) {
105             result.append("name: ").append(name).append(", ");
106         }
107         result.append("value: ").append(get()).append("]");
108         return result.toString();
109     }
110 
111     /**
112      * Returns a {@code FloatProperty} that wraps a
113      * {@link javafx.beans.property.Property} and is
114      * bidirectionally bound to it.
115      * Changing this property will result in a change of the original property.
116      *
117      * <p>
118      * This is very useful when bidirectionally binding an ObjectProperty&lt;Float&gt; and
119      * a FloatProperty.
120      *
121      * <blockquote><pre>
122      *   FloatProperty floatProperty = new SimpleFloatProperty(1.0f);
123      *   ObjectProperty&lt;Float&gt; objectProperty = new SimpleObjectProperty&lt;&gt;(2.0f);
124      *
125      *   // Need to keep the reference as bidirectional binding uses weak references
126      *   FloatProperty objectAsFloat = FloatProperty.floatProperty(objectProperty);
127      *
128      *   floatProperty.bindBidirectional(objectAsFloat);
129      *
130      * </pre></blockquote>
131      *
132      * Another approach is to convert the FloatProperty to ObjectProperty using
133      * {@link #asObject()} method.
134      *
135      * <p>
136      * Note: null values in the source property will be interpreted as 0f
137      *
138      * @param property
139      *            The source {@code Property}
140      * @return A {@code FloatProperty} that wraps the
141      *         {@code Property}
142      * @throws NullPointerException
143      *             if {@code property} is {@code null}
144      * @see #asObject()
145      * @since JavaFX 8.0
146      */
147     public static FloatProperty floatProperty(final Property<Float> property) {
148         Objects.requireNonNull(property, "Property cannot be null");
149         return new FloatPropertyBase() {
150             {
151                 BidirectionalBinding.bindNumber(this, property);
152             }
153 
154             @Override
155             public Object getBean() {
156                 return null; // Virtual property, no bean
157             }
158 
159             @Override
160             public String getName() {
161                 return property.getName();
162             }
163         };
164     }
165 
166     /**
167      * Creates an {@link javafx.beans.property.ObjectProperty} that
168      * bidirectionally bound to this {@code FloatProperty}. If the value of
169      * this {@code FloatProperty} changes, the value of the
170      * {@code ObjectProperty} will be updated automatically and vice-versa.
171      *
172      * <p>
173      * Can be used for binding an ObjectProperty to FloatProperty.
174      *
175      * <blockquote><pre>
176      *   FloatProperty floatProperty = new SimpleFloatProperty(1.0f);
177      *   ObjectProperty&lt;Float&gt; objectProperty = new SimpleObjectProperty&lt;&gt;(2.0f);
178      *
179      *   objectProperty.bind(floatProperty.asObject());
180      * </pre></blockquote>
181      *
182      * @return the new {@code ObjectProperty}
183      * @since JavaFX 8.0
184      */
185     @Override
186     public ObjectProperty<Float> asObject() {
187         return new ObjectPropertyBase<> () {
188             {
189                 BidirectionalBinding.bindNumber(this, FloatProperty.this);
190             }
191 
192             @Override
193             public Object getBean() {
194                 return null; // Virtual property, does not exist on a bean
195             }
196 
197             @Override
198             public String getName() {
199                 return FloatProperty.this.getName();
200             }
201         };
202     }
203 }
