26 package javafx.beans.property;
27 
28 import javafx.beans.InvalidationListener;
29 import javafx.beans.value.ChangeListener;
30 
31 import com.sun.javafx.binding.ExpressionHelper;
32 
33 /**
34  * Base class for all readonly properties wrapping a {@code float}. This class provides a default
35  * implementation to attach listener.
36  *
37  * @see ReadOnlyFloatProperty
38  * @since JavaFX 2.0
39  */
40 public abstract class ReadOnlyFloatPropertyBase extends ReadOnlyFloatProperty {
41 
42     ExpressionHelper<Number> helper;
43 






44     @Override
45     public void addListener(InvalidationListener listener) {
46         helper = ExpressionHelper.addListener(helper, this, listener);
47     }
48 
49     @Override
50     public void removeListener(InvalidationListener listener) {
51         helper = ExpressionHelper.removeListener(helper, listener);
52     }
53 
54     @Override
55     public void addListener(ChangeListener<? super Number> listener) {
56         helper = ExpressionHelper.addListener(helper, this, listener);
57     }
58 
59     @Override
60     public void removeListener(ChangeListener<? super Number> listener) {
61         helper = ExpressionHelper.removeListener(helper, listener);
62     }
63 
64     /**
65      * Sends notifications to all attached
66      * {@link javafx.beans.InvalidationListener InvalidationListeners} and
67      * {@link javafx.beans.value.ChangeListener ChangeListeners}.
68      *
69      * This method needs to be called, if the value of this property changes.
70      */
71     protected void fireValueChangedEvent() {
72         ExpressionHelper.fireValueChangedEvent(helper);
73     }
74 
75 }
