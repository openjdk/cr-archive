26 package javafx.util.converter;
27 
28 import java.math.BigInteger;
29 import javafx.util.StringConverter;
30 
31 /**
32  * <p>{@link StringConverter} implementation for {@link BigInteger} values.</p>
33  * @since JavaFX 2.1
34  */
35 public class BigIntegerStringConverter extends StringConverter<BigInteger> {
36 
37     /**
38      * Creates a default {@code BigIntegerStringConverter}.
39      */
40     public BigIntegerStringConverter() {
41     }
42 
43     /** {@inheritDoc} */
44     @Override public BigInteger fromString(String value) {
45         // If the specified value is null or zero-length, return null
46         if (value == null) {
47             return null;
48         }
49 
50         value = value.trim();
51 
52         if (value.length() < 1) {
53             return null;
54         }
55 
56         return new BigInteger(value);
57     }
58 
59     /** {@inheritDoc} */
60     @Override public String toString(BigInteger value) {
61         // If the specified value is null, return a zero-length String
62         if (value == null) {
63             return "";
64         }
65 
66         return ((BigInteger)value).toString();
67     }
68 }
