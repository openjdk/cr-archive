29  * Converter defines conversion behavior between strings and objects.
30  * The type of objects and formats of strings are defined by the subclasses
31  * of Converter.
32  * @since JavaFX 2.0
33  */
34 public abstract class StringConverter<T> {







35     /**
36     * Converts the object provided into its string form.
37     * Format of the returned string is defined by the specific converter.
38     * @param object the object of type {@code T} to convert
39     * @return a string representation of the object passed in.
40     */
41     public abstract String toString(T object);
42

48     */
49     public abstract T fromString(String string);
50 }
