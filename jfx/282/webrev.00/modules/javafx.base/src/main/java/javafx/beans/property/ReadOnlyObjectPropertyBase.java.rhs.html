26 package javafx.beans.property;
27 
28 import javafx.beans.InvalidationListener;
29 import javafx.beans.value.ChangeListener;
30 
31 import com.sun.javafx.binding.ExpressionHelper;
32 
33 /**
34  * Base class for all readonly properties wrapping an arbitrary {@code Object}. This class provides a default
35  * implementation to attach listener.
36  *
37  * @see ReadOnlyObjectProperty
38  *
39  * @param <T> the type of the wrapped {@code Object}
40  * @since JavaFX 2.0
41  */
42 public abstract class ReadOnlyObjectPropertyBase<T> extends ReadOnlyObjectProperty<T> {
43 
44     ExpressionHelper<T> helper;
45 
46     /**
47      * Creates a default {@code ReadOnlyObjectPropertyBase}.
48      */
49     public ReadOnlyObjectPropertyBase() {
50     }
51 
52     @Override
53     public void addListener(InvalidationListener listener) {
54         helper = ExpressionHelper.addListener(helper, this, listener);
55     }
56 
57     @Override
58     public void removeListener(InvalidationListener listener) {
59         helper = ExpressionHelper.removeListener(helper, listener);
60     }
61 
62     @Override
63     public void addListener(ChangeListener<? super T> listener) {
64         helper = ExpressionHelper.addListener(helper, this, listener);
65     }
66 
67     @Override
68     public void removeListener(ChangeListener<? super T> listener) {
69         helper = ExpressionHelper.removeListener(helper, listener);
70     }
71 
72     /**
73      * Sends notifications to all attached
74      * {@link javafx.beans.InvalidationListener InvalidationListeners} and
75      * {@link javafx.beans.value.ChangeListener ChangeListeners}.
76      *
77      * This method needs to be called, if the value of this property changes.
78      */
79     protected void fireValueChangedEvent() {
80         ExpressionHelper.fireValueChangedEvent(helper);
81     }
82 
83 }
