26 package javafx.collections;
27 
28 import java.util.Map;
29 
30 import javafx.beans.Observable;
31 
32 /**
33  * A map that allows observers to track changes when they occur.
34  *
35  * @see MapChangeListener
36  * @see MapChangeListener.Change
37  * @since JavaFX 2.0
38  */
39 public interface ObservableMap<K, V> extends Map<K, V>, Observable {
40     /**
41      * Add a listener to this observable map.
42      * @param listener the listener for listening to the list changes
43      */
44     public void addListener(MapChangeListener<? super K, ? super V> listener);
45     /**
46      * Tries to removed a listener from this observable map. If the listener is not
47      * attached to this map, nothing happens.
48      * @param listener a listener to remove
49      */
50     public void removeListener(MapChangeListener<? super K, ? super V> listener);
51 }
