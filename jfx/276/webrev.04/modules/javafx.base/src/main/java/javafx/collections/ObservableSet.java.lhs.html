26 package javafx.collections;
27 
28 
29 import java.util.Set;
30 
31 import javafx.beans.Observable;
32 
33 /**
34  * A set that allows observers to track changes when they occur.


35  *
36  * @see SetChangeListener
37  * @see SetChangeListener.Change

38  * @since JavaFX 2.1
39  */
40 public interface ObservableSet<E> extends Set<E>, Observable {
41     /**
42      * Add a listener to this observable set.
43      * @param listener the listener for listening to the set changes
44      */
45     public void addListener(SetChangeListener<? super E> listener);
46     /**
47      * Tries to removed a listener from this observable set. If the listener is not
48      * attached to this list, nothing happens.
49      * @param listener a listener to remove
50      */
51     public void removeListener(SetChangeListener<? super E> listener);
52 }
