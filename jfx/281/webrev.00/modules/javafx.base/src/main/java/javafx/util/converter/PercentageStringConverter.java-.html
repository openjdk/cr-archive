26 package javafx.util.converter;
27 
28 import java.text.NumberFormat;
29 import java.util.Locale;
30 import javafx.util.StringConverter;
31 
32 /**
33  * A {@link StringConverter} implementation for {@link Number} values that represent percentages. Instances of this class are
34  * immutable.
35  *
36  * @see CurrencyStringConverter
37  * @see NumberStringConverter
38  * @see StringConverter
39  * @since JavaFX 2.1
40  */
41 public class PercentageStringConverter extends NumberStringConverter {
42 
43     /**
44      * Constructs a {@code PercentageStringConverter} with the default locale and format.
45      */
46     public PercentageStringConverter() {
47         this(Locale.getDefault());
48     }
49 
50     /**
51      * Constructs a {@code PercentageStringConverter} with the given locale and the default format.
52      *
53      * @param locale the locale used in determining the number format used to format the string
54      */
55     public PercentageStringConverter(Locale locale) {
56         super(locale, null, null);
57     }
58 
59     /**
60      * Constructs a {@code PercentageStringConverter} with the given number format.
61      *
62      * @param numberFormat the number format used to format the string
63      */
64     public PercentageStringConverter(NumberFormat numberFormat) {
65         super(null, null, numberFormat);
66     }
67 
68     /** {@inheritDoc} */
69     @Override public NumberFormat getNumberFormat() {
70         Locale _locale = locale == null ? Locale.getDefault() : locale;
71 
72         if (numberFormat != null) {
73             return numberFormat;
74         } else {
75             return NumberFormat.getPercentInstance(_locale);
76         }
77     }
78 }
