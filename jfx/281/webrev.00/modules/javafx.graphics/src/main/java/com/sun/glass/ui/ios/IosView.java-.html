28 import com.sun.glass.ui.Pixels;
29 import com.sun.glass.ui.View;
30 import java.util.Map;
31 
32 /**
33  * iOS View platform implementation.
34  */
35 final class IosView extends View {
36 
37     private long nativePtr;
38 
39     public IosView() {
40         super();
41     }
42 
43     // Constants /reusing Mac OS X values
44     private static final long multiClickTime =  300;
45     private static final int multiClickMaxX = 2;
46     private static final int multiClickMaxY = 2;
47 
48     static long _getMultiClickTime() {
49         return multiClickTime;
50     }
51 
52     static int _getMultiClickMaxX() {
53         return multiClickMaxX;
54     }
55 
56     static int _getMultiClickMaxY() {
57         return multiClickMaxY;
58     }
59 
60     // See View
61     @Override protected void _enableInputMethodEvents(long ptr, boolean enable) { }
62 
63     @Override native protected int _getNativeFrameBuffer(long ptr);
64     @Override native protected long _create(Map caps);
65 
66     @Override native protected long _getNativeView(long ptr);
67 
68     @Override native protected int _getX(long ptr);
69     @Override protected native int _getY(long ptr);
70 
71     @Override native protected boolean _close(long ptr);
72 
73     @Override native protected void _scheduleRepaint(long ptr);
74 
75     @Override native protected void _begin(long ptr);
76     @Override native protected void _end(long ptr);
77 
78     @Override native protected boolean _enterFullscreen(long ptr, boolean animate, boolean keepRatio, boolean hideCursor);
79     @Override native protected void _exitFullscreen(long ptr, boolean animate);
80 
81     @Override native protected void _setParent(long ptr, long parentPtr);
82     @Override protected void _uploadPixels(long ptr, Pixels pixels) {
83         throw new RuntimeException("IosView._uploadPixels() UNIMPLEMENTED.");
84     }
85 
86     private void notifyUnicode(int type, int keyCode, int unicode, int modifiers) {
87         notifyKey(type, keyCode, new char[] {(char) unicode}, modifiers);
88     }
89 
90 }
