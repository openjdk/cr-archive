37 D3DLight::D3DLight() {
38     color[0] = 0;
39     color[1] = 0;
40     color[2] = 0;
41     position[0] = 0;
42     position[1] = 0;
43     position[2] = 0;
44     position[3] = 0; // padding since SetPixelShaderConstantF only takes vec4f; position[3] is unused
45     w = 0;
46 }

48 void D3DLight::setColor(float r, float g, float b) {
49     color[0] = r;
50     color[1] = g;
51     color[2] = b;
52 }
53 
54 void D3DLight::setPosition(float x, float y, float z) {
55     position[0] = x;
56     position[1] = y;
57     position[2] = z;
58 }
