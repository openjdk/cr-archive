26 package javafx.scene;
27 
28 import com.sun.javafx.scene.DirtyBits;
29 import com.sun.javafx.scene.NodeHelper;
30 import com.sun.javafx.scene.PointLightHelper;
31 import com.sun.javafx.sg.prism.NGNode;
32 import com.sun.javafx.sg.prism.NGPointLight;
33 import javafx.scene.paint.Color;
34 
35 /**
36  * Defines a point light source object. A light source that has a
37  * fixed point in space and radiates light equally in all directions
38  * away from itself.
39  *
40  * @since JavaFX 8.0
41  */
42 public class PointLight extends LightBase {
43     static {
44         PointLightHelper.setPointLightAccessor(new PointLightHelper.PointLightAccessor() {
45             @Override
46             public NGNode doCreatePeer(Node node) {
47                 return ((PointLight) node).doCreatePeer();
48             }
49         });
50     }
51 
52     {
53         // To initialize the class helper at the begining each constructor of this class
54         PointLightHelper.initHelper(this);
55     }
56 
57     /**
58      * Creates a new instance of {@code PointLight} class with a default Color.WHITE light source.
59      */
60     public PointLight() {
61         super();
62     }
63 
64     /**
65      * Creates a new instance of {@code PointLight} class using the specified color.
66      *
67      * @param color the color of the light source
68      */
69     public PointLight(Color color) {
70         super(color);
71     }
72 
73     /*
74      * Note: This method MUST only be called via its accessor method.
75      */
76     private NGNode doCreatePeer() {
77         return new NGPointLight();
78     }
79 }
