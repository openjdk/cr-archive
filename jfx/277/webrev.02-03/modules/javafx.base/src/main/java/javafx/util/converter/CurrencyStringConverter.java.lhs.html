26 package javafx.util.converter;
27 
28 import java.text.DecimalFormat;
29 import java.text.DecimalFormatSymbols;
30 import java.text.NumberFormat;
31 import java.util.Locale;
32 import javafx.util.StringConverter;
33 
34 /**
35  * A {@link StringConverter} implementation for {@link Number} values that represent currency. Instances of this class are immutable.
36  *
37  * @see PercentageStringConverter
38  * @see NumberStringConverter
39  * @see StringConverter
40  * @since JavaFX 2.1
41  */
42 public class CurrencyStringConverter extends NumberStringConverter {
43 
44     /**
45      * Constructs a {@code CurrencyStringConverter} with the default locale and format.
46      */
47     public CurrencyStringConverter() {
48         this(Locale.getDefault());
49     }
50 
51     /**
52      * Constructs a {@code CurrencyStringConverter} with the given locale and the default format.


53      */
54     public CurrencyStringConverter(Locale locale) {
55         this(locale, null);
56     }
57 
58     /**
59      * Constructs a {@code CurrencyStringConverter} with the default locale and the given decimal format pattern.
60      *


61      * @see java.text.DecimalFormat
62      */
63     public CurrencyStringConverter(String pattern) {
64         this(Locale.getDefault(), pattern);
65     }
66 
67     /**
68      * Constructs a {@code CurrencyStringConverter} with the given locale and decimal format pattern.
69      *



70      * @see java.text.DecimalFormat
71      */
72     public CurrencyStringConverter(Locale locale, String pattern) {
73         super(locale, pattern, null);
74     }
75 
76     /**
77      * Constructs a {@code CurrencyStringConverter} with the given number format.


78      */
79     public CurrencyStringConverter(NumberFormat numberFormat) {
80         super(null, null, numberFormat);
81     }
82 
83     /** {@inheritDoc} */
84     @Override protected NumberFormat getNumberFormat() {
85         Locale _locale = locale == null ? Locale.getDefault() : locale;
86 
87         if (numberFormat != null) {
88             return numberFormat;
89         } else if (pattern != null) {
90             DecimalFormatSymbols symbols = new DecimalFormatSymbols(_locale);
91             return new DecimalFormat(pattern, symbols);
92         } else {
93             return NumberFormat.getCurrencyInstance(_locale);
94         }
95     }
96 }
