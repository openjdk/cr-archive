26 package javafx.util.converter;
27 
28 import java.text.DecimalFormat;
29 import java.text.DecimalFormatSymbols;
30 import java.text.NumberFormat;
31 import java.util.Locale;
32 import javafx.util.StringConverter;
33 
34 /**
35  * <p>{@link StringConverter} implementation for {@link Number} values
36  * that represent currency.</p>
37  *
38  * @see PercentageStringConverter
39  * @see NumberStringConverter
40  * @see StringConverter
41  * @since JavaFX 2.1
42  */
43 public class CurrencyStringConverter extends NumberStringConverter {
44 
45     // ------------------------------------------------------------ Constructors
46     public CurrencyStringConverter() {
47         this(Locale.getDefault());
48     }
49 
50     public CurrencyStringConverter(Locale locale) {
51         this(locale, null);
52     }
53 
54     public CurrencyStringConverter(String pattern) {
55         this(Locale.getDefault(), pattern);
56     }
57 
58     public CurrencyStringConverter(Locale locale, String pattern) {
59         super(locale, pattern, null);
60     }
61 
62     public CurrencyStringConverter(NumberFormat numberFormat) {
63         super(null, null, numberFormat);
64     }
65 
66 
67     // ---------------------------------------------------------------0- Methods
68 
69     /** {@inheritDoc} */
70     @Override protected NumberFormat getNumberFormat() {
71         Locale _locale = locale == null ? Locale.getDefault() : locale;
72 
73         if (numberFormat != null) {
74             return numberFormat;
75         } else if (pattern != null) {
76             DecimalFormatSymbols symbols = new DecimalFormatSymbols(_locale);
77             return new DecimalFormat(pattern, symbols);
78         } else {
79             return NumberFormat.getCurrencyInstance(_locale);
80         }
81     }
82 }
