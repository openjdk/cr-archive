52 
53 public class EnvironmentVariablesPage extends DistinctItemsPage {
54 	public static class EnvironmentVariablesPageFactory implements IDataPageFactory {
55 
56 		@Override
57 		public String getName(IState state) {
58 			return Messages.EnvironmentVariablesPage_PAGE_NAME;
59 		}
60 
61 		@Override
62 		public ImageDescriptor getImageDescriptor(IState state) {
63 			return FlightRecorderUI.getDefault().getMCImageDescriptor(ImageConstants.PAGE_ENVIRONMENT_VARIABLES);
64 		}
65 
66 		@Override
67 		public String[] getTopics(IState state) {
68 			return new String[] {JfrRuleTopics.ENVIRONMENT_VARIABLES_TOPIC};
69 		}
70 
71 		@Override
72 		public IDisplayablePage createPage(IPageDefinition dpd, StreamModel items, IPageContainer editor) {
73 			return new EnvironmentVariablesPage(dpd, items, editor);
74 		}
75 	}
76 
77 	public EnvironmentVariablesPage(IPageDefinition dpd, StreamModel items, IPageContainer editor) {
78 		super(dpd, items, editor);
79 		setTableDefinition(JdkQueries.ENVIRONMENT_VARIABLE);
80 		setLabelProvider(DataPageToolkit.createTableHighlightProvider(PasswordsInArgumentsRule.PASSWORD_PATTERN, true));
81 	}
82 
83 	@Override
84 	public IItemFilter getDefaultSelectionFilter() {
85 		return JdkFilters.ENVIRONMENT_VARIABLE;
86 	}
87 
88 }
