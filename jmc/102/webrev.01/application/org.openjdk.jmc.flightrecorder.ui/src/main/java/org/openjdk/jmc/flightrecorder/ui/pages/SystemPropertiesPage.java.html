52 
53 public class SystemPropertiesPage extends DistinctItemsPage {
54 	public static class SystemPropertiesPageFactory implements IDataPageFactory {
55 		@Override
56 		public String getName(IState state) {
57 			return Messages.SystemPropertiesPage_PAGE_NAME;
58 		}
59 
60 		@Override
61 		public ImageDescriptor getImageDescriptor(IState state) {
62 			return FlightRecorderUI.getDefault().getMCImageDescriptor(ImageConstants.PAGE_SYSTEM_PROPERTIES);
63 		}
64 
65 		@Override
66 		public String[] getTopics(IState state) {
67 			return new String[] {JfrRuleTopics.SYSTEM_PROPERTIES};
68 		}
69 
70 		@Override
71 		public IDisplayablePage createPage(IPageDefinition dpd, StreamModel items, IPageContainer editor) {
72 			return new SystemPropertiesPage(dpd, items, editor);
73 		}
74 	}
75 
76 	public SystemPropertiesPage(IPageDefinition dpd, StreamModel items, IPageContainer editor) {
77 		super(dpd, items, editor);
78 		setTableDefinition(JdkQueries.SYSTEM_PROPERTIES);
79 		setLabelProvider(DataPageToolkit.createTableHighlightProvider(PasswordsInArgumentsRule.PASSWORD_PATTERN, true));
80 	}
81 
82 	@Override
83 	public IItemFilter getDefaultSelectionFilter() {
84 		return JdkFilters.SYSTEM_PROPERTIES;
85 	}
86 }
