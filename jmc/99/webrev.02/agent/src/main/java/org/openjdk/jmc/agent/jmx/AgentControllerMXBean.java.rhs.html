33 package org.openjdk.jmc.agent.jmx;
34 
35 import org.openjdk.jmc.agent.jfr.JFRTransformDescriptor;
36 
37 public interface AgentControllerMXBean {
38 
39 	/**
40 	 * Reverts all existing instrumentation and then defines a new set of event probes to transform.
41 	 *
42 	 * @param xmlDescription
43 	 *           the XML snippet describing event probes to transform.
44 	 * @throws Exception
45 	 *           thrown when an event probe's specified class can not be retransformed.
46 	 */
47 	public void defineEventProbes(String xmlDescription) throws Exception;
48 
49 	/**
50 	 * Returns the most recent XML configuration that was successfully applied.
51 	 *
52 	 * @return an XML snippet of the current configuration.
53 	 */
54 	public String retrieveEventProbes();
55 
56 	public JFRTransformDescriptor[] retrieveCurrentTransforms();
57 }
