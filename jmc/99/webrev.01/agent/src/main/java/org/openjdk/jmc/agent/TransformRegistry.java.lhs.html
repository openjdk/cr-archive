33 package org.openjdk.jmc.agent;
34 
35 import java.util.List;
36 import java.util.Set;
37 
38 public interface TransformRegistry {
39 	/**
40 	 * The named class has transforms that have not been executed yet.
41 	 *
42 	 * @param className
43 	 *            the name of the class to be transformed.
44 	 * @return true if the class has transforms scheduled. false if not.
45 	 */
46 	boolean hasPendingTransforms(String className);
47 
48 	/**
49 	 * Returns the list of {@link TransformDescriptor}s for the named class.
50 	 *
51 	 * @param className
52 	 *            the class for which to retrieve the transformation metadata.
53 	 * @return the list of transformation metadata for the named class.
54 	 */
55 	List<TransformDescriptor> getTransformData(String className);
56 
57 	/**
58 	 * Returns the names of all classes stored in the registry.
59 	 *
60 	 * @return the unmodifiable set of class names.
61 	 */
62 	Set<String> getClassNames();
63 















64 	/**
65 	 * Modifies class information in the registry according to the xml description.
66 	 *
67 	 * @param xmlDescription
68 	 *           an XML snippet describing the wanted modifications.
69 	 *
70 	 * @return a set of class names associated with modified {@link TransformDescriptor}s.
71 	 */
72 	Set<String> modify(String xmlDescription);
73 
74 	/**
75 	 * Clears all classes and their corresponding transforms in the registry.
76 	 *
77 	 * @return the set of class names that were cleared.
78 	 */
79 	Set<String> clearAllTransformData();
80 
81 	/**
82 	 * Signify classes are or are not being reverted to their pre instrumentation versions.
83 	 * @param shouldRevert
84 	 *           true if class instrumentation should be reverted, false otherwise.
85 	 */
86 	void setRevertInstrumentation(boolean shouldRevert);
87 
88 	/**
89 	 * Determines if classes should be reverted to their pre instrumentation versions.
90 	 * @return true, if classes should be reverted and false otherwise.
91 	 */
92 	boolean isRevertIntrumentation();
93 
94 }
