151 	/**
152 	 * Calculates the click point of the Chart Canvas
153 	 *
154 	 * @return the Point of the Chart Canvas
155 	 */
156 	private Point getRelativeClickPoint() {
157 		Fetcher<Point> fetcher = new Fetcher<Point>() {
158 			@Override
159 			public void run() {
160 				setOutput(new Point(control.getScreenBounds().x / 2, control.getScreenBounds().y / 2));
161 			}
162 		};
163 		Display.getDefault().syncExec(fetcher);
164 		return fetcher.getOutput();
165 	}
