548 	public static String stackTraceMessage(int itemCount, int totalCount, String frameFraction) {
549 		String message;
550 		if (itemCount == 1) {
551 			message = Messages.STACKTRACE_VIEW_TRACE_OF_TOTAL;
552 		} else {
553 			message = Messages.STACKTRACE_VIEW_TRACES_OF_TOTAL;
554 		}
555 		return NLS.bind(message, new Object[] {itemCount, frameFraction, totalCount});
556 	}

558 	public static String siblingMessage(int itemsInSiblings, int nSiblings) {
559 		String message;
560 		if (itemsInSiblings == 1) {
561 			if (nSiblings == 1) {
562 				message = Messages.STACKTRACE_VIEW_TRACE_IN_GROUP;
563 			} else {
564 				message = Messages.STACKTRACE_VIEW_TRACE_IN_GROUPS;
565 			}
566 		} else {
567 			if (nSiblings == 1) {
568 				message = Messages.STACKTRACE_VIEW_TRACES_IN_GROUP;
569 			} else {
570 				message = Messages.STACKTRACE_VIEW_TRACES_IN_GROUPS;
571 			}
572 		}
573 		return NLS.bind(message, itemsInSiblings, nSiblings);
574 	}
