 44 	public static Color getColor(String typeId) {
 45 		switch (typeId) {
 46 		case JdkTypeIDs.RECORDINGS:
 47 			return Palette.PF_CYAN_300.getAWTColor();
 48 		case JdkTypeIDs.RECORDING_SETTING:
 49 			return Palette.PF_GREEN_300.getAWTColor();
 50 		case JdkTypeIDs.THROWABLES_STATISTICS:
 51 			return Palette.PF_GOLD_500.getAWTColor();
 52 		case JdkTypeIDs.BIASED_LOCK_CLASS_REVOCATION:
 53 			return Palette.PF_PURPLE_400.getAWTColor();
 54 		case JdkTypeIDs.BIASED_LOCK_REVOCATION:
 55 			return Palette.PF_LIGHT_GREEN_300.getAWTColor();
 56 		case JdkTypeIDs.BIASED_LOCK_SELF_REVOCATION:
 57 			return Palette.PF_LIGHT_GREEN_400.getAWTColor();
 58 		case JdkTypeIDs.FILE_READ:
 59 			return Palette.PF_ORANGE_400.getAWTColor();
 60 		case JdkTypeIDs.FILE_WRITE:
 61 			return Palette.PF_CYAN_600.getAWTColor();
 62 		case JdkTypeIDs.ERRORS_THROWN:
 63 			return Palette.PF_RED_100.getAWTColor();
 64 		case JdkTypeIDs.EXCEPTIONS_THROWN:
 65 			return Palette.PF_RED_300.getAWTColor();
 66 		case JdkTypeIDs.MONITOR_ENTER:
 67 			return Palette.PF_ORANGE_200.getAWTColor();
 68 		case JdkTypeIDs.MONITOR_WAIT:
 69 			return Palette.PF_GOLD_200.getAWTColor();
 70 		case JdkTypeIDs.THREAD_PARK:
 71 			return Palette.PF_BLACK_500.getAWTColor();
 72 		case JdkTypeIDs.THREAD_SLEEP:
 73 			return Palette.PF_BLUE_500.getAWTColor();
 74 		case JdkTypeIDs.OLD_OBJECT_SAMPLE:
 75 			return Palette.PF_CYAN_200.getAWTColor();
 76 		case JdkTypeIDs.SWEEP_CODE_CACHE:
 77 			return Palette.PF_LIGHT_GREEN_500.getAWTColor();
 78 		case JdkTypeIDs.SOCKET_READ:
 79 			return Palette.PF_RED_200.getAWTColor();
 80 		case JdkTypeIDs.SOCKET_WRITE:
 81 			return Palette.PF_LIGHT_BLUE_500.getAWTColor();
 82 		case JdkTypeIDs.CLASS_LOAD:
 83 			return Palette.PF_PURPLE_100.getAWTColor();
 84 		case JdkTypeIDs.COMPILATION:
 85 			return Palette.PF_GOLD_300.getAWTColor();
 86 		case JdkTypeIDs.GC_PAUSE:
 87 			return Palette.PF_ORANGE_500.getAWTColor();
 88 		case JdkTypeIDs.GC_PAUSE_L1:
 89 			return Palette.PF_GOLD_400.getAWTColor();
 90 		case JdkTypeIDs.GC_PAUSE_L2:
 91 			return Palette.PF_BLUE_400.getAWTColor();
 92 		case JdkTypeIDs.GC_PAUSE_L3:
 93 			return Palette.PF_PURPLE_300.getAWTColor();
 94 		case JdkTypeIDs.GC_PAUSE_L4:
 95 			return Palette.PF_LIGHT_GREEN_600.getAWTColor();
 96 		case JdkTypeIDs.SAFEPOINT_BEGIN:
 97 			return Palette.PF_PURPLE_200.getAWTColor();
 98 		case JdkTypeIDs.SAFEPOINT_CLEANUP:

102 		case JdkTypeIDs.SAFEPOINT_END:
103 			return Palette.PF_GREEN_400.getAWTColor();
104 		case JdkTypeIDs.SAFEPOINT_STATE_SYNC:
105 			return Palette.PF_LIGHT_GREEN_200.getAWTColor();
106 		case JdkTypeIDs.SAFEPOINT_WAIT_BLOCKED:
107 			return Palette.PF_GREEN_500.getAWTColor();
108 		case JdkTypeIDs.VM_OPERATIONS:
109 			return Palette.PF_ORANGE_500.getAWTColor();
110 		case JdkTypeIDs.ALLOC_INSIDE_TLAB:
111 			return new Color(0xFF8000);
112 		case JdkTypeIDs.ALLOC_OUTSIDE_TLAB:
113 			return new Color(0x808000);
114 		case JdkTypeIDs.JAVA_THREAD_END:
115 			return new Color(0x408040);
116 		case JdkTypeIDs.JAVA_THREAD_START:
117 			return new Color(0x80FF80);
118 		case JdkTypeIDs.CLASS_UNLOAD:
119 			return new Color(0x00FF00);
120 		case JdkTypeIDs.COMPILER_FAILURE:
121 			return new Color(0xE67245);
122 		case JdkTypeIDs.GARBAGE_COLLECTION:
123 			return new Color(0xD50000);
124 		case JdkTypeIDs.GC_COLLECTOR_OLD_GARBAGE_COLLECTION:
125 			return new Color(0x800000);
126 		case JdkTypeIDs.GC_COLLECTOR_YOUNG_GARBAGE_COLLECTION:
127 			return new Color(0xFF8080);
128 		case JdkTypeIDs.GC_DETAILED_EVACUATION_FAILED:
129 			return new Color(0xFF0000);
130 		case JdkTypeIDs.GC_DETAILED_PROMOTION_FAILED:
131 			return new Color(0xD04E4E);
132 		case JdkTypeIDs.EXECUTION_SAMPLE:
133 			return new Color(0xCC66FF);
134 		case JdkTypeIDs.EXECUTION_SAMPLING_INFO_EVENT_ID:
135 			return new Color(0xE6C940);
136 		case JdkTypeIDs.PROCESSES:
137 			return new Color(0xE37A44);
138 		case JdkTypeIDs.CONCURRENT_MODE_FAILURE:
139 			return new Color(0xFF0000);
140 		case JdkTypeIDs.CONTEXT_SWITCH_RATE:
141 			return new Color(0x7940E6);
142 		case JdkTypeIDs.CPU_INFORMATION:
143 			return new Color(0xBB97FF);
144 		case JdkTypeIDs.CPU_LOAD:
145 			return new Color(0x000096);
146 		case JdkTypeIDs.THREAD_DUMP:
147 			return new Color(0xFFA800);
148 		default:
149 			// "http://www.oracle.com/hotspot/jvm/vm/gc/collector/old_garbage_collection" -> new Color(0x800000);
150 			// "http://www.oracle.com/hotspot/jvm/vm/gc/collector/young_garbage_collection" -> new Color(0xFF8080);
151 			// "http://www.oracle.com/hotspot/jvm/vm/gc/detailed/evacuation_failed" -> new Color(0xFF0000);
152 			// "http://www.oracle.com/hotspot/jvm/vm/gc/detailed/promotion_failed" -> new Color(0xFF0000);
153 			return null;
154 		}
155 	}
156 
157 	public static Color getColorOrDefault(String string) {
158 		Color color = getColor(string);
159 		return color == null ? ColorToolkit.getDistinguishableColor(string) : color;
160 	}
161 }
