34 package org.openjdk.jmc.ui.charts;
35 
36 import org.eclipse.jface.resource.JFaceResources;
37 import org.eclipse.swt.SWT;
38 import org.eclipse.swt.layout.GridData;
39 import org.eclipse.swt.layout.GridLayout;
40 import org.eclipse.swt.widgets.Composite;
41 import org.eclipse.swt.widgets.Label;
42 import org.eclipse.swt.widgets.Listener;
43 
44 import org.openjdk.jmc.common.unit.IQuantity;
45 import org.openjdk.jmc.common.unit.IRange;
46 import org.openjdk.jmc.ui.misc.ChartCanvas;
47 import org.openjdk.jmc.ui.misc.PatternFly.Palette;
48 import org.openjdk.jmc.ui.misc.TimeFilter;
49 
50 public class ChartFilterControlBar extends Composite {
51 
52 	private static final String THREADS_LABEL = "Threads";
53 
54 	private TimeFilter timeFilter;
55 
56 	public ChartFilterControlBar(Composite parent, Listener resetListener, IRange<IQuantity> recordingRange) {
57 		super(parent, SWT.NONE);
58 		this.setLayout(new GridLayout(3, false));
59 		this.setBackground(Palette.getThreadsPageBackgroundColor());
60 		Label nameLabel = new Label(this, SWT.CENTER | SWT.HORIZONTAL);
61 		nameLabel.setText(THREADS_LABEL);
62 		nameLabel.setBackground(Palette.getThreadsPageBackgroundColor());
63 		GridData gd = new GridData(SWT.FILL, SWT.CENTER, false, true);
64 		gd.widthHint = 180;
65 		nameLabel.setLayoutData(gd);
66 		nameLabel.setFont(JFaceResources.getFontRegistry().get(JFaceResources.BANNER_FONT));
67 
68 		timeFilter = new TimeFilter(this, recordingRange, resetListener);
69 		timeFilter.setLayoutData(new GridData(SWT.LEFT, SWT.CENTER, false, false));
70 	}
71 
72 	public void setChart(XYChart chart) {
73 		timeFilter.setChart(chart);
74 	}
75 
76 	public void setChartCanvas(ChartCanvas canvas) {
77 		timeFilter.setChartCanvas(canvas);
78 	}
79 
80 	public void setStartTime(IQuantity startTime) {
81 		timeFilter.setStartTime(startTime);
82 	}
83 
84 	public void setEndTime(IQuantity endTime) {
85 		timeFilter.setEndTime(endTime);
86 	}
87 }
