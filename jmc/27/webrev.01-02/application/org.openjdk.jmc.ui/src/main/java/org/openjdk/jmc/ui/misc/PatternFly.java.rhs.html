147 		private final String color;
148 
149 		Palette(String color) {
150 			this.color = color;
151 		}
152 
153 		/**
154 		 * Return a color object of the type corresponding to the constant passed into the method.
155 		 * This function first converts the PatternFly color hex value to an AWT Color object. Next,
156 		 * it either returns the AWT Color object if the constant type is AWT, or uses the RGB
157 		 * values to generate an SWT Color object.
158 		 *
159 		 * @param type
160 		 *            String constant: AWT or SWT
161 		 * @return a RGB color (as a regular Object to be casted later)
162 		 */
163 		private Object parseRGB(String type) {
164 			java.awt.Color awtColor = new java.awt.Color(java.awt.Color.decode(this.color).getRGB());
165 			switch (type) {
166 			case (AWT):
167 				return awtColor;
168 			case (SWT):
169 				return new org.eclipse.swt.graphics.Color(Display.getCurrent(), awtColor.getRed(), awtColor.getGreen(),
170 						awtColor.getBlue());
171 			default:
172 				return null;
173 			}
174 		}
175 
176 		/**
177 		 * Converts the PatternFly hex color value to an AWT Color object
178 		 * 
179 		 * @return AWT Color of the selected PatternFly color
180 		 */
181 		public java.awt.Color getAWTColor() {
182 			return (java.awt.Color) parseRGB(AWT);
183 		}
184 
185 		/**
186 		 * Converts the PatternFly hex color value to an SWT Color object
187 		 * 
188 		 * @return SWT Color of the selected PatternFly color
189 		 */
190 		public org.eclipse.swt.graphics.Color getSWTColor() {
191 			return (org.eclipse.swt.graphics.Color) parseRGB(SWT);
192 		}

194 		/**

196 		 */

198 			return PF_BLACK_200.getSWTColor();
