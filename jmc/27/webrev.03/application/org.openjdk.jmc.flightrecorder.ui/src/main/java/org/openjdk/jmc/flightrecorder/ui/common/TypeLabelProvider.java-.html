 33 package org.openjdk.jmc.flightrecorder.ui.common;
 34 
 35 import java.awt.Color;
 36 
 37 import org.openjdk.jmc.common.util.ColorToolkit;
 38 import org.openjdk.jmc.flightrecorder.jdk.JdkTypeIDs;
 39 
 40 // May evolve into a label provider
 41 public class TypeLabelProvider {
 42 
 43 	public static Color getColor(String typeId) {
 44 		switch (typeId) {
 45 		case JdkTypeIDs.ERRORS_THROWN:
 46 			return new Color(0xFF0000);
 47 		case JdkTypeIDs.EXCEPTIONS_THROWN:
 48 			return new Color(0xC80000);
 49 		case JdkTypeIDs.FILE_READ:
 50 			return new Color(0xBE4422);
 51 		case JdkTypeIDs.FILE_WRITE:
 52 			return new Color(0x2C689E);
 53 		case JdkTypeIDs.MONITOR_ENTER:
 54 			return new Color(0xFF6D64);
 55 		case JdkTypeIDs.MONITOR_WAIT:
 56 			return new Color(0xFFE75B);
 57 		case JdkTypeIDs.ALLOC_INSIDE_TLAB:
 58 			return new Color(0xFF8000);
 59 		case JdkTypeIDs.ALLOC_OUTSIDE_TLAB:
 60 			return new Color(0x808000);
 61 		case JdkTypeIDs.SOCKET_READ:
 62 			return new Color(0xC8321E);
 63 		case JdkTypeIDs.SOCKET_WRITE:
 64 			return new Color(0x4678C8);
 65 		case JdkTypeIDs.THREAD_PARK:
 66 			return new Color(0x808080);
 67 		case JdkTypeIDs.JAVA_THREAD_END:
 68 			return new Color(0x408040);
 69 		case JdkTypeIDs.JAVA_THREAD_START:
 70 			return new Color(0x80FF80);
 71 		case JdkTypeIDs.CLASS_LOAD:
 72 			return new Color(0x9B81DB);
 73 		case JdkTypeIDs.CLASS_UNLOAD:
 74 			return new Color(0x00FF00);
 75 		case JdkTypeIDs.COMPILER_FAILURE:
 76 			return new Color(0xE67245);
 77 		case JdkTypeIDs.GARBAGE_COLLECTION:
 78 			return new Color(0xD50000);
 79 		case JdkTypeIDs.GC_COLLECTOR_OLD_GARBAGE_COLLECTION:
 80 			return new Color(0x800000);
 81 		case JdkTypeIDs.GC_COLLECTOR_YOUNG_GARBAGE_COLLECTION:
 82 			return new Color(0xFF8080);
 83 		case JdkTypeIDs.GC_DETAILED_EVACUATION_FAILED:
 84 			return new Color(0xFF0000);
 85 		case JdkTypeIDs.GC_DETAILED_PROMOTION_FAILED:
 86 			return new Color(0xD04E4E);
 87 		case JdkTypeIDs.GC_PAUSE:
 88 			return new Color(0xDC3C00);
 89 		case JdkTypeIDs.GC_PAUSE_L1:
 90 			return new Color(0xE6CB45);
 91 		case JdkTypeIDs.GC_PAUSE_L2:
 92 			return new Color(0x458AE6);
 93 		case JdkTypeIDs.GC_PAUSE_L3:
 94 			return new Color(0xE645E2);
 95 		case JdkTypeIDs.GC_PAUSE_L4:
 96 			return new Color(0x85A115);
 97 		case JdkTypeIDs.EXECUTION_SAMPLE:
 98 			return new Color(0xCC66FF);
 99 		case JdkTypeIDs.EXECUTION_SAMPLING_INFO_EVENT_ID:
100 			return new Color(0xE6C940);
101 		case JdkTypeIDs.VM_OPERATIONS:
102 			return new Color(0xBA6F1F);
103 		case JdkTypeIDs.PROCESSES:
104 			return new Color(0xE37A44);
105 		case JdkTypeIDs.COMPILATION:
106 			return new Color(0xF7EA2A);
107 		case JdkTypeIDs.CONCURRENT_MODE_FAILURE:
108 			return new Color(0xFF0000);
109 		case JdkTypeIDs.CONTEXT_SWITCH_RATE:
110 			return new Color(0x7940E6);
111 		case JdkTypeIDs.CPU_INFORMATION:
112 			return new Color(0xBB97FF);
113 		case JdkTypeIDs.CPU_LOAD:
114 			return new Color(0x000096);
115 		case JdkTypeIDs.THREAD_DUMP:
116 			return new Color(0xFFA800);
117 		default:
118 			// "http://www.oracle.com/hotspot/jvm/vm/gc/collector/old_garbage_collection" -> new Color(0x800000);
119 			// "http://www.oracle.com/hotspot/jvm/vm/gc/collector/young_garbage_collection" -> new Color(0xFF8080);
120 			// "http://www.oracle.com/hotspot/jvm/vm/gc/detailed/evacuation_failed" -> new Color(0xFF0000);
121 			// "http://www.oracle.com/hotspot/jvm/vm/gc/detailed/promotion_failed" -> new Color(0xFF0000);
122 			return null;
123 		}
124 	}
125 
126 	public static Color getColorOrDefault(String string) {
127 		Color color = getColor(string);
128 		return color == null ? ColorToolkit.getDistinguishableColor(string) : color;
129 	}
130 }
