192 		/**
193 		 * Hides a thread from the chart and rebuilds the chart
194 		 */
195 		private void hideThread(Object thread) {
196 			if (this.threadRows != null && this.threadRows.size() > 0 && thread instanceof IMCThread) {
197 				int index = indexOfThread(thread);
198 				if (index != -1) {
199 					this.threadRows.remove(index);
200 					this.reloadThreads = false;
201 					buildChart();
202 					if (!this.isChartModified) {
203 						this.isChartModified = true;
204 						setResetChartActionEnablement(true);
205 					}
206 				}
207 				if (this.threadRows.size() == 0) {
208 					setHideThreadActionEnablement(false);
209 				}
210 			}
211 		}
212 
213 		/**
214 		 * Locates the index of the target Thread in the current selection list
215 		 *
216 		 * @param thread
217 		 *            the thread of interest
218 		 * @return the index of the thread in the current selection, or -1 if not found
219 		 */
220 		private int indexOfThread(Object thread) {
221 			for (int i = 0; i < this.threadRows.size() && thread != null; i++) {
222 				if (this.threadRows.get(i) instanceof QuantitySpanRenderer) {
223 					if (thread.equals(((QuantitySpanRenderer) this.threadRows.get(i)).getData())) {
224 						return i;
225 					}
226 				}
227 			}
228 			return -1;
229 		}

296 							.getSelectedRows((object, items) -> lanes.buildThreadRenderer(object, items))

299 					if (this.isChartMenuActionsInit) {
300 						setResetChartActionEnablement(false);
301 						setHideThreadActionEnablement(true);
302 					}
303 				} else {
304 					this.reloadThreads = true;
305 				}
306 
307 				double threadsWeight = Math.sqrt(threadRows.size()) * 0.15;
308 				double otherRowWeight = Math.max(threadsWeight * 0.1, (1 - threadsWeight) / rows.size());

314 				rows = Arrays.asList(RendererToolkit.weightedRows(rows, weights));
315 			}

258 		private void performToolbarAction(String action, boolean selected) {
259 			switch (action) {
260 			case FOLD_TABLE_ACTION:
261 				if (selected) {
262 					sash.setWeights(this.getStoredSashWeights());
263 					foldTableAction.setToolTipText(Messages.ThreadsPage_FOLD_TABLE_TOOLTIP);

268 					} else {

271 						foldTableAction.setToolTipText(Messages.ThreadsPage_SHOW_TABLE_TOOLTIP);
272 					}
273 				}
274 				break;
275 			case FOLD_CHART_ACTION:
276 				if (selected) {
277 					sash.setWeights(this.getStoredSashWeights());
278 					foldChartAction.setToolTipText(Messages.ThreadsPage_FOLD_CHART_TOOLTIP);
279 				} else {
280 					// if the table is folded, don't fold the chart
281 					if (sash.getWeights()[0] == 0) {
282 						this.foldChartAction.setChecked(true);
283 					} else {

316 		/**
317 		 * Hides a thread from the chart and rebuilds the chart
318 		 */
319 		private void hideThread(Object thread) {
320 			if (this.threadRows != null && this.threadRows.size() > 0 && thread instanceof IMCThread) {
321 				int index = indexOfThread(thread);
322 				if (index != -1) {
323 					this.threadRows.remove(index);
324 					this.reloadThreads = false;
325 					buildChart(false);
326 					if (!this.isChartModified) {
327 						this.isChartModified = true;
328 						setResetChartActionEnablement(true);
329 					}
330 				}
331 				if (this.threadRows.size() == 0) {
332 					setHideThreadActionEnablement(false);
333 				}
334 			}
335 		}

337 		/**
338 		 * Locates the index of the target Thread in the current selection list
339 		 *
340 		 * @param thread
341 		 *            the thread of interest
342 		 * @return the index of the thread in the current selection, or -1 if not found
343 		 */
344 		private int indexOfThread(Object thread) {
345 			for (int i = 0; i < this.threadRows.size() && thread != null; i++) {
346 				if (this.threadRows.get(i) instanceof QuantitySpanRenderer) {
347 					if (thread.equals(((QuantitySpanRenderer) this.threadRows.get(i)).getData())) {
348 						return i;
349 					}
350 				}
351 			}
352 			return -1;
353 		}
