542 	public static String stackTraceMessage(int itemCount, int totalCount, String frameFraction) {
543 		String message;
544 		if (itemCount == 1) {
545 			message = Messages.STACKTRACE_VIEW_TRACE_OF_TOTAL;
546 		} else {
547 			message = Messages.STACKTRACE_VIEW_TRACES_OF_TOTAL;
548 		}
549 		return NLS.bind(message, new Object[] {itemCount, frameFraction, totalCount});
550 	}

552 	public static String siblingMessage(int itemsInSiblings, int nSiblings) {
553 		String message;
554 		if (itemsInSiblings == 1) {
555 			if (nSiblings == 1) {
556 				message = Messages.STACKTRACE_VIEW_TRACE_IN_GROUP;
557 			} else {
558 				message = Messages.STACKTRACE_VIEW_TRACE_IN_GROUPS;
559 			}
560 		} else {
561 			if (nSiblings == 1) {
562 				message = Messages.STACKTRACE_VIEW_TRACES_IN_GROUP;
563 			} else {
564 				message = Messages.STACKTRACE_VIEW_TRACES_IN_GROUPS;
565 			}
566 		}
567 		return NLS.bind(message, itemsInSiblings, nSiblings);
568 	}
