 88 		@Override
 89 		public void mouseDown(MouseEvent e) {
 90 			/*
 91 			 * On Mac OS X, CTRL + left mouse button can be used to trigger a context menu. (This is
 92 			 * for historical reasons when the primary input device on Macs were a mouse with a
 93 			 * single physical button. All modern Macs have other means to bring up the context
 94 			 * menu, typically a two finger tap.)
 95 			 * 
 96 			 * Although I think it would be best to check that this MouseEvent does not cause a
 97 			 * platform specific popup trigger, like java.awt.event.MouseEvent.isPopupTrigger() for
 98 			 * AWT, SWT doesn't seem to have something as simple. It has the MenuDetectEvent, but
 99 			 * the order in relation to this MouseEvent is unspecified.
100 			 * 
101 			 * The code below instead relies on ignoring mouse down events when SWT.MOD4 is
102 			 * depressed. Since MOD4 is CTRL on OS X and 0 on all other current platforms, this
103 			 * suffices. Except for an additional platform check, this approach is also used in
104 			 * org.eclipse.swt.custom.StyledText.handleMouseDown(Event).
105 			 */
106 			if ((e.button == 1) && ((e.stateMask & SWT.MOD4) == 0)) {

107 				selectionStartX = e.x;
108 				selectionStartY = e.y;


109 				selectionIsClick = true;
110 				toggleSelect(selectionStartX, selectionStartY);


































111 			}
112 		}

114 		@Override
115 		public void mouseMove(MouseEvent e) {
116 			if (selectionStartX >= 0) {
117 				highlightRects = null;
118 				updateSelectionState(e);
119 			} else {
120 				lastMouseX = e.x;
121 				lastMouseY = e.y;
122 				updateHighlightRects();
123 			}
124 		}

126 		private void updateSelectionState(MouseEvent e) {
127 			int x = e.x;
128 			int y = e.y;
129 			if (selectionIsClick && ((Math.abs(x - selectionStartX) > 3) || (Math.abs(y - selectionStartY) > 3))) {
130 				selectionIsClick = false;
131 			}
132 			if (!selectionIsClick) {
133 				select((int) (selectionStartX / xScale), (int) (x / xScale), (int) (selectionStartY / yScale),
134 						(int) (y / yScale));
135 			}
136 		}

139 		public void mouseUp(MouseEvent e) {
140 			if (selectionStartX >= 0 && (e.button == 1)) {
141 				updateSelectionState(e);

142 				selectionStartX = -1;
143 				selectionStartY = -1;



144 				if (selectionListener != null) {
145 					selectionListener.run();



146 				}
147 			}

167 	class Painter implements PaintListener {
168 
169 		@Override
170 		public void paintControl(PaintEvent e) {
171 			Rectangle rect = getClientArea();












172 			if (awtNeedsRedraw || !awtCanvas.hasImage(rect.width, rect.height)) {
173 				Graphics2D g2d = awtCanvas.getGraphics(rect.width, rect.height);
174 				g2d.setColor(Color.WHITE);
175 				g2d.fillRect(0, 0, rect.width, rect.height);
176 				Point adjusted = translateDisplayToImageCoordinates(rect.width, rect.height);


177 				render(g2d, adjusted.x, adjusted.y);
178 				if (highlightRects != null) {
179 					updateHighlightRects();


180 				}
181 				awtNeedsRedraw = false;
182 			}
183 			awtCanvas.paint(e, 0, 0);
184 			// Crude, flickering highlight of areas also delivered to tooltips.
185 			// FIXME: Remove flicker by drawing in a buffered stage (AWT or SWT).
186 			List<Rectangle2D> rs = highlightRects;
187 			if (rs != null) {
188 				GC gc = e.gc;
189 				gc.setForeground(getForeground());
190 				for (Rectangle2D r : rs) {
191 					int x = (int) (((int) r.getX()) * xScale);
192 					int y = (int) (((int) r.getY()) * yScale);
193 					if ((r.getWidth() == 0) && (r.getHeight() == 0)) {
194 						int width = (int) Math.round(4 * xScale);
195 						int height = (int) Math.round(4 * yScale);
196 						gc.drawOval(x - (int) Math.round(2 * xScale), y - (int) Math.round(2 * yScale), width, height);
197 					} else {
198 						int width = (int) Math.round(r.getWidth() * xScale);
199 						int height = (int) Math.round(r.getHeight() * yScale);
200 						gc.drawRectangle(x, y, width, height);
201 					}
202 				}

203 			}
204 		}

373 	 * Translates display coordinates into image coordinates for the chart.
374 	 *
375 	 * @param x
376 	 *            the provided x coordinate
377 	 * @param y
378 	 *            the provided y coordinate
379 	 * @return a Point that represents the (x,y) coordinates in the chart's coordinate space

382 		int xImage = (int) Math.round(x / xScale);

411 		List<Rectangle2D> newRects = new ArrayList<>();
412 		infoAt(new IChartInfoVisitor.Adapter() {
413 			@Override
414 			public void visit(IBucket bucket) {

418 			@Override

420 				Point2D target = point.getTarget();
421 				newRects.add(new Rectangle2D.Double(target.getX(), target.getY(), 0, 0));
422 			}
423 
424 			@Override

429 			@Override

433 			}
434 
435 			@Override
436 			public void visit(ILane lane) {
437 				// FIXME: Do we want this highlighted?
438 			}

441 			public void hover(Object data) {
442 				if (data != null) {
443 					setHoveredItemData(data);

445 			}

451 		}

494 	private void toggleSelect(int x, int y) {
495 		Point p = translateDisplayToImageCoordinates(x, y);
496 		if (awtChart != null) {
497 			final IQuantity[] range = new IQuantity[2];
498 			infoAt(new IChartInfoVisitor.Adapter() {
499 				@Override
500 				public void visit(IBucket bucket) {
501 					if (range[0] == null) {
502 						range[0] = (IQuantity) bucket.getStartX();
503 						range[1] = (IQuantity) bucket.getEndX();
504 					}
505 				}
506 
507 				@Override
508 				public void visit(ISpan span) {
509 					if (range[0] == null) {
510 						IDisplayable x0 = span.getStartX();
511 						IDisplayable x1 = span.getEndX();
512 						range[0] = (x0 instanceof IQuantity) ? (IQuantity) x0 : null;
513 						range[1] = (x1 instanceof IQuantity) ? (IQuantity) x1 : null;
514 					}
515 				}
516 			}, x, y);
517 			if ((range[0] != null) || (range[1] != null)) {
518 				if (!awtChart.select(range[0], range[1], p.y, p.y)) {
519 					awtChart.clearSelection();
520 				}
521 			} else {
522 				if (!awtChart.select(p.x, p.x, p.y, p.y)) {
523 					awtChart.clearSelection();
524 				}
525 			}

526 			redrawChart();

527 		}
528 	}

101 			/*
102 			 * On Mac OS X, CTRL + left mouse button can be used to trigger a context menu. (This is
103 			 * for historical reasons when the primary input device on Macs were a mouse with a
104 			 * single physical button. All modern Macs have other means to bring up the context
105 			 * menu, typically a two finger tap.)
106 			 * 
107 			 * Although I think it would be best to check that this MouseEvent does not cause a
108 			 * platform specific popup trigger, like java.awt.event.MouseEvent.isPopupTrigger() for
109 			 * AWT, SWT doesn't seem to have something as simple. It has the MenuDetectEvent, but
110 			 * the order in relation to this MouseEvent is unspecified.
111 			 * 
112 			 * The code below instead relies on ignoring mouse down events when SWT.MOD4 is
113 			 * depressed. Since MOD4 is CTRL on OS X and 0 on all other current platforms, this
114 			 * suffices. Except for an additional platform check, this approach is also used in
115 			 * org.eclipse.swt.custom.StyledText.handleMouseDown(Event).
116 			 */
117 			if ((e.button == 1) && ((e.stateMask & SWT.MOD4) == 0) && ((e.stateMask & SWT.CTRL) == 0)
118 					&& ((e.stateMask & SWT.SHIFT) == 0)) {
119 				selectionStartX = e.x;
120 				selectionStartY = e.y;
121 				highlightSelectionEnd = new Point(-1, -1);
122 				lastSelection = new Point(-1, -1);
123 				selectionIsClick = true;
124 				toggleSelect(selectionStartX, selectionStartY);
125 			} else if (((e.stateMask & SWT.CTRL) != 0) && (e.button == 1)) {
126 				select(e.x, e.x, e.y, e.y, false);
127 				if (selectionListener != null) {
128 					selectionListener.run();
129 				}
130 			} else if (((e.stateMask & SWT.SHIFT) != 0) && (e.button == 1)) {
131 				if (highlightSelectionEnd.y == -1) {
132 					highlightSelectionEnd = new Point(e.x, e.y);
133 					lastSelection = highlightSelectionEnd;
134 					if (highlightSelectionStart.y > highlightSelectionEnd.y) {
135 						Point temp = highlightSelectionStart;
136 						highlightSelectionStart = highlightSelectionEnd;
137 						highlightSelectionEnd = temp;
138 					}
139 				} else {
140 					if (e.y > highlightSelectionStart.y && e.y < highlightSelectionEnd.y) {
141 						if (e.y < lastSelection.y) {
142 							highlightSelectionEnd = new Point(e.x, e.y);
143 						} else if (e.y > lastSelection.y) {
144 							highlightSelectionStart = new Point(e.x, e.y);
145 						}
146 					} else if (e.y < highlightSelectionStart.y) {
147 						highlightSelectionStart = new Point(e.x, e.y);
148 						lastSelection = highlightSelectionStart;
149 					} else if (e.y > highlightSelectionEnd.y) {
150 						highlightSelectionEnd = new Point(e.x, e.y);
151 						lastSelection = highlightSelectionEnd;
152 					}
153 				}
154 				select(highlightSelectionStart.x, highlightSelectionEnd.x, highlightSelectionStart.y,
155 						highlightSelectionEnd.y, true);
156 				if (selectionListener != null) {
157 					selectionListener.run();
158 				}
159 			}
160 		}

224 		@Override
225 		public void paintControl(PaintEvent e) {
226 			Rectangle rect = new Rectangle(0, 0, getParent().getSize().x, getParent().getSize().y);
227 			if (minLaneHeight == -1) {
228 				minLaneHeight = calculateMinLaneHeight(rect);
229 				laneHeight = minLaneHeight;
230 			}
231 			if (getNumItems() == 0) {
232 				rect = getClientArea();
233 			} else if (getNumItems() == 1 || (laneHeight * getNumItems() < rect.height)) {
234 				// it fills the height
235 			} else {
236 				rect.height = laneHeight * getNumItems();
237 			}
238 
239 			if (awtNeedsRedraw || !awtCanvas.hasImage(rect.width, rect.height)) {
240 				Graphics2D g2d = awtCanvas.getGraphics(rect.width, rect.height);


241 				Point adjusted = translateDisplayToImageCoordinates(rect.width, rect.height);
242 				g2d.setColor(Palette.PF_BLACK_100.getAWTColor());
243 				g2d.fillRect(0, 0, adjusted.x, adjusted.y);
244 				render(g2d, adjusted.x, adjusted.y);
245 				if (isScrollableChart()) {
246 					minReadableLaneHeight = (int) (g2d.getFontMetrics().getHeight() * xScale);
247 					minLaneHeight = initMinLaneHeight();
248 					((ScrolledComposite) getParent()).setMinSize(rect.width, rect.height);
249 				}
250 				awtNeedsRedraw = false;
251 			}
252 			awtCanvas.paint(e, 0, 0);
253 			// Crude, flickering highlight of areas also delivered to tooltips.
254 			// FIXME: Remove flicker by drawing in a buffered stage (AWT or SWT).
255 			List<Rectangle2D> rs = highlightRects;
256 			if (rs != null) {
257 				GC gc = e.gc;
258 				gc.setForeground(getForeground());
259 				for (Rectangle2D r : rs) {
260 					int x = (int) (((int) r.getX()) * xScale);
261 					int y = (int) (((int) r.getY()) * yScale);
262 					if ((r.getWidth() == 0) && (r.getHeight() == 0)) {
263 						int width = (int) Math.round(4 * xScale);
264 						int height = (int) Math.round(4 * yScale);
265 						gc.drawOval(x - (int) Math.round(2 * xScale), y - (int) Math.round(2 * yScale), width, height);
266 					} else {
267 						int width = (int) Math.round(r.getWidth() * xScale);
268 						int height = (int) Math.round(r.getHeight() * yScale);
269 						gc.drawRectangle(x, y, width, height);
270 					}
271 				}
272 				updateHighlightRects();
273 			}
274 		}

676 	private void toggleSelect(int x, int y) {
677 		Point p = translateDisplayToImageCoordinates(x, y);
678 		if (awtChart != null) {
679 			final IQuantity[] range = new IQuantity[2];
680 			infoAt(new IChartInfoVisitor.Adapter() {
681 				@Override
682 				public void visit(IBucket bucket) {
683 					if (range[0] == null) {
684 						range[0] = (IQuantity) bucket.getStartX();
685 						range[1] = (IQuantity) bucket.getEndX();
686 					}
687 				}
688 
689 				@Override
690 				public void visit(ISpan span) {
691 					if (range[0] == null) {
692 						IDisplayable x0 = span.getStartX();
693 						IDisplayable x1 = span.getEndX();
694 						range[0] = (x0 instanceof IQuantity) ? (IQuantity) x0 : null;
695 						range[1] = (x1 instanceof IQuantity) ? (IQuantity) x1 : null;
696 					}
697 				}
698 			}, x, y);
699 			if ((range[0] != null) || (range[1] != null)) {
700 				if (!awtChart.select(range[0], range[1], p.y, p.y, true)) {
701 					awtChart.clearSelection();
702 				}
703 			} else {
704 				if (!awtChart.select(p.x, p.x, p.y, p.y, true)) {
705 					awtChart.clearSelection();
706 				}
707 			}
708 			notifyZoomOnClickListener(SWT.MouseDown);
709 			redrawChart();
710 			redrawChartText();
711 		}
