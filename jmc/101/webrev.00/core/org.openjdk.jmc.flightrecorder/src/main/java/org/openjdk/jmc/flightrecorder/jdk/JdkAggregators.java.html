417 	 * Aggregator for getting the first value, ie. the value from the event with the first occurring
418 	 * start time.
419 	 *
420 	 * @param attribute
421 	 *            attribute to get value from
422 	 * @return the value provided by the very first event
423 	 */
424 	public static <V> IAggregator<V, ?> first(IAttribute<V> attribute) {
425 		return new Aggregators.AdvancedMinAggregator<>(
426 				MessageFormat.format(Messages.getString(Messages.AGGR_FIRST_ATTRIBUTE), attribute.getName()),
427 				MessageFormat.format(Messages.getString(Messages.AGGR_FIRST_ATTRIBUTE_DESC), attribute.getName()),
428 				attribute, JfrAttributes.START_TIME);
